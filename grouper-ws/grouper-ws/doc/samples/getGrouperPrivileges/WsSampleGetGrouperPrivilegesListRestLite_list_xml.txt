        .getWsLiteResponseContentType().parseString(response);

      //lets make sure admin is in there
      for (WsGrouperPrivilegeResult wsGrouperPrivilegeResult : wsGetGrouperPrivilegesLiteResult.getPrivilegeResults()) {

          foundAdmin = true;
