24    public static Set<Stem> retrieveStemsForService(String idOfAttributeDef, QueryOptions queryOptions) {
25      return new StemFinder().assignIdOfAttributeDefName(idOfAttributeDef).assignAttributeCheckReadOnAttributeDef(false)
26          .assignQueryOptions(queryOptions).findStems();
