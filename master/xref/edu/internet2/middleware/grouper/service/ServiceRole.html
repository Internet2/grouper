16  package edu.internet2.middleware.grouper.service;
17  
18  import java.util.Collection;
19  import java.util.HashSet;
20  import java.util.Set;
21  
22  import edu.internet2.middleware.grouper.Field;
23  import edu.internet2.middleware.grouper.Group;
24  import edu.internet2.middleware.grouper.privs.AccessPrivilege;
25  import edu.internet2.middleware.grouper.util.GrouperUtil;
26  
27  /**
28   * role in a service, admin (update or admin in service) or user (member of group/role or assignee of permissions)
29   * @author mchyzer
30   *
31   */
32  public enum ServiceRole {
33    
34    /** admin of a service */
35    admin {
36  
37      /**
38       * @see ServiceRole#fieldsForGroupQuery()
39       */
40      @Override
41      public Collection<Field> fieldsForGroupQuery() {
42        return GrouperUtil.toSet(AccessPrivilege.ADMIN.getField(),
43            AccessPrivilege.UPDATE.getField());
44      }
45    },
46    
47    /** user of a service (might include admins) */
48    user {
49  
50      /**
51       * @see ServiceRole#fieldsForGroupQuery()
52       */
53      @Override
54      public Collection<Field> fieldsForGroupQuery() {
55        return GrouperUtil.toSet(Group.getDefaultList());
56      }
57    };
58    
59    /**
60     * get all fields for any role
61     * @return the fields
62     */
63    public static Collection<Field> allFieldsForGroupQuery() {
64      Set<Field> allFields = new HashSet<Field>();
65      for (ServiceRole serviceRole : ServiceRole.values()) {
66        allFields.addAll(serviceRole.fieldsForGroupQuery());
67      }
68      return allFields;
69    }
70    
71    /**
72     * e.g. return the fields for this service role
73     * @return the fields for the query
74     */
75    public abstract Collection<Field> fieldsForGroupQuery();
76    
77    /**
78     * convert a string to the service role enum
79     * @param string
80     * @param exceptionOnNull
81     * @return service role
82     */
83    public static ServiceRole valueOfIgnoreCase(String string, boolean exceptionOnNull) {
84      return GrouperUtil.enumValueOfIgnoreCase(ServiceRole.class, string, exceptionOnNull, true);
85    }
86    
87  }
