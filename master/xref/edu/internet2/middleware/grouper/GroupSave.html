326   /**
327    * <pre>
328    * create or update a group.  Do not throw checked exceptions, wrap in unchecked
329    * 
330    * Note this will not rename a group at this time (might in future)
331    * 
332    * This is a static method since setters to Group objects persist to the DB
333    * 
334    * Steps:
335    * 
336    * 1. Find the group by groupNameToEdit
337    * 2. Internally set all the fields of the stem (no need to reset if already the same)
338    * 3. Store the group (insert or update) if needed
339    * 4. Return the group object
340    * 
341    * This runs in a tx so that if part of it fails the whole thing fails, and potentially the outer
342    * transaction too
343    * </pre>
344    * @return the group
345    * @deprecated use save() instead
346    */
347   @Deprecated
348   public Group saveUnchecked() {
349     return this.save();
350   }

353   /**
354    * <pre>
355    * create or update a group.  Note this will not rename a group at this time (might in future)
356    * 
357    * This is a static method since setters to Group objects persist to the DB
358    * 
359    * Steps:
360    * 
361    * 1. Find the group by groupNameToEdit
362    * 2. Internally set all the fields of the stem (no need to reset if already the same)
363    * 3. Store the group (insert or update) if needed
364    * 4. Return the group object
365    * 
366    * This runs in a tx so that if part of it fails the whole thing fails, and potentially the outer
367    * transaction too
368    * </pre>
369    * @return the group
370    * @throws StemNotFoundException
371    * @throws InsufficientPrivilegeException
372    * @throws StemAddException
373    * @throws GroupModifyException
374    * @throws GroupNotFoundException
375    * @throws GroupAddException
376    */
377   public Group save() 
378         throws StemNotFoundException, InsufficientPrivilegeException, StemAddException, 
379         GroupModifyException, GroupNotFoundException, GroupAddException {
380 
381     //help with incomplete entries
382     if (StringUtils.isBlank(this.name)) {
383       this.name = this.groupNameToEdit;
384     }
385 
386     //get from uuid since could be a rename
387     if (StringUtils.isBlank(this.groupNameToEdit) && !StringUtils.isBlank(this.uuid)) {
388       Group group = GroupFinder.findByUuid(GrouperSession.staticGrouperSession(), this.uuid, true);
389       this.groupNameToEdit = group.getName();
390     }
391     
392     if (StringUtils.isBlank(this.groupNameToEdit)) {
393       this.groupNameToEdit = this.name;
394     }
395     
396     //validate
397     //get the stem name
398     if (!StringUtils.contains(GroupSave.this.name, ":")) {
399       throw new RuntimeException("Group name must exist and must contain at least one stem name (separated by colons): '" + GroupSave.this.name + "'" );
400     }
401 
402     //default to insert or update
403     GroupSave.this.saveMode = (SaveMode)ObjectUtils.defaultIfNull(GroupSave.this.saveMode, SaveMode.INSERT_OR_UPDATE);
404     final SaveMode SAVE_MODE = GroupSave.this.saveMode;
405 
406     try {
407       //do this in a transaction
408       Group group = (Group)GrouperTransaction.callbackGrouperTransaction(new GrouperTransactionHandler() {
409   
410         @SuppressWarnings("cast")
411         @Override
412         public Object callback(GrouperTransaction grouperTransaction)
413             throws GrouperDAOException {
414           
415           grouperTransaction.setCachingEnabled(false);
416           
417           return (Group)GrouperSession.callbackGrouperSession(GroupSave.this.grouperSession, new GrouperSessionHandler() {
418 
419               @Override
420               public Object callback(GrouperSession theGrouperSession)
421                   throws GrouperSessionException {
422                 
423                 String groupNameForError = GrouperUtil.defaultIfBlank(GroupSave.this.groupNameToEdit, GroupSave.this.name);
424                 
425                 int lastColonIndex = GroupSave.this.name.lastIndexOf(':');
426                 boolean topLevelGroup = lastColonIndex < 0;
427         
428                 //empty is root stem
429                 String parentStemNameNew = GrouperUtil.parentStemNameFromName(GroupSave.this.name);
430                 
431                 //note, this might be blank
432                 String parentStemDisplayNameNew = GrouperUtil.parentStemNameFromName(GroupSave.this.displayName);
433                 String extensionNew = GrouperUtil.extensionFromName(GroupSave.this.name);
434                 
435                 String displayExtensionFromDisplayNameNew = GrouperUtil.extensionFromName(GroupSave.this.displayName);
436 
437                 //figure out the display extension from the extension or the name (or both!)
438                 String theDisplayExtension = null;
439                 
440                 //if blank, and display name blank, then, use extension
441                 if (StringUtils.isBlank(GroupSave.this.displayExtension) && StringUtils.isBlank(GroupSave.this.displayName)) {
442                   theDisplayExtension = extensionNew;
443                 } else if (!StringUtils.isBlank(GroupSave.this.displayExtension) && !StringUtils.isBlank(GroupSave.this.displayName)) {
444                   //if neither blank
445                   if (!StringUtils.equals(displayExtensionFromDisplayNameNew, GroupSave.this.displayExtension)) {
446                     throw new RuntimeException("The display extension '" + GroupSave.this.displayExtension 
447                         + "' is not consistent with the last part of the group name '" 
448                         + displayExtensionFromDisplayNameNew + "', display name: " + GroupSave.this.displayName);
449                   }
450                   theDisplayExtension = displayExtensionFromDisplayNameNew;
451                 } else if (!StringUtils.isBlank(GroupSave.this.displayExtension)) {
452                   theDisplayExtension = GroupSave.this.displayExtension;
453                 } else if (!StringUtils.isBlank(GroupSave.this.displayName)) {
454                   theDisplayExtension = displayExtensionFromDisplayNameNew;
455                 } else {
456                   throw new RuntimeException("Shouldnt get here");
457                 }
458 
459                 
460                 //lets find the stem
461                 Stem parentStem = null;
462                 
463                 try {
464                   parentStem = topLevelGroup ? StemFinder.findRootStem(theGrouperSession) 
465                       : StemFinder.findByName(theGrouperSession, parentStemNameNew, true);
466                 } catch (StemNotFoundException snfe) {
467                   
468                   //see if we should fix this problem
469                   if (GroupSave.this.createParentStemsIfNotExist) {
470                     
471                     //at this point the stem should be there (and is equal to currentStem), 
472                     //just to be sure, query again
473                     if (GrouperLoader.isDryRun()) {
474                       parentStem = StemFinder.findByName(theGrouperSession, parentStemNameNew, false);
475                       if (parentStem == null) {
476                         parentStem = new Stem();
477                         parentStem.setNameDb(parentStemNameNew);
478                         parentStem.setExtensionDb(GrouperUtil.extensionFromName(parentStemNameNew));
479                       }
480                     } else {
481                       parentStem = Stem._createStemAndParentStemsIfNotExist(theGrouperSession, parentStemNameNew, parentStemDisplayNameNew);
482                     }
483                   } else {
484                     throw new GrouperSessionException(new StemNotFoundException("Cant find stem: '" + parentStemNameNew 
485                         + "' (from update on stem name: '" + groupNameForError + "')"));
486                   }
487                 }
488                 
489                 Group theGroup = null;
490                 //see if update
491                 boolean isUpdate = SAVE_MODE.isUpdate(GroupSave.this.groupNameToEdit, GroupSave.this.name);
492         
493                 if (isUpdate) {
494                   String parentStemNameLookup = GrouperUtil.parentStemNameFromName(GroupSave.this.groupNameToEdit);
495                   if (!StringUtils.equals(parentStemNameLookup, parentStemNameNew)) {
496                     throw new GrouperSessionException(new GroupModifyException("Can't move a group.  Existing parentStem: '"
497                         + parentStemNameLookup + "', new stem: '" + parentStemNameNew + "'"));
498                 }    
499                 try {
500                     theGroup = GroupFinder.findByName(theGrouperSession, GroupSave.this.groupNameToEdit, true);
501                     
502                     //while we are here, make sure uuid's match if passed in
503                     if (!StringUtils.isBlank(GroupSave.this.uuid) && !StringUtils.equals(GroupSave.this.uuid, theGroup.getUuid())) {
504                       throw new RuntimeException("UUID group changes are not supported: new: " + GroupSave.this.uuid + ", old: " 
505                           + theGroup.getUuid() + ", " + groupNameForError);
506                     }
507                     
508                   } catch (GroupNotFoundException gnfe) {
509                     if (SAVE_MODE.equals(SaveMode.INSERT_OR_UPDATE) || SAVE_MODE.equals(SaveMode.INSERT)) {
510                       isUpdate = false;
511                     } else {
512                         throw new GrouperSessionException(gnfe);
513                     }
514                   }
515                 }
516                 //default
517                 GroupSave.this.saveResultType = SaveResultType.NO_CHANGE;
518                 boolean needsSave = false;
519                 //if inserting
520                 if (!isUpdate) {
521                   GroupSave.this.saveResultType = SaveResultType.INSERT;
522                   if (GroupSave.this.typeOfGroup == null || GroupSave.this.typeOfGroup == TypeOfGroup.group) {
523                     theGroup = parentStem.internal_addChildGroup(extensionNew, theDisplayExtension, GroupSave.this.uuid);
524                   } else if (GroupSave.this.typeOfGroup == TypeOfGroup.role) {
525                     theGroup = (Group)parentStem.internal_addChildRole(extensionNew, theDisplayExtension, GroupSave.this.uuid);
526                   } else if (GroupSave.this.typeOfGroup == TypeOfGroup.entity) {
527                     theGroup = (Group)parentStem.internal_addChildEntity(extensionNew, theDisplayExtension, GroupSave.this.uuid);
528                   } else {
529                     throw new RuntimeException("Not expecting type of group: " + GroupSave.this.typeOfGroup);
530                   }
531                   
532                 } else {
533                   //check if different so it doesnt make unneeded queries
534                   if (!StringUtils.equals(theGroup.getExtension(), extensionNew)) {
535                       
536                       //lets just confirm that one doesnt exist
537                       String newName = GrouperUtil.parentStemNameFromName(theGroup.getName()) + ":" + extensionNew;
538                       
539                       Group existingGroup = GroupFinder.findByName(theGrouperSession.internal_getRootSession(), newName, false);
540                       
541                       if (existingGroup != null && !StringUtils.equals(theGroup.getUuid(), existingGroup.getUuid())) {
542                         throw new GroupModifyAlreadyExistsException("Group already exists: " + newName);
543                       }
544                       
545                       theGroup.setExtension(extensionNew);
546                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
547                     needsSave = true;
548                   }
549                   if (!StringUtils.equals(theGroup.getDisplayExtension(), theDisplayExtension)) {
550                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
551                     theGroup.setDisplayExtension(theDisplayExtension);
552                     needsSave = true;
553                   }
554                 }
555 
556                 if (GroupSave.this.idIndex != null) {
557                   if (GroupSave.this.saveResultType == SaveResultType.INSERT) {
558                     
559                     if (theGroup.assignIdIndex(GroupSave.this.idIndex)) {
560                       needsSave = true;
561                     }
562                     
563                   } else {
564                     //maybe they are equal...
565                     throw new RuntimeException("Cannot update idIndex for an already created group: " + GroupSave.this.idIndex + ", " + theGroup.getName());
566                   }
567                 }
568 
569                 //now compare and put all attributes (then store if needed)
570                 //null throws exception? hmmm.  remove attribute if blank
571                 if (!StringUtils.equals(StringUtils.defaultString(theGroup.getDescription()), 
572                     StringUtils.defaultString(StringUtils.trim(GroupSave.this.description)))) {
573                   needsSave = true;
574                   if (GroupSave.this.saveResultType == SaveResultType.NO_CHANGE) {
575                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
576                   }
577                   theGroup.setDescription(GroupSave.this.description);
578                 }
579 
580                 //compare type of group
581                 if (GroupSave.this.typeOfGroup != null && GroupSave.this.typeOfGroup != theGroup.getTypeOfGroup()) {
582                   needsSave = true;
583                   if (GroupSave.this.saveResultType == SaveResultType.NO_CHANGE) {
584                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
585                   }
586                   theGroup.setTypeOfGroup(GroupSave.this.typeOfGroup);
587                 }
588 
589                 //only store once
590                 if (needsSave) {
591                   theGroup.store();
592                 }
593 
594                 boolean changedPrivs = false;
595                 
596                 boolean adminDefaultChecked = theGroup.hasAdmin(SubjectFinder.findAllSubject());
597                 
598                 boolean updateDefaultChecked = theGroup.hasUpdate(SubjectFinder.findAllSubject());
599                 
600                 boolean readDefaultChecked = theGroup.hasRead(SubjectFinder.findAllSubject());
601       
602                 boolean viewDefaultChecked = theGroup.hasView(SubjectFinder.findAllSubject());
603       
604                 boolean optinDefaultChecked = theGroup.hasOptin(SubjectFinder.findAllSubject());
605       
606                 boolean optoutDefaultChecked = theGroup.hasOptout(SubjectFinder.findAllSubject());
607       
608                 boolean attrReadDefaultChecked = theGroup.hasGroupAttrRead(SubjectFinder.findAllSubject());
609       
610                 boolean attrUpdateDefaultChecked = theGroup.hasGroupAttrUpdate(SubjectFinder.findAllSubject());
611       
612                 if (GroupSave.this.privAllAdmin != null && GroupSave.this.privAllAdmin != adminDefaultChecked) {
613                   if (GroupSave.this.privAllAdmin) {
614                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.ADMIN, false);
615                   } else {
616                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.ADMIN, false);
617                   }
618                 }
619                             
620                 if (GroupSave.this.privAllView != null && GroupSave.this.privAllView != viewDefaultChecked) {
621                   if (GroupSave.this.privAllView) {
622                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
623                   } else {
624                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
625                   }
626                 }
627       
628                 if (GroupSave.this.privAllRead != null && GroupSave.this.privAllRead != readDefaultChecked) {
629                   if (GroupSave.this.privAllRead) {
630                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
631                   } else {
632                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
633                   }
634                 }
635                 if (GroupSave.this.privAllUpdate != null && GroupSave.this.privAllUpdate != updateDefaultChecked) {
636                   if (GroupSave.this.privAllUpdate) {
637                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.UPDATE, false);
638                   } else {
639                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.UPDATE, false);
640                   }
641                 }
642                 if (GroupSave.this.privAllOptin != null && GroupSave.this.privAllOptin != optinDefaultChecked) {
643                   if (GroupSave.this.privAllOptin) {
644                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTIN, false);
645                   } else {
646                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTIN, false);
647                   }
648                 }
649                 if (GroupSave.this.privAllOptout != null && GroupSave.this.privAllOptout != optoutDefaultChecked) {
650                   if (GroupSave.this.privAllOptout) {
651                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTOUT, false);
652                   } else {
653                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTOUT, false);
654                   }
655                 }
656                 if (GroupSave.this.privAllAttrRead != null && GroupSave.this.privAllAttrRead != attrReadDefaultChecked) {

658                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.GROUP_ATTR_READ, false);
659                   } else {

666                   } else {

668                   }
669                 }
670                 
671                 if (changedPrivs) {
672                   if (GroupSave.this.saveResultType == SaveResultType.NO_CHANGE) {
673                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
674                   }
675                 }
676                 
677 			          return theGroup;
678               }
679           });
680         }
681       });
682       return group;
