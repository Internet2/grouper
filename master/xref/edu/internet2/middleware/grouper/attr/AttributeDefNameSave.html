193   /**
194    * <pre>
195    * create or update a attributeDefName.  Note this will not rename an attributeDefName at this time (might in future)
196    * 
197    * This is a static method since setters to AttributeDefName objects persist to the DB
198    * 
199    * Steps:
200    * 
201    * 1. Find the attributeDefName by attributeDefNameNameToEdit
202    * 2. Internally set all the fields of the attributeDefName (no need to reset if already the same)
203    * 3. Store the attributeDefName (insert or update) if needed
204    * 4. Return the attributeDefName object
205    * 
206    * This runs in a tx so that if part of it fails the whole thing fails, and potentially the outer
207    * transaction too
208    * </pre>
209    * @return the attributeDefName
210    * @throws StemNotFoundException
211    * @throws InsufficientPrivilegeException
212    * @throws StemAddException
213    */
214   public AttributeDefName save() 
215         throws StemNotFoundException, InsufficientPrivilegeException, StemAddException {
216   
217     //help with incomplete entries
218     if (StringUtils.isBlank(this.name)) {
219       this.name = this.attributeDefNameNameToEdit;
220     }
221     
222     //get from uuid since could be a rename
223     if (StringUtils.isBlank(this.attributeDefNameNameToEdit) && !StringUtils.isBlank(this.id)) {
224       AttributeDefName attributeDefName = AttributeDefNameFinder.findById(this.id, true);
225       this.attributeDefNameNameToEdit = attributeDefName.getName();
226     }
227     
228     if (StringUtils.isBlank(this.attributeDefNameNameToEdit)) {
229       this.attributeDefNameNameToEdit = this.name;
230     }
231     
232     //validate
233     //get the attributeDefName name
234     if (!StringUtils.contains(this.name, ":")) {
235       throw new RuntimeException("AttributeDefName name must exist and must contain at least one stem name (separated by colons): '" + name + "'" );
236     }
237 
238     //default to insert or update
239     saveMode = (SaveMode)ObjectUtils.defaultIfNull(saveMode, SaveMode.INSERT_OR_UPDATE);
240     final SaveMode SAVE_MODE = saveMode;
241 
242     try {
243       //do this in a transaction
244       AttributeDefName attributeDefName = (AttributeDefName)GrouperTransaction.callbackGrouperTransaction(new GrouperTransactionHandler() {
245   
246         public Object callback(GrouperTransaction grouperTransaction)
247             throws GrouperDAOException {
248           
249           grouperTransaction.setCachingEnabled(false);
250           
251           return (AttributeDefName)GrouperSession.callbackGrouperSession(AttributeDefNameSave.this.grouperSession, new GrouperSessionHandler() {
252 
253               public Object callback(GrouperSession grouperSession)
254                   throws GrouperSessionException {
255                 
256                 String attributeDefNameNameForError = GrouperUtil.defaultIfBlank(AttributeDefNameSave.this.attributeDefNameNameToEdit, AttributeDefNameSave.this.name);
257                 
258                 int lastColonIndex = AttributeDefNameSave.this.name.lastIndexOf(':');
259                 boolean topLevelAttributeDefName = lastColonIndex < 0;
260         
261                 //empty is root stem
262                 String parentStemNameNew = GrouperUtil.parentStemNameFromName(AttributeDefNameSave.this.name);
263                 
264                 //note, this might be blank
265                 String parentStemDisplayNameNew = GrouperUtil.parentStemNameFromName(displayName);
266                 String extensionNew = GrouperUtil.extensionFromName(AttributeDefNameSave.this.name);
267                 
268                 String displayExtensionFromDisplayNameNew = GrouperUtil.extensionFromName(AttributeDefNameSave.this.displayName);
269 
270                 //figure out the display extension from the extension or the name (or both!)
271                 String theDisplayExtension = null;
272                 
273                 //if blank, and display name blank, then, use extension
274                 if (StringUtils.isBlank(AttributeDefNameSave.this.displayExtension) && StringUtils.isBlank(AttributeDefNameSave.this.displayName)) {
275                   theDisplayExtension = extensionNew;
276                 } else if (!StringUtils.isBlank(AttributeDefNameSave.this.displayExtension) && !StringUtils.isBlank(AttributeDefNameSave.this.displayName)) {
277                   //if neither blank
278                   if (!StringUtils.equals(displayExtensionFromDisplayNameNew, AttributeDefNameSave.this.displayExtension)) {
279                     throw new RuntimeException("The display extension '" + AttributeDefNameSave.this.displayExtension 
280                         + "' is not consistent with the last part of the attributeDefName name '" 
281                         + displayExtensionFromDisplayNameNew + "', display name: " + AttributeDefNameSave.this.displayName);
282                   }
283                   theDisplayExtension = displayExtensionFromDisplayNameNew;
284                 } else if (!StringUtils.isBlank(AttributeDefNameSave.this.displayExtension)) {
285                   theDisplayExtension = AttributeDefNameSave.this.displayExtension;
286                 } else if (!StringUtils.isBlank(AttributeDefNameSave.this.displayName)) {
287                   theDisplayExtension = displayExtensionFromDisplayNameNew;
288                 } else {
289                   throw new RuntimeException("Shouldnt get here");
290                 }
291 
292                 
293                 //lets find the stem
294                 Stem parentStem = null;
295                 
296                 try {
297                   parentStem = topLevelAttributeDefName ? StemFinder.findRootStem(grouperSession) 
298                       : StemFinder.findByName(grouperSession, parentStemNameNew, true);
299                 } catch (StemNotFoundException snfe) {
300                   
301                   //see if we should fix this problem
302                   if (AttributeDefNameSave.this.createParentStemsIfNotExist) {
303                     
304                     //at this point the stem should be there (and is equal to currentStem), 
305                     //just to be sure, query again
306                     parentStem = Stem._createStemAndParentStemsIfNotExist(grouperSession, parentStemNameNew, parentStemDisplayNameNew);
307                   } else {
308                     throw new GrouperSessionException(new StemNotFoundException("Cant find stem: '" + parentStemNameNew 
309                         + "' (from update on stem name: '" + attributeDefNameNameForError + "')"));
310                   }
311                 }
312                 
313                 AttributeDefName theAttributeDefName = null;
314                 //see if update
315                 boolean isUpdate = SAVE_MODE.isUpdate(AttributeDefNameSave.this.attributeDefNameNameToEdit, AttributeDefNameSave.this.name);
316         
317                 if (isUpdate) {
318                   String parentStemNameLookup = GrouperUtil.parentStemNameFromName(AttributeDefNameSave.this.attributeDefNameNameToEdit);
319                   if (!StringUtils.equals(parentStemNameLookup, parentStemNameNew)) {
320                     throw new GrouperSessionException(new RuntimeException("Can't move an attributeDefName.  Existing parentStem: '"
321                         + parentStemNameLookup + "', new stem: '" + parentStemNameNew + "'"));
322                   }
323                 }    
324                 theAttributeDefName = GrouperDAOFactory.getFactory().getAttributeDefName().findByNameSecure(
325                     AttributeDefNameSave.this.attributeDefNameNameToEdit, false, new QueryOptions().secondLevelCache(false));
326                 
327                 if (theAttributeDefName != null) {
328 
329                   //while we are here, make sure id's match if passed in
330                   if (!StringUtils.isBlank(AttributeDefNameSave.this.id) && !StringUtils.equals(AttributeDefNameSave.this.id, theAttributeDefName.getId())) {
331                     throw new RuntimeException("ID attributeDefName changes are not supported: new: " + AttributeDefNameSave.this.id + ", old: " 
332                         + theAttributeDefName.getId() + ", " + attributeDefNameNameForError);
333                   }
334                   
335                 } else {
336                   if (SAVE_MODE.equals(SaveMode.INSERT_OR_UPDATE) || SAVE_MODE.equals(SaveMode.INSERT)) {
337                     isUpdate = false;
338                   } else {
339                     throw new RuntimeException("Cant find attributeDefName: " + attributeDefNameNameForError);
340                   }
341                   
342                 }
343                 //default
344                 AttributeDefNameSave.this.saveResultType = SaveResultType.NO_CHANGE;
345                 boolean needsSave = false;
346                 //if inserting
347                 if (!isUpdate) {
348                   saveResultType = SaveResultType.INSERT;
349                   theAttributeDefName = parentStem.internal_addChildAttributeDefName(grouperSession, 
350                       AttributeDefNameSave.this.attributeDef, extensionNew, theDisplayExtension, 
351                       AttributeDefNameSave.this.id, AttributeDefNameSave.this.description);
352                 } else {
353 
354                   //check if different so it doesnt make unneeded queries
355                   if (!StringUtils.equals(theAttributeDefName.getExtension(), extensionNew)) {
356                       
357                     //lets just confirm that one doesnt exist
358                     final String newName = GrouperUtil.parentStemNameFromName(theAttributeDefName.getName()) + ":" + extensionNew;
359                     
360                     AttributeDefName existingAttributeDefName = (AttributeDefName)GrouperSession.callbackGrouperSession(
361                         grouperSession.internal_getRootSession(), new GrouperSessionHandler() {
362 
363                       public Object callback(GrouperSession grouperSession)
364                           throws GrouperSessionException {
365                         return AttributeDefNameFinder.findByName(newName, false);
366                       }
367                       
368                     });
369                     
370                     if (existingAttributeDefName != null && !StringUtils.equals(theAttributeDefName.getId(), existingAttributeDefName.getId())) {
371                       throw new RuntimeException("AttributeDefNameName already exists: " + newName);
372                     }
373                     
374                     theAttributeDefName.setExtension(extensionNew);
375 
376                     AttributeDefNameSave.this.saveResultType = SaveResultType.UPDATE;
377                     needsSave = true;
378                   }
379                   if (!StringUtils.equals(theAttributeDefName.getDisplayExtension(), theDisplayExtension)) {
380                     AttributeDefNameSave.this.saveResultType = SaveResultType.UPDATE;
381                     theAttributeDefName.setDisplayExtensionDb(theDisplayExtension);
382                     theAttributeDefName.setDisplayNameDb(GrouperUtil.parentStemNameFromName(theAttributeDefName.getDisplayName()) + ":" + theDisplayExtension);
383                     needsSave = true;
384                   }
385                 }
386                 
387                 if (AttributeDefNameSave.this.idIndex != null) {
388                   if (AttributeDefNameSave.this.saveResultType == SaveResultType.INSERT) {
389 
390                     if (theAttributeDefName.assignIdIndex(AttributeDefNameSave.this.idIndex)) {
391                       needsSave = true;
392                     }
393                     
394                   } else {
395                     //maybe they are equal...
396                     throw new RuntimeException("Cannot update idIndex for an already created AttributeDefName: " + AttributeDefNameSave.this.idIndex + ", " + theAttributeDefName.getName());
397                   }
398                 }
399 
400 
401                 //now compare and put all attributes (then store if needed)
402                 //null throws exception? hmmm.  remove attribute if blank
403                 if (!StringUtils.equals(StringUtils.defaultString(theAttributeDefName.getDescription()), 
404                     StringUtils.defaultString(StringUtils.trim(AttributeDefNameSave.this.description)))) {
405                   needsSave = true;
406                   if (AttributeDefNameSave.this.saveResultType == SaveResultType.NO_CHANGE) {
407                     AttributeDefNameSave.this.saveResultType = SaveResultType.UPDATE;
408                   }
409                   theAttributeDefName.setDescription(AttributeDefNameSave.this.description);
410                 }
411                 if (!StringUtils.equals(StringUtils.defaultString(theAttributeDefName.getAttributeDefId()), 
412                     StringUtils.defaultString(StringUtils.trim(AttributeDefNameSave.this.attributeDef.getId())))) {
413                   throw new RuntimeException("Cannot change attributeDefId of an attributeDefName! " 
414                       + AttributeDefNameSave.this.attributeDef.getId() + ", " + theAttributeDefName.getAttributeDefId());
415                 }
416 
417 
418                 //only store once
419                 if (needsSave) {
420                   theAttributeDefName.store();
421                 }
422                 
423                 return theAttributeDefName;
424               }
425               
426             });
427             
428         }
429       });
430       return attributeDefName;
431     } catch (RuntimeException re) {
432       
433       Throwable throwable = re.getCause();
434       if (throwable instanceof StemNotFoundException) {
435         throw (StemNotFoundException)throwable;
436       }
437       if (throwable instanceof InsufficientPrivilegeException) {
438         throw (InsufficientPrivilegeException)throwable;
439       }
440       if (throwable instanceof StemAddException) {
441         throw (StemAddException)throwable;
442       }
443       //must just be runtime
444       throw re;
445     }

447   }
