219   /**
220    * make sure this attribute def can admin from grouper session
221    */
222   public void assertCanAdminAttributeDefStatic() {
223     
224     Subject subject = GrouperSession.staticGrouperSession().getSubject();
225     MultiKey cacheKey = new MultiKey(this.attributeDefId, subject.getSourceId(), subject.getId());
226     Boolean result = canAdminAttributeDef.get(cacheKey);
227     AttributeDef attributeDef = null;
228     
229     //if not in cache, calculate
230     if (result == null) {
231       attributeDef = this.getAttributeDef();
232       result = attributeDef.getPrivilegeDelegate().canAttrAdmin(subject);
233       
234       //add back to cache since wasnt cached
235       canAdminAttributeDef.put(cacheKey, result);
236     }
237     
238     //false means cant admin
239     if (!result) {
240       
241       attributeDef = attributeDef == null ? this.getAttributeDef() : attributeDef;
242       
243       throw new InsufficientPrivilegeException(GrouperUtil
244           .subjectToString(subject)
245           + " is not attrAdmin on attributeDef: " + attributeDef.getName() 
246           + ", concerning attributeDefName: " + this.getName());
247     }
248     
249   }

635   /**
636    * save or update this object
637    */
638   public void delete() {
639     
640     HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, AuditControl.WILL_AUDIT, new HibernateHandler() {
641       
642       public Object callback(HibernateHandlerBean hibernateHandlerBean)
643           throws GrouperDAOException {
644 
645         HibernateSession hibernateSession = hibernateHandlerBean.getHibernateSession();
646 
647         hibernateSession.setCachingEnabled(false);
648         
649         //make sure subject is allowed to do this
650         Subject subject = GrouperSession.staticGrouperSession().getSubject();
651         AttributeDef attributeDef2 = AttributeDefName.this.getAttributeDef();
652         if (!attributeDef2.getPrivilegeDelegate().canAttrAdmin(subject)) {
653           throw new InsufficientPrivilegeException(GrouperUtil
654               .subjectToString(subject)
655               + " is not attrAdmin on attributeDef: " + attributeDef2.getName());
656         }
657         
658         //we need to find all sets related to this node, and delete them (in reverse order since
659         //parents might point back)
660         Set<AttributeDefNameSet> attributeDefNameSets = GrouperDAOFactory.getFactory()
661           .getAttributeDefNameSet()
662           .findByIfThenHasAttributeDefNameId(AttributeDefName.this.id, 
663               AttributeDefName.this.id);
664 
665         List<AttributeDefNameSet> attributeDefNameSetsList = 
666           new ArrayList<AttributeDefNameSet>(attributeDefNameSets);
667         
668         //sort in reverse depth
669         Collections.sort(attributeDefNameSetsList, new Comparator<AttributeDefNameSet>() {
670           
671           /**
672            * compare two items
673            * @param first first item
674            * @param second item
675            * @return -1, 0, 1
676            */
677           public int compare(AttributeDefNameSet first, AttributeDefNameSet second) {
678             if (first == second) {
679               return 0;
680             }
681             if (first == null) {
682               return -1;
683             }
684             if (second == null) {
685               return 1;
686             }
687             return ((Integer)first.getDepth()).compareTo(second.getDepth());
688           }
689         });
690         
691         Collections.reverse(attributeDefNameSetsList);
692         
693         for(AttributeDefNameSet attributeDefNameSet : attributeDefNameSetsList) {
694           //I believe mysql has problems deleting self referential foreign keys
695           attributeDefNameSet.setParentAttrDefNameSetId(null);
696           attributeDefNameSet.saveOrUpdate();
697           attributeDefNameSet.delete();
698         }
699 
700         Set<AttributeAssign> attributeAssigns = GrouperDAOFactory.getFactory().getAttributeAssign().findByAttributeDefNameId(AttributeDefName.this.getId());
701         
702         //delete all assignments
703         for(AttributeAssign attributeAssign : attributeAssigns) {
704           attributeAssign.delete();
705         }
706         
707         GrouperDAOFactory.getFactory().getAttributeDefName().delete(AttributeDefName.this);
708         
709         if (!hibernateHandlerBean.isCallerWillCreateAudit()) {
710           AuditEntry auditEntry = new AuditEntry(AuditTypeBuiltin.ATTRIBUTE_DEF_NAME_DELETE, "id", 
711               AttributeDefName.this.getId(), "name", AttributeDefName.this.getName(), 
712               "displayName", AttributeDefName.this.getDisplayName(),
713               "description", AttributeDefName.this.getDescription(),
714               "parentStemId", AttributeDefName.this.getStemId(), 
715               "parentAttributeDefId", attributeDef2.getId(),
716               "parentAttributeDefName", attributeDef2.getName());
717           
718           auditEntry.setDescription("Deleted attributeDefName: " + AttributeDefName.this.getName());
719           auditEntry.saveOrUpdate(true);
720         }
721

1320   /**
1321    * assign different id index
1322    * @param theIdIndex
1323    * @return if it was changed
1324    */
1325   public boolean assignIdIndex(final long theIdIndex) {
1326 
1327     TableIndex.assertCanAssignIdIndex();
1328 
1329     boolean needsSave = false;
1330     synchronized (TableIndexType.attributeDefName) {
1331 
1332       //ok, if the index is not in use (not, it could be reserved... hmmm)
1333       AttributeDefName tempAttributeDefName = GrouperDAOFactory.getFactory().getAttributeDefName().findByIdIndex(theIdIndex, false, null);
1334       if (tempAttributeDefName == null) {
1335         
1336         this.setIdIndex(theIdIndex);
1337         TableIndex.clearReservedId(TableIndexType.attributeDefName, theIdIndex);
1338         needsSave = true;
1339         
1340         //do a new session so we don hold on too long
1341         HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_NEW, AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
1342           
1343           @Override
1344           public Object callback(HibernateHandlerBean hibernateHandlerBean)
1345               throws GrouperDAOException {
1346             //now we might need to increment the index
1347             TableIndex tableIndex = GrouperDAOFactory.getFactory().getTableIndex().findByType(TableIndexType.attributeDefName);
1348             if (tableIndex != null && tableIndex.getLastIndexReserved() < theIdIndex) {
1349               tableIndex.setLastIndexReserved(theIdIndex);
1350               tableIndex.saveOrUpdate();
1351             }
1352             return null;
1353           }
1354         });
1355       }      
1356     }
1357     return needsSave;
1358   }

1416   /**
1417    * @see GrouperObject#matchesLowerSearchStrings(Set)
1418    */
1419   @Override
1420   public boolean matchesLowerSearchStrings(Set<String> filterStrings) {
1421 
1422     if (GrouperUtil.length(filterStrings) == 0) {
1423       return true;
1424     }
1425 
1426     String lowerId = this.getId().toLowerCase();
1427     String lowerName = StringUtils.defaultString(this.getName()).toLowerCase();
1428     String lowerDisplayName = StringUtils.defaultString(this.getDisplayName()).toLowerCase();
1429     String lowerDescription = StringUtils.defaultString(this.getDescription()).toLowerCase();
1430     
1431     for (String filterString : GrouperUtil.nonNull(filterStrings)) {
1432       
1433       //if all dont match, return false
1434       if (!lowerId.contains(filterString)
1435           && !lowerName.contains(filterString)
1436           && !lowerDisplayName.contains(filterString)
1437           && !lowerDescription.contains(filterString)) {
1438         return false;
1439       }
1440       
1441     }
1442     return true;
1443   }
