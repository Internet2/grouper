20  package edu.internet2.middleware.grouper.audit;
21  
22  import org.apache.commons.lang.StringUtils;
23  
24  import edu.internet2.middleware.grouper.util.GrouperUtil;
25  
26  
27  
28  /**
29   * built in grouper engines.  Note, if we prefix builtins with "grouper"
30   * there will not be conflicts with custom engines
31   */
32  public enum GrouperEngineBuiltin implements GrouperEngineIdentifier {
33  
34    /** gsh engine */
35    GSH("grouperShell"), 
36    
37    /** default group ui engine */
38    UI("grouperUI"), 
39    
40    /** web service engine */
41    WS("grouperWS"), 
42  
43    /** loader engine */
44    LOADER("grouperLoader"), 
45  
46    /** ldappc engine */
47    LDAPPC("grouperLdappc"),
48  
49    /** ldappc engine */
50    IMPORT("grouperImport"),
51  
52    /** usdu engine engine */
53    USDU("grouperUsdu"),
54  
55    /** junit engine */
56    JUNIT("grouperJunit");
57  
58    /**
59     * have the label different from name
60     * @param theGrouperEngine
61     */
62    private GrouperEngineBuiltin(String theGrouperEngine) {
63      this.grouperEngine = theGrouperEngine;
64    }
65    
66    /**
67     * string that goes to the DB
68     */
69    private String grouperEngine;
70  
71    /**
72     * @see edu.internet2.middleware.grouper.audit.GrouperEngineIdentifier#getGrouperEngine()
73     */
74    public String getGrouperEngine() {
75      return this.grouperEngine;
76    }
77    
78    /**
79     * do a case-insensitive matching
80     * 
81     * @param string
82     * @param exceptionOnNull will not allow null or blank entries
83     * @return the enum or null or exception if not found
84     */
85    public static GrouperEngineBuiltin valueOfIgnoreCase(String string, boolean exceptionOnNull) {
86      
87      return valueOfIgnoreCase(string, exceptionOnNull, true);
88    }
89  
90    /**
91     * do a case-insensitive matching
92     * 
93     * @param string
94     * @param exceptionOnNull will not allow null or blank entries
95     * @param exceptionIfNotFound if string isnt found should there be an exception
96     * @return the enum or null or exception if not found
97     */
98    public static GrouperEngineBuiltin valueOfIgnoreCase(String string, boolean exceptionOnNull, boolean exceptionIfNotFound) {
99      
100     //check the string
101     for (GrouperEngineBuiltin grouperEngineBuiltin : values()) {
102       if (StringUtils.equalsIgnoreCase(string, grouperEngineBuiltin.grouperEngine)) {
103         return grouperEngineBuiltin;
104       }
105     }
106     
107     return GrouperUtil.enumValueOfIgnoreCase(GrouperEngineBuiltin.class, 
108         string, exceptionOnNull, exceptionIfNotFound);
109 
110   }

112 }
