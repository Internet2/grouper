20  package edu.internet2.middleware.grouper.esb.listener;

22  import java.io.IOException;
23  
24  import org.apache.commons.logging.Log;
25  import org.mortbay.jetty.Handler;
26  import org.mortbay.jetty.Server;
27  import org.mortbay.jetty.nio.SelectChannelConnector;
28  import org.mortbay.jetty.security.Constraint;
29  import org.mortbay.jetty.security.ConstraintMapping;
30  import org.mortbay.jetty.security.HashUserRealm;
31  import org.mortbay.jetty.security.SecurityHandler;
32  import org.mortbay.jetty.security.SslSocketConnector;
33  import org.quartz.Job;
34  import org.quartz.JobDataMap;
35  import org.quartz.JobExecutionContext;
36  import org.quartz.JobExecutionException;
37  
38  import edu.internet2.middleware.grouper.util.GrouperUtil;

40  /**
41   * 
42   * Class to start a simple HTTP/HTTPS server to listen
43   * for incoming event notifications to process as changes in the
44   * Grouper registry. Server is started as a quartz job
45   * <p>
46   * SSL and basic auth supported for security. All configuration in grouper-loader.properties
47   *
48   */
49  
50  public class EsbHttpServer implements Job {
51  
52    private static final Log LOG = GrouperUtil.getLog(EsbHttpServer.class);
53  
54    /**
55     * Method to start the Jetty server
56     * @param jobDataMap
57     */
58    private void startServer(JobDataMap jobDataMap) {
59      if (LOG.isDebugEnabled()) {
60        LOG.debug("Initialising HTTP server");
61      }
62      int port = Integer.parseInt(jobDataMap.getString("port"));
63      String bindAddress = jobDataMap.getString("bindAddress");
64      String authConfigFile = jobDataMap.getString("authConfigFile");
65      String keystore = jobDataMap.getString("keystore");
66      Server server = new Server();
67      if (keystore == null || keystore.equals("")) {
68        LOG.info("Starting with HTTP (non-encrypted) protocol");
69        SelectChannelConnector connector = new SelectChannelConnector();
70        connector.setHost(bindAddress);
71        connector.setPort(port);
72        server.addConnector(connector);
73      } else {
74        LOG.info("Starting with HTTPS (encrypted) protocol");
75        SslSocketConnector sslConnector = new SslSocketConnector();
76        sslConnector.setHost(bindAddress);
77        sslConnector.setPort(port);
78        sslConnector.setKeystore(jobDataMap.getString("keystore"));
79        sslConnector.setKeyPassword(jobDataMap.getString("keyPassword"));
80        sslConnector.setTruststore(jobDataMap.getString("trustStore"));
81        sslConnector.setTrustPassword(jobDataMap.getString("trustPassword"));
82        sslConnector.setPassword(jobDataMap.getString("password"));
83        server.addConnector(sslConnector);
84      }
85  
86      if (authConfigFile != null && !(authConfigFile.equals(""))) {
87        if (LOG.isDebugEnabled()) {
88          LOG.debug("Requiring basic auth");
89        }
90        Constraint constraint = new Constraint();
91        constraint.setName(Constraint.__BASIC_AUTH);
92        ;
93        constraint.setRoles(new String[] { "user", "grouper" });
94        constraint.setAuthenticate(true);
95  
96        ConstraintMapping cm = new ConstraintMapping();
97        cm.setConstraint(constraint);
98        cm.setPathSpec("/*");
99  
100       SecurityHandler sh = new SecurityHandler();
101       try {
102         sh.setUserRealm(new HashUserRealm("Grouper", authConfigFile));
103       } catch (IOException e) {
104         // TODO Auto-generated catch block
105         e.printStackTrace();
106       }
107       sh.setConstraintMappings(new ConstraintMapping[] { cm });
108 
109       Handler[] handlers = new Handler[] { sh, new EsbHttpHandler() };
110       server.setHandlers(handlers);
111 
112     } else {
113       if (LOG.isDebugEnabled()) {
114         LOG.debug("Not requiring basic auth");
115       }
116       server.setHandler(new EsbHttpHandler());
117     }
118     try {
119       server.start();
120       LOG.info("HTTP server started on address " + bindAddress + " port " + port);
121       server.join();
122     } catch (Exception e) {
123       e.printStackTrace();
124     }
125 
126   }
127 
128   /**
129    * Method called by quartz to start the server
130    */
131   public void execute(JobExecutionContext context)
132       throws JobExecutionException {
133     // TODO Auto-generated method stub
134     JobDataMap jobDataMap = context.getJobDetail().getJobDataMap();
135     this.startServer(jobDataMap);
136   }
137 }
