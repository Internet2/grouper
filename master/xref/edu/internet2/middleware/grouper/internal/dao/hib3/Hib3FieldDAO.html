64    /**
65     * @param name 
66     * @return if exists
67     * @throws GrouperDAOException 
68     * @since   @HEAD@
69     */
70    public boolean existsByName(String name) 
71      throws  GrouperDAOException
72    {
73      Object id = null;
74      
75      try {
76        id = HibernateSession.byHqlStatic()
77        .createQuery("select f.id from Field f where f.name = :name")
78        .setString("name", name).uniqueResult(Object.class);
79      } catch (GrouperDAOException gde) {
80        Throwable throwable = gde.getCause();
81        //CH 20080218 this was legacy error handling
82        if (throwable instanceof HibernateException) {
83          LOG.fatal( throwable.getMessage() );
84        }
85        throw gde;
86      }
87      boolean rv  = false;
88      if ( id != null ) {
89        rv = true; 
90      }
91      return rv;
92    } // public boolean existsByName(name)
93    
94    /**
95     * @return set of fields
96     * @throws GrouperException 
97     * @since   @HEAD@
98     */
99    public Set<Field> findAll() 
100     throws  GrouperException
101   {
102     return HibernateSession.byHqlStatic()
103       .createQuery("from Field order by name asc")
104       .setCacheable(true)
105       .setCacheRegion(KLASS + ".FindAll").listSet(Field.class);
106   } // public Set findAll()

126   /**
127    * @param f 
128    * @return if in use
129    * @throws GrouperDAOException 
130    * @throws SchemaException 
131    * @since   @HEAD@
132    */
133   public boolean isInUse(Field f) 
134     throws  GrouperDAOException,
135             SchemaException
136   {
137     ByHqlStatic qry = HibernateSession.byHqlStatic();
138     if ( f.getType().equals(FieldType.LIST) )      {
139       qry.createQuery("select ms from MembershipEntry as ms, Field as field where field.name = :name and field.uuid = ms.fieldId");
140     } else {
141       throw new SchemaException( f.getType().toString() );
142     }
143     qry.setCacheable(false);
144     qry.setString("name", f.getName() );
145     if (qry.list(Object.class).size() > 0) {
146       return true;
147     }
148     return false;
149   } // public boolean isInUse(f)

159   /**
160    * @see edu.internet2.middleware.grouper.internal.dao.FieldDAO#findByUuidOrName(java.lang.String, java.lang.String, boolean)
161    */
162   public Field findByUuidOrName(String uuid, String name, 
163       boolean exceptionIfNull) throws GrouperDAOException {
164     return findByUuidOrName(uuid, name, exceptionIfNull, null);
165   }
166 
167   /**
168    * @see edu.internet2.middleware.grouper.internal.dao.FieldDAO#findByUuidOrName(java.lang.String, java.lang.String, boolean, QueryOptions)
169    */
170   public Field findByUuidOrName(String uuid, String name,
171       boolean exceptionIfNull, QueryOptions queryOptions) throws GrouperDAOException {
172     try {
173       Field field = HibernateSession.byHqlStatic()
174         .createQuery("from Field as theField where theField.uuid = :uuid or theField.name = :name")
175         .setCacheable(true)
176         .setCacheRegion(KLASS + ".FindByUuidOrName")
177         .options(queryOptions)
178         .setString("uuid", uuid)
179         .setString("name", name)
180         .uniqueResult(Field.class);
181       if (field == null && exceptionIfNull) {
182         throw new RuntimeException("Can't find field by uuid: '" + uuid + "' or name '" + name + "'");
183       }
184       return field;
185     }
186     catch (GrouperDAOException e) {
187       String error = "Problem find field by uuid: '" 
188         + uuid + "' or name '" + name + "', " + e.getMessage();
189       throw new GrouperDAOException( error, e );
190     }
191   }

193   /**
194    * @see edu.internet2.middleware.grouper.internal.dao.FieldDAO#saveUpdateProperties(edu.internet2.middleware.grouper.Field)
195    */
196   public void saveUpdateProperties(Field field) {
197     //run an update statement since the business methods affect these properties
198     HibernateSession.byHqlStatic().createQuery("update Field " +
199         "set hibernateVersionNumber = :theHibernateVersionNumber, " +
200         "contextId = :theContextId " +
201         "where uuid = :theUuid")
202         .setLong("theHibernateVersionNumber", field.getHibernateVersionNumber())
203         .setString("theContextId", field.getContextId())
204         .setString("theUuid", field.getUuid())
205         .executeUpdate();
206   }

230    *

232    * @throws HibernateException
233    */
234   protected static void reset(HibernateSession hibernateSession)
235     throws  HibernateException {
236 
237     //delete custom lists
238     hibernateSession.byHql().createQuery("delete from Field field where typeString='list' and name <> 'members'").executeUpdate();   
239   }
