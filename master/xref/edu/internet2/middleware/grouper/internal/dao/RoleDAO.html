22  import edu.internet2.middleware.grouper.exception.RoleNotFoundException;
23  import edu.internet2.middleware.grouper.permissions.role.Role;
24  
25  /**
26   * role data access methods
27   */
28  public interface RoleDAO extends GrouperDAO {
29    
30    /** 
31     * insert or update a role object 
32     * @param role
33     */
34    public void saveOrUpdate(Role role);
35    
36    /**
37     * @param id
38     * @param exceptionIfNotFound
39     * @return the role or null if not there
40     */
41    public Role findById(String id, boolean exceptionIfNotFound);
42    
43    /**
44     * find an attribute def name by name
45     * @param name 
46     * @param exceptionIfNotFound 
47     * @return  name
48     * @throws GrouperDAOException 
49     * @throws RoleNotFoundException 
50     */
51    public Role findByName(String name, boolean exceptionIfNotFound) 
52      throws GrouperDAOException, RoleNotFoundException;
53  
54  }
