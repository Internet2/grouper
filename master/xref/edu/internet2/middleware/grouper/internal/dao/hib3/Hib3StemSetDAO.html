55    /**
56     * @param hibernateSession
57     */
58    static void reset(HibernateSession hibernateSession) {
59      
60      if (GrouperDdlUtils.isMysql() || GrouperDdlUtils.isHsql()) {
61        //do this since mysql cant handle self-referential foreign keys
62        // restrict this only to mysql since in oracle this might cause unique constraint violations
63        hibernateSession.byHql().createQuery("update StemSet set parentStemSetId = null where ifHasStemId not in (select uuid from Stem where name = ':')").executeUpdate();
64      }
65      
66      hibernateSession.byHql().createQuery("delete from StemSet where ifHasStemId not in (select uuid from Stem where name = ':')").executeUpdate();
67    }

108   /**
109    * @see edu.internet2.middleware.grouper.internal.dao.StemSetDAO#findByIfHasStemIds(java.util.Collection)
110    */
111   public Set<StemSet> findByIfHasStemIds(Collection ids) {
112     int idsSize = GrouperUtil.length(ids);
113     
114     Set<StemSet> results = new HashSet<StemSet>();
115     
116     if (idsSize == 0) {
117       return results;
118     }
119     
120     List<String> idsList = new ArrayList<String>(ids);
121     
122     int numberOfBatches = GrouperUtil.batchNumberOfBatches(idsSize, 100);
123     
124     //if there are more than 100, batch these up and return them
125     for (int i = 0; i < numberOfBatches; i++) {
126       
127       ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
128 
129       StringBuilder sql = new StringBuilder("from StemSet where ");
130       
131       List<String> currentBatch = GrouperUtil.batchList(idsList, 100, i);
132       
133       sql.append(" ifHasStemId in (");
134       sql.append(HibUtils.convertToInClause(currentBatch, byHqlStatic));
135       sql.append(") ");
136     
137       Set<StemSet> localResult = byHqlStatic.createQuery(sql.toString())
138         .listSet(StemSet.class);
139       
140       results.addAll(localResult);
141     }
142 
143     return results;
144   }

168   /**
169    * @see edu.internet2.middleware.grouper.internal.dao.StemSetDAO#delete(edu.internet2.middleware.grouper.stem.StemSet)
170    */
171   public void delete(final StemSet stemSet) {
172     HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, 
173         AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
174 
175           public Object callback(HibernateHandlerBean hibernateHandlerBean)
176               throws GrouperDAOException {
177             
178             hibernateHandlerBean.getHibernateSession().setCachingEnabled(false);
179 
180             if (stemSet.getParentStemSetId() != null && (GrouperDdlUtils.isMysql() || GrouperDdlUtils.isHsql())) {
181               //set parent to null so mysql doest get mad
182               //http://bugs.mysql.com/bug.php?id=15746
183               hibernateHandlerBean.getHibernateSession().byHql()
184                 .createQuery("update StemSet set parentStemSetId = null where id = :id")
185                 .setString("id", stemSet.getId()).executeUpdate();
186             }
187             
188             hibernateHandlerBean.getHibernateSession().byObject().delete(stemSet);
189             return null;
190           }
191     });
192   }

197   public StemSet findByIfThenImmediate(String stemIdIf, String stemIdThen, boolean exceptionIfNotFound) 
198     throws StemSetNotFoundException {
199     
200     StemSet stemSet = HibernateSession.byHqlStatic()
201       .createQuery("from StemSet where ifHasStemId = :ifId and thenHasStemId = :thenId and depth = 1")
202       .setString("ifId", stemIdIf)
203       .setString("thenId", stemIdThen)
204       .uniqueResult(StemSet.class);
205     
206     if (stemSet == null && exceptionIfNotFound) {
207       throw new StemSetNotFoundException("StemSet immediate if " + stemIdIf + ", then: " + stemIdThen);
208     }
209     
210     return stemSet;
211   }

214    * @see edu.internet2.middleware.grouper.internal.dao.StemSetDAO#findAllChildren(java.util.Collection, edu.internet2.middleware.grouper.internal.dao.QueryOptions)
215    */
216   public Set<StemSet> findAllChildren(Collection<StemSet> stemSets, QueryOptions queryOptions) {
217     
218     ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
219     Set<String> stemSetIds = new LinkedHashSet<String>();
220     for (StemSet stemSet : stemSets) {
221       stemSetIds.add(stemSet.getId());
222     }
223     
224     StringBuilder sql = new StringBuilder();
225     sql.append("select ss from StemSet ss where ss.parentStemSetId in (");
226     sql.append(HibUtils.convertToInClause(stemSetIds, byHqlStatic));
227     sql.append(") ");
228     
229     Set<StemSet> result = byHqlStatic
230       .createQuery(sql.toString())
231       .options(queryOptions)
232       .listSet(StemSet.class);
233     
234     return result; 
235   }

251   /**
252    * @see edu.internet2.middleware.grouper.internal.dao.StemSetDAO#deleteByIfHasStemId(java.lang.String)
253    */
254   public void deleteByIfHasStemId(final String id) {
255     HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, 
256         AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
257 
258           public Object callback(HibernateHandlerBean hibernateHandlerBean)
259               throws GrouperDAOException {
260             
261             hibernateHandlerBean.getHibernateSession().setCachingEnabled(false);
262 
263             if (GrouperDdlUtils.isMysql() || GrouperDdlUtils.isHsql()) {
264               //set parent to null so mysql doest get mad
265               //http://bugs.mysql.com/bug.php?id=15746
266               hibernateHandlerBean.getHibernateSession().byHql()
267                 .createQuery("update StemSet set parentStemSetId = null where ifHasStemId = :id")
268                 .setString("id", id).executeUpdate();
269             }
270             
271             List<StemSet> stemSets = new ArrayList<StemSet>(findByIfHasStemId(id));
272             
273             // sort by depth desc
274             Collections.sort(stemSets, new Comparator<StemSet>() {
275 
276               public int compare(StemSet o1, StemSet o2) {
277                 return ((Integer)o2.getDepth()).compareTo(o1.getDepth());
278               }
279             });
280             
281             for (StemSet stemSet : stemSets) {
282               stemSet.delete();
283             }
284             
285             return null;
286           }
287     });
288   }
