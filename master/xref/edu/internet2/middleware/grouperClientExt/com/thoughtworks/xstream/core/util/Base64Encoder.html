27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.core.util;
28  
29  import java.io.ByteArrayOutputStream;
30  import java.io.IOException;
31  import java.io.Reader;
32  import java.io.StringReader;
33  
34  /**
35   * Encodes binary data to plain text as Base64.
36   *
37   * <p>Despite there being a gazillion other Base64 implementations out there, this has been written as part of XStream as
38   * it forms a core part but is too trivial to warrant an extra dependency.</p>
39   *
40   * <p>This meets the standard as described in RFC 1521, section 5.2 <http://www.freesoft.org/CIE/RFC/1521/7.htm>, allowing
41   * other Base64 tools to manipulate the data.</p>
42   *
43   * @author Joe Walnes
44   */
45  public class Base64Encoder {
46  
47      // Here's how encoding works:
48      //
49      // 1) Incoming bytes are broken up into groups of 3 (each byte having 8 bits).
50      //
51      // 2) The combined 24 bits (3 * 8) are split into 4 groups of 6 bits.
52      //
53      // input  |------||------||------| (3 values each with 8 bits)
54      //        101010101010101010101010
55      // output |----||----||----||----| (4 values each with 6 bits)
56      //
57      // 3) Each of these 4 groups of 6 bits are converted back to a number, which will fall in the range of 0 - 63.
58      //
59      // 4) Each of these 4 numbers are converted to an alphanumeric char in a specified mapping table, to create
60      //    a 4 character string.
61      //
62      // 5) This is repeated for all groups of three bytes.
63      //
64      // 6) Special padding is done at the end of the stream using the '=' char.
65  
66      private static final char[] SIXTY_FOUR_CHARS = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/".toCharArray();
67      private static final int[] REVERSE_MAPPING = new int[123];
68  
69      static {
70          for (int i = 0; i < SIXTY_FOUR_CHARS.length; i++) REVERSE_MAPPING[SIXTY_FOUR_CHARS[i]] = i + 1;
71      }
72  
73      public String encode(byte[] input) {
74          StringBuffer result = new StringBuffer();
75          int outputCharCount = 0;
76          for (int i = 0; i < input.length; i += 3) {
77              int remaining = Math.min(3, input.length - i);
78              int oneBigNumber = (input[i] & 0xff) << 16 | (remaining <= 1 ? 0 : input[i + 1] & 0xff) << 8 | (remaining <= 2 ? 0 : input[i + 2] & 0xff);
79              for (int j = 0; j < 4; j++) result.append(remaining + 1 > j ? SIXTY_FOUR_CHARS[0x3f & oneBigNumber >> 6 * (3 - j)] : '=');
80              if ((outputCharCount += 4) % 76 == 0) result.append('\n');
81          }
82          return result.toString();
83      }
84  
85      public byte[] decode(String input) {
86          try {
87              ByteArrayOutputStream out = new ByteArrayOutputStream();
88              StringReader in = new StringReader(input);
89              for (int i = 0; i < input.length(); i += 4) {
90                  int a[] = {mapCharToInt(in), mapCharToInt(in), mapCharToInt(in), mapCharToInt(in)};
91                  int oneBigNumber = (a[0] & 0x3f) << 18 | (a[1] & 0x3f) << 12 | (a[2] & 0x3f) << 6 | (a[3] & 0x3f);
92                  for (int j = 0; j < 3; j++) if (a[j + 1] >= 0) out.write(0xff & oneBigNumber >> 8 * (2 - j));
93              }
94              return out.toByteArray();
95          } catch (IOException e) {
96              throw new Error(e + ": " + e.getMessage());
97          }
98      }
99  
100     private int mapCharToInt(Reader input) throws IOException {
101         int c;
102         while ((c = input.read()) != -1) {
103             int result = REVERSE_MAPPING[c];
104             if (result != 0) return result -1;
105             if (c == '=') return -1;
106         }
107         return -1;
108     }
109 }
