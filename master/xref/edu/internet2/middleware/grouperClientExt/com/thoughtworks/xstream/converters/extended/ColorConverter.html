33  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamReader;
34  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamWriter;

36  import java.awt.Color;
37  import java.util.HashMap;
38  import java.util.Map;
39  
40  /**
41   * Converts a java.awt.Color to XML, using four nested elements:
42   * red, green, blue, alpha.
43   *
44   * @author Joe Walnes
45   */
46  public class ColorConverter implements Converter {
47  
48      public boolean canConvert(Class type) {
49          // String comparison is used here because Color.class loads the class which in turns instantiates AWT,
50          // which is nasty if you don't want it.
51          return type.getName().equals("java.awt.Color");
52      }
53  
54      public void marshal(Object source, HierarchicalStreamWriter writer, MarshallingContext context) {
55          Color color = (Color) source;
56          write("red", color.getRed(), writer);
57          write("green", color.getGreen(), writer);
58          write("blue", color.getBlue(), writer);
59          write("alpha", color.getAlpha(), writer);
60      }
61  
62      public Object unmarshal(HierarchicalStreamReader reader, UnmarshallingContext context) {
63          Map elements = new HashMap();
64          while (reader.hasMoreChildren()) {
65              reader.moveDown();
66              elements.put(reader.getNodeName(), Integer.valueOf(reader.getValue()));
67              reader.moveUp();
68          }
69          return new Color(((Integer) elements.get("red")).intValue(),
70                  ((Integer) elements.get("green")).intValue(),
71                  ((Integer) elements.get("blue")).intValue(),
72                  ((Integer) elements.get("alpha")).intValue());
73      }
74  
75      private void write(String fieldName, int value, HierarchicalStreamWriter writer) {
76          ExtendedHierarchicalStreamWriterHelper.startNode(writer, fieldName, int.class);
77          writer.setValue(String.valueOf(value));
78          writer.endNode();
79      }
80  
81  }
