39  /**
40   * <p>
41   * An utility class providing functionality related to the {@code
42   * java.util.concurrent} package.
43   * </p>
44   *
45   * @since 3.0
46   * @version $Id: ConcurrentUtils.java 1199894 2011-11-09 17:53:59Z ggregory $
47   */
48  public class ConcurrentUtils {
49  
50      /**
51       * Private constructor so that no instances can be created. This class
52       * contains only static utility methods.
53       */
54      private ConcurrentUtils() {
55      }
56  
57      /**
58       * Inspects the cause of the specified {@code ExecutionException} and
59       * creates a {@code ConcurrentException} with the checked cause if
60       * necessary. This method performs the following checks on the cause of the
61       * passed in exception:
62       * <ul>
63       * <li>If the passed in exception is <b>null</b> or the cause is
64       * <b>null</b>, this method returns <b>null</b>.</li>
65       * <li>If the cause is a runtime exception, it is directly thrown.</li>
66       * <li>If the cause is an error, it is directly thrown, too.</li>
67       * <li>In any other case the cause is a checked exception. The method then
68       * creates a {@link ConcurrentException}, initializes it with the cause, and
69       * returns it.</li>
70       * </ul>
71       *
72       * @param ex the exception to be processed
73       * @return a {@code ConcurrentException} with the checked cause
74       */
75      public static ConcurrentException extractCause(ExecutionException ex) {
76          if (ex == null || ex.getCause() == null) {
77              return null;
78          }
79  
80          throwCause(ex);
81          return new ConcurrentException(ex.getMessage(), ex.getCause());
82      }
83  
84      /**
85       * Inspects the cause of the specified {@code ExecutionException} and
86       * creates a {@code ConcurrentRuntimeException} with the checked cause if
87       * necessary. This method works exactly like
88       * {@link #extractCause(ExecutionException)}. The only difference is that
89       * the cause of the specified {@code ExecutionException} is extracted as a
90       * runtime exception. This is an alternative for client code that does not
91       * want to deal with checked exceptions.
92       *
93       * @param ex the exception to be processed
94       * @return a {@code ConcurrentRuntimeException} with the checked cause
95       */
96      public static ConcurrentRuntimeException extractCauseUnchecked(
97              ExecutionException ex) {
98          if (ex == null || ex.getCause() == null) {
99              return null;
100         }
101 
102         throwCause(ex);
103         return new ConcurrentRuntimeException(ex.getMessage(), ex.getCause());
104     }
105 
106     /**
107      * Handles the specified {@code ExecutionException}. This method calls
108      * {@link #extractCause(ExecutionException)} for obtaining the cause of the
109      * exception - which might already cause an unchecked exception or an error
110      * being thrown. If the cause is a checked exception however, it is wrapped
111      * in a {@code ConcurrentException}, which is thrown. If the passed in
112      * exception is <b>null</b> or has no cause, the method simply returns
113      * without throwing an exception.
114      *
115      * @param ex the exception to be handled
116      * @throws ConcurrentException if the cause of the {@code
117      * ExecutionException} is a checked exception
118      */
119     public static void handleCause(ExecutionException ex)
120             throws ConcurrentException {
121         ConcurrentException cex = extractCause(ex);
122 
123         if (cex != null) {
124             throw cex;
125         }
126     }
127 
128     /**
129      * Handles the specified {@code ExecutionException} and transforms it into a
130      * runtime exception. This method works exactly like
131      * {@link #handleCause(ExecutionException)}, but instead of a
132      * {@link ConcurrentException} it throws a
133      * {@link ConcurrentRuntimeException}. This is an alternative for client
134      * code that does not want to deal with checked exceptions.
135      *
136      * @param ex the exception to be handled
137      * @throws ConcurrentRuntimeException if the cause of the {@code
138      * ExecutionException} is a checked exception; this exception is then
139      * wrapped in the thrown runtime exception
140      */
141     public static void handleCauseUnchecked(ExecutionException ex) {
142         ConcurrentRuntimeException crex = extractCauseUnchecked(ex);
143 
144         if (crex != null) {
145             throw crex;
146         }
147     }
148 
149     /**
150      * Tests whether the specified {@code Throwable} is a checked exception. If
151      * not, an exception is thrown.
152      *
153      * @param ex the {@code Throwable} to check
154      * @return a flag whether the passed in exception is a checked exception
155      * @throws IllegalArgumentException if the {@code Throwable} is not a
156      * checked exception
157      */
158     static Throwable checkedException(Throwable ex) {
159         if (ex != null && !(ex instanceof RuntimeException)
160                 && !(ex instanceof Error)) {
161             return ex;
162         } else {
163             throw new IllegalArgumentException("Not a checked exception: " + ex);
164         }
165     }
166 
167     /**
168      * Tests whether the cause of the specified {@code ExecutionException}
169      * should be thrown and does it if necessary.
170      *
171      * @param ex the exception in question
172      */
173     private static void throwCause(ExecutionException ex) {
174         if (ex.getCause() instanceof RuntimeException) {
175             throw (RuntimeException) ex.getCause();
176         }
177 
178         if (ex.getCause() instanceof Error) {
179             throw (Error) ex.getCause();
180         }
181     }

183     //-----------------------------------------------------------------------
184     /**
185      * Invokes the specified {@code ConcurrentInitializer} and returns the
186      * object produced by the initializer. This method just invokes the {@code
187      * get()} method of the given {@code ConcurrentInitializer}. It is
188      * <b>null</b>-safe: if the argument is <b>null</b>, result is also
189      * <b>null</b>.
190      *
191      * @param <T> the type of the object produced by the initializer
192      * @param initializer the {@code ConcurrentInitializer} to be invoked
193      * @return the object managed by the {@code ConcurrentInitializer}
194      * @throws ConcurrentException if the {@code ConcurrentInitializer} throws
195      * an exception
196      */
197     public static <T> T initialize(ConcurrentInitializer<T> initializer)
198             throws ConcurrentException {
199         return initializer != null ? initializer.get() : null;
200     }
201 
202     /**
203      * Invokes the specified {@code ConcurrentInitializer} and transforms
204      * occurring exceptions to runtime exceptions. This method works like
205      * {@link #initialize(ConcurrentInitializer)}, but if the {@code
206      * ConcurrentInitializer} throws a {@link ConcurrentException}, it is
207      * caught, and the cause is wrapped in a {@link ConcurrentRuntimeException}.
208      * So client code does not have to deal with checked exceptions.
209      *
210      * @param <T> the type of the object produced by the initializer
211      * @param initializer the {@code ConcurrentInitializer} to be invoked
212      * @return the object managed by the {@code ConcurrentInitializer}
213      * @throws ConcurrentRuntimeException if the initializer throws an exception
214      */
215     public static <T> T initializeUnchecked(ConcurrentInitializer<T> initializer) {
216         try {
217             return initialize(initializer);
218         } catch (ConcurrentException cex) {
219             throw new ConcurrentRuntimeException(cex.getCause());
220         }
221     }

223     //-----------------------------------------------------------------------
224     /**
225      * <p>
226      * Puts a value in the specified {@code ConcurrentMap} if the key is not yet
227      * present. This method works similar to the {@code putIfAbsent()} method of
228      * the {@code ConcurrentMap} interface, but the value returned is different.
229      * Basically, this method is equivalent to the following code fragment:
230      *
231      * <pre>
232      * if (!map.containsKey(key)) {
233      *     map.put(key, value);
234      *     return value;
235      * } else {
236      *     return map.get(key);
237      * }
238      * </pre>
239      *
240      * except that the action is performed atomically. So this method always
241      * returns the value which is stored in the map.
242      * </p>
243      * <p>
244      * This method is <b>null</b>-safe: It accepts a <b>null</b> map as input
245      * without throwing an exception. In this case the return value is
246      * <b>null</b>, too.
247      * </p>
248      *
249      * @param <K> the type of the keys of the map
250      * @param <V> the type of the values of the map
251      * @param map the map to be modified
252      * @param key the key of the value to be added
253      * @param value the value to be added
254      * @return the value stored in the map after this operation
255      */
256     public static <K, V> V putIfAbsent(ConcurrentMap<K, V> map, K key, V value) {
257         if (map == null) {
258             return null;
259         }
260 
261         V result = map.putIfAbsent(key, value);
262         return result != null ? result : value;
263     }
264 
265     /**
266      * Checks if a concurrent map contains a key and creates a corresponding
267      * value if not. This method first checks the presence of the key in the
268      * given map. If it is already contained, its value is returned. Otherwise
269      * the {@code get()} method of the passed in {@link ConcurrentInitializer}
270      * is called. With the resulting object
271      * {@link #putIfAbsent(ConcurrentMap, Object, Object)} is called. This
272      * handles the case that in the meantime another thread has added the key to
273      * the map. Both the map and the initializer can be <b>null</b>; in this
274      * case this method simply returns <b>null</b>.
275      *
276      * @param <K> the type of the keys of the map
277      * @param <V> the type of the values of the map
278      * @param map the map to be modified
279      * @param key the key of the value to be added
280      * @param init the {@link ConcurrentInitializer} for creating the value
281      * @return the value stored in the map after this operation; this may or may
282      * not be the object created by the {@link ConcurrentInitializer}
283      * @throws ConcurrentException if the initializer throws an exception
284      */
285     public static <K, V> V createIfAbsent(ConcurrentMap<K, V> map, K key,
286             ConcurrentInitializer<V> init) throws ConcurrentException {
287         if (map == null || init == null) {
288             return null;
289         }
290 
291         V value = map.get(key);
292         if (value == null) {
293             return putIfAbsent(map, key, init.get());
294         }
295         return value;
296     }
297 
298     /**
299      * Checks if a concurrent map contains a key and creates a corresponding
300      * value if not, suppressing checked exceptions. This method calls
301      * {@code createIfAbsent()}. If a {@link ConcurrentException} is thrown, it
302      * is caught and re-thrown as a {@link ConcurrentRuntimeException}.
303      *
304      * @param <K> the type of the keys of the map
305      * @param <V> the type of the values of the map
306      * @param map the map to be modified
307      * @param key the key of the value to be added
308      * @param init the {@link ConcurrentInitializer} for creating the value
309      * @return the value stored in the map after this operation; this may or may
310      * not be the object created by the {@link ConcurrentInitializer}
311      * @throws ConcurrentRuntimeException if the initializer throws an exception
312      */
313     public static <K, V> V createIfAbsentUnchecked(ConcurrentMap<K, V> map,
314             K key, ConcurrentInitializer<V> init) {
315         try {
316             return createIfAbsent(map, key, init);
317         } catch (ConcurrentException cex) {
318             throw new ConcurrentRuntimeException(cex.getCause());
319         }
320     }

322     //-----------------------------------------------------------------------
323     /**
324      * <p>
325      * Gets an implementation of <code>Future</code> that is immediately done
326      * and returns the specified constant value.
327      * </p>
328      * <p>
329      * This can be useful to return a simple constant immediately from the
330      * concurrent processing, perhaps as part of avoiding nulls.
331      * A constant future can also be useful in testing.
332      * </p>
333      *
334      * @param <T> the type of the value used by this {@code Future} object
335      * @param value  the constant value to return, may be null
336      * @return an instance of Future that will return the value, never null
337      */
338     public static <T> Future<T> constantFuture(T value) {
339         return new ConstantFuture<T>(value);
340     }
341 
342     /**
343      * A specialized {@code Future} implementation which wraps a constant value.
344      * @param <T> the type of the value wrapped by this class
345      */
346     static final class ConstantFuture<T> implements Future<T> {
347         /** The constant value. */
348         private final T value;
349 
350         /**
351          * Creates a new instance of {@code ConstantFuture} and initializes it
352          * with the constant value.
353          *
354          * @param value the value (may be <b>null</b>)
355          */
356         ConstantFuture(T value) {
357             this.value = value;
358         }
359 
360         /**
361          * {@inheritDoc} This implementation always returns <b>true</b> because
362          * the constant object managed by this {@code Future} implementation is
363          * always available.
364          */
365         public boolean isDone() {
366             return true;
367         }
368 
369         /**
370          * {@inheritDoc} This implementation just returns the constant value.
371          */
372         public T get() {
373             return value;
374         }
375 
376         /**
377          * {@inheritDoc} This implementation just returns the constant value; it
378          * does not block, therefore the timeout has no meaning.
379          */
380         public T get(long timeout, TimeUnit unit) {
381             return value;
382         }
383 
384         /**
385          * {@inheritDoc} This implementation always returns <b>false</b>; there
386          * is no background process which could be cancelled.
387          */
388         public boolean isCancelled() {
389             return false;
390         }
391 
392         /**
393          * {@inheritDoc} The cancel operation is not supported. This
394          * implementation always returns <b>false</b>.
395          */
396         public boolean cancel(boolean mayInterruptIfRunning) {
397             return false;
398         }
399     }
