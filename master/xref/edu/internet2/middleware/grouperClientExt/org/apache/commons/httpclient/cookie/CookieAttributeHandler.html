45  package edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.cookie;
46  
47  import edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie;
48  
49  /**
50   * Ths interface represents a cookie attribute handler responsible
51   * for parsing, validating, and matching a specific cookie attribute, 
52   * such as path, domain, port, etc.
53   *
54   * Different cookie specifications can provide a specific
55   * implementation for this class based on their cookie handling
56   * rules.
57   *
58   * @author jain.samit@gmail.com (Samit Jain)
59   * 
60   * @since 3.1
61   */
62  public interface CookieAttributeHandler {
63  
64    /**
65     * Parse the given cookie attribute value and update the corresponding
66     * {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} property.
67     *
68     * @param cookie {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} to be updated
69     * @param value cookie attribute value from the cookie response header
70     */
71    void parse(Cookie cookie, String value)
72            throws MalformedCookieException;
73  
74    /**
75     * Peforms cookie validation for the given attribute value.
76     *
77     * @param cookie {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} to validate
78     * @param origin the cookie source to validate against
79     * @throws MalformedCookieException if cookie validation fails for this attribute
80     */
81    void validate(Cookie cookie, CookieOrigin origin)
82            throws MalformedCookieException;
83  
84    /**
85     * Matches the given value (property of the destination host where request is being
86     * submitted) with the corresponding cookie attribute.
87     *
88     * @param cookie {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} to match
89     * @param origin the cookie source to match against
90     * @return <tt>true</tt> if the match is successful; <tt>false</tt> otherwise
91     */
92    boolean match(Cookie cookie, CookieOrigin origin);
93  
94  }
