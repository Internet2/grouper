32  package edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.reflect;
33  
34  import java.lang.reflect.AccessibleObject;
35  import java.lang.reflect.Member;
36  import java.lang.reflect.Modifier;
37  
38  import edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.ClassUtils;
39  
40  /**
41   * Contains common code for working with Methods/Constructors, extracted and
42   * refactored from <code>MethodUtils</code> when it was imported from Commons
43   * BeanUtils.
44   *
45   * @since 2.5
46   * @version $Id: MemberUtils.java 1143537 2011-07-06 19:30:22Z joehni $
47   */
48  abstract class MemberUtils {
49      // TODO extract an interface to implement compareParameterSets(...)?
50  
51      private static final int ACCESS_TEST = Modifier.PUBLIC | Modifier.PROTECTED | Modifier.PRIVATE;
52  
53      /** Array of primitive number types ordered by "promotability" */
54      private static final Class<?>[] ORDERED_PRIMITIVE_TYPES = { Byte.TYPE, Short.TYPE,
55              Character.TYPE, Integer.TYPE, Long.TYPE, Float.TYPE, Double.TYPE };
56  
57      /**
58       * XXX Default access superclass workaround
59       *
60       * When a public class has a default access superclass with public members,
61       * these members are accessible. Calling them from compiled code works fine.
62       * Unfortunately, on some JVMs, using reflection to invoke these members
63       * seems to (wrongly) prevent access even when the modifier is public.
64       * Calling setAccessible(true) solves the problem but will only work from
65       * sufficiently privileged code. Better workarounds would be gratefully
66       * accepted.
67       * @param o the AccessibleObject to set as accessible
68       */
69      static void setAccessibleWorkaround(AccessibleObject o) {
70          if (o == null || o.isAccessible()) {
71              return;
72          }
73          Member m = (Member) o;
74          if (Modifier.isPublic(m.getModifiers())
75                  && isPackageAccess(m.getDeclaringClass().getModifiers())) {
76              try {
77                  o.setAccessible(true);
78              } catch (SecurityException e) { // NOPMD
79                  // ignore in favor of subsequent IllegalAccessException
80              }
81          }
82      }
83  
84      /**
85       * Returns whether a given set of modifiers implies package access.
86       * @param modifiers to test
87       * @return true unless package/protected/private modifier detected
88       */
89      static boolean isPackageAccess(int modifiers) {
90          return (modifiers & ACCESS_TEST) == 0;
91      }
92  
93      /**
94       * Returns whether a Member is accessible.
95       * @param m Member to check
96       * @return true if <code>m</code> is accessible
97       */
98      static boolean isAccessible(Member m) {
99          return m != null && Modifier.isPublic(m.getModifiers()) && !m.isSynthetic();
100     }
101 
102     /**
103      * Compares the relative fitness of two sets of parameter types in terms of
104      * matching a third set of runtime parameter types, such that a list ordered
105      * by the results of the comparison would return the best match first
106      * (least).
107      *
108      * @param left the "left" parameter set
109      * @param right the "right" parameter set
110      * @param actual the runtime parameter types to match against
111      * <code>left</code>/<code>right</code>
112      * @return int consistent with <code>compare</code> semantics
113      */
114     static int compareParameterTypes(Class<?>[] left, Class<?>[] right, Class<?>[] actual) {
115         float leftCost = getTotalTransformationCost(actual, left);
116         float rightCost = getTotalTransformationCost(actual, right);
117         return leftCost < rightCost ? -1 : rightCost < leftCost ? 1 : 0;
118     }
119 
120     /**
121      * Returns the sum of the object transformation cost for each class in the
122      * source argument list.
123      * @param srcArgs The source arguments
124      * @param destArgs The destination arguments
125      * @return The total transformation cost
126      */
127     private static float getTotalTransformationCost(Class<?>[] srcArgs, Class<?>[] destArgs) {
128         float totalCost = 0.0f;
129         for (int i = 0; i < srcArgs.length; i++) {
130             Class<?> srcClass, destClass;
131             srcClass = srcArgs[i];
132             destClass = destArgs[i];
133             totalCost += getObjectTransformationCost(srcClass, destClass);
134         }
135         return totalCost;
136     }
137 
138     /**
139      * Gets the number of steps required needed to turn the source class into
140      * the destination class. This represents the number of steps in the object
141      * hierarchy graph.
142      * @param srcClass The source class
143      * @param destClass The destination class
144      * @return The cost of transforming an object
145      */
146     private static float getObjectTransformationCost(Class<?> srcClass, Class<?> destClass) {
147         if (destClass.isPrimitive()) {
148             return getPrimitivePromotionCost(srcClass, destClass);
149         }
150         float cost = 0.0f;
151         while (srcClass != null && !destClass.equals(srcClass)) {
152             if (destClass.isInterface() && ClassUtils.isAssignable(srcClass, destClass)) {
153                 // slight penalty for interface match.
154                 // we still want an exact match to override an interface match,
155                 // but
156                 // an interface match should override anything where we have to
157                 // get a superclass.
158                 cost += 0.25f;
159                 break;
160             }
161             cost++;
162             srcClass = srcClass.getSuperclass();
163         }
164         /*
165          * If the destination class is null, we've travelled all the way up to
166          * an Object match. We'll penalize this by adding 1.5 to the cost.
167          */
168         if (srcClass == null) {
169             cost += 1.5f;
170         }
171         return cost;
172     }
173 
174     /**
175      * Gets the number of steps required to promote a primitive number to another
176      * type.
177      * @param srcClass the (primitive) source class
178      * @param destClass the (primitive) destination class
179      * @return The cost of promoting the primitive
180      */
181     private static float getPrimitivePromotionCost(final Class<?> srcClass, final Class<?> destClass) {
182         float cost = 0.0f;
183         Class<?> cls = srcClass;
184         if (!cls.isPrimitive()) {
185             // slight unwrapping penalty
186             cost += 0.1f;
187             cls = ClassUtils.wrapperToPrimitive(cls);
188         }
189         for (int i = 0; cls != destClass && i < ORDERED_PRIMITIVE_TYPES.length; i++) {
190             if (cls == ORDERED_PRIMITIVE_TYPES[i]) {
191                 cost += 0.1f;
192                 if (i < ORDERED_PRIMITIVE_TYPES.length - 1) {
193                     cls = ORDERED_PRIMITIVE_TYPES[i + 1];
194                 }
195             }
196         }
197         return cost;
198     }
