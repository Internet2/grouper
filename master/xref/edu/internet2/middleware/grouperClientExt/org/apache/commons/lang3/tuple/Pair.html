32  package edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.tuple;
33  
34  import java.io.Serializable;
35  import java.util.Map;
36  
37  import edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.ObjectUtils;
38  import edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.builder.CompareToBuilder;
39  
40  /**
41   * <p>A pair consisting of two elements.</p>
42   * 
43   * <p>This class is an abstract implementation defining the basic API.
44   * It refers to the elements as 'left' and 'right'. It also implements the
45   * {@code Map.Entry} interface where the key is 'left' and the value is 'right'.</p>
46   * 
47   * <p>Subclass implementations may be mutable or immutable.
48   * However, there is no restriction on the type of the stored objects that may be stored.
49   * If mutable objects are stored in the pair, then the pair itself effectively becomes mutable.</p>
50   *
51   * @param <L> the left element type
52   * @param <R> the right element type
53   *
54   * @since Lang 3.0
55   * @version $Id: Pair.java 1142401 2011-07-03 08:30:12Z bayard $
56   */
57  public abstract class Pair<L, R> implements Map.Entry<L, R>, Comparable<Pair<L, R>>, Serializable {
58  
59      /** Serialization version */
60      private static final long serialVersionUID = 4954918890077093841L;
61  
62      /**
63       * <p>Obtains an immutable pair of from two objects inferring the generic types.</p>
64       * 
65       * <p>This factory allows the pair to be created using inference to
66       * obtain the generic types.</p>
67       * 
68       * @param <L> the left element type
69       * @param <R> the right element type
70       * @param left  the left element, may be null
71       * @param right  the right element, may be null
72       * @return a pair formed from the two parameters, not null
73       */
74      public static <L, R> Pair<L, R> of(L left, R right) {
75          return new ImmutablePair<L, R>(left, right);
76      }
77  
78      //-----------------------------------------------------------------------
79      /**
80       * <p>Gets the left element from this pair.</p>
81       * 
82       * <p>When treated as a key-value pair, this is the key.</p>
83       * 
84       * @return the left element, may be null
85       */
86      public abstract L getLeft();
87  
88      /**
89       * <p>Gets the right element from this pair.</p>
90       * 
91       * <p>When treated as a key-value pair, this is the value.</p>
92       * 
93       * @return the right element, may be null
94       */
95      public abstract R getRight();
96  
97      /**
98       * <p>Gets the key from this pair.</p>
99       * 
100      * <p>This method implements the {@code Map.Entry} interface returning the
101      * left element as the key.</p>
102      * 
103      * @return the left element as the key, may be null
104      */
105     public final L getKey() {
106         return getLeft();
107     }
108 
109     /**
110      * <p>Gets the value from this pair.</p>
111      * 
112      * <p>This method implements the {@code Map.Entry} interface returning the
113      * right element as the value.</p>
114      * 
115      * @return the right element as the value, may be null
116      */
117     public R getValue() {
118         return getRight();
119     }
120 
121     //-----------------------------------------------------------------------
122     /**
123      * <p>Compares the pair based on the left element followed by the right element.
124      * The types must be {@code Comparable}.</p>
125      * 
126      * @param other  the other pair, not null
127      * @return negative if this is less, zero if equal, positive if greater
128      */
129     public int compareTo(Pair<L, R> other) {
130       return new CompareToBuilder().append(getLeft(), other.getLeft())
131               .append(getRight(), other.getRight()).toComparison();
132     }
133 
134     /**
135      * <p>Compares this pair to another based on the two elements.</p>
136      * 
137      * @param obj  the object to compare to, null returns false
138      * @return true if the elements of the pair are equal
139      */
140     @Override
141     public boolean equals(Object obj) {
142         if (obj == this) {
143             return true;
144         }
145         if (obj instanceof Map.Entry<?, ?>) {
146             Map.Entry<?, ?> other = (Map.Entry<?, ?>) obj;
147             return ObjectUtils.equals(getKey(), other.getKey())
148                     && ObjectUtils.equals(getValue(), other.getValue());
149         }
150         return false;
151     }
152 
153     /**
154      * <p>Returns a suitable hash code.
155      * The hash code follows the definition in {@code Map.Entry}.</p>
156      * 
157      * @return the hash code
158      */
159     @Override
160     public int hashCode() {
161         // see Map.Entry API specification
162         return (getKey() == null ? 0 : getKey().hashCode()) ^
163                 (getValue() == null ? 0 : getValue().hashCode());
164     }
165 
166     /**
167      * <p>Returns a String representation of this pair using the format {@code ($left,$right)}.</p>
168      * 
169      * @return a string describing this object, not null
170      */
171     @Override
172     public String toString() {
173         return new StringBuilder().append('(').append(getLeft()).append(',').append(getRight()).append(')').toString();
174     }
175 
176     /**
177      * <p>Formats the receiver using the given format.</p>
178      * 
179      * <p>This uses {@link java.util.Formattable} to perform the formatting. Two variables may
180      * be used to embed the left and right elements. Use {@code %1$s} for the left
181      * element (key) and {@code %2$s} for the right element (value).
182      * The default format used by {@code toString()} is {@code (%1$s,%2$s)}.</p>
183      * 
184      * @param format  the format string, optionally containing {@code %1$s} and {@code %2$s}, not null
185      * @return the formatted string, not null
186      */
187     public String toString(String format) {
188         return String.format(format, getLeft(), getRight());
189     }
190 
191 }
