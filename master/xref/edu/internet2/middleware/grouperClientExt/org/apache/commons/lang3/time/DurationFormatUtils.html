124     /**
125      * <p>Formats the time gap as a string, using the specified format.
126      * Padding the left hand side of numbers with zeroes is optional and 
127      * the timezone may be specified.</p>
128      * 
129      * <p>This method formats durations using the days and lower fields of the
130      * format pattern. Months and larger are not used.</p>
131      * 
132      * @param durationMillis  the duration to format
133      * @param format  the way in which to format the duration, not null
134      * @param padWithZeros  whether to pad the left hand side of numbers with 0's
135      * @return the formatted duration, not null
136      */
137     public static String formatDuration(long durationMillis, String format, boolean padWithZeros) {
138 
139         Token[] tokens = lexx(format);
140 
141         int days         = 0;
142         int hours        = 0;
143         int minutes      = 0;
144         int seconds      = 0;
145         int milliseconds = 0;
146         
147         if (Token.containsTokenWithValue(tokens, d) ) {
148             days = (int) (durationMillis / DateUtils.MILLIS_PER_DAY);
149             durationMillis = durationMillis - (days * DateUtils.MILLIS_PER_DAY);
150         }
151         if (Token.containsTokenWithValue(tokens, H) ) {
152             hours = (int) (durationMillis / DateUtils.MILLIS_PER_HOUR);
153             durationMillis = durationMillis - (hours * DateUtils.MILLIS_PER_HOUR);
154         }
155         if (Token.containsTokenWithValue(tokens, m) ) {
156             minutes = (int) (durationMillis / DateUtils.MILLIS_PER_MINUTE);
157             durationMillis = durationMillis - (minutes * DateUtils.MILLIS_PER_MINUTE);
158         }
159         if (Token.containsTokenWithValue(tokens, s) ) {
160             seconds = (int) (durationMillis / DateUtils.MILLIS_PER_SECOND);
161             durationMillis = durationMillis - (seconds * DateUtils.MILLIS_PER_SECOND);
162         }
163         if (Token.containsTokenWithValue(tokens, S) ) {
164             milliseconds = (int) durationMillis;
165         }
166 
167         return format(tokens, 0, 0, days, hours, minutes, seconds, milliseconds, padWithZeros);
168     }

170     /**
171      * <p>Formats an elapsed time into a plurialization correct string.</p>
172      * 
173      * <p>This method formats durations using the days and lower fields of the
174      * format pattern. Months and larger are not used.</p>
175      * 
176      * @param durationMillis  the elapsed time to report in milliseconds
177      * @param suppressLeadingZeroElements  suppresses leading 0 elements
178      * @param suppressTrailingZeroElements  suppresses trailing 0 elements
179      * @return the formatted text in days/hours/minutes/seconds, not null
180      */
181     public static String formatDurationWords(
182         long durationMillis,
183         boolean suppressLeadingZeroElements,
184         boolean suppressTrailingZeroElements) {
185 
186         // This method is generally replacable by the format method, but 
187         // there are a series of tweaks and special cases that require 
188         // trickery to replicate.
189         String duration = formatDuration(durationMillis, "d' days 'H' hours 'm' minutes 's' seconds'");
190         if (suppressLeadingZeroElements) {
191             // this is a temporary marker on the front. Like ^ in regexp.
192             duration = " " + duration;
193             String tmp = StringUtils.replaceOnce(duration, " 0 days", "");
194             if (tmp.length() != duration.length()) {
195                 duration = tmp;
196                 tmp = StringUtils.replaceOnce(duration, " 0 hours", "");
197                 if (tmp.length() != duration.length()) {
198                     duration = tmp;
199                     tmp = StringUtils.replaceOnce(duration, " 0 minutes", "");
200                     duration = tmp;
201                     if (tmp.length() != duration.length()) {
202                         duration = StringUtils.replaceOnce(tmp, " 0 seconds", "");
203                     }
204                 }
205             }
206             if (duration.length() != 0) {
207                 // strip the space off again
208                 duration = duration.substring(1);
209             }
210         }
211         if (suppressTrailingZeroElements) {
212             String tmp = StringUtils.replaceOnce(duration, " 0 seconds", "");
213             if (tmp.length() != duration.length()) {
214                 duration = tmp;
215                 tmp = StringUtils.replaceOnce(duration, " 0 minutes", "");
216                 if (tmp.length() != duration.length()) {
217                     duration = tmp;
218                     tmp = StringUtils.replaceOnce(duration, " 0 hours", "");
219                     if (tmp.length() != duration.length()) {
220                         duration = StringUtils.replaceOnce(tmp, " 0 days", "");
221                     }
222                 }
223             }
224         }
225         // handle plurals
226         duration = " " + duration;
227         duration = StringUtils.replaceOnce(duration, " 1 seconds", " 1 second");
228         duration = StringUtils.replaceOnce(duration, " 1 minutes", " 1 minute");
229         duration = StringUtils.replaceOnce(duration, " 1 hours", " 1 hour");
230         duration = StringUtils.replaceOnce(duration, " 1 days", " 1 day");
231         return duration.trim();
232     }

261     /**
262      * <p>Formats the time gap as a string, using the specified format.
263      * Padding the left hand side of numbers with zeroes is optional and 
264      * the timezone may be specified. </p>
265      *
266      * <p>When calculating the difference between months/days, it chooses to 
267      * calculate months first. So when working out the number of months and 
268      * days between January 15th and March 10th, it choose 1 month and 
269      * 23 days gained by choosing January->February = 1 month and then 
270      * calculating days forwards, and not the 1 month and 26 days gained by 
271      * choosing March -> February = 1 month and then calculating days 
272      * backwards. </p>
273      *
274      * <p>For more control, the <a href="http://joda-time.sf.net/">Joda-Time</a>
275      * library is recommended.</p>
276      * 
277      * @param startMillis  the start of the duration
278      * @param endMillis  the end of the duration
279      * @param format  the way in which to format the duration, not null
280      * @param padWithZeros  whether to pad the left hand side of numbers with 0's
281      * @param timezone  the millis are defined in
282      * @return the formatted duration, not null
283      */
284     public static String formatPeriod(long startMillis, long endMillis, String format, boolean padWithZeros, 
285             TimeZone timezone) {
286 
287         // Used to optimise for differences under 28 days and 
288         // called formatDuration(millis, format); however this did not work 
289         // over leap years. 
290         // TODO: Compare performance to see if anything was lost by 
291         // losing this optimisation. 
292         
293         Token[] tokens = lexx(format);
294 
295         // timezones get funky around 0, so normalizing everything to GMT 
296         // stops the hours being off
297         Calendar start = Calendar.getInstance(timezone);
298         start.setTime(new Date(startMillis));
299         Calendar end = Calendar.getInstance(timezone);
300         end.setTime(new Date(endMillis));
301 
302         // initial estimates
303         int milliseconds = end.get(Calendar.MILLISECOND) - start.get(Calendar.MILLISECOND);
304         int seconds = end.get(Calendar.SECOND) - start.get(Calendar.SECOND);
305         int minutes = end.get(Calendar.MINUTE) - start.get(Calendar.MINUTE);
306         int hours = end.get(Calendar.HOUR_OF_DAY) - start.get(Calendar.HOUR_OF_DAY);
307         int days = end.get(Calendar.DAY_OF_MONTH) - start.get(Calendar.DAY_OF_MONTH);
308         int months = end.get(Calendar.MONTH) - start.get(Calendar.MONTH);
309         int years = end.get(Calendar.YEAR) - start.get(Calendar.YEAR);
310 
311         // each initial estimate is adjusted in case it is under 0
312         while (milliseconds < 0) {
313             milliseconds += 1000;
314             seconds -= 1;
315         }
316         while (seconds < 0) {
317             seconds += 60;
318             minutes -= 1;
319         }
320         while (minutes < 0) {
321             minutes += 60;
322             hours -= 1;
323         }
324         while (hours < 0) {
325             hours += 24;
326             days -= 1;
327         }
328        
329         if (Token.containsTokenWithValue(tokens, M)) {
330             while (days < 0) {
331                 days += start.getActualMaximum(Calendar.DAY_OF_MONTH);
332                 months -= 1;
333                 start.add(Calendar.MONTH, 1);
334             }
335 
336             while (months < 0) {
337                 months += 12;
338                 years -= 1;
339             }
340 
341             if (!Token.containsTokenWithValue(tokens, y) && years != 0) {
342                 while (years != 0) {
343                     months += 12 * years;
344                     years = 0;
345                 }
346             }
347         } else {
348             // there are no M's in the format string
349 
350             if( !Token.containsTokenWithValue(tokens, y) ) {
351                 int target = end.get(Calendar.YEAR);
352                 if (months < 0) {
353                     // target is end-year -1
354                     target -= 1;
355                 }
356                 
357                 while ( (start.get(Calendar.YEAR) != target)) {
358                     days += start.getActualMaximum(Calendar.DAY_OF_YEAR) - start.get(Calendar.DAY_OF_YEAR);
359                     
360                     // Not sure I grok why this is needed, but the brutal tests show it is
361                     if (start instanceof GregorianCalendar &&
362                             start.get(Calendar.MONTH) == Calendar.FEBRUARY &&
363                             start.get(Calendar.DAY_OF_MONTH) == 29) {
364                         days += 1;
365                     }
366                     
367                     start.add(Calendar.YEAR, 1);
368                     
369                     days += start.get(Calendar.DAY_OF_YEAR);
370                 }
371                 
372                 years = 0;
373             }
374             
375             while( start.get(Calendar.MONTH) != end.get(Calendar.MONTH) ) {
376                 days += start.getActualMaximum(Calendar.DAY_OF_MONTH);
377                 start.add(Calendar.MONTH, 1);
378             }
379             
380             months = 0;            
381 
382             while (days < 0) {
383                 days += start.getActualMaximum(Calendar.DAY_OF_MONTH);
384                 months -= 1;
385                 start.add(Calendar.MONTH, 1);
386             }
387             
388         }
389 
390         // The rest of this code adds in values that 
391         // aren't requested. This allows the user to ask for the 
392         // number of months and get the real count and not just 0->11.
393 
394         if (!Token.containsTokenWithValue(tokens, d)) {
395             hours += 24 * days;
396             days = 0;
397         }
398         if (!Token.containsTokenWithValue(tokens, H)) {
399             minutes += 60 * hours;
400             hours = 0;
401         }
402         if (!Token.containsTokenWithValue(tokens, m)) {
403             seconds += 60 * minutes;
404             minutes = 0;
405         }
406         if (!Token.containsTokenWithValue(tokens, s)) {
407             milliseconds += 1000 * seconds;
408             seconds = 0;
409         }
410 
411         return format(tokens, years, months, days, hours, minutes, seconds, milliseconds, padWithZeros);
412     }

415     /**
416      * <p>The internal method to do the formatting.</p>
417      * 
418      * @param tokens  the tokens
419      * @param years  the number of years
420      * @param months  the number of months
421      * @param days  the number of days
422      * @param hours  the number of hours
423      * @param minutes  the number of minutes
424      * @param seconds  the number of seconds
425      * @param milliseconds  the number of millis
426      * @param padWithZeros  whether to pad
427      * @return the formatted string
428      */
429     static String format(Token[] tokens, int years, int months, int days, int hours, int minutes, int seconds,
430             int milliseconds, boolean padWithZeros) {
431         StringBuffer buffer = new StringBuffer();
432         boolean lastOutputSeconds = false;
433         int sz = tokens.length;
434         for (int i = 0; i < sz; i++) {
435             Token token = tokens[i];
436             Object value = token.getValue();
437             int count = token.getCount();
438             if (value instanceof StringBuffer) {
439                 buffer.append(value.toString());
440             } else {
441                 if (value == y) {
442                     buffer.append(padWithZeros ? StringUtils.leftPad(Integer.toString(years), count, '0') : Integer
443                             .toString(years));
444                     lastOutputSeconds = false;
445                 } else if (value == M) {
446                     buffer.append(padWithZeros ? StringUtils.leftPad(Integer.toString(months), count, '0') : Integer
447                             .toString(months));
448                     lastOutputSeconds = false;
449                 } else if (value == d) {
450                     buffer.append(padWithZeros ? StringUtils.leftPad(Integer.toString(days), count, '0') : Integer
451                             .toString(days));
452                     lastOutputSeconds = false;
453                 } else if (value == H) {
454                     buffer.append(padWithZeros ? StringUtils.leftPad(Integer.toString(hours), count, '0') : Integer
455                             .toString(hours));
456                     lastOutputSeconds = false;
457                 } else if (value == m) {
458                     buffer.append(padWithZeros ? StringUtils.leftPad(Integer.toString(minutes), count, '0') : Integer
459                             .toString(minutes));
460                     lastOutputSeconds = false;
461                 } else if (value == s) {
462                     buffer.append(padWithZeros ? StringUtils.leftPad(Integer.toString(seconds), count, '0') : Integer
463                             .toString(seconds));
464                     lastOutputSeconds = true;
465                 } else if (value == S) {
466                     if (lastOutputSeconds) {
467                         milliseconds += 1000;
468                         String str = padWithZeros
469                                 ? StringUtils.leftPad(Integer.toString(milliseconds), count, '0')
470                                 : Integer.toString(milliseconds);
471                         buffer.append(str.substring(1));
472                     } else {
473                         buffer.append(padWithZeros
474                                 ? StringUtils.leftPad(Integer.toString(milliseconds), count, '0')
475                                 : Integer.toString(milliseconds));
476                     }
477                     lastOutputSeconds = false;
478                 }
479             }
480         }
481         return buffer.toString();
482     }

492     /**
493      * Parses a classic date format string into Tokens
494      *
495      * @param format  the format to parse, not null
496      * @return array of Token[]
497      */
498     static Token[] lexx(String format) {
499         char[] array = format.toCharArray();
500         ArrayList<Token> list = new ArrayList<Token>(array.length);
501 
502         boolean inLiteral = false;
503         StringBuffer buffer = null;
504         Token previous = null;
505         int sz = array.length;
506         for(int i=0; i<sz; i++) {
507             char ch = array[i];
508             if(inLiteral && ch != '\'') {
509                 buffer.append(ch); // buffer can't be null if inLiteral is true
510                 continue;
511             }
512             Object value = null;
513             switch(ch) {
514                 // TODO: Need to handle escaping of '
515                 case '\'' : 
516                   if(inLiteral) {
517                       buffer = null;
518                       inLiteral = false;
519                   } else {
520                       buffer = new StringBuffer();
521                       list.add(new Token(buffer));
522                       inLiteral = true;
523                   }
524                   break;
525                 case 'y'  : value = y; break;
526                 case 'M'  : value = M; break;
527                 case 'd'  : value = d; break;
528                 case 'H'  : value = H; break;
529                 case 'm'  : value = m; break;
530                 case 's'  : value = s; break;
531                 case 'S'  : value = S; break;
532                 default   : 
533                   if(buffer == null) {
534                       buffer = new StringBuffer();
535                       list.add(new Token(buffer));
536                   }
537                   buffer.append(ch);
538             }
539 
540             if(value != null) {
541                 if(previous != null && previous.getValue() == value) {
542                     previous.increment();
543                 } else {
544                     Token token = new Token(value);
545                     list.add(token); 
546                     previous = token;
547                 }
548                 buffer = null; 
549             }
550         }
551         return list.toArray( new Token[list.size()] );
552     }

554     //-----------------------------------------------------------------------
555     /**
556      * Element that is parsed from the format pattern.
557      */
558     static class Token {
559 
560         /**
561          * Helper method to determine if a set of tokens contain a value
562          *
563          * @param tokens set to look in
564          * @param value to look for
565          * @return boolean <code>true</code> if contained
566          */
567         static boolean containsTokenWithValue(Token[] tokens, Object value) {
568             int sz = tokens.length;
569             for (int i = 0; i < sz; i++) {
570                 if (tokens[i].getValue() == value) {
571                     return true;
572                 }
573             }
574             return false;
575         }
576 
577         private final Object value;
578         private int count;
579 
580         /**
581          * Wraps a token around a value. A value would be something like a 'Y'.
582          *
583          * @param value to wrap
584          */
585         Token(Object value) {
586             this.value = value;
587             this.count = 1;
588         }
589 
590         /**
591          * Wraps a token around a repeated number of a value, for example it would 
592          * store 'yyyy' as a value for y and a count of 4.
593          *
594          * @param value to wrap
595          * @param count to wrap
596          */
597         Token(Object value, int count) {
598             this.value = value;
599             this.count = count;
600         }
601 
602         /**
603          * Adds another one of the value
604          */
605         void increment() { 
606             count++;
607         }
608 
609         /**
610          * Gets the current number of values represented
611          *
612          * @return int number of values represented
613          */
614         int getCount() {
615             return count;
616         }
617 
618         /**
619          * Gets the particular value this token represents.
620          * 
621          * @return Object value
622          */
623         Object getValue() {
624             return value;
625         }
626 
627         /**
628          * Supports equality of this Token to another Token.
629          *
630          * @param obj2 Object to consider equality of
631          * @return boolean <code>true</code> if equal
632          */
633         @Override
634         public boolean equals(Object obj2) {
635             if (obj2 instanceof Token) {
636                 Token tok2 = (Token) obj2;
637                 if (this.value.getClass() != tok2.value.getClass()) {
638                     return false;
639                 }
640                 if (this.count != tok2.count) {
641                     return false;
642                 }
643                 if (this.value instanceof StringBuffer) {
644                     return this.value.toString().equals(tok2.value.toString());
645                 } else if (this.value instanceof Number) {
646                     return this.value.equals(tok2.value);
647                 } else {
648                     return this.value == tok2.value;
649                 }
650             }
651             return false;
652         }
653 
654         /**
655          * Returns a hash code for the token equal to the 
656          * hash code for the token's value. Thus 'TT' and 'TTTT' 
657          * will have the same hash code. 
658          *
659          * @return The hash code for the token
660          */
661         @Override
662         public int hashCode() {
663             return this.value.hashCode();
664         }
665 
666         /**
667          * Represents this token as a String.
668          *
669          * @return String representation of the token
670          */
671         @Override
672         public String toString() {
673             return StringUtils.repeat(this.value.toString(), this.count);
674         }
675     }
