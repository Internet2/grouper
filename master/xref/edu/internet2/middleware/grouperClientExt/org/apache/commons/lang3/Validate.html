39  /**
40   * <p>This class assists in validating arguments. The validation methods are
41   * based along the following principles:
42   * <ul>
43   *   <li>An invalid {@code null} argument causes a {@link NullPointerException}.</li>
44   *   <li>A non-{@code null} argument causes an {@link IllegalArgumentException}.</li>
45   *   <li>An invalid index into an array/collection/map/string causes an {@link IndexOutOfBoundsException}.</li>
46   * </ul>
47   *
48   * <p>All exceptions messages are
49   * <a href="http://java.sun.com/j2se/1.5.0/docs/api/java/util/Formatter.html#syntax">format strings</a>
50   * as defined by the Java platform. For example:</p>
51   *
52   * <pre>
53   * Validate.isTrue(i > 0, "The value must be greater than zero: %d", i);
54   * Validate.notNull(surname, "The surname must not be %s", null);
55   * </pre>
56   *
57   * <p>#ThreadSafe#</p>
58   * @version $Id: Validate.java 1199983 2011-11-09 21:41:24Z ggregory $
59   * @see java.lang.String#format(String, Object...)
60   * @since 2.0
61   */

100     /**
101      * <p>Validate that the argument condition is {@code true}; otherwise
102      * throwing an exception with the specified message. This method is useful when
103      * validating according to an arbitrary boolean expression, such as validating a
104      * primitive number or using your own custom validation expression.</p>
105      *
106      * <pre>Validate.isTrue(i > 0.0, "The value must be greater than zero: %d", i);</pre>
107      *
108      * <p>For performance reasons, the long value is passed as a separate parameter and
109      * appended to the exception message only in the case of an error.</p>
110      *
111      * @param expression  the boolean expression to check
112      * @param message  the {@link String#format(String, Object...)} exception message if invalid, not null
113      * @param value  the value to append to the message when invalid
114      * @throws IllegalArgumentException if expression is {@code false}
115      * @see #isTrue(boolean)
116      * @see #isTrue(boolean, String, double)
117      * @see #isTrue(boolean, String, Object...)
118      */
119     public static void isTrue(boolean expression, String message, long value) {
120         if (expression == false) {
121             throw new IllegalArgumentException(String.format(message, Long.valueOf(value)));
122         }
123     }

498      * <p>Validate that the specified argument array is neither
499      * {@code null} nor contains any elements that are {@code null};
500      * otherwise throwing an exception with the specified message.
501      *
502      * <pre>Validate.noNullElements(myArray, "The array contain null at position %d");</pre>
503      *
504      * <p>If the array is {@code null}, then the message in the exception
505      * is &quot;The validated object is null&quot;.</p>
506      *
507      * <p>If the array has a {@code null} element, then the iteration
508      * index of the invalid element is appended to the {@code values}
509      * argument.</p>
510      *
511      * @param <T> the array type
512      * @param array  the array to check, validated not null by this method
513      * @param message  the {@link String#format(String, Object...)} exception message if invalid, not null
514      * @param values  the optional values for the formatted exception message, null array not recommended
515      * @return the validated array (never {@code null} method for chaining)
516      * @throws NullPointerException if the array is {@code null}
517      * @throws IllegalArgumentException if an element is {@code null}
518      * @see #noNullElements(Object[])
519      */
520     public static <T> T[] noNullElements(T[] array, String message, Object... values) {
521         Validate.notNull(array);
522         for (int i = 0; i < array.length; i++) {
523             if (array[i] == null) {
524                 Object[] values2 = ArrayUtils.add(values, Integer.valueOf(i));
525                 throw new IllegalArgumentException(String.format(message, values2));
526             }
527         }
528         return array;
529     }

559     /**
560      * <p>Validate that the specified argument iterable is neither
561      * {@code null} nor contains any elements that are {@code null};
562      * otherwise throwing an exception with the specified message.
563      *
564      * <pre>Validate.noNullElements(myCollection, "The collection contains null at position %d");</pre>
565      *
566      * <p>If the iterable is {@code null}, then the message in the exception
567      * is &quot;The validated object is null&quot;.</p>
568      *
569      * <p>If the iterable has a {@code null} element, then the iteration
570      * index of the invalid element is appended to the {@code values}
571      * argument.</p>
572      *
573      * @param <T> the iterable type
574      * @param iterable  the iterable to check, validated not null by this method
575      * @param message  the {@link String#format(String, Object...)} exception message if invalid, not null
576      * @param values  the optional values for the formatted exception message, null array not recommended
577      * @return the validated iterable (never {@code null} method for chaining)
578      * @throws NullPointerException if the array is {@code null}
579      * @throws IllegalArgumentException if an element is {@code null}
580      * @see #noNullElements(Iterable)
581      */
582     public static <T extends Iterable<?>> T noNullElements(T iterable, String message, Object... values) {
583         Validate.notNull(iterable);
584         int i = 0;
585         for (Iterator<?> it = iterable.iterator(); it.hasNext(); i++) {
