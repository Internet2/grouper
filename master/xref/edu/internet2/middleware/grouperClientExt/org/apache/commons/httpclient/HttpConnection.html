488     /**
489      * Determines whether this connection is "stale", which is to say that either
490      * it is no longer open, or an attempt to read the connection would fail.
491      *
492      * <p>Unfortunately, due to the limitations of the JREs prior to 1.4, it is
493      * not possible to test a connection to see if both the read and write channels
494      * are open - except by reading and writing.  This leads to a difficulty when
495      * some connections leave the "write" channel open, but close the read channel
496      * and ignore the request.  This function attempts to ameliorate that
497      * problem by doing a test read, assuming that the caller will be doing a
498      * write followed by a read, rather than the other way around.
499      * </p>
500      *
501      * <p>To avoid side-effects, the underlying connection is wrapped by a
502      * {@link BufferedInputStream}, so although data might be read, what is visible
503      * to clients of the connection will not change with this call.</p.
504      *
505      * @throws IOException if the stale connection test is interrupted.
506      * 
507      * @return <tt>true</tt> if the connection is already closed, or a read would
508      * fail.
509      */
510     protected boolean isStale() throws IOException {
511         boolean isStale = true;
512         if (isOpen) {
513             // the connection is open, but now we have to see if we can read it
514             // assume the connection is not stale.
515             isStale = false;
516             try {
517                 if (inputStream.available() <= 0) {
518                     try {
519                         socket.setSoTimeout(1);
520                         inputStream.mark(1);
521                         int byteRead = inputStream.read();
522                         if (byteRead == -1) {
523                             // again - if the socket is reporting all data read,
524                             // probably stale
525                             isStale = true;
526                         } else {
527                             inputStream.reset();
528                         }
529                     } finally {
530                         socket.setSoTimeout(this.params.getSoTimeout());
531                     }
532                 }
533             } catch (InterruptedIOException e) {
534                 if (!ExceptionUtil.isSocketTimeoutException(e)) {
535                     throw e;
536                 }
537                 // aha - the connection is NOT stale - continue on!
538             } catch (IOException e) {
539                 // oops - the connection is stale, the read or soTimeout failed.
540                 LOG.debug(
541                     "An error occurred while reading from the socket, is appears to be stale",
542                     e
543                 );
544                 isStale = true;
545             }
546         }
547 
548         return isStale;
549     }

691     /**
692      * Establishes a connection to the specified host and port
693      * (via a proxy if specified).
694      * The underlying socket is created from the {@link ProtocolSocketFactory}.
695      *
696      * @throws IOException if an attempt to establish the connection results in an
697      *   I/O error.
698      */
699     public void open() throws IOException {
700         LOG.trace("enter HttpConnection.open()");
701 
702         final String host = (proxyHostName == null) ? hostName : proxyHostName;
703         final int port = (proxyHostName == null) ? portNumber : proxyPortNumber;
704         assertNotOpen();
705         
706         if (LOG.isDebugEnabled()) {
707             LOG.debug("Open connection to " + host + ":" + port);
708         }
709         
710         try {
711             if (this.socket == null) {
712                 usingSecureSocket = isSecure() && !isProxied();
713                 // use the protocol's socket factory unless this is a secure
714                 // proxied connection
715                 ProtocolSocketFactory socketFactory = null;
716                 if (isSecure() && isProxied()) {
717                     Protocol defaultprotocol = Protocol.getProtocol("http");
718                     socketFactory = defaultprotocol.getSocketFactory();
719                 } else {
720                     socketFactory = this.protocolInUse.getSocketFactory();
721                 }
722                 this.socket = socketFactory.createSocket(
723                             host, port, 
724                             localAddress, 0,
725                             this.params);
726             }
727 
728             /*
729             "Nagling has been broadly implemented across networks, 
730             including the Internet, and is generally performed by default 
731             - although it is sometimes considered to be undesirable in 
732             highly interactive environments, such as some client/server 
733             situations. In such cases, nagling may be turned off through 
734             use of the TCP_NODELAY sockets option." */
735 
736             socket.setTcpNoDelay(this.params.getTcpNoDelay());
737             socket.setSoTimeout(this.params.getSoTimeout());
738             
739             int linger = this.params.getLinger();
740             if (linger >= 0) {
741                 socket.setSoLinger(linger > 0, linger);
742             }
743             
744             int sndBufSize = this.params.getSendBufferSize();
745             if (sndBufSize >= 0) {
746                 socket.setSendBufferSize(sndBufSize);
747             }        
748             int rcvBufSize = this.params.getReceiveBufferSize();
749             if (rcvBufSize >= 0) {
750                 socket.setReceiveBufferSize(rcvBufSize);
751             }        
752             int outbuffersize = socket.getSendBufferSize();
753             if ((outbuffersize > 2048) || (outbuffersize <= 0)) {
754                 outbuffersize = 2048;
755             }
756             int inbuffersize = socket.getReceiveBufferSize();
757             if ((inbuffersize > 2048) || (inbuffersize <= 0)) {
758                 inbuffersize = 2048;
759             }
760             inputStream = new BufferedInputStream(socket.getInputStream(), inbuffersize);
761             outputStream = new BufferedOutputStream(socket.getOutputStream(), outbuffersize);
762             isOpen = true;
763         } catch (IOException e) {
764             // Connection wasn't opened properly
765             // so close everything out
766             closeSocketAndStreams();
767             throw e;
768         }
769     }

771     /**
772      * Instructs the proxy to establish a secure tunnel to the host. The socket will 
773      * be switched to the secure socket. Subsequent communication is done via the secure 
774      * socket. The method can only be called once on a proxied secure connection.
775      *
776      * @throws IllegalStateException if connection is not secure and proxied or
777      * if the socket is already secure.
778      * @throws IOException if an attempt to establish the secure tunnel results in an
779      *   I/O error.
780      */
781     public void tunnelCreated() throws IllegalStateException, IOException {
782         LOG.trace("enter HttpConnection.tunnelCreated()");
783 
784         if (!isSecure() || !isProxied()) {
785             throw new IllegalStateException(
786                 "Connection must be secure "
787                     + "and proxied to use this feature");
788         }
789 
790         if (usingSecureSocket) {
791             throw new IllegalStateException("Already using a secure socket");
792         }
793         
794         if (LOG.isDebugEnabled()) {
795             LOG.debug("Secure tunnel to " + this.hostName + ":" + this.portNumber);
796         }
797 
798         SecureProtocolSocketFactory socketFactory =
799             (SecureProtocolSocketFactory) protocolInUse.getSocketFactory();
800 
801         socket = socketFactory.createSocket(socket, hostName, portNumber, true);
802         int sndBufSize = this.params.getSendBufferSize();
803         if (sndBufSize >= 0) {
804             socket.setSendBufferSize(sndBufSize);
805         }        
806         int rcvBufSize = this.params.getReceiveBufferSize();
807         if (rcvBufSize >= 0) {
808             socket.setReceiveBufferSize(rcvBufSize);
809         }        
810         int outbuffersize = socket.getSendBufferSize();
811         if (outbuffersize > 2048) {
812             outbuffersize = 2048;
813         }
814         int inbuffersize = socket.getReceiveBufferSize();
815         if (inbuffersize > 2048) {
816             inbuffersize = 2048;
817         }
818         inputStream = new BufferedInputStream(socket.getInputStream(), inbuffersize);
819         outputStream = new BufferedOutputStream(socket.getOutputStream(), outbuffersize);
820         usingSecureSocket = true;
821         tunnelEstablished = true;
822     }

897     /**
898      * Tests if input data becomes available within the given period time in milliseconds.
899      * 
900      * @param timeout The number milliseconds to wait for input data to become available 
901      * @return boolean <tt>true</tt> if input data is availble, 
902      *                 <tt>false</tt> otherwise.
903      * 
904      * @throws IOException If an IO problem occurs
905      * @throws IllegalStateException If the connection isn't open.
906      */
907     public boolean isResponseAvailable(int timeout) 
908         throws IOException {
909         LOG.trace("enter HttpConnection.isResponseAvailable(int)");
910         assertOpen();
911         boolean result = false;
912         if (this.inputStream.available() > 0) {
913             result = true;
914         } else {
915             try {
916                 this.socket.setSoTimeout(timeout);
917                 inputStream.mark(1);
918                 int byteRead = inputStream.read();
919                 if (byteRead != -1) {
920                     inputStream.reset();
921                     LOG.debug("Input data available");
922                     result = true;
923                 } else {
924                     LOG.debug("Input data not available");
925                 }
926             } catch (InterruptedIOException e) {
927                 if (!ExceptionUtil.isSocketTimeoutException(e)) {
928                     throw e;
929                 }
930                 if (LOG.isDebugEnabled()) {
931                     LOG.debug("Input data not available after " + timeout + " ms");
932                 }
933             } finally {
934                 try {
935                     socket.setSoTimeout(this.params.getSoTimeout());
936                 } catch (IOException ioe) {
937                     LOG.debug("An error ocurred while resetting soTimeout, we will assume that"
938                         + " no response is available.",
939                         ioe);
940                     result = false;
941                 }
942             }
943         }
944         return result;
945     }

1235         lastResponseInputStream = null;
