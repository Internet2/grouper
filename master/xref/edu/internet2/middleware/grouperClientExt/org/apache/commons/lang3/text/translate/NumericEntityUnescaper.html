32  package edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.text.translate;
33  
34  import java.io.IOException;
35  import java.io.Writer;
36  import java.util.Arrays;
37  import java.util.EnumSet;
38  
39  /**
40   * Translate XML numeric entities of the form &#[xX]?\d+;? to 
41   * the specific codepoint.
42   *
43   * Note that the semi-colon is optional.
44   * 
45   * @since 3.0
46   * @version $Id: NumericEntityUnescaper.java 1199894 2011-11-09 17:53:59Z ggregory $
47   */
48  public class NumericEntityUnescaper extends CharSequenceTranslator {
49  
50      public static enum OPTION { semiColonRequired, semiColonOptional, errorIfNoSemiColon }
51  
52      // TODO?: Create an OptionsSet class to hide some of the conditional logic below
53      private final EnumSet<OPTION> options;
54  
55      /**
56       * Create a UnicodeUnescaper.
57       *
58       * The constructor takes a list of options, only one type of which is currently 
59       * available (whether to allow, error or ignore the semi-colon on the end of a 
60       * numeric entity to being missing).
61       *
62       * For example, to support numeric entities without a ';':
63       *    new NumericEntityUnescaper(NumericEntityUnescaper.OPTION.semiColonOptional)
64       * and to throw an IllegalArgumentException when they're missing:
65       *    new NumericEntityUnescaper(NumericEntityUnescaper.OPTION.errorIfNoSemiColon)
66       *
67       * Note that the default behaviour is to ignore them. 
68       *
69       * @param options to apply to this unescaper
70       */
71      public NumericEntityUnescaper(OPTION... options) {
72          if(options.length > 0) {
73              this.options = EnumSet.copyOf(Arrays.asList(options));
74          } else {
75              this.options = EnumSet.copyOf(Arrays.asList(new OPTION[] { OPTION.semiColonRequired }));
76          }
77      }
78  
79      /**
80       * Whether the passed in option is currently set.
81       *
82       * @param option to check state of
83       * @return whether the option is set
84       */
85      public boolean isSet(OPTION option) { 
86          return options == null ? false : options.contains(option);
87      }
88  
89      /**
90       * {@inheritDoc}
91       */
92      @Override
93      public int translate(CharSequence input, int index, Writer out) throws IOException {
94          int seqEnd = input.length();
95          // Uses -2 to ensure there is something after the &#
96          if(input.charAt(index) == '&' && index < seqEnd - 2 && input.charAt(index + 1) == '#') {
97              int start = index + 2;
98              boolean isHex = false;
99  
100             char firstChar = input.charAt(start);
101             if(firstChar == 'x' || firstChar == 'X') {
102                 start++;
103                 isHex = true;
104 
105                 // Check there's more than just an x after the &#
106                 if(start == seqEnd) {
107                     return 0;
108                 }
109             }
110 
111             int end = start;
112             // Note that this supports character codes without a ; on the end
113             while(end < seqEnd && ( input.charAt(end) >= '0' && input.charAt(end) <= '9' ||
114                                     input.charAt(end) >= 'a' && input.charAt(end) <= 'f' ||
115                                     input.charAt(end) >= 'A' && input.charAt(end) <= 'F' ) )
116             {
117                 end++;
118             }
119 
120             boolean semiNext = end != seqEnd && input.charAt(end) == ';';
121 
122             if(!semiNext) {
123                 if(isSet(OPTION.semiColonRequired)) {
124                     return 0;
125                 } else
126                 if(isSet(OPTION.errorIfNoSemiColon)) {
127                     throw new IllegalArgumentException("Semi-colon required at end of numeric entity");
128                 }
129             }
130 
131             int entityValue;
132             try {
133                 if(isHex) {
134                     entityValue = Integer.parseInt(input.subSequence(start, end).toString(), 16);
135                 } else {
136                     entityValue = Integer.parseInt(input.subSequence(start, end).toString(), 10);
137                 }
138             } catch(NumberFormatException nfe) {
139                 return 0;
140             }
141 
142             if(entityValue > 0xFFFF) {
143                 char[] chrs = Character.toChars(entityValue);
144                 out.write(chrs[0]);
145                 out.write(chrs[1]);
146             } else {
147                 out.write(entityValue);
148             }
149 
150             return 2 + end - start + (isHex ? 1 : 0) + (semiNext ? 1 : 0);
151         }
152         return 0;
153     }
154 }
