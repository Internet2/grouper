53  /**
54   * Establishes a tunneled HTTP connection via the CONNECT method.
55   *
56   * @author Ortwin Gl???ck
57   * @author dIon Gillard
58   * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
59   * @author <a href="mailto:oleg@ural.ru">Oleg Kalnichevski</a>
60   * @since 2.0
61   * @version $Revision: 1.1 $ $Date: 2008-11-30 10:57:19 $
62   */
63  public class ConnectMethod extends HttpMethodBase {
64      
65      /** the name of this method */
66      public static final String NAME = "CONNECT";
67  
68      private final HostConfiguration targethost;
69  
70      /**
71       * @deprecated use #ConnectMethod(HttpHost);
72       * 
73       * Create a connect method.
74       * 
75       * @since 3.0
76       */
77      public ConnectMethod() {
78          super();
79          this.targethost = null;
80      }
81  
82      /**
83       * @deprecated the wrapped method is no longer used
84       * 
85       * Create a connect method wrapping the existing method
86       *
87       * @param method the {@link HttpMethod method} to execute after connecting
88       *      to the server
89       */
90      public ConnectMethod(HttpMethod method) {
91          super();
92          this.targethost = null;
93      }
94  
95      /**
96       * Create a connect method.
97       * 
98       * @since 3.0
99       */
100     public ConnectMethod(final HostConfiguration targethost) {
101         super();
102         if (targethost == null) {
103             throw new IllegalArgumentException("Target host may not be null");
104         }
105         this.targethost = targethost;
106     }
107 
108     /**
109      * Provide the {@link #NAME name} of this method.
110      *
111      * @return the String "CONNECT"
112      */
113     public String getName() {
114         return NAME;
115     }
116     
117     public String getPath() {
118         if (this.targethost != null) {
119             StringBuffer buffer = new StringBuffer();
120             buffer.append(this.targethost.getHost()); 
121             int port = this.targethost.getPort();
122             if (port == -1) {
123                 port = this.targethost.getProtocol().getDefaultPort();  
124             }
125             buffer.append(':'); 
126             buffer.append(port);
127             return buffer.toString();
128         } else {
129             return "/";
130         }
131     }
132 
133     public URI getURI() throws URIException {
134         String charset = getParams().getUriCharset();
135         return new URI(getPath(), true, charset);
136     }
137 
138     /**
139      * This method does nothing. <tt>CONNECT</tt> request is not supposed 
140      * to contain <tt>Cookie</tt> request header.
141      *
142      * @param state current state of http requests
143      * @param conn the connection to use for I/O
144      *
145      * @throws IOException when errors occur reading or writing to/from the
146      *         connection
147      * @throws HttpException when a recoverable error occurs
148      *  
149      * @see HttpMethodBase#addCookieRequestHeader(HttpState, HttpConnection)
150      */
151     protected void addCookieRequestHeader(HttpState state, HttpConnection conn)
152         throws IOException, HttpException {
153         // Do nothing. Not applicable to CONNECT method
154     }
155 
156 
157     /**
158      * Populates the request headers map to with additional {@link Header
159      * headers} to be submitted to the given {@link HttpConnection}.
160      *
161      * <p>
162      * This implementation adds <tt>User-Agent</tt>, <tt>Host</tt>,
163      * and <tt>Proxy-Authorization</tt> headers, when appropriate.
164      * </p>
165      *
166      * @param state the client state
167      * @param conn the {@link HttpConnection} the headers will eventually be
168      *        written to
169      * @throws IOException when an error occurs writing the request
170      * @throws HttpException when a HTTP protocol error occurs
171      *
172      * @see #writeRequestHeaders
173      */
174     protected void addRequestHeaders(HttpState state, HttpConnection conn)
175         throws IOException, HttpException {
176         LOG.trace("enter ConnectMethod.addRequestHeaders(HttpState, "
177             + "HttpConnection)");
178         addUserAgentRequestHeader(state, conn);
179         addHostRequestHeader(state, conn);
180         addProxyConnectionHeader(state, conn);
181     }
182 
183     /**
184      * Execute this method and create a tunneled HttpConnection.  If the method
185      * is successful (i.e. the status is a 2xx) tunnelCreated() will be called
186      * on the connection.
187      *
188      * @param state the current http state
189      * @param conn the connection to write to
190      * @return the http status code from execution
191      * @throws HttpException when an error occurs writing the headers
192      * @throws IOException when an error occurs writing the headers
193      * 
194      * @see HttpConnection#tunnelCreated()
195      */
196     public int execute(HttpState state, HttpConnection conn) 
197     throws IOException, HttpException {
198 
199         LOG.trace("enter ConnectMethod.execute(HttpState, HttpConnection)");
200         int code = super.execute(state, conn);
201         if (LOG.isDebugEnabled()) {
202             LOG.debug("CONNECT status code " + code);
203         }
204         return code;
205     }
206 
207     /**
208      * Special Connect request.
209      *
210      * @param state the current http state
211      * @param conn the connection to write to
212      * @throws IOException when an error occurs writing the request
213      * @throws HttpException when an error occurs writing the request
214      */
215     protected void writeRequestLine(HttpState state, HttpConnection conn)
216     throws IOException, HttpException {
217         StringBuffer buffer = new StringBuffer();
218         buffer.append(getName()); 
219         buffer.append(' '); 
220         if (this.targethost != null) {
221             buffer.append(getPath()); 
222         } else {
223             int port = conn.getPort();
224             if (port == -1) {
225                 port = conn.getProtocol().getDefaultPort();  
226             }
227             buffer.append(conn.getHost()); 
228             buffer.append(':'); 
229             buffer.append(port); 
230         }
231         buffer.append(" "); 
232         buffer.append(getEffectiveVersion()); 
233         String line = buffer.toString();
234         conn.printLine(line, getParams().getHttpElementCharset());
235         if (Wire.HEADER_WIRE.enabled()) {
236             Wire.HEADER_WIRE.output(line);
237         }
238     }
239 
240     /**
241      * Returns <code>true</code> if the status code is anything other than
242      * SC_OK, <code>false</code> otherwise.
243      * 
244      * @see HttpMethodBase#shouldCloseConnection(HttpConnection)
245      * @see HttpStatus#SC_OK
246      * 
247      * @return <code>true</code> if the connection should be closed
248      */
249     protected boolean shouldCloseConnection(HttpConnection conn) {
250         if (getStatusCode() == HttpStatus.SC_OK) {
251             Header connectionHeader = null;
252             if (!conn.isTransparent()) {
253                 connectionHeader = getResponseHeader("proxy-connection");
254             }
255             if (connectionHeader == null) {
256                 connectionHeader = getResponseHeader("connection");
257             }
258             if (connectionHeader != null) {
259                 if (connectionHeader.getValue().equalsIgnoreCase("close")) {
260                     if (LOG.isWarnEnabled()) {
261                         LOG.warn("Invalid header encountered '" + connectionHeader.toExternalForm() 
262                         + "' in response " + getStatusLine().toString());
263                     }
264                 }
265             }
266             return false;
267         } else {
268             return super.shouldCloseConnection(conn);
269         }
270     }
271     
272     /** Log object for this class. */
273     private static final Log LOG = LogFactory.getLog(ConnectMethod.class);
