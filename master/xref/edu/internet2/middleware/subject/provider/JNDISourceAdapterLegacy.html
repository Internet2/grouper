145   @Override
146   public Subject getSubject(String id1, boolean exceptionIfNull) throws SubjectNotFoundException,
147       SubjectNotUniqueException {
148     Subject subject = null;
149     Search search = getSearch("searchSubject");
150     if (search == null) {
151       log.error("searchType: \"searchSubject\" not defined.");
152       return subject;
153     }
154     String[] attributeNames = { this.nameAttributeName, this.descriptionAttributeName,
155         this.subjectIDAttributeName };
156     try {
157       Attributes attributes1 = getLdapUnique(search, id1, attributeNames);
158       subject = createSubject(attributes1);
159     } catch (SubjectNotFoundException snfe) {
160       if (exceptionIfNull) {
161         throw snfe;
162       }
163     }
164     if (subject == null && exceptionIfNull) {
165       throw new SubjectNotFoundException("Subject " + id1 + " not found.");
166     }
167 
168     return subject;
169   }

204   /**
205    * 
206    * @param searchValue 
207    * @param firstPageOnly 
208    * @return  the result and if too many
209    * @see edu.internet2.middleware.subject.provider.BaseSourceAdapter#search(java.lang.String)
210    */
211   private SearchPageResult searchHelper(String searchValue, boolean firstPageOnly) {
212     Set<Subject> result = new LinkedHashSet<Subject>();
213     boolean tooManyResults = false;
214     Search search = getSearch("search");
215     if (search == null) {
216       log.error("searchType: \"search\" not defined.");
217       return new SearchPageResult(tooManyResults, result);
218     }
219     String throwErrorOnFindAllFailureString = this.getInitParam("throwErrorOnFindAllFailure");
220     boolean throwErrorOnFindAllFailure = SubjectUtils.booleanValue(throwErrorOnFindAllFailureString, true);
221 
222     try {
223       String[] attributeNames = { this.nameAttributeName, this.subjectIDAttributeName,
224           this.descriptionAttributeName };
225       NamingEnumeration ldapResults = getLdapResults(search, searchValue, attributeNames, firstPageOnly);
226       if (ldapResults == null) {
227         return new SearchPageResult(tooManyResults, result);
228       }
229 
230       if (failOnSearchForTesting) {
231         throw new RuntimeException("failOnSearchForTesting");
232       }
233       
234       while (ldapResults.hasMore()) {
235         SearchResult si = (SearchResult) ldapResults.next();
236         Attributes attributes1 = si.getAttributes();
237         Subject subject = createSubject(attributes1);
238         result.add(subject);
239         //if we are at the end of the page
240         if (firstPageOnly && this.maxPage != null && result.size() >= this.maxPage) {
241           tooManyResults = true;
242           break;
243         }
244         if (this.maxResults != null && result.size() >= this.maxResults) {
245           throw new SubjectTooManyResults(
246               "More results than allowed: " + this.maxResults 
247               + " for search '" + search + "'");
248         }
249       }
250     } catch (Exception ex) {
251       if (ex instanceof SubjectTooManyResults) {
252         throw (SubjectTooManyResults)ex;
253       }
254       if (!throwErrorOnFindAllFailure) {
255         log.error("LDAP Naming Except: " 
256             + ex.getMessage() + ", " + this.id + ", " + searchValue, ex);
257       } else {
258         throw new SourceUnavailableException(ex.getMessage() + ", source: " + this.getId() + ", sql: "
259             + search.getParam("sql"), ex);
260       }
261     }
262 
263     return new SearchPageResult(tooManyResults, result);
264   }

266   /**
267    * 
268    * @param attributes1
269    * @return subject
270    */
271   private Subject createSubject(Attributes attributes1) {
272     String name1 = "";
273     String subjectID = "";
274     String description = "";
275     try {
276       Attribute attribute = attributes1.get(this.subjectIDAttributeName);
277       if (attribute == null) {
278         log
279             .error("The LDAP attribute \""
280                 + this.subjectIDAttributeName
281                 + "\" does not have a value. It is beging used as the Grouper special attribute \"SubjectID\".");
282         return null;
283       }
284       subjectID = (String) attribute.get();
285       attribute = attributes1.get(this.nameAttributeName);
286       if (attribute == null) {
287         log
288             .error("The LDAP attribute \""
289                 + this.nameAttributeName
290                 + "\" does not have a value. It is being used as the Grouper special attribute \"name\".");
291         return null;
292       }
293       name1 = (String) attribute.get();
294       attribute = attributes1.get(this.descriptionAttributeName);
295       if (attribute == null) {
296         log
297             .error("The LDAP attribute \""
298                 + this.descriptionAttributeName
299                 + "\" does not have a value. It is being used as the Grouper special attribute \"description\".");
300       } else {
301         description = (String) attribute.get();
302       }
303     } catch (NamingException ex) {
304       throw new SourceUnavailableException("LDAP Naming Except: " + ex.getMessage(), ex);
305     }
306     
307     return new JNDISubject(subjectID, name1, description, 
308         this.getSubjectType().getName(), this.getId(), null);
309   }

331   @Override
332   public void init() throws SourceUnavailableException {
333     try {
334       Properties props = getInitParams();
335       setupEnvironment(props);
336       
337       {
338         String maxResultsString = props.getProperty("maxResults");
339         if (!StringUtils.isBlank(maxResultsString)) {
340           try {
341             this.maxResults = Integer.parseInt(maxResultsString);
342           } catch (NumberFormatException nfe) {
343             throw new SourceUnavailableException("Cant parse maxResults: " + maxResultsString, nfe);
344           }
345         }
346       }
347       
348       {
349         String maxPageString = props.getProperty("maxPageSize");
350         if (!StringUtils.isBlank(maxPageString)) {
351           try {
352             this.maxPage = Integer.parseInt(maxPageString);
353           } catch (NumberFormatException nfe) {
354             throw new SourceUnavailableException("Cant parse maxPage: " + maxPageString, nfe);
355           }
356         }
357       }
358 
359     } catch (Exception ex) {
360       throw new SourceUnavailableException("Unable to init JNDI source", ex);
361     }

364   /**
365    * Setup environment.
366    * @param props 
367    * @throws SourceUnavailableException
368    */
369   protected void setupEnvironment(Properties props) throws SourceUnavailableException {
370 
371     this.environment.put(Context.INITIAL_CONTEXT_FACTORY, props
372         .getProperty("INITIAL_CONTEXT_FACTORY"));
373     this.environment.put(Context.PROVIDER_URL, props.getProperty("PROVIDER_URL"));
374     this.environment.put(Context.SECURITY_AUTHENTICATION, props
375         .getProperty("SECURITY_AUTHENTICATION"));
376     
377     if (props.getProperty("SECURITY_PRINCIPAL") != null) {
378     this.environment.put(Context.SECURITY_PRINCIPAL, props.getProperty("SECURITY_PRINCIPAL"));
379     }
380 
381     String password = props.getProperty("SECURITY_CREDENTIALS");
382     password = Morph.decryptIfFile(password);
383 
384     if (password != null) {
385     this.environment.put(Context.SECURITY_CREDENTIALS, password);
386     }
387     
388     if (props.getProperty("SECURITY_PROTOCOL") != null) {
389       //this used to be hardcoded to SSL!!!
390       this.environment.put(Context.SECURITY_PROTOCOL, props.getProperty("SECURITY_PROTOCOL"));
391     }
392     Context context = null;
393     try {
394       log.debug("Creating Directory Context");
395       context = new InitialDirContext(this.environment);
396     } catch (AuthenticationException ex) {
397       log.error("Error with Authentication " + ex.getMessage(), ex);
398       throw new SourceUnavailableException("Error with Authentication ", ex);
399     } catch (NamingException ex) {
400       log.error("Naming Error " + ex.getMessage(), ex);
401       throw new SourceUnavailableException("Naming Error", ex);
402     } finally {
403       if (context != null) {
404         try {
405           context.close();
406         } catch (NamingException ne) {
407           // squelch, since it is already closed
408         }
409       }
410     }
411     log.info("Success in connecting to LDAP");
412 
413     this.nameAttributeName = props.getProperty("Name_AttributeType");
414     if (this.nameAttributeName == null) {
415       log.error("Name_AttributeType not defined");
416     }
417     this.subjectIDAttributeName = props.getProperty("SubjectID_AttributeType");
418     if (this.subjectIDAttributeName == null) {
419       log.error("SubjectID_AttributeType not defined");
420     }
421     this.descriptionAttributeName = props.getProperty("Description_AttributeType");
422     if (this.descriptionAttributeName == null) {
423       log.error("Description_AttributeType not defined");
424     }
425 
426   }

428   /**
429    * Loads attributes for the argument subject.
430    * @param subject 
431    * @return the map
432    */
433   protected Map loadAttributes(SubjectImpl subject) {
434     Map<String, Set<String>> attributes1 = new SubjectCaseInsensitiveMapImpl<String, Set<String>>();
435     Search search = getSearch("searchSubject");
436     if (search == null) {
437       log.error("searchType: \"search\" not defined.");
438       return attributes1;
439     }
440     //setting attributeNames to null will cause all attributes for a subject to be returned
441     String[] attributeNames = null;
442     Set attributeNameSet = this.getAttributes();
443     if (attributeNameSet.size() == 0) {
444       attributeNames = null;
445     } else {
446       attributeNames = new String[attributeNameSet.size()];
447       int i = 0;
448       for (Iterator it = attributeNameSet.iterator(); it.hasNext(); attributeNames[i++] = (String) it
449           .next())
450         ;
451     }
452     try {
453       Attributes ldapAttributes = getLdapUnique(search, subject.getId(), attributeNames);
454       for (NamingEnumeration e = ldapAttributes.getAll(); e.hasMore();) {
455         Attribute attr = (Attribute) e.next();
456         String name1 = attr.getID();
457         Set<String> values = new HashSet<String>();
458         for (NamingEnumeration en = attr.getAll(); en.hasMore();) {
459           Object value = en.next();
460           values.add(value.toString());
461         }
462         attributes1.put(name1, values);
463       }
464       subject.setAttributes(attributes1);
465     } catch (NamingException ex) {
466       throw new SourceUnavailableException("LDAP Naming Except: " + ex.getMessage(), ex);
467     }
468     return attributes1;
469   }

483   /**
484    * 
485    * @param search
486    * @param searchValue
487    * @param attributeNames
488    * @return naming enumeration
489    */
490   protected NamingEnumeration getLdapResults(Search search, String searchValue,
491       String[] attributeNames, boolean firstPageOnly) {
492     
493     //if this is a search and not by id or identifier, strip out the status part
494     boolean subjectStatusQuery = StringUtils.equals("search", search.getSearchType());
495     if (subjectStatusQuery) {
496       SubjectStatusResult subjectStatusResult = null;
497       
498       //see if we are doing status
499       SubjectStatusProcessor subjectStatusProcessor = new SubjectStatusProcessor(searchValue, this.getSubjectStatusConfig());
500       subjectStatusResult = subjectStatusProcessor.processSearch();
501 
502       //strip out status parts
503       searchValue = subjectStatusResult.getStrippedQuery();
504     }      
505     
506     DirContext context = null;
507     NamingEnumeration results = null;
508     String filter = search.getParam("filter");
509     if (filter == null) {
510       log.error("Search filter not found for search type:  " + search.getSearchType());
511       return results;
512     }
513     
514     filter = filter.replaceAll("%TERM%", escapeSearchFilter(searchValue));
515     String base = search.getParam("base");
516     if (base == null) {
517       base = "";
518       log.error("Search base not found for:  " + search.getSearchType()
519           + ". Using base \"\" ");
520 
521     }
522     int scopeNum = -1;
523     String scope = search.getParam("scope");
524     if (scope != null) {
525       scopeNum = getScope(scope);
526     }
527     if (scopeNum == -1) {
528       scopeNum = SearchControls.SUBTREE_SCOPE;
529       log.error("Search scope not found for: " + search.getSearchType()
530           + ". Using scope SUBTREE_SCOPE.");
531     }
532     log.debug("searchType: " + search.getSearchType() + " filter: " + filter + " base: "
533         + base + " scope: " + scope);
534     try {
535       context = new InitialDirContext(this.environment);
536       SearchControls constraints = new SearchControls();
537       
538       if ((firstPageOnly && this.maxPage != null) || this.maxResults != null) {
539         int pagesize = (firstPageOnly && this.maxPage != null) ? (this.maxPage+1) : -1;
540         if (pagesize == -1) {
541           pagesize = this.maxResults + 1;
542         } else if (this.maxResults != null){
543           pagesize = Math.min(pagesize, this.maxResults+1);
544         }
545         constraints.setCountLimit(pagesize);
546       }
547 
548       constraints.setSearchScope(scopeNum);
549       constraints.setReturningAttributes(attributeNames);
550       results = context.search(base, filter, constraints);
551     } catch (AuthenticationException ex) {
552       throw new SourceUnavailableException("Ldap Authentication Exception: " + ex.getMessage(), ex);
553     } catch (NamingException ex) {
554       throw new SourceUnavailableException("Ldap NamingException: " + ex.getMessage(), ex);
555 
556     } finally {
557       if (context != null) {
558         try {
559           context.close();
560         } catch (NamingException ne) {
561           // squelch, since it is already closed
562         }
563       }
564     }
565     return results;
566 
567   }

569   /**
570    * 
571    * @param search
572    * @param searchValue
573    * @param attributeNames
574    * @return attributes
575    * @throws SubjectNotFoundException
576    * @throws SubjectNotUniqueException
577    */
578   protected Attributes getLdapUnique(Search search, String searchValue,
579       String[] attributeNames) throws SubjectNotFoundException, SubjectNotUniqueException {
580     Attributes attributes1 = null;
581     NamingEnumeration results = getLdapResults(search, searchValue, attributeNames);
582 
583     try {
584       if (results == null || !results.hasMore()) {
585         String errMsg = "No results: " + search.getSearchType() + " filter:"
586             + search.getParam("filter") + " searchValue: " + searchValue;
587         throw new SubjectNotFoundException(errMsg);
588       }
589 
590       SearchResult si = (SearchResult) results.next();
591       attributes1 = si.getAttributes();
592       if (results.hasMore()) {
593         si = (SearchResult) results.next();
594         String errMsg = "Search is not unique:" + si.getName() + "\n";
595         throw new SubjectNotUniqueException(errMsg);
596       }
597     } catch (NamingException ex) {
598       throw new SourceUnavailableException("Ldap NamingException: " + ex.getMessage(), ex);
599     }
600     return attributes1;
601   }

603   /**
604    * Escape a search filter to prevent LDAP injection.
605    * From http://www.owasp.org/index.php/Preventing_LDAP_Injection_in_Java
606    * 
607    * @param filter
608    * @return escaped filter
609    */
610   protected String escapeSearchFilter(String filter) {
611     //From RFC 2254
612     String escapedStr = new String(filter);
613     escapedStr = escapedStr.replaceAll("\\\\", "\\\\5c");
614     // We want people to be able to use wildcards.
615     // escapedStr = escapedStr.replaceAll("\\*","\\\\2a");
616     escapedStr = escapedStr.replaceAll("\\(", "\\\\28");
617     escapedStr = escapedStr.replaceAll("\\)", "\\\\29");
618     escapedStr = escapedStr.replaceAll("\\" + Character.toString('\u0000'), "\\\\00");
619     return escapedStr;
620   }
