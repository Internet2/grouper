20  package edu.internet2.middleware.subject.provider;
21  
22  import java.sql.Connection;
23  import java.sql.SQLException;
24  
25  
26  /**
27   * bean that wraps connections
28   */
29  public interface JdbcConnectionBean {
30  
31    /**
32     * get a connection (dont close this when done, just call "doneWithConnection()"
33     * @return the connection
34     * @throws SQLException if there is a problem
35     */
36    public Connection connection() throws SQLException;
37    
38    /**
39     * call this when the connection is done.  This will do any cleanup
40     * this is a null-safe method.  
41     * @throws SQLException if there is a problem
42     */
43    public void doneWithConnection() throws SQLException;
44  
45    /**
46     * call this when the connection is not needed, in the finally block
47     * this might return to pool.  In general, this shouldnt throw exceptions
48     * since it is done in a finally block, it should only log them
49     */
50    public void doneWithConnectionFinally();
51  
52    /**
53     * call this when the connection is done but there was an error, will pass
54     * an exception.  This should do whatever and rethrow the exception as runtime
55     * @param t 
56     */
57    public void doneWithConnectionError(Throwable t);
58  }
