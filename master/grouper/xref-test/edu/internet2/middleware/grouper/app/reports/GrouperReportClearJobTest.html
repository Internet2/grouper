34    public void testClearOldReports() {
35          
36      //Given
37      GrouperSessionResult grouperSessionResult = GrouperSession.startRootSessionIfNotStarted();
38      GrouperSession grouperSession = grouperSessionResult.getGrouperSession();
39      
40      GrouperConfig.retrieveConfig().propertiesOverrideMap().put("grouperReporting.enable", "true");
41  
42      Stem stem0 = new StemSave(grouperSession).assignCreateParentStemsIfNotExist(true).assignName("test").save();
43      
44      AttributeAssign configAttributeAssign = saveReportConfigAttributeMetadata(stem0, "test config");
45      saveReportInstanceAttributeMetadata(stem0, configAttributeAssign.getId());
46      
47      //When 
48      GrouperReportClearJob.clearOldReports();
49      
50      //Then
51      List<GrouperReportInstance> reportInstances = GrouperReportInstanceService.getReportInstances(stem0, configAttributeAssign.getId());
52      assertEquals(0, reportInstances.size());
53      
54    }

107   private static AttributeAssign saveReportInstanceAttributeMetadata(Stem stem,
108       String attributeAssignOfReportConfig) {
109     
110     AttributeAssign attributeAssign = stem.getAttributeDelegate().
111         addAttribute(GrouperReportInstanceAttributeNames.retrieveAttributeDefNameBase()).getAttributeAssign();
112     
113     AttributeDefName attributeDefName = AttributeDefNameFinder.findByName(reportConfigStemName()+":"+GROUPER_REPORT_INSTANCE_STATUS, true);
114     attributeAssign.getAttributeValueDelegate().assignValue(attributeDefName.getName(), STATUS_SUCCESS);
115     
116     attributeDefName = AttributeDefNameFinder.findByName(reportConfigStemName()+":"+GROUPER_REPORT_INSTANCE_CONFIG_MARKER_ASSIGNMENT_ID, true);
117     attributeAssign.getAttributeValueDelegate().assignValue(attributeDefName.getName(), attributeAssignOfReportConfig);
118     
119     attributeDefName = AttributeDefNameFinder.findByName(reportConfigStemName()+":"+GROUPER_REPORT_INSTANCE_MILLIS_SINCE_1970, true);
120     attributeAssign.getAttributeValueDelegate().assignValue(attributeDefName.getName(), String.valueOf(System.currentTimeMillis() - (35 * 24 * 60 * 60 * 1000L)));
121     
122     attributeDefName = AttributeDefNameFinder.findByName(reportConfigStemName()+":"+GROUPER_REPORT_INSTANCE_FILE_NAME, true);
123     attributeAssign.getAttributeValueDelegate().assignValue(attributeDefName.getName(), "data.csv");
124     
125     attributeDefName = AttributeDefNameFinder.findByName(reportConfigStemName()+":"+GROUPER_REPORT_INSTANCE_FILE_POINTER, true);
126     attributeAssign.getAttributeValueDelegate().assignValue(attributeDefName.getName(), "/tmp/reports/data.csv");
127     
128     return attributeAssign;
