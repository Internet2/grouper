44      GroupSet gs1 = GrouperDAOFactory.getFactory().getGroupSet().findSelfGroup(testEntity1.getId(), FieldFinder.find("groupAttrReaders", true).getId());

49      gs1.delete(false);

67        // good
68      }
69      
70      try {
71        GrouperDAOFactory.getFactory().getGroupSet().findSelfGroup(testEntity2.getId(), FieldFinder.find("groupAttrReaders", true).getId());
