66    /**
67     * <pre>
68     * load test duke data (written by Shilen, adapted by Chris
69     * 
70     * I've attached the script I used to load test data in Grouper.  
71     * I have the following settings set to true.
72     * 
73     * groups.create.grant.all.read
74     * groups.create.grant.all.view
75     * 
76     * I mentioning that because it will cause there to be more rows in the grouper_memberships table.  After running the script, I ended up with the following rows in the tables.
77     * 
78     * grouper_memberships = 1,067,103
79     * grouper_stems = 105,704
80     * grouper_groups = 125,400
81     * grouper_members = 128,329
82     * 
83     * </pre>
84     * @throws Exception
85     */
86    public static void loadDukeData() throws Exception {
87      GrouperSession session = GrouperSession.startRootSession();
88  
89      Stem eduStem = Stem.saveStem(session, "edu", null, "edu", "edu", "", 
90          null, false);
91  
92      AttributeDef attributeDef = new AttributeDefSave(session).assignName("edu:attributeDef").assignAttributeDefType(AttributeDefType.perm).assignToGroup(true).save();
93  
94      Stem dukeStem = Stem.saveStem(session, "duke", null, "duke", "Duke University", "", 
95          null, false);
96      Stem sissStem = Stem.saveStem(session, dukeStem.getName() + ":siss",null,  dukeStem.getName() + ":siss",  "siss","",
97          null, false);
98      Stem coursesStem = Stem.saveStem(session, sissStem.getName() + ":courses",  null,sissStem.getName() + ":courses", "courses","",  
99          null, false);
100 
101     int stemsCreated = 3;
102     int groupsCreated = 0;
103     int membersAdded = 0;
104 
105     for (int subjCount = 1; subjCount <= 100; subjCount++) {
106       String subjStemName = coursesStem.getName() + ":SUBJECT" + subjCount;
107       Stem subjStem = Stem.saveStem(session, subjStemName, null, subjStemName, "SUBJECT" + subjCount, "", 
108            null, false);
109       stemsCreated++;
110       //for (int catelogCount = 100; catelogCount <= 110; catelogCount++) {
111       for (int catelogCount = 100; catelogCount <= 110; catelogCount++) {
112         String catelogStemName = subjStem.getName() + ":" + catelogCount;
113         Stem catelogStem = Stem.saveStem(session, catelogStemName, null,catelogStemName,""+catelogCount, "", 
114                null, false);
115          stemsCreated++;
116          //for (int secCount = 10; secCount <= 28; secCount++) {
117          for (int secCount = 10; secCount <= 28; secCount++) {
118            String secStemName = catelogStem.getName() + ":" + secCount;
119           Stem secStem = Stem.saveStem(session, secStemName, null,secStemName, "" + secCount,  "", 
120                null, false);
121            stemsCreated++;
122            for (int classCount = 1000; classCount <= 1001; classCount++) {
123              Stem classStem = null;
124              try {
125                classStem = secStem.addChildStem("" + classCount, "" + classCount); 
126                stemsCreated++;
127              } catch (StemAddException sae) {
128                //hopefully it is because the group already exists
129                classStem = StemFinder.findByName(session, secStem.getName() + ":" + classCount, true);
130              }
131              for (int termCount = 2000; termCount <= 2000; termCount++) {
132                Stem termStem = null;
133                try {
134                  termStem = classStem.addChildStem("" + termCount, "" + termCount);
135                  stemsCreated++;
136                } catch (StemAddException sae) {
137                  //hopefully it is because the group already exists
138                  termStem = StemFinder.findByName(session, classStem.getName() + ":" + termCount, true);
139                }
140                
141                Role instructors = null;
142                try {
143                  instructors = termStem.addChildRole("instructors", "instructors");
144                  groupsCreated++;
145                } catch (GroupAddException gae) {
146                  //hopefully it is because the group already exists
147                  instructors = GroupFinder.findByName(session, termStem.getName() + ":instructors", true);
148                }
149                
150                Role ta = null;
151                try {
152                  ta = termStem.addChildRole("TAs", "TAs");
153                  groupsCreated++;
154                } catch (GroupAddException gae) {
155                  //hopefully it is because the group already exists
156                  ta = GroupFinder.findByName(session, termStem.getName() + ":TAs", true);
157                }
158                  
159                Role students = null;
160                try {
161                  students = termStem.addChildRole("students", "students");
162                  groupsCreated++;
163                } catch (GroupAddException gae) {
164                  //hopefully it is because the group already exists
165                  students = GroupFinder.findByName(session, termStem.getName() + ":students", true);
166                }
167 
168                AttributeDefName attributeDefName = new AttributeDefNameSave(session, attributeDef).assignName(termStem.getName() + ":testAttribute").save();
169                instructors.getPermissionRoleDelegate().assignRolePermission(attributeDefName);
170                students.getPermissionRoleDelegate().assignRolePermission(attributeDefName);
171                ta.getPermissionRoleDelegate().assignRolePermission(attributeDefName);
172 
173                for (int memberCount = 0; memberCount <= 0; memberCount++) { 
174                  Subject subject = subjectFindCreateById("0" + catelogCount + secCount + memberCount);
175                  try {
176                    instructors.addMember(subject, false);
177                    membersAdded++;
178                  } catch (MemberAddException mae) {
179                    //hopefully it is because the member is already a member
180                  }
181                }
182 
183                for (int memberCount = 0; memberCount <= 0; memberCount++) { 
184                  Subject subject = subjectFindCreateById("1" + catelogCount + secCount + memberCount);
185                  try {
186                    ta.addMember(subject, false);
187                    membersAdded++;
188                  } catch (MemberAddException mae) {
189                    //hopefully it is because the member is already a member
190                  }
191                    
192                }
193 
194                for (int memberCount = 0; memberCount <= 11; memberCount++) { 
195                  Subject subject = subjectFindCreateById("2" + catelogCount + secCount + memberCount);
196                  try {
197                    students.addMember(subject, false);
198                    membersAdded++;
199                  } catch (MemberAddException mae) {
200                    //hopefully it is because the member is already a member
201                  }
202                }
203              }
204            }
205 
206            System.out.println(new Date() + ", stem: " + secStem.getName() + ", stems: " + stemsCreated + ", Groups: " + groupsCreated + ", Memberships: " + membersAdded);
207          }
208       }
209     }
210     System.out.println("Stems created: " + stemsCreated);
211     System.out.println("Groups created: " + groupsCreated);
212     System.out.println("Memberships added: " + membersAdded);
213 
214   }

216   /**
217    * find or create subject in the default subject api tables
218    * @param subjectId
219    * @return the subject
220    * @throws SubjectNotFoundException 
221    * @throws SubjectNotUniqueException 
222    */
223   private static Subject subjectFindCreateById(final String subjectId) 
224       throws SubjectNotFoundException, SubjectNotUniqueException {
225 
226     try {
227       return SubjectFinder.findById(subjectId, true);
228     } catch (SubjectNotFoundException snfe) {
229       //create;
230       
231       HibernateSession.callbackHibernateSession(
232           GrouperTransactionType.READ_WRITE_NEW, AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
233 
234         @SuppressWarnings("deprecation")
235         public Object callback(HibernateHandlerBean hibernateHandlerBean)
236             throws GrouperDAOException {
237           HibernateSession hibernateSession = hibernateHandlerBean.getHibernateSession();
238           Session session = hibernateSession.getSession();
239           Connection connection = session.connection();
240           PreparedStatement preparedStatement = null;
241           try {
242 
243             //INSERT INTO SUBJECT ( SUBJECTID, SUBJECTTYPEID, NAME ) VALUES ('10000000', 'person', '10000000') 
244             String query = "INSERT INTO SUBJECT ( SUBJECTID, SUBJECTTYPEID, NAME ) VALUES (?, 'person', ?)";
245             preparedStatement = connection.prepareStatement(query);
246             preparedStatement.setString(1, subjectId);
247             preparedStatement.setString(2, subjectId);
248             preparedStatement.execute();
249             
250             //INSERT INTO SUBJECTATTRIBUTE ( SUBJECTID, NAME, VALUE, SEARCHVALUE ) VALUES ( 
251             //'10000000', 'name', 'Nancy Eggers', 'nancy eggers')
252             //INSERT INTO SUBJECTATTRIBUTE ( SUBJECTID, NAME, VALUE, SEARCHVALUE ) VALUES ( 
253             //'10000000', 'loginid', 'Nancy Eggers', 'nancy eggers')
254             //INSERT INTO SUBJECTATTRIBUTE ( SUBJECTID, NAME, VALUE, SEARCHVALUE ) VALUES ( 
255             //'10000000', 'description', 'Nancy Eggers', 'nancy eggers'); 
256             for (String name : new String[]{"name", "loginid", "description"}) {
257               
258               query = "INSERT INTO SUBJECTATTRIBUTE ( SUBJECTID, NAME, VALUE, SEARCHVALUE ) VALUES ( " +
259                   "?, ?, ?, ?)";
260               preparedStatement = connection.prepareStatement(query);
261               preparedStatement.setString(1, subjectId);
262               preparedStatement.setString(2, name);
263               preparedStatement.setString(3, subjectId);
264               preparedStatement.setString(4, subjectId);
265               preparedStatement.execute();
266             }
267             
268             connection.commit();
269           } catch (Exception e) {
270             throw new RuntimeException(e);
271           } finally {
272             HibUtils.closeQuietly(preparedStatement);
273           }
274           return null;
275         }
276         
277       });
278       
279     }
280     return SubjectFinder.findById(subjectId, true);
281   }
