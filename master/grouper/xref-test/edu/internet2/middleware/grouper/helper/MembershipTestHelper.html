376   private static void _testEff(Group g, Subject subj, Member m) {
377     // The basics
378     Assert.assertTrue("g hasMember m",    g.hasMember(subj));
379     Assert.assertTrue("g hasEffMember m", g.hasEffectiveMember(subj));

385       Membership ms = (Membership) iter.next();
386       try {
387         Assert.assertNotNull("eff ms g !null", ms.getGroup());

389           "eff ms g",

392       }
393       catch (GroupNotFoundException eGNF) {
394         Assert.fail("eff ms g: " + eGNF.getMessage());
395       }
396       try {
397         Assert.assertNotNull("eff ms m !null", ms.getMember());
398         Assert.assertTrue(
399           "eff ms m", 
400           ms.getMember().getUuid().equals(m.getUuid())
401         );
402       }
403       catch (MemberNotFoundException eMNF) {
404         Assert.fail("eff ms m: " + eMNF.getMessage());

406       Assert.assertNotNull("eff ms l !null", ms.getList());  
407       Assert.assertTrue( "eff ms l", ms.getList().equals(Group.getDefaultList()));

409       try {
410         ms.getViaGroup();
411         Assert.assertTrue("eff ms has via", true);
412       }
413       catch (GroupNotFoundException eGNF) {
414         Assert.fail("eff ms has no via");
415       }

511     try {
512       Set<Membership> memberships = MembershipFinder.findEffectiveMemberships(s, g, subj,
513         f, via, depth);
514       
515       Iterator<Membership> it = memberships.iterator();
516       while (it.hasNext()) {
517         Membership curr = it.next();
518         Membership parentOfCurr = curr.getParentMembership();
519         if (parent.equals(parentOfCurr)) {
520           return curr;
521         }
522       }
523     } catch (Exception e) {
524       // do nothing 
525     }
526     
527     return null;
