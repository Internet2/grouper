95    /**
96     * Add test subjects to registry.
97     * <p/>
98     * @since   1.2.1
99     */
100   public void addSubjects(int number) {
101     RegistrySubjectDAO  dao   = GrouperDAOFactory.getFactory().getRegistrySubject();
102     RegistrySubject  _subj;
103     for (int i=0; i<number; i++) {
104       String  id    = _getSuffix(i);
105       String  name  = "subject " + id;
106       _subj = new RegistrySubject();
107       _subj.setId(id);
108       _subj.setName(name);
109       _subj.setTypeString("person");
110       dao.create(_subj);
111       this.subjects.put(id, _subj);  
112       LOG.debug("created subject: " + _subj);
113     }
114   }

141   /**
142    * Adds a child {@link Group} beneath <i>parent</i>.
143    * <p/>
144    * @return  Child {@link Group}. 
145    * @throws  GrouperException

148   public Group addGroup(Stem parent, String extn, String displayExtn) 
149     throws  GrouperException
150   {
151     try {
152       Group child = parent.addChildGroup(extn, displayExtn);

158     }
159     catch (InsufficientPrivilegeException eIP) {

337   /**
338    * <pre>
339    * This will make stems and groups in the stems.
340    * e.g. group: i2:a:a, i2:a:b, then i2:b:a, i2:b:b
341    * </pre>
342    * @param nStems
343    * @param nGroups
344    * @param nSubjects
345    * @return R which is the 
346    * @throws Exception
347    */
348   public static R populateRegistry(int nStems, int nGroups, int nSubjects) 
349    
350   {
351     LOG.info("populateRegistry");   
352     R r  = new R();
353     r.rs    = SessionHelper.getRootSession();
354     r.root  = StemFinder.findRootStem(r.rs);
355     r.ns    = new StemSave(r.rs).assignName("i2").assignDisplayExtension("internet2").save();
356     for (int i=0; i<nStems; i++) {
357       String  nsExtn  = _getSuffix(i);
358       Stem    ns      = r.ns.addChildStem(nsExtn, "stem " + nsExtn);
359       LOG.debug("created stem: " + ns);
360       r.stems.put(nsExtn, ns);
361       for (int j=0; j<nGroups; j++) {
362         String  gExtn = _getSuffix(j);
363         String  key   = nsExtn + ":" + gExtn;
364         Group   g     = ns.addChildGroup(gExtn, "group " + gExtn);
365         LOG.debug("created group: " + g);
366         r.groups.put(key, g);
367       }
368     }
369     r.addSubjects(nSubjects);
370 
371     return r;
372   } // public static R populateRegistry(nStems, nGroups, nSubjects)
