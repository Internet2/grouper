61    public void testHibernate() {
62  
63      GrouperSession grouperSession = GrouperSession.start( SubjectFinder.findRootSubject() );
64      Stem root = StemFinder.findRootStem(grouperSession);
65      Stem top = root.addChildStem("top", "top display name");
66      AttributeDef attributeDef = top.addChildAttributeDef("test", AttributeDefType.attr);
67  
68      AttributeAssignAction attributeAssignAction = new AttributeAssignAction();
69      attributeAssignAction.setId(GrouperUuid.getUuid());
70      attributeAssignAction.setAttributeDefId(attributeDef.getId());
71  
72      attributeAssignAction.save();
73      
74      attributeAssignAction = GrouperDAOFactory.getFactory().getAttributeAssignAction()
75        .findById(attributeAssignAction.getId(), true);
76      
77      attributeAssignAction.delete();
78  
79      attributeAssignAction = GrouperDAOFactory.getFactory().getAttributeAssignAction()
80        .findById(attributeAssignAction.getId(), false);
81      
82      assertNull(attributeAssignAction);
83    }

115   public static AttributeAssignAction exampleAttributeAssignActionDb(String actionName) {
116     
117     AttributeDef attributeDef = AttributeDefFinder.findByName("test:attributeDefAction", false);
118 
119     if (attributeDef == null) {
120       Stem stem = new StemSave(GrouperSession.staticGrouperSession()).assignSaveMode(SaveMode.INSERT_OR_UPDATE)
121         .assignStemNameToEdit("test").assignName("test").assignCreateParentStemsIfNotExist(true)
122         .assignDescription("description").save();
123       attributeDef = stem.addChildAttributeDef("attributeDefAction", AttributeDefType.attr);
124     }
125     
126     attributeDef.getAttributeDefActionDelegate().addAction(actionName);
127     
128     return GrouperDAOFactory.getFactory().getAttributeAssignAction().findByUuidOrKey(null, attributeDef.getId(), actionName, true);
129

151   /**
152    * make sure update properties are detected correctly
153    */
154   public void testXmlInsert() {
155     
156     GrouperSession.startRootSession();
157     
158     AttributeAssignAction attributeAssignActionOriginal =  exampleAttributeAssignActionDb("testInsert");
159     
160     //do this because last membership update isnt there, only in db
161     attributeAssignActionOriginal = exampleRetrieveAttributeAssignActionDb("testInsert");
162     AttributeAssignAction attributeAssignActionCopy = exampleRetrieveAttributeAssignActionDb("testInsert");
163     AttributeAssignAction attributeAssignActionCopy2 = exampleRetrieveAttributeAssignActionDb("testInsert");
164     attributeAssignActionCopy.delete();
165     
166     //lets insert the original
167     attributeAssignActionCopy2.xmlSaveBusinessProperties(null);
168     attributeAssignActionCopy2.xmlSaveUpdateProperties();
169 
170     //refresh from DB
171     attributeAssignActionCopy = exampleRetrieveAttributeAssignActionDb("testInsert");
172     
173     assertFalse(attributeAssignActionCopy == attributeAssignActionOriginal);
174     assertFalse(attributeAssignActionCopy.xmlDifferentBusinessProperties(attributeAssignActionOriginal));
175     assertFalse(attributeAssignActionCopy.xmlDifferentUpdateProperties(attributeAssignActionOriginal));
176     
177   }

179   /**
180    * make sure update properties are detected correctly
181    */
182   public void testXmlDifferentUpdateProperties() {
183     
184     @SuppressWarnings("unused")
185     GrouperSession grouperSession = GrouperSession.startRootSession();
186     
187     AttributeAssignAction attributeAssignAction = null;
188     AttributeAssignAction exampleAttributeAssignAction = null;
189 
190     
191     //TEST UPDATE PROPERTIES
192     {
193       attributeAssignAction = exampleAttributeAssignActionDb();
194       exampleAttributeAssignAction = exampleRetrieveAttributeAssignActionDb();
195       
196       attributeAssignAction.setContextId("abc");
197       
198       assertFalse(attributeAssignAction.xmlDifferentBusinessProperties(exampleAttributeAssignAction));
199       assertTrue(attributeAssignAction.xmlDifferentUpdateProperties(exampleAttributeAssignAction));
200 
201       attributeAssignAction.setContextId(exampleAttributeAssignAction.getContextId());
202       attributeAssignAction.xmlSaveUpdateProperties();
203 
204       attributeAssignAction = exampleRetrieveAttributeAssignActionDb();
205       
206       assertFalse(attributeAssignAction.xmlDifferentBusinessProperties(exampleAttributeAssignAction));
207       assertFalse(attributeAssignAction.xmlDifferentUpdateProperties(exampleAttributeAssignAction));
208       
209     }
210     
211     {
212       attributeAssignAction = exampleAttributeAssignActionDb();
213       exampleAttributeAssignAction = exampleRetrieveAttributeAssignActionDb();
214 
215       attributeAssignAction.setCreatedOnDb(99L);
216       
217       assertFalse(attributeAssignAction.xmlDifferentBusinessProperties(exampleAttributeAssignAction));
218       assertTrue(attributeAssignAction.xmlDifferentUpdateProperties(exampleAttributeAssignAction));
219 
220       attributeAssignAction.setCreatedOnDb(exampleAttributeAssignAction.getCreatedOnDb());
221       attributeAssignAction.xmlSaveUpdateProperties();
222       
223       attributeAssignAction = exampleRetrieveAttributeAssignActionDb();
224       
225       assertFalse(attributeAssignAction.xmlDifferentBusinessProperties(exampleAttributeAssignAction));
226       assertFalse(attributeAssignAction.xmlDifferentUpdateProperties(exampleAttributeAssignAction));

229     {

237

273       assertTrue(attributeAssignAction.xmlDifferentBusinessProperties(exampleAttributeAssignAction));
274       assertFalse(attributeAssignAction.xmlDifferentUpdateProperties(exampleAttributeAssignAction));
275 
276       attributeAssignAction.setAttributeDefId(exampleAttributeAssignAction.getAttributeDefId());
277       attributeAssignAction.xmlSaveBusinessProperties(exampleAttributeAssignAction.clone());
278       attributeAssignAction.xmlSaveUpdateProperties();
279       
280       attributeAssignAction = exampleRetrieveAttributeAssignActionDb();
281       
282       assertFalse(attributeAssignAction.xmlDifferentBusinessProperties(exampleAttributeAssignAction));
283       assertFalse(attributeAssignAction.xmlDifferentUpdateProperties(exampleAttributeAssignAction));
284     
285     }
286

293       assertTrue(attributeAssignAction.xmlDifferentBusinessProperties(exampleAttributeAssignAction));
294       assertFalse(attributeAssignAction.xmlDifferentUpdateProperties(exampleAttributeAssignAction));
