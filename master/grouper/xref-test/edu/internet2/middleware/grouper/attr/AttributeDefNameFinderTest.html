85    public void testFindByIdIndexSecure() {
86      
87      GrouperSession grouperSession = GrouperSession.startRootSession();
88      
89      AttributeDef attributeDef = new AttributeDefSave(grouperSession).assignName("test:attributeDef")
90        .assignAttributeDefType(AttributeDefType.attr).assignCreateParentStemsIfNotExist(true).save();
91      
92      AttributeDefName attributeDefName = new AttributeDefNameSave(grouperSession, attributeDef)
93        .assignCreateParentStemsIfNotExist(true)
94        .assignName("test:attributeDefName").save();
95      
96      attributeDef.getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_VIEW, false);
97      attributeDef.getPrivilegeDelegate().revokePriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_VIEW, false);
98      attributeDef.getPrivilegeDelegate().revokePriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_READ, false);
99      
100     GrouperSession.stopQuietly(grouperSession);
101     
102     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ0);
103 
104     AttributeDefName found = AttributeDefNameFinder.findByIdIndexSecure(attributeDefName.getIdIndex(), true, null);
105     
106     assertEquals(found.getName(), attributeDefName.getName());
107     
108     GrouperSession.stopQuietly(grouperSession);
109     
110     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ1);
111     
112     found = AttributeDefNameFinder.findByIdIndexSecure(attributeDefName.getIdIndex(), false, null);
113     
114     assertNull(found);
115     
116     try {
117       AttributeDefNameFinder.findByIdIndexSecure(attributeDefName.getIdIndex(), true, null);
118       fail("shouldnt get here");
119     } catch (AttributeDefNameNotFoundException gnfe) {
120       //good
121     }
122     
123     try {
124       AttributeDefNameFinder.findByIdIndexSecure(123456789L, true, null);
125       fail("shouldnt get here");
126     } catch (AttributeDefNameNotFoundException gnfe) {
127       //good
128     }
129

307    * make sure update properties are detected correctly

309   @SuppressWarnings("unused")
310   public void testFinderByStem() {
311     
312     GrouperSession grouperSession = GrouperSession.startRootSession();
313     
314     AttributeDef attributeDef1 = new AttributeDefSave(grouperSession).assignName("test:attributeDef1")
315       .assignAttributeDefType(AttributeDefType.attr).assignCreateParentStemsIfNotExist(true)
316       .assignToGroup(true).save();
317     
318     AttributeDef attributeDef2 = new AttributeDefSave(grouperSession).assignName("test:attributeDef2")
319       .assignAttributeDefType(AttributeDefType.attr).assignCreateParentStemsIfNotExist(true)
320       .assignToGroup(true).save();
321   
322     AttributeDefName attributeDefName1_1_7 = new AttributeDefNameSave(grouperSession, attributeDef1)
323       .assignCreateParentStemsIfNotExist(true)
324       .assignName("test1:test2:serviceDefName1_7").save();
325     
326     Stem stemTest1 = StemFinder.findByName(grouperSession, "test1", true);
327     
328     AttributeDefName attributeDefName1_1_8 = new AttributeDefNameSave(grouperSession, attributeDef1)
329       .assignCreateParentStemsIfNotExist(true)
330       .assignName("test1:serviceDefName1_8").save();
331     
332     AttributeDefName attributeDefName1_1_9 = new AttributeDefNameSave(grouperSession, attributeDef1)
333       .assignCreateParentStemsIfNotExist(true)
334       .assignName("test1:test3:test5:serviceDefName1_9").save();
335   
336     AttributeDefName attributeDefName1_1_10 = new AttributeDefNameSave(grouperSession, attributeDef1)
337       .assignCreateParentStemsIfNotExist(true)
338       .assignName("test2:test3:test5:serviceDefName1_10").save();
339 
340     
341     AttributeDefName attributeDefName1_1_1 = new AttributeDefNameSave(grouperSession, attributeDef2)
342       .assignCreateParentStemsIfNotExist(true)
343       .assignName("test2:attributeDefName1_1").save();
344     AttributeDefName attributeDefName1_1_2 = new AttributeDefNameSave(grouperSession, attributeDef2)
345       .assignCreateParentStemsIfNotExist(true)
346       .assignName("test2:test3:attributeDefName1_2").save();
347 
348     Stem stemTest2 = StemFinder.findByName(grouperSession, "test2", true);
349 
350     attributeDef1.getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, false);
351     attributeDef1.getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_UPDATE, false);
352     attributeDef2.getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ1, AttributeDefPrivilege.ATTR_UPDATE, false);
353     
354     Set<AttributeDefName> attributeDefNames = new AttributeDefNameFinder()
355       .addPrivilege(AttributeDefPrivilege.ATTR_ADMIN).assignSubject(SubjectTestHelper.SUBJ0).findAttributeNames();
356     
357     assertEquals(0, attributeDefNames.size());
358 
359     attributeDefNames = new AttributeDefNameFinder().assignScope("test")
360       .addPrivilege(AttributeDefPrivilege.ATTR_UPDATE).assignSubject(SubjectTestHelper.SUBJ0).findAttributeNames();
361     
362     assertEquals(GrouperUtil.toStringForLog(attributeDefNames), 4, attributeDefNames.size());
363     assertTrue(attributeDefNames.contains(attributeDefName1_1_7));
364     assertTrue(attributeDefNames.contains(attributeDefName1_1_8));
365     assertTrue(attributeDefNames.contains(attributeDefName1_1_9));
366     assertTrue(attributeDefNames.contains(attributeDefName1_1_10));
367     
368     attributeDefNames = new AttributeDefNameFinder().assignStemScope(Scope.ONE).assignParentStemId(stemTest1.getId())
369       .addPrivilege(AttributeDefPrivilege.ATTR_UPDATE).assignSubject(SubjectTestHelper.SUBJ0).findAttributeNames();
370     
371     assertEquals(GrouperUtil.toStringForLog(attributeDefNames), 1, attributeDefNames.size());
372     assertTrue(attributeDefNames.contains(attributeDefName1_1_8));
373       
374     attributeDefNames = new AttributeDefNameFinder().assignStemScope(Scope.ONE).assignParentStemId(stemTest2.getId())
375       .addPrivilege(AttributeDefPrivilege.ATTR_UPDATE).assignSubject(SubjectTestHelper.SUBJ0).findAttributeNames();
376       
377     assertEquals(GrouperUtil.toStringForLog(attributeDefNames), 0, attributeDefNames.size());
378         
379     attributeDefNames = new AttributeDefNameFinder().assignStemScope(Scope.SUB).assignParentStemId(stemTest1.getId())
380       .addPrivilege(AttributeDefPrivilege.ATTR_UPDATE).assignSubject(SubjectTestHelper.SUBJ0).findAttributeNames();
381     
382     assertEquals(GrouperUtil.toStringForLog(attributeDefNames), 3, attributeDefNames.size());
383     assertTrue(attributeDefNames.contains(attributeDefName1_1_7));
384     assertTrue(attributeDefNames.contains(attributeDefName1_1_8));
385     assertTrue(attributeDefNames.contains(attributeDefName1_1_9));
386
