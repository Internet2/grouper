100     super.setUp();
101     try {

103       
104       String wheelGroupName = GrouperConfig.getProperty("groups.wheel.group");
105       
106       if (StringUtils.isBlank(wheelGroupName)) {
107         throw new RuntimeException("grouper.properties must have an extry for " +
108         		"groups.wheel.group, e.g. etc:sysadmingroup");
109       }
110       GrouperSession grouperSession = r.startRootSession();
111       String extension = GrouperUtil.extensionFromName(wheelGroupName);
112       wheel = Group.saveGroup(grouperSession, wheelGroupName, null, wheelGroupName, extension, "description for " + extension, SaveMode.INSERT_OR_UPDATE, true);
113       dev   = r.getGroup("i2mi:grouper", "grouper-dev");
114       subjA = r.getSubject("a");
115     }
116     catch (Exception e) {
117       throw new GrouperException( "test setUp() error: " + e.getMessage(), e );

144     throws  GrantPrivilegeException,

155     Group wheelGroup = GroupFinder.findByUuid(grouperSession,wheelUuid, true);

158       wheelGroup.addMember( this.subjA );
159     } finally {
160     }
161     
162     // start session and turn on wheel
163     GrouperSession s = GrouperSession.start(this.subjA);
164     GrouperConfig.retrieveConfig().propertiesOverrideMap().put( GrouperConfig.PROP_USE_WHEEL_GROUP, "true" );

166     assertTrue( s.getMember().canAdmin(dev) );
167     GrouperConfig.retrieveConfig().propertiesOverrideMap().put( GrouperConfig.PROP_USE_WHEEL_GROUP, "false" );

175    //Since I added caching for wheel group members it can take up to 2 minutes before 
176    //privileges will work
177   public void test_addingMemberToWheelGroupShouldElevatePrivilegesWithinSession() 
178     throws  GroupNotFoundException,
179             InsufficientPrivilegeException,
180             MemberAddException,
181             MemberNotFoundException,
182             SessionException
183   {
184     GrouperSession s = GrouperSession.start( SubjectFinder.findRootSubject() );
185     GrouperConfig.retrieveConfig().propertiesOverrideMap().put( GrouperConfig.PROP_USE_WHEEL_GROUP, "true" );
186     Member  mA  = MemberFinder.findBySubject(s, subjA, true);
187     Group   g   = GroupFinder.findByUuid( s, wheel.getUuid(), true );
188     //WheelAccessResolver caches wheel group membership
189     //so doing a pre-check breaks the subsequent ADMIN check
190     assertFalse( "does not have ADMIN", g.hasAdmin(subjA) );
191     assertFalse( "cannot ADMIN", mA.canAdmin(g) );
192     s.getAccessResolver().flushCache();
193     g.addMember(subjA);
194     assertTrue( "now has ADMIN", g.hasAdmin(subjA) );
195     assertTrue( "now can ADMIN", mA.canAdmin(g) );
196 
197     GrouperConfig.retrieveConfig().propertiesOverrideMap().put( GrouperConfig.PROP_USE_WHEEL_GROUP, "false" );
198   }

201    * If a subject is a member of the wheel group and we're in self mode, the 
202    * subject should not have wheel access.
203    * @throws GrantPrivilegeException
204    * @throws GroupNotFoundException
205    * @throws InsufficientPrivilegeException
206    * @throws MemberAddException
207    * @throws SchemaException
208    * @throws SessionException

210   public void test_preventAdminWhenMemberOfWheelAndSelfMode()
211       throws GrantPrivilegeException, GroupNotFoundException,
212       InsufficientPrivilegeException, MemberAddException, SchemaException,
213       SessionException {
214     // make this.subjA a member of wheel
215     Subject rootSubject = SubjectFinder.findRootSubject();
216     GrouperSession grouperSession = GrouperSession.start(rootSubject);
217     String wheelUuid = wheel.getUuid();
218     Group wheelGroup = GroupFinder.findByUuid(grouperSession, wheelUuid, true);
219 
220     wheelGroup.addMember(this.subjA);
221 
222     // start session and turn on wheel
223     GrouperSession s = GrouperSession.start(this.subjA);
224     GrouperConfig.retrieveConfig().propertiesOverrideMap().put(GrouperConfig.PROP_USE_WHEEL_GROUP, "true");
225     s.setConsiderIfWheelMember(false);
226     // verify no admin privilege
227     assertFalse(s.getMember().canAdmin(dev));
228     GrouperConfig.retrieveConfig().propertiesOverrideMap().put(GrouperConfig.PROP_USE_WHEEL_GROUP, "false");
