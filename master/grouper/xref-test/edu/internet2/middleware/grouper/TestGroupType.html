89    public void testAddAndDeleteCustomTypeAsNonRoot() {
90      LOG.info("testAddAndDeleteCustomTypeAsNonRoot");
91      try {
92        R r = R.populateRegistry(1, 1, 1);
93    
94        String    tName = "test type";
95        // Create the custom type as root
96        GroupType type  = GroupType.createType(r.rs, tName);
97    
98        // Now allow a non-root subj to admin this group
99        Group   g     = r.getGroup("a", "a");
100       Subject subj  = r.getSubject("a");
101       g.grantPriv(subj, AccessPrivilege.ADMIN);
102       type.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(subj, AttributeDefPrivilege.ATTR_READ, false);
103       type.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(subj, AttributeDefPrivilege.ATTR_UPDATE, false);
104   
105       // Now start non-root session and add+delete group type as non-root
106       GrouperSession nrs = GrouperSession.start(subj);
107       Assert.assertFalse( "no custom type"      , g.hasType(type) );
108       g.addType(type);
109       Assert.assertTrue(  "now has custom type" , g.hasType(type) );
110       g.deleteType(type);
111       Assert.assertFalse( "custom type removed" , g.hasType(type) );
112       nrs.stop();
113   
114       r.rs.stop();
115     }
116     catch (Exception e) {
117       T.e(e);
118     }

302   public void testUseCustomAttributeAsNonRoot() {
303     LOG.info("testUseCustomAttributeAsNonRoot");
304     try {
305       R         r       = R.populateRegistry(1, 1, 1);
306       Group     gA      = r.getGroup("a", "a");
307       String    name    = gA.getName();
308       Subject   subjA   = r.getSubject("a");
309       GroupType custom  = GroupType.createType(r.rs, "custom");
310       AttributeDefName     attr    = custom.addAttribute(
311         r.rs, "custom a", false
312       );
313       gA.addType(custom);
314       gA.grantPriv(subjA, AccessPrivilege.ADMIN);
315       custom.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(subjA, AttributeDefPrivilege.ATTR_READ, false);
316       custom.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(subjA, AttributeDefPrivilege.ATTR_UPDATE, false);
317       custom.internal_getAttributeDefForAttributes().getPrivilegeDelegate().grantPriv(subjA, AttributeDefPrivilege.ATTR_READ, false);
318       custom.internal_getAttributeDefForAttributes().getPrivilegeDelegate().grantPriv(subjA, AttributeDefPrivilege.ATTR_UPDATE, false);
319       r.rs.stop();
320   
321       // Now test-and-set attribute as !root
322       GrouperSession  s = GrouperSession.start(subjA);
323       Group           g = GroupFinder.findByName(s, name, true);
324       Assert.assertTrue(
325         "group has custom type", g.hasType(custom)
326       );         
327       Assert.assertTrue(
328         "group does not have attribute set - yet",
329         g.getAttributeValue(attr.getLegacyAttributeName(true), false, false).equals(GrouperConfig.EMPTY_STRING)
330       );
331       try {
332         g.setAttribute(attr.getLegacyAttributeName(true), name);
333 
334         Assert.assertTrue("set attribute", true);
335       }
336       catch (Exception e) {
337         Assert.fail("exception while setting custom attribute! - " + e.getMessage());
338       }
339       T.string("now group has attribute set", name, g.getAttributeValue(attr.getLegacyAttributeName(true), false, true));
340       s.stop();
341   
342       // Now make sure it was properly persisted
343       GrouperSession  S = GrouperSession.start(subjA);
344       Group           G = GroupFinder.findByName(S, name, true);
345       T.string("attribute was persisted", name, G.getAttributeValue(attr.getLegacyAttributeName(true), false, true));
346       S.stop();
347     }
348     catch (Exception e) {
349       T.e(e);
350     }
351   } // public void testUseCustomAttributeAsNonRoot()
