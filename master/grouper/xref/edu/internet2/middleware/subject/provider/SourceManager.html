136     /**
137      * loop through config beans from subject.properties 
138      */
139     public void processConfigBeans() {
140       
141       for (SubjectStatusConfig subjectStatusConfig : this.sourceIdToStatusConfigs.values()) {
142 
143         {
144           String statusLabel = subjectStatusConfig.getStatusLabel();
145           if (!StringUtils.isBlank(statusLabel)) {
146             this.statusLabels.add(statusLabel);
147           }
148         }
149         
150         {
151           Set<String> statusesFromUser = subjectStatusConfig.getStatusesFromUser();
152           this.statusesFromUser.addAll(statusesFromUser);
153         }
154 
155         {
156           String statusAllFromUser = subjectStatusConfig.getStatusAllFromUser();
157           if (!StringUtils.isBlank(statusAllFromUser)) {
158             this.statusAllFromUser.add(statusAllFromUser);
159           }
160         }
161       }
162     }

202    * print out the config for the subject API
203    * @return the config
204    */
205   public String printConfig() {
206     try {
207       StringBuilder result = new StringBuilder();
208 
209       File subjectPropertiesFile = SubjectUtils.fileFromResourceName("subject.properties");
210       String subjectPropertiesFileLocation = subjectPropertiesFile == null ? " [cant find subject.properties]"
211           : SubjectUtils.fileCanonicalPath(subjectPropertiesFile);
212       result.append("subject.properties read from: " + subjectPropertiesFileLocation + "\n");
213 
214       result.append("sources configured in:        subject.properties\n");
215       File sourcesXmlFile = SubjectUtils.fileFromResourceName("sources.xml");
216       if (sourcesXmlFile != null && sourcesXmlFile.exists() && sourcesXmlFile.isFile()) {
217         String sourcesError = "NON-FATAL ERROR:              subject sources are read from subject.properties but you "
218             + "still have a sources.xml on the classpath which is confusing, please backup and remove this file: " + sourcesXmlFile.getAbsolutePath();
219         result.append(sourcesError + "\n");
220         log.error(sourcesError);
221       }
222       
223       //at this point, we have a subject.properties...  now check it out
224       Collection<Source> sources = SourceManager.getInstance().getSources();
225       for (Source source : sources) {
226         result.append(source.printConfig()).append("\n");
227       }
228       //dont end in newline
229       if (result.toString().endsWith("\n")) {
230         result.deleteCharAt(result.length() - 1);
231       }
232       return result.toString();
233     } catch (Exception e) {
234       log.error("Cant print subject API configs", e);
235     }
236     return "Cant print subject API configs";
237

271   /**
272    * Gets Source for the argument source ID.
273    * @param sourceId
274    * @return Source
275    * @throws SourceUnavailableException
276    */
277   public Source getSource(String sourceId) throws SourceUnavailableException {
278     Source source = this.sourceMap.get(sourceId);
279     if (source == null) {
280       
281       StringBuilder allSources = new StringBuilder();
282       int i=0;
283       for (String theSourceId : this.sourceMap.keySet()) {
284         allSources.append(theSourceId);
285         if (i != this.sourceMap.size() - 1) {
286           allSources.append(", ");
287         }
288         i++;
289       }
290       
291       throw new SourceUnavailableException("Source not found: '" + sourceId + "', available sources are: " + allSources);
292     }
293     return source;
294   }

334   public void loadSource(Source source) {
335     log.debug("Loading source: " + source.getId());
336     
337     //put in map before initting
338     this.sourceMap.put(source.getId(), source);
339 
340     for (Iterator it = source.getSubjectTypes().iterator(); it.hasNext();) {
341       SubjectType../../../../edu/internet2/middleware/subject/SubjectType.html#SubjectType">SubjectType type = (SubjectType) it.next();
342       Set<Source> sources = this.source2TypeMap.get(type);
343       if (sources == null) {
344         sources = new HashSet<Source>();
345         this.source2TypeMap.put(type, sources);
346       }
347       sources.add(source);
348     }
349 
350     //do this last in case it throws exceptions...
351     source.init();
352       
353     source.getSearchAttributes();
354     source.getSortAttributes();
355     source.getSubjectIdentifierAttributes();
356   }
