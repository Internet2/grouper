501   /**
502    * @see edu.internet2.middleware.grouper.GrouperAPI#onPreSave(edu.internet2.middleware.grouper.hibernate.HibernateSession)
503    */
504   @Override
505   public void onPreSave(HibernateSession hibernateSession) {
506     super.onPreSave(hibernateSession);
507     
508     // add change log entry for flat memberships
509     if (this.isActive() && (this.getFlatMembershipNotificationsOnSaveOrUpdate() || this.getFlatPrivilegeNotificationsOnSaveOrUpdate())) {
510       Set<ChangeLogEntry> changeLogEntryBatch = new LinkedHashSet<ChangeLogEntry>();
511       int batchSize = GrouperConfig.getHibernatePropertyInt("hibernate.jdbc.batch_size", 20);
512       if (batchSize <= 0) {
513         batchSize = 1;
514       }
515 
516       PITField pitField = null;
517       String ownerId = null;
518       String ownerName = null;
519       String privilegeName = null;
520       String privilegeType = null;
521       String ownerType = null;
522       boolean isMembership = false;
523       Set<PITMembership> pitMemberships = GrouperDAOFactory.getFactory().getPITMembershipView().findPITMembershipsJoinedWithNewPITGroupSet(this);
524       
525       if (pitMemberships.size() > 0) {
526         pitField = GrouperDAOFactory.getFactory().getPITField().findById(this.getFieldId(), true);
527         if (this.getOwnerGroupId() != null) {
528           PITGroup pitGroup = GrouperDAOFactory.getFactory().getPITGroup().findById(this.getOwnerId(), true);
529           ownerId = pitGroup.getSourceId();
530           ownerName = pitGroup.getName();
531           if (pitField.getType().equals(FieldType.LIST.getType())) {
532             isMembership = true;
533           } else {
534             privilegeType = FieldType.ACCESS.getType();
535             privilegeName = AccessPrivilege.listToPriv(pitField.getName()).getName();
536             ownerType = Membership.OWNER_TYPE_GROUP;
537           }
538         } else if (this.getOwnerStemId() != null) {
539           PITStem pitStem = GrouperDAOFactory.getFactory().getPITStem().findById(this.getOwnerId(), true);
540           ownerId = pitStem.getSourceId();
541           ownerName = pitStem.getName();
542           privilegeType = FieldType.NAMING.getType();
543           privilegeName = NamingPrivilege.listToPriv(pitField.getName()).getName();
544           ownerType = Membership.OWNER_TYPE_STEM;
545         } else if (this.getOwnerAttrDefId() != null) {
546           PITAttributeDef pitAttributeDef = GrouperDAOFactory.getFactory().getPITAttributeDef().findById(this.getOwnerId(), true);
547           ownerId = pitAttributeDef.getSourceId();
548           ownerName = pitAttributeDef.getName();
549           privilegeType = FieldType.ATTRIBUTE_DEF.getType();
550           privilegeName = AttributeDefPrivilege.listToPriv(pitField.getName()).getName();
551           ownerType = Membership.OWNER_TYPE_ATTRIBUTE_DEF;
552         } else {
553           throw new RuntimeException("Unable to determine owner of PIT Group Set: " + this.getId());
554         }
555       }
556       
557       Iterator<PITMembership> iter = pitMemberships.iterator();
558       while (iter.hasNext()) {
559         PITMembership pitMembership = iter.next();
560         
561         ChangeLogEntry changeLogEntry = null;
562         
563         if (isMembership && this.getFlatMembershipNotificationsOnSaveOrUpdate()) {
564           changeLogEntry = new ChangeLogEntry(false, ChangeLogTypeBuiltin.MEMBERSHIP_ADD,
565               ChangeLogLabels.MEMBERSHIP_ADD.id.name(), pitMembership.getSourceId(),
566               ChangeLogLabels.MEMBERSHIP_ADD.fieldName.name(), pitField.getName(),
567               ChangeLogLabels.MEMBERSHIP_ADD.fieldId.name(), pitField.getSourceId(),
568               ChangeLogLabels.MEMBERSHIP_ADD.memberId.name(), pitMembership.getMember().getSourceId(),
569               ChangeLogLabels.MEMBERSHIP_ADD.subjectId.name(), pitMembership.getMember().getSubjectId(),
570               ChangeLogLabels.MEMBERSHIP_ADD.sourceId.name(), pitMembership.getMember().getSubjectSourceId(),
571               ChangeLogLabels.MEMBERSHIP_ADD.groupId.name(), ownerId,
572               ChangeLogLabels.MEMBERSHIP_ADD.membershipType.name(), "flattened",
573               ChangeLogLabels.MEMBERSHIP_ADD.groupName.name(), ownerName,
574               ChangeLogLabels.MEMBERSHIP_ADD.subjectIdentifier0.name(), pitMembership.getMember().getSubjectIdentifier0());
575         } else if (!isMembership && this.getFlatPrivilegeNotificationsOnSaveOrUpdate()) {
576           changeLogEntry = new ChangeLogEntry(false, ChangeLogTypeBuiltin.PRIVILEGE_ADD,
577               ChangeLogLabels.PRIVILEGE_ADD.id.name(), pitMembership.getSourceId(),
578               ChangeLogLabels.PRIVILEGE_ADD.privilegeName.name(), privilegeName,
579               ChangeLogLabels.PRIVILEGE_ADD.fieldId.name(), pitField.getSourceId(),
580               ChangeLogLabels.PRIVILEGE_ADD.memberId.name(), pitMembership.getMember().getSourceId(),
581               ChangeLogLabels.PRIVILEGE_ADD.subjectId.name(), pitMembership.getMember().getSubjectId(),
582               ChangeLogLabels.PRIVILEGE_ADD.sourceId.name(), pitMembership.getMember().getSubjectSourceId(),
583               ChangeLogLabels.PRIVILEGE_ADD.privilegeType.name(), privilegeType,
584               ChangeLogLabels.PRIVILEGE_ADD.ownerType.name(), ownerType,
585               ChangeLogLabels.PRIVILEGE_ADD.ownerId.name(), ownerId,
586               ChangeLogLabels.PRIVILEGE_ADD.membershipType.name(), "flattened",
587               ChangeLogLabels.PRIVILEGE_ADD.ownerName.name(), ownerName);
588         }
589 
590         if (changeLogEntry != null) {
591           changeLogEntry.setContextId(this.getContextId());
592           changeLogEntry.setCreatedOnDb(this.getStartTimeDb());
593           changeLogEntryBatch.add(changeLogEntry);
594           
595           if (changeLogEntryBatch.size() % batchSize == 0) {
596             GrouperDAOFactory.getFactory().getChangeLogEntry().saveBatch(changeLogEntryBatch, false);
597             changeLogEntryBatch.clear();
598           }
599         }
600       }
601       
602       // make sure all changes get made      
603       if (changeLogEntryBatch.size() > 0) {
604         GrouperDAOFactory.getFactory().getChangeLogEntry().saveBatch(changeLogEntryBatch, false);
605         changeLogEntryBatch.clear();
606       }
607     }
608   }

610   /**
611    * @see edu.internet2.middleware.grouper.GrouperAPI#onPreUpdate(edu.internet2.middleware.grouper.hibernate.HibernateSession)
612    */
613   @Override
614   public void onPreUpdate(HibernateSession hibernateSession) {
615     super.onPreUpdate(hibernateSession);
616     
617     // add change log entry for flat memberships
618     if (!this.isActive() && this.dbVersion().isActive() && (this.getFlatMembershipNotificationsOnSaveOrUpdate() || this.getFlatPrivilegeNotificationsOnSaveOrUpdate())) {
619       Set<ChangeLogEntry> changeLogEntryBatch = new LinkedHashSet<ChangeLogEntry>();
620       int batchSize = GrouperConfig.getHibernatePropertyInt("hibernate.jdbc.batch_size", 20);
621       if (batchSize <= 0) {
622         batchSize = 1;
623       }
624 
625       PITField pitField = null;
626       String ownerId = null;
627       String ownerName = null;
628       String privilegeName = null;
629       String privilegeType = null;
630       String ownerType = null;
631       boolean isMembership = false;
632       Set<PITMembership> pitMemberships = GrouperDAOFactory.getFactory().getPITMembershipView().findPITMembershipsJoinedWithOldPITGroupSet(this);
633       
634       if (pitMemberships.size() > 0) {
635         pitField = GrouperDAOFactory.getFactory().getPITField().findById(this.getFieldId(), true);
636         if (this.getOwnerGroupId() != null) {
637           PITGroup pitGroup = GrouperDAOFactory.getFactory().getPITGroup().findById(this.getOwnerId(), true);
638           ownerId = pitGroup.getSourceId();
639           ownerName = pitGroup.getName();
640           if (pitField.getType().equals(FieldType.LIST.getType())) {
641             isMembership = true;
642           } else {
643             privilegeType = FieldType.ACCESS.getType();
644             privilegeName = AccessPrivilege.listToPriv(pitField.getName()).getName();
645             ownerType = Membership.OWNER_TYPE_GROUP;
646           }
647         } else if (this.getOwnerStemId() != null) {
648           PITStem pitStem = GrouperDAOFactory.getFactory().getPITStem().findById(this.getOwnerId(), true);
649           ownerId = pitStem.getSourceId();
650           ownerName = pitStem.getName();
651           privilegeType = FieldType.NAMING.getType();
652           privilegeName = NamingPrivilege.listToPriv(pitField.getName()).getName();
653           ownerType = Membership.OWNER_TYPE_STEM;
654         } else if (this.getOwnerAttrDefId() != null) {
655           PITAttributeDef pitAttributeDef = GrouperDAOFactory.getFactory().getPITAttributeDef().findById(this.getOwnerId(), true);
656           ownerId = pitAttributeDef.getSourceId();
657           ownerName = pitAttributeDef.getName();
658           privilegeType = FieldType.ATTRIBUTE_DEF.getType();
659           privilegeName = AttributeDefPrivilege.listToPriv(pitField.getName()).getName();
660           ownerType = Membership.OWNER_TYPE_ATTRIBUTE_DEF;
661         } else {
662           throw new RuntimeException("Unable to determine owner of PIT Group Set: " + this.getId());
663         }
664       }
665       
666       Iterator<PITMembership> iter = pitMemberships.iterator();
667       while (iter.hasNext()) {
668         PITMembership pitMembership = iter.next();
669         
670         ChangeLogEntry changeLogEntry = null;
671         
672         if (isMembership && this.getFlatMembershipNotificationsOnSaveOrUpdate()) {
673           String subjectName = null;
674           
675           // get the subject name if the subject is a group
676           if (pitMembership.getMember().getSubjectTypeId().equals("group")) {
677             PITGroup memberGroup = GrouperDAOFactory.getFactory().getPITGroup().findBySourceIdActive(pitMembership.getMember().getSubjectId(), false);
678             
679             if (memberGroup != null) {
680               subjectName = memberGroup.getName();
681             }
682           }  
683           
684           changeLogEntry = new ChangeLogEntry(false, ChangeLogTypeBuiltin.MEMBERSHIP_DELETE,
685               ChangeLogLabels.MEMBERSHIP_DELETE.id.name(), pitMembership.getSourceId(),
686               ChangeLogLabels.MEMBERSHIP_DELETE.fieldName.name(), pitField.getName(),
687               ChangeLogLabels.MEMBERSHIP_DELETE.fieldId.name(), pitField.getSourceId(),
688               ChangeLogLabels.MEMBERSHIP_DELETE.memberId.name(), pitMembership.getMember().getSourceId(),
689               ChangeLogLabels.MEMBERSHIP_DELETE.subjectId.name(), pitMembership.getMember().getSubjectId(),
690               ChangeLogLabels.MEMBERSHIP_DELETE.sourceId.name(), pitMembership.getMember().getSubjectSourceId(),
691               ChangeLogLabels.MEMBERSHIP_DELETE.groupId.name(), ownerId,
692               ChangeLogLabels.MEMBERSHIP_DELETE.membershipType.name(), "flattened",
693               ChangeLogLabels.MEMBERSHIP_DELETE.subjectName.name(), subjectName,
694               ChangeLogLabels.MEMBERSHIP_DELETE.groupName.name(), ownerName,
695               ChangeLogLabels.MEMBERSHIP_DELETE.subjectIdentifier0.name(), pitMembership.getMember().getSubjectIdentifier0());
696         } else if (!isMembership && this.getFlatPrivilegeNotificationsOnSaveOrUpdate()) {
697           changeLogEntry = new ChangeLogEntry(false, ChangeLogTypeBuiltin.PRIVILEGE_DELETE,
698               ChangeLogLabels.PRIVILEGE_DELETE.id.name(), pitMembership.getSourceId(),
699               ChangeLogLabels.PRIVILEGE_DELETE.privilegeName.name(), privilegeName,
700               ChangeLogLabels.PRIVILEGE_DELETE.fieldId.name(), pitField.getSourceId(),
701               ChangeLogLabels.PRIVILEGE_DELETE.memberId.name(), pitMembership.getMember().getSourceId(),
702               ChangeLogLabels.PRIVILEGE_DELETE.subjectId.name(), pitMembership.getMember().getSubjectId(),
703               ChangeLogLabels.PRIVILEGE_DELETE.sourceId.name(), pitMembership.getMember().getSubjectSourceId(),
704               ChangeLogLabels.PRIVILEGE_DELETE.privilegeType.name(), privilegeType,
705               ChangeLogLabels.PRIVILEGE_DELETE.ownerType.name(), ownerType,
706               ChangeLogLabels.PRIVILEGE_DELETE.ownerId.name(), ownerId,
707               ChangeLogLabels.PRIVILEGE_DELETE.membershipType.name(), "flattened",
708               ChangeLogLabels.PRIVILEGE_DELETE.ownerName.name(), ownerName);
709         }
710 
711         if (changeLogEntry != null) {
712           changeLogEntry.setContextId(this.getContextId());
713           changeLogEntry.setCreatedOnDb(this.getEndTimeDb());
714           changeLogEntryBatch.add(changeLogEntry);
715           
716           if (changeLogEntryBatch.size() % batchSize == 0) {
717             GrouperDAOFactory.getFactory().getChangeLogEntry().saveBatch(changeLogEntryBatch, false);
718             changeLogEntryBatch.clear();
719           }
720         }
721       }
722       
723       // make sure all changes get made      
724       if (changeLogEntryBatch.size() > 0) {
725         GrouperDAOFactory.getFactory().getChangeLogEntry().saveBatch(changeLogEntryBatch, false);
726         changeLogEntryBatch.clear();
727       }
728     }

769    * the children of a parent PIT group set.
770    * @param members
771    * @return the map
772    */
773   private Map<String, Set<PITGroupSet>> getParentToChildrenMap(Set<PITGroupSet> members) {
774     Map<String, Set<PITGroupSet>> parentToChildrenMap = new HashMap<String, Set<PITGroupSet>>();
775 
776     Iterator<PITGroupSet> iterator = members.iterator();
777     while (iterator.hasNext()) {
778       PITGroupSet pitGroupSet = iterator.next();
779       String parentId = pitGroupSet.getParentId();
780 
781       if (parentId != null && !parentId.equals("")) {
782         Set<PITGroupSet> children = parentToChildrenMap.get(parentId);
783         if (children == null) {
784           children = new LinkedHashSet<PITGroupSet>();
785         }
786 
787         children.add(pitGroupSet);
788         parentToChildrenMap.put(parentId, children);
789       }
790     }
791 
792     return parentToChildrenMap;
793   }

817   /**
818    * @param pitGroupSet
819    * @param parentPITGroupSet
820    * @param parentToChildrenMap
821    * @return set
822    */
823   private Set<PITGroupSet> addHasMembersRecursively(PITGroupSet pitGroupSet, PITGroupSet parentPITGroupSet, Map<String, Set<PITGroupSet>> parentToChildrenMap, PITField pitMemberField) {
824 
825     PITGroupSet newPITGroupSet = new PITGroupSet();
826     newPITGroupSet.setId(GrouperUuid.getUuid());
827     newPITGroupSet.setFieldId(parentPITGroupSet.getFieldId());
828     newPITGroupSet.setMemberFieldId(pitMemberField.getId());
829     newPITGroupSet.setOwnerAttrDefId(parentPITGroupSet.getOwnerAttrDefId());
830     newPITGroupSet.setOwnerGroupId(parentPITGroupSet.getOwnerGroupId());
831     newPITGroupSet.setOwnerStemId(parentPITGroupSet.getOwnerStemId());
832     newPITGroupSet.setMemberGroupId(pitGroupSet.getMemberGroupId());
833     newPITGroupSet.setDepth(parentPITGroupSet.getDepth() + 1);
834     newPITGroupSet.setParentId(parentPITGroupSet.getId());
835     newPITGroupSet.setActiveDb("T");
836     newPITGroupSet.setStartTimeDb(this.getStartTimeDb());
837     newPITGroupSet.setContextId(this.getContextId());
838 
839     String ownerSourceId = null;
840     if (newPITGroupSet.getOwnerGroupId() != null) {
841       ownerSourceId = GrouperDAOFactory.getFactory().getPITGroup().findById(newPITGroupSet.getOwnerGroupId(), true).getSourceId();
842     } else if (newPITGroupSet.getOwnerStemId() != null) {
843       ownerSourceId = GrouperDAOFactory.getFactory().getPITStem().findById(newPITGroupSet.getOwnerStemId(), true).getSourceId();
844     } else if (newPITGroupSet.getOwnerAttrDefId() != null) {
845       ownerSourceId = GrouperDAOFactory.getFactory().getPITAttributeDef().findById(newPITGroupSet.getOwnerAttrDefId(), true).getSourceId();
846     } else {
847       throw new RuntimeException("Unexpected");
848     }
849     
850     // now get the existing group set so we have the source id.
851     PITGroup pitMemberGroup = GrouperDAOFactory.getFactory().getPITGroup().findById(newPITGroupSet.getMemberId(), true);
852     PITField pitField = GrouperDAOFactory.getFactory().getPITField().findById(newPITGroupSet.getFieldId(), true);
853     GroupSet gs = GrouperDAOFactory.getFactory().getGroupSet().findByOwnerMemberFieldParentAndType(
854         ownerSourceId, pitMemberGroup.getSourceId(), pitField.getSourceId(), 
855         parentPITGroupSet.getSourceId(), "effective", false);
856     
857     if (gs == null) {
858       // either the group was deleted or it was never added (because it formed a circular path for instance)
859       return new LinkedHashSet<PITGroupSet>();
860     }
861     
862     if (!GrouperUtil.isEmpty(this.getContextId()) && !GrouperUtil.isEmpty(gs.getContextId()) &&
863         !this.getContextId().equals(gs.getContextId())) {
864       // this group set must have been added, deleted and then readded.  Don't add to point in time
865       return new LinkedHashSet<PITGroupSet>();
866     }
867     
868     newPITGroupSet.setSourceId(gs.getId());
869 
870     Set<PITGroupSet> newPITGroupSets = new LinkedHashSet<PITGroupSet>();
871     newPITGroupSets.add(newPITGroupSet);
872 
873     Set<PITGroupSet> children = parentToChildrenMap.get(pitGroupSet.getId());
874     if (children != null) {
875       Iterator<PITGroupSet> it = children.iterator();
876       while (it.hasNext()) {
877         PITGroupSet nextPITGroupSet = it.next();
878         Set<PITGroupSet> newAdditions = addHasMembersRecursively(nextPITGroupSet, newPITGroupSet, parentToChildrenMap, pitMemberField);
879         newPITGroupSets.addAll(newAdditions);
880       }
881     }
882 
883     return newPITGroupSets;
884   }

892   private Set<PITGroupSet> addHasMembersToWhereGroupIsMember(String memberGroupId,
893       Set<PITGroupSet> pitGroupSetIsMember, Set<PITGroupSet> pitGroupSetHasMembers, PITField pitMemberField) {
894     
895     Set<PITGroupSet> pitGroupSets = new LinkedHashSet<PITGroupSet>();
896 
897     Iterator<PITGroupSet> isMembersIter = pitGroupSetIsMember.iterator();
898     Map<String, Set<PITGroupSet>> parentToChildrenMap = getParentToChildrenMap(pitGroupSetHasMembers);
899 
900     // lets get all the hasMembers with a depth of 1 before the while loop
901     Set<PITGroupSet> hasMembersOneDepth = new LinkedHashSet<PITGroupSet>();
902     Iterator<PITGroupSet> hasMembersIter = pitGroupSetHasMembers.iterator();
903     while (hasMembersIter.hasNext()) {
904       PITGroupSet pitGroupSet = hasMembersIter.next();
905       if (pitGroupSet.getDepth() == 1) {
906         hasMembersOneDepth.add(pitGroupSet);
907       }
908     }
909

911       PITGroupSet isPITGS = isMembersIter.next();
912

914       String ownerStemId = isPITGS.getOwnerStemId();
915       String ownerAttrDefId = isPITGS.getOwnerAttrDefId();
916       String fieldId = isPITGS.getFieldId();
917       String id = isPITGS.getId();
918       int depth = isPITGS.getDepth();
919 
920       PITGroupSet pitGroupSet = new PITGroupSet();
921       pitGroupSet.setId(GrouperUuid.getUuid());
922       pitGroupSet.setDepth(depth + 1);
923       pitGroupSet.setParentId(id);
924       pitGroupSet.setFieldId(fieldId);
925       pitGroupSet.setMemberFieldId(pitMemberField.getId());
926       pitGroupSet.setMemberGroupId(memberGroupId);
927       pitGroupSet.setOwnerGroupId(ownerGroupId);
928       pitGroupSet.setOwnerAttrDefId(ownerAttrDefId);
929       pitGroupSet.setOwnerStemId(ownerStemId);
930       pitGroupSet.setActiveDb("T");
931       pitGroupSet.setStartTimeDb(this.getStartTimeDb());
932       pitGroupSet.setContextId(this.getContextId());
933 
934       String ownerSourceId = null;
935       if (pitGroupSet.getOwnerGroupId() != null) {
936         ownerSourceId = GrouperDAOFactory.getFactory().getPITGroup().findById(pitGroupSet.getOwnerGroupId(), true).getSourceId();
937       } else if (pitGroupSet.getOwnerStemId() != null) {
938         ownerSourceId = GrouperDAOFactory.getFactory().getPITStem().findById(pitGroupSet.getOwnerStemId(), true).getSourceId();
939       } else if (pitGroupSet.getOwnerAttrDefId() != null) {
940         ownerSourceId = GrouperDAOFactory.getFactory().getPITAttributeDef().findById(pitGroupSet.getOwnerAttrDefId(), true).getSourceId();
941       } else {
942         throw new RuntimeException("Unexpected");
943       }
944       
945       // now get the existing group set so we have the source id.
946       PITGroup pitMemberGroup = GrouperDAOFactory.getFactory().getPITGroup().findById(pitGroupSet.getMemberId(), true);
947       PITField pitField = GrouperDAOFactory.getFactory().getPITField().findById(pitGroupSet.getFieldId(), true);
948       GroupSet gs = GrouperDAOFactory.getFactory().getGroupSet().findByOwnerMemberFieldParentAndType(
949           ownerSourceId, pitMemberGroup.getSourceId(), pitField.getSourceId(),
950           isPITGS.getSourceId(), "effective", false);
951       
952       if (gs == null) {
953         // either the group was deleted or it was never added (because it formed a circular path for instance)
954         continue;
955       }
956       
957       if (!GrouperUtil.isEmpty(this.getContextId()) && !GrouperUtil.isEmpty(gs.getContextId()) &&
958           !this.getContextId().equals(gs.getContextId())) {
959         // this group set must have been added, deleted and then readded.  Don't add to point in time
960         continue;
961       }
962 
963       pitGroupSet.setSourceId(gs.getId());
964       pitGroupSets.add(pitGroupSet);
965 
966       Iterator<PITGroupSet> itHM = hasMembersOneDepth.iterator();
967       while (itHM.hasNext()) {
968         PITGroupSet hasPITGS = itHM.next();
969         Set<PITGroupSet> newAdditions = addHasMembersRecursively(hasPITGS, pitGroupSet, parentToChildrenMap, pitMemberField);
970         pitGroupSets.addAll(newAdditions);
971       }
972     }
973 
974     return pitGroupSets;
975   }

1020       }
