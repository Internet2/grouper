38    /**
39     * note, call this in your grouper session, this runs in current thread
40     * @param callablesWithProblems
41     */
42    public static void tryCallablesWithProblems(Collection<GrouperCallable> callablesWithProblems) {
43      
44      RuntimeException problem = null;
45  
46      int callablesWithProblemsCount = 0;
47      
48      for (GrouperCallable grouperCallable : GrouperUtil.nonNull(callablesWithProblems)) {
49        
50        try {
51          grouperCallable.callLogic();
52        } catch (RuntimeException re) {
53          
54          callablesWithProblemsCount++;
55          
56          if (problem == null) {
57            problem = re;
58          } else {
59            LOG.error("Problem with callable: " + grouperCallable.getLogLabel());
60          }
61        }
62        
63      }
64      
65      if (problem != null) {
66        GrouperUtil.injectInException(problem, callablesWithProblemsCount 
67            + " callables out of problem count " + callablesWithProblems.size() + " had problems running outside of threads!");
68        throw problem;
69      }
70    }
71    
72    /**
73     * convert exception
74     * @param throwable
75     */
76    public static void throwRuntimeException(Throwable throwable) {
77  
78      //this isnt good, exception
79      if (throwable instanceof ExecutionException) {
80        //unwrap it
81        ExecutionException executionException = (ExecutionException)throwable;
82        if (executionException.getCause() != null) {
83          //the underlying exception is here... might be runtime
84          throwable = executionException.getCause();
85        }
86      }
87      if (throwable instanceof RuntimeException) {
88        throw (RuntimeException)throwable;
89      }
90      throw new RuntimeException(throwable);
91  
92    }

102   private Subject grouperSessionSubject;

104   /** loglabel */
105   private String logLabel;

107   /**
108    * 
109    */
110   static int numberOfThreads = 0;
111   
112   /**
113    * @see java.util.concurrent.Callable#call()
114    */
115   public final T call() throws Exception {
116     
117     long subStartNanos = -1;
118     
119     //store the old
120     GrouperThreadLocalState oldGrouperThreadLocalState = new GrouperThreadLocalState();
121     oldGrouperThreadLocalState.storeCurrentThreadLocals();
122     
123     try {
124       if (LOG.isDebugEnabled()) {
125         subStartNanos = System.nanoTime();
126         synchronized (GrouperCallable.class) {
127           numberOfThreads++;
128         }
129       }
130       //propagate thread locals
131       this.grouperThreadLocalState.assignCurrentThreadLocals();
132       return this.callLogicWithSessionIfExists();
133     } finally {
134       if (LOG.isDebugEnabled()) {
135         synchronized (GrouperCallable.class) {
136           long nanos = System.nanoTime() - subStartNanos;
137           long millis = nanos / 1000000;
138           LOG.debug("Threads: " + numberOfThreads + ", " + this.logLabel + ", time in millis: " + millis);
139           numberOfThreads--;
140         }
141       }
142       //assign the old thread local state, shouldnt really matter since thread if going back in pool
143       oldGrouperThreadLocalState.assignCurrentThreadLocals();
144     }
145     
146   }
147 
148   /**
149    * Computes a result
150    *
151    * @return computed result
152    */
153   public final T callLogicWithSessionIfExists() {
154     
155     GrouperSession grouperSession = GrouperSession.staticGrouperSession(false);
156 
157     //if we already have the same session open, use it
158     boolean sameSubjectInSession = grouperSession != null && SubjectHelper.eq(grouperSession.getSubject(), this.grouperSessionSubject);
159     if (this.grouperSessionSubject != null && !sameSubjectInSession) {
160       grouperSession = GrouperSession.start(this.grouperSessionSubject, false);
161       return (T)GrouperSession.callbackGrouperSession(grouperSession, new GrouperSessionHandler() {
162         
163         public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
164           return GrouperCallable.this.callLogic();
165         }
166       });
167       
168     }
169     return this.callLogic();
170   }

172   /**

177   public abstract T callLogic();

179   /**
180    * @param theGrouperSession
181    * @param theLogLabel
182    */
183   public GrouperCallable(String theLogLabel, GrouperSession theGrouperSession) {
184     this.logLabel = theLogLabel;
185     this.grouperSessionSubject = theGrouperSession == null ? null : theGrouperSession.getSubject();
186     //keep track of important thread locals
187     this.grouperThreadLocalState.storeCurrentThreadLocals();
188   }

194   public GrouperCallable(String theLogLabel) {

196     GrouperSession grouperSession = GrouperSession.staticGrouperSession(false);
197     this.grouperSessionSubject = grouperSession == null ? null : grouperSession.getSubject();
198     //keep track of important thread locals
199     this.grouperThreadLocalState.storeCurrentThreadLocals();
