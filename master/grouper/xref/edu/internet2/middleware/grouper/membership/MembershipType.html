20  package edu.internet2.middleware.grouper.membership;
21  
22  import edu.internet2.middleware.grouper.util.GrouperUtil;
23  
24  /**
25   * type of membership
26   */
27  public enum MembershipType {
28  
29    /**
30     * An immediate member is directly assigned to a group.
31     * A composite group has no immediate members.  Note that a 
32     * member can have 0 to 1 immediate memberships
33     * to a single group, and 0 to many effective memberships to a group.
34     * A group can have potentially unlimited effective 
35     * memberships
36     */
37    IMMEDIATE("immediate") {
38  
39      /**
40       * query clause for this membership type, e.g. = 'immediate'
41       */
42      public String queryClause() {
43        return " = 'immediate' ";
44      }
45    },
46    
47    /**
48     * An effective member has an indirect membership to a group
49     * (e.g. in a group within a group).  All subjects in a
50     * composite group are effective members (since the composite
51     * group has two groups and a set operator and no other immediate
52     * members).  Note that a member can have 0 to 1 immediate memberships
53     * to a single group, and 0 to many effective memberships to a group.
54     * 'group within a group' can be nested to any level so long as it does 
55     * not become circular.  A group can have potentially unlimited effective 
56     * memberships
57     */
58    EFFECTIVE("effective") {
59  
60      /**
61       * query clause for this membership type, e.g. = 'immediate'
62       */
63      public String queryClause() {
64        return " = 'effective' ";
65      }
66    },
67    
68    /**
69     * composite memberships are due to union, intersection, minus
70     * A member of a group (aka composite member) has either or both of
71     * an immediate (direct) membership, or an effective (indirect) membership 
72     */
73    COMPOSITE("composite") {
74  
75      /**
76       * query clause for this membership type, e.g. = 'immediate'
77       */
78      public String queryClause() {
79        return " = 'composite' ";
80      }
81    },
82    
83    /**
84     * everything except immediate
85     * An immediate member is directly assigned to a group.
86     * A composite group has no immediate members.  Note that a 
87     * member can have 0 to 1 immediate memberships
88     * to a single group, and 0 to many effective memberships to a group.
89     * A group can have potentially unlimited effective 
90     * memberships
91     */
92    NONIMMEDIATE("nonImmediate") {
93  
94      /**
95       * query clause for this membership type, e.g. = 'immediate'
96       */
97      public String queryClause() {
98        return " != 'immediate' ";
99      }
100   };
101   
102   /**
103    * query clause for this membership type, e.g. = 'immediate'
104    * @return the query clause
105    */
106   public abstract String queryClause();
107   
108   /**
109    * construct with type
110    * @param theType
111    */
112   private MembershipType(String theType) {
113     this.typeString = theType; 
114   }
115   
116   /** type */
117   private String typeString;
118 
119   /**
120    * return the type string
121    * @return the type
122    */
123   public String getTypeString() {
124     return this.typeString;
125   }
126   
127   /**
128    * do a case-insensitive matching
129    * 
130    * @param string
131    * @param exceptionOnNull will not allow null or blank entries
132    * @return the enum or null or exception if not found
133    */
134   public static MembershipType valueOfIgnoreCase(String string, boolean exceptionOnNull) {
135     
136     return GrouperUtil.enumValueOfIgnoreCase(MembershipType.class, 
137         string, exceptionOnNull);
138   }
