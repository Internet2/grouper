29  /**
30   * ddl versions and stuff for grouper.  All ddl classes must have a currentVersion method that
31   * returns the current version
32   */
33  public enum SubjectDdl implements DdlVersionable {
34  
35    /** first version of grouper, make sure the ddl table is there */
36    V1 {
37      /**
38       * add the subject table
39       * @see SubjectDdl#updateVersionFromPrevious(org.apache.ddlutils.model.Database, 
40        DdlVersionBean)
41       */
42      @Override
43      public void updateVersionFromPrevious(Database database, DdlVersionBean ddlVersionBean) {
44  
45        {
46          Table subjectTable = GrouperDdlUtils.ddlutilsFindOrCreateTable(database,
47              "subject");
48    
49          GrouperDdlUtils.ddlutilsFindOrCreateColumn(subjectTable, "subjectId", 
50              Types.VARCHAR, "255", true, true);
51    
52          GrouperDdlUtils.ddlutilsFindOrCreateColumn(subjectTable, "subjectTypeId", 
53              Types.VARCHAR, "32", false, true);
54    
55          GrouperDdlUtils.ddlutilsFindOrCreateColumn(subjectTable, "name", 
56              Types.VARCHAR, "255", false, false);
57    
58        }
59        {
60          Table subjectTable = GrouperDdlUtils.ddlutilsFindOrCreateTable(database,
61              "subjectattribute");
62    
63          GrouperDdlUtils.ddlutilsFindOrCreateColumn(subjectTable, "subjectId", 
64              Types.VARCHAR, "255", true, true);
65    
66          GrouperDdlUtils.ddlutilsFindOrCreateColumn(subjectTable, "name", 
67              Types.VARCHAR, "255", true, true);
68    
69          GrouperDdlUtils.ddlutilsFindOrCreateColumn(subjectTable, "value", 
70              Types.VARCHAR, "255", true, true);
71    
72          GrouperDdlUtils.ddlutilsFindOrCreateColumn(subjectTable, "searchValue", 
73              Types.VARCHAR, "255", false, false);
74    
75          GrouperDdlUtils.ddlutilsFindOrCreateIndex(database, subjectTable.getName(), 
76              "searchattribute_value_idx", false, "value");
77  
78          GrouperDdlUtils.ddlutilsFindOrCreateIndex(database, subjectTable.getName(), 
79              "searchattribute_id_name_idx", true, "subjectId", "name");
80          
81          GrouperDdlUtils.ddlutilsFindOrCreateIndex(database, subjectTable.getName(), 
82              "searchattribute_name_idx", false, "name");
83  
84        }
85        
86      }
87    };
88  
89    /**
90     * @see edu.internet2.middleware.grouper.ddl.DdlVersionable#getVersion()
91     */
92    public int getVersion() {
93      return GrouperDdlUtils.versionIntFromEnum(this);
94    }
95  
96    /**
97     * cache this
98     */
99    private static int currentVersion = -1;
100   
101   /**
102    * keep the current version here, increment as things change
103    * @return the current version
104    */
105   public static int currentVersion() {
106     if (currentVersion == -1) {
107       int max = -1;
108       for (SubjectDdl grouperDdl : SubjectDdl.values()) {
109         String number = grouperDdl.name().substring(1);
110         int theInt = Integer.parseInt(number);
111         max = Math.max(max, theInt);
112       }
113       currentVersion = max;
114     }
115     return currentVersion;
116   }

146   /**
147    * add all foreign keys
148    * @param ddlVersionBean 
149    */
150   public void addAllForeignKeysViewsEtc(DdlVersionBean ddlVersionBean) {
151     Database database = ddlVersionBean.getDatabase();
152 
153     
154     GrouperDdlUtils.ddlutilsFindOrCreateForeignKey(database, "subjectattribute", 
155         "fk_subjectattr_subjectid", "subject", "subjectId", "subjectId");
156 
157     GrouperDdlUtils.ddlutilsTableComment(ddlVersionBean, 
158         "subject", "sample subject table for grouper unit tests");
159 
160     GrouperDdlUtils.ddlutilsColumnComment(ddlVersionBean, "subject", "subjectId", 
161         "subject id of row");
162 
163     GrouperDdlUtils.ddlutilsColumnComment(ddlVersionBean, "subject", "subjectTypeId", 
164         "subject type e.g. person");
165 
166     GrouperDdlUtils.ddlutilsColumnComment(ddlVersionBean, "subject", "name", 
167         "name of this subject");
168 
169     
170     GrouperDdlUtils.ddlutilsTableComment(ddlVersionBean, 
171         "subjectattribute", "attribute data for each subject");
172 
173     GrouperDdlUtils.ddlutilsColumnComment(ddlVersionBean, 
174         "subjectattribute",  "subjectId", 
175         "subject id of row");
176 
177     GrouperDdlUtils.ddlutilsColumnComment(ddlVersionBean, 
178         "subjectattribute",  "name", 
179         "name of attribute");
180 
181     GrouperDdlUtils.ddlutilsColumnComment(ddlVersionBean, 
182         "subjectattribute",  "value", 
183         "value of attribute");
184 
185     GrouperDdlUtils.ddlutilsColumnComment(ddlVersionBean, 
186         "subjectattribute",  "searchValue", 
187         "search value (e.g. all lower)");
188 
189   }
