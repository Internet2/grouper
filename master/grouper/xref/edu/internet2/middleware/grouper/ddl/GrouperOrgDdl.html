269   /**
270    * create the assignments
271    * @param ddlVersionBean
272    */
273   public static void generateOrgAssignment(DdlVersionBean ddlVersionBean) {
274     
275     List<String> orgIds = GrouperUtil.toList("1", "2", "3", "11", "12", "13", "21", "22", "23", "31", "32", "33",
276         "111", "112", "113", "121", "122", "123", "131", "132", "133", "211", "212", "213", "221", "222", "223", 
277         "231", "232", "233", "311", "312", "313", "321", "322", "323", "331", "332", "333");
278     
279     //make this deterministic
280     Random random = new Random(1);
281 
282     int id = 0;
283     
284     for (String orgId : orgIds) {
285 
286       //pick a random number of people in here, from 0 to 9
287       int memberCount = (int)(random.nextDouble() * 10);
288       
289       for (int i=0;i<memberCount;i++) {
290         int subjectIndex = (int) (random.nextDouble() * subjectIds.size());
291         String subjectId = subjectIds.get(subjectIndex);
292         ddlVersionBean.appendAdditionalScriptUnique("insert " +
293         		"into grouperorgs_poc_org_assign (id, org_id, subject_id) " +
294         		"values ('" + id++ + "', '" + orgId + "', '" + subjectId + "');\n");        
295       }
296       
297     }

331   /**
332    * keep the current version here, increment as things change
333    * @return the current version
334    */
335   public static int currentVersion() {
336     if (currentVersion == -1) {
337       int max = -1;
338       for (GrouperOrgDdl grouperDdl : GrouperOrgDdl.values()) {
339         String number = grouperDdl.name().substring(1);
340         int theInt = Integer.parseInt(number);
341         max = Math.max(max, theInt);
342       }
343       currentVersion = max;
344     }
345     return currentVersion;
346   }

386              + "from grouperorgs_hierarchical gh, grouperorgs_hierarchical gh_member, grouperorgs_poc_orgs gpo "
