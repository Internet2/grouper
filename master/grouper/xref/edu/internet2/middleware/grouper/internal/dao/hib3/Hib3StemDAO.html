249   /**
250    * @param stemId
251    * @param parentStemId
252    */
253   public void createStemSetsForStem(String stemId, String parentStemId) {
254     
255     List<StemSet> allStemSets = new LinkedList<StemSet>();
256 
257     // batch to help performance
258     int batchSize = GrouperConfig.getHibernatePropertyInt("hibernate.jdbc.batch_size", 20);
259     if (batchSize <= 0) {
260       batchSize = 1;
261     }
262     
263     // create self one first
264     StemSet selfStemSet = new StemSet();
265     selfStemSet.setId(GrouperUuid.getUuid());
266     selfStemSet.setDepth(0);
267     selfStemSet.setIfHasStemId(stemId);
268     selfStemSet.setThenHasStemId(stemId);
269     selfStemSet.setType(StemHierarchyType.self);
270     selfStemSet.setParentStemSetId(selfStemSet.getId());
271     
272     allStemSets.add(selfStemSet);
273    
274     // now find ancestor stems
275     if (parentStemId != null) {
276       Set<StemSet> stemSets = GrouperDAOFactory.getFactory().getStemSet().findByIfHasStemId(parentStemId);
277       allStemSets.addAll(createNonSelfStemSetsForStem(new LinkedList<StemSet>(stemSets), stemId, selfStemSet));
278     }
279     
280     for (int i = 0; i < GrouperUtil.batchNumberOfBatches(allStemSets, batchSize); i++) {
281       List<StemSet> currentBatch = GrouperUtil.batchList(allStemSets, batchSize, i);
282       GrouperDAOFactory.getFactory().getStemSet().saveBatch(currentBatch);
283     }
284   }

286   /**
287    * @param ifHasStemSetsOfParentStem
288    * @param ifHasStemId
289    * @param firstParentStemSet
290    * @return new child stemSets
291    */
292   private List<StemSet> createNonSelfStemSetsForStem(List<StemSet> ifHasStemSetsOfParentStem, String ifHasStemId, StemSet firstParentStemSet) {
293     
294     String nextParentStemSetId = firstParentStemSet.getId();
295     
296     // sort by depth
297     Collections.sort(ifHasStemSetsOfParentStem, new Comparator<StemSet>() {
298 
299       public int compare(StemSet o1, StemSet o2) {
300         return ((Integer)o1.getDepth()).compareTo(o2.getDepth());
301       }
302     });
303     
304     
305     List<StemSet> newChildren = new LinkedList<StemSet>();
306     
307     for (StemSet currIfHasStemSet : ifHasStemSetsOfParentStem) {
308       StemSet childStemSet = new StemSet();
309       childStemSet.setId(GrouperUuid.getUuid());
310       childStemSet.setDepth(currIfHasStemSet.getDepth() + 1 + firstParentStemSet.getDepth());
311       childStemSet.setThenHasStemId(currIfHasStemSet.getThenHasStemId());
312       childStemSet.setIfHasStemId(ifHasStemId);
313       childStemSet.setType(childStemSet.getDepth() == 1 ? StemHierarchyType.immediate : StemHierarchyType.effective);
314       childStemSet.setParentStemSetId(nextParentStemSetId);
315       
316       newChildren.add(childStemSet);
317       
318       nextParentStemSetId = childStemSet.getId();
319     }
320     
321     return newChildren;
322   }

324   /**
325    * @see edu.internet2.middleware.grouper.internal.dao.StemDAO#moveStemSets(java.util.List, java.util.List, java.lang.String, int)
326    */
327   public void moveStemSets(List<StemSet> ifHasStemSetsOfParentStem, List<StemSet> oldStemSets, String currentStemId, int depthOfFirstParent) {
328 
329     // we want to keep one of the oldStemSets -- with the lowest depth .. remove from the collection
330     // also sort desc so they can be deleted later in order
331     Collections.sort(oldStemSets, new Comparator<StemSet>() {
332 
333       public int compare(StemSet o1, StemSet o2) {
334         return ((Integer)o2.getDepth()).compareTo(o1.getDepth());
335       }
336     });
337     
338     StemSet firstParentStemSet = oldStemSets.remove(oldStemSets.size() - 1);
339     
340     List<StemSet> allStemSetsForCurrentNode = createNonSelfStemSetsForStem(new LinkedList<StemSet>(ifHasStemSetsOfParentStem), currentStemId, firstParentStemSet);
341     
342     // batch to help performance
343     int batchSize = GrouperConfig.getHibernatePropertyInt("hibernate.jdbc.batch_size", 20);
344     if (batchSize <= 0) {
345       batchSize = 1;
346     }
347     
348     // add new stem sets
349     for (int i = 0; i < GrouperUtil.batchNumberOfBatches(allStemSetsForCurrentNode, batchSize); i++) {
350       List<StemSet> currentBatch = GrouperUtil.batchList(allStemSetsForCurrentNode, batchSize, i);
351       GrouperDAOFactory.getFactory().getStemSet().saveBatch(currentBatch);
352     }
353     
354     // take care of children, separately for each ifHasStemId
355     Set<StemSet> allOldStemSetChildren = GrouperDAOFactory.getFactory().getStemSet().findByIfHasStemOfStemChildrenAndMinDepth(currentStemId, depthOfFirstParent + 1, new QueryOptions().sortAsc("ss.ifHasStemId"));
356 
357     if (allOldStemSetChildren.size() > 0) {
358       List<StemSet> oldStemSetChildrenSameIfId = new LinkedList<StemSet>();
359       String lastIfId = null;
360       for (StemSet oldStemSetChild : allOldStemSetChildren) {
361         if (lastIfId != null && !oldStemSetChild.getIfHasStemId().equals(lastIfId)) {
362           moveStemSets(ifHasStemSetsOfParentStem, oldStemSetChildrenSameIfId, lastIfId, depthOfFirstParent + 1);
363 
364           oldStemSetChildrenSameIfId.clear();
365         }
366         
367         lastIfId = oldStemSetChild.getIfHasStemId();
368         oldStemSetChildrenSameIfId.add(oldStemSetChild);
369       }
370       
371       if (oldStemSetChildrenSameIfId.size() > 0) {
372         moveStemSets(ifHasStemSetsOfParentStem, oldStemSetChildrenSameIfId, lastIfId, depthOfFirstParent + 1);
373       }
374     }
375     
376     // delete old stem sets
377     for (StemSet stemSetToDelete : oldStemSets) {
378       stemSetToDelete.delete();
379     }
380   }

383    * @param group
384    */
385   private void createGroupSetsForGroup(Group group) {
386     
387     // add group sets
388     Set<Field> fields = FieldFinder.findAll();
389     for (Field field : fields) {
390       if (field.getType().equals(FieldType.ACCESS) || Group.getDefaultList().getUuid().equals(field.getUuid())) {
391         if (group.getTypeOfGroup() != null && group.getTypeOfGroup().supportsField(field)) {
392           GroupSet groupSet = new GroupSet();
393           groupSet.setId(GrouperUuid.getUuid());
394           groupSet.setCreatorId(GrouperSession.staticGrouperSession().getMemberUuid());
395           groupSet.setDepth(0);
396           groupSet.setMemberGroupId(group.getUuid());
397           groupSet.setOwnerGroupId(group.getUuid());
398           groupSet.setParentId(groupSet.getId());
399           groupSet.setFieldId(field.getUuid());
400           GrouperDAOFactory.getFactory().getGroupSet().save(groupSet);
401         }
402       }
403     }
404   }
