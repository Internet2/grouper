196   /**
197    * reset group set
198    * @param hibernateSession
199    */
200   public static void reset(HibernateSession hibernateSession) {
201     Session hs = hibernateSession.getSession();
202 
203     // Find the root stem
204     Stem rootStem = GrouperDAOFactory.getFactory().getStem().findByName(Stem.ROOT_INT, true);
205     
206     // get the max depth
207     int maxDepth = HibernateSession.byHqlStatic().createQuery("select max(depth) from GroupSet").uniqueResult(Integer.class);
208     
209     // delete the rows that don't have self reference by the depth desc
210     for (int i = maxDepth; i > 0; i--) {
211       hs.createQuery("delete from GroupSet where depth = :depth").setInteger("depth", i).executeUpdate();
212     }
213     
214     // set parent_id to null due to mysql bug -- http://bugs.mysql.com/bug.php?id=15746
215     // also note that parent_id is part of a unique constraint, but now that we have only self groupSets, that doesn't matter.
216     hs.createQuery("update GroupSet set parentId = null where ownerStemId not like :owner or ownerStemId is null")
217         .setString("owner", rootStem.getUuid())
218         .executeUpdate();
219     
220     hs.createQuery("delete from GroupSet where parentId = null")
221         .executeUpdate();
222   }

739   /**
740    * @see edu.internet2.middleware.grouper.internal.dao.GroupSetDAO#findDuplicateSelfGroupSets()
741    */
742   public Set<GroupSet> findDuplicateSelfGroupSets() {
743     Set<GroupSet> duplicates = new LinkedHashSet<GroupSet>();
744     
745     String sql = "select ownerId, memberId, fieldId from GroupSet where depth='0' group by ownerId, memberId, fieldId having count(*) > 1";
746     
747     Set<Object[]> results = HibernateSession.byHqlStatic()
748       .createQuery(sql)
749       .setCacheable(false)
750       .listSet(Object[].class);
751     
752     for (Object[] result : results) {
753       String ownerId = (String)result[0];
754       String memberId = (String)result[1];
755       String fieldId = (String)result[2];
756 
757       Set<GroupSet> groupSets = HibernateSession.byHqlStatic()
758         .createQuery("select gs from GroupSet gs where memberId = :memberId and fieldId = :fieldId and ownerId = :ownerId and depth='0'")
759         .setCacheable(false)
760         .setString("ownerId", ownerId)
761         .setString("memberId", memberId)
762         .setString("fieldId", fieldId)
763         .listSet(GroupSet.class);
764       
765       Set<GroupSet> noForeignKeys = new LinkedHashSet<GroupSet>();
766       for (GroupSet groupSet : groupSets) {
767         long count = HibernateSession.byHqlStatic()
768           .createQuery("select count(*) from GroupSet where parentId = :parentId")
769           .setCacheable(false)
770           .setString("parentId", groupSet.getId())
771           .uniqueResult(Long.class);
772         
773         if (count == 1) {
774           noForeignKeys.add(groupSet);
775         }
776       }
777       
778       List<GroupSet> currentDuplicates = new ArrayList<GroupSet>();
779       currentDuplicates.addAll(noForeignKeys);
780       if (groupSets.size() == currentDuplicates.size()) {
781         currentDuplicates.remove(0);
782       }
783       
784       duplicates.addAll(currentDuplicates);
785     }
786     
787     return duplicates;  
788   }
