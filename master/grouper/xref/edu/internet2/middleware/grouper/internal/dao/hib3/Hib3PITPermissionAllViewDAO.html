16  package edu.internet2.middleware.grouper.internal.dao.hib3;
17  
18  import java.sql.Timestamp;
19  import java.util.Collection;
20  import java.util.Set;
21  
22  import edu.internet2.middleware.grouper.GrouperSession;
23  import edu.internet2.middleware.grouper.cfg.GrouperConfig;
24  import edu.internet2.middleware.grouper.hibernate.ByHqlStatic;
25  import edu.internet2.middleware.grouper.hibernate.HibUtils;
26  import edu.internet2.middleware.grouper.hibernate.HibernateSession;
27  import edu.internet2.middleware.grouper.internal.dao.PITPermissionAllViewDAO;
28  import edu.internet2.middleware.grouper.permissions.PermissionEntry;
29  import edu.internet2.middleware.grouper.privs.AccessPrivilege;
30  import edu.internet2.middleware.grouper.privs.AttributeDefPrivilege;
31  import edu.internet2.middleware.grouper.util.GrouperUtil;
32  import edu.internet2.middleware.subject.Subject;
33  /**
34   * @author shilen
35   * $Id$
36   */
37  public class Hib3PITPermissionAllViewDAO extends Hib3DAO implements PITPermissionAllViewDAO {
38  
39    /**
40     *
41     */
42    private static final String KLASS = Hib3PITPermissionAllViewDAO.class.getName();
43    
44    /**
45     * @see edu.internet2.middleware.grouper.internal.dao.PITPermissionAllViewDAO#findPermissions(java.util.Collection, java.util.Collection, java.util.Collection, java.util.Collection, java.util.Collection, java.sql.Timestamp, java.sql.Timestamp)
46     */
47    public Set<PermissionEntry> findPermissions(Collection<String> attributeDefSourceIds, Collection<String> attributeDefNameSourceIds, 
48        Collection<String> roleSourceIds, Collection<String> actions, Collection<String> memberSourceIds, Timestamp pointInTimeFrom, 
49        Timestamp pointInTimeTo) {
50      
51      int memberIdsSize = GrouperUtil.length(memberSourceIds);
52      int roleIdsSize = GrouperUtil.length(roleSourceIds);
53      int actionsSize = GrouperUtil.length(actions);
54      int attributeDefIdsSize = GrouperUtil.length(attributeDefSourceIds);
55      int attributeDefNameIdsSize = GrouperUtil.length(attributeDefNameSourceIds);
56      
57      if (memberIdsSize == 0 && roleIdsSize == 0 && attributeDefIdsSize == 0 && attributeDefNameIdsSize == 0) {
58        throw new RuntimeException("Illegal query, you need to pass in members and/or attributeDefId(s) and/or roleId(s) and/or attributeDefNameIds");
59      }
60      
61      //too many bind vars
62      if (memberIdsSize + roleIdsSize + attributeDefIdsSize + attributeDefNameIdsSize + actionsSize > 100) {
63        throw new RuntimeException("Too many memberIdsSize " + memberIdsSize 
64            + " roleIdsSize " + roleIdsSize + " or attributeDefIdsSize " 
65            + attributeDefIdsSize + " or attributeDefNameIds " + attributeDefNameIdsSize + " or actionsSize " + actionsSize );
66      }
67  
68      ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
69  
70      String selectPrefix = "select distinct pea ";
71      
72      StringBuilder sqlTables = new StringBuilder(" from PITPermissionAllView pea ");
73      
74      StringBuilder sqlWhereClause = new StringBuilder("");
75      
76      GrouperSession grouperSession = GrouperSession.staticGrouperSession();
77      
78      Subject grouperSessionSubject = grouperSession.getSubject();
79      
80      grouperSession.getAttributeDefResolver().hqlFilterAttrDefsWhereClause(
81        grouperSessionSubject, byHqlStatic, 
82        sqlTables, sqlWhereClause, "pea.attributeDefSourceId", AttributeDefPrivilege.ATTR_READ_PRIVILEGES);
83      
84      boolean changedQuery = grouperSession.getAccessResolver().hqlFilterGroupsWhereClause(
85          grouperSessionSubject, byHqlStatic, 
86          sqlTables, "pea.roleSourceId", AccessPrivilege.ATTRIBUTE_READ_PRIVILEGES);
87  
88      StringBuilder sql;
89      if (sqlWhereClause.length() > 0) {
90        if (changedQuery) {
91          sql = sqlTables.append(" and ").append(sqlWhereClause);
92          
93        } else {
94          sql = sqlTables.append(" where ").append(sqlWhereClause);
95          
96        }
97      } else {
98        //where and will be removed later on
99        sql = sqlTables.append(" where ");
100     }
101     
102     if (actionsSize > 0) {
103       sql.append(" and pea.action in (");
104       sql.append(HibUtils.convertToInClause(actions, byHqlStatic));
105       sql.append(") ");
106     }
107     if (roleIdsSize > 0) {
108       sql.append(" and pea.roleSourceId in (");
109       sql.append(HibUtils.convertToInClause(roleSourceIds, byHqlStatic));
110       sql.append(") ");
111     }
112     if (attributeDefIdsSize > 0) {
113       sql.append(" and pea.attributeDefSourceId in (");
114       sql.append(HibUtils.convertToInClause(attributeDefSourceIds, byHqlStatic));
115       sql.append(") ");
116     }
117     if (attributeDefNameIdsSize > 0) {
118       sql.append(" and pea.attributeDefNameSourceId in (");
119       sql.append(HibUtils.convertToInClause(attributeDefNameSourceIds, byHqlStatic));
120       sql.append(") ");
121     }
122     if (memberIdsSize > 0) {
123       sql.append(" and pea.memberSourceId in (");
124       sql.append(HibUtils.convertToInClause(memberSourceIds, byHqlStatic));
125       sql.append(") ");
126     }
127     
128     if (pointInTimeFrom != null) {
129       Long endDateAfter = pointInTimeFrom.getTime() * 1000;
130       sql.append(" and (pea.membershipEndTimeDb is null or pea.membershipEndTimeDb > '" + endDateAfter + "')");
131       sql.append(" and (pea.groupSetEndTimeDb is null or pea.groupSetEndTimeDb > '" + endDateAfter + "')");
132       sql.append(" and (pea.actionSetEndTimeDb is null or pea.actionSetEndTimeDb > '" + endDateAfter + "')");
133       sql.append(" and (pea.attributeDefNameSetEndTimeDb is null or pea.attributeDefNameSetEndTimeDb > '" + endDateAfter + "')");
134       sql.append(" and (pea.roleSetEndTimeDb is null or pea.roleSetEndTimeDb > '" + endDateAfter + "')");
135       sql.append(" and (pea.attributeAssignEndTimeDb is null or pea.attributeAssignEndTimeDb > '" + endDateAfter + "')");
136     }
137     
138     if (pointInTimeTo != null) {
139       Long startDateBefore = pointInTimeTo.getTime() * 1000;
140       sql.append(" and pea.membershipStartTimeDb < '" + startDateBefore + "'");
141       sql.append(" and pea.groupSetStartTimeDb < '" + startDateBefore + "'");
142       sql.append(" and pea.actionSetStartTimeDb < '" + startDateBefore + "'");
143       sql.append(" and pea.attributeDefNameSetStartTimeDb < '" + startDateBefore + "'");
144       sql.append(" and pea.roleSetStartTimeDb < '" + startDateBefore + "'");
145       sql.append(" and pea.attributeAssignStartTimeDb < '" + startDateBefore + "'");
146     }
147     
148     byHqlStatic
149       .setCacheable(false)
150       .setCacheRegion(KLASS + ".findPermissions");
151 
152     int maxAssignments = GrouperConfig.retrieveConfig().propertyValueInt("ws.findPermissions.maxResultSize", 30000);
153     
154     String sqlString = sql.toString();
155     
156     //if we did where and, then switch to where
157     sqlString = sqlString.replaceAll("where\\s+and", "where");
158     
159     Set<PermissionEntry> results = byHqlStatic.createQuery(selectPrefix + sqlString).listSet(PermissionEntry.class);
160 
161     int size = GrouperUtil.length(results);
162     if (maxAssignments >= 0) {
163       
164       //see if too many
165       if (size > maxAssignments) {
166         throw new RuntimeException("Too many results: " + size);
167       }
168     }
169 
170     //nothing to filter
171     if (size == 0) {
172       return results;
173     }
174     
175     //if the hql didnt filter, we need to do that here
176     results = grouperSession.getAttributeDefResolver().postHqlFilterPermissions(grouperSessionSubject, results);
177     
178     //we should be down to the secure list
179     return results;
180   }
