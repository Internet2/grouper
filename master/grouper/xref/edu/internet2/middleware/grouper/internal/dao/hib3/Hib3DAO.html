95    /**
96     * init hibernate if not initted
97     */
98    public synchronized static void initHibernateIfNotInitted() {
99      if (hibernateInitted) {
100       return;
101     }
102     
103     //this might not be completely accurate
104     hibernateInitted = true;
105 
106     try {
107       // Find the custom configuration file
108       Properties  p   = GrouperHibernateConfig.retrieveConfig().properties();
109       
110       //unencrypt pass
111       if (p.containsKey("hibernate.connection.password")) {
112         String newPass = Morph.decryptIfFile(p.getProperty("hibernate.connection.password"));
113         p.setProperty("hibernate.connection.password", newPass);
114       }
115       
116       String connectionUrl = StringUtils.defaultString(GrouperUtil.propertiesValue(p,"hibernate.connection.url"));
117 
118       {
119         String dialect = StringUtils.defaultString(GrouperUtil.propertiesValue(p,"hibernate.dialect"));
120         dialect = GrouperDdlUtils.convertUrlToHibernateDialectIfNeeded(connectionUrl, dialect);
121         p.setProperty("hibernate.dialect", dialect);
122       }
123       
124       {
125         String driver = StringUtils.defaultString(GrouperUtil.propertiesValue(p,"hibernate.connection.driver_class"));
126         driver = GrouperDdlUtils.convertUrlToDriverClassIfNeeded(connectionUrl, driver);
127         p.setProperty("hibernate.connection.driver_class", driver);
128       }      
129       
130       // And now load all configuration information
131       CFG = new Configuration()
132         .addProperties(p);
133       addClass(CFG, Hib3AttributeAssignActionDAO.class);
134       addClass(CFG, Hib3AttributeAssignActionSetDAO.class);
135       addClass(CFG, Hib3AttributeAssignActionSetViewDAO.class);
136       addClass(CFG, Hib3AttributeAssignDAO.class);
137       addClass(CFG, Hib3AttributeAssignValueDAO.class);
138       addClass(CFG, Hib3AttributeDefDAO.class);
139       addClass(CFG, Hib3AttributeDefNameDAO.class);
140       addClass(CFG, Hib3AttributeDefNameSetDAO.class);
141       addClass(CFG, Hib3AttributeDefNameSetViewDAO.class);
142       addClass(CFG, Hib3AttributeDefScopeDAO.class);
143       addClass(CFG, Hib3AuditEntryDAO.class);
144       addClass(CFG, Hib3AuditTypeDAO.class);
145       addClass(CFG, Hib3ChangeLogEntryDAO.class);
146       addClass(CFG, Hib3ChangeLogEntryDAO.class, "Hib3ChangeLogEntryTempDAO");
147       addClass(CFG, Hib3ChangeLogConsumerDAO.class);
148       addClass(CFG, Hib3ChangeLogTypeDAO.class);
149       addClass(CFG, Hib3CompositeDAO.class);
150       addClass(CFG, Hib3ExternalSubjectDAO.class);

152       addClass(CFG, Hib3FieldDAO.class);
153       addClass(CFG, Hib3GroupDAO.class);

155       addClass(CFG, Hib3MembershipDAO.class);
156       addClass(CFG, Hib3MembershipDAO.class, "Hib3ImmediateMembershipDAO");

186       addClass(CFG, Hib3ServiceRoleViewDAO.class);
187       addClass(CFG, Hib3StemSetDAO.class);
188       addClass(CFG, Hib3TableIndexDAO.class);
189       CFG.setInterceptor(new Hib3SessionInterceptor());
190       
191       //if we are testing, map these classes to the table (which may or may not exist)
192       try {
193         Class<?> hibernatableClass = Class.forName("edu.internet2.middleware.grouper.app.loader.TestgrouperLoader");
194         addClass(CFG, hibernatableClass);
195         hibernatableClass = Class.forName("edu.internet2.middleware.grouper.app.loader.TestgrouperLoaderGroups");
196         addClass(CFG, hibernatableClass);
197         
198         hibernatableClass = Class.forName("edu.internet2.middleware.grouper.subj.TestgrouperSubjAttr");
199         addClass(CFG, hibernatableClass);
200         
201       } catch (ClassNotFoundException cnfe) {
202         //this is ok
203       }
204       
205       GrouperHooksUtils.callHooksIfRegistered(GrouperHookType.LIFECYCLE, 
206           LifecycleHooks.METHOD_HIBERNATE_INIT, HooksLifecycleHibInitBean.class, 
207           CFG, Configuration.class, null);
208       
209       // And finally create our session factory
210       //trying to avoid warning of using the same dir
211       String tmpDir = GrouperUtil.tmpDir();
212       try {
213         String newTmpdir = StringUtils.trimToEmpty(tmpDir);
214         if (!newTmpdir.endsWith("\\") && !newTmpdir.endsWith("/")) {
215           newTmpdir += File.separator;
216         }
217         newTmpdir += "grouper_ehcache_auto_" + GrouperUtil.uniqueId();
218         System.setProperty(GrouperUtil.JAVA_IO_TMPDIR, newTmpdir);
219         
220         //now it should be using a unique directory
221         FACTORY = CFG.buildSessionFactory();
222       } finally {
223         
224         //put tmpdir back
225         if (tmpDir == null) {
226           System.clearProperty(GrouperUtil.JAVA_IO_TMPDIR);
227         } else {
228           System.setProperty(GrouperUtil.JAVA_IO_TMPDIR, tmpDir);
229         }
230       }
231 
232     } catch (Throwable t) {
233       String msg = "unable to initialize hibernate: " + t.getMessage();
234       LOG.fatal(msg, t);
235       throw new RuntimeException(msg, t);
236     }

249   /**
250    * 
251    * @param _CFG
252    * @param mappedClass
253    * @param entityNameXmlFileNameOverride send in an entity name if the entity name and xml file are different than
254    * the class file.
255    */
256   private static void addClass(Configuration _CFG, Class<?> mappedClass, String entityNameXmlFileNameOverride) {
257     String resourceName = resourceNameFromClassName(mappedClass, entityNameXmlFileNameOverride);
258     String xml = GrouperUtil.readResourceIntoString(resourceName, false);
259     
260     if (xml.contains("<version")) {
261       
262       //if versioned, then make sure the setting in class is there
263       String optimisiticLockVersion = "optimistic-lock=\"version\"";
264       
265       if (!StringUtils.contains(xml, optimisiticLockVersion)) {
266         throw new RuntimeException("If there is a versioned class, it must contain " +
267         		"the class level attribute: optimistic-lock=\"version\": " + mappedClass.getName() + ", " + resourceName);
268       }
269       
270       //if versioned, then see if we are disabling
271       boolean optimisiticLocking = GrouperConfig.retrieveConfig().propertyValueBoolean("dao.optimisticLocking", true);
272       
273       if (!optimisiticLocking) {
274         xml = StringUtils.replace(xml, optimisiticLockVersion, "optimistic-lock=\"none\"");
275       }
276     }
277     _CFG.addXML(xml);
278 
279   }

281   /**
282    * class is e.g. edu.internet2.middleware.grouper.internal.dto.Attribute,
283    * must return e.g. edu.internet2.middleware.grouper.internal.dao.hib3.Hib3AttributeDAO
284    * @param theClass
285    * @param entityNameXmlFileNameOverride pass in an override if the entity name and xml file are different than
286    * the class file
287    * @return the string of resource
288    */
289   public static String resourceNameFromClassName(Class theClass, String entityNameXmlFileNameOverride) {
290     String daoClass = theClass.getName();
291     if (!StringUtils.isBlank(entityNameXmlFileNameOverride)) {
292       daoClass = StringUtils.replace(daoClass, theClass.getSimpleName(), entityNameXmlFileNameOverride);
293     }
294     //replace with hbm
295     String result = StringUtils.replace(daoClass, ".", "/") + ".hbm.xml";
296     
297     return result;
298   }
