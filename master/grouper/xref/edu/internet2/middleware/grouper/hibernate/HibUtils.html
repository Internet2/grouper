75    /**
76     * Whether the type of the class means that it should be handled as a primitive
77     * query vs a hibernate object query.
78     * @param clazz is the class to check.
79     * @return true if so.
80     */
81    static boolean handleAsPrimitive(Class clazz) {
82      
83      if (clazz.isArray()) {
84        clazz = clazz.getComponentType();
85      }
86      
87      if (clazz.isPrimitive() || clazz == java.lang.String.class
88          || clazz == BigDecimal.class || clazz == Integer.class
89          || clazz == java.sql.Date.class || clazz == java.util.Date.class
90          || clazz == java.sql.Time.class || clazz == java.sql.Timestamp.class
91          || clazz == java.sql.Clob.class || clazz == java.sql.Blob.class
92          || clazz == java.sql.Ref.class || clazz == Boolean.class || clazz == Byte.class
93          || clazz == Short.class || clazz == Integer.class || clazz == Float.class
94          || clazz == Double.class || clazz == Long.class) {
95        return true;
96      }
97      return false;
98    }

100   /**
101    * Convert the params to friendly strings
102    * 
103    * @param params
104    * @param types
105    * @return the string of the params (for logging)
106    */
107   @SuppressWarnings("unchecked")
108   public static String paramsToString(Object params, Object types) {
109     //nothing to do if nothing to do
110     if (params == null && types == null) {
111       return "null";
112     }
113 
114     List<Object> paramsList = GrouperUtil.toList(params);
115     List<Type> typesList = (List<Type>)(Object)GrouperUtil.toList(types);
116 
117     int paramLength = GrouperUtil.length(paramsList);
118     int typeLength = GrouperUtil.length(typesList);
119 
120     if (paramLength != typeLength) {
121       throw new RuntimeException("The params length " + paramLength
122           + " must equal the types length " + typeLength);
123     }
124     StringBuilder result = new StringBuilder();
125     //loop through, set the params
126     Type currentType = null;
127     for (int i = 0; i < paramLength; i++) {
128       try {
129         currentType = typesList.get(i);
130         result.append(currentType.toLoggableString(paramsList.get(i), null)).append(",");
131       } catch (HibernateException he) {
132         result.append("<error>");
133       }
134     }
135 
136     return result.toString();
137   }

139   /**
140    * Parses an alias out of a sql query.
141    * @param text is the text to find the alias in.
142    * @param exceptionIfNotFound if true, when an alias is not found, and exception will be thrown.
143    * @return the alias.
144    */
145   public static String parseAlias(String text, boolean exceptionIfNotFound) {
146     Pattern pattern = Pattern.compile("\\{.*?\\}");
147     Matcher matcher = pattern.matcher(text);
148     String alias = null;
149     if (matcher.find()) {
150       alias = matcher.group();
151       alias = StringUtils.replace(alias, "{", "");
152       alias = StringUtils.replace(alias, "}", "");
153       if (alias != null && alias.indexOf(".") > -1) {
154         alias = alias.substring(0, alias.indexOf("."));
155       }
156       return alias;
157     }
158 
159     if (exceptionIfNotFound) {
160       throw new RuntimeException("Cannot find a sql alias in the text: " + text
161           + "An alias must be in the format {anything} or {anything.anything}. "
162           + "Each sql query must contain at least one alias.");
163     }
164     return null;
165   }

182    * 
183    * @param cacheable
184    * @param queryOptions
185    * @return if caching
186    */
187   public static boolean secondLevelCaching(Boolean cacheable, QueryOptions queryOptions) {
188 
189     HibernateSession hibernateSession = HibernateSession._internal_hibernateSession();
190     
191     //if hibernate session says no, then no
192     if (hibernateSession != null && !hibernateSession.isCachingEnabled()) {
193       return false;
194     }
195     
196     //cant find answer
197     if (cacheable == null && (queryOptions == null || queryOptions.getSecondLevelCache() == null)) {
198       return false;
199     }
200     
201     //if no options, but has cacheable
202     if (queryOptions == null || queryOptions.getSecondLevelCache() == null) {
203       return cacheable;
204     }
205 
206     //this one trumps all if not null
207     return queryOptions.getSecondLevelCache();
208   }

239   /**
240    * convert an hql to a count hql
241    * @param hql
242    * @return the hql of the count query
243    */
244   public static String convertHqlToCountHql(String hql) {
245     
246     if (startsWithFrom.matcher(hql).matches()) {
247       return "select count(*) " + hql;
248     }
249     Matcher selectAndFromMatcher = hasSelectAndFrom.matcher(hql);
250     if (selectAndFromMatcher.matches()) {
251       String selectPart = selectAndFromMatcher.group(1);
252       String endOfQuery = selectAndFromMatcher.group(2);
253       return "select count( " + selectPart + " ) " + endOfQuery;
254     }
255     throw new RuntimeException("Cant convert query to count query: " + hql);
256   }

342    * 1. If you edit an object that is in the hibernate session, and commit, it will

346    * reference, e.g. if the DTO went through it, then you will get an exception:
347    * "a different object with the same identifier value was already associated with the session"
348    * </pre>

351    * @param onlyEvictIfNotNew true to only evict if this is a nested tx
352    */

354       Object object, boolean onlyEvictIfNotNew) {
355     evict(hibernateSession, object, onlyEvictIfNotNew, true);

358   /**
359    * <pre>
360    * evict a list of objects from hibernate.  do this always for two reasons:
361    * 1. If you edit an object that is in the hibernate session, and commit, it will
362    * commit those changes magically.  Only objects called session.save(obj) or 
363    * update etc should be committed
364    * 2. If you select an object, then try to store it back (but have a different
365    * reference, e.g. if the DTO went through it, then you will get an exception:
366    * "a different object with the same identifier value was already associated with the session"
367    * </pre>
368    * @param hibernateSession grouper hibernateSession, can be null if not known
369    * @param object to evict that was just retrieved, can be list or array
370    * @param onlyEvictIfNotNew true to only evict if this is a nested tx
371    * @param evictBeforeFlush if evict before flush (dont do this if iterating through list)
372    */
373   public static void evict(HibernateSession hibernateSession,
374       Object object, boolean onlyEvictIfNotNew, boolean evictBeforeFlush) {
375     if (object instanceof Collection) {
376       HibUtils.evict(hibernateSession, (Collection)object, onlyEvictIfNotNew);
377       return;
378     }
379     
380     //dont worry about it if new and only evicting if not new
381     if (hibernateSession != null && hibernateSession.isNewHibernateSession() && onlyEvictIfNotNew) {
382       return;
383     }
384     
385     //if array, loop through
386     if (object != null && object.getClass().isArray()) {
387       if (evictBeforeFlush) {
388         hibernateSession.getSession().flush();
389       }
390       for (int i=0;i<Array.getLength(object);i++) {
391         HibUtils.evict(hibernateSession, Array.get(object, i), onlyEvictIfNotNew, false);
392       }
393       return;
394     }
395     
396     //not sure it could ever be null...
397     if (object != null) {
398       if (evictBeforeFlush) {
399         hibernateSession.getSession().flush();
400       }
401       
402       try {
403         hibernateSession.getSession().evict(object);
404       } catch (IllegalArgumentException e) {
405         // ignore
406         
407         // In hibernate 5, seems to sometimes throw:
408         // java.lang.IllegalArgumentException: Non-entity object instance passed to evict : cd620430b6204390be5e55e375a5cdb7
409         // at org.hibernate.event.internal.DefaultEvictEventListener.onEvict(DefaultEvictEventListener.java:95)
410         
411         // comment in hibernate 5 code:
412         
413         // see if the passed object is even an entity, and if not throw an exception
414         //              this is different than legacy Hibernate behavior, but what JPA 2.1 is calling for
415         //              with EntityManager.detach
416 
417         // TODO need to look at later
418       }
419     }
420   }

422   /**
423    * <pre>
424    * evict a list of objects from hibernate.  do this always for two reasons:
425    * 1. If you edit an object that is in the hibernate session, and commit, it will
426    * commit those changes magically.  Only objects called session.save(obj) or 
427    * update etc should be committed
428    * 2. If you select an object, then try to store it back (but have a different
429    * reference, e.g. if the DTO went through it, then you will get an exception:
430    * "a different object with the same identifier value was already associated with the session"
431    * </pre>
432    * @param hibernateSession grouper hibernateSession
433    * @param list of objects from hibernate to evict
434    * @param onlyEvictIfNotNew true to only evict if this is a nested tx
435    */
436   public static void evict(HibernateSession hibernateSession,
437       Collection<Object> list, boolean onlyEvictIfNotNew) {
438     if (list == null) {
439       return;
440     }
441     hibernateSession.getSession().flush();
442     for (Object object : list) {
443       evict(hibernateSession, object, onlyEvictIfNotNew, false);
444     }
445   }

492   /**
493    * make a list of criterions.  e.g. listCrit(crit1, crit2, etc).  will AND or OR them together
494    * this is null and empty safe
495    * @param junction either conjunction or disjunction
496    * @param criterions
497    * @return the criterion containing the list or null if none passed in
498    */
499   private static Criterion listCritHelper(Junction junction, Criterion... criterions) {
500     int criterionsLength = GrouperUtil.length(criterions);
501     if (criterionsLength == 0) {
502       return null;
503     }
504     
505     //if one no need for junction
506     if (criterionsLength == 1 && criterions[0] != null) {
507       return criterions[0];
508     }
509     
510     //count to see if any added
511     int resultsCount = 0;
512     for (int i=0;i<criterionsLength;i++) {
513       Criterion current = criterions[i];
514       if (current != null) {
515         resultsCount++;
516         junction.add(current);
517       }
518     }
519     if (resultsCount == 0) {
520       return null;
521     }
522     return junction;
523     
524   }

610   /**
611    * Attach params for a prepared statement.  The type of the params and types must be the
612    * same (e.g. either both array or list, but not one is Array, and the other list
613    * @param statement
614    * @param params either null, Object, Object[], or List of Objects
615    * @throws HibernateException
616    * @throws SQLException
617    */
618   @SuppressWarnings("unchecked")
619   public static void attachParams(PreparedStatement statement, Object params)
620       throws HibernateException, SQLException {
621     if (GrouperUtil.length(params) == 0) {
622       return;
623     }
624     List<Object> paramList = listObject(params);
625     List<Type> typeList = hibernateTypes(paramList);
626     attachParams(statement, paramList, typeList);
627   }

629   /**
630    * Attach params for a prepared statement.  The type of the params and types must be the
631    * same (e.g. either both array or list, but not one is Array, and the other list
632    * @param statement
633    * @param params either null, Object, Object[], or List of Objects
634    * @param types either null, Type, Type[], or List of Objects
635    * @throws HibernateException
636    * @throws SQLException
637    */
638   static void attachParams(PreparedStatement statement, Object params, Object types)
639       throws HibernateException, SQLException {
640     int paramLength = GrouperUtil.length(params);
641     int typeLength = GrouperUtil.length(types);
642     
643     //nothing to do if nothing to do
644     if (paramLength == 0 && typeLength == 0) {
645       return;
646     }
647   
648       if (paramLength != typeLength) {
649       throw new RuntimeException("The params length must equal the types length and params " +
650       "and types must either both or neither be null");
651       }
652     
653       List paramList = listObject(params);
654       List typeList = listObject(types);
655   
656       //loop through, set the params
657       Type currentType = null;
658       for (int i = 0; i < paramLength; i++) {
659         //not sure why the session implementer is null, if this ever fails for a type, 
660         //might want to not use hibernate and brute force it
661         currentType = (Type) typeList.get(i);
662         currentType.nullSafeSet(statement, paramList.get(i), i + 1, null);
663       }
664   
665   }

684   /**
685    * Returns a Hibernate Type for the given java type. Handles both primitives and Objects.
686    * Will throw an exception if the given object is null or if a type cannot be found for it.
687    * @param o is the object to find the Type for.
688    * @return the Type.
689    */
690   public static Type hibernateType(Object o) {
691     if (o == null) {
692       //its possible to bind null (e.g. for an update), so just use object to do this
693       return ObjectType.INSTANCE;
694     }
695     Class clazz = o.getClass();
696   
697     if (clazz == int.class || o instanceof Integer) {
698       return IntegerType.INSTANCE;
699     } else if (clazz == double.class || clazz == Double.class) {
700       return DoubleType.INSTANCE;
701     } else if (clazz == long.class || clazz == Long.class) {
702       return LongType.INSTANCE;
703     } else if (clazz == float.class || clazz == Float.class) {
704       return FloatType.INSTANCE;
705     } else if (clazz == byte.class || clazz == Byte.class) {
706       return ByteType.INSTANCE;
707     } else if (clazz == boolean.class || clazz == Boolean.class) {
708       return TrueFalseType.INSTANCE;
709     } else if (clazz == char.class || clazz == Character.class) {
710       return CharacterType.INSTANCE;
711     } else if (clazz == short.class || clazz == Short.class) {
712       return ShortType.INSTANCE;
713     } else if (clazz == java.util.Date.class || clazz == java.sql.Date.class) {
714       //return Hibernate.TIMESTAMP;
715       return DateType.INSTANCE;
716     } else if (clazz == Timestamp.class) {
717       return TimestampType.INSTANCE;
718     } else if (clazz == String.class) {
719       return StringType.INSTANCE;
720     }
721     throw new RuntimeException(
722         "Cannot find a hibernate type to associate with java type " + clazz);
723   }

726    * Returns a list of Hibernate types corresponding to the given params.
727    * @param params are the objects to get the types for. Can be list, Object, or array.
728    * @return the corresponding types.
729    */
730   public static List<Type> hibernateTypes(List<Object> params) {
731   
732     int length = GrouperUtil.length(params);
733   
734     //if null, make sure the same (or exception later)
735     if (length == 0) {
736       return null;
737     }
738   
739     // Get the types.
740     // Create a list of hibernate types.
741     List<Type> types = new ArrayList<Type>();
742   
743     for (int i = 0; i < length; i++) {
744       Object o = params.get(i);
745       types.add(hibernateType(o));
746     }
747     return types;
748   }

784   public static String convertToSubjectInClause(Collection<Subject> subjects, HqlQuery hqlQuery, String memberAlias) {

847   /**
848    * convert a collection of multikeys to an in clause with multiple args.  currently this only
849    * works with strings, though we could add support for more types in future
850    * @param collection
851    * @param scalarable to set the string
852    * @param columnNames names of columns in multikey
853    * @param whereClause
854    */
855   public static void convertToMultiKeyInClause(Collection<MultiKey> collection, HqlQuery scalarable, 
856       Collection<String> columnNames, StringBuilder whereClause) {
857     
858     String unique = GrouperUtil.uniqueId();
859     
860     int collectionSize = collection.size();
861     int columnNamesSize = columnNames.size();
862     int i = 0;
863     
864     if (GrouperUtil.length(collection) == 0) {
865       return;
866     }
867     
868     whereClause.append(" and ( ");
869     
870     for (MultiKey multiKey : collection) {
871       
872       whereClause.append(" ( ");
873       
874       int j = 0;
875       
876       for (String columnName : columnNames) {
877 
878         String var = unique + i + "_" + j;
879 
880         whereClause.append(" ").append(columnName).append(" = :").append(var).append(" ");
881         
882         //add to query
883         scalarable.setString(var, (String)multiKey.getKey(j));
884 
885         if (j < columnNamesSize-1) {
886           whereClause.append(" and ");
887         }
888         j++;
889 
890       }
891       
892       whereClause.append(" ) ");
893 
894       if (i < collectionSize-1) {
895         whereClause.append(" or ");
896       }
897       i++;
898     }
899     whereClause.append(" ) ");
900   }

933    * @param sourceColumnName 
934    */
935   public static void convertSourcesToSqlInString(Set<Source> sources, 
936       HqlQuery hqlQuery, StringBuilder sql, String sourceColumnName) {
937     if (sources == null || sources.size() == 0) {
938       return;
939     }
940     
941     //simplify if 1
942     if (sources.size() == 1) {
943       String bindVar = "sourceId_" + GrouperUtil.uniqueId();
944       sql.append(" and ").append(sourceColumnName).append(" = :").append(bindVar).append(" ");
945       hqlQuery.setString(bindVar, sources.iterator().next().getId());
946       return;
947     }
948     
949     List<String> sourcesStrings = new ArrayList<String>();
950     for (Source source : sources) {
951       sourcesStrings.add(source.getId());
952     }
953     
954     String questions = HibUtils.convertToInClause(sourcesStrings, hqlQuery);
955     
956     sql.append(" and ").append(sourceColumnName).append(" in ( ").append(questions).append(" ) ");
957     
958   }

960   /**
961    * e.g. ('g:gsa', 'jdbc')
962    * @param sources
963    * @return the in string, of sources sorted alphabetically
964    */
965   public static String convertSourcesToSqlInString(Set<Source> sources) {
966     if (sources == null || sources.size() == 0) {
967       return null;
968     }
969     
970     //simplify if 1
971     if (sources.size() == 1) {
972       return " ('" + sources.iterator().next().getId() + "') ";
973     }
974     
975     List<String> sourcesStrings = new ArrayList<String>();
976     for (Source source : sources) {
977       sourcesStrings.add(source.getId());
978     }
979     
980     //sort 
981     Collections.sort(sourcesStrings);
982     
983     StringBuilder result = new StringBuilder();
984     result.append(" (");
985     for (int i=0;i<sourcesStrings.size();i++) {
986       result.append("'").append(sourcesStrings.get(i)).append("'");
987       if (i != sourcesStrings.size()-1) {
988         result.append(", ");
989       }
990     }
991     result.append(") ");
992     return result.toString();
993   }
