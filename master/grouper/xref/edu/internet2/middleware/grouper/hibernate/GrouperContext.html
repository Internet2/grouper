92    /**
93     * assign fields in audit entry from outer audit (note, this might not exist)
94     * @param auditEntry
95     */
96    public static void assignAuditEntryFieldsOuter(AuditEntry auditEntry) {
97  
98      auditEntry.setCreatedOn(new Timestamp(System.currentTimeMillis()));
99      auditEntry.setEnvName(GrouperConfig.retrieveConfig().propertyValueString("grouper.env.name"));
100     auditEntry.setGrouperVersion(GrouperVersion.GROUPER_VERSION);
101     auditEntry.setServerUserName(System.getProperty("user.name"));
102     String serverHost = null;
103     try {
104       serverHost = InetAddress.getLocalHost().getHostName();
105     } catch (Exception e) {
106       //dont worry about it
107       LOG.info(e);
108     }
109     auditEntry.setServerHost(serverHost);
110     GrouperContext grouperDefaultContext = defaultContext.get();
111     if (grouperDefaultContext != null) {
112       
113       auditEntry.setGrouperEngine(grouperDefaultContext.grouperEngine);
114       auditEntry.setUserIpAddress(grouperDefaultContext.callerIpAddress);
115       auditEntry.setLoggedInMemberId(grouperDefaultContext.loggedInMemberId);
116       auditEntry.setActAsMemberId(grouperDefaultContext.loggedInMemberIdActAs);
117       if(auditEntry.getActAsMemberId()==null) {
118     	  GrouperSession s = GrouperSession.staticGrouperSession();
119     	  //If there isn't an ActAsMemberId, but there is an active GrouperSession
120     	  //and its subject doesn't match the loggedInMemberId, use it for the actAsMemberId.
121     	  //Means that from GSH, where there is no loggedInMemberId, the current GrouperSession
122     	  //determines the actAsMemberId which means that audit log entries can be filtered by
123     	  //the nominal subject performing the action
124     	  if(s != null && !s.getMemberUuid().equals(auditEntry.getLoggedInMemberId())) {
125     		  auditEntry.setActAsMemberId(s.getMemberUuid());
126     	  }
127     	}
128 
129     }
130 
131     GrouperContext grouperOuterContext = currentOuterContext.get();
132     if (grouperOuterContext != null) {
133       
134       if (!StringUtils.isBlank(grouperOuterContext.callerIpAddress)) {
135         auditEntry.setUserIpAddress(grouperOuterContext.callerIpAddress);
136       }
137       
138       if (!StringUtils.isBlank(grouperOuterContext.loggedInMemberId)) {
139         auditEntry.setLoggedInMemberId(grouperDefaultContext.loggedInMemberId);
140       }
141       if (!StringUtils.isBlank(grouperOuterContext.loggedInMemberIdActAs)) {       
142         auditEntry.setActAsMemberId(grouperDefaultContext.loggedInMemberIdActAs);
143       }
144     }
145   }

155   /**
156    * assign fields in audit entry
157    * @param auditEntry
158    */
159   public static void assignAuditEntryFields(AuditEntry auditEntry) {
160 
161     assignAuditEntryFieldsOuter(auditEntry);
162     
163     GrouperContext grouperInnerContext = currentInnerContext.get();
164     if (grouperInnerContext == null) {
165       throw new NullPointerException("grouperInnerContext is null, was it not started?");
166     }
167     auditEntry.setContextId(grouperInnerContext.getContextId());
168     //divide nanos by 1000 to get micros
169     auditEntry.setDurationMicroseconds((System.nanoTime() - grouperInnerContext.startedNanos)/1000);
170     auditEntry.setQueryCount(grouperInnerContext.queryCount);
171     
172     if (StringUtils.isBlank(auditEntry.getGrouperEngine())) {
173       auditEntry.setGrouperEngine(grouperInnerContext.grouperEngine);
174     }
175     
176   }

264   /**
265    * retrieve current context id
266    * @param requireContext true to require context (if required in grouper.properties)
267    * @return context id
268    */
269   public static String retrieveContextId(boolean requireContext) {
270     GrouperContext grouperContextInner = currentInnerContext.get();
271     if (grouperContextInner == null) {
272       //TODO make these default to true before 1.5
273       if (requireContext) {
274         if (GrouperConfig.retrieveConfig().propertyValueBoolean("audit.requireAuditsForAllActions", false)) {
275           throw new RuntimeException("No context found");
276         }
277         if (GrouperConfig.retrieveConfig().propertyValueBoolean("audit.logAuditsForMissingActions", false)) {
278           LOG.warn("No context found here is the stack", new RuntimeException("Not an exception, just need the stack"));
279         }
280       }
281       return null;
282     }
283     return grouperContextInner.getContextId();
284   }

286   /**
287    * create a new context if one doesnt already exist
288    * @return true if created one, false if already existed
289    */
290   static boolean createNewInnerContextIfNotExist() {
291     if (currentInnerContext.get() != null) {
292       return false;
293     }
294     GrouperContext grouperContextPrivate = new GrouperContext();
295     currentInnerContext.set(grouperContextPrivate);
296     return true;

306   /**
307    * @param exceptionIfDefaultAlreadyExists
308    * @param grouperEngineIdentifier
309    * @param useServerIpAddressAsUserIpAddress if the users use this server, use that for user ip address
310    * @return the context
311    */
312   public static GrouperContext createNewDefaultContext(GrouperEngineIdentifier grouperEngineIdentifier, 
313       boolean exceptionIfDefaultAlreadyExists, boolean useServerIpAddressAsUserIpAddress) {
314     
315     if (exceptionIfDefaultAlreadyExists && defaultContext.get() != null) {
316       throw new RuntimeException("Default context is already set: " 
317           + defaultContext.get() + ", " + grouperEngineIdentifier);
318     }
319     
320     GrouperContext grouperContext = new GrouperContext();
321     grouperContext.grouperEngine = grouperEngineIdentifier.getGrouperEngine();
322     defaultContext.set(grouperContext);
323     
324     if (useServerIpAddressAsUserIpAddress) {
325       try {
326         grouperContext.setCallerIpAddress(InetAddress.getLocalHost().getHostAddress());
327       } catch (UnknownHostException uhe) {
328         //not the end of the world
329         LOG.info(uhe);
330       }
331     }
332     
333     return grouperContext;
334   }

336   /**
337    * tell the context another query occurred
338    */
339   public static void incrementQueryCount() {
340     
341     totalQueryCount++;
342     
343     GrouperContext grouperContextInner = currentInnerContext.get();
344     if (grouperContextInner != null) {
345       grouperContextInner.queryCount++;
346     }
347     
348     GrouperContext grouperContextOuter = currentOuterContext.get();
349     if (grouperContextOuter != null) {
350       grouperContextOuter.queryCount++;
351     }
352   }
