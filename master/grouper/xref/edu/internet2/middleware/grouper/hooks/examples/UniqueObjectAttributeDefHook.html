29  /**
30   * When attribute defs are created or saved, make sure there are no name conflicts with other object types
31   */
32  public class UniqueObjectAttributeDefHook extends AttributeDefHooks {
33  
34  
35    /**
36     * @see edu.internet2.middleware.grouper.hooks.AttributeDefHooks#attributeDefPreUpdate(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksAttributeDefBean)
37     */
38    @Override
39    public void attributeDefPreUpdate(HooksContext hooksContext,
40        HooksAttributeDefBean preUpdateBean) {
41      checkNameOfAttributeDef(preUpdateBean.getAttributeDef().getName());
42    }
43  
44    /**
45     * @see edu.internet2.middleware.grouper.hooks.AttributeDefHooks#attributeDefPreInsert(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksAttributeDefBean)
46     */
47    @Override
48    public void attributeDefPreInsert(HooksContext hooksContext,
49        HooksAttributeDefBean preInsertBean) {
50      checkNameOfAttributeDef(preInsertBean.getAttributeDef().getName());
51    }
52  
53    /**
54     * make sure name is not used in other object types
55     * @param name
56     */
57    private void checkNameOfAttributeDef(String name) {
58      UniqueObjectGroupHook.assertNoGroupsWithThisNameExist(name);
59      UniqueObjectStemHook.assertNoStemsWithThisNameExist(name);
60      //dont check attribute defs
61      UniqueObjectAttributeDefNameHook.assertNoAttributeDefNamesWithThisNameExist(name);
62    }
63  
64    
65    /**
66     * make sure no groups by this name exist
67     * @param name
68     */
69    public static void assertNoAttributeDefsWithThisNameExist(final String name) {
70      
71      GrouperSession.internal_callbackRootGrouperSession(new GrouperSessionHandler() {
72        
73        public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
74          
75          AttributeDef attributeDef = AttributeDefFinder.findByName(name, false);
76          
77          if (attributeDef != null) {
78            throw new HookVeto("veto.uniqueObject.attributeDef.name", "The ID is already in use by an attribute definition, please use a different ID");
79          }
80          
81          return null;
82        }
83      });
84          
85    }
