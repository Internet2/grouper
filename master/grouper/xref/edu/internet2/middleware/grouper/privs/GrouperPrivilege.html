20  package edu.internet2.middleware.grouper.privs;
21  
22  import java.util.Collection;
23  
24  import edu.internet2.middleware.grouper.GrouperAPI;
25  import edu.internet2.middleware.subject.Subject;
26  
27  /**
28   * combines AccessPrivilege and NamingPrivilege
29   */
30  public interface GrouperPrivilege {
31    
32    /** get the object this privilege is assigned to (e.g. group or stem object) 
33     * @return the group or stem
34     */
35    public GrouperAPI getGrouperApi();
36    
37    /**
38     * Get name of implementation class for this privilege type. (e.g. from grouper.properties)
39     * @return the name
40     */
41    public String getImplementationName();
42    
43    /**
44     * get type of privilege (e.g. access or naming)
45     * @return the type
46     */
47    public String getType();
48    
49    /**
50     * Get name of privilege.
51     * @return  Name of privilege.
52     */
53    public String getName();
54  
55    /**
56     * Get subject which was granted privilege on this object.
57     * @return  {@link Subject} that was granted privilege.
58     */
59    public Subject getOwner();
60  
61    /**
62     * Get subject which has this privilege.
63     * @return  {@link Subject} that has this privilege.
64     */
65    public Subject getSubject();
66  
67    /**
68     * Returns true if privilege can be revoked.
69     * @return  Boolean true if privilege can be revoked.
70     */
71    public boolean isRevokable();
72    
73    /**
74     * if we are caching subject objects, then set it here...  do not change the subject here
75     * @param subject
76     */
77    public void internalSetSubject(Subject subject);
78  }
