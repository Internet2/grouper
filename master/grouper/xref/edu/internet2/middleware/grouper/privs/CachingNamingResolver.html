74    /** */
75    public static final String CACHE_HASPRIV = CachingNamingResolver.class.getName()
76        + ".HasPrivilege";
77  
78    /**
79     * @param resolver 
80     * @since   1.2.1
81     */
82    public CachingNamingResolver(NamingResolver resolver) {
83      super(resolver);
84    }
85  
86    /**
87     * Retrieve boolean from cache for <code>hasPrivilege(...)</code>.
88     * @param ns 
89     * @param subj 
90     * @param priv 
91     * @return  Cached return value or null.
92     * @since   1.2.1
93     */
94    private Boolean getFromHasPrivilegeCache(Stem ns, Subject subj, Privilege priv) {
95      // TODO 20070823 are these the right element keys to use?
96      Element el = EhcacheController.ehcacheController().getCache(CACHE_HASPRIV).get(
97          new MultiKey(ns.getUuid(), subj.getSourceId(), subj.getId(), priv));
98      if (el != null) {
99        return (Boolean) el.getObjectValue();
100     }
101     return null;
102   }

104   /**
105    * @param cache 
106    * @return  ehcache statistics for <i>cache</i>.
107    * @since   1.2.1
108    */
109   public CacheStats getStats(String cache) {
110     return EhcacheController.ehcacheController().getStats(CACHE_HASPRIV);
111   }

113   /**
114    * @see     NamingResolver#grantPrivilege(Stem, Subject, Privilege, String)
115    * @since   1.2.1
116    */
117   public void grantPrivilege(Stem stem, Subject subject, Privilege privilege, String uuid)
118       throws IllegalArgumentException,
119       UnableToPerformException {
120     // TODO 20070816 add caching
121     super.getDecoratedResolver().grantPrivilege(stem, subject, privilege, uuid);
122     this.flushCache();
123     this.putInHasPrivilegeCache(stem, subject, privilege, Boolean.TRUE);
124   }
125 
126   /**
127    * @see     NamingResolver#hasPrivilege(Stem, Subject, Privilege)
128    * @since   1.2.1
129    */
130   public boolean hasPrivilege(Stem stem, Subject subject, Privilege privilege)
131       throws IllegalArgumentException {
132     Boolean rv = this.getFromHasPrivilegeCache(stem, subject, privilege);
133     if (rv == null) {
134       rv = super.getDecoratedResolver().hasPrivilege(stem, subject, privilege);
135       this.putInHasPrivilegeCache(stem, subject, privilege, rv);
136     }
137     return rv;
138   }
139 
140   /**
141    * Put boolean into cache for <code>hasPrivilege(...)</code>.
142    * @param ns 
143    * @param subj 
144    * @param priv 
145    * @param rv 
146    * @since   1.2.1
147    */
148   private void putInHasPrivilegeCache(Stem ns, Subject subj, Privilege priv, Boolean rv) {
149     EhcacheController.ehcacheController().getCache(CACHE_HASPRIV).put(
150         new Element(new MultiKey(ns.getUuid(), subj.getSourceId(), subj.getId(), priv), rv));
151   }

195   /**
196    * @see edu.internet2.middleware.grouper.privs.NamingResolver#postHqlFilterStems(java.util.Set, edu.internet2.middleware.subject.Subject, java.util.Set)
197    */
198   public Set<Stem> postHqlFilterStems(Set<Stem> stems, Subject subject,
199       Set<Privilege> privInSet) {
200     Set<Stem> filteredStems = super.getDecoratedResolver().postHqlFilterStems(stems,
201         subject, privInSet);
202 
203     //add to cache
204     for (Stem stem : stems) {
205       putInHasPrivilegeCache(stem, subject, AccessPrivilege.VIEW, filteredStems
206           .contains(stem));
207     }
208 
209     //return filtered groups
210     return filteredStems;
211   }
