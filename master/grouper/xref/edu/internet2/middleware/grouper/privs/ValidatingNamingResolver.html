54     */
55    public Set<Stem> getStemsWhereSubjectDoesntHavePrivilege(String stemId, Scope scope, 
56        Subject subject, Privilege privilege, boolean considerAllSubject, 
57        String sqlLikeString)
58        throws IllegalArgumentException {
59      this.param.notNullSubject(subject).notNullPrivilege(privilege);
60      return super.getDecoratedResolver().getStemsWhereSubjectDoesntHavePrivilege(stemId, scope,

62    }

177   /**
178    * @see edu.internet2.middleware.grouper.privs.NamingResolver#hqlFilterStemsWhereClause(edu.internet2.middleware.subject.Subject, edu.internet2.middleware.grouper.hibernate.HqlQuery, java.lang.StringBuilder, java.lang.String, java.util.Set)
179    */
180   public boolean hqlFilterStemsWhereClause(Subject subject, HqlQuery hqlQuery,
181       StringBuilder hql, String stemColumn, Set<Privilege> privInSet) {
182 
183     this.param.notNullSubject(subject).notNullHqlQuery(hqlQuery);
184 
185     NamingResolver decoratedResolver = super.getDecoratedResolver();
186     //CachingNamingResolver
187     return decoratedResolver.hqlFilterStemsWhereClause(subject, hqlQuery, hql,
188         stemColumn, privInSet);
189   }
190 
191   /**
192    * @see edu.internet2.middleware.grouper.privs.NamingResolver#postHqlFilterStems(java.util.Set, edu.internet2.middleware.subject.Subject, java.util.Set)
193    */
194   public Set<Stem> postHqlFilterStems(Set<Stem> stems, Subject subject,
195       Set<Privilege> privInSet) {
196 
197     if (GrouperUtil.length(privInSet) == 0) {
198       return stems;
199     }
200     this.param.notNullSubject(subject);
201 
202     Set<Stem> filteredStems = super.getDecoratedResolver().postHqlFilterStems(stems,
203         subject, privInSet);
204 
205     //return filtered groups
206     return filteredStems;
207   }

219    */

227     //CachingAccessResolver

229         stemColumn, privilege, considerAllSubject);
230   }
