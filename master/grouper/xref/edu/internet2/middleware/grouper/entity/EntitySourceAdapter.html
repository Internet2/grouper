33  package edu.internet2.middleware.grouper.entity;
34  import java.util.LinkedHashSet;
35  import java.util.Set;
36  
37  import edu.internet2.middleware.grouper.GrouperSourceAdapter;
38  import edu.internet2.middleware.grouper.group.TypeOfGroup;
39  import edu.internet2.middleware.subject.provider.SubjectTypeEnum;
40  
41  /** 
42   * Source adapter for Entities
43   * <p/>
44   * <p>
45   * @author  chris hyzer.
46   * @version $Id: GrouperSourceAdapter.java,v 1.31 2009-08-12 04:52:21 mchyzer Exp $
47   */
48  public class EntitySourceAdapter extends GrouperSourceAdapter {
49  
50    /** instance */
51    private static EntitySourceAdapter instance = null;
52  
53    /**
54     * type of groups to search on
55     * @return type of groups
56     */
57    @Override
58    public Set<TypeOfGroup> typeOfGroups() {
59      return TypeOfGroup.ENTITY_SET;
60    }
61  
62    /** types */
63    private Set _types  = new LinkedHashSet();
64    
65  
66    /**
67     * Gets the SubjectTypes supported by this source.
68     * <pre class="eg">
69     * SourceAdapter  sa    = new GrouperSourceAdapter();
70     * Set            types = sa.getSubjectTypes();
71     * </pre>
72     * @return  Subject types supported by this source.
73     */
74    public Set getSubjectTypes() {
75      if (_types.size() != 1) {
76        _types.add( SubjectTypeEnum.valueOf("application") );
77      }
78      return _types;
79    } // public Set getSubjectTypes()
80  
81    /**
82     * instance
83     * @return instance
84     */
85    public static EntitySourceAdapter instance() {
86      if (instance == null) {
87        synchronized (EntitySourceAdapter.class) {
88          if (instance == null) {
89            EntitySourceAdapter newInstance = new EntitySourceAdapter();
90            newInstance.setId("grouperEntities");
91            newInstance.setName("grouperEntities");
92            newInstance.addSubjectType("application");
93            
94            newInstance.addInitParam("subjectVirtualAttribute_0_searchAttribute0", "${subject.getName()},${subject.getDescription()}");
95            newInstance.addInitParam("sortAttribute0", "name");
96            newInstance.addInitParam("searchAttribute0", "searchAttribute0");
97            newInstance.addInternalAttribute("searchAttribute0");
98            
99            instance = newInstance;
100           
101         }
102       }
103     }
104     return instance;
105   }
106 }
