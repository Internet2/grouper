278    * Export Groups Registry to XML output.
279    * <p/>

282   public static void main(String args[]) {
283     if (XmlArgs.internal_wantsHelp(args)) {
284       System.out.println( _getUsage() );
285       //System.exit(0);
286       return;
287     }
288     Properties rc = new Properties();
289     try {
290       rc = XmlArgs.internal_getXmlExportArgs(args);
291     }
292     catch (Exception e) {
293       e.printStackTrace();
294       System.err.println();
295       System.err.println( _getUsage() );
296       //System.exit(1);
297       return;
298     }
299     XmlExporter exporter = null;
300     try {
301       GrouperSession session = GrouperSession.startRootSession();
302       Subject subj = SubjectFinder.findByIdentifier( rc.getProperty(XmlArgs.RC_SUBJ), true );
303       GrouperSession.stopQuietly(session);
304       
305       exporter = new XmlExporter(
306         GrouperSession.start(
307           subj, false),
308         XmlUtils.internal_getUserProperties(LOG, rc.getProperty(XmlArgs.RC_UPROPS)));
309       _handleArgs(exporter, rc);
310       LOG.debug("Finished export to [" + rc.getProperty(XmlArgs.RC_EFILE) + "]");
311     }
312     catch (Exception e) {
313       LOG.fatal("unable to export to xml: " + e.getMessage());
314       try {
315       exporter.xml.internal_close();
316       }catch(IOException ioe){}
317       //System.exit(1);
318       return;
319     }
320     finally {
321       if (exporter != null) {
322         try {
323           exporter.s.stop();
324         }
325         catch (SessionException eS) {
326           LOG.error(eS.getMessage());
327         }
328       }
329     }
330     //System.exit(0);
331     return;
332   } // public static void main(args)

475       sb.append( Quote.single( XML.escape( g.getName() ) ) );

480     sb.append( GrouperConfig.NL );

579   private static Set _getListFieldsForGroup(Group g)
580     throws  SchemaException 
581   {
582     Set       lists   = new LinkedHashSet();
583     Field     defList = Group.getDefaultList();
584     Field     f;
585     GroupType type;
586     Iterator  iterF;
587     Iterator  iter    = g.getTypes().iterator();
588     while (iter.hasNext()) {
589       type  = (GroupType) iter.next();
590       iterF = type.getFields().iterator();
591       while (iterF.hasNext()) { 
592         f = (Field) iterF.next();
593         if (
594               f.getType().equals(FieldType.LIST)
595           &&  !f.equals(defList)
596           &&  g.canReadField(f)
597         )
598         {
599           lists.add(f.getName());
600         }
601       }
602     }
603     return lists;
604   } // private static Set _getListFieldsForGroup(group)

638   private static void _handleArgs(XmlExporter exporter, Properties rc) 
639     throws  Exception {
640     
641     String userAuditFilename = rc.getProperty("userAuditFilename");
642     if (!StringUtils.isBlank(userAuditFilename)) {
643       
644       XmlUserAuditExport.writeUserAudits(new File(userAuditFilename));
645       
646     }
647     
648     String userAuditOnly = rc.getProperty("userAuditOnly");
649     if (StringUtils.equalsIgnoreCase("true", userAuditOnly)) {
650       //we are done
651       return;
652     }
653     
654     Writer writer = new PrintWriter( new FileWriter( rc.getProperty(XmlArgs.RC_EFILE) ));
655     if (rc.getProperty(XmlArgs.RC_UUID) == null && rc.getProperty(XmlArgs.RC_NAME) == null) {
656       exporter.export(writer);
657     } 
658     else {
659       Group group = null;
660       Stem  stem  = null;
661       if (rc.getProperty(XmlArgs.RC_UUID) != null) {
662         String uuid = rc.getProperty(XmlArgs.RC_UUID);
663         try {
664           group = GroupFinder.findByUuid(exporter.s, uuid, true);
665           LOG.debug("Found group with uuid [" + uuid + "]");
666         } 
667         catch (GroupNotFoundException eGNF) {
668           // Look for stem instead
669           try {
670             stem = StemFinder.findByUuid(exporter.s, uuid, true);
671             LOG.debug("Found stem with uuid [" + uuid + "]");
672           } 
673           catch (StemNotFoundException eNSNF) {
674             throw new IllegalArgumentException(
675               "Could not find group or stem with uuid [" + uuid + "]"
676             );
677           }
678         }
679       } 
680       else {
681         String name = rc.getProperty(XmlArgs.RC_NAME);
682         try {
683           group = GroupFinder.findByName(exporter.s, name, true);
684           LOG.debug("Found group with name [" + name + "]");
685         } 
686         catch (GroupNotFoundException eGNF) {
687           // Look for stem instead
688           try {
689             stem = StemFinder.findByName(exporter.s, name, true);
690             LOG.debug("Found stem with name [" + name + "]");
691           } catch (StemNotFoundException eNSNF) {
692             // No group or stem
693             throw new IllegalArgumentException(
694               "Could not find group or stem with name [" + name + "]"
695             );
696           }
697         }
698       }
699       if (group != null) {
700         exporter.export(
701           writer, group,
702           Boolean.valueOf( rc.getProperty(XmlArgs.RC_RELATIVE) ).booleanValue(),
703           Boolean.valueOf( rc.getProperty(XmlArgs.RC_PARENT) ).booleanValue()
704         );
705       } 
706       else {
707         exporter.export(writer, stem, Boolean.valueOf( rc.getProperty(XmlArgs.RC_RELATIVE) ).booleanValue(),Boolean.valueOf( rc.getProperty(XmlArgs.RC_CHILDREN) ).booleanValue() );
708       }
709     }
