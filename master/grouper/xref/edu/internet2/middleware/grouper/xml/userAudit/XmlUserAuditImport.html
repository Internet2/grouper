115   /**
116    * 
117    * @param importFile 
118    */
119   public void readUserAudits(final File importFile) {
120   
121     HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, 
122         AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
123   
124           public Object callback(HibernateHandlerBean hibernateHandlerBean)
125               throws GrouperDAOException {
126   
127             try {
128               
129               xStream = XmlUserAuditExport.xstream();
130               
131               XmlUserAuditImport.this.hibernateSession = hibernateHandlerBean.getHibernateSession();
132               XmlUserAuditImport.this.session = hibernateSession.getSession();
133               
134               SAXReader reader = new SAXReader();
135               
136               reader.addHandler( "/userAudits", 
137                   new ElementHandler() {
138                       public void onStart(ElementPath path) {
139                         Element userAudit = path.getCurrent();
140                         XmlUserAuditImport.this.importFileVersion = userAudit.attributeValue("version");
141                         XmlUserAuditExport.logInfoAndPrintToScreen(
142                             "userAudit import: reading document: " + GrouperUtil.fileCanonicalPath(importFile) 
143                              + ", version: " + XmlUserAuditImport.this.importFileVersion);
144                       }
145                       public void onEnd(ElementPath path) {
146                         //we done!
147                       }
148                   }
149               );
150 
151               reader.addHandler( "/userAudits/xmlMembers", 
152                   new ElementHandler() {
153                       public void onStart(ElementPath path) {
154                           clearCounts();    
155                           initMembers();
156                       }
157                       public void onEnd(ElementPath path) {
158                           // process a ROW element
159                           Element row = path.getCurrent();
160           
161                           // prune the tree
162                           row.detach();
163 
164                           logInfoAndPrintCounts("members");
165                       }
166                   }
167               );
168 
169               reader.addHandler( "/userAudits/xmlMembers/XmlMember", 
170                   new ElementHandler() {
171                       public void onStart(ElementPath path) {
172                           // do nothing here...    
173                       }
174                       public void onEnd(ElementPath path) {
175                           // process a ROW element
176                           Element row = path.getCurrent();
177           
178                           // prune the tree
179                           row.detach();
180 
181                           XmlMember xmlMemberFromFile = (XmlMember)xStream.unmarshal(new Dom4JReader(row));
182                           
183                           XmlUserAuditImport.this.processXmlMemberFromFile(xmlMemberFromFile);
184                           
185                       }
186                   }
187               );
188 
189               //outer audit types element
190               reader.addHandler( "/userAudits/xmlAuditTypes", 
191                   new ElementHandler() {
192                       public void onStart(ElementPath path) {
193                         clearCounts();    
194                         //get all xml audit type from the DB
195                         XmlUserAuditImport.this.allAuditTypesInDb = GrouperDAOFactory.getFactory().getAuditType().findAll();
196                         XmlUserAuditImport.this.auditTypeTranslation = new HashMap<String, String>();
197                         
198                       }
199                       public void onEnd(ElementPath path) {
200                           // process a ROW element
201                           Element row = path.getCurrent();
202           
203                           // prune the tree
204                           row.detach();
205 
206                           logInfoAndPrintCounts("auditTypes");
207                       }
208                   }
209               );
210 
211               //inner audit type element
212               reader.addHandler( "/userAudits/xmlAuditTypes/XmlAuditType", 
213                   new ElementHandler() {
214                       public void onStart(ElementPath path) {
215                           // do nothing here...    
216                       }
217                       public void onEnd(ElementPath path) {
218                           // process a ROW element
219                           Element row = path.getCurrent();
220           
221                           // prune the tree
222                           row.detach();
223 
224                           XmlAuditType xmlAuditTypeFromFile = (XmlAuditType)xStream.unmarshal(new Dom4JReader(row));
225                           
226                           XmlUserAuditImport.this.processXmlAuditTypeFromFile(xmlAuditTypeFromFile);
227                           
228                       }
229                   }
230               );
231 
232               //outer audit entries element
233               reader.addHandler( "/userAudits/xmlAuditEntries", 
234                   new ElementHandler() {
235                       public void onStart(ElementPath path) {
236                         clearCounts();
237                         
238                       }
239                       public void onEnd(ElementPath path) {
240                           // process a ROW element
241                           Element row = path.getCurrent();
242           
243                           // prune the tree
244                           row.detach();
245 
246                           logInfoAndPrintCounts("auditEntries");
247                       }
248                   }
249               );
250 
251               //inner audit entry element
252               reader.addHandler( "/userAudits/xmlAuditEntries/XmlAuditEntry", 
253                   new ElementHandler() {
254                       public void onStart(ElementPath path) {
255                           // do nothing here...    
256                       }
257                       public void onEnd(ElementPath path) {
258                           // process a ROW element
259                           Element row = path.getCurrent();
260           
261                           // prune the tree
262                           row.detach();
263 
264                           XmlAuditEntry xmlAuditEntryFromFile = (XmlAuditEntry)xStream.unmarshal(new Dom4JReader(row));
265                           
266                           XmlUserAuditImport.this.processXmlAuditEntryFromFile(xmlAuditEntryFromFile);
267                           
268                       }
269                   }
270               );
271 
272               reader.read(importFile);
273             } catch (DocumentException de) {
274               throw new RuntimeException("Problem reading file: " + GrouperUtil.fileCanonicalPath(importFile));
275             }
276             return null;
277           }
278     });
279   }

281   /**
282    * init members from db
283    */
284   private void initMembers() {
285     //lets cache the current members in the DB
286     allMembersInRegistry = XmlMember.retrieveAllMembers(session);
287     
288     if (LOG.isInfoEnabled()) {
289       LOG.info("userAudit import: Read in " + allMembersInRegistry.size() + " members into cache");
290     }
291 
292     //lets make a reverse lookup too based on sourceId and subjectId
293     allMembersInRegistryBySubject = new HashMap<MultiKey, XmlMember>();
294     
295     for (XmlMember xmlMember : allMembersInRegistry.values()) {
296       
297       allMembersInRegistryBySubject.put(new MultiKey(xmlMember.getSourceId(), xmlMember.getSubjectId()), xmlMember);
298       
299     }
300     
301     if (LOG.isInfoEnabled()) {
302       LOG.info("userAudit import: Created reverse lookup member cache");
303     }
304     
305     //map from old member id to new member id
306     memberIdTranslation = new HashMap<String, String>();
307 
308   }

310   /**
311    * @param xmlMemberFromFile
312    */
313   private void processXmlMemberFromFile(XmlMember xmlMemberFromFile) {
314     
315     this.totalCount++;
316     
317     XmlMember xmlMemberInDb = allMembersInRegistry.get(xmlMemberFromFile.getUuid());
318     
319     if (xmlMemberInDb != null) {
320       LOG.debug("userAudit import: read existing member: " + xmlMemberFromFile.toString());
321       XmlUserAuditImport.this.existingCount++;
322       return;
323     }
324     
325     xmlMemberInDb = allMembersInRegistryBySubject.get(
326         new MultiKey(xmlMemberFromFile.getSourceId(), xmlMemberFromFile.getSubjectId()));
327     
328     if (xmlMemberInDb != null) {
329       //change in uuid
330       LOG.debug("userAudit import: read member with changed uuid: " 
331           + xmlMemberFromFile.toString() + ", to uuid: " + xmlMemberInDb.getUuid());
332       
333       XmlUserAuditImport.this.changeCount++;
334       memberIdTranslation.put(xmlMemberFromFile.getUuid(), xmlMemberInDb.getUuid());
335       return;
336     }
337 
338     XmlUserAuditImport.this.newCount++;
339     
340     //this is a new member
341     LOG.debug("userAudit import: read new member: " 
342         + xmlMemberFromFile.toString());
343     
344     SubjectImpl simpleSubject = new SubjectImpl(
345         xmlMemberFromFile.getSubjectId(), xmlMemberFromFile.getSubjectId(), 
346         xmlMemberFromFile.getSubjectId(), SubjectTypeEnum.PERSON.getName(), xmlMemberFromFile.getSourceId());
347     Member member = MemberFinder.internal_findBySubject(simpleSubject, xmlMemberFromFile.getUuid(), true);
348  
349     if (!StringUtils.equals(member.getUuid(), xmlMemberFromFile.getUuid())) {
350       throw new RuntimeException("Why is member uuid not inserted??? " + xmlMemberFromFile);
351     }
352   }

354   /**
355    * @param xmlAuditTypeFromFile
356    */
357   private void processXmlAuditTypeFromFile(XmlAuditType xmlAuditTypeFromFile) {
358     AuditType auditTypeInDb = null;
359     
360     for (AuditType auditType : this.allAuditTypesInDb) {
361       
362       if (StringUtils.equals(xmlAuditTypeFromFile.getId(), auditType.getId())) {
363         auditTypeInDb = auditType;
364         break;
365       }
366       
367       if (StringUtils.equals(xmlAuditTypeFromFile.getAuditCategory(), auditType.getAuditCategory())
368           && StringUtils.equals(xmlAuditTypeFromFile.getActionName(), auditType.getActionName())) {
369         auditTypeInDb = auditType;
370         break;
371       }
372       
373     }
374     
375     AuditType auditTypeFromFile = xmlAuditTypeFromFile.toAuditType();
376   
377     GrouperUtil.substituteStrings(this.memberIdTranslation, auditTypeFromFile);
378     
379     XmlUserAuditImport.this.totalCount++;
380     
381     //if new entry
382     if (auditTypeInDb == null) {
383       LOG.debug("userAudit import: read new auditType: " + auditTypeFromFile.toStringDeep()); 
384       GrouperDAOFactory.getFactory().getAuditType().saveOrUpdate(auditTypeFromFile);
385       XmlUserAuditImport.this.newCount++;
386       return;
387     }
388     
389     //see if translating
390     boolean translatingId = !StringUtils.equals(auditTypeFromFile.getId(), auditTypeInDb.getId());
391     if (translatingId) {
392       this.auditTypeTranslation.put(auditTypeFromFile.getId(), auditTypeInDb.getId());
393     }
394     
395     if (auditTypeFromFile.equalsDeep(auditTypeInDb)) {
396       XmlUserAuditImport.this.existingCount++;
397       if (!translatingId) {
398         //they are equal, let it be
399         LOG.debug("userAudit import: auditType exists in DB: " + auditTypeFromFile); 
400         return;
401       }
402       
403       //they are equal except for id
404       LOG.debug("userAudit import: auditType exists in DB with different UUID: " + auditTypeFromFile 
405           + ", new id: " + auditTypeInDb.getId()); 
406       
407       return;
408     }
409   
410     //at this point, it exists, but its not the same
411     XmlUserAuditImport.this.changeCount++;
412     LOG.debug("userAudit import: auditType exists in DB but is different, will be: " + auditTypeFromFile); 
413     
414     auditTypeFromFile.setId(auditTypeInDb.getId());
415     auditTypeFromFile.setHibernateVersionNumber(auditTypeInDb.getHibernateVersionNumber());
416     GrouperDAOFactory.getFactory().getAuditType().saveOrUpdate(auditTypeFromFile);
417     
418   }

420   /**
421    * @param xmlAuditEntryFromFile
422    */
423   private void processXmlAuditEntryFromFile(XmlAuditEntry xmlAuditEntryFromFile) {
424 
425     AuditEntry auditEntryInDb = GrouperDAOFactory.getFactory().getAuditEntry()
426       .findById(xmlAuditEntryFromFile.getId(), false);
427 
428     AuditEntry auditEntryFromFile = xmlAuditEntryFromFile.toAuditEntry();
429 
430     boolean newMemberId = GrouperUtil.substituteStrings(this.memberIdTranslation, auditEntryFromFile);
431 
432     XmlUserAuditImport.this.totalCount++;
433     
434     //if new entry
435     if (auditEntryInDb == null) {
436       LOG.debug("userAudit import: read new auditEntry: " + auditEntryFromFile.toStringDeep()); 
437       GrouperDAOFactory.getFactory().getAuditEntry().saveOrUpdate(auditEntryFromFile);
438       XmlUserAuditImport.this.newCount++;
439       return;
440     }
441     
442     if (auditEntryFromFile.equalsDeep(auditEntryInDb)) {
443       //they are equal, let it be
444       XmlUserAuditImport.this.existingCount++;
445       LOG.debug("userAudit import: auditEntry exists in DB: " + auditEntryFromFile); 
446       return;
447     }
448       
449     XmlUserAuditImport.this.changeCount++;
450       
451     //lets see if the auditType changed
452     String newAuditTypeId = this.auditTypeTranslation.get(auditEntryFromFile.getAuditTypeId());
453     boolean needsNewAuditTypeId = !StringUtils.isBlank(newAuditTypeId);
454     if (needsNewAuditTypeId) {
455       auditEntryFromFile.setAuditTypeId(newAuditTypeId);
456     }
457 
458     //see if there are member uuids to change
459 
460     
461     //they are equal except for id
462     LOG.debug("userAudit import: auditEntry exists in DB with same UUID and different values " +
463     		"(typeChange? " + (needsNewAuditTypeId ? "T" : "F") + ", memberIdChange? " 
464     		+ (newMemberId ? "T" : "F") + "): " + auditEntryFromFile); 
465     
466     auditEntryFromFile.setId(auditEntryInDb.getId());
467     auditEntryFromFile.setHibernateVersionNumber(auditEntryInDb.getHibernateVersionNumber());
468     GrouperDAOFactory.getFactory().getAuditEntry().saveOrUpdate(auditEntryFromFile);
469     
470   }

476     XmlUserAuditImport.this.changeCount = 0;    
477     XmlUserAuditImport.this.newCount = 0;

486     XmlUserAuditExport.logInfoAndPrintToScreen(
487         "userAudit import: completed with " + StringUtils.rightPad(label + ":", 13) + " total: " 
488         + XmlUserAuditImport.this.totalCount + ", new: " 
489         + XmlUserAuditImport.this.newCount + ", existing: " 
490         + XmlUserAuditImport.this.existingCount + ", changed: "
491         + XmlUserAuditImport.this.changeCount);
492   }
