70     * assign grouper object finder types (null or empty is all)
71     * @param grouperObjectFinderTypes1
72     * @return this for chaining
73     */
74    public GrouperFavoriteFinder assignGrouperObjectFinderType(Collection<GrouperFavoriteFinderType> grouperObjectFinderTypes1) {
75      
76      if (grouperObjectFinderTypes1 == null) {
77        grouperObjectFinderTypes1 = new HashSet<GrouperFavoriteFinderType>();
78      } else {
79      
80        this.grouperFavoriteFinderTypes = new HashSet<GrouperFavoriteFinderType>(grouperObjectFinderTypes1);
81      }
82      
83      return this;
84    }

86    /**

89     */
90    public static enum GrouperFavoriteFinderType {
91  
92      /** search includes stems */
93      stems,
94      
95      /** groups */
96      groups,
97      
98      /** subjects */
99      subjects,
100     
101     /** attribute def names */
102     attributeDefNames,
103     
104     /** attribute defs */
105     attributeDefs;
106 
107     /** all types */
108     public static final Set<GrouperFavoriteFinderType> ALL_GROUPER_FINDER_TYPES =
109         Collections.unmodifiableSet(GrouperUtil.toSet(stems, groups, subjects, attributeDefNames, attributeDefs));
110     
111     /**
112      * do a case-insensitive matching
113      * 
114      * @param string
115      * @param exceptionOnNull will not allow null or blank entries
116      * @return the enum or null or exception if not found
117      */
118     public static GrouperFavoriteFinderType valueOfIgnoreCase(String string, boolean exceptionOnNull) {
119       
120       return GrouperUtil.enumValueOfIgnoreCase(GrouperFavoriteFinderType.class, 
121           string, exceptionOnNull);
122 
123     }

202   /**
203    * find grouper objects based on the paramet
204    * @return the set of objects
205    */
206   public Set<GrouperObject> findFavorites() {
207     
208     boolean findGroups = GrouperUtil.length(this.grouperFavoriteFinderTypes) == 0 ? true : 
209       this.grouperFavoriteFinderTypes.contains(GrouperFavoriteFinderType.groups);
210     
211     boolean findStems = GrouperUtil.length(this.grouperFavoriteFinderTypes) == 0 ? true : 
212       this.grouperFavoriteFinderTypes.contains(GrouperFavoriteFinderType.stems);
213     
214     boolean findSubjects = GrouperUtil.length(this.grouperFavoriteFinderTypes) == 0 ? true : 
215       this.grouperFavoriteFinderTypes.contains(GrouperFavoriteFinderType.subjects);
216     
217     boolean findAttributeDefs = GrouperUtil.length(this.grouperFavoriteFinderTypes) == 0 ? true : 
218       this.grouperFavoriteFinderTypes.contains(GrouperFavoriteFinderType.attributeDefs);
219     
220     boolean findAttributeDefNames = GrouperUtil.length(this.grouperFavoriteFinderTypes) == 0 ? true : 
221       this.grouperFavoriteFinderTypes.contains(GrouperFavoriteFinderType.attributeDefNames);
222     
223     Set<GrouperObject> results = new LinkedHashSet<GrouperObject>();
224     
225     if (StringUtils.isBlank(this.userDataGroupName)) {
226       throw new RuntimeException("userDataGroupName is required");
227     }
228     Subject theSubject = this.subject;
229     if (theSubject == null) {
230       theSubject = GrouperSession.staticGrouperSession().getSubject();
231     }
232 
233     Set<String> filterTokensLower = new HashSet<String>();
234     
235     if (!StringUtils.isBlank(this.filterText)) {
236       
237       String filterTokenLower = this.filterText.toLowerCase();
238       
239       if (this.splitScope != null && this.splitScope) {
240         filterTokensLower.addAll(GrouperUtil.splitTrimToSet(filterTokenLower, " "));
241       } else {
242         filterTokensLower.add(filterTokenLower);
243       }
244       
245     }
246 
247     if (findGroups) {
248       results.addAll(GrouperUtil.nonNull(GrouperUserDataApi.favoriteGroups(this.userDataGroupName, theSubject)));
249     }
250     if (findStems) {
251       results.addAll(GrouperUtil.nonNull(GrouperUserDataApi.favoriteStems(this.userDataGroupName, theSubject)));
252     }
253     if (findAttributeDefNames) {
254       results.addAll(GrouperUtil.nonNull(GrouperUserDataApi.favoriteAttributeDefNames(this.userDataGroupName, theSubject)));
255     }
256     if (findAttributeDefs) {
257       results.addAll(GrouperUtil.nonNull(GrouperUserDataApi.favoriteAttributeDefs(this.userDataGroupName, theSubject)));
258     }
259     if (findSubjects) {
260       Set<Member> members = GrouperUtil.nonNull(GrouperUserDataApi.favoriteMembers(this.userDataGroupName, theSubject));
261       
262       for (Member member : members) {
263         results.add(new GrouperObjectSubjectWrapper(member.getSubject()));
264       }
265       
266     }
267 
268     Iterator<GrouperObject> iterator = results.iterator();
269     
270     //if filtering, remove the ones that dont match
271     if (GrouperUtil.length(filterTokensLower) > 0) {
272       while (iterator.hasNext()) {
273         GrouperObject grouperObject = iterator.next();
274         if (!grouperObject.matchesLowerSearchStrings(filterTokensLower)) {
275           iterator.remove();
276         }
277       }
278     }
279     
280     if (this.queryOptions != null && this.queryOptions.getQueryPaging() != null) {
281       
282       //set the total record count
283       this.queryOptions.getQueryPaging().setTotalRecordCount(GrouperUtil.length(results));
284       this.queryOptions.setCount(new Long(GrouperUtil.length(results)));
285       
286       int pageNumber = this.queryOptions.getQueryPaging().getPageNumber();
287       int pageSize = this.queryOptions.getQueryPaging().getPageSize();
288       
289       results = new LinkedHashSet<GrouperObject>(GrouperUtil.batchList(new ArrayList<GrouperObject>(results), pageSize, pageNumber-1));
290       
291     }
292     
293     return results;
294   }
