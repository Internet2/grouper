123   /**
124    * Run {@link GrouperShell}.
125    * <pre class="eg">
126    * // Launch GrouperShell in interactive mode
127    * % gsh.sh
128    *
129    * // Run GrouperShell script
130    * % gsh.sh script.gsh
131    * 
132    * // Read commands from STDIN
133    * % gsh.sh - 
134    * </pre>
135    * @param args 
136    * @since 0.0.1
137    */
138   public static void main(String args[]) {
139     //set this and leave it...
140     @SuppressWarnings("unused")
141     GrouperContext grouperContext = GrouperContext.createNewDefaultContext(
142         GrouperEngineBuiltin.GSH, false, true);
143     
144     boolean wasSpecialCase = handleSpecialCase(args);
145     if(wasSpecialCase) {
146     	return;
147     }
148 	  runFromGsh = true;
149 	
150     GrouperStartup.runFromMain = true;
151     GrouperStartup.startup();
152     //turn on logging
153     Log bshLogger = LogFactory.getLog("bsh");
154     if (bshLogger.isTraceEnabled()) {
155       Interpreter.TRACE = true;
156     }
157     if (bshLogger.isDebugEnabled()) {
158       Interpreter.DEBUG = true;
159     }
160     exitOnFailure = true;
161     try {
162       grouperShellHelper(args, null);
163     }
164     catch (GrouperShellException eGS) {
165       eGS.printStackTrace();
166       LOG.error("GSH is exiting: " + eGS.getMessage(), eGS);
167       System.exit(1);
168     }
169     System.exit(0);
170   } // public static void main(args)

173  /**
174  * Avoid GSH initialisation for special cases 
175  * @param args - command line arguments
176  * @return whether this was handled as a special case
177  */
178 private static boolean handleSpecialCase(String[] args) {
179 	  if(args == null || args.length==0) {
180 		  return false;
181 	  }
182 	  if("-h".equalsIgnoreCase(args[0])||"-help".equalsIgnoreCase(args[0])) {
183 		  System.out.println(_getUsage());
184 		  return true;
185 	  }
186 	  
187 	  if("-nocheck".equalsIgnoreCase(args[0])) {
188 		  GrouperStartup.ignoreCheckConfig = true;
189 		  return false;
190 	  }
191 	  String mainClass = mainLookups.get(args[0].toLowerCase());
192 	  if(mainClass==null) {
193 		  return false;
194 	  }
195 	  String[] mainArgs = new String[args.length -1];
196 	  for(int i=1;i<args.length;i++) {
197 		  mainArgs[i-1] = args[i];
198 	  }
199 	  
200 	  try {
201 		  Class claz = Class.forName(mainClass);
202 		  
203 		  Method method = claz.getMethod("main", String[].class);
204 		  method.invoke(null, (Object)mainArgs);
205 	  }catch(Exception e) {
206 	    if (ExceptionUtils.getFullStackTrace(e).contains("PSPCLI")) {
207 	      String error = "Make sure you have run 'ant dist' in ldappcng, and 'ant ldappcng' in grouper to copy the libs over";
208         LOG.fatal(error);
209         System.err.println(error);
210 	    }
211 		  if(e instanceof RuntimeException) {
212 			  throw (RuntimeException)e;
213 		  }
214 		  throw new RuntimeException(e);
215 	  }
216 	  
217 	  return true;
218   } //private static boolean handleSpecialCase(args)

220   /**
221    * helper method to kick off GSH without exiting
222    * @param args
223    * @param inputStreamParam if passing in an inputStream
224    * @throws GrouperShellException
225    */
226   static void grouperShellHelper(String args[], InputStream inputStreamParam) throws GrouperShellException {
227     
228     System.out.println("Type help() for instructions");
229     
230     GrouperContextTypeBuiltIn.setDefaultContext(GrouperContextTypeBuiltIn.GSH);
231     
232     new GrouperShell( new ShellCommandReader(args, inputStreamParam )).run();
233   }

283   // @since   0.0.1
284   protected static GrouperSession getSession(Interpreter i) 
285     throws  GrouperShellException
286   {
287     try {
288       GrouperSession s = (GrouperSession) GrouperShell.get(i, GSH_SESSION);
289       if (s != null && s.getSubjectDb() == null) {
290         s = null;
291         GrouperShell.set(i, GSH_SESSION, s);
292       }
293       if (s == null) {
294         s = GrouperSession.staticGrouperSession(false);
295         
296         if (s == null) {
297           s = GrouperSession.start(
298             SubjectFinder.findRootSubject()
299           );
300         }
301         GrouperShell.set(i, GSH_SESSION, s);
302       }
303       return s;
304     }
305     catch (Exception e) {
306       if (i != null) {
307         i.error(e.getMessage());
308       }
309       throw new GrouperShellException(e);
310     }
311   } // protected static GrouperSession getSession(i)

364   // @since   0.1.1
365   protected void run() 
366     throws  GrouperShellException
367   {
368     String cmd = new String();
369     try {
370       this.interpreter.eval(  "importCommands(\"edu.internet2.middleware.grouper\")");

373       this.interpreter.eval(  "importCommands(\"edu.internet2.middleware.grouper.privs\")");
374       //this.i.eval(  "importCommands(\"edu.internet2.middleware.grouper.registry\")");
375       this.interpreter.eval(  "importCommands(\"edu.internet2.middleware.subject\")");
376       this.interpreter.eval(  "importCommands(\"edu.internet2.middleware.subject.provider\")");
377       this.interpreter.eval(  "import edu.internet2.middleware.grouper.*;");
378       this.interpreter.eval(  "import edu.internet2.middleware.grouper.app.loader.ldap.*;");
379       this.interpreter.eval(  "import edu.internet2.middleware.grouper.attr.*;");
380       this.interpreter.eval(  "import edu.internet2.middleware.grouper.attr.assign.*;");
381       this.interpreter.eval(  "import edu.internet2.middleware.grouper.attr.finder.*;");
382       this.interpreter.eval(  "import edu.internet2.middleware.grouper.attr.value.*;");
383       this.interpreter.eval(  "import edu.internet2.middleware.grouper.audit.*;");
384       this.interpreter.eval(  "import edu.internet2.middleware.grouper.client.*;");
385       this.interpreter.eval(  "import edu.internet2.middleware.grouper.entity.*;");
386       this.interpreter.eval(  "import edu.internet2.middleware.grouper.externalSubjects.*;");
387       this.interpreter.eval(  "import edu.internet2.middleware.grouper.group.*;");
388       this.interpreter.eval(  "import edu.internet2.middleware.grouper.ldap.*;");
389       this.interpreter.eval(  "import edu.internet2.middleware.grouper.app.loader.*;");
390       this.interpreter.eval(  "import edu.internet2.middleware.grouper.xml.*;");
391       this.interpreter.eval(  "import edu.internet2.middleware.grouper.registry.*;");
392       this.interpreter.eval(  "import edu.internet2.middleware.grouper.app.usdu.*;");
393       this.interpreter.eval(  "import edu.internet2.middleware.grouper.app.gsh.*;");
394       this.interpreter.eval(  "import edu.internet2.middleware.grouper.app.misc.*;");
395       this.interpreter.eval(  "import edu.internet2.middleware.grouper.privs.*;");
396       this.interpreter.eval(  "import edu.internet2.middleware.grouper.rules.*;");
397       this.interpreter.eval(  "import edu.internet2.middleware.grouper.misc.*;");
398       this.interpreter.eval(  "import edu.internet2.middleware.grouper.hibernate.*;");
399       this.interpreter.eval(  "import edu.internet2.middleware.grouper.permissions.*;");
400       this.interpreter.eval(  "import edu.internet2.middleware.grouper.util.*;");
401       this.interpreter.eval(  "import edu.internet2.middleware.subject.*;");
402       this.interpreter.eval(  "import edu.internet2.middleware.subject.provider.*;");
403       this.interpreter.eval(  "import edu.internet2.middleware.grouper.userData.*;");
404       
405     }
406     catch (bsh.EvalError eBBB) {
407       throw new GrouperShellException(GshErrorMessages.I_IMPORT + eBBB.getMessage(), eBBB);
408     }
409     while ( ( cmd = r.getNext() ) != null) {
410       if ( this._isComment(cmd) ) {
411         continue;
412       }
413       if ( this._isTimeToExit(cmd) ) {
414         int txSize = HibernateSession._internal_staticSessions().size();
415         boolean hasTransactions = txSize>0;
416         if (hasTransactions) {
417           String error = "Exiting in the middle of " + txSize + " open transactions, they will be rolled back and closed";
418           this.interpreter.println(error);
419           LOG.error(error);
420           HibernateSession._internal_closeAllHibernateSessions(new RuntimeException());
421         }
422         this._stopSession();
423         if (hasTransactions) {
424           ShellHelper.exitDueToOpenTransaction(this.interpreter);
425         }
426         break;
427       }
428       // Now try to eval the command
429       cmd = ShellHelper.eval(interpreter, cmd);
430     }
431   }

436   /**
437    * 
438    * @param i
439    * @param var
440    * @return true if istrue
441    */
442   private static boolean _isTrue(Interpreter i, String var) {
443     boolean rv = false;
444     try {
445       Object  obj = GrouperShell.get(i, var);
446     if (
447                 (obj != null)
448             &&  (obj instanceof Boolean)
449             &&  (Boolean.TRUE.equals( obj ))
450          )
451       {
452         rv = true;
453       }
454     }
455     catch (bsh.EvalError eBEE) {
456       i.error(eBEE.getMessage());
457     }
458     return rv;
459   } // private static boolean _isTrue(i, var)

482   // @since   0.0.1
483   private void _stopSession() 
484     throws  GrouperShellException
485   {
486     try {
487       // `GrouperShell.getSession()` will start the session if it doesn't exist.
488       // That's just slow.  And wrong.
489       GrouperSession s = (GrouperSession) GrouperShell.get(this.interpreter, GSH_SESSION);
490       if (s != null) {
491         s.stop();
492         this.interpreter.unset(GSH_SESSION);
493       }
494     }
495     catch (Exception e) {
496       if (interpreter != null) {
497         this.interpreter.error(e.getMessage());
498       }
499       throw new GrouperShellException(e);
500     }
501   } // private void _stopSession()
