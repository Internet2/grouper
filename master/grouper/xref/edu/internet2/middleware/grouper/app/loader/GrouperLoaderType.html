207 
208         Hib3GrouperLoaderLog hib3GrouploaderLog = loaderJobBean.getHib3GrouploaderLogOverall();
209         
210         if (StringUtils.equals(MAINTENANCE_CLEAN_LOGS, hib3GrouploaderLog.getJobName())) {
211           StringBuilder jobMessage = new StringBuilder();
212           {
213             int daysToKeepLogs = GrouperLoaderConfig.retrieveConfig().propertyValueInt(GrouperLoaderConfig.LOADER_RETAIN_DB_LOGS_DAYS, 7);
214             if (daysToKeepLogs != -1) {
215               //lets get a date
216               Calendar calendar = GregorianCalendar.getInstance();
217               //get however many days in the past
218               calendar.add(Calendar.DAY_OF_YEAR, -1 * daysToKeepLogs);
219               //run a query to delete (note, dont retrieve records to java, just delete)
220               int records = HibernateSession.bySqlStatic().executeSql("delete from grouper_loader_log where last_updated < ?", 
221                   (List<Object>)(Object)GrouperUtil.toList(new Timestamp(calendar.getTimeInMillis())));
222               jobMessage.append("Deleted " + records + " records from grouper_loader_log older than " + daysToKeepLogs + " days old.  ");
223               
224             } else {
225               jobMessage.append("Configured to not delete records from grouper_loader_log table.  ");
226             }
227           }
228           {
229             int daysToKeepLogs = GrouperLoaderConfig.retrieveConfig().propertyValueInt("loader.retain.db.change_log_entry.days", 14);
230             if (daysToKeepLogs != -1) {
231               //lets get a date
232               Calendar calendar = GregorianCalendar.getInstance();
233               //get however many days in the past
234               calendar.add(Calendar.DAY_OF_YEAR, -1 * daysToKeepLogs);
235               //note, this is *1000 so that we can differentiate conflicting records
236               long time = calendar.getTimeInMillis()*1000L;
237               //run a query to delete (note, dont retrieve records to java, just delete)
238               int records = HibernateSession.bySqlStatic().executeSql("delete from grouper_change_log_entry where created_on < ?", 
239                   (List<Object>)(Object)GrouperUtil.toList(new Long(time)));
240               jobMessage.append("Deleted " + records + " records from grouper_change_log_entry older than " + daysToKeepLogs + " days old. (" + time + ")  ");
241             } else {
242               jobMessage.append("Configured to not delete records from grouper_change_log_entry table.  ");
243             }
244             
245           }
246           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.SUCCESS.name());
247           hib3GrouploaderLog.setJobMessage(jobMessage.toString());
248         } else if (StringUtils.equals(GROUPER_REPORT, hib3GrouploaderLog.getJobName())) {
249 
250 
251           //how often to run usdu
252           String usduSchedule = StringUtils.defaultString(GrouperLoaderConfig.retrieveConfig().propertyValueString(
253               "daily.report.usdu.daysToRun")).toLowerCase();
254           String badMemberSchedule = StringUtils.defaultString(GrouperLoaderConfig.retrieveConfig().propertyValueString(
255               "daily.report.badMembership.daysToRun")).toLowerCase();
256           
257           boolean isRunUsdu = dayListContainsToday(usduSchedule);
258           boolean isRunBadMember = dayListContainsToday(badMemberSchedule);
259           
260           String emailTo = GrouperLoaderConfig.retrieveConfig().propertyValueString("daily.report.emailTo");
261           String reportDirectory = GrouperLoaderConfig.retrieveConfig().propertyValueString("daily.report.saveInDirectory");
262           
263           if (StringUtils.isBlank(emailTo) && StringUtils.isBlank(reportDirectory)) {
264             throw new RuntimeException("grouper-loader.properties property daily.report.emailTo " +
265             		"or daily.report.saveInDirectory needs to be filled in");
266           }
267           
268           String report = null;
269           //keep track if ends up being error
270           RuntimeException re = null;
271           try {
272             report = new GrouperReport().findBadMemberships(isRunBadMember).findUnresolvables(isRunUsdu)
273               .runReport();
274           } catch (RuntimeException e) {
275             report = e.toString() + "\n\n" + GrouperUtil.getFullStackTrace(e) + "\n\n";
276             if (e instanceof GrouperReportException) {
277               report += ((GrouperReportException)e).getResult();
278             }
279             re = e;
280           }
281           
282           //if we are emailing
283           if (!StringUtils.isBlank(emailTo)) {
284             new GrouperEmail().setBody(report).setSubject("Grouper report").setTo(emailTo).send();
285           }
286           
287           //if we are saving to dir on server
288           if (!StringUtils.isBlank(reportDirectory)) {
289             reportDirectory = GrouperUtil.stripLastSlashIfExists(reportDirectory) + File.separator;
290             GrouperUtil.mkdirs(new File(reportDirectory));
291             File reportFile = new File(reportDirectory + "grouperReport_" 
292                 + new SimpleDateFormat("yyyyMMdd_HH_mm_ss").format(new Date()) + ".txt");
293             GrouperUtil.saveStringIntoFile(reportFile, report);
294           }
295           
296           //end in error if error
297           if (re != null) {
298             throw re;
299           }
300           
301           hib3GrouploaderLog.setJobMessage("Ran the grouper report, isRunUnresolvable: " 
302               + isRunUsdu + ", isRunBadMembershipFinder: " + isRunBadMember + ", sent to: " + emailTo);
303           
304           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.SUCCESS.name());
305         } else if (StringUtils.equals(GROUPER_ENABLED_DISABLED, hib3GrouploaderLog.getJobName())) {
306 
307           int records = Membership.internal_fixEnabledDisabled();
308           records += AttributeAssign.internal_fixEnabledDisabled();
309           records += ExternalSubject.internal_fixDisabled();
310 
311           hib3GrouploaderLog.setUpdateCount(records);
312 
313           hib3GrouploaderLog.setJobMessage("Ran enabled/disabled daemon, changed " + records + " records");
314           
315           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.SUCCESS.name());
316         } else if (StringUtils.equals(GROUPER_BUILTIN_MESSAGING_DAEMON, hib3GrouploaderLog.getJobName())) {
317 
318           int processedRecords = GrouperBuiltinMessagingSystem.cleanOldProcessedMessages();
319           
320           int unprocessedRecords = GrouperBuiltinMessagingSystem.cleanOldUnprocessedMessages();
321           
322           hib3GrouploaderLog.setUpdateCount(processedRecords + unprocessedRecords);
323 
324           hib3GrouploaderLog.setJobMessage("Ran builtin messaging daemon, deleted " + processedRecords + " processed records, deleted " + unprocessedRecords + " unprocessed records.");
325           
326           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.SUCCESS.name());
327 
328         } else if (StringUtils.equals(GROUPER_EXTERNAL_SUBJ_CALC_FIELDS, hib3GrouploaderLog.getJobName())) {
329 
330           int records = ExternalSubject.internal_daemonCalcFields();
331 
332           hib3GrouploaderLog.setUpdateCount(records);
333 
334           hib3GrouploaderLog.setJobMessage("Ran ext subj calc fields daemon, changed " + records + " records");
335           
336           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.SUCCESS.name());
337         } else if (StringUtils.equals(GROUPER_RULES, hib3GrouploaderLog.getJobName())) {
338 
339           int records = RuleEngine.daemon();
340           hib3GrouploaderLog.setUpdateCount(records);
341 
342           hib3GrouploaderLog.setJobMessage("Ran rules daemon, changed " + records + " records");
343           
344           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.SUCCESS.name());
345         } else if (hib3GrouploaderLog.getJobName().startsWith(GrouperLoaderType.GROUPER_GROUP_SYNC)) {
346

348           String localGroupName = hib3GrouploaderLog.getJobName().substring(GrouperLoaderType.GROUPER_GROUP_SYNC.length()+2);
349           int records = GroupSyncDaemon.syncGroup(localGroupName);
350           hib3GrouploaderLog.setUpdateCount(records);
351 
352           hib3GrouploaderLog.setJobMessage("Ran group sync daemon, changed " + records + " records");
353           
354           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.SUCCESS.name());
355         } else {
356           throw new RuntimeException("Cant find implementation for job: " + hib3GrouploaderLog.getJobName());
357         }

401       public void runJob(LoaderJobBean loaderJobBean) {
402         
403         GrouperContext.createNewDefaultContext(GrouperEngineBuiltin.LOADER, false, true);
404 
405         String groupNameOverall = loaderJobBean.getGroupNameOverall();
406         GrouperLoaderDb grouperLoaderDb = loaderJobBean.getGrouperLoaderDb();
407         String query = loaderJobBean.getQuery();
408         Hib3GrouperLoaderLog hib3GrouploaderLogOverall = loaderJobBean.getHib3GrouploaderLogOverall();
409         long startTime = loaderJobBean.getStartTime();
410         
411         GrouperSession grouperSession = loaderJobBean.getGrouperSession();
412         List<Group> andGroups = loaderJobBean.getAndGroups();
413         List<GroupType> groupTypes = GrouperUtil.nonNull(loaderJobBean.getGroupTypes());
414         String groupLikeString = loaderJobBean.getGroupLikeString();
415         String groupQuery = loaderJobBean.getGroupQuery();
416 
417         if (LOG.isDebugEnabled()) {
418           LOG.debug(groupNameOverall + ": start syncing membership");
419         }
420         
421         GrouperLoaderStatus[] statusOverall = new GrouperLoaderStatus[]{GrouperLoaderStatus.SUCCESS};
422         
423         try {
424           //get a resultset from the db
425           final GrouperLoaderResultset grouperLoaderResultsetOverall = new GrouperLoaderResultset(grouperLoaderDb, 
426               query + " order by group_name", loaderJobBean.getHib3GrouploaderLogOverall().getJobName(), 
427               loaderJobBean.getHib3GrouploaderLogOverall());
428           
429           if (LOG.isDebugEnabled()) {
430             LOG.debug(groupNameOverall + ": found " + grouperLoaderResultsetOverall + " members overall");
431           }
432 
433           hib3GrouploaderLogOverall.setMillisGetData((int)(System.currentTimeMillis()-startTime));
434 
435           grouperLoaderResultsetOverall.bulkLookupSubjects();
436           
437           //#######################################
438           //Get group metadata
439           int groupMetadataNumberOfRows = 0;
440           Map<String, String> groupNameToDisplayName = new LinkedHashMap<String, String>();
441           Map<String, String> groupNameToDescription = new LinkedHashMap<String, String>();
442           Map<String, Subject> subjectCache = new HashMap<String, Subject>();
443           Map<String, Map<Privilege, List<Subject>>> privsToAdd = new LinkedHashMap<String, Map<Privilege, List<Subject>>>();
444           Set<String> groupNamesFromGroupQuery = null;
445           if (!StringUtils.isBlank(groupQuery)) {
446             
447             groupNamesFromGroupQuery = new LinkedHashSet<String>();
448             //get a resultset from the db
449             final GrouperLoaderResultset grouperLoaderGroupsResultset = new GrouperLoaderResultset(
450                 grouperLoaderDb, groupQuery + " order by group_name", hib3GrouploaderLogOverall.getJobName(), 
451                 hib3GrouploaderLogOverall);
452             
453             groupMetadataNumberOfRows = grouperLoaderGroupsResultset.numberOfRows();
454             for (int i=0;i<groupMetadataNumberOfRows;i++) {
455               
456               String groupName = (String)grouperLoaderGroupsResultset.getCell(i, GrouperLoaderResultset.GROUP_NAME_COL, true);
457               groupNamesFromGroupQuery.add(groupName);
458               String groupDisplayName = (String)grouperLoaderGroupsResultset.getCell(i, GrouperLoaderResultset.GROUP_DISPLAY_NAME_COL, false);
459               groupNameToDisplayName.put(groupName, groupDisplayName);
460               String groupDescription = (String)grouperLoaderGroupsResultset.getCell(i, GrouperLoaderResultset.GROUP_DESCRIPTION_COL, false);
461               groupNameToDescription.put(groupName, groupDescription);
462               Map<Privilege, List<Subject>> privsToAddForGroup = new HashMap<Privilege, List<Subject>>();
463               privsToAdd.put(groupName, privsToAddForGroup);
464               {
465                 String groupViewers = (String)grouperLoaderGroupsResultset.getCell(i, GrouperLoaderResultset.GROUP_VIEWERS_COL, false);
466                 if (!StringUtils.isBlank(groupViewers)) {
467                   List<Subject> viewerSubjects = lookupSubject(subjectCache, groupViewers);
468                   privsToAddForGroup.put(AccessPrivilege.VIEW, viewerSubjects);
469                 }
470             }
471               {
472                 String groupReaders = (String)grouperLoaderGroupsResultset.getCell(i, GrouperLoaderResultset.GROUP_READERS_COL, false);
473                 if (!StringUtils.isBlank(groupReaders)) {
474                   List<Subject> readerSubjects = lookupSubject(subjectCache, groupReaders);
475                   privsToAddForGroup.put(AccessPrivilege.READ, readerSubjects);
476                 }
477               }
478               {
479                 String groupUpdaters = (String)grouperLoaderGroupsResultset.getCell(i, GrouperLoaderResultset.GROUP_UPDATERS_COL, false);
480                 if (!StringUtils.isBlank(groupUpdaters)) {
481                   List<Subject> updaterSubjects = lookupSubject(subjectCache, groupUpdaters);
482                   privsToAddForGroup.put(AccessPrivilege.UPDATE, updaterSubjects);
483                 }
484               }
485               {
486                 String groupAdmins = (String)grouperLoaderGroupsResultset.getCell(i, GrouperLoaderResultset.GROUP_ADMINS_COL, false);

1522       final boolean useThreads = GrouperLoaderConfig.retrieveConfig().propertyValueBoolean("loader.use.groupThreads", true);
1523 
1524       //see when threads are done processing
1525       List<GrouperFuture> futures = new ArrayList<GrouperFuture>();
1526 
1527       //if there were thread problems, run those again
1528       List<GrouperCallable> callablesWithProblems = new ArrayList<GrouperCallable>();
1529 
1530       int groupThreadPoolSize = GrouperLoaderConfig.retrieveConfig().propertyValueInt("loader.groupThreadPoolSize", 20);
1531 
1532       for (final String groupName : groupNamesToSync) {
1533         
1534         if (LOG.isDebugEnabled()) {
1535           LOG.debug(groupNameOverall + ": syncing membership for " + groupName + " " + count + " out of " + groupNamesToSync.size() + " groups");
1536         }
1537         
1538         GrouperCallable<Void> grouperCallable = new GrouperCallable<Void>("syncLogicForOneGroup: " + groupName) {
1539 
1540           @Override
1541           public Void callLogic() {
1542             syncGroupLogicForOneGroup(grouperLoaderResultsetOverall,
1543                 GrouperSession.staticGrouperSession(), andGroups, groupTypes, hib3GrouploaderLogOverall,
1544                 statusOverall, groupNameToDisplayName, groupNameToDescription, privsToAdd,
1545                 groupStartedMillis, membershipsInRegistry, groupName);
1546             return null;
1547           }
1548         };
1549 
1550         if (!useThreads) {
1551           grouperCallable.callLogic();
1552         } else {
1553           GrouperFuture<Void> future = GrouperUtil.executorServiceSubmit(GrouperUtil.retrieveExecutorService(), grouperCallable);
1554           futures.add(future);
1555           
1556           GrouperFuture.waitForJob(futures, groupThreadPoolSize, callablesWithProblems);
1557 
1558         }
1559 
1560         count++;
1561       }
1562 
1563       //wait for the rest
1564       GrouperFuture.waitForJob(futures, 0, callablesWithProblems);
1565 
1566       GrouperCallable.tryCallablesWithProblems(callablesWithProblems);
1567       
1568       if (LOG.isDebugEnabled()) {
1569         LOG.debug(groupNameOverall + ": done syncing membership");
1570       }
1571 
1572     } finally {
1573       hib3GrouploaderLogOverall.setMillisLoadData((int)(System.currentTimeMillis()-startTimeLoadData));
1574       statusOverall[0] = getFinalStatusAfterUnresolvables(statusOverall[0], 
1575           hib3GrouploaderLogOverall.getTotalCount(), hib3GrouploaderLogOverall.getUnresolvableSubjectCount());
1576     }
1577     
1578   }
