49  public class XmlExport {

55    private Stem stem = null;

57    /** groups and stems to export */

61    private String collectionMessage = null;

64    private File userProperties = null;

179   /**
180    * export to string
181    * @return the string of xml
182    */
183   public String exportToString() {
184     Writer writer         = new StringWriter();
185     this.exportHelper(writer);
186     return writer.toString();

188   }

190   /**
191    * export to string
192    * @param file 
193    */
194   public void exportToFile(File file) {
195     Writer writer = null;
196     try {
197       writer = new BufferedWriter( new FileWriter( file ) );
198       this.exportHelper(writer);
199     } catch (Exception e) {
200       if (e instanceof RuntimeException) {
201         throw (RuntimeException)e;
202       }
203       throw new RuntimeException("Error with export: " + e.getMessage(), e);
204     } finally {
205       GrouperUtil.closeQuietly(writer);
206     }
207     System.out.println("Exported " + file.length() 
208         + " bytes to file: " + GrouperUtil.fileCanonicalPath(file));
209   }

211   /**
212    * logger
213    */
214   private static final Log LOG = GrouperUtil.getLog(XmlExport.class);
215 
216   /**
217    * @param writer
218    */
219   private void exportHelper(Writer writer) {
220     boolean startedRootSession = false;
221     //dont assign if creating a new one
222     GrouperSession theGrouperSession = this.grouperSession;
223     try {
224       if (theGrouperSession == null) {
225         startedRootSession = true;
226         Subject subject = SubjectFinder.findRootSubject();
227         try {
228           theGrouperSession = GrouperSession.start(subject);
229         } catch (SessionException gse) {
230           throw new RuntimeException("Problem starting session: " + gse.getMessage(), gse);
231         }
232       }
233       Properties properties = new Properties();
234       if (this.userProperties != null) {
235         String userPropertiesPath = GrouperUtil.fileCanonicalPath(this.userProperties);
236         try {
237           properties = XmlUtils.internal_getUserProperties(LOG, userPropertiesPath);
238         } catch (IOException ioe) {
239           throw new RuntimeException("Problem processing user properties: " + userPropertiesPath + ", " + ioe.getMessage(), ioe);
240         }
241       }
242       XmlExporter     exporter  = new XmlExporter(theGrouperSession, properties);
243       
244       try {
245         if (this.group == null && this.stem == null && this.groupsAndStems == null) {
246           exporter.export(writer);
247         } else if (this.group != null) {
248           exporter.export(writer, group, relative, includeParent);
249         } else if (this.stem != null) {
250           if (this.includeParent) {
251             throw new RuntimeException("Cannot include parent when exporting stems");
252           }
253           exporter.export(writer, this.stem, this.relative,this.childrenOnly);
254         } else if (this.groupsAndStems != null) {
255           exporter.export(writer, this.groupsAndStems, this.collectionMessage);
256         }
257       } catch (GrouperException ge) {
258         throw new RuntimeException("Problem with export: " + ge.getMessage());
259       }
260     } finally {
261       if (startedRootSession) {
262         GrouperSession.stopQuietly(theGrouperSession);
263       }
264     }
265   }
