68    /**
69     * get a pooled data source by url and user
70     * @param url
71     * @param user
72     * @return the data source or null if not found
73     */
74    static DataSource retrieveDataSourceFromC3P0(String url, String user) {
75      for (Object dataSourceObject : C3P0Registry.getPooledDataSources()) {
76        WrapperConnectionPoolDataSource wrapperConnectionPoolDataSource = null;
77        if (dataSourceObject instanceof ComboPooledDataSource) {
78          ComboPooledDataSource comboPooledDataSource = (ComboPooledDataSource)dataSourceObject;
79          wrapperConnectionPoolDataSource = (WrapperConnectionPoolDataSource)comboPooledDataSource.getConnectionPoolDataSource();
80        } else {
81          PoolBackedDataSource poolBackedDataSource = (PoolBackedDataSource)dataSourceObject;
82          wrapperConnectionPoolDataSource = (WrapperConnectionPoolDataSource)poolBackedDataSource.getConnectionPoolDataSource();
83        }
84        DriverManagerDataSource driverManagerDataSource = (DriverManagerDataSource)wrapperConnectionPoolDataSource.getNestedDataSource();
85        String c3p0jdbcUrl = driverManagerDataSource.getJdbcUrl();
86        String c3p0user = driverManagerDataSource.getUser();
87        if (LOG.isDebugEnabled()) {
88          LOG.debug("c3p0 pool user@url: " + c3p0user + "@" + c3p0jdbcUrl);
89        }
90        if (StringUtils.equals(url, c3p0jdbcUrl) && StringUtils.equals(user, c3p0user)) {
91          return (DataSource)dataSourceObject;
92        }
93      }
94      return null;
95    }

97    /**
98     * retrieve the config name for a url and user
99     * @param url
100    * @param user
101    * @return the config name from grouper-loader.properties or null if not there
102    */
103   static String retrieveConfigName(String url, String user) {
104     
105     GrouperLoaderConfig grouperLoaderConfig = GrouperLoaderConfig.retrieveConfig();
106 
107     Pattern pattern = Pattern.compile("^db.([^.]+).user$");
108     
109     for (String propertyName : grouperLoaderConfig.propertyNames()) {
110       Matcher matcher = pattern.matcher(propertyName);
111       if (!matcher.matches()) {
112         continue;
113       }
114       String configName = matcher.group(1);
115       String configUser = grouperLoaderConfig.propertyValueString(propertyName);
116       String configUrl = grouperLoaderConfig.propertyValueString("db." + configName + ".url");
117       if (StringUtils.equals(url, configUrl) && StringUtils.equals(user, configUser)) {
118         return configName;
119       }
120     }
121     return null;
122   }

124   /**
125    * get a connection from the db
126    * @return the connection
127    */
128   public Connection connection() {
129 
130     try {
131 
132       if (!GrouperHibernateConfig.retrieveConfig().propertyValueBoolean("grouperLoader.db.connections.pool", true)) {
133         
134         //not pooling
135         Class.forName (this.driver);
136     
137         // connect
138         Connection connection = DriverManager.getConnection(this.url,this.user, this.pass);
139         return connection;
140       }
141       
142       //pooling
143   
144       //see if there is a pool
145       DataSource dataSource = retrieveDataSourceFromC3P0(this.url, this.user);
146       
147       //if so, we are good, this will cover the grouper built in database case with hibernate
148       if (dataSource != null) {
149         return dataSource.getConnection();
150       }
151   
152       //add a pool entry
153       //find name in grouper-loader.properties
154       String configName = retrieveConfigName(this.url, this.user);
155       
156       ComboPooledDataSource comboPooledDataSource = StringUtils.isBlank(configName) ? new ComboPooledDataSource() : new ComboPooledDataSource(configName);
157       
158       comboPooledDataSource.setDriverClass(this.driver); //loads the jdbc driver
159       comboPooledDataSource.setJdbcUrl(this.url);
160       comboPooledDataSource.setUser(this.user);
161       comboPooledDataSource.setPassword(this.pass);
162       
163       Integer minSize = null;
164       Integer maxSize = null;
165       Integer timeout = null;
166       Integer maxStatements = null;
167       Integer idleTestPeriod = null;
168       Integer acquireIncrement = null;
169       Boolean validate = null;
170 
171       //lets see whats in the loader config
172       //  ##optional pooling params, these will default to the grouper.hibernate(.base).properties pooling settings
173       //  #db.warehouse.c3p0.max_size = 100
174       //  #db.warehouse.c3p0.min_size = 0
175       //  ##seconds
176       //  #db.warehouse.c3p0.timeout = 100
177       //  #db.warehouse.c3p0.max_statements = 0
178       //  #db.warehouse.c3p0.idle_test_period = 100
179       //  #db.warehouse.c3p0.acquire_increment = 1
180       //  #db.warehouse.c3p0.validate = false
181       if (!StringUtils.isBlank(configName)) {
182         minSize = GrouperLoaderConfig.retrieveConfig().propertyValueInt("db." + configName + ".c3p0.min_size");
183         maxSize = GrouperLoaderConfig.retrieveConfig().propertyValueInt("db." + configName + ".c3p0.max_size");
184         timeout = GrouperLoaderConfig.retrieveConfig().propertyValueInt("db." + configName + ".c3p0.timeout");
185         maxStatements = GrouperLoaderConfig.retrieveConfig().propertyValueInt("db." + configName + ".c3p0.max_statements");
186         idleTestPeriod = GrouperLoaderConfig.retrieveConfig().propertyValueInt("db." + configName + ".c3p0.idle_test_period");
187         acquireIncrement = GrouperLoaderConfig.retrieveConfig().propertyValueInt("db." + configName + ".c3p0.acquire_increment");
188         validate = GrouperHibernateConfig.retrieveConfig().propertyValueBoolean("db." + configName + ".c3p0.validate");
189       }
190 
191       //if null, then default to grouper.hibernate.properties
192       //  # Use c3p0 connection pooling (since dbcp not supported in hibernate anymore)
193       //  # http://www.hibernate.org/214.html, http://www.hibernate.org/hib_docs/reference/en/html/session-configuration.html
194       //  hibernate.c3p0.max_size = 100
195       //  hibernate.c3p0.min_size = 0
196       //  #seconds
197       //  hibernate.c3p0.timeout = 100
198       //  hibernate.c3p0.max_statements = 0
199       //  hibernate.c3p0.idle_test_period = 100
200       //  hibernate.c3p0.acquire_increment = 1
201       //  hibernate.c3p0.validate = false
202       if (minSize == null) {
203         minSize = GrouperHibernateConfig.retrieveConfig().propertyValueInt("hibernate.c3p0.min_size");
204       }
205       if (maxSize == null) {
206         maxSize = GrouperHibernateConfig.retrieveConfig().propertyValueInt("hibernate.c3p0.max_size");
207       }
208       if (timeout == null) {
209         timeout = GrouperHibernateConfig.retrieveConfig().propertyValueInt("hibernate.c3p0.timeout");
210       }
211       if (maxStatements == null) {
212         maxStatements = GrouperHibernateConfig.retrieveConfig().propertyValueInt("hibernate.c3p0.max_statements");
213       }
214       if (idleTestPeriod == null) {
215         idleTestPeriod = GrouperHibernateConfig.retrieveConfig().propertyValueInt("hibernate.c3p0.idle_test_period");
216       }
217       if (acquireIncrement == null) {
218         acquireIncrement = GrouperHibernateConfig.retrieveConfig().propertyValueInt("hibernate.c3p0.acquire_increment");
219       }
220       if (validate == null) {
221         validate = GrouperHibernateConfig.retrieveConfig().propertyValueBoolean("hibernate.c3p0.validate");
222       }
223 
224       //if set, set them, otherwise defaults
225       if (minSize != null) {
226         comboPooledDataSource.setMinPoolSize(minSize);
227       }
228       if (maxSize != null) {
229         comboPooledDataSource.setMaxPoolSize(maxSize);
230       }
231       if (timeout != null) {
232         comboPooledDataSource.setMaxIdleTime(timeout);
233       }
234       if (maxStatements != null) {
235         comboPooledDataSource.setMaxStatements(maxStatements);
236       }
237       if (idleTestPeriod != null) {
238         comboPooledDataSource.setIdleConnectionTestPeriod(idleTestPeriod);
239       }
240       if (acquireIncrement != null) {
241         comboPooledDataSource.setAcquireIncrement(acquireIncrement);
242       }
243       if (validate != null) {
244         //i assume this is the setting... hmmm
245         comboPooledDataSource.setTestConnectionOnCheckout(validate);
246       }
247 
248       //is it there now????
249       dataSource = retrieveDataSourceFromC3P0(this.url, this.user);
250       
251       return dataSource.getConnection();
252       
253     } catch (Exception e) {
254       throw new RuntimeException("Problems with db: " + this, e);
255     }
256     
257   }
