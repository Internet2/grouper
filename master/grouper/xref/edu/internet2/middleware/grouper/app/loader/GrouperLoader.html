281   /**
282    * lazy load (and start the scheduler) the scheduler factory
283    * @return the scheduler factory
284    */
285   public static SchedulerFactory schedulerFactory() {
286     if (schedulerFactory == null) {
287       
288       Properties props = new Properties();
289       for (String key : GrouperLoaderConfig.retrieveConfig().propertyNames()) {
290         if (key.startsWith("org.quartz.")) {
291           String value = GrouperLoaderConfig.retrieveConfig().propertyValueString(key);
292           if (value == null) {
293             value = "";
294           }
295           props.put(key, value);
296         }
297       }
298 
299       String url = GrouperHibernateConfig.retrieveConfig().propertyValueString("hibernate.connection.url");
300       
301       if (GrouperUtil.isEmpty((String)props.get("org.quartz.dataSource.myDS.driver"))) {
302         String driver = GrouperHibernateConfig.retrieveConfig().propertyValueString("hibernate.connection.driver_class");
303         driver = GrouperDdlUtils.convertUrlToDriverClassIfNeeded(url, driver);
304         props.put("org.quartz.dataSource.myDS.driver", driver);
305       }
306       
307       if (GrouperUtil.isEmpty((String)props.get("org.quartz.jobStore.driverDelegateClass"))) {
308         String driver = GrouperDdlUtils.convertUrlToQuartzDriverDelegateClassIfNeeded(url, null);
309         props.put("org.quartz.jobStore.driverDelegateClass", driver);
310       }
311       
312       if (GrouperUtil.isEmpty((String)props.get("org.quartz.dataSource.myDS.URL"))) {
313         props.put("org.quartz.dataSource.myDS.URL", url);
314       }
315       
316       if (GrouperUtil.isEmpty((String)props.get("org.quartz.dataSource.myDS.user"))) {
317         props.put("org.quartz.dataSource.myDS.user", GrouperHibernateConfig.retrieveConfig().propertyValueString("hibernate.connection.username"));
318       }
319       
320       if (GrouperUtil.isEmpty((String)props.get("org.quartz.dataSource.myDS.password"))) {
321         String pass = GrouperHibernateConfig.retrieveConfig().propertyValueString("hibernate.connection.password");
322         pass = Morph.decryptIfFile(pass);
323         props.put("org.quartz.dataSource.myDS.password", pass);
324       }
325       
326       try {
327         schedulerFactory = new StdSchedulerFactory(props);
328       } catch (SchedulerException se) {
329         throw new RuntimeException(se);
330       }
331     }
332     return schedulerFactory;
333   }

356   /**
357    * schedule maintenance job for moving records from change log to change log temp
358    */
359   public static void scheduleChangeLogTempToChangeLogJob() {
360 
361     String cronString = null;
362 
363     //this is a medium priority job
364     int priority = 5;
365 
366     //schedule the log delete job
367     try {
368       Scheduler scheduler = GrouperLoader.schedulerFactory().getScheduler();
369       String triggerName = "triggerChangeLog_grouperChangeLogTempToChangeLog";
370       
371       if (!GrouperLoaderConfig.retrieveConfig().propertyValueBoolean("changeLog.changeLogTempToChangeLog.enable", false)) {
372         LOG.warn("grouper-loader.properties key: changeLog.changeLogTempToChangeLog.enable is not " +
373           "filled in or false so the change log temp to change log daemon will not run");
374         scheduler.unscheduleJob(TriggerKey.triggerKey(triggerName));
375         return;
376       }
377       
378       cronString = GrouperLoaderConfig.retrieveConfig().propertyValueString("changeLog.changeLogTempToChangeLog.quartz.cron");
379 
380       if (StringUtils.isBlank(cronString)) {
381         cronString = "50 * * * * ?";
382         
383       }
384       
385       //at this point we have all the attributes and we know the required ones are there, and logged when 
386       //forbidden ones are there
387 
388       //the name of the job must be unique, so use the group name since one job per group (at this point)
389       JobDetail jobDetail = JobBuilder.newJob(GrouperLoaderJob.class)
390         .withIdentity(GrouperLoaderType.GROUPER_CHANGE_LOG_TEMP_TO_CHANGE_LOG)
391         .build();
392 
393       //schedule this job
394       GrouperLoaderScheduleType grouperLoaderScheduleType = GrouperLoaderScheduleType.CRON;
395 
396       Trigger trigger = grouperLoaderScheduleType.createTrigger(triggerName, priority, cronString, null);
397 
398       scheduleJobIfNeeded(jobDetail, trigger);
399 
400 
401     } catch (Exception e) {
402       String errorMessage = "Could not schedule job: '" + GrouperLoaderType.GROUPER_CHANGE_LOG_TEMP_TO_CHANGE_LOG + "'";
403       LOG.error(errorMessage, e);
404       errorMessage += "\n" + ExceptionUtils.getFullStackTrace(e);
405       try {
406         //lets enter a log entry so it shows up as error in the db
407         Hib3GrouperLoaderLog hib3GrouploaderLog = new Hib3GrouperLoaderLog();
408         hib3GrouploaderLog.setHost(GrouperUtil.hostname());
409         hib3GrouploaderLog.setJobMessage(errorMessage);
410         hib3GrouploaderLog.setJobName(GrouperLoaderType.GROUPER_CHANGE_LOG_TEMP_TO_CHANGE_LOG);
411         hib3GrouploaderLog.setJobSchedulePriority(priority);
412         hib3GrouploaderLog.setJobScheduleQuartzCron(cronString);
413         hib3GrouploaderLog.setJobScheduleType(GrouperLoaderScheduleType.CRON.name());
414         hib3GrouploaderLog.setJobType(GrouperLoaderType.CHANGE_LOG.name());
415         hib3GrouploaderLog.setStatus(GrouperLoaderStatus.CONFIG_ERROR.name());
416         hib3GrouploaderLog.store();
417         
418       } catch (Exception e2) {
419         LOG.error("Problem logging to loader db log", e2);
420       }
421     }

425   /**
426    * schedule change log consumer jobs
427    */
428   public static void scheduleChangeLogConsumers() {
429 
430     //changeLog.consumer.ldappc.class = 
431     //changeLog.consumer.ldappc.quartz.cron
432     
433     //make sure sequences are ok
434     Map<String, String> consumerMap = GrouperLoaderConfig.retrieveConfig().propertiesMap( 
435         GrouperCheckConfig.grouperLoaderConsumerPattern);
436     
437     Set<String> changeLogJobNames = new HashSet<String>();
438     
439     int index = 0;
440     
441     while (consumerMap.size() > 0) {
442       
443       //get one
444       String consumerKey = consumerMap.keySet().iterator().next();
445       //get the database name
446       Matcher matcher = GrouperCheckConfig.grouperLoaderConsumerPattern.matcher(consumerKey);
447       matcher.matches();
448       String consumerName = matcher.group(1);
449       boolean missingOne = false;
450       //now find all 4 required keys
451       String classKey = "changeLog.consumer." + consumerName + ".class";
452       if (!consumerMap.containsKey(classKey)) {
453         String error = "cannot find grouper-loader.properties key: " + classKey; 
454         System.out.println("Grouper error: " + error);
455         LOG.error(error);
456         missingOne = true;
457       }
458       String cronKey = "changeLog.consumer." + consumerName + ".quartzCron";
459 
460       //check the classname
461       Class<?> theClass = null;
462       String className = consumerMap.get(classKey);
463       String cronString = consumerMap.get(cronKey);
464       
465       String jobName = GrouperLoaderType.GROUPER_CHANGE_LOG_CONSUMER_PREFIX + consumerName;
466       changeLogJobNames.add(jobName);
467       
468       //this is a medium priority job
469       int priority = 5;
470 
471       try {
472         if (missingOne) {
473           throw new RuntimeException("Cant find config param" );
474         }
475         
476         theClass = GrouperUtil.forName(className);
477         if (!ChangeLogConsumerBase.class.isAssignableFrom(theClass)) {
478           throw new RuntimeException("not a subclass of ChangeLogConsumerBase");
479         }
480 
481         //default to every minute on the minute, though add a couple of seconds for each one
482         //        if (StringUtils.isBlank(cronString) || StringUtils.equals("0 * * * * ?", cronString)) {
483         //        dont change the crons that are explicitly set... only blank ones
484         if (StringUtils.isBlank(cronString)) {
485           cronString = ((index * 2) % 60) + " * * * * ?";
486         }
487         //at this point we have all the attributes and we know the required ones are there, and logged when 
488         //forbidden ones are there
489 
490         //the name of the job must be unique, so use the group name since one job per group (at this point)
491         JobDetail jobDetail = JobBuilder.newJob(GrouperLoaderJob.class)
492           .withIdentity(jobName)
493           .build();
494         
495         //schedule this job
496         GrouperLoaderScheduleType grouperLoaderScheduleType = GrouperLoaderScheduleType.CRON;
497 
498         Trigger trigger = grouperLoaderScheduleType.createTrigger("triggerChangeLog_" + jobName, priority, cronString, null);
499 
500         scheduleJobIfNeeded(jobDetail, trigger);
501 
502       } catch (Exception e) {
503 
504         String errorMessage = "Could not schedule job: '" + jobName + "'";
505         LOG.error(errorMessage, e);
506         errorMessage += "\n" + ExceptionUtils.getFullStackTrace(e);
507         try {
508           //lets enter a log entry so it shows up as error in the db
509           Hib3GrouperLoaderLog hib3GrouploaderLog = new Hib3GrouperLoaderLog();
510           hib3GrouploaderLog.setHost(GrouperUtil.hostname());
511           hib3GrouploaderLog.setJobMessage(errorMessage);
512           hib3GrouploaderLog.setJobName(jobName);
513           hib3GrouploaderLog.setJobSchedulePriority(priority);
514           hib3GrouploaderLog.setJobScheduleQuartzCron(cronString);
515           hib3GrouploaderLog.setJobScheduleType(GrouperLoaderScheduleType.CRON.name());
516           hib3GrouploaderLog.setJobType(GrouperLoaderType.CHANGE_LOG.name());
517           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.CONFIG_ERROR.name());
518           hib3GrouploaderLog.store();
519           
520         } catch (Exception e2) {
521           LOG.error("Problem logging to loader db log", e2);
522         }
523 
524       }
525       
526       consumerMap.remove(classKey);
527       consumerMap.remove(cronKey);
528       index++;
529     }
530     
531     // check to see if anything should be unscheduled.
532     try {
533       Scheduler scheduler = GrouperLoader.schedulerFactory().getScheduler();
534 
535       for (JobKey jobKey : scheduler.getJobKeys(GroupMatcher.jobGroupEquals("DEFAULT"))) {
536         
537         String jobName = jobKey.getName();
538         
539         if (jobName.startsWith(GrouperLoaderType.GROUPER_CHANGE_LOG_CONSUMER_PREFIX) && !changeLogJobNames.contains(jobName)) {
540           try {
541             String triggerName = "triggerChangeLog_" + jobName;
542             scheduler.unscheduleJob(TriggerKey.triggerKey(triggerName));
543           } catch (Exception e) {
544             String errorMessage = "Could not unschedule job: '" + jobName + "'";
545             LOG.error(errorMessage, e);
546             errorMessage += "\n" + ExceptionUtils.getFullStackTrace(e);
547             try {
548               //lets enter a log entry so it shows up as error in the db
549               Hib3GrouperLoaderLog hib3GrouploaderLog = new Hib3GrouperLoaderLog();
550               hib3GrouploaderLog.setHost(GrouperUtil.hostname());
551               hib3GrouploaderLog.setJobMessage(errorMessage);
552               hib3GrouploaderLog.setJobName(jobName);
553               hib3GrouploaderLog.setJobType(GrouperLoaderType.CHANGE_LOG.name());
554               hib3GrouploaderLog.setStatus(GrouperLoaderStatus.CONFIG_ERROR.name());
555               hib3GrouploaderLog.store();
556               
557             } catch (Exception e2) {
558               LOG.error("Problem logging to loader db log", e2);
559             }
560           }
561         }
562       }
563     } catch (Exception e) {
564       
565       String errorMessage = "Could not query change log jobs to see if any should be unscheduled.";
566       LOG.error(errorMessage, e);
567       errorMessage += "\n" + ExceptionUtils.getFullStackTrace(e);
568       try {
569         //lets enter a log entry so it shows up as error in the db
570         Hib3GrouperLoaderLog hib3GrouploaderLog = new Hib3GrouperLoaderLog();
571         hib3GrouploaderLog.setHost(GrouperUtil.hostname());
572         hib3GrouploaderLog.setJobMessage(errorMessage);
573         hib3GrouploaderLog.setJobType(GrouperLoaderType.CHANGE_LOG.name());
574         hib3GrouploaderLog.setStatus(GrouperLoaderStatus.CONFIG_ERROR.name());
575         hib3GrouploaderLog.store();
576         
577       } catch (Exception e2) {
578         LOG.error("Problem logging to loader db log", e2);
579       }
580     }
581   }

2057   /**
2058    * Schedule job if new or something has changed
2059    * @param jobDetail
2060    * @param trigger
2061    * @return true if needed an update
2062    * @throws SchedulerException 
2063    */
2064   public static boolean scheduleJobIfNeeded(JobDetail jobDetail, Trigger trigger) throws SchedulerException {
2065     Scheduler scheduler = GrouperLoader.schedulerFactory().getScheduler();
2066     
2067     boolean scheduleJob = false;
2068     
2069     JobDetail oldJobDetail = scheduler.getJobDetail(jobDetail.getKey());
2070     Trigger oldTrigger = scheduler.getTrigger(trigger.getKey());
2071 
2072     if (oldJobDetail == null || oldTrigger == null) {
2073       scheduleJob = true;
2074     }
2075     
2076     if (!scheduleJob && oldJobDetail.getJobClass() != jobDetail.getJobClass()) {
2077       scheduleJob = true;
2078     }
2079     
2080     if (!scheduleJob && oldTrigger.getPriority() != trigger.getPriority()) {
2081       scheduleJob = true;
2082     }
2083     
2084     if (!scheduleJob) {
2085 
2086       if (oldTrigger instanceof SimpleTrigger && trigger instanceof SimpleTrigger) {
2087         if (((SimpleTrigger)oldTrigger).getRepeatInterval() != ((SimpleTrigger)trigger).getRepeatInterval()) {
2088           scheduleJob = true;
2089         }
2090       } else if (oldTrigger instanceof CronTrigger && trigger instanceof CronTrigger) {
2091         if (!((CronTrigger)oldTrigger).getCronExpression().equals(((CronTrigger)trigger).getCronExpression())) {
2092           scheduleJob = true;
2093         }
2094       } else {
2095         scheduleJob = true;
2096       }
2097     }
2098     
2099     if (!scheduleJob) {
2100       if (oldJobDetail.getJobDataMap() == null && jobDetail.getJobDataMap() == null) {
2101         // ok
2102       } else if (oldJobDetail.getJobDataMap() == null || jobDetail.getJobDataMap() == null) {
2103         scheduleJob = true;
2104       } else if (!oldJobDetail.getJobDataMap().equals(jobDetail.getJobDataMap())) {
2105         scheduleJob = true;
2106       }
2107     }
2108     
2109     if (scheduleJob) {
2110       // apparently running this for a job that already exists when bringing the daemon up on a different host may fire some
2111       // jobs concurrently with jobs on another host.  so only updating if there's something to update.
2112       scheduler.scheduleJob(jobDetail, GrouperUtil.toSet(trigger), true);
2113       LOG.info("Scheduled quartz job: " + jobDetail.getKey().getName());
2114       return true;
2115     }
2116     
2117     return false;
2118   }
