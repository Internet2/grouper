23  package edu.internet2.middleware.grouper.app.gsh;
24  import bsh.CallStack;
25  import bsh.Interpreter;
26  import edu.internet2.middleware.grouper.Group;
27  import edu.internet2.middleware.grouper.GroupFinder;
28  import edu.internet2.middleware.grouper.GrouperSession;
29  import edu.internet2.middleware.grouper.exception.GroupDeleteException;
30  import edu.internet2.middleware.grouper.exception.GroupNotFoundException;
31  import edu.internet2.middleware.grouper.exception.InsufficientPrivilegeException;
32  
33  /**
34   * Delete a group.
35   * <p/>
36   * @author  blair christensen.
37   * @version $Id: delGroup.java,v 1.3 2009-03-15 06:37:23 mchyzer Exp $
38   * @since   0.0.1
39   */
40  public class delGroup {
41  
42    // PUBLIC CLASS METHODS //
43  
44    /**
45     * Delete a  stem.
46     * <p/>
47     * @param   i     BeanShell interpreter.
48     * @param   stack BeanShell call stack.
49     * @param   name  <i>name</i> of {@link Group} to delete.
50     * @return  True if {@link Group} deleted.
51     * @throws  GrouperShellException
52     * @since   0.0.1
53     */
54    public static boolean invoke(Interpreter i, CallStack stack, String name) 
55      throws  GrouperShellException
56    {
57      GrouperShell.setOurCommand(i, true);
58      try {
59        GrouperSession  s = GrouperShell.getSession(i);
60        Group           g = GroupFinder.findByName(s, name, true);
61        g.delete();
62        return true;
63      }
64      catch (GroupDeleteException eGD)            {
65        GrouperShell.error(i, eGD);
66      }
67      catch (GroupNotFoundException eGNF)         {
68        GrouperShell.error(i, eGNF);
69      }
70      catch (InsufficientPrivilegeException eIP)  {
71        GrouperShell.error(i, eIP);
72      }
73      return false;
74    } // public static boolean invoke(i, stack, name)
75  
76  } // public class delGroup
