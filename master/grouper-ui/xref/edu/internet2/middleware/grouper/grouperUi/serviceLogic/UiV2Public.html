39    /** logger */
40    @SuppressWarnings("unused")
41    private static final Log LOG = LogFactory.getLog(UiV2Public.class);

43    /**
44     * index page of application, or public operation used for posts.  since we only have one
45     * request path that is a servlet in the public space, then this will be
46     * the gatekeeper for all public operations either the index or ajax operations
47     * @param request
48     * @param response
49     */
50    public void postIndex(HttpServletRequest request, HttpServletResponse response) {
51  
52      //since this is the only public servlet, tunnel the operation through
53      String function = request.getParameter("function");
54      
55      //dont use reflection here for security reasons
56      if (StringUtils.equals(function, UiV2Public.class.getSimpleName() + ".error")) {
57  
58        error(request, response);
59  
60      } else {
61        throw new RuntimeException("Invalid function: " + function);
62      }
63  
64  
65    }
66  
67    /**
68     * index page of application, or public operation.  since we only have one
69     * request path that is a servlet in the public space, then this will be
70     * the gatekeeper for all public operations either the index or ajax operations
71     * @param request
72     * @param response
73     */
74    public void index(HttpServletRequest request, HttpServletResponse response) {
75      
76      //initialize the bean
77      GrouperRequestContainer.retrieveFromRequestOrCreate();
78  
79      //if there is an operation, then we need the main page
80      String operation = request.getParameter("operation");
81      
82      if (!StringUtils.isBlank(operation)) {
83        //if this is an error, it could be ajax, so ajax will need to redirect to right place.
84        String url = request.getRequestURL().toString() + "?" + request.getQueryString();
85        
86        //strip off the http / https
87        url = url.substring(8);
88        //go to the next slash
89        url = url.substring(url.indexOf('/')+1);
90        //the next slash is the end of the contextroot
91        url = url.substring(url.indexOf('/')+1);
92        url = "../../" + url;
93        
94        response.addHeader("X-Grouper-path", GrouperUtil.escapeUrlEncode(url));
95    
96        //just show a jsp
97        showJsp("/WEB-INF/grouperUi2/public/index.jsp");
98    
99        throw new ControllerDone();
100     }  
101   }
102 
103   /**
104    * ajax operation if there is an error
105    * @param request
106    * @param response
107    */
108   public void error(HttpServletRequest request, HttpServletResponse response) {
109     String code = request.getParameter("code");
110 
111     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
112 
113     guiResponseJs.addAction(GuiScreenAction.newInnerHtml("#preMessaging", 
114         TextContainer.retrieveFromRequest().getText().get("guiErrorHeader")));
115 
116     guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#grouperMainContentDivId", 
117         "/WEB-INF/grouperUi2/public/startOver.jsp"));
118 
119     guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
120         TextContainer.retrieveFromRequest().getText().get("errorCode_" + code)));
121 
122   }
