84    /**
85     * filter groups to pick one to edit
86     * @param httpServletRequest
87     * @param httpServletResponse
88     */
89    public void filterGroups(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
90      final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
91      
92      
93      GrouperSession grouperSession = null;
94    
95      String searchTerm = httpServletRequest.getParameter("mask");
96    
97      try {
98        grouperSession = GrouperSession.start(loggedInSubject);
99        
100       
101       Set<Group> groups = null;
102       
103       StringBuilder xmlBuilder = new StringBuilder(GrouperUiUtils.DHTMLX_OPTIONS_START);
104   
105       QueryOptions queryOptions = null;
106       
107       if (StringUtils.defaultString(searchTerm).length() < 2) {
108         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, "", 
109             GrouperUiUtils.message("simpleMembershipUpdate.errorNotEnoughGroupChars", false), "bullet_error.png");
110       } else {
111         queryOptions = new QueryOptions().paging(GrouperUiConfig.retrieveConfig().propertyValueInt("simpleMembershipUpdate.groupComboboxResultSize", 200), 1, true).sortAsc("theGroup.displayNameDb");
112         groups = GrouperDAOFactory.getFactory().getGroup().getAllGroupsSecure("%" + searchTerm + "%", grouperSession, loggedInSubject, 
113             GrouperUtil.toSet(AccessPrivilege.ADMIN, AccessPrivilege.UPDATE), queryOptions, TypeOfGroup.GROUP_OR_ROLE_SET);
114         
115         if (GrouperUtil.length(groups) == 0) {
116           GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, "", 
117               GrouperUiUtils.message("simpleMembershipUpdate.errorNoGroupsFound", false), "bullet_error.png");
118         }
119       }
120       
121       for (Group group : GrouperUtil.nonNull(groups)) {
122   
123         String value = group.getUuid();
124         String label = GrouperUiUtils.escapeHtml(group.getDisplayName(), true);
125         String imageName = null;
126         if (group.getTypeOfGroup() == TypeOfGroup.role) {
127           imageName = GrouperUiUtils.imageFromSubjectSource("g:rsa");
128         } else {
129           imageName = GrouperUiUtils.imageFromSubjectSource("g:gsa");
130         }
131   
132         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, value, label, imageName);
133       }
134   
135       //add one more for more options if we didnt get them all
136       if (queryOptions != null && queryOptions.getCount() != null 
137           && groups != null && queryOptions.getCount() > groups.size()) {
138         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, "", 
139             GrouperUiUtils.message("simpleMembershipUpdate.errorTooManyGroups", false), "bullet_error.png");
140       }
141       
142       
143       xmlBuilder.append(GrouperUiUtils.DHTMLX_OPTIONS_END);
144       
145       GrouperUiUtils.printToScreen(xmlBuilder.toString(), HttpContentType.TEXT_XML, false, false);
146   
147     } catch (Exception se) {
148       LOG.error("Error searching for groups: '" + GrouperUiUtils.escapeHtml(searchTerm, true) + "', " + se.getMessage(), se);
149       
150       //dont rethrow or the control will get confused
151       StringBuilder xmlBuilder = new StringBuilder(GrouperUiUtils.DHTMLX_OPTIONS_START);
152       GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, null, 
153           GrouperUiUtils.escapeHtml("Error searching for groups: " + searchTerm + ", " + se.getMessage(), true), null);
154       xmlBuilder.append(GrouperUiUtils.DHTMLX_OPTIONS_END);
155       GrouperUiUtils.printToScreen(xmlBuilder.toString(), HttpContentType.TEXT_XML, false, false);
156     } finally {
157       GrouperSession.stopQuietly(grouperSession); 
158     }
159   
160     //dont print the regular JSON
161     throw new ControllerDone();
162   
163   }

165   /**
166    * when text is typed into the combobox, this activates the list
167    * @param httpServletRequest
168    * @param httpServletResponse
169    */
170   public void filterMembers(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
171   
172     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
173     SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
174     GrouperSession grouperSession = null;
175   
176     String searchTerm = httpServletRequest.getParameter("mask");
177     Group group = null;
178     try {
179       grouperSession = GrouperSession.start(loggedInSubject);
180       
181       group = new SimpleMembershipUpdate().retrieveGroup(grouperSession);
182             
183       Set<Subject> subjects = null;
184       
185       StringBuilder xmlBuilder = new StringBuilder(GrouperUiUtils.DHTMLX_OPTIONS_START);
186       boolean notEnoughChars = false;
187       //minimum input length
188       boolean[] tooManyResults = new boolean[]{false};
189       if (StringUtils.defaultString(searchTerm).length() < 3) {
190         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, "", 
191             simpleMembershipUpdateContainer.getText().getErrorNotEnoughFilterChars(), null);
192         notEnoughChars = true;
193       } else {
194         GuiPaging guiPaging = new GuiPaging();
195         guiPaging.setPageNumber(1);
196         int maxSubjectsDropDown = GrouperUiConfig.retrieveConfig().propertyValueInt("simpleMembershipUpdate.subjectComboboxResultSize", 50);
197         guiPaging.setPageSize(maxSubjectsDropDown);
198         Set<Member> members = retrieveMembersFilter(guiPaging, group, searchTerm, tooManyResults);
199         subjects = GrouperUiUtils.convertMembersToSubject(members);
200       }
201       
202       //convert to XML for DHTMLX
203       for (Subject subject : GrouperUtil.nonNull(subjects)) {
204         String value = GrouperUiUtils.convertSubjectToValue(subject);
205   
206         String imageName = GrouperUiUtils.imageFromSubjectSource(subject.getSource().getId());
207         String label = GrouperUiUtils.escapeHtml(GrouperUiUtils.convertSubjectToLabelConfigured(subject), true);
208   
209         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, value, label, imageName);
210       }
211       int maxSubjectsSort = GrouperUiConfig.retrieveConfig().propertyValueInt("comparator.sort.limit", 200);
212       
213       //this might not be correct, but probably is
214       if (tooManyResults[0] || (!notEnoughChars && GrouperUtil.length(subjects) == maxSubjectsSort)) {
215         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, null, simpleMembershipUpdateContainer.getText().getErrorUserSearchTooManyResults(), 
216             "bullet_error.png");
217       } else if (!notEnoughChars && GrouperUtil.length(subjects) == 0) {
218         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, "", simpleMembershipUpdateContainer.getText().getErrorUserSearchNoResults(), "bullet_error.png");
219       }
220   
221       xmlBuilder.append(GrouperUiUtils.DHTMLX_OPTIONS_END);
222       
223       GrouperUiUtils.printToScreen(xmlBuilder.toString(), HttpContentType.TEXT_XML, false, false);
224   
225     } catch (Exception se) {
226       LOG.error("Error searching for members: '" + searchTerm + "', " + se.getMessage(), se);
227 
228       //dont rethrow or the control will get confused
229       StringBuilder xmlBuilder = new StringBuilder(GrouperUiUtils.DHTMLX_OPTIONS_START);
230       GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, null, 
231           GrouperUiUtils.escapeHtml("Error searching for members: " + searchTerm + ", " + se.getMessage(), true), null);
232       xmlBuilder.append(GrouperUiUtils.DHTMLX_OPTIONS_END);
233       GrouperUiUtils.printToScreen(xmlBuilder.toString(), HttpContentType.TEXT_XML, false, false);
234     } finally {
235       GrouperSession.stopQuietly(grouperSession); 
236     }
237   
238     //dont print the regular JSON
239     throw new ControllerDone();
240   
241   }

243   /**
244    * called in the combobox to list the users
245    * @param httpServletRequest
246    * @param httpServletResponse
247    */
248   @SuppressWarnings("unchecked")
249   public void filterUsers(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
250     
251     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
252     
253     SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
254     
255     GrouperSession grouperSession = null;
256   
257     String searchTerm = httpServletRequest.getParameter("mask");
258     
259     try {
260       grouperSession = GrouperSession.start(loggedInSubject);
261       
262       Group group = new SimpleMembershipUpdate().retrieveGroup(grouperSession);
263       String stemName = group.getParentStemName();
264       
265       Set<Subject> subjects = null;
266       
267       StringBuilder xmlBuilder = new StringBuilder(GrouperUiUtils.DHTMLX_OPTIONS_START);
268       QueryPaging queryPaging = null;
269       
270       //minimum input length
271       boolean tooManyResults = false;
272       if (StringUtils.defaultString(searchTerm).length() < 2) {
273         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, "", 
274             simpleMembershipUpdateContainer.getText().getErrorNotEnoughSubjectChars(), null);
275       } else {
276         try {
277           
278           final String requireGroup = simpleMembershipUpdateContainer.configValue(
279               "simpleMembershipUpdate.subjectSearchRequireGroup", false);
280           
281           final String requireSources = simpleMembershipUpdateContainer.configValue(
282               "simpleMembershipUpdate.subjectSearchRequireSources", false);
283           
284           if (!StringUtils.isBlank(requireSources)) {
285             Set<Source> sources = GrouperUtil.convertSources(requireSources);
286             subjects = SubjectFinder.findPage(searchTerm, sources).getResults();
287           } else {
288             subjects = SubjectFinder.findPageInStem(stemName, searchTerm).getResults();            
289           }
290           
291           
292           if (!StringUtils.isBlank(requireGroup)) {
293 
294             final Set<Subject> SUBJECTS = subjects;
295             
296             subjects = (Set<Subject>)GrouperSession.callbackGrouperSession(
297                 grouperSession.internal_getRootSession(), new GrouperSessionHandler() {
298               
299               @Override
300               public Object callback(GrouperSession rootGrouperSession) throws GrouperSessionException {
301                 
302                 Group groupFilter = GroupFinder.findByName(rootGrouperSession, requireGroup, true);
303                 
304                 return SubjectFinder.findBySubjectsInGroup(rootGrouperSession, SUBJECTS, groupFilter, Group.getDefaultList(), null);
305               }
306             });
307             
308           }
309           
310           String maxSubjectsDropDownString = GrouperUiConfig.retrieveConfig().propertyValueString("simpleMembershipUpdate.subjectComboboxResultSize");
311           int maxSubjectsDropDown = GrouperUtil.intValue(maxSubjectsDropDownString, 250);
312     
313           queryPaging = new QueryPaging(maxSubjectsDropDown, 1, true);
314         
315           //sort and page the results
316           subjects = GrouperUiUtils.subjectsSortedPaged(subjects, queryPaging, searchTerm);
317         } catch (SubjectTooManyResults stmr) {
318           tooManyResults = true;
319         }
320       }
321       
322       //convert to XML for DHTMLX
323       for (Subject subject : GrouperUtil.nonNull(subjects)) {
324         String value = GrouperUiUtils.convertSubjectToValue(subject);
325   
326         String imageName = GrouperUiUtils.imageFromSubjectSource(subject.getSource().getId());
327         String label = GrouperUiUtils.escapeHtml(GrouperUiUtils.convertSubjectToLabelConfigured(subject), true);
328   
329         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, value, label, imageName);
330       }
331   
332       //maybe add one more if we hit the limit
333       if (tooManyResults || queryPaging != null && GrouperUtil.length(subjects) < queryPaging.getTotalRecordCount()) {
334         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, null, 
335             simpleMembershipUpdateContainer.getText().getErrorUserSearchTooManyResults(), 
336             "bullet_error.png");
337       } else if (GrouperUtil.length(subjects) == 0) {
338         GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, "", 
339             simpleMembershipUpdateContainer.getText().getErrorUserSearchNoResults(), "bullet_error.png");
340       }
341   
342       xmlBuilder.append(GrouperUiUtils.DHTMLX_OPTIONS_END);
343       
344       GrouperUiUtils.printToScreen(xmlBuilder.toString(), HttpContentType.TEXT_XML, false, false);
345   
346     } catch (Exception se) {
347       LOG.error("Error searching for members: '" + searchTerm + "', " + se.getMessage(), se);
348 
349       //dont rethrow or the control will get confused
350       StringBuilder xmlBuilder = new StringBuilder(GrouperUiUtils.DHTMLX_OPTIONS_START);
351       GrouperUiUtils.dhtmlxOptionAppend(xmlBuilder, null, 
352           GrouperUiUtils.escapeHtml("Error searching for members: " + searchTerm + ", " + se.getMessage(), true), null);
353       xmlBuilder.append(GrouperUiUtils.DHTMLX_OPTIONS_END);
354       GrouperUiUtils.printToScreen(xmlBuilder.toString(), HttpContentType.TEXT_XML, false, false);
355     } finally {
356       GrouperSession.stopQuietly(grouperSession); 
357     }
358   
359     //dont print the regular JSON
360     throw new ControllerDone();
361   
362   }

364   /**
365    * @param guiPaging
366    * @param group
367    * @param filterString filter by this string (could be sourceId__subjectId
368    * @param tooManyResultsArray array of size one to pass back too many results
369    * @return the set of members
370    * @throws SchemaException
371    */
372   @SuppressWarnings("unchecked")
373   Set<Member> retrieveMembersFilter(GuiPaging guiPaging, Group group, String filterString, boolean[] tooManyResultsArray)
374       throws SchemaException {
375     Set<Member> members = new LinkedHashSet<Member>();
376     
377     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
378     
379     SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = 
380       SimpleMembershipUpdateContainer.retrieveFromSession();
381     
382     //lets do the subject search
383     if (StringUtils.defaultString(filterString).length() < GrouperUiConfig.retrieveConfig().propertyValueInt("simpleMembershipUpdate.filterComboMinChars", 3)) {
384       guiResponseJs.addAction(GuiScreenAction.newAlert(
385           simpleMembershipUpdateContainer.getText().getErrorNotEnoughFilterCharsAlert()));
386       return members;
387     }
388     
389     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
390     int maxSubjectsSort = GrouperUiConfig.retrieveConfig().propertyValueInt("comparator.sort.limit", 200);  // used if we're not doing sorting using member attributes
391     SortStringEnum sortStringEnum = simpleMembershipUpdateContainer.getSelectedSortStringEnum();
392     SearchStringEnum searchStringEnum = simpleMembershipUpdateContainer.getSearchStringEnum();
393     boolean useMemberSortAndSearch = sortStringEnum != null && searchStringEnum != null;
394     
395     Set<Subject> subjects = null;
396     boolean filterByOneSubject = false;
397     
398     //lets see if there is a subject in there
399     Subject subject = GrouperUiUtils.findSubject(filterString, false);
400     if (subject != null) {
401       //if there was a subject selected...
402       filterByOneSubject = true;
403       subjects = GrouperUtil.toSet(subject);
404       //show filter string on screen
405       simpleMembershipUpdateContainer.setMemberFilterForScreen(GrouperUiUtils.convertSubjectToLabelConfigured(subject));
406     } else {
407       //show filter string on screen
408       simpleMembershipUpdateContainer.setMemberFilterForScreen(filterString); 
409     }
410     
411     // if we're not using member attributes for sorting/searching and the filter value isn't for a single subject
412     if (!filterByOneSubject && !useMemberSortAndSearch) {
413       boolean tooManyResults = false;
414       try {
415         subjects = SubjectFinder.findPage(filterString).getResults();
416       } catch (SubjectTooManyResults stmr) {
417         tooManyResults = true;
418       }
419       if (tooManyResults || GrouperUtil.length(subjects) > GrouperUiConfig.retrieveConfig().propertyValueInt("simpleMembershipUpdate.filterMaxSearchSubjects", 1000)) {
420         if (GrouperUtil.length(tooManyResultsArray) > 0) {
421           tooManyResultsArray[0] = true;
422         }
423         simpleMembershipUpdateContainer.setMemberFilterForScreen(filterString);
424         guiResponseJs.addAction(GuiScreenAction.newAlert(
425             simpleMembershipUpdateContainer.getText().getErrorMemberFilterTooManyResults()));
426         return members;
427       }
428     }
429   
430     GrouperSession grouperSession = null;
431     String groupName = null;
432     boolean startedSession = false;
433     try {
434       grouperSession = GrouperSession.staticGrouperSession();
435       if (grouperSession == null) {
436         startedSession = true;
437         grouperSession = GrouperSession.start(loggedInSubject);
438       }
439   
440       groupName = group.getName();
441       
442       if (useMemberSortAndSearch && !filterByOneSubject) {
443         QueryOptions queryOptions = new QueryOptions().retrieveCount(true).retrieveResults(false);
444         group.getImmediateMembers(Group.getDefaultList(), null, queryOptions, 
445             sortStringEnum, searchStringEnum, filterString);
446         int totalSize = queryOptions.getCount() == null ? 0 : queryOptions.getCount().intValue();
447         
448         // check if too many results...
449         if (totalSize > GrouperUiConfig.retrieveConfig().propertyValueInt("simpleMembershipUpdate.filterMaxSearchSubjects", 1000)) {
450           if (GrouperUtil.length(tooManyResultsArray) > 0) {
451             tooManyResultsArray[0] = true;
452           }
453 
454           guiResponseJs.addAction(GuiScreenAction.newAlert(simpleMembershipUpdateContainer.getText().getErrorMemberFilterTooManyResults()));
455           return members;
456         }
457         
458         int pageSize = guiPaging.getPageSize();
459         
460         guiPaging.setTotalRecordCount(totalSize);
461         
462         QueryPaging queryPaging = new QueryPaging();
463         queryPaging.setPageSize(pageSize);
464         queryPaging.setPageNumber(guiPaging.getPageNumber());
465         
466         queryOptions = new QueryOptions().paging(queryPaging);
467   
468         members = group.getImmediateMembers(Group.getDefaultList(), null, queryOptions, 
469             sortStringEnum, searchStringEnum, filterString);
470       } else {
471       
472         members = GrouperUiUtils.convertSubjectsToMembers(grouperSession, group, subjects, true);
473       
474         QueryPaging queryPaging = new QueryPaging(maxSubjectsSort, 1, true);
475         
476         int totalSize = members.size();
477         
478         int pageSize = guiPaging.getPageSize();
479         
480         guiPaging.setTotalRecordCount(totalSize);
481         guiPaging.setPageSize(pageSize);
482         
483         //if there are less than the sort limit, then just get all, no problem
484         queryPaging = new QueryPaging();
485         queryPaging.setPageSize(pageSize);
486         queryPaging.setPageNumber(guiPaging.getPageNumber());
487         
488         if (totalSize <= maxSubjectsSort) {
489           members = GrouperUiUtils.membersSortedPaged(members, queryPaging);
490         } else {
491           members = new LinkedHashSet<Member>(GrouperUtil.batchList(GrouperUtil.listFromCollection(members), 
492               pageSize, guiPaging.getPageNumber()-1));
493         }
494       }
495     } catch (Exception se) {
496       throw new RuntimeException("Error searching for members: '" + filterString 
497           + "', " + groupName + ", " + se.getMessage(), se);
498     } finally {
499       if (startedSession) {
500         GrouperSession.stopQuietly(grouperSession);
501       }
502     } 
503     
504     return members;
505   }
