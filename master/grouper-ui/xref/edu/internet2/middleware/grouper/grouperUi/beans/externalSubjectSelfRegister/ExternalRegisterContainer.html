46  /**
47   * bean for simple membership update.  holds all state for this module
48   */
49  @SuppressWarnings("serial")
50  public class ExternalRegisterContainer implements Serializable {
51  
52    /**
53     * 
54     */
55    public ExternalRegisterContainer() {
56      
57      this.initFields();
58      
59    }
60  
61    /** cache if this is an insert or not */
62    private Boolean insert = null;
63    
64    /**
65     * if should show delete button
66     * @return if should show
67     */
68    public boolean isShowDeleteButton() {
69      return !this.isInsert() && GrouperUiConfig.retrieveConfig().propertyValueBoolean("externalMembers.allowSelfDelete", false);
70    }
71    
72    /**
73     * if this record exists in the DB then it is an update.  Else it is an insert
74     * @return if this is an insert or update
75     */
76    public boolean isInsert() {
77      
78      if (this.insert == null) {
79      
80        final String identifier = this.getUserLoggedInIdentifier();
81        
82        GrouperSession grouperSession = GrouperSession.startRootSession(false);
83        ExternalSubject externalSubject = null;
84        try {
85          externalSubject = (ExternalSubject)GrouperSession.callbackGrouperSession(grouperSession, new GrouperSessionHandler() {
86            
87            @Override
88            public Object callback(GrouperSession theGrouperSession) throws GrouperSessionException {
89              return ExternalSubjectStorageController.findByIdentifier(identifier, false, null);
90            }
91          });
92              
93              //if its null then it is an insert
94          this.insert = externalSubject == null;
95  
96        } finally {
97            GrouperSession.stopQuietly(grouperSession);
98          }
99        }
100     return this.insert;
101   }
102   
103   /**
104    * 
105    */
106   private void initFields() {
107     
108     GrouperSession grouperSession = GrouperSession.startRootSession(false);
109     try { 
110       GrouperSession.callbackGrouperSession(grouperSession, new GrouperSessionHandler() {
111   
112         /**
113          * 
114          */
115         @Override
116         public Object callback(GrouperSession theGrouperSession) throws GrouperSessionException {
117           
118           ExternalRegisterContainer.this.registerFields = new ArrayList<RegisterField>();
119           RegisterField registerField = null;
120   
121           ExternalSubject externalSubject = null;
122           
123           {
124             registerField = new RegisterField();
125             registerField.setSystemName("identifier");
126             registerField.setParamName("param_identifier");
127             String identifierLabel = GrouperUiUtils.message("externalSubjectSelfRegister.register.field.identifier.label");
128             registerField.setLabel(identifierLabel);
129             
130             registerField.setFieldNotAttribute(true);
131             registerField.setReadonly(true);
132             registerField.setRequired(false);
133             
134             String identifierTooltip = GrouperUiUtils.message("externalSubjectSelfRegister.register.field.identifier.tooltip", true);
135             
136             registerField.setTooltip(identifierTooltip);
137             
138             String identifier = getUserLoggedInIdentifier();
139             
140             registerField.setValue(identifier);
141             ExternalRegisterContainer.this.registerFields.add(registerField);
142   
143             //get the current subject so we can prepopulate data, if it exists
144             externalSubject = ExternalSubjectStorageController.findByIdentifier(identifier, false, new QueryOptions().secondLevelCache(false));
145             
146           }
147           
148           ExternalSubjectConfigBean externalSubjectConfigBean = ExternalSubjectConfig.externalSubjectConfigBean();
149           
150           
151           {
152             registerField = new RegisterField();
153             registerField.setSystemName("name");
154             
155             if (externalSubjectConfigBean.isNameRequired()) {
156               registerField.setRequired(true);
157             }
158             registerField.setFieldNotAttribute(true);
159             registerField.setParamName("param_name");
160   
161             String label = GrouperUiUtils.message("externalSubjectSelfRegister.register.field.name.label", true);
162             String tooltip = GrouperUiUtils.message("externalSubjectSelfRegister.register.field.name.tooltip", true);
163             
164             registerField.setLabel(label);
165             registerField.setTooltip(tooltip);
166             
167             if (externalSubject != null) {
168               registerField.setValue(externalSubject.getName());
169             }
170             
171             ExternalRegisterContainer.this.registerFields.add(registerField);
172             
173           }
174           
175           {
176   
177             if (externalSubjectConfigBean.isInstitutionEnabled()) {
178               registerField = new RegisterField();
179               registerField.setSystemName("institution");
180               if (externalSubjectConfigBean.isInstitutionRequired()) {
181                 registerField.setRequired(true);
182               }
183               registerField.setFieldNotAttribute(true);
184   
185               String label = GrouperUiUtils.message("externalSubjectSelfRegister.register.field.institution.label", true);
186               String tooltip = GrouperUiUtils.message("externalSubjectSelfRegister.register.field.institution.tooltip", true);
187               
188               registerField.setLabel(label);
189               registerField.setTooltip(tooltip);
190   
191               registerField.setParamName("param_institution");
192   
193               if (externalSubject != null) {
194                 registerField.setValue(externalSubject.getInstitution());
195               }
196   
197               ExternalRegisterContainer.this.registerFields.add(registerField);
198             }
199           }
200   
201   
202           {
203             if (externalSubjectConfigBean.isEmailEnabled()) {
204               registerField = new RegisterField();
205               registerField.setSystemName("email");
206   
207               if (externalSubjectConfigBean.isEmailRequired()) {
208                 registerField.setRequired(true);
209               }
210   
211               registerField.setFieldNotAttribute(true);
212   
213               String label = GrouperUiUtils.message("externalSubjectSelfRegister.register.field.email.label", true);
214               String tooltip = GrouperUiUtils.message("externalSubjectSelfRegister.register.field.email.tooltip", true);
215               
216               registerField.setLabel(label);
217               registerField.setTooltip(tooltip);
218               registerField.setParamName("param_email");
219   
220               if (externalSubject != null) {
221                 registerField.setValue(externalSubject.getEmail());
222               }
223   
224               ExternalRegisterContainer.this.registerFields.add(registerField);
225             }
226           }
227   
228           for (ExternalSubjectAttributeConfigBean externalSubjectAttributeConfigBean 
229               : externalSubjectConfigBean.getExternalSubjectAttributeConfigBeans()) {
230   
231             registerField = new RegisterField();
232             registerField.setSystemName(externalSubjectAttributeConfigBean.getSystemName());
233             registerField.setRequired(externalSubjectAttributeConfigBean.isRequired());
234             registerField.setFieldNotAttribute(false);
235   
236             String label = GrouperUiUtils.message("externalSubjectSelfRegister.register.field." + registerField.getSystemName() + ".label", true);
237             String tooltip = GrouperUiUtils.message("externalSubjectSelfRegister.register.field." + registerField.getSystemName() + ".tooltip", true);
238   
239             registerField.setLabel(label);
240             registerField.setTooltip(tooltip);
241   
242             if (externalSubject != null) {
243               ExternalSubjectAttribute externalSubjectAttribute = externalSubject.retrieveAttribute(externalSubjectAttributeConfigBean.getSystemName(), false);
244               if (externalSubjectAttribute != null) {
245                 registerField.setValue(externalSubjectAttribute.getAttributeValue());
246               }
247             }
248   
249             registerField.setParamName("param_" + registerField.getSystemName());
250             ExternalRegisterContainer.this.registerFields.add(registerField);
251   
252           }
253   
254           return null;
255         }
256       });
257     } finally {      
258         GrouperSession.stopQuietly(grouperSession);
259       }
260     }

262   /**
263    * get the identifier of the user logged in
264    * @return the identifier
265    */
266   public String getUserLoggedInIdentifier() {
267     return GrouperUiFilter.remoteUser(GrouperUiFilter.retrieveHttpServletRequest());
268   }

283   /**
284    * store to session scope
285    */
286   public void storeToRequest() {
287     HttpServletRequest httpServletRequest = GrouperUiFilter.retrieveHttpServletRequest();
288     httpServletRequest.setAttribute("externalRegisterContainer", this);
289   }
290 
291   /**
292    * retrieveFromSession, cannot be null
293    * @return the app state in request scope
294    */
295   public static ExternalRegisterContainer retrieveFromRequest() {
296     HttpServletRequest httpServletRequest = GrouperUiFilter.retrieveHttpServletRequest();
297 
298     ExternalRegisterContainer externalRegisterContainer = (ExternalRegisterContainer)httpServletRequest
299       .getAttribute("externalRegisterContainer");
300     if (externalRegisterContainer == null) {
301       throw new NoSessionException(GrouperUiUtils.message("externalSubjectSelfRegister.noContainer"));
302     }
303     return externalRegisterContainer;
304   }
