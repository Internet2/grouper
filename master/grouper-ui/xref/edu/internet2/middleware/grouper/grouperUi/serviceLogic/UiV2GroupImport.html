94    /**
95     * validate import list
96     * @param request
97     * @param response
98     */
99    public void groupImportValidateList(HttpServletRequest request, HttpServletResponse response) {
100     
101     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
102     
103     GrouperSession grouperSession = null;
104   
105     try {
106       
107       grouperSession = GrouperSession.start(loggedInSubject);
108 
109       GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
110 
111       String entityList = StringUtils.defaultString(request.getParameter("entityList"));
112       
113       //split trim by comma, semi, or whitespace
114       entityList = StringUtils.replace(entityList, ",", " ");
115       entityList = StringUtils.replace(entityList, ";", " ");
116       
117       String[] entityIdOrIdentifiers = GrouperUtil.splitTrim(entityList, null, true);
118 
119       if (GrouperUtil.length(entityIdOrIdentifiers) == 0) {
120 
121         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
122             "#entityListId",
123             TextContainer.retrieveFromRequest().getText().get("groupImportNoEntitiesSpecified")));
124         return;
125 
126       }
127       
128       if (GrouperUtil.length(entityIdOrIdentifiers) > 100) {
129 
130         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
131             TextContainer.retrieveFromRequest().getText().get("groupImportTooManyEntitiesToValidate")));
132         return;
133 
134       }
135       
136       //extra source ids and subjects ids
137       Set<GuiSubject> extraGuiSubjects = new LinkedHashSet<GuiSubject>();
138       
139       GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer().setGroupImportExtraGuiSubjects(extraGuiSubjects);
140 
141       String source = request.getParameter("searchEntitySourceName");
142       
143       List<String> entityIdOrIdentifiersList = new ArrayList<String>(Arrays.asList(GrouperUtil.nonNull(
144           entityIdOrIdentifiers, String.class)));
145       
146       Map<String, Subject> entityIdOrIdentifierMap = null;
147       
148       if (StringUtils.equals("all", source)) {
149 
150         entityIdOrIdentifierMap = SubjectFinder.findByIdsOrIdentifiers(entityIdOrIdentifiersList);
151         
152       } else {
153 
154         entityIdOrIdentifierMap = SubjectFinder.findByIdsOrIdentifiers(entityIdOrIdentifiersList, source);
155 
156       }
157       
158       //lets add all the subjects
159       for (Subject subject : GrouperUtil.nonNull(entityIdOrIdentifierMap).values()) {
160         extraGuiSubjects.add(new GuiSubject(subject));
161       }
162 
163       //lets see which are missing
164       entityIdOrIdentifiersList.removeAll(GrouperUtil.nonNull(entityIdOrIdentifierMap).keySet());
165       
166       if (entityIdOrIdentifiersList.size() > 0) {
167         GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer().setEntityIdsNotFound(GrouperUtil.join(entityIdOrIdentifiersList.iterator(), ", "));
168         
169         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
170             TextContainer.retrieveFromRequest().getText().get("groupImportEntityIdsNotFound")));
171 
172       }
173 
174       //clear out combobox
175       guiResponseJs.addAction(GuiScreenAction.newScript(
176           "dijit.byId('groupAddMemberComboId').set('displayedValue', ''); " +
177           "dijit.byId('groupAddMemberComboId').set('value', '');"));
178 
179       //select the option for enter in list
180       guiResponseJs.addAction(GuiScreenAction.newFormFieldValue("bulkAddOptions", "input"));
181 
182       //fill in the extra subjects
183       guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#groupImportExtraMembersDivId", 
184           "/WEB-INF/grouperUi2/groupImport/groupImportExtraSubjects.jsp"));
185 
186     } finally {
187       GrouperSession.stopQuietly(grouperSession);
188     }
189 
190 
191   }

194   /**
195    * export a group
196    * @param request
197    * @param response
198    */
199   public void groupExportSubmit(HttpServletRequest request, HttpServletResponse response) {
200 
201     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
202     
203     GrouperSession grouperSession = null;
204   
205     Group group = null;
206   
207     try {
208   
209       grouperSession = GrouperSession.start(loggedInSubject);
210 
211       List<String> urlStrings = GrouperUiRestServlet.extractUrlStrings(request);
212       
213       //groupId=721e4e8ae6e54c4087db092f0a6372f7
214       String groupIdString = urlStrings.get(2);
215       
216       String groupId = GrouperUtil.prefixOrSuffix(groupIdString, "=", false);
217       
218       group = GroupFinder.findByUuid(grouperSession, groupId, false);
219 
220       if (group == null) {
221         throw new RuntimeException("Cant find group by id: " + groupId);
222       }
223       
224       GroupContainer groupContainer = GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupContainer();
225       GroupImportContainer groupImportContainer = GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer();
226       
227       groupContainer.setGuiGroup(new GuiGroup(group));
228       
229       if (!groupContainer.isCanRead()) {
230         throw new RuntimeException("Cant read group: " + group.getName());
231       }
232       
233       //ids
234       String groupExportOptions = urlStrings.get(3);
235       
236       boolean exportAll = false;
237       if (StringUtils.equals("all", groupExportOptions)) {
238         groupImportContainer.setExportAll(true);
239         exportAll = true;
240       } else if (StringUtils.equals("ids", groupExportOptions)) {
241         groupImportContainer.setExportAll(false);
242       } else {
243         throw new RuntimeException("Not expecting group-export-options value: '" + groupExportOptions + "'");
244       }
245 
246       
247       //groupExportSubjectIds_removeAllMembers.csv
248       @SuppressWarnings("unused")
249       String fileName = urlStrings.get(4);
250       
251       if (exportAll) {
252         String headersCommaSeparated = GrouperUiConfig.retrieveConfig().propertyValueString(
253             "uiV2.group.exportAllSubjectFields");
254         
255         String exportAllSortField = GrouperUiConfig.retrieveConfig().propertyValueString(
256             "uiV2.group.exportAllSortField");
257   
258         SimpleMembershipUpdateImportExport.exportGroupAllFieldsToBrowser(group, headersCommaSeparated, exportAllSortField, false);
259       } else {
260         
261         SimpleMembershipUpdateImportExport.exportGroupSubjectIdsCsv(group, false);
262         
263       }
264       
265       GrouperUserDataApi.recentlyUsedGroupAdd(GrouperUiUserData.grouperUiGroupNameForUserData(), 
266           loggedInSubject, group);
267 
268     } finally {
269       GrouperSession.stopQuietly(grouperSession);
270     }
271 
272   }

353   /**
354    * setup the extra groups (other than combobox), and maybe move the combobox down
355    * @param loggedInSubject
356    * @param request
357    * @param removeGroupId if removing one
358    * @param includeCombobox
359    * @param allGroups, pass in a blank linked hash set, and all groups will be populated including combobox
360    * @param errorOnNullCombobox true if an error should appear if there is nothing in the combobox
361    * @return true if ok, false if not
362    */
363   private boolean groupImportSetupExtraGroups(Subject loggedInSubject, 
364       HttpServletRequest request, GuiResponseJs guiResponseJs, boolean considerRemoveGroupId, boolean includeCombobox,
365       Set<Group> allGroups, boolean errorOnNullCombobox) {
366 
367     Set<GuiGroup> extraGuiGroups = new LinkedHashSet<GuiGroup>();
368     
369     GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer().setGroupImportExtraGuiGroups(extraGuiGroups);
370     
371     String removeGroupId = null;
372 
373     //if removing a group id
374     if (considerRemoveGroupId) {
375       removeGroupId = request.getParameter("removeGroupId");
376       if (StringUtils.isBlank(removeGroupId)) {
377         throw new RuntimeException("Why would removeGroupId be empty????");
378       }
379     }
380 
381     //if moving combobox down to extra list or getting all groups
382     String comboValue = request.getParameter("groupImportGroupComboName");
383     
384     if (StringUtils.isBlank(comboValue)) {
385       //if didnt pick one from results
386       comboValue = request.getParameter("groupImportGroupComboNameDisplay");
387     }
388     
389     Group theGroup = StringUtils.isBlank(comboValue) ? null : new GroupFinder()
390         .assignPrivileges(AccessPrivilege.UPDATE_PRIVILEGES)
391         .assignSubject(loggedInSubject)
392         .assignFindByUuidOrName(true).assignScope(comboValue).findGroup();
393 
394     boolean success = true;
395     
396     if (theGroup == null) {
397       if (includeCombobox && errorOnNullCombobox) {
398         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
399             "#groupImportGroupComboErrorId",
400             TextContainer.retrieveFromRequest().getText().get("groupImportGroupNotFound")));
401         success = false;
402       }
403       
404     } else {
405       if (includeCombobox) {
406         extraGuiGroups.add(new GuiGroup(theGroup));
407       }
408       //always add to all groups
409       allGroups.add(theGroup);
410     }
411 
412     //loop through all the hidden fields (max 100)
413     for (int i=0;i<100;i++) {
414       String extraGroupId = request.getParameter("extraGroupId_" + i);
415       
416       //we are at the end
417       if (StringUtils.isBlank(extraGroupId)) {
418         break;
419       }
420       
421       //might be removing this one
422       if (considerRemoveGroupId && StringUtils.equals(removeGroupId, extraGroupId)) {
423         continue;
424       }
425       
426       theGroup = new GroupFinder()
427         .assignPrivileges(AccessPrivilege.UPDATE_PRIVILEGES)
428         .assignSubject(loggedInSubject)
429         .assignFindByUuidOrName(true).assignScope(extraGroupId).findGroup();
430       
431       extraGuiGroups.add(new GuiGroup(theGroup));
432 
433       //always add to all groups
434       allGroups.add(theGroup);
435       
436     }
437     return success;
438   }

440   /**
441    * submit a group import
442    * @param request
443    * @param response
444    */
445   public void groupImportSubmit(HttpServletRequest request, HttpServletResponse response) {
446     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
447     
448     GrouperSession grouperSession = null;
449   
450     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
451 
452     GrouperRequestContainer grouperRequestContainer = GrouperRequestContainer.retrieveFromRequestOrCreate();
453     GroupImportContainer groupImportContainer = grouperRequestContainer.getGroupImportContainer();
454 
455     try {
456       grouperSession = GrouperSession.start(loggedInSubject);
457 
458       Set<Group> groups = new LinkedHashSet<Group>();
459       boolean success = groupImportSetupExtraGroups(loggedInSubject, request, guiResponseJs, false, true, groups, false);
460       
461       if (!success) {
462         //error message already shown
463         return;
464       }
465 
466       if (groups.size() == 0) {
467         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
468             "#groupImportGroupComboErrorId",
469             TextContainer.retrieveFromRequest().getText().get("groupImportGroupNotFound")));
470         return;
471       }
472       
473       // can be import, input, list
474       String bulkAddOption = request.getParameter("bulkAddOptions");
475       Map<String, Integer> listInvalidSubjectIdsAndRow = new LinkedHashMap<String, Integer>();
476       
477       final Set<Subject> subjectSet = new LinkedHashSet<Subject>();
478 
479       String fileName = null;
480       if (StringUtils.equals(bulkAddOption, "import")) {
481 
482         GrouperRequestWrapper grouperRequestWrapper = (GrouperRequestWrapper)request;
483         
484         FileItem importCsvFile = grouperRequestWrapper.getParameterFileItem("importCsvFile");
485 
486         if (importCsvFile == null) {
487           
488           guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
489               "#importCsvFileId",
490               TextContainer.retrieveFromRequest().getText().get("groupImportUploadFile")));
491           return;
492         }
493         
494         Reader reader = null;
495         reader = new InputStreamReader(importCsvFile.getInputStream());
496         
497         fileName = StringUtils.defaultString(importCsvFile == null ? "" : importCsvFile.getName());
498 
499         try {
500           subjectSet.addAll(SimpleMembershipUpdateImportExport.parseCsvImportFile(reader, fileName, new ArrayList<String>(), 
501               listInvalidSubjectIdsAndRow, true));
502         } catch (GrouperImportException gie) {
503           if (LOG.isDebugEnabled()) {
504             LOG.debug("error in import", gie);
505           }
506           guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
507               "#importCsvFileId", GrouperUtil.xmlEscape(gie.getMessage())));
508           return;
509         }
510         
511       } else if (StringUtils.equals(bulkAddOption, "input")) {
512 
513         //combobox
514         success = groupImportSetupExtraSubjects(loggedInSubject, request, guiResponseJs, false, true, subjectSet, false);
515         
516         if (!success) {
517           //error message already shown
518           return;
519         }
520         
521         if (subjectSet.size() == 0) {
522           guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
523               "#groupAddMemberComboErrorId", 
524               TextContainer.retrieveFromRequest().getText().get("groupImportSubjectNotFound")));
525           return;
526         }
527 
528 
529       } else if (StringUtils.equals(bulkAddOption, "list")) {
530 
531         String entityList = StringUtils.defaultString(request.getParameter("entityList"));
532         
533         //split trim by comma, semi, or whitespace
534         entityList = StringUtils.replace(entityList, ",", " ");
535         entityList = StringUtils.replace(entityList, ";", " ");
536         
537         String[] entityIdOrIdentifiers = GrouperUtil.splitTrim(entityList, null, true);
538 
539         if (GrouperUtil.length(entityIdOrIdentifiers) == 0) {
540 
541           guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
542               "#entityListId",
543               TextContainer.retrieveFromRequest().getText().get("groupImportNoEntitiesSpecified")));
544           return;
545 
546         }
547         
548         String source = request.getParameter("searchEntitySourceName");
549         
550         List<String> entityIdOrIdentifiersList = new ArrayList<String>(Arrays.asList(GrouperUtil.nonNull(
551             entityIdOrIdentifiers, String.class)));
552         
553         Map<String, Subject> entityIdOrIdentifierMap = null;
554         
555         if (StringUtils.equals("all", source)) {
556 
557           entityIdOrIdentifierMap = SubjectFinder.findByIdsOrIdentifiers(entityIdOrIdentifiersList);
558           
559         } else {
560 
561           entityIdOrIdentifierMap = SubjectFinder.findByIdsOrIdentifiers(entityIdOrIdentifiersList, source);
562 
563         }
564         
565         //lets add all the subjects
566         subjectSet.addAll(GrouperUtil.nonNull(entityIdOrIdentifierMap).values());
567 
568         //lets see which are missing
569         List<String> originalIdList = new ArrayList<String>(entityIdOrIdentifiersList);
570 
571         //lets see which are missing
572         entityIdOrIdentifiersList.removeAll(GrouperUtil.nonNull(entityIdOrIdentifierMap).keySet());
573 
574         //keep trac of the index of the invalid ids
575         for (String invalidId : entityIdOrIdentifiersList) {
576           int index = originalIdList.indexOf(invalidId);
577           listInvalidSubjectIdsAndRow.put(invalidId, index == -1 ? null : index);
578         }
579         
580       } else {
581         throw new RuntimeException("Not expecting bulk add option: " + bulkAddOption);
582       }
583       
584       {
585         Group group = UiV2Group.retrieveGroupHelper(request, AccessPrivilege.UPDATE, false).getGroup();
586         if (group != null) {
587           groupImportContainer.setImportFromGroup(true);
588         }
589       }
590       {
591         Subject subject = UiV2Subject.retrieveSubjectHelper(request, false);
592         if (subject != null) {
593           groupImportContainer.setImportFromSubject(true);
594         }
595       }
596 
597       Set<GuiGroup> guiGroups = new LinkedHashSet<GuiGroup>();
598       groupImportContainer.setGuiGroups(guiGroups);
599       
600       Map<String, String> reportByGroupName = new HashMap<String, String>();
601       groupImportContainer.setReportForGroupNameMap(reportByGroupName);
602 
603       int totalAdded = 0;
604       int totalDeleted = 0;
605       
606       Iterator<Group> groupIterator = groups.iterator();
607 
608       boolean importReplaceMembers = GrouperUtil.booleanValue(request.getParameter("replaceExistingMembers"), false);
609 
610       //lets go through the groups that were submitted
611       while (groupIterator.hasNext()) {
612 
613         final Group group = groupIterator.next();
614 
615         {
616           //remove groups that cannot be viewed
617           boolean canView = (Boolean)GrouperSession.callbackGrouperSession(
618             GrouperSession.staticGrouperSession().internal_getRootSession(), new GrouperSessionHandler() {
619   
620               @Override
621               public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
622                 return group.canHavePrivilege(loggedInSubject, AccessPrivilege.VIEW.getName(), false);
623               }
624             });
625   
626           if (!canView) {
627             guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error,  
628                 TextContainer.retrieveFromRequest().getText().get("groupImportGroupCantView")));
629             groupIterator.remove();
630             continue;
631           }
632         }
633 
634         guiGroups.add(new GuiGroup(group));
635         StringBuilder report = new StringBuilder("<ul>\n");
636 
637         {
638           //give error if cant update
639           boolean canUpdate = (Boolean)GrouperSession.callbackGrouperSession(
640             GrouperSession.staticGrouperSession().internal_getRootSession(), new GrouperSessionHandler() {
641 
642               @Override
643               public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
644                 return group.canHavePrivilege(loggedInSubject, AccessPrivilege.UPDATE.getName(), false);
645               }
646             });
647 
648           if (!canUpdate) {
649             report.append("<li>" +
650               TextContainer.retrieveFromRequest().getText().get("groupImportGroupCantUpdate") + "</li></ul>");
651             continue;
652           }
653         }
654 
655         //<ul>
656         //  <li>Before importing, the membership count was 10 and is now 12.</li>
657         //  <li>You successfully added 2 members and deleted 0 members.</li>
658         //  <li>2 members were not imported due to errors, as shown below.</li>
659         //</ul>
660         //<h5>Errors</h5>
661         //<ul>
662         //  <li><span class="label label-important">Error</span>&nbsp;on row 2. Subject not found: "foo-bar-user"</li>
663         //</ul>
664 
665         List<Member> existingMembers = new ArrayList<Member>(GrouperUtil.nonNull(group.getImmediateMembers()));
666         List<Subject> subjectList = new ArrayList<Subject>(GrouperUtil.nonNull(subjectSet));
667         int existingCount = GrouperUtil.length(existingMembers);
668         groupImportContainer.setGroupCountOriginal(existingCount);
669         
670         GrouperUiUtils.removeOverlappingSubjects(existingMembers, subjectList);
671 
672         int addedCount = 0;
673         int errorsCount = 0;
674 
675         StringBuilder errors = new StringBuilder();
676 
677         // figure out subject not founds
678         if (listInvalidSubjectIdsAndRow.size() > 0) {
679           for (String subjectLabel : listInvalidSubjectIdsAndRow.keySet()) {
680             int rowNumber = listInvalidSubjectIdsAndRow.get(subjectLabel);
681             String errorLine = errorLine(subjectLabel, TextContainer.retrieveFromRequest().getText().get(
682                 "groupImportProblemFindingSubjectError"), rowNumber);
683             errors.append(errorLine).append("\n");
684             errorsCount++;
685           }
686         }
687         
688         //first lets add some members
689         for (int i=0;i<subjectList.size();i++) {
690           
691           Subject subject = subjectList.get(i);
692           
693           if (subject instanceof ImportSubjectWrapper) {
694             try {
695               subject = ((ImportSubjectWrapper)subject).wrappedSubject();
696             } catch (Exception e) {
697               //ignore
698             }
699           }
700           
701           try {
702               
703             group.addMember(subject, false);
704             
705             addedCount++;
706           } catch (Exception e) {
707             
708             String errorLine = errorLine(subject, GrouperUtil.xmlEscape(e.getMessage()));
709             errors.append(errorLine).append("\n");
710             errorsCount++;
711             LOG.warn(errorLine, e);
712           }
713     
714         }
715     
716         boolean didntImportDueToSubjects = errorsCount > 0;
717         int deletedCount = 0;
718     
719         //remove the ones which are already there
720         if (importReplaceMembers && !didntImportDueToSubjects) {
721           
722           for (Member existingMember : existingMembers) {
723             
724             try {
725               group.deleteMember(existingMember, false);
726               deletedCount++;
727             } catch (Exception e) {
728               String errorLine = errorLine(existingMember.getSubject(), GrouperUtil.xmlEscape(e.getMessage()));
729               errors.append(errorLine).append("\n");
730               errorsCount++;
731               LOG.warn(errorLine, e);
732             
733             }
734           }
735         }
736 
737         if (importReplaceMembers && didntImportDueToSubjects) {
738           report.append(TextContainer.retrieveFromRequest().getText().get("groupImportReportNoReplaceError")).append("\n");
739         }
740         
741         //this might be a little wasteful, but I think it is a good sanity check
742         int newSize = group.getImmediateMembers().size();
743 
744         // = Errors
745         //groupImportReportErrorLine = <li><span class="label label-important">Error</span>&nbsp;on row ${grouperRequestContainer.groupImportContainer.errorRowNumber}. ${grouperRequestContainer.groupImportContainer.errorText}: "${grouperUtil.xmlEscape(grouperRequestContainer.groupImportContainer.errorSubject)}"</li>
746 
747         //set stuff for text to use
748         groupImportContainer.setGroupCountAdded(addedCount);
749         groupImportContainer.setGroupCountDeleted(deletedCount);
750         groupImportContainer.setGroupCountErrors(errorsCount);
751         groupImportContainer.setGroupCountOriginal(existingCount);
752         groupImportContainer.setGroupCountNew(newSize);
753         
754         totalAdded += addedCount;
755         totalDeleted += deletedCount;
756         
757         report.append(TextContainer.retrieveFromRequest().getText().get("groupImportReportSummary")).append("\n");
758         report.append(TextContainer.retrieveFromRequest().getText().get("groupImportReportSuccess")).append("\n");
759         
760         // dont add the error report line if there are no errors 
761         if (errorsCount > 0) {
762           report.append(TextContainer.retrieveFromRequest().getText().get("groupImportReportErrorSummary")).append("\n");
763         }
764         report.append("</ul>\n");
765         
766         //only add the errors section if there are errors
767         if (errorsCount > 0) {
768           report.append("<h5>").append(TextContainer.retrieveFromRequest().getText().get("groupImportReportErrorsTitle")).append("</h5>\n");
769           report.append("<ul>\n");
770           report.append(errors.toString());
771           report.append("</ul>\n");
772         }
773         
774         reportByGroupName.put(group.getName(), report.toString());
775 
776         GrouperUserDataApi.recentlyUsedGroupAdd(GrouperUiUserData.grouperUiGroupNameForUserData(), 
777             loggedInSubject, group);
778         
779       }
780       
781       if (StringUtils.equals(bulkAddOption, "import")) {
782     	  auditImport(subjectSet.size(), fileName, totalAdded, totalDeleted);
783       }
784       
785       //show the report screen
786       guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#grouperMainContentDivId", 
787           "/WEB-INF/grouperUi2/groupImport/groupImportReport.jsp"));
788 
789       guiResponseJs.addAction(GuiScreenAction.newScript("guiScrollTop()"));
790 
791 
792     } catch (NoSessionException se) {
793       throw se;
794     } catch (ControllerDone cd) {
795       throw cd;
796     } catch (Exception e) {
797       throw new RuntimeException("error", e);
798     } finally {
799       GrouperSession.stopQuietly(grouperSession);
800     }
801 
802   }

1105   /**
1106    * setup the extra members (other than combobox), and maybe move the combobox down
1107    * @param loggedInSubject
1108    * @param request
1109    * @param considerRemoveSubjectSourceAndId if removing one
1110    * @param includeCombobox
1111    * @param allSubjects is a LinkedHashSet of subjects
1112    * @param errorOnNullCombobox is true if an error should appear if there is nothing in the combobox
1113    * @return true if ok, false if not
1114    */
1115   private boolean groupImportSetupExtraSubjects(Subject loggedInSubject, 
1116       HttpServletRequest request, GuiResponseJs guiResponseJs, boolean considerRemoveSubjectSourceAndId, 
1117       boolean includeCombobox, Set<Subject> allSubjects, boolean errorOnNullCombobox) {
1118 
1119     //extra source ids and subjects ids
1120     Set<MultiKey> extraSubjectSourceAndIds = new HashSet<MultiKey>();
1121     Set<GuiSubject> extraGuiSubjects = new LinkedHashSet<GuiSubject>();
1122     
1123     GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer().setGroupImportExtraGuiSubjects(extraGuiSubjects);
1124     
1125     Set<MultiKey> allSubjectsSourceAndIds = new HashSet<MultiKey>();
1126   
1127     String removeSubjectSourceAndId = null;
1128   
1129     //if removing a group id
1130     if (considerRemoveSubjectSourceAndId) {
1131       removeSubjectSourceAndId = request.getParameter("removeSubjectSourceAndId");
1132       if (StringUtils.isBlank(removeSubjectSourceAndId)) {
1133         throw new RuntimeException("Why would removeSubjectSourceAndId be empty????");
1134       }
1135     }
1136 
1137     Subject theSubject = null;
1138     
1139     {
1140       //if moving combobox down to extra list or getting all groups
1141       String comboValue = request.getParameter("groupAddMemberComboName");
1142       
1143       if (StringUtils.isBlank(comboValue)) {
1144         //if didnt pick one from results
1145         comboValue = request.getParameter("groupAddMemberComboDisplay");
1146       }
1147       
1148       try {
1149         GrouperSourceAdapter.searchForGroupsWithReadPrivilege(true);
1150         if (comboValue != null && comboValue.contains("||")) {
1151           String sourceId = GrouperUtil.prefixOrSuffix(comboValue, "||", true);
1152           String subjectId = GrouperUtil.prefixOrSuffix(comboValue, "||", false);
1153           theSubject =  SubjectFinder.findByIdOrIdentifierAndSource(subjectId, sourceId, false);
1154         } else {
1155           theSubject = StringUtils.isBlank(comboValue) ? null : SubjectFinder.findByIdOrIdentifier(comboValue, false);
1156         }
1157       } finally {
1158         GrouperSourceAdapter.clearSearchForGroupsWithReadPrivilege();
1159       }
1160     }
1161     boolean success = true;
1162     if (theSubject == null) {
1163       if (includeCombobox && errorOnNullCombobox) {
1164         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
1165             "#groupAddMemberComboErrorId", 
1166             TextContainer.retrieveFromRequest().getText().get("groupImportSubjectNotFound")));
1167         success = false;
1168       }
1169       
1170     } else {
1171       MultiKey multiKey = new MultiKey(theSubject.getSourceId(), theSubject.getId());
1172       if (includeCombobox) {
1173         if (!extraSubjectSourceAndIds.contains(multiKey)) {
1174           extraGuiSubjects.add(new GuiSubject(theSubject));
1175           extraSubjectSourceAndIds.add(multiKey);
1176         }
1177       }
1178       //always add to all groups
1179       if (!allSubjectsSourceAndIds.contains(multiKey)) {
1180         allSubjects.add(theSubject);
1181         allSubjectsSourceAndIds.add(multiKey);
1182       }
1183     }
1184   
1185     //loop through all the hidden fields (max 100)
1186     for (int i=0;i<100;i++) {
1187       String extraSourceIdSubjectId = request.getParameter("extraSourceIdSubjectId_" + i);
1188       
1189       //we are at the end
1190       if (StringUtils.isBlank(extraSourceIdSubjectId)) {
1191         break;
1192       }
1193       
1194       //might be removing this one
1195       if (considerRemoveSubjectSourceAndId && StringUtils.equals(removeSubjectSourceAndId, extraSourceIdSubjectId)) {
1196         continue;
1197       }
1198       
1199       theSubject = null;
1200 
1201       try {
1202         GrouperSourceAdapter.searchForGroupsWithReadPrivilege(true);
1203         if (extraSourceIdSubjectId != null && extraSourceIdSubjectId.contains("||")) {
1204           String sourceId = GrouperUtil.prefixOrSuffix(extraSourceIdSubjectId, "||", true);
1205           String subjectId = GrouperUtil.prefixOrSuffix(extraSourceIdSubjectId, "||", false);
1206           theSubject =  SubjectFinder.findByIdOrIdentifierAndSource(subjectId, sourceId, false);
1207         }
1208       } finally {
1209         GrouperSourceAdapter.clearSearchForGroupsWithReadPrivilege();
1210       }
1211       
1212       if (theSubject != null) {
1213         MultiKey multiKey = new MultiKey(theSubject.getSourceId(), theSubject.getId());
1214         if (!extraSubjectSourceAndIds.contains(multiKey)) {
1215           extraGuiSubjects.add(new GuiSubject(theSubject));
1216           extraSubjectSourceAndIds.add(multiKey);
1217         }
1218   
1219         if (!allSubjectsSourceAndIds.contains(multiKey)) {
1220           allSubjects.add(theSubject);
1221           allSubjectsSourceAndIds.add(multiKey);
1222         }
1223       }      
1224     }
1225     return success;
1226   }
