255   /**
256    * init screen labels
257    */
258   private void initScreenLabels() {
259     if (this.screenLabelLong == null && this.screenLabelShort == null && this.guiSubject != null) {
260       
261       String screenLabel = this.guiSubject.getScreenLabelLong();
262             
263       this.screenLabelLong = screenLabel;
264       
265       screenLabel = this.guiSubject.getScreenLabel();
266       
267       int maxWidth = GrouperUiConfig.retrieveConfig().propertyValueInt("simplePermissionUpdate.maxOwnerSubjectChars", 50);
268       if (maxWidth == -1) {
269         this.screenLabelShort = screenLabel;
270       } else {
271         this.screenLabelShort = StringUtils.abbreviate(screenLabel, maxWidth);
272       }
273     }
274   }

308   /**
309    * process raw entries
310    * @param actions 
311    * @param permissionEntryLimitBeanMap is the map of permission entry to the set of limit beans (limits and values)
312    */
313   public void processRawEntries(List<String> actions, Map<PermissionEntry, Set<PermissionLimitBean>> permissionEntryLimitBeanMap) {
314     
315     this.actionToGuiPermissionEntryMap = new HashMap<String, GuiPermissionEntry>();
316 
317     //lets get an entry for each action if there is an assignment or not
318     for (String action: actions) {
319       GuiPermissionEntry guiPermissionEntry = new GuiPermissionEntry();
320       this.actionToGuiPermissionEntryMap.put(action, guiPermissionEntry);
321       guiPermissionEntry.setPermissionType(this.permissionType);
322       guiPermissionEntry.setRawGuiPermissionEntries(new ArrayList<GuiPermissionEntry>());
323     }
324 
325     //we only want one entry for each limit, so keep track
326     Map<AttributeAssign, GuiPermissionLimitBeanContainer> limitToGuiPermissionLimitBeanContainerMap = 
327       new HashMap<AttributeAssign, GuiPermissionLimitBeanContainer>();
328 
329     for (PermissionEntry permissionEntry : this.getRawPermissionEntries()) {
330       GuiPermissionEntry guiPermissionEntry = this.actionToGuiPermissionEntryMap.get(permissionEntry.getAction());
331       
332       //if not found, why?
333       if (guiPermissionEntry == null) {
334         throw new RuntimeException("Why no guiPermissionEntry for action: " 
335             + permissionEntry.getAction() + ", " + GrouperUtil.stringValue(actions));
336       }
337       
338       GuiPermissionEntry currentGui = new GuiPermissionEntry();
339       currentGui.setPermissionEntry(permissionEntry);
340       
341       guiPermissionEntry.getRawGuiPermissionEntries().add(currentGui);
342       
343       //lets setup the limits
344       Set<PermissionLimitBean> permissionLimitBeanSet = permissionEntryLimitBeanMap.get(permissionEntry);
345       
346       //if we got some
347       if (GrouperUtil.length(permissionLimitBeanSet) > 0) {
348         if (this.guiPermissionLimitBeanContainers == null) {
349           this.guiPermissionLimitBeanContainers = new TreeSet<GuiPermissionLimitBeanContainer>();
350         }
351         for (PermissionLimitBean permissionLimitBean : permissionLimitBeanSet) {
352           //see if we have a row for this limit yet
353           GuiPermissionLimitBeanContainer guiPermissionLimitBeanContainer = limitToGuiPermissionLimitBeanContainerMap.get(permissionLimitBean.getLimitAssign());
354           
355           //if not, make one
356           if (guiPermissionLimitBeanContainer == null) {
357             guiPermissionLimitBeanContainer = new GuiPermissionLimitBeanContainer();
358             guiPermissionLimitBeanContainer.setPermissionLimitBean(permissionLimitBean);
359             limitToGuiPermissionLimitBeanContainerMap.put(permissionLimitBean.getLimitAssign(), guiPermissionLimitBeanContainer);
360             this.guiPermissionLimitBeanContainers.add(guiPermissionLimitBeanContainer);
361           }
362           
363           //this limit can apply to multiple actions
364           guiPermissionLimitBeanContainer.getActions().add(permissionEntry.getAction());
365 
366           //for the limit to be immediate, it must be an assignment on the permission, and the permission needs to be immediate
367           if (permissionEntry.isImmediate(this.permissionType)) {
368             AttributeAssign limitAssign = permissionLimitBean.getLimitAssign();
369             if (StringUtils.equals(limitAssign.getOwnerAttributeAssignId(), permissionEntry.getAttributeAssignId())) {
370               guiPermissionLimitBeanContainer.setImmediate(true);
371             }
372           }
373           
374         }
375       }
376       
377     }
378     
379     //now lets process the inner objects
380     for (GuiPermissionEntry guiPermissionEntry : this.actionToGuiPermissionEntryMap.values()) {
381       guiPermissionEntry.processRawEntries();
382     }
383     
384   }

386   /**
387    * sort on the screen from left to right on the columns
388    */
389   @Override
390   public int compareTo(GuiPermissionEntryContainer other) {
391     CompareToBuilder compareToBuilder = new CompareToBuilder().append(this.getRole().getDisplayExtension(), other.getRole().getDisplayExtension());
392     if (this.permissionType == PermissionType.role_subject) {
393       compareToBuilder.append(this.getScreenLabelShort(), other.getScreenLabelShort());
394     }
395     compareToBuilder.append(this.getPermissionResource().getDisplayExtension(), other.getPermissionResource().getDisplayExtension());
396     return compareToBuilder.toComparison();
397   }
