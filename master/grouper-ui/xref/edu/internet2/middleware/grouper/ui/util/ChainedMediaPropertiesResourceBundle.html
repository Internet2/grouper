16  package edu.internet2.middleware.grouper.ui.util;
17  
18  import java.util.Enumeration;
19  import java.util.ResourceBundle;
20  import java.util.Vector;
21  
22  import org.apache.commons.lang.StringUtils;
23  
24  /**
25   * do a resource bundle which consults the properties file before 
26   * consulting the media.properties resource bundle
27   * @author mchyzer
28   *
29   */
30  public class ChainedMediaPropertiesResourceBundle extends ResourceBundle {
31  
32    /**
33     * constructor with the parent resource bundle
34     * @param theParentBundle
35     */
36    public ChainedMediaPropertiesResourceBundle(ResourceBundle theParentBundle) {
37      this.setParent(theParentBundle);
38    }
39    
40    /**
41     * @see ResourceBundle#getKeys()
42     */
43    @Override
44    public Enumeration<String> getKeys() {
45      Vector<String> keys = new Vector<String>();
46      
47      keys.addAll(GrouperUiConfig.retrieveConfig().propertyNames());
48      keys.addAll(this.parent.keySet());
49      
50      return keys.elements();
51      
52    }
53  
54    /**
55     * @see ResourceBundle#handleGetObject()
56     */
57    @Override
58    protected Object handleGetObject(String key) {
59      GrouperUiConfig uiConfig = GrouperUiConfig.retrieveConfig();
60      if (uiConfig.containsKey(key)) {
61        //do a default string since null means go up the chain
62        return StringUtils.defaultString(uiConfig.propertyValueString(key));
63      }
64      return null;
65    }
66  
67  }
