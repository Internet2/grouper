65  /**
66   * Reads an XML configuration file and exports membership data as specified. The configuration 
67   * file is determined by looking up the media.properties key 'membership-export.config'.
68   * The base Grouper UI distribution does not have a value set. If you want to allow 
69   * membership export you must configure one or more formats appropriate to your site. 
70   * <pre>&lt;membership-export&gt;
71      &lt;format  name=&quot;CSV (Open with Excel)&quot; separator=&quot;,&quot; 
72              quote=&quot;true&quot; 
73          extension=&quot;.csv&quot; 
74          content-type=&quot;application/ms-excel&quot;
75      &gt;<br>		&lt;headers&gt;<br>			&lt;header name=&quot;Id&quot;/&gt;<br>			&lt;header name=&quot;Name&quot;/&gt;<br>			&lt;header name=&quot;Type&quot;/&gt;<br>		&lt;/headers&gt;<br>		&lt;source id=&quot;g:gsa&quot;&gt;<br>			&lt;field name=&quot;id&quot;/&gt;<br>			&lt;field name=&quot;displayName&quot;/&gt;<br>			&lt;field value=&quot;group&quot;/&gt;<br>		&lt;/source&gt;<br>		&lt;source id=&quot;qsuob&quot;&gt;<br>			&lt;field name=&quot;id&quot;/&gt;<br>			&lt;field name=&quot;name&quot;/&gt;<br>			&lt;field value=&quot;person&quot;/&gt;<br>		&lt;/source&gt;<br>    &lt;/format&gt;
76  &lt;/membership-export&gt;  </pre>
77  <p>Currently only simple delimited files are supported</p>
78  <p><strong>format tag</strong></p>
79  <table width="100%" border="1" cellspacing="5px">
80    <tr> 
81      <td><font face="Arial, Helvetica, sans-serif"><strong>name</strong></font></td>
82      <td><font face="Arial, Helvetica, sans-serif">is the text seen by the user 
83        to identify this format - unless there is only one format, in which case 
84        it is used as the default</font></td>
85    </tr>
86    <tr> 
87      <td><font face="Arial, Helvetica, sans-serif"><strong>quote=true</strong></font></td>
88      <td><font face="Arial, Helvetica, sans-serif">indicates that double-quotes 
89        will surround each exported field. </font></td>
90    </tr>
91    <tr> 
92      <td><font face="Arial, Helvetica, sans-serif"><strong>extension</strong></font></td>
93      <td><font face="Arial, Helvetica, sans-serif">determines the file extension 
94        that will be presented to the web browser - which helps the browser choose 
95        the correct application to open, and in the case where an application recognizes 
96        different formats, lets the application know the format to expect</font></td>
97    </tr>
98    <tr> 
99      <td><font face="Arial, Helvetica, sans-serif"><strong>content-type</strong></font></td>
100     <td><font face="Arial, Helvetica, sans-serif">if configured, is sent as an 
101       HTTP header. This will determine how a web browser tries to handle the data. 
102       If no content-type is specified, it will be displayed in the UI as a normal 
103       page.</font></td>
104   </tr>
105   <tr> 
106     <td><font face="Arial, Helvetica, sans-serif"><strong>separator</strong></font></td>
107     <td><font face="Arial, Helvetica, sans-serif">String used to separate fields 
108       - typically a comma or tab (\t)</font></td>
109   </tr>
110 </table>
111 <p><strong>headers tag</strong></p>
112 <p>Optionally specifies column headings</p>
113 <p><strong>source tag</strong></p>
114 <p>Specifies which fields should be exported for Subjects with the specified source 
115   id. If a Subject to be exported has a source which has no configuration it is 
116   ignored. </p>
117 <p>Each source should specify the same number of fields, which should, if they 
118   are present, match the number of header fields specified.</p>
119 <p><strong>field tag</strong></p>
120 <table width="100%" border="1" cellspacing="5px">
121   <tr> 
122     <td><font face="Arial, Helvetica, sans-serif"><strong>name</strong></font></td>
123     <td><font face="Arial, Helvetica, sans-serif">the name of the subject attribute, 
124       the value of which will be exported</font></td>
125   </tr>
126   <tr> 
127     <td><font face="Arial, Helvetica, sans-serif"><strong>value</strong></font></td>
128     <td><font face="Arial, Helvetica, sans-serif">if name is not specified then 
129       the text in the value attribute will be used, as is. This allows 'padding' 
130       where Subjects from different sources may nnot always have equivalent fields</font></td>
131   </tr>
132 </table>
133  * <p />
134  * 
135  * @author Gary Brown.
136  * @version $Id: MembershipExporter.java,v 1.7 2009-10-16 10:30:08 isgwb Exp $
137  */

149 	public static void main(String[] args) throws Exception{
150 		Subject gs = SubjectFinder.findById("GrouperSystem", true);
151 		GrouperSession s = GrouperSession.start(gs);
152 		Group g = GroupFinder.findByName(s,"qsuob:all", true);
153 		Set members=g.getMembers();
154 		PrintWriter writer = new PrintWriter(System.out);
155 		ResourceBundle bundle = ResourceBundle.getBundle("resources/grouper/media");
156 		MembershipExporter export= new MembershipExporter();
157 		export.export("Minimal",members,writer);
158 		writer.flush();
159 		s.stop();
160 	}

171 	private void init() throws Exception{
172 		if(configXml!=null) {
173 			return;
174 		}
175 		
176 		String configResource = null;
177 		try {
178 			configResource=GrouperUiFilter.retrieveSessionMediaResourceBundle().getString("membership-export.config");
179 		}catch(MissingResourceException e){
180 			return;
181 		}
182 		configXml = DOMHelper.getDomFromResourceOnClassPath(configResource);
183 		List available = getAvailableFormats();
184 		if(!available.isEmpty()) active=true;
185 	}
186 	
187 	public boolean isActive() {
188 		return active;
189 	}

191 	public List getAvailableFormats() throws Exception{
192 		List res = new ArrayList();
193 		init();
194 		NodeList nl = configXml.getElementsByTagName("format");
195 		Element el =null;
196 		for (int i=0; i<nl.getLength();i++) {
197 			el = (Element) nl.item(i);
198 			res.add(el.getAttribute("name"));
199 			formatCache.put(el.getAttribute("name"),el);
200 		}
201 		return res;
202 	}
203 	
204 	public int getNumberOfAvailableFormats() throws Exception{
205 		return getAvailableFormats().size();
206 	}

208 	public void export(String name,Collection subjects,PrintWriter writer) throws Exception,IOException {
209 		init();
210 		if(getFormat(name)==null) throw new IllegalArgumentException(name  +" is not a valid format");
211 		String sep = getFormat(name).getAttribute("separator");
212 		quote=Boolean.parseBoolean(getFormat(name).getAttribute("quote"));
213 		if("".equals(sep) || sep==null) sep=separator;
214 		if("\\t".equals(sep)) sep="\t";
215 		printHeader(writer,name,sep);
216 		Iterator it= subjects.iterator();
217 		Subject subject = null;
218 		Object item;
219 		while(it.hasNext()) {
220 			item=it.next();
221 			try {
222 				if(item instanceof Subject) {
223 					subject = (Subject)item;
224 				}else if(item instanceof Membership) {
225 					
226 						subject = ((Membership)item).getMember().getSubject();
227 				}else if(item instanceof Member) {
228 					
229 						subject = ((Member)item).getSubject();
230 				}else{
231 					continue;
232 				}
233 			}catch(MemberNotFoundException mnfe) {
234 				continue;
235 			}catch(SubjectNotFoundException snfe) {
236 				continue;
237 			}
238 			printSubject(writer,name,subject,sep);
239 		}
240 		return;
241 	}
242 	
243 	private void printHeader(PrintWriter writer,String name,String sep) throws Exception,IOException {
244 		List headers = getHeaders(name);
245 		for(int i=0;i<headers.size();i++) {
246 			if(i>0) writer.print(sep);
247 			if(quote) writer.print('"');
248 			writer.print(headers.get(i));
249 			if(quote) writer.print('"');
250 		}
251 		if(!headers.isEmpty()) writer.print("\n");
252 	}
253 	
254 	private void printSubject(PrintWriter writer,String name,Subject subject,String sep) throws Exception,IOException {
255 		List fields = getFields(name,subject);
256 		if(fields==null) return;
257 		Element field;
258 		String val="???";
259 		String fieldName;
260 		for(int i=0;i<fields.size();i++) {
261 			field=(Element)fields.get(i);
262 			
263 			if(i>0) writer.print(sep);
264 			if(quote) writer.print('"');
265 			if(field.hasAttribute("value")) {
266 				writer.print(field.getAttribute("value"));
267 			}else{
268 				val="???";
269 				fieldName=field.getAttribute("name");
270 				if("name".equals(fieldName)) {
271 					val=subject.getName();
272 				}else if("description".equals(fieldName)) {
273 					val=subject.getDescription();
274 				}else if("id".equals(fieldName)) {
275 						val=subject.getId();
276 				}else {
277 					try {
278 						val=subject.getAttributeValue(fieldName);
279 					}catch(Exception e){}
280 				}
281 				if(val==null) val="???";
282 				if(quote && val.indexOf("\"") > -1) {
283 					writer.print(val.replaceAll("\"","\"\""));
284 				}else{
285 					writer.print(val);
286 				}
287 			}
288 			if(quote) writer.print('"');
289 		}
290 		writer.print("\n");
291 	}

293 	private Element getFormat(String name) throws Exception{
294 		init();
295 		return (Element)formatCache.get(name);
296 	}
297 	
298 	private List getFields(String format,Subject subject) throws Exception{
299 		String lookup = format + ":" + subject.getSource().getId();
300 		if("".equals(fieldsCache.get(lookup))) return null;
301 		List fields = (List)fieldsCache.get(lookup);
302 		if(fields==null) {
303 			boolean ok=false;
304 			Element formatElement = getFormat(format);
305 			NodeList nl = formatElement.getElementsByTagName("source");
306 			NodeList fieldList;
307 			Element fieldElement;
308 			Element sourceElement;
309 			
310 			for (int i=0;i<nl.getLength();i++) {
311 				sourceElement = (Element)nl.item(i);
312 				fieldList = sourceElement.getElementsByTagName("field");
313 				List sourceFields = new ArrayList();
314 				fieldsCache.put(format + ":" + sourceElement.getAttribute("id"),sourceFields);
315 				if(subject.getSource().getId().equals(sourceElement.getAttribute("id"))) {
316 					ok=true;
317 					fields = sourceFields;
318 				}
319 				for (int j=0;j<fieldList.getLength();j++) {
320 					sourceFields.add(fieldList.item(j));
321 				}
322 			}
323 			if(!ok) {
324 				fieldsCache.put(lookup,"");
325 				return null;
326 			}
327 		}
328 		return fields;
329 	}
330 	
331 	private List getHeaders(String format) throws Exception{
332 		List headers=new ArrayList();
333 		Element fe = getFormat(format);
334 		NodeList nl = fe.getElementsByTagName("header");
335 		Element he;
336 		for(int i=0;i<nl.getLength();i++) {
337 			he = (Element) nl.item(i);
338 			headers.add(he.getAttribute("name"));
339 		}
340 		return headers;
341 	}

343 	public String getContentType(String format) throws Exception{
344 		String ct = getFormat(format).getAttribute("content-type");
345 		if("".equals(ct)) ct=null;
346 		return ct;
347 	}

349 	public String getExtension(String format) throws Exception{
350 		String ext = getFormat(format).getAttribute("extension");
351 		if("".equals(ext)) ext=null;
352 		return ext;
353 	}
