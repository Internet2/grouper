117   /**
118    * 
119    * @return pro
120    */
121   public int searchDelayProcessed() {
122     if (this.searchDelay == null || this.searchDelay < 0) {
123       return 500;
124     }
125     return this.searchDelay;
126   }
127   
128   /**
129    * default downarrow to false
130    * @return if has down arrow
131    */
132   public boolean hasDownArrowProcessed() {
133     if (this.hasDownArrow == null) {
134       return false;
135     }
136     return this.hasDownArrow;
137   }
