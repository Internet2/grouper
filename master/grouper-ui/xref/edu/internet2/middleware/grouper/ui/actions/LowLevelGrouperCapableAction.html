122 	/**
123 	 * Makes HttpSession and GrouperSession available to subclasses
124 	 * Also handles pageSize parameter and times how long is spent in an action
125 	 */
126 	public ActionForward execute(ActionMapping mapping, ActionForm form,
127 		      HttpServletRequest request, HttpServletResponse response) throws Exception{
128 	  
129 	   boolean needsThreadLocalInit = GrouperUiFilter.retrieveHttpServletRequest() == null;
130 	    if (needsThreadLocalInit) {
131 	      request = new GrouperRequestWrapper(request);
132         GrouperUiFilter.initRequest((GrouperRequestWrapper)request, response);
133 	    }
134 	    try {
135 	  
136   			HttpSession session = request.getSession();
137   			GrouperSession grouperSession = (GrouperSession) session.getAttribute("edu.intenet2.middleware.grouper.ui.GrouperSession");
138   						
139   			DynaActionForm dummyForm = (DynaActionForm)form;
140   			if(form!=null)request.setAttribute("grouperForm",form);
141   			//tell grouper session if we are in admin mode or user mode
142   	    Boolean originalConsiderIfWheelMember = grouperSession == null ? null : grouperSession.isConsiderIfWheelMember();
143   			
144   	    //are we acting as self or wheel?
145   	    if (grouperSession != null) {
146   	      
147 	        if (isWheelGroupMember(session)) {
148 	          grouperSession.setConsiderIfWheelMember(Boolean.TRUE.equals(UIThreadLocal.get("isActiveWheelGroupMember")));
149 	        } else {
150 	          // we'll set this back to the default
151 	          grouperSession.setConsiderIfWheelMember(true);
152 	        }
153   	      
154   	    }	    
155   	    try {
156   			ActionForward forward =  grouperExecute(mapping,form,request,response,session,grouperSession);
157   			
158   			return forward;
159   	    } finally {
160   	      if (grouperSession != null && originalConsiderIfWheelMember != null) {
161   	        grouperSession.setConsiderIfWheelMember(originalConsiderIfWheelMember);
162   	}
163   	    }
164 	    } finally {
165 	      if (needsThreadLocalInit) {
166 	        GrouperUiFilter.finallyRequest();
167 	      }
168 	    }
169 	}

393 	protected boolean processFlatForMode(String mode,HttpServletRequest request,HttpSession session) {
394 		GrouperSession s = SessionInitialiser.getGrouperSession(session);
395 		RepositoryBrowser rb = getRepositoryBrowser(s, session);
396 		if(!rb.isFlatCapable()) {
397 			request.setAttribute("isFlat",Boolean.FALSE);
398 			return false;
399 		}
400 		String flat = request.getParameter("flat");
401 		Boolean isFlat = (Boolean)session.getAttribute("isFlat" + mode);
402 		if((flat==null || flat.length()==0)) {
403 			if(isFlat==null) isFlat=Boolean.FALSE; 
404 		}else isFlat=new Boolean("true".equals(flat));
405 		session.setAttribute("isFlat" + mode,isFlat);
406 		request.setAttribute("isFlat",isFlat);
407 		return isFlat.booleanValue();
408 	}

439 	protected boolean isWheelGroupMember(HttpSession session) {
440 		if(isEmpty(SessionInitialiser.getAuthUser(session))) return false;
441 		if(Boolean.TRUE.equals(session.getAttribute("isWheelGroupMember"))) return true;
442 		if(Boolean.FALSE.equals(session.getAttribute("isWheelGroupMember"))) return false;
443 		Object obj= session.getServletContext().getAttribute("wheelGroup");
444 		if(Boolean.FALSE.equals(obj)) {
445 			session.setAttribute("isWheelGroupMember",Boolean.FALSE);
446 			return false;
447 		}
448 		if(obj==null) {
449 			if("true".equals(GrouperConfig.getProperty("groups.wheel.use"))) {
450 				try {
451 					Subject subj = SubjectFinder.findById("GrouperSystem", true);
452 					GrouperSession s = GrouperSession.start(subj);
453 					obj=GroupFinder.findByName(s,GrouperConfig.getProperty("groups.wheel.group"), true);
454 					session.getServletContext().setAttribute("wheelGroup",obj);
455 				}catch(Exception e) {
456 					session.setAttribute("isWheelGroupMember",Boolean.FALSE);
457 					return false;
458 				}
459 			}
460 		}
461 		
462 		Group wheelGroup = (Group)obj;
463 		GrouperSession s  =SessionInitialiser.getGrouperSession(session);
464 		if(s !=null && obj!=null && wheelGroup.hasMember(s.getSubject())) {
465 			session.setAttribute("isWheelGroupMember",Boolean.TRUE);
466 			return true;
467 		}
468 		session.setAttribute("isWheelGroupMember",Boolean.FALSE);
469 		return false;
470 		
471 	}

478 		HttpSession session = request.getSession();
479 		List savedAsMaps =  new ArrayList();
480 		Set savedSubjects=getSavedSubjects(session);
481 		Subject subj;
482 		Group group;
483 		Iterator it = savedSubjects.iterator();
484 		GrouperSession grouperSession = (GrouperSession) session.getAttribute("edu.intenet2.middleware.grouper.ui.GrouperSession");
485 		
486 		while(it.hasNext()) {
487 			subj=(Subject)it.next();
488 			if("group".equals(subj.getType().getName())) {
489 				try {
490 					group=GroupFinder.findByUuid(grouperSession,subj.getId(), true);
491 					savedAsMaps.add(GrouperHelper.group2Map(grouperSession,group));
492 				}catch(GroupNotFoundException e) {
493 					it.remove();
494 				}
495 			}
496 		}
497 		String sortContext="search";
498 		String field = (String)session.getAttribute("groupSearchResultField");
499 		if(field !=null) {
500 			sortContext="search:" + field;
501 		}
502 		savedAsMaps=sort(savedAsMaps,request,sortContext, -1, null);
503 		request.setAttribute("savedSubjects",new ArrayList(savedAsMaps));
504 		request.setAttribute("savedSubjectsSize",new Integer(savedAsMaps.size()));	
505 	}

598 	 * This method assumes that there is a HttpSession attribute, 'GrouperComparator',
599 	 * which is an implementation of the {@link GrouperComparator} interface. If not present
600 	 * this method will instantiate an implementation instance using the value for the
601 	 * media.properties key 'comparator.impl'. The media.properties key 'comparator.sort.limit'
602 	 * defines the maximum input size that will be sorted. If the input Collection is larger
603 	 * than this limit, it is effectively returned as is, though if the Collection was not a List
604 	 * it will be returned as an ArrayList.
605 	 * @see edu.internet2.middleware.grouper.ui.DefaultComparatorImpl
606 	 * @param input the Collection to sort
607 	 * @param request the current request object
608 	 * @param context the context in which sorting is taking place
609 	 * @param collectionSize is the total size of the collection, or -1 if the collection is the entire thing
610 	 * @param searchTerm is the term to search for or null if not applicable
611 	 * @return the input Collection as a sorted List
612 	 */
613 	public static List sort(Collection input,HttpServletRequest request,
614 	    String context, int collectionSize, String searchTerm) {
615 		HttpSession session = request.getSession();
616 		GrouperComparator gc = (GrouperComparator)session.getAttribute("GrouperComparator");
617 		ResourceBundle config = GrouperUiFilter.retrieveSessionMediaResourceBundle();
618 		String maxStr=config.getString("comparator.sort.limit");
619 		int max=Integer.parseInt(maxStr);
620 		if(gc==null) {
621 			String comparatorClass = null;
622 			try {
623 				comparatorClass=config.getString("comparator.impl");
624 				
625 			}catch(Exception e){}
626 			if(comparatorClass==null) comparatorClass="edu.internet2.middleware.grouper.ui.DefaultComparatorImpl";
627 			try {
628 				gc = (GrouperComparator)Class.forName(comparatorClass).newInstance();
629 			}catch(Exception e) {
630 				throw new IllegalStateException("Cannot create " + comparatorClass + " instance");
631 			}
632 			session.setAttribute("GrouperComparator",gc);
633 		}
634 		gc.setContext(context);
635 		
636 		List toSort=null;
637 		if(input instanceof List) {
638 			toSort=(List)input;
639 		}else{
640 			toSort=new ArrayList(input);
641 		}
642 		int toSortSize = collectionSize == -1 ? toSort.size() : collectionSize;
643     if(toSortSize<=max) Collections.sort(toSort,gc);
644     
645     //we need to bring important matches to the top
646     boolean isSubjects = true;
647     for (Object item : toSort) {
648       if (!(item instanceof Subject)) {
649         isSubjects = false;
650       }
651     }
652     if (isSubjects) {
653       Set<Subject> subjectsOut = SubjectHelper.sortSetForSearch(toSort, searchTerm);
654       //avoid a null pointer...
655       toSort = GrouperUtil.length(subjectsOut) > 0 ? new ArrayList<Subject>(subjectsOut) : toSort;
656     }
657 		return toSort;
658 	}
