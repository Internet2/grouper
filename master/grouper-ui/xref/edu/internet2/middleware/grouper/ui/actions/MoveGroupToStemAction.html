20  import javax.servlet.http.HttpSession;

22  import org.apache.struts.action.ActionForm;
23  import org.apache.struts.action.ActionForward;
24  import org.apache.struts.action.ActionMapping;
25  import org.apache.struts.action.DynaActionForm;
26  
27  import edu.internet2.middleware.grouper.Group;
28  import edu.internet2.middleware.grouper.GroupFinder;
29  import edu.internet2.middleware.grouper.GrouperHelper;
30  import edu.internet2.middleware.grouper.GrouperSession;
31  import edu.internet2.middleware.grouper.Stem;
32  import edu.internet2.middleware.grouper.StemFinder;
33  import edu.internet2.middleware.grouper.ui.Message;
34  
35  /**
36   * @author shilen
37   * @version $Id: MoveGroupToStemAction.java,v 1.1 2009-05-08 12:03:37 shilen Exp $
38   */
39  public class MoveGroupToStemAction extends GrouperCapableAction {
40  
41    static final private String FORWARD_MoveGroup = "MoveGroupToStem";
42  
43    public ActionForward grouperExecute(ActionMapping mapping, ActionForm form,
44        HttpServletRequest request, HttpServletResponse response,
45        HttpSession session, GrouperSession grouperSession)
46        throws Exception {
47      
48      DynaActionForm groupForm = (DynaActionForm) form;
49      
50      String curNode = (String)groupForm.get("stemId");
51      Stem destinationStem = StemFinder.findByUuid(grouperSession, curNode, true);
52  
53      // get the options selected by the user for the group move
54      String[] selections = request.getParameterValues("selections");
55      
56      // find the group to move
57      String groupSelection = request.getParameter("groupSelection");
58      if (groupSelection.equals("other")) {
59        groupSelection = request.getParameter("otherGroupSelection");
60      }
61      
62      if (groupSelection == null || groupSelection.equals("")) {
63        request.setAttribute("message", new Message(
64            "groups.message.error.invalid-group", true));
65        return mapping.findForward(FORWARD_MoveGroup);
66      }
67  
68      Group group = GroupFinder.findByName(grouperSession, groupSelection, false);
69      
70      if (group == null) {
71        request.setAttribute("message", new Message(
72            "groups.message.error.invalid-group", true));
73        return mapping.findForward(FORWARD_MoveGroup);
74      }
75  
76      GrouperHelper.moveGroup(group, destinationStem, selections);
77  
78      request.setAttribute("message", new Message(
79          "groups.message.group-moved", group.getName()));
80      
81      return new ActionForward("/populate" + getBrowseMode(session) + "Groups.do");
82  
83    }
84  
85  }
