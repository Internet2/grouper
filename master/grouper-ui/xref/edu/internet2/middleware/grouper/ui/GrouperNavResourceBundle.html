4   package edu.internet2.middleware.grouper.ui;
5   
6   import java.util.Collections;
7   import java.util.Enumeration;
8   import java.util.Locale;
9   import java.util.ResourceBundle;
10  
11  import javax.servlet.http.HttpServletRequest;
12  
13  import edu.internet2.middleware.grouper.util.GrouperUtil;
14  import edu.internet2.middleware.grouperClient.config.GrouperUiTextConfig;
15  
16  
17  /**
18   * @author mchyzer
19   *
20   */
21  public class GrouperNavResourceBundle extends ResourceBundle {
22  
23    /**
24     * 
25     */
26    private GrouperUiTextConfig grouperUiTextConfig = null;
27    
28    /**
29     * @param locale 
30     * 
31     */
32    public GrouperNavResourceBundle(Locale locale) {
33      //for some reason this isnt always the request one
34      HttpServletRequest httpServletRequest = GrouperUiFilter.retrieveHttpServletRequest();
35      if (httpServletRequest != null) {
36        locale = GrouperUtil.defaultIfNull(httpServletRequest.getLocale(), locale);
37      }
38      this.grouperUiTextConfig = GrouperUiTextConfig.retrieveText(locale); 
39    }
40  
41    /**
42     * @see java.util.ResourceBundle#handleGetObject(java.lang.String)
43     */
44    @Override
45    protected Object handleGetObject(String key) {
46      return this.grouperUiTextConfig.propertyValueString(key);
47    }
48  
49    /**
50     * @see java.util.ResourceBundle#getKeys()
51     */
52    @Override
53    public Enumeration<String> getKeys() {
54      return Collections.enumeration(this.grouperUiTextConfig.propertyNames());
55    }
56  
57  }
