57  /**
58   * Wraps a GrouperStem- allows non persistent values to be stored for the UI and
59   * works well with JSTL
60   * <p />
61   * 
62   * @author Gary Brown.
63   * @version $Id: AuditEntryAsMap.java,v 1.4 2009-10-01 13:43:13 isgwb Exp $
64   */
65  public class AuditEntryAsMap extends ObjectAsMap {
66  	
67  	
68  	protected AuditEntry entry = null;
69  
70  	protected final static String objType = "AuditEntry";
71  
72  	private GrouperSession grouperSession = null;
73  	
74  	private static Set<String> groupIdLabels = new HashSet<String>();
75  	private static Set<String> stemIdLabels = new HashSet<String>();
76  	private static Set<String> memberIdLabels = new HashSet<String>();
77  	private static Set<String> idLabels = new HashSet<String>();
78  	static {
79  		groupIdLabels.add("groupId");
80  		groupIdLabels.add("groupUuid");		
81  		groupIdLabels.add("oldGroupId");
82  		groupIdLabels.add("newGroupId");
83  		groupIdLabels.add("leftFactorId");
84  		groupIdLabels.add("rightFactorId");
85  		groupIdLabels.add("ownerId");
86  		stemIdLabels.add("stemId");
87  		stemIdLabels.add("oldStemId");
88  		stemIdLabels.add("newStemId");
89  		stemIdLabels.add("stemUuid");
90  		stemIdLabels.add("newStemUuid");
91  		memberIdLabels.add("memberId");
92  		memberIdLabels.add("newMemberId");
93  		memberIdLabels.add("oldMemberId");
94  		idLabels.add("group");
95  		idLabels.add("stem");
96  		idLabels.add("member");
97  		
98  	}
99  	
100 	protected AuditEntryAsMap() {}
101 
102 	/**
103 	 * @param stem Stem to wrap
104 	 * @param s GrouperSession for authenticated user
105 	 */
106 	public AuditEntryAsMap(AuditEntry entry, GrouperSession s) {
107 		super();
108 		grouperSession=s;
109 		init(entry);
110 	}
111 	
112 	protected void init(AuditEntry entry) {
113 		dynaBean = new WrapDynaBean(entry);
114 		super.objType = objType;
115 		if (entry == null)
116 			throw new NullPointerException(
117 					"Cannot create AuditEntryAsMap with a null AuditEntry");
118 		this.entry = entry;
119 		wrappedObject = entry;
120 
121 		Map<String, Object> fields = new HashMap<String, Object>();
122 		Map<String, Object> fieldObjects = new HashMap<String, Object>();
123 		for (String label :  entry.getAuditType().labels()) {
124 			String fieldName = entry.getAuditType().retrieveAuditEntryFieldForLabel(label);
125 	        //Object value = GrouperUtil.fieldValue(this, fieldName);
126 			fields.put(label,entry.fieldValue(fieldName));
127 			Object obj = instantiateFieldObject(label,entry.fieldValue(fieldName));
128 			if(obj!=null) {
129 				fieldObjects.put(label, obj);
130 			}
131 		}
132 		this.put("fields", fields);
133 		this.put("fieldObjects", fieldObjects);
134 	}
135 	
136 	protected Object instantiateFieldObject(String label,Object value) {
137 		if(value==null) {
138 			return null;
139 		}
140 		if(groupIdLabels.contains(label) || 
141 				("id".equals(label) 
142 						&& ("group".equals(entry.getAuditType().getAuditCategory()) 
143 								))) {
144 			Group g = GroupFinder.findByUuid(grouperSession, (String)value, false);
145 			if(g==null) {
146 				return g;
147 			}
148 			return ObjectAsMap.getInstance("GroupAsMap", g,grouperSession);
149 		}
150 		
151 		if(stemIdLabels.contains(label) || ("id".equals(label) && "stem".equals(entry.getAuditType().getAuditCategory()))) {
152 			Stem s = StemFinder.findByUuid(grouperSession, (String)value, false);
153 			if(s==null) {
154 				return s;
155 			}
156 			return ObjectAsMap.getInstance("StemAsMap", s, grouperSession);
157 		}
158 		
159 		if(memberIdLabels.contains(label) || ("id".equals(label) && "member".equals(entry.getAuditType().getAuditCategory()))) {
160 			Member m = MemberFinder.findByUuid(grouperSession, (String)value, false);
161 			if(m==null) {
162 				return m;
163 			}
164 			return ObjectAsMap.getInstance("SubjectAsMap", m.getSubject());
165 		}
166 		
167 		if(("id".equals(label) && "groupType".equals(entry.getAuditType().getAuditCategory())) ||
168 				("groupTypeId".equals(label) && "groupField".equals(entry.getAuditType().getAuditCategory()))) {
169 			return GroupTypeFinder.findByUuid((String)value, false);
170 		}
171 		
172 		return null;
173 	}
174 
175 	/*
176 	 * (non-Javadoc)
177 	 * 
178 	 * @see java.util.Map#get(java.lang.Object)
179 	 */
180 	public Object get(Object key) {
181 		//Map would override GrouperGroup values
182 		Object obj = super.get(key);
183 		if (obj == null) {
184 			if("duration".equals(key)) {
185 				double l = Math.round(((AuditEntry)wrappedObject).getDurationMicroseconds()/10000000)/10.0;
186 				return l * 10.0;
187 				
188 			}
189 			if("loggedInMember".equals(key)) {
190 				Member m = MemberFinder.findByUuid(grouperSession, entry.getLoggedInMemberId(), false);
191 				if(m==null) {
192 					return m;
193 				}
194 				return ObjectAsMap.getInstance("SubjectAsMap", m.getSubject());
195 			}
196 			if("formatLastUpdated".equals(key)) {
197 				Date date = entry.getLastUpdated();
198 				String df = getDateFormat();
199 				SimpleDateFormat sdf = new SimpleDateFormat(df);
200 				return sdf.format(date);
201 			}
202 			if("actAsMember".equals(key)) {
203 				String actAs = entry.getActAsMemberId();
204 				if(actAs==null) {
205 					return null;
206 				}
207 				Member m = MemberFinder.findByUuid(grouperSession, actAs, false);
208 				if(m==null) {
209 					return m;
210 				}
211 				return ObjectAsMap.getInstance("SubjectAsMap", m.getSubject());
212 			}
213 			obj=getByIntrospection(key);
214 		}
215 		if (obj == null)
216 			obj = "";
217 		return obj;
218 	}
