101   /**
102    * try to change a paging query, note it will add one to the resultSetLimit so that
103    * the caller can see if there are too many records
104    * @param query
105    * @param conn
106    * @param resultSetLimit
107    * @return the query with paging or original query if cant change it
108    */
109   public String tryToChangeQuery(String query, Connection conn, int resultSetLimit) {
110     
111     //lets see if we are restricted
112     if (!this.changeSearchQueryForMaxResults || resultSetLimit <= 0) {
113       return query;
114     }
115     
116     JdbcDatabaseType jdbcDatabaseType = JdbcDatabaseType.resolveDatabaseType(conn);
117 
118     if (jdbcDatabaseType == null) {
119       return query;
120     }
121     
122     //we need to add one so we know if there are too many results...
123     resultSetLimit++;
124     
125     String newQuery = jdbcDatabaseType.pageQuery(query, resultSetLimit);
126     
127     if (StringUtils.isBlank(newQuery)) {
128       return query;
129     }
130     
131     return newQuery;
132   }

169   @Override
170   public Subject getSubject(String id1, boolean exceptionIfNull)
171       throws SubjectNotFoundException, SubjectNotUniqueException {
172     
173     //do the batched one
174     if (this.useInClauseForIdAndIdentifier) {
175       
176       Map<String, Subject> result = this.getSubjectsByIds(SubjectUtils.toSet(id1));
177       
178       if (SubjectUtils.length(result) == 1) {
179         return result.get(id1);
180       }
181       
182       if (SubjectUtils.length(result) > 1) {
183         throw new SubjectNotUniqueException("Not unique by id: " + id1);
184       }
185       
186       if (exceptionIfNull) {
187         throw new SubjectNotFoundException("Cant find subject by id: " + id1);
188       }
189       
190       return null;
191     }
192     try {
193       return uniqueSearch(id1, "searchSubject");
194     } catch (SubjectNotFoundException snfe) {
195       if (exceptionIfNull) {
196         throw snfe;
197       }
198       return null;
199     }
200     
201   }

239   /**
240    * Perform a search for a unique subject.
241    * 
242    * @param id1
243    * @param searchType
244    * @return subject
245    * @throws SubjectNotFoundException
246    * @throws SubjectNotUniqueException
247    * @throws InvalidQueryException 
248    */
249   private Subject uniqueSearch(String id1, String searchType)
250       throws SubjectNotFoundException, SubjectNotUniqueException, InvalidQueryException {
251     Subject subject = null;
252     
253     Search search = getSearch(searchType);
254     if (search == null) {
255       log.error("searchType: \"" + searchType + "\" not defined.");
256       return subject;
257     }
258     Connection conn = null;
259     PreparedStatement stmt = null;
260     JdbcConnectionBean jdbcConnectionBean = null;
261     try {
262       jdbcConnectionBean = this.jdbcConnectionProvider.connectionBean();
263       conn = jdbcConnectionBean.connection();
264       
265       queryCountforTesting++;
266       
267       stmt = prepareStatement(search, conn, false, false);
268       ResultSet rs = getSqlResults(id1, stmt, search);
269       subject = createUniqueSubject(rs, search, id1, search.getParam("sql"));
270       jdbcConnectionBean.doneWithConnection();
271     } catch (SQLException ex) {
272       String error = "problem in sources.xml source: " + this.getId() + ", sql: "
273           + search.getParam("sql") + ", " + id1 + ", " + searchType;
274       try {
275         jdbcConnectionBean.doneWithConnectionError(ex);
276       } catch (RuntimeException e) {
277         log.error(error, e);
278       }
279       throw new SourceUnavailableException(error, ex);
280     } finally {
281       closeStatement(stmt);
282       if (jdbcConnectionBean != null) {
283         jdbcConnectionBean.doneWithConnectionFinally();
284       }
285     }
286     if (subject == null) {
287       throw new SubjectNotFoundException("Subject " + id1 + " not found.");
288     }
289     return subject;
290   }

314   /**
315    * helper for query
316    * @param searchValue 
317    * @param firstPageOnly 
318    * @return the result object, never null
319    */
320   private SearchPageResult searchHelper(String searchValue, boolean firstPageOnly) {
321 
322     //if this is a search and not by id or identifier, strip out the status part
323     {
324       SubjectStatusResult subjectStatusResult = null;
325       
326       //see if we are doing status
327       SubjectStatusProcessor subjectStatusProcessor = new SubjectStatusProcessor(searchValue, this.getSubjectStatusConfig());
328       subjectStatusResult = subjectStatusProcessor.processSearch();
329 
330       //strip out status parts
331       searchValue = subjectStatusResult.getStrippedQuery();
332     }      
333     
334 
335     Set<Subject> result = new LinkedHashSet<Subject>();
336     boolean tooManyResults = false;
337     Search search = getSearch("search");
338     if (search == null) {
339       log.error("searchType: \"search\" not defined.");
340       return new SearchPageResult(false, result);
341     }
342     String throwErrorOnFindAllFailureString = this.getInitParam("throwErrorOnFindAllFailure");
343     boolean throwErrorOnFindAllFailure = SubjectUtils.booleanValue(throwErrorOnFindAllFailureString, true);
344 
345     Connection conn = null;
346     PreparedStatement stmt = null;
347     JdbcConnectionBean jdbcConnectionBean = null;
348     try {
349       jdbcConnectionBean = this.jdbcConnectionProvider.connectionBean();
350 
351       if (failOnSearchForTesting) {
352         throw new RuntimeException("failOnSearchForTesting");
353       }
354       
355       conn = jdbcConnectionBean.connection();
356       
357       queryCountforTesting++;
358       
359       stmt = prepareStatement(search, conn, true, firstPageOnly);
360       ResultSet rs = getSqlResults(searchValue, stmt, search);
361       if (rs == null) {
362         return new SearchPageResult(false, result);
363       }
364       while (rs.next()) {
365         Subject subject = createSubject(rs, search.getParam("sql"));
366         result.add(subject);
367         //if we are at the end of the page
368         if (firstPageOnly && this.maxPage != null && result.size() >= this.maxPage) {
369           tooManyResults = true;
370           break;
371         }
372         if (this.maxResults != null && result.size() > this.maxResults) {
373           throw new SubjectTooManyResults(
374               "More results than allowed: " + this.maxResults 
375               + " for search '" + search + "'");
376         }
377       }
378       jdbcConnectionBean.doneWithConnection();
379     } catch (Exception ex) {
380       try {
381         jdbcConnectionBean.doneWithConnectionError(ex);
382       } catch (RuntimeException re) {
383         if (!(ex instanceof SubjectTooManyResults)) {
384           log.error("Problem with source: " + this.getId() + ", and query: '" + searchValue + "'", re);
385         }
386       }
387       if (ex instanceof SubjectTooManyResults) {
388         throw (SubjectTooManyResults)ex;
389       }
390       if (!throwErrorOnFindAllFailure) {
391         log.error(ex.getMessage() + ", source: " + this.getId() + ", sql: "
392           + search.getParam("sql"), ex);
393       } else {
394         throw new SourceUnavailableException(ex.getMessage() + ", source: " + this.getId() + ", sql: "
395             + search.getParam("sql"), ex);
396       }
397     } finally {
398       closeStatement(stmt);
399       if (jdbcConnectionBean != null) {
400         jdbcConnectionBean.doneWithConnectionFinally();
401       }
402     }
403     return new SearchPageResult(tooManyResults, result);
404   }

406   /**
407    * Create a subject from the current row in the resultSet
408    * 
409    * @param rs
410    * @param sql 
411    * @return subject
412    */
413   private Subject createSubject(ResultSet rs, String sql) {
414     String name1 = "";
415     String subjectID = "";
416     String description = "";
417     Subject subject = null;
418     try {
419       subjectID = retrieveString(rs, this.subjectIDAttributeName,
420           "SubjectID_AttributeType", sql);
421       name1 = retrieveString(rs, this.nameAttributeName, "Name_AttributeType", sql);
422       if (!this.descriptionAttributeName.equals("")) {
423         description = retrieveString(rs, this.descriptionAttributeName,
424             "Description_AttributeType", sql);
425       }
426       Map<String, Set<String>> attributes1 = loadAttributes(rs);
427       subject = new SubjectImpl(subjectID, name1, description, this.getSubjectType().getName(),
428           this.getId(), attributes1);
429     } catch (SQLException ex) {
430       throw new SourceUnavailableException("SQLException occurred: " + ex.getMessage() + ": " + sql, ex);
431     }
432     return subject;
433   }

459   /**
460    * Create a unique subject from the resultSet.
461    * 
462    * @param rs
463    * @param search
464    * @param searchValue
465    * @param sql 
466    * @return subject
467    * @throws SubjectNotFoundException
468    * @throws SubjectNotUniqueException
469    */
470   private Subject createUniqueSubject(ResultSet rs, Search search, String searchValue,
471       String sql) throws SubjectNotFoundException, SubjectNotUniqueException {
472     Subject subject = null;
473     try {
474       if (rs == null || !rs.next()) {
475         String errMsg = "No results: " + search.getSearchType() + " searchValue: "
476             + searchValue;
477         throw new SubjectNotFoundException(errMsg);
478       }
479       subject = createSubject(rs, sql);
480       if (rs.next()) {
481         String errMsg = "Search is not unique:"
482             + rs.getString(this.subjectIDAttributeName) + "\n";
483         throw new SubjectNotUniqueException(errMsg);
484       }
485     } catch (SQLException ex) {
486       throw new SourceUnavailableException("SQLException occurred: " + ex.getMessage() + ": " + sql, ex);
487     }
488     return subject;
489 
490   }

492   /**
493    * Create unique subjects from the resultSet.
494    * 
495    * @param rs
496    * @param search
497    * @param searchValue
498    * @param sql 
499    * @return subject
500    * @throws SubjectNotFoundException
501    * @throws SubjectNotUniqueException
502    */
503   private Map<String, Subject> createUniqueSubjects(ResultSet rs, Search search, List<String> searchValues,
504       String sql, boolean useIdentifiersInMatch) throws SubjectNotFoundException, SubjectNotUniqueException {
505 
506     Map<String, Subject> results = new HashMap<String, Subject>();
507     Set<String> searchValuesSet = new HashSet<String>(searchValues);
508 
509     if (rs != null) {
510       try {
511         RESULT_SET_LOOP: while (rs.next()) {
512           
513           Subject subject = createSubject(rs, sql);
514 
515           //maybe id match
516           if (searchValues.contains(subject.getId())) {
517             results.put(subject.getId(), subject);
518           } else {
519             
520             //find out which id or identifier found this one
521             for (String identifierAttribute : this.identifierAttributes) {
522               Set<String> values = subject.getAttributeValues(identifierAttribute);
523               if (SubjectUtils.length(values) > 0) {
524                 for (String value: values) {
525                   //we found a match
526                   if (searchValuesSet.contains(value)) {
527                     results.put(value, subject);
528                     //dont want dupes
529                     searchValuesSet.remove(value);
530                     continue RESULT_SET_LOOP;
531                   }
532                 }
533               }
534               
535             }
536 
537             //if we made it this far there is a problem
538             throw new InvalidQueryException("Why is this subject not able to be " +
539                 "referenced by id or identifier (do you need to add " +
540                 "identifierAttributes to your sources.xml???) " + SubjectUtils.subjectToString(subject) );
541 
542           }
543           
544         }
545       } catch (SQLException ex) {
546         throw new SourceUnavailableException("SQLException occurred: " + ex.getMessage() + ": " + sql, ex);
547       }
548 
549     }
550     return results;
551 
552   }

554   /**
555    * Prepare a statement handle from the search object.
556    * 
557    * @param search
558    * @param conn
559    * @param searchAll true if a searchAll method
560    * @param firstPageOnly 
561    * @return the prepared statement
562    * @throws InvalidQueryException
563    * @throws SQLException
564    */
565   protected PreparedStatement prepareStatement(Search search, Connection conn, boolean searchAll, boolean firstPageOnly)
566       throws InvalidQueryException, SQLException {
567     String sql = search.getParam("sql");
568     if (sql == null) {
569       throw new InvalidQueryException("No sql parameter for search type "
570           + search.getSearchType() + ", source: " + this.getId());
571     }
572     if (sql.contains("%TERM%")) {
573       throw new InvalidQueryException("%TERM%. Possibly old style SQL query, source: "
574           + this.getId() + ", sql: " + sql);
575     }
576     String numParametersString = search.getParam("numParameters");
577     //not required since it will just count the number of question marks
578     if (!StringUtils.isBlank(numParametersString)) {
579       try {
580         Integer.parseInt(numParametersString);
581       } catch (NumberFormatException e) {
582         throw new InvalidQueryException(
583             "Non-numeric numParameters parameter specified, source: " + this.getId()
584                 + ", sql: " + sql);
585       }
586     }
587     
588     if (searchAll) {
589       Integer pageSize = resultSetLimit(firstPageOnly, this.getMaxPage(), this.maxResults);
590       if (pageSize != null && pageSize > 0) {
591         
592         sql = this.tryToChangeQuery(sql, conn, pageSize);
593       }
594     }
595     
596     PreparedStatement stmt = conn.prepareStatement(sql);
597     return stmt;
598   }

600   /**
601    * Set the parameters in the prepared statement and execute the query.
602    * 
603    * @param searchValue
604    * @param stmt
605    * @param search
606    * @return resultSet
607    * @throws SQLException 
608    */
609   protected ResultSet getSqlResults(String searchValue, PreparedStatement stmt,
610       Search search) throws SQLException {
611     ResultSet rs = null;
612     String sql = search.getParam("sql");
613     String numParametersString = search.getParam("numParameters");
614     //default to the number of question marks
615     int numParameters = StringUtils.isBlank(numParametersString) ? StringUtils
616         .countMatches(sql, "?") : Integer.parseInt(numParametersString);
617     for (int i = 1; i <= numParameters; i++) {
618       try {
619         stmt.setString(i, searchValue);
620       } catch (SQLException e) {
621         SubjectUtils
622             .injectInException(
623                 e,
624                 "Error setting param: "
625                     + i
626                     + " in source: "
627                     + this.getId()
628                     + ", in query: "
629                     + sql
630                     + ", "
631                     + e.getMessage()
632                     + ", maybe not enough question marks "
633                     + "(bind variables) are in query, or the number of question marks in the query is "
634                     + "not the same as the number of parameters (might need to set the optional param numParameters), "
635                     + "or the param 'numParameters' in sources.xml for that query is incorrect");
636         throw e;
637       }
638     }
639     rs = stmt.executeQuery();
640     return rs;
641   }

643   /**
644    * Set the parameters in the prepared statement and execute the query.
645    * 
646    * @param searchValue
647    * @param stmt
648    * @param search
649    * @return resultSet
650    * @throws SQLException 
651    */
652   protected ResultSet getSqlResults(List<String> batchIdsOrIdentifiers, PreparedStatement stmt, 
653       int numParameters, String sql) throws SQLException {
654 
655     ResultSet rs = null;
656     int paramIndex = 1;
657     for (String idOrIdentifier : batchIdsOrIdentifiers) {
658       
659       //there might be more than one param for the same value
660       for (int i = 1; i <= numParameters; i++) {
661         try {
662           stmt.setString(paramIndex++, idOrIdentifier);
663         } catch (SQLException e) {
664           SubjectUtils
665               .injectInException(
666                   e,
667                   "Error setting param: "
668                       + i
669                       + " in source: "
670                       + this.getId()
671                       + ", in query: "
672                       + sql
673                       + ", "
674                       + e.getMessage()
675                       + ", maybe not enough question marks "
676                       + "(bind variables) are in query, or the number of question marks in the query is "
677                       + "not the same as the number of parameters (might need to set the optional param numParameters), "
678                       + "or the param 'numParameters' in sources.xml for that query is incorrect, " + this.getId());
679           throw e;
680         }
681       }
682     }
683     rs = stmt.executeQuery();
684     return rs;
685   }

715   /**
716    * Loads attributes for the argument subject.
717    * @param rs 
718    * @return attributes
719    */
720   protected Map<String, Set<String>> loadAttributes(ResultSet rs) {
721     Map<String, Set<String>> attributes1 = new SubjectCaseInsensitiveMapImpl<String, Set<String>>();
722     try {
723       ResultSetMetaData rsmd = rs.getMetaData();
724       int colCount = rsmd.getColumnCount();
725       String[] colNames = new String[colCount];
726       for (int i = 0; i < colCount; i++) {
727         colNames[i] = rsmd.getColumnName(i + 1);
728       }
729       for (int i = 0; i < colCount; i++) {
730         String name1 = colNames[i];
731         /*
732         if (name1.toLowerCase().equals(this.subjectIDAttributeName.toLowerCase())) {
733           continue;
734         }
735         if (name1.toLowerCase().equals(this.nameAttributeName.toLowerCase())) {
736           continue;
737         }
738         if (name1.toLowerCase().equals(this.descriptionAttributeName.toLowerCase())) {
739           continue;
740         }
741         */
742         String value = rs.getString(i + 1);

1178   /**
1179    * Perform a search for a unique subject.
1180    * 
1181    * @param id1
1182    * @param searchType
1183    * @return subjects
1184    * @throws SubjectNotFoundException
1185    * @throws SubjectNotUniqueException
1186    * @throws InvalidQueryException 
1187    */
1188   private Map<String, Subject> uniqueSearchBatch(Collection<String> idsOrIdentifiers, String searchType, boolean useIdentifiersInMatch)
1189       throws SubjectNotFoundException, SubjectNotUniqueException, InvalidQueryException {
1190 
1191     Map<String, Subject> results = new LinkedHashMap<String, Subject>();
1192     
1193     if (SubjectUtils.length(idsOrIdentifiers) > 0) {
1194       
1195       Search search = getSearch(searchType);
1196 
1197       if (search == null) {
1198         log.error("searchType: \"" + searchType + "\" not defined.");
1199         return results;
1200       }
1201 
1202       String sql = search.getParam("sql");
1203       
1204       if (StringUtils.isBlank(sql)) {
1205         throw new SourceUnavailableException("Why is there no sql for sourceId: " + this.getId());
1206       }
1207 
1208       if (StringUtils.countMatches(sql, "?") > 0) {
1209         throw new SourceUnavailableException("Why are there parameters in the sql? " + this.getId());
1210       }
1211 
1212       if (!StringUtils.contains(sql, "{inclause}")) {
1213         throw new SourceUnavailableException("Why does the SQL not have an {inclause} param? " + this.getId() + ", " + sql);
1214       }
1215       
1216       if (sql.contains("%TERM%")) {
1217         throw new InvalidQueryException("%TERM%. Possibly old style SQL query, source: "
1218             + this.getId() + ", sql: " + sql);
1219       }
1220 
1221       String inclause = search.getParam("inclause");
1222       
1223       if (StringUtils.isBlank(inclause)) {
1224         throw new SourceUnavailableException("Why is there no inclause? " + this.getId());
1225       }
1226       
1227       String numParametersString = search.getParam("numParameters");
1228 
1229       //default to the number of question marks
1230       int numParameters = StringUtils.isBlank(numParametersString) ? StringUtils
1231           .countMatches(inclause, "?") : Integer.parseInt(numParametersString);
1232           
1233       if (numParameters == 0) {
1234         throw new SourceUnavailableException("Why are there no parameters in the inclause? " + this.getId());
1235       }
1236           
1237       //we dont want more than 180 bind variables per batch
1238       int batchSize = 180 / numParameters;
1239       
1240       List<String> idsOrIdentifiersList = SubjectUtils.listFromCollection(idsOrIdentifiers);
1241       
1242       int numberOfBatches = SubjectUtils.batchNumberOfBatches(idsOrIdentifiersList, batchSize);
1243       
1244       for (int i=0;i<numberOfBatches;i++) {
1245 
1246         List<String> batchIdsOrIdentifiers = SubjectUtils.batchList(idsOrIdentifiersList, batchSize, i);
1247         
1248         Connection conn = null;
1249         PreparedStatement stmt = null;
1250         JdbcConnectionBean jdbcConnectionBean = null;
1251         String aggregateSql = sql;
1252         try {
1253           jdbcConnectionBean = this.jdbcConnectionProvider.connectionBean();
1254           conn = jdbcConnectionBean.connection();
1255           
1256           queryCountforTesting++;
1257           aggregateSql = uniqueSearchBatchSql(sql, inclause, batchIdsOrIdentifiers);
1258           
1259           stmt = conn.prepareStatement(aggregateSql);
1260           
1261           ResultSet rs = getSqlResults(batchIdsOrIdentifiers, stmt, numParameters, aggregateSql);
1262           Map<String, Subject> resultBatch = createUniqueSubjects(rs, search, batchIdsOrIdentifiers, aggregateSql, useIdentifiersInMatch);
1263           results.putAll(resultBatch);
1264 
1265           jdbcConnectionBean.doneWithConnection();
1266         } catch (SQLException ex) {
1267           String error = "problem in sources.xml source: " + this.getId() + ", sql: "
1268               + aggregateSql + ", id size: " + SubjectUtils.length(idsOrIdentifiers) + ", " + searchType;
1269           try {
1270             jdbcConnectionBean.doneWithConnectionError(ex);
1271           } catch (RuntimeException e) {
1272             log.error(error, e);
1273           }
1274           throw new SourceUnavailableException(error, ex);
1275         } finally {
1276           closeStatement(stmt);
1277           if (jdbcConnectionBean != null) {
1278             jdbcConnectionBean.doneWithConnectionFinally();
1279           }
1280         }
1281         
1282       }
1283     }
1284     return results;
1285   }

1287   /**
1288    * turn a sql, and inclause, and batch, into a sql statement
1289    * @param sql
1290    * @param inclause
1291    * @param numParameters
1292    * @param batchIdsOrIdentifiers
1293    * @return unique search sql
1294    */
1295   private String uniqueSearchBatchSql(String sql, String inclause, List<String> batchIdsOrIdentifiers) {
1296     StringBuilder result = new StringBuilder();
1297     result.append(" ( ");
1298     for (int i=0;i<SubjectUtils.length(batchIdsOrIdentifiers);i++) {
1299       if (i != 0) {
1300         result.append(" or ");
1301       }
1302 
1303       result.append(" ( ");
1304       result.append(inclause);
1305       result.append(" ) ");
1306     }
1307     result.append(" ) ");
1308     String aggregateInclause = result.toString();
1309     
1310     String aggregateSql = StringUtils.replace(sql, "{inclause}", aggregateInclause);
1311     
1312     return aggregateSql;
1313   }
