38  /**
39   * represents a connection to an XMPP server.  There is one connection per server/port/user/resource
40   * @author mchyzer
41   */
42  public class XmppConnectionBean {
43  
44    /** chat map, holds the jabberid chatting with and the chat object */
45    private Map<String, Chat> chatMap = Collections.synchronizedMap(new HashMap<String, Chat>());
46  
47    /** keep a reference here so we dont have to keep logging in.  The multikey is server, port, user */
48    private static Map<XmppConnectionBean, XMPPConnection> xmppConnectionMap = Collections.synchronizedMap(new HashMap<XmppConnectionBean, XMPPConnection>());

73    public XmppConnectionBean(String theServer, int thePort, String theUser, String theResource, String thePass) {
74      this.resource = theResource;
75      this.port = thePort;
76      this.pass = thePass;
77      this.user = theUser;

80    }

82    /**
83     * @see Object#equals(Object)
84     */
85    @Override
86    public boolean equals(Object other) {
87      if (this == other) {
88        return true;
89      }
90      if (!(other instanceof XmppConnectionBean)) {
91        return false;
92      }
93      
94      XmppConnectionBean otherBean = (XmppConnectionBean)other;
95      
96      return new EqualsBuilder()
97        .append( this.xmppPort(), otherBean.xmppPort() )
98        .append( this.xmppUser(), otherBean.xmppUser() )
99        .append( this.xmppResource(), otherBean.xmppResource() )
100       .append( this.xmppServer(), otherBean.xmppServer() )
101       .isEquals();
102   }

117   /**
118    * get a chat for a jabber id.  Note this doesnt listen on the channel
119    * @param jabberIds
120    * @param payload of the message
121    */
122   public synchronized void sendMessage(String jabberIds, String payload) {
123     
124     for (String jabberId : GrouperUtil.splitTrim(jabberIds, ",")) {
125       
126       try {
127         
128         Chat chat = this.chat(jabberId, null);
129         try {
130           chat.sendMessage(payload);
131         } catch (XMPPException xe) {
132           
133           if (LOG.isDebugEnabled()) {
134             LOG.debug("Problem in first pass to " + jabberId, xe);
135           }
136           //after first try, clear out the chat and try again
137           try {
138             chatMap.clear();
139             this.xmppConnection().disconnect();
140           } catch (Exception e) {
141             //this is ok
142           }
143           //try again
144           chat = this.chat(jabberId, null);
145           chat.sendMessage(payload);
146         }
147       } catch (XMPPException xmppException) {
148         //lets chill out for a while, there might be big problems
149         GrouperUtil.sleep(30000);
150         
151         throw new RuntimeException("Problem with XMPP to " + jabberId, xmppException);
152       }
153     }  
154   }
155 
156   /**
157    * get a connection object
158    * @return the connection object
159    */
160   public XMPPConnection xmppConnection() {
161     synchronized(XmppConnectionBean.class) {
162       
163       XMPPConnection xmppConnection = xmppConnectionMap.get(this);
164       
165       if (xmppConnection == null || !xmppConnection.isAuthenticated() || !xmppConnection.isConnected()) {
166         try {
167           if (xmppConnection != null) {
168             try {
169               xmppConnection.disconnect();
170             } catch (Exception e) {
171               //this is ok
172             }
173           }
174           ConnectionConfiguration config = new ConnectionConfiguration(this.xmppServer(), this.xmppPort());
175     
176           config.setDebuggerEnabled(false);
177           config.setReconnectionAllowed(true);
178           
179           config.setSASLAuthenticationEnabled(true);
180           SASLAuthentication.supportSASLMechanism("PLAIN");
181           config.setSecurityMode(ConnectionConfiguration.SecurityMode.enabled);
182           xmppConnection = new XMPPConnection(config);
183           xmppConnection.connect();
184     
185           xmppConnection.login(this.xmppUser(), this.xmppPass(), this.xmppResource());
186           
187           //clear this out so we dont try to use the old connections
188           this.chatMap.clear();
189   
190         } catch (XMPPException xe) {
191           throw new RuntimeException(xe);
192         }
193       }
194       
195       xmppConnectionMap.put(this, xmppConnection);
196       
197       return xmppConnection;
198     }
199   }

201   /**
202    * xmpp pass (decrypted if file)
203    * @return the pass
204    */
205   public String xmppPass() {
206     String pass = StringUtils.isBlank(this.pass) ? GrouperLoaderConfig.retrieveConfig().propertyValueStringRequired("xmpp.pass") : this.pass;
207     return Morph.decryptIfFile(pass);
208   }

210   /**
211    * port to connect to, or 1522 as default 
212    * @return port
213    */
214   public int xmppPort() {
215     return this.port == -1 ? GrouperLoaderConfig.retrieveConfig().propertyValueInt("xmpp.server.port", 1522) : this.port;
216   }

219    * xmpp resource
220    * @return the resource
221    */
222   public String xmppResource() {
223     return StringUtils.isBlank(this.resource) ? GrouperLoaderConfig.retrieveConfig().propertyValueString("xmpp.resource") : this.resource;
224   }

231     return StringUtils.isBlank(this.server) ? GrouperLoaderConfig.retrieveConfig().propertyValueStringRequired("xmpp.server.host") : this.server;
232   }

235    * xmpp user

240   }

242   /**
243    * get or make a chat
244    * @param jabberId to chat with
245    * @param messageListener to get callbacks on messages or null to just send.  Note if there is already one registered, this
246    * one wont be used
247    * @return the chat
248    */
249   public synchronized Chat chat(String jabberId, MessageListener messageListener) {
250     
251     Chat chat = this.chatMap.get(jabberId);
252     if (chat == null) {
253       
254       ChatManager chatManager = xmppConnection().getChatManager();
255       chat = chatManager.createChat(jabberId, null);
256       chatMap.put(jabberId, chat);
257     }
258     return chat;
259     
260   }
