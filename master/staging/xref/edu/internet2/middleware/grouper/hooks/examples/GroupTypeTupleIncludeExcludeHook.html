75    private static boolean registered = false;

77    /** if the hook was registered and being used */
78    private static boolean registeredSuccess = false;

80    /**
81     * see if this is configured in the grouper.properties, if so, register this hook
82     * @param tryAgainIfNotBefore 
83     */
84    public static void registerHookIfNecessary(boolean tryAgainIfNotBefore) {
85      
86      if (registered && !tryAgainIfNotBefore) {
87        return;
88      }
89      
90      //if trying again, but already registered, fine
91      if (tryAgainIfNotBefore && registeredSuccess) {
92        return;
93      }
94      
95      //see if there are config entries
96      Properties grouperProperties = GrouperConfig.retrieveConfig().properties();
97  
98      boolean useIncludeExcludeOverride = StringUtils.equals(GrouperConfig.retrieveConfig().propertiesOverrideMap().get("grouperIncludeExclude.use"), "true");
99      
100     boolean useGrouperIncludeExclude = useIncludeExcludeOverride || GrouperUtil.propertiesValueBoolean(
101         grouperProperties, "grouperIncludeExclude.use", false);
102 
103     boolean useRequireGroupsOverride = StringUtils.equals(GrouperConfig.retrieveConfig().propertiesOverrideMap().get("grouperIncludeExclude.requireGroups.use"), "true");
104     
105     boolean useRequireGroups = useRequireGroupsOverride || GrouperUtil.propertiesValueBoolean(
106         grouperProperties, "grouperIncludeExclude.requireGroups.use", false);
107 
108     //register the hook
109     if (useGrouperIncludeExclude || useRequireGroups) {
110       registeredSuccess = true;
111       //register this hooks
112       GrouperHooksUtils.addHookManual(GrouperHookType.GROUP_TYPE_TUPLE.getPropertyFileKey(), GroupTypeTupleIncludeExcludeHook.class);
113       GrouperHooksUtils.addHookManual(GrouperHookType.ATTRIBUTE.getPropertyFileKey(), AttributeIncludeExcludeHook.class);
114     }
115     
116     registered = true;
117 
118   }

120   /**
121    * 
122    * @param attributeName
123    * @return the group name from the config file
124    */
125   public static String groupNameFromAndGroupAttributeName(String attributeName) {
126     int i=0;
127     while (true) {
128       String propertyName = "grouperIncludeExclude.requireGroup.name." + i;
129       String propertyValue = GrouperConfig.retrieveConfig().propertyValueString(propertyName);
130       if (StringUtils.isBlank(propertyValue)) {
131         break;
132       }
133       if (StringUtils.equals(attributeName, propertyValue)) {
134         //this is the right index, get the group name
135         return GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.group." + i);
136       }
137       i++;
138     }
139     throw new RuntimeException("Cant find config entry for andGroup attribute name: " + attributeName 
140         + ", e.g. config name: grouperIncludeExclude.requireGroup.name.{i}");  }

164   /**
165    * @param postInsertBean
166    * @param isInsert
167    */
168   private void groupTypeTupleHelper(HooksGroupTypeTupleBean postInsertBean, boolean isInsert) {
169     boolean useGrouperIncludeExclude = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.use", false);
170     boolean useRequireGroups = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.requireGroups.use", false);
171     
172     //dont do anything if not using this type
173     if (!useGrouperIncludeExclude && !useRequireGroups) {
174       return;
175     }
176 
177     GroupTypeTuple groupTypeTuple = postInsertBean.getGroupTypeTuple();
178 
179     String groupUuid = groupTypeTuple.getGroupUuid();
180 
181     //make sure this is the right type
182     String includeExcludeTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.type.name");
183     String requireGroupsTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.type.name");
184 
185     //there better be a session now!
186     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
187 
188     try {
189       
190       GroupType includeExcludeType = useGrouperIncludeExclude ? GroupTypeFinder.find(includeExcludeTypeName, false) : null;
191       boolean fireHook = (useGrouperIncludeExclude && isInsert)
192         ? StringUtils.equals(groupTypeTuple.getTypeUuid(), 
193             includeExcludeType == null ? null : includeExcludeType.getUuid()) : false;
194 
195       GroupType requireGroupsType = useRequireGroups ? GroupTypeFinder.find(requireGroupsTypeName, false) : null;
196       fireHook = fireHook || (useRequireGroups 
197         ? StringUtils.equals(groupTypeTuple.getTypeUuid(), 
198             requireGroupsType == null ? null : requireGroupsType.getUuid()) : false);
199         
200       //see if a custom group type
201       if (!fireHook) {
202         
203         //#grouperIncludeExclude.requireGroup.name.0 = requireActiveEmployee
204         //#grouperIncludeExclude.requireGroup.attributeOrType.0 = type
205         int i = 0;
206         while (true) {
207           String name = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.name." + i);
208           if (StringUtils.isBlank(name)) {
209             break;
210           }
211           String attributeOrType = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.attributeOrType." + i);
212           if (StringUtils.equalsIgnoreCase("type", attributeOrType)) {
213             GroupType type = GroupTypeFinder.find(name, false);
214             if (type != null) {
215               fireHook = fireHook ||StringUtils.equals(groupTypeTuple.getTypeUuid(), type.getUuid());
216               if (fireHook) {
217                 break;
218               }
219             }
220           }
221           i++;
222         }
223         
224       }
225       if (!fireHook) {
226         return;
227       }
228         
229       //not sure why this would be null, could be a stale state problem if so
230       Group typedGroup = postInsertBean.getGroupTypeTuple().retrieveGroup(true);
231 
232       manageIncludesExcludesAndGroups(grouperSession, typedGroup, includeExcludeTypeName + " changed on group: " + typedGroup.getName());
233       
234     } catch (Exception e) {
235       throw new RuntimeException("Error doing include/exclude on group: " + groupUuid, e);
236     }
237   }

239   /**
240    * convert a system of record extension to an overall extension
241    * @param extension is overall extension or system of record extension
242    * @return the overall extension
243    */
244   public static String convertToOverall(String extension) {
245     //strip system of record suffix (i.e. we are putting the type on the system of record group)
246     if (extension.endsWith(systemOfRecordExtensionSuffix())) {
247       extension = extension.substring(0, 
248           extension.length() - systemOfRecordExtensionSuffix().length());
249     }
250     return extension;
251     
252   }

254   /**
255    * return a set of groups including the one passed in, related to this group.
256    * if the groups arent found, dont worry
257    * i.e. if include/exclude or requireGroups, find related groups
258    * @param group
259    * @return the set of groups
260    */
261   public static Set<Group> relatedGroups(Group group) {
262     Set<Group> groups = new LinkedHashSet<Group>();
263     
264     groups.add(group);
265     
266     String name = group.getName();
267     
268     //baseName is name without suffix if applicable
269     String baseName = name;
270     
271     boolean useGrouperIncludeExclude = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.use", false);
272     
273     if (useGrouperIncludeExclude) {
274       
275       if (name.endsWith(systemOfRecordExtensionSuffix())) {
276         baseName = GrouperUtil.stripSuffix(name, systemOfRecordExtensionSuffix());
277       } else if (name.endsWith(includeExtensionSuffix())) {
278         baseName = GrouperUtil.stripSuffix(name, includeExtensionSuffix());
279       } else if (name.endsWith(includesMinusExcludesExtensionSuffix())) {
280         baseName = GrouperUtil.stripSuffix(name, includesMinusExcludesExtensionSuffix());
281       } else if (name.endsWith(systemOfRecordAndIncludesExtensionSuffix())) {
282         baseName = GrouperUtil.stripSuffix(name, systemOfRecordAndIncludesExtensionSuffix());
283       }
284     }
285     boolean useRequireGroups = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.requireGroups.use", false);
286     
287     //strip off something like _requireGroups15
288     if (useRequireGroups) {
289       Matcher matcher = requireGroupsPattern.matcher(name);
290       if (matcher.matches()) {
291         baseName = matcher.group(1);
292       }
293     }
294 
295     if (LOG.isDebugEnabled()) {
296       LOG.debug("name: " + name + ", baseName: " + baseName);
297     }
298     
299     GrouperSession grouperSession = GrouperSession.staticGrouperSession().internal_getRootSession();
300     
301     Group groupFind = null;
302     
303     //now lets find all the groups
304     if (!StringUtils.equals(name, baseName)) {
305       groupFind = GroupFinder.findByName(grouperSession, baseName,false);
306       if (groupFind != null) {
307         groups.add(groupFind);
308       }
309     }
310 
311     if (useGrouperIncludeExclude) {
312       
313       //system of record
314       groupFind = GroupFinder.findByName(grouperSession, baseName + systemOfRecordExtensionSuffix(),false);
315       if (groupFind != null) {
316         groups.add(groupFind);
317       }
318     
319       //include
320       groupFind = GroupFinder.findByName(grouperSession, baseName + includeExtensionSuffix(),false);
321       
322       if (LOG.isDebugEnabled()) {
323         LOG.debug("Searching for group: " + (baseName + includeExtensionSuffix()) + ", found: " + (groupFind != null));
324       }
325       
326       if (groupFind != null) {
327         groups.add(groupFind);
328       }
329     
330       //exclude
331       groupFind = GroupFinder.findByName(grouperSession, baseName + excludeExtensionSuffix(),false);
332       if (groupFind != null) {
333         groups.add(groupFind);
334       }
335 
336       //include and sor
337       groupFind = GroupFinder.findByName(grouperSession, baseName + includesMinusExcludesExtensionSuffix(),false);
338       if (groupFind != null) {
339         groups.add(groupFind);
340       }
341       
342     }
343     
344     if (useRequireGroups) {
345       
346       //any require groups
347       for (int i=1;i<200;i++) {
348         groupFind = GroupFinder.findByName(grouperSession, baseName + requireGroupsExtensionSuffix(1),false);
349         if (groupFind == null) {
350           break;
351         }
352         groups.add(groupFind);
353       }
354     }
355     
356     return groups;
357   }

373   /**
374    * @param grouperSession
375    * @param typedGroup
376    * @param summaryForLog some string that will be logged to debug...
377    */
378   public static void manageIncludesExcludesAndGroups(GrouperSession grouperSession, Group typedGroup, String summaryForLog) {
379     
380     //make sure this is the right type
381     String includeExcludeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.type.name");
382     String groupTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.type.name");
383 
384     //make these sorted so that it is more easily testable
385     Set<Group> andGroups = new TreeSet<Group>();
386     
387     boolean includeExclude = false;
388     GroupType includesExcludesType = null;
389     GroupType requireGroupsType = null;
390     try {
391       
392       includesExcludesType = GroupTypeFinder.find(includeExcludeName, true);
393       
394       includeExclude = typedGroup.hasType(includesExcludesType, false);
395       
396       //if other groups are there, then this is include/exclude.  we dont remove when the checkbox is
397       //unchecked
398       String overallExtension = typedGroup.getExtension();
399       
400       overallExtension = convertToOverall(overallExtension);
401 
402       String stemName = GrouperUtil.parentStemNameFromName(typedGroup.getName());
403       String overallName = stemName + ":" + overallExtension;
404 
405       includeExclude = includeExclude || GroupFinder.findByName(grouperSession, 
406           overallName + includeExtensionSuffix(), false) != null;
407       includeExclude = includeExclude || GroupFinder.findByName(grouperSession, 
408           overallName + excludeExtensionSuffix(), false) != null;
409       
410       requireGroupsType = GroupTypeFinder.find(groupTypeName, true);
411   
412       boolean hasRequireGroupsType = typedGroup.hasType(requireGroupsType, false);
413       
414       String andGroupsAttributeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.attributeName");
415       boolean hasAndGroupsAttributeName = StringUtils.isNotBlank(andGroupsAttributeName);
416       
417       if (hasRequireGroupsType && hasAndGroupsAttributeName) {
418         String groupNames = typedGroup.getAttributeValue(andGroupsAttributeName, false, false);
419         if (!StringUtils.isBlank(groupNames)) {
420           String[] andGroupNames = GrouperUtil.splitTrim(groupNames, ",");
421           for (String andGroupName: andGroupNames) {
422             andGroups.add(GroupFinder.findByName(grouperSession, andGroupName, true));
423           }
424         }
425       }
426       
427       if (hasRequireGroupsType) {
428         Set<AttributeDefName> attrs = GrouperDAOFactory.getFactory().getAttributeDefName().findByAttributeDef(requireGroupsType.internal_getAttributeDefForAttributes().getId());
429         
430         for (AttributeDefName attr : GrouperUtil.nonNull(attrs)) {
431           
432           //see if this is not a custom require group
433           if (hasAndGroupsAttributeName && StringUtils.equals(andGroupsAttributeName, attr.getLegacyAttributeName(true))) {
434             continue;
435           }
436           
437           String valueString = typedGroup.getAttributeValue(attr.getLegacyAttributeName(true), false, false);
438     
439           boolean valueBoolean = GrouperUtil.booleanValue(valueString, false);
440           
441           if (valueBoolean) {
442             String groupName = groupNameFromAndGroupAttributeName(attr.getLegacyAttributeName(true));
443             andGroups.add(GroupFinder.findByName(grouperSession, groupName, true));
444           }
445           
446         }
447       }
448       //now try custom types
449       //#grouperIncludeExclude.requireGroup.name.0 = requireActiveEmployee
450       //#grouperIncludeExclude.requireGroup.attributeOrType.0 = type
451       //#grouperIncludeExclude.requireGroup.group.0 = school:community:activeEmployee
452       int i=0;
453       while (true) {
454         
455         String typeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.name." + i);
456         if (StringUtils.isBlank(typeName)) {
457           break;
458         }
459         String attributeOrType = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.attributeOrType." + i);
460         //just check types
461         if (StringUtils.equalsIgnoreCase("type", attributeOrType)) {
462           String groupName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.group." + i);
463           
464           GroupType type = GroupTypeFinder.find(typeName, true);
465           //if the hooked group has this type, then good to go
466           if (typedGroup.getTypesDb().contains(type)) {
467             Group group = GroupFinder.findByName(grouperSession, groupName, true);
468             andGroups.add(group);
469           }
470         }
471         i++;
472       }
473       
474     } catch (SchemaException se) {
475       throw new RuntimeException(se);
476     } catch (GroupNotFoundException gnfe) {
477       throw new RuntimeException(gnfe);
478     }
479     manageIncludesExcludesAndGroups(typedGroup, includeExclude, andGroups, summaryForLog);
480   }

517   /**
518    * change a typed group into include and exclude group lists and andGroups also
519    * @param typedGroup 
520    * @param isIncludeExclude 
521    * @param andGroups 
522    * @param calledFromForLog summary of where this is coming from for debug log
523    */
524   public static void manageIncludesExcludesAndGroups(Group typedGroup, boolean isIncludeExclude, 
525       Set<Group> andGroups, String calledFromForLog) {
526 
527     String groupUuid = null;
528     
529     if (LOG.isDebugEnabled()) {
530       StringBuilder summary = new StringBuilder();
531       summary.append("manageIncludeExclude called for group: " + typedGroup.getName() 
532           + ", includeExclude? " + isIncludeExclude + " and " + GrouperUtil.length(andGroups) + " andGroups: ");
533       Iterator<Group> iterator = GrouperUtil.nonNull(andGroups).iterator();
534       boolean first = true;
535       while (iterator.hasNext()) {
536         if (!first) {
537           summary.append(", ");
538         }
539         first = false;
540         summary.append(iterator.next().getExtension());
541       }
542       summary.append(".  ").append(calledFromForLog);
543       LOG.debug(summary.toString());
544      }
545     
546     //there better be a session now!
547     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
548     try {
549       
550       groupUuid = typedGroup.getUuid();
551       
552       String overallExtension = typedGroup.getExtension();
553       
554       String overallDisplayExtension = typedGroup.getDisplayExtension();
555 
556       //strip system of record suffix (i.e. we are putting the type on the system of record group)
557       if (overallExtension.endsWith(systemOfRecordExtensionSuffix())) {
558         overallExtension = overallExtension.substring(0, 
559             overallExtension.length() - systemOfRecordExtensionSuffix().length());
560 
561         //strip system of record suffixes
562         if (overallDisplayExtension.endsWith(systemOfRecordExtensionSuffix())) {
563           overallDisplayExtension = overallDisplayExtension.substring(0, 
564               overallDisplayExtension.length() - systemOfRecordExtensionSuffix().length());
565         }
566         if (overallDisplayExtension.endsWith(systemOfRecordDisplayExtensionSuffix())) {
567           overallDisplayExtension = overallDisplayExtension.substring(0, 
568               overallDisplayExtension.length() - systemOfRecordDisplayExtensionSuffix().length());
569         }
570         
571       }
572       
573       String stemName = GrouperUtil.parentStemNameFromName(typedGroup.getName());
574       String overallName = stemName + ":" + overallExtension;
575 
576       //create the overall group if not exist already
577       Group overallGroup = null;
578       
579       if (StringUtils.equals(typedGroup.getName(), overallName)) {
580         overallGroup = typedGroup;
581       } else {
582         overallGroup = GroupFinder.findByName(grouperSession, overallName, false);
583       }
584       
585       String overallDescription = overallDescription(overallExtension, overallDisplayExtension);
586       if (overallGroup == null) {
587         LOG.debug("Adding overall group: " + overallName);
588         overallGroup = Group.saveGroup(grouperSession, null, null, overallName, overallDisplayExtension, 
589             overallDescription, null, false);
590       } else if (StringUtils.isBlank(overallGroup.getDescription())) {
591         overallGroup.setDescription(overallDescription);
592         overallGroup.store();
593       }
594 
595       //create system of record if not exist
596       String systemOfRecordName = overallName + systemOfRecordExtensionSuffix();
597 
598       Group systemOfRecordGroup = GroupFinder.findByName(grouperSession, systemOfRecordName, false);
599       if (systemOfRecordGroup == null ) {
600         
601         String systemOfRecordDescription = systemOfRecordDescription(overallExtension, overallDisplayExtension);
602         String systemOfRecordDisplayExtension = overallDisplayExtension + systemOfRecordDisplayExtensionSuffix();
603         LOG.debug("Adding system of record group: " + systemOfRecordName);
604         systemOfRecordGroup = Group.saveGroup(grouperSession, null, null, systemOfRecordName, systemOfRecordDisplayExtension, 
605             systemOfRecordDescription, null, true);
606         
607       }
608       
609       //create includes group if not exist
610       String includesName = overallName + includeExtensionSuffix();
611 
612       Group includesGroup = GroupFinder.findByName(grouperSession, includesName, false);
613       if (isIncludeExclude && includesGroup == null ) {
614         
615         String includesDescription = includeDescription(overallExtension, overallDisplayExtension);
616         String includesDisplayExtension = overallDisplayExtension + includeDisplayExtensionSuffix();
617         LOG.debug("Adding includes group: " + includesName);
618         includesGroup = Group.saveGroup(grouperSession, null, null, includesName, includesDisplayExtension, 
619             includesDescription, null, true);
620         
621       }
622       
623       //create excludes group if not exist
624       String excludesName = overallName + excludeExtensionSuffix();
625 
626       Group excludesGroup = GroupFinder.findByName(grouperSession, excludesName, false);
627       if (isIncludeExclude && excludesGroup == null ) {
628         
629         String excludesDescription = excludeDescription(overallExtension, overallDisplayExtension);
630         String excludesDisplayExtension = overallDisplayExtension + excludeDisplayExtensionSuffix();
631         LOG.debug("Adding excludes group: " + excludesName);
632         excludesGroup = Group.saveGroup(grouperSession, null, null, excludesName, excludesDisplayExtension, 
633             excludesDescription, null, true);
634         
635       }
636       
637       //create excludes group if not exist
638       String systemOfRecordAndIncludesName = overallName + systemOfRecordAndIncludesExtensionSuffix();
639 
640       Group systemOfRecordAndIncludesGroup = GroupFinder.findByName(grouperSession, systemOfRecordAndIncludesName, false);
641       if (isIncludeExclude && systemOfRecordAndIncludesGroup == null ) {
642         
643         String systemOfRecordAndIncludesDescription = systemOfRecordAndIncludesDescription(overallExtension, overallDisplayExtension);
644         String systemOfRecordAndIncludesDisplayExtension = overallDisplayExtension + systemOfRecordAndIncludesDisplayExtensionSuffix();
645         LOG.debug("Adding system of record group: " + systemOfRecordAndIncludesName);
646         systemOfRecordAndIncludesGroup = Group.saveGroup(grouperSession, null, null, systemOfRecordAndIncludesName, systemOfRecordAndIncludesDisplayExtension, 
647             systemOfRecordAndIncludesDescription, null, true);
648         
649       }
650 
651       //dont need to do this if not include/exclude
652       if (isIncludeExclude) {
653         //lets make sure systemOfRecord and includes are members
654         systemOfRecordAndIncludesGroup.addMember(SubjectFinder.findById(systemOfRecordGroup.getUuid(), true), false);
655         systemOfRecordAndIncludesGroup.addMember(SubjectFinder.findById(includesGroup.getUuid(), true), false);
656       }
657       
658       //at this point, the main group must have no members
659       if (overallGroup.getComposite(false) == null) {
660         
661         //lets get all the existing members and add to the system of record group
662         //lets do this if the overall group has members, and if the system of record group
663         //doesnt have any members yet
664         Set<Member> existingOverallMembers = overallGroup.getImmediateMembers();
665         
666         if (existingOverallMembers.size() > 0) {
667           
668           //note: lists arent covered here...
669           for (Member member : existingOverallMembers) {
670             LOG.debug("Removing member from overall, and adding to system of record: " 
671                 + member.getSubjectSourceIdDb() + " - " + member.getSubjectIdDb());
672             //dont add system of record to itself
673             if (!StringUtils.equals(member.getSubjectIdDb(), systemOfRecordGroup.getUuid())) {
674               systemOfRecordGroup.addMember(member.getSubject(), false);
675             }
676             overallGroup.deleteMember(member, false);
677           }
678         }
679 
680       }
681       int andGroupsLength = GrouperUtil.length(andGroups);
682 
683       //################################################
684       //lets remove unneeded groups
685       int i=andGroupsLength;
686       int highestIndexToDelete = -1;
687       while(true) {
688         
689         String requireGroupsSuffixExtension = requireGroupsExtensionSuffix(i);
690         String requireGroupsName = overallName + requireGroupsSuffixExtension;
691         Group requireGroupsSuffix = GroupFinder.findByName(grouperSession, requireGroupsName, false);
692         if (requireGroupsSuffix != null) {
693           highestIndexToDelete = i;
694         }
695         if (requireGroupsSuffix == null && i > 10) {
696           break;
697         }
698         i++;
699       }
700 
701       String includesMinusExcludesName = overallName + includesMinusExcludesExtensionSuffix();
702       Group includesMinusExcludesGroup = GroupFinder.findByName(grouperSession, includesMinusExcludesName, false);
703 
704       //#############################################
705       //make final structure
706       if (andGroupsLength == 0) {
707             
708         if (isIncludeExclude) {
709           //now the overall group needs to be a composite complement
710           overallGroup.assignCompositeMember(CompositeType.COMPLEMENT, systemOfRecordAndIncludesGroup, excludesGroup);
711         } else {
712           //make sure not a composite group from before
713           if (overallGroup.hasComposite()) {
714             overallGroup.deleteCompositeMember();
715           }
716           overallGroup.addMember(SubjectFinder.findById(systemOfRecordGroup.getUuid(), true), false);
717           LOG.debug("Adding system of record group (" + systemOfRecordName + ") to the overall group: "  
718               + overallGroup.getName());
719 
720         }
721         
722       } else {
723 
724         //######################################
725         //includesMinusExcludes is composite complement: systemOfRecordAndIncludesGroup minus excludesGroup
726         includesMinusExcludesGroup = GroupFinder.findByName(grouperSession, includesMinusExcludesName, false);
727         //if no groups we dont need this one
728         if (isIncludeExclude && includesMinusExcludesGroup == null ) {
729 
730           String includesMinusExcludesGroupDescription = systemOfRecordAndIncludesDescription(overallExtension, overallDisplayExtension);
731           String includesMinusExcludesGroupDisplayExtension = overallDisplayExtension + includesMinusExcludesDisplayExtensionSuffix();
732           includesMinusExcludesGroup = Group.saveGroup(grouperSession, null, null, includesMinusExcludesName, 
733               includesMinusExcludesGroupDisplayExtension, 
734               includesMinusExcludesGroupDescription, null, true);
735         }
736         if (isIncludeExclude) {
737           includesMinusExcludesGroup.assignCompositeMember(CompositeType.COMPLEMENT, systemOfRecordAndIncludesGroup, excludesGroup);
738         }
739 
740         //for one andGroup, structure will be
741         //includesMinusExcludes is composite complement: systemOfRecordAndIncludesGroup minus excludesGroup
742         //overallGroup is composite complement: includesMinusExcludes minus aStem:activeEmployee
743         Group[] andGroupsArray = GrouperUtil.toArray(andGroups, Group.class);
744         if (andGroups.size() == 1) {
745           if (isIncludeExclude) {
746             overallGroup.assignCompositeMember(CompositeType.INTERSECTION, includesMinusExcludesGroup, andGroupsArray[0]);
747           } else {
748             overallGroup.assignCompositeMember(CompositeType.INTERSECTION, systemOfRecordGroup, andGroupsArray[0]);
749           }
750         } else {
751         
752           Group previousGroup = null;
753           
754           //for more than one andGroup, structure will be
755           //includeExcludeMinusAnd2 is composite complement: includesMinusExcludes minus aStem:activeEmployee
756           //includeExcludeMinusAnd1 is composite complement: includeExcludeMinusAnd2 minus aStem:anotherGroup
757           //overallGroup is composite complement: includeExcludeMinusAnd1 minus aStem:yetAnotherGroup
758           int arrayIndex = 0;
759           for (i=andGroupsLength-1;i>=1;i--) {
760             String requireGroupsName = overallName + requireGroupsExtensionSuffix(i);
761             Group requireGroup = GroupFinder.findByName(grouperSession, requireGroupsName, false);
762             //if no groups we dont need this one
763             if (requireGroup == null ) {
764   
765               String requireGroupsDescription = requireGroupsDescription(i, overallExtension, overallDisplayExtension);
766               String requireGroupsDisplayExtension = overallDisplayExtension + requireGroupsDisplayExtensionSuffix(i);
767               requireGroup = Group.saveGroup(grouperSession, null, null, requireGroupsName, requireGroupsDisplayExtension, 
768                   requireGroupsDescription, null, true);
769               LOG.debug("Adding requireGroups group: " + requireGroup.getExtension());
770             }
771             Group leftGroup = i == andGroupsLength-1 
772               ? (isIncludeExclude ? includesMinusExcludesGroup : systemOfRecordGroup) : previousGroup;
773             requireGroup.assignCompositeMember(CompositeType.INTERSECTION, leftGroup, andGroupsArray[arrayIndex]);
774             previousGroup = requireGroup;
775             arrayIndex++;
776           }
777 
778           //cant assign if has one
779           if (overallGroup.hasComposite()) {
780             overallGroup.deleteCompositeMember();
781           }
782           
783           overallGroup.assignCompositeMember(CompositeType.INTERSECTION, previousGroup, andGroupsArray[andGroupsLength-1]);
784             
785         }
786         
787       }
788       
789       for (i=andGroupsLength;i<=highestIndexToDelete;i++) {
790         String requireGroupsSuffixExtension = requireGroupsExtensionSuffix(i);
791         String requireGroupsName = overallName + requireGroupsSuffixExtension;
792         Group includeExcludeMinusAndGroup = GroupFinder.findByName(grouperSession, requireGroupsName, false);
793         if (includeExcludeMinusAndGroup != null) {
794           LOG.debug("Deleting unneeded group: " + includeExcludeMinusAndGroup.getName());
795           includeExcludeMinusAndGroup.delete();
796         }
797       }
798 
799       //if no groups we dont need this one
800       if (andGroupsLength == 0 || !isIncludeExclude) {
801         if (includesMinusExcludesGroup != null ) {
802           LOG.debug("Deleting unneeded group: " + includesMinusExcludesGroup.getName());
803           includesMinusExcludesGroup.delete();
804         }
805       }
806       if (!isIncludeExclude && systemOfRecordAndIncludesGroup != null ) {
807           LOG.debug("Deleting unneeded group: " + systemOfRecordAndIncludesGroup.getName());
808           systemOfRecordAndIncludesGroup.delete();
809       }
810 
811     } catch (Exception e) {
812       throw new RuntimeException("Error doing include/exclude on group: " + groupUuid, e);
813     }
814   }

826    * see if a group name has an include/exclude or requireGroup suffix
827    * @param groupName
828    * @return true if include/exclude
829    */
830   public static boolean nameIsIncludeExcludeRequireGroup(String groupName) {
831     boolean useGrouperIncludeExclude = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.use", false);
832 
833     //check include exclude
834     if (useGrouperIncludeExclude) {
835       
836       if (groupName.endsWith(excludeExtensionSuffix())
837           || groupName.endsWith(includeExtensionSuffix())
838           || groupName.endsWith(includesMinusExcludesExtensionSuffix())
839           || groupName.endsWith(systemOfRecordAndIncludesExtensionSuffix())
840           || groupName.endsWith(systemOfRecordExtensionSuffix())) {
841         return true;
842       }
843       
844     }
845     //check 10 require groups
846     boolean useRequireGroups = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.requireGroups.use", false);
847     
848     if (useRequireGroups) {
849       for (int i=1;i<=10;i++) {
850         if (groupName.endsWith(requireGroupsExtensionSuffix(i))) {
851           return true;
852         }
853       }
854     }
855     return false;
856   }

927   /**
928    * 
929    * @param grouperSession 
930    * @param groupName if the overall or system of record group
931    * @param reason if you want a reason passed back
932    * @param saveIncludesExcludesIfMembers true if not delete only the includes group and excludes group
933    * if they have members.  false, delete anyway
934    * @return the number of groups removed
935    */
936   public static int deleteGroupsIfNotUsed(GrouperSession grouperSession, 
937       String groupName, StringBuilder reason, boolean saveIncludesExcludesIfMembers) {
938     if (reason == null) {
939       reason = new StringBuilder();
940     }
941     int count = 0;
942     try {
943       String overallName = convertToOverall(groupName);
944       Group group = GroupFinder.findByName(grouperSession, overallName, false);
945       if (group != null) {
946         
947         Member member = MemberFinder.internal_findBySubject(SubjectFinder.findById(group.getUuid(), true), null, false);
948         
949         if (member != null) {
950           //get any membership
951           Set<Membership> memberships = GrouperDAOFactory.getFactory()
952             .getMembership().findAllImmediateByMember(member.getUuid(), false);
953           if (memberships.size() > 0) {
954             String message = "Not deleting group: " + overallName + " since used in " 
955               + memberships.size() + " immediate memberships";
956             LOG.debug(message);
957             reason.append(message);
958             return 0;
959           }
960           Set<Composite> composites = CompositeFinder.findAsFactor(group);
961           if (composites.size() > 0) {
962             String message = "Not deleting group: " + overallName + " since used in " 
963               + composites.size() + " composites";
964             LOG.debug(message);
965             reason.append(message);
966             return 0;
967           }
968         } 
969         
970       }
971       
972       //we can delete the overall group
973       group.delete();
974       count++;
975       
976       //now lets pick apart the composites and such
977       for (int i=1;i<10;i++) {
978         String requireGroupsName = overallName + requireGroupsExtensionSuffix(i);
979         group = GroupFinder.findByName(grouperSession, requireGroupsName, false);
980         if (group != null) {
981           String logString = "deleting group: " + requireGroupsName;
982           GrouperUtil.append(reason, ", ", logString);
983           LOG.debug(logString);
984           group.delete();
985           count++;
986         }
987       }
988       
989       {
990         //now if the includesMinusExcludes
991         String includesMinuxExcludesName = overallName + includesMinusExcludesExtensionSuffix();
992         group = GroupFinder.findByName(grouperSession, 
993             includesMinuxExcludesName, false);
994   
995         if (group != null) {
996           String logString = "deleting group: " + includesMinuxExcludesName;
997           GrouperUtil.append(reason, ", ", logString);
998           LOG.debug(logString);
999           group.delete();
1000           count++;
1001         }
1002       }
1003       
1004       {
1005         //now if the includesMinusExcludes
1006         String sorAndIncludesName = overallName + systemOfRecordAndIncludesExtensionSuffix();
1007         group = GroupFinder.findByName(grouperSession, 
1008             sorAndIncludesName, false);
1009   
1010         if (group != null) {
1011           String logString = "deleting group: " + sorAndIncludesName;
1012           GrouperUtil.append(reason, ", ", logString);
1013           LOG.debug(logString);
1014           group.delete();
1015           count++;
1016         }
1017       }
1018       
1019       {
1020         //now excludes if no members
1021         String excludesName = overallName + excludeExtensionSuffix();
1022         group = GroupFinder.findByName(grouperSession, 
1023             excludesName, false);
1024   
1025         if (group != null) {
1026           Set<Member> members = group.getMembers();
1027           if (members.size() > 0 && saveIncludesExcludesIfMembers) {
1028             String logString = "not deleting group: " + excludesName + " since has " + members.size() + " members";
1029             GrouperUtil.append(reason, ", ", logString);
1030             LOG.debug(logString);
1031           } else {
1032             String logString = "deleting group: " + excludesName;
1033             GrouperUtil.append(reason, ", ", logString);
1034             LOG.debug(logString);
1035             group.delete();
1036             count++;
1037           }
1038         }
1039       }
1040       
1041       {
1042         //now includes if no members
1043         String includesName = overallName + includeExtensionSuffix();
1044         group = GroupFinder.findByName(grouperSession, 
1045             includesName, false);
1046   
1047         if (group != null) {
1048           Set<Member> members = group.getMembers();
1049           if (members.size() > 0 && saveIncludesExcludesIfMembers) {
1050             String logString = "not deleting group: " + includesName + " since has " + members.size() + " members";
1051             GrouperUtil.append(reason, ", ", logString);
1052             LOG.debug(logString);
1053           } else {
1054             String logString = "deleting group: " + includesName;
1055             GrouperUtil.append(reason, ", ", logString);
1056             LOG.debug(logString);
1057             group.delete();
1058             count++;
1059           }
1060         }
1061       }
1062       
1063       {
1064         //finally system of record
1065         String systemOfRecordName = overallName + systemOfRecordExtensionSuffix();
1066         group = GroupFinder.findByName(grouperSession, 
1067             systemOfRecordName, false);
1068   
1069         if (group != null) {
1070           String logString = "deleting group: " + systemOfRecordName;
1071           GrouperUtil.append(reason, ", ", logString);
1072           LOG.debug(logString);
1073           group.delete();
1074           count++;
1075         }
1076       }      
1077       GrouperUtil.append(reason, ", ", "deleted " + count + " groups");
1078       
1079       return count;
1080     } catch (Exception e) {
1081       throw new RuntimeException("Problem deleting groups for name: " + groupName 
1082           + ", though did delete " + count + " groups", e);
1083     }
1084   }
