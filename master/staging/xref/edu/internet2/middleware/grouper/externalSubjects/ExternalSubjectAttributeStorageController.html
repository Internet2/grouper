16  package edu.internet2.middleware.grouper.externalSubjects;
17  
18  import java.util.Set;
19  
20  import org.apache.commons.lang.StringUtils;
21  
22  import edu.internet2.middleware.grouper.cfg.GrouperConfig;
23  import edu.internet2.middleware.grouper.internal.dao.ExternalSubjectAttributeDAO;
24  import edu.internet2.middleware.grouper.internal.dao.QueryOptions;
25  import edu.internet2.middleware.grouper.util.GrouperUtil;
26  
27  /**
28   * @see ExternalSubjectAttributeStorable
29   * @author mchyzer
30   *
31   */
32  public class ExternalSubjectAttributeStorageController {
33  
34    /**
35     * store the implementation, lazy load
36     */
37    private static ExternalSubjectAttributeStorable externalSubjectAttributeStorable = null;
38    
39    /**
40     * 
41     * @return the external subject storable
42     */
43    private static ExternalSubjectAttributeStorable externalSubjectAttributeStorable() {
44      if (externalSubjectAttributeStorable == null) {
45  
46        //externalSubjects.storage.ExternalSubjectAttributeStorable.class = edu.internet2.middleware.grouper.externalSubjects.ExternalSubjectAttributeDbStorage
47        String externalSubjectAttributeStorableClassName = StringUtils.defaultIfEmpty(
48            GrouperConfig.retrieveConfig().propertyValueString("externalSubjects.storage.ExternalSubjectAttributeStorable.class"), ExternalSubjectAttributeDbStorage.class.getName());
49        Class<ExternalSubjectAttributeStorable> externalSubjectAttributeStorableClass = GrouperUtil.forName(externalSubjectAttributeStorableClassName);
50        externalSubjectAttributeStorable = GrouperUtil.newInstance(externalSubjectAttributeStorableClass);
51        
52      }
53      return externalSubjectAttributeStorable;
54    }
55  
56    /**
57     * @see ExternalSubjectAttributeDAO#delete(ExternalSubjectAttribute)
58     * @param externalSubjectAttribute
59     */
60    public static void delete(ExternalSubjectAttribute externalSubjectAttribute) {
61      externalSubjectAttributeStorable().delete(externalSubjectAttribute);
62    }
63  
64    /**
65     * @see ExternalSubjectAttributeDAO#findBySubject(String, QueryOptions)
66     * @param subjectUuid
67     * @param queryOptions
68     * @return attributes
69     */
70    public static Set<ExternalSubjectAttribute> findBySubject(String subjectUuid,
71        QueryOptions queryOptions) {
72      return externalSubjectAttributeStorable().findBySubject(subjectUuid, queryOptions);
73    }
74  
75    /**
76     * @see ExternalSubjectAttributeDAO#findByUuid(String, boolean, QueryOptions)
77     * @param uuid
78     * @param exceptionIfNotFound
79     * @param queryOptions
80     * @return attribute
81     */
82    public static ExternalSubjectAttribute findByUuid(String uuid, boolean exceptionIfNotFound,
83        QueryOptions queryOptions) {
84      return externalSubjectAttributeStorable().findByUuid(uuid, exceptionIfNotFound, queryOptions);
85    }
86  
87    /**
88     * @see ExternalSubjectAttributeDAO#saveOrUpdate(ExternalSubjectAttribute)
89     * @param externalSubjectAttribute
90     */
91    public static void saveOrUpdate(ExternalSubjectAttribute externalSubjectAttribute) {
92      externalSubjectAttributeStorable().saveOrUpdate(externalSubjectAttribute);
93    }
94  
95  }
