27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.extended;
28  
29  import java.lang.reflect.InvocationHandler;
30  import java.lang.reflect.Proxy;
31  import java.util.ArrayList;
32  import java.util.List;
33  
34  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.ConversionException;
35  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.Converter;
36  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.MarshallingContext;
37  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.UnmarshallingContext;
38  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamReader;
39  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamWriter;
40  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.mapper.DynamicProxyMapper;
41  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.mapper.Mapper;
42  
43  /**
44   * Converts a dynamic proxy to XML, storing the implemented
45   * interfaces and handler.
46   *
47   * @author Joe Walnes
48   */
49  public class DynamicProxyConverter implements Converter {
50  
51      private ClassLoader classLoader;
52      private Mapper mapper;
53  
54      public DynamicProxyConverter(Mapper mapper) {
55          this(mapper, DynamicProxyConverter.class.getClassLoader());
56      }
57  
58      public DynamicProxyConverter(Mapper mapper, ClassLoader classLoader) {
59          this.classLoader = classLoader;
60          this.mapper = mapper;
61      }
62  
63      public boolean canConvert(Class type) {
64          return type.equals(DynamicProxyMapper.DynamicProxy.class) || Proxy.isProxyClass(type);
65      }
66  
67      public void marshal(Object source, HierarchicalStreamWriter writer, MarshallingContext context) {
68          InvocationHandler invocationHandler = Proxy.getInvocationHandler(source);
69          addInterfacesToXml(source, writer);
70          writer.startNode("handler");
71          writer.addAttribute("class", mapper.serializedClass(invocationHandler.getClass()));
72          context.convertAnother(invocationHandler);
73          writer.endNode();
74      }
75  
76      private void addInterfacesToXml(Object source, HierarchicalStreamWriter writer) {
77          Class[] interfaces = source.getClass().getInterfaces();
78          for (int i = 0; i < interfaces.length; i++) {
79              Class currentInterface = interfaces[i];
80              writer.startNode("interface");
81              writer.setValue(mapper.serializedClass(currentInterface));
82              writer.endNode();
83          }
84      }
85  
86      public Object unmarshal(HierarchicalStreamReader reader, UnmarshallingContext context) {
87          List interfaces = new ArrayList();
88          InvocationHandler handler = null;
89          while (reader.hasMoreChildren()) {
90              reader.moveDown();
91              String elementName = reader.getNodeName();
92              if (elementName.equals("interface")) {
93                  interfaces.add(mapper.realClass(reader.getValue()));
94              } else if (elementName.equals("handler")) {
95                  Class handlerType = mapper.realClass(reader.getAttribute("class"));
96                  handler = (InvocationHandler) context.convertAnother(null, handlerType);
97              }
98              reader.moveUp();
99          }
100         if (handler == null) {
101             throw new ConversionException("No InvocationHandler specified for dynamic proxy");
102         }
103         Class[] interfacesAsArray = new Class[interfaces.size()];
104         interfaces.toArray(interfacesAsArray);
105         return Proxy.newProxyInstance(classLoader, interfacesAsArray, handler);
106     }
107 }
