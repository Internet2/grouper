32  import java.beans.BeanInfo;
33  import java.beans.IntrospectionException;
34  import java.beans.Introspector;
35  import java.beans.PropertyDescriptor;
36  import java.lang.reflect.Constructor;
37  import java.lang.reflect.InvocationTargetException;
38  import java.lang.reflect.Method;
39  import java.lang.reflect.Modifier;
40  import java.util.ArrayList;
41  import java.util.Comparator;
42  import java.util.Iterator;
43  import java.util.List;
44  import java.util.Map;
45  import java.util.Set;
46  import java.util.TreeSet;
47  import java.util.WeakHashMap;

49  
50  public class BeanProvider {
51  
52      protected static final Object[] NO_PARAMS = new Object[0];
53      private final Comparator propertyNameComparator;
54      private final transient Map propertyNameCache = new WeakHashMap();
55  
56      public BeanProvider() {
57          this(null);
58      }
59  
60      public BeanProvider(final Comparator propertyNameComparator) {
61          this.propertyNameComparator = propertyNameComparator;
62      }
63      
64      public Object newInstance(Class type) {
65          try {
66              return getDefaultConstrutor(type).newInstance(NO_PARAMS);
67          } catch (InstantiationException e) {
68              throw new ObjectAccessException("Cannot construct " + type.getName(), e);
69          } catch (IllegalAccessException e) {
70              throw new ObjectAccessException("Cannot construct " + type.getName(), e);
71          } catch (InvocationTargetException e) {
72              if (e.getTargetException() instanceof RuntimeException) {
73                  throw (RuntimeException)e.getTargetException();
74              } else if (e.getTargetException() instanceof Error) {
75                  throw (Error)e.getTargetException();
76              } else {
77                  throw new ObjectAccessException("Constructor for "
78                      + type.getName()
79                      + " threw an exception", e);
80              }
81          }
82      }
83  
84      public void visitSerializableProperties(Object object, Visitor visitor) {
85          PropertyDescriptor[] propertyDescriptors = getSerializableProperties(object);
86          for (int i = 0; i < propertyDescriptors.length; i++) {
87              PropertyDescriptor property = propertyDescriptors[i];
88              try {
89                  Method readMethod = property.getReadMethod();
90                  Object value = readMethod.invoke(object, new Object[0]);
91                  visitor.visit(property.getName(), property.getPropertyType(), readMethod
92                      .getDeclaringClass(), value);
93              } catch (IllegalArgumentException e) {
94                  throw new ObjectAccessException("Could not get property "
95                      + object.getClass()
96                      + "."
97                      + property.getName(), e);
98              } catch (IllegalAccessException e) {
99                  throw new ObjectAccessException("Could not get property "
100                     + object.getClass()
101                     + "."
102                     + property.getName(), e);
103             } catch (InvocationTargetException e) {
104                 throw new ObjectAccessException("Could not get property "
105                     + object.getClass()
106                     + "."
107                     + property.getName(), e);
108             }
109         }
110     }
111 
112     public void writeProperty(Object object, String propertyName, Object value) {
113         PropertyDescriptor property = getProperty(propertyName, object.getClass());
114         try {
115             property.getWriteMethod().invoke(object, new Object[]{value});
116         } catch (IllegalArgumentException e) {
117             throw new ObjectAccessException("Could not set property "
118                 + object.getClass()
119                 + "."
120                 + property.getName(), e);
121         } catch (IllegalAccessException e) {
122             throw new ObjectAccessException("Could not set property "
123                 + object.getClass()
124                 + "."
125                 + property.getName(), e);
126         } catch (InvocationTargetException e) {
127             throw new ObjectAccessException("Could not set property "
128                 + object.getClass()
129                 + "."
130                 + property.getName(), e);
131         }
132     }
133 
134     public Class getPropertyType(Object object, String name) {
135         return getProperty(name, object.getClass()).getPropertyType();
136     }
137 
138     public boolean propertyDefinedInClass(String name, Class type) {
139         return getProperty(name, type) != null;
140     }
141 
142     /**
143      * Returns true if the Bean provider can instantiate the specified class
144      */
145     public boolean canInstantiate(Class type) {
146         return getDefaultConstrutor(type) != null;
147     }
148 
149     /**
150      * Returns the default constructor, or null if none is found
151      * 
152      * @param type
153      */
154     protected Constructor getDefaultConstrutor(Class type) {
155         Constructor[] constructors = type.getConstructors();
156         for (int i = 0; i < constructors.length; i++) {
157             Constructor c = constructors[i];
158             if (c.getParameterTypes().length == 0 && Modifier.isPublic(c.getModifiers()))
159                 return c;
160         }
161         return null;
162     }
163 
164     private PropertyDescriptor[] getSerializableProperties(Object object) {
165         Map nameMap = getNameMap(object.getClass());
166         List result = new ArrayList(nameMap.size());
167         Set names = nameMap.keySet();
168         if (propertyNameComparator != null) {
169             Set sortedSet = new TreeSet(propertyNameComparator);
170             sortedSet.addAll(names);
171             names = sortedSet;
172         }
173         for (final Iterator iter = names.iterator(); iter.hasNext();) {
174             final PropertyDescriptor descriptor = (PropertyDescriptor)nameMap.get(iter.next());
175             if (canStreamProperty(descriptor)) {
176                 result.add(descriptor);
177             }
178         }
179         return (PropertyDescriptor[])result.toArray(new PropertyDescriptor[result.size()]);
180     }
181 
182     protected boolean canStreamProperty(PropertyDescriptor descriptor) {
183         return descriptor.getReadMethod() != null && descriptor.getWriteMethod() != null;
184     }
185 
186     public boolean propertyWriteable(String name, Class type) {
187         PropertyDescriptor property = getProperty(name, type);
188         return property.getWriteMethod() != null;
189     }
190 
191     private PropertyDescriptor getProperty(String name, Class type) {
192         return (PropertyDescriptor)getNameMap(type).get(name);
193     }
194 
195     private Map getNameMap(Class type) {
196         Map nameMap = (Map)propertyNameCache.get(type);
197         if (nameMap == null) {
198             BeanInfo beanInfo;
199             try {
200                 beanInfo = Introspector.getBeanInfo(type, Object.class);
201             } catch (IntrospectionException e) {
202                 throw new ObjectAccessException("", e);
203             }
204             nameMap = new OrderRetainingMap();
205             propertyNameCache.put(type, nameMap);
206             PropertyDescriptor[] propertyDescriptors = beanInfo.getPropertyDescriptors();
207             for (int i = 0; i < propertyDescriptors.length; i++) {
208                 PropertyDescriptor descriptor = propertyDescriptors[i];
209                 nameMap.put(descriptor.getName(), descriptor);
210             }
211         }
212         return nameMap;
213     }
214 
215     interface Visitor {
216         void visit(String name, Class type, Class definedIn, Object value);
217     }
218 }
