27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.collections;

29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.Converter;
30  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.MarshallingContext;
31  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.UnmarshallingContext;
32  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.core.util.Fields;
33  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamReader;
34  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamWriter;
35  
36  import java.lang.reflect.Field;
37  import java.util.Iterator;
38  import java.util.Map;
39  import java.util.Properties;
40  import java.util.TreeMap;
41  
42  /**
43   * Special converter for java.util.Properties that stores
44   * properties in a more compact form than java.util.Map.
45   * <p/>
46   * <p>Because all entries of a Properties instance
47   * are Strings, a single element is used for each property
48   * with two attributes; one for key and one for value.</p>
49   * <p>Additionally, default properties are also serialized, if they are present.</p>
50   *
51   * @author Joe Walnes
52   * @author Kevin Ring
53   */
54  public class PropertiesConverter implements Converter {
55  
56      private final static Field defaultsField = Fields.find(Properties.class, "defaults");
57      private final boolean sort;
58  
59      public PropertiesConverter() {
60          this(false);
61      }
62  
63      public PropertiesConverter(boolean sort) {
64          this.sort = sort;
65      }
66  
67      public boolean canConvert(Class type) {
68          return Properties.class == type;
69      }
70  
71      public void marshal(Object source, HierarchicalStreamWriter writer, MarshallingContext context) {
72          Properties properties = (Properties) source;
73          Map map = sort ? (Map)new TreeMap(properties) : (Map)properties;
74          for (Iterator iterator = map.entrySet().iterator(); iterator.hasNext();) {
75              Map.Entry entry = (Map.Entry) iterator.next();
76              writer.startNode("property");
77              writer.addAttribute("name", entry.getKey().toString());
78              writer.addAttribute("value", entry.getValue().toString());
79              writer.endNode();
80          }
81          Properties defaults = (Properties) Fields.read(defaultsField, properties);
82          if (defaults != null) {
83              writer.startNode("defaults");
84              marshal(defaults, writer, context);
85              writer.endNode();
86          }
87      }
88  
89      public Object unmarshal(HierarchicalStreamReader reader, UnmarshallingContext context) {
90          Properties properties = new Properties();
91          while (reader.hasMoreChildren()) {
92              reader.moveDown();
93              if (reader.getNodeName().equals("defaults")) {
94                  Properties defaults = (Properties) unmarshal(reader, context);
95                  Fields.write(defaultsField, properties, defaults);
96              } else {
97                  String name = reader.getAttribute("name");
98                  String value = reader.getAttribute("value");
99                  properties.setProperty(name, value);
100             }
101             reader.moveUp();
102         }
103         return properties;
104     }
105 
106 }
