44  /**
45   * <p>
46   * Assists in implementing {@link Object#hashCode()} methods.
47   * </p>
48   *
49   * <p>
50   * This class enables a good <code>hashCode</code> method to be built for any class. It follows the rules laid out in
51   * the book <a href="http://java.sun.com/docs/books/effective/index.html">Effective Java</a> by Joshua Bloch. Writing a
52   * good <code>hashCode</code> method is actually quite difficult. This class aims to simplify the process.
53   * </p>
54   *
55   * <p>
56   * The following is the approach taken. When appending a data field, the current total is multiplied by the
57   * multiplier then a relevant value
58   * for that data type is added. For example, if the current hashCode is 17, and the multiplier is 37, then
59   * appending the integer 45 will create a hashcode of 674, namely 17 * 37 + 45.
60   * </p>
61   *
62   * <p>
63   * All relevant fields from the object should be included in the <code>hashCode</code> method. Derived fields may be
64   * excluded. In general, any field used in the <code>equals</code> method must be used in the <code>hashCode</code>
65   * method.
66   * </p>
67   *
68   * <p>
69   * To use this class write code as follows:
70   * </p>
71   *
72   * <pre>
73   * public class Person {
74   *   String name;
75   *   int age;
76   *   boolean smoker;
77   *   ...
78   *
79   *   public int hashCode() {
80   *     // you pick a hard-coded, randomly chosen, non-zero, odd number
81   *     // ideally different for each class
82   *     return new HashCodeBuilder(17, 37).
83   *       append(name).
84   *       append(age).
85   *       append(smoker).
86   *       toHashCode();
87   *   }
88   * }
89   * </pre>
90   *
91   * <p>
92   * If required, the superclass <code>hashCode()</code> can be added using {@link #appendSuper}.
93   * </p>
94   *
95   * <p>
96   * Alternatively, there is a method that uses reflection to determine the fields to test. Because these fields are
97   * usually private, the method, <code>reflectionHashCode</code>, uses <code>AccessibleObject.setAccessible</code>
98   * to change the visibility of the fields. This will fail under a security manager, unless the appropriate permissions
99   * are set up correctly. It is also slower than testing explicitly.
100  * </p>
101  *
102  * <p>
103  * A typical invocation for this method would look like:
104  * </p>
105  *
106  * <pre>
107  * public int hashCode() {
108  *   return HashCodeBuilder.reflectionHashCode(this);
109  * }
110  * </pre>
111  *
112  * @since 1.0
113  * @version $Id: HashCodeBuilder.java 1144929 2011-07-10 18:26:16Z ggregory $
114  */
115 public class HashCodeBuilder implements Builder<Integer> {
116     /**
117      * <p>
118      * A registry of objects used by reflection methods to detect cyclical object references and avoid infinite loops.
119      * </p>
120      *
121      * @since 2.3
122      */
123     private static final ThreadLocal<Set<IDKey>> REGISTRY = new ThreadLocal<Set<IDKey>>();
124 
125     /*
126      * NOTE: we cannot store the actual objects in a HashSet, as that would use the very hashCode()
127      * we are in the process of calculating.
128      *
129      * So we generate a one-to-one mapping from the original object to a new object.
130      *
131      * Now HashSet uses equals() to determine if two elements with the same hashcode really
132      * are equal, so we also need to ensure that the replacement objects are only equal
133      * if the original objects are identical.
134      *
135      * The original implementation (2.4 and before) used the System.indentityHashCode()
136      * method - however this is not guaranteed to generate unique ids (e.g. LANG-459)
137      *
138      * We now use the IDKey helper class (adapted from org.apache.axis.utils.IDKey)
139      * to disambiguate the duplicate ids.
140      */
141 
142     /**
143      * <p>
144      * Returns the registry of objects being traversed by the reflection methods in the current thread.
145      * </p>
146      *
147      * @return Set the registry of objects being traversed
148      * @since 2.3
149      */
150     static Set<IDKey> getRegistry() {
151         return REGISTRY.get();
152     }
153 
154     /**
155      * <p>
156      * Returns <code>true</code> if the registry contains the given object. Used by the reflection methods to avoid
157      * infinite loops.
158      * </p>
159      *
160      * @param value
161      *            The object to lookup in the registry.
162      * @return boolean <code>true</code> if the registry contains the given object.
163      * @since 2.3
164      */
165     static boolean isRegistered(Object value) {
166         Set<IDKey> registry = getRegistry();
167         return registry != null && registry.contains(new IDKey(value));
168     }

170     /**
171      * <p>
172      * Appends the fields and values defined by the given object of the given <code>Class</code>.
173      * </p>
174      *
175      * @param object
176      *            the object to append details of
177      * @param clazz
178      *            the class to append details of
179      * @param builder
180      *            the builder to append to
181      * @param useTransients
182      *            whether to use transient fields
183      * @param excludeFields
184      *            Collection of String field names to exclude from use in calculation of hash code
185      */
186     private static void reflectionAppend(Object object, Class<?> clazz, HashCodeBuilder builder, boolean useTransients,
187             String[] excludeFields) {
188         if (isRegistered(object)) {
189             return;
190         }
191         try {
192             register(object);
193             Field[] fields = clazz.getDeclaredFields();
194             AccessibleObject.setAccessible(fields, true);
195             for (Field field : fields) {
196                 if (!ArrayUtils.contains(excludeFields, field.getName())
197                     && (field.getName().indexOf('$') == -1)
198                     && (useTransients || !Modifier.isTransient(field.getModifiers()))
199                     && (!Modifier.isStatic(field.getModifiers()))) {
200                     try {
201                         Object fieldValue = field.get(object);
202                         builder.append(fieldValue);
203                     } catch (IllegalAccessException e) {
204                         // this can't happen. Would get a Security exception instead
205                         // throw a runtime exception in case the impossible happens.
206                         throw new InternalError("Unexpected IllegalAccessException");
207                     }
208                 }
209             }
210         } finally {
211             unregister(object);
212         }
213     }

216      * <p>
217      * This method uses reflection to build a valid hash code.
218      * </p>
219      *
220      * <p>
221      * It uses <code>AccessibleObject.setAccessible</code> to gain access to private fields. This means that it will
222      * throw a security exception if run under a security manager, if the permissions are not set up correctly. It is
223      * also not as efficient as testing explicitly.
224      * </p>
225      *
226      * <p>
227      * Transient members will be not be used, as they are likely derived fields, and not part of the value of the
228      * <code>Object</code>.
229      * </p>
230      *
231      * <p>
232      * Static fields will not be tested. Superclass fields will be included.
233      * </p>
234      *
235      * <p>
236      * Two randomly chosen, non-zero, odd numbers must be passed in. Ideally these should be different for each class,
237      * however this is not vital. Prime numbers are preferred, especially for the multiplier.
238      * </p>
239      *
240      * @param initialNonZeroOddNumber
241      *            a non-zero, odd number used as the initial value
242      * @param multiplierNonZeroOddNumber
243      *            a non-zero, odd number used as the multiplier
244      * @param object
245      *            the Object to create a <code>hashCode</code> for
246      * @return int hash code
247      * @throws IllegalArgumentException
248      *             if the Object is <code>null</code>
249      * @throws IllegalArgumentException
250      *             if the number is zero or even
251      */
252     public static int reflectionHashCode(int initialNonZeroOddNumber, int multiplierNonZeroOddNumber, Object object) {
253         return reflectionHashCode(initialNonZeroOddNumber, multiplierNonZeroOddNumber, object, false, null);
254     }

300     /**
301      * <p>
302      * This method uses reflection to build a valid hash code.
303      * </p>
304      *
305      * <p>
306      * It uses <code>AccessibleObject.setAccessible</code> to gain access to private fields. This means that it will
307      * throw a security exception if run under a security manager, if the permissions are not set up correctly. It is
308      * also not as efficient as testing explicitly.
309      * </p>
310      *
311      * <p>
312      * If the TestTransients parameter is set to <code>true</code>, transient members will be tested, otherwise they
313      * are ignored, as they are likely derived fields, and not part of the value of the <code>Object</code>.
314      * </p>
315      *
316      * <p>
317      * Static fields will not be included. Superclass fields will be included up to and including the specified
318      * superclass. A null superclass is treated as java.lang.Object.
319      * </p>
320      *
321      * <p>
322      * Two randomly chosen, non-zero, odd numbers must be passed in. Ideally these should be different for each class,
323      * however this is not vital. Prime numbers are preferred, especially for the multiplier.
324      * </p>
325      *
326      * @param <T>
327      *            the type of the object involved
328      * @param initialNonZeroOddNumber
329      *            a non-zero, odd number used as the initial value
330      * @param multiplierNonZeroOddNumber
331      *            a non-zero, odd number used as the multiplier
332      * @param object
333      *            the Object to create a <code>hashCode</code> for
334      * @param testTransients
335      *            whether to include transient fields
336      * @param reflectUpToClass
337      *            the superclass to reflect up to (inclusive), may be <code>null</code>
338      * @param excludeFields
339      *            array of field names to exclude from use in calculation of hash code
340      * @return int hash code
341      * @throws IllegalArgumentException
342      *             if the Object is <code>null</code>
343      * @throws IllegalArgumentException
344      *             if the number is zero or even
345      * @since 2.0
346      */
347     public static <T> int reflectionHashCode(int initialNonZeroOddNumber, int multiplierNonZeroOddNumber, T object,
348             boolean testTransients, Class<? super T> reflectUpToClass, String... excludeFields) {
349 
350         if (object == null) {
351             throw new IllegalArgumentException("The object to build a hash code for must not be null");
352         }
353         HashCodeBuilder builder = new HashCodeBuilder(initialNonZeroOddNumber, multiplierNonZeroOddNumber);
354         Class<?> clazz = object.getClass();
355         reflectionAppend(object, clazz, builder, testTransients, excludeFields);
356         while (clazz.getSuperclass() != null && clazz != reflectUpToClass) {
357             clazz = clazz.getSuperclass();
358             reflectionAppend(object, clazz, builder, testTransients, excludeFields);
359         }
360         return builder.toHashCode();
361     }

473     /**
474      * <p>
475      * Registers the given object. Used by the reflection methods to avoid infinite loops.
476      * </p>
477      *
478      * @param value
479      *            The object to register.
480      */
481     static void register(Object value) {
482         synchronized (HashCodeBuilder.class) {
483             if (getRegistry() == null) {
484                 REGISTRY.set(new HashSet<IDKey>());
485             }
486         }
487         getRegistry().add(new IDKey(value));
488     }
489 
490     /**
491      * <p>
492      * Unregisters the given object.
493      * </p>
494      *
495      * <p>
496      * Used by the reflection methods to avoid infinite loops.
497      *
498      * @param value
499      *            The object to unregister.
500      * @since 2.3
501      */
502     static void unregister(Object value) {
503         Set<IDKey> registry = getRegistry();
504         if (registry != null) {
505             registry.remove(new IDKey(value));
506             synchronized (HashCodeBuilder.class) {
507                 //read again
508                 registry = getRegistry();
509                 if (registry != null && registry.isEmpty()) {
510                     REGISTRY.remove();
511                 }
512             }
513         }
514     }

536     /**
537      * <p>
538      * Two randomly chosen, non-zero, odd numbers must be passed in. Ideally these should be different for each class,
539      * however this is not vital.
540      * </p>
541      *
542      * <p>
543      * Prime numbers are preferred, especially for the multiplier.
544      * </p>
545      *
546      * @param initialNonZeroOddNumber
547      *            a non-zero, odd number used as the initial value
548      * @param multiplierNonZeroOddNumber
549      *            a non-zero, odd number used as the multiplier
550      * @throws IllegalArgumentException
551      *             if the number is zero or even
552      */
553     public HashCodeBuilder(int initialNonZeroOddNumber, int multiplierNonZeroOddNumber) {
554         if (initialNonZeroOddNumber == 0) {
555             throw new IllegalArgumentException("HashCodeBuilder requires a non zero initial value");
556         }
557         if (initialNonZeroOddNumber % 2 == 0) {
558             throw new IllegalArgumentException("HashCodeBuilder requires an odd initial value");
559         }
560         if (multiplierNonZeroOddNumber == 0) {
561             throw new IllegalArgumentException("HashCodeBuilder requires a non zero multiplier");
562         }
563         if (multiplierNonZeroOddNumber % 2 == 0) {
564             throw new IllegalArgumentException("HashCodeBuilder requires an odd multiplier");
565         }
566         iConstant = multiplierNonZeroOddNumber;
567         iTotal = initialNonZeroOddNumber;
568     }

570     /**
571      * <p>
572      * Append a <code>hashCode</code> for a <code>boolean</code>.
573      * </p>
574      * <p>
575      * This adds <code>1</code> when true, and <code>0</code> when false to the <code>hashCode</code>.
576      * </p>
577      * <p>
578      * This is in contrast to the standard <code>java.lang.Boolean.hashCode</code> handling, which computes
579      * a <code>hashCode</code> value of <code>1231</code> for <code>java.lang.Boolean</code> instances
580      * that represent <code>true</code> or <code>1237</code> for <code>java.lang.Boolean</code> instances
581      * that represent <code>false</code>.
582      * </p>
583      * <p>
584      * This is in accordance with the <quote>Effective Java</quote> design.
585      * </p>
586      *
587      * @param value
588      *            the boolean to add to the <code>hashCode</code>
589      * @return this
590      */
591     public HashCodeBuilder append(boolean value) {
592         iTotal = iTotal * iConstant + (value ? 0 : 1);
593         return this;
594     }
595 
596     /**
597      * <p>
598      * Append a <code>hashCode</code> for a <code>boolean</code> array.
599      * </p>
600      *
601      * @param array
602      *            the array to add to the <code>hashCode</code>
603      * @return this
604      */
605     public HashCodeBuilder append(boolean[] array) {
606         if (array == null) {
607             iTotal = iTotal * iConstant;
608         } else {
609             for (boolean element : array) {
610                 append(element);
611             }
612         }
613         return this;
614     }

827     /**
828      * <p>
829      * Append a <code>hashCode</code> for an <code>Object</code>.
830      * </p>
831      *
832      * @param object
833      *            the Object to add to the <code>hashCode</code>
834      * @return this
835      */
836     public HashCodeBuilder append(Object object) {
837         if (object == null) {
838             iTotal = iTotal * iConstant;
839 
840         } else {
841             if(object.getClass().isArray()) {
842                 // 'Switch' on type of array, to dispatch to the correct handler
843                 // This handles multi dimensional arrays
844                 if (object instanceof long[]) {
845                     append((long[]) object);
846                 } else if (object instanceof int[]) {
847                     append((int[]) object);
848                 } else if (object instanceof short[]) {
849                     append((short[]) object);
850                 } else if (object instanceof char[]) {
851                     append((char[]) object);
852                 } else if (object instanceof byte[]) {
853                     append((byte[]) object);
854                 } else if (object instanceof double[]) {
855                     append((double[]) object);
856                 } else if (object instanceof float[]) {
857                     append((float[]) object);
858                 } else if (object instanceof boolean[]) {
859                     append((boolean[]) object);
860                 } else {
861                     // Not an array of primitives
862                     append((Object[]) object);
863                 }
864             } else {
865                 iTotal = iTotal * iConstant + object.hashCode();
866             }
867         }
868         return this;
869     }
