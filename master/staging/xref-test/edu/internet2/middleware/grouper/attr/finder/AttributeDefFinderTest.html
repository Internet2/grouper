36      super(name);

39    /**
40     * 
41     */
42    public void testFindByIdIndex() {
43      GrouperSession grouperSession = GrouperSession.startRootSession();
44      
45      AttributeDef attributeDef = new AttributeDefSave(grouperSession).assignName("test:testAttributeDef")
46          .assignCreateParentStemsIfNotExist(true).save();
47  
48      attributeDef.getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_VIEW, false);
49      attributeDef.getPrivilegeDelegate().revokePriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_VIEW, false);
50      attributeDef.getPrivilegeDelegate().revokePriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_READ, false);
51      
52      GrouperSession.stopQuietly(grouperSession);
53      
54      grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ0);
55  
56      AttributeDef found = AttributeDefFinder.findByIdIndexSecure(attributeDef.getIdIndex(), true, null);
57      
58      assertEquals(found.getName(), attributeDef.getName());
59      
60      GrouperSession.stopQuietly(grouperSession);
61      
62      grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ1);
63      
64      found = AttributeDefFinder.findByIdIndexSecure(attributeDef.getIdIndex(), false, null);
65      
66      assertNull(found);
67      
68      try {
69        AttributeDefFinder.findByIdIndexSecure(attributeDef.getIdIndex(), true, null);
70        fail("shouldnt get here");
71      } catch (AttributeDefNotFoundException gnfe) {
72        //good
73      }
74      
75      try {
76        AttributeDefFinder.findByIdIndexSecure(123456789L, true, null);
77        fail("shouldnt get here");
78      } catch (AttributeDefNotFoundException gnfe) {
79        //good
80      }
81  
82    }
