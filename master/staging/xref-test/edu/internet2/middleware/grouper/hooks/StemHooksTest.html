66    public void testStemPreInsert() {
67      
68      StemHooksImpl.mostRecentPreInsertStemExtension = null;
69      
70      Stem stem = StemHelper.addChildStem(this.edu, "test1", "the test1");
71      
72      assertEquals("test1", stem.getExtension());
73      assertEquals("test1", StemHooksImpl.mostRecentPreInsertStemExtension);
74      
75      try {
76        stem = StemHelper.addChildStem(this.edu, "test2", "the test2");
77        fail("Should veto test2");
78      } catch (HookVeto hookVeto) {
79        assertEquals("name cannot be test2", hookVeto.getReason());
80        assertEquals(VetoTypeGrouper.STEM_PRE_INSERT, hookVeto.getVetoType());
81      }
82      
83    }

86     * @throws StemModifyException 
87     * @throws InsufficientPrivilegeException 
88     * 
89     */
90    public void testStemPreUpdate() throws StemModifyException, InsufficientPrivilegeException {
91      
92      
93      Stem stem = StemHelper.addChildStem(this.edu, "test9", "the test9");
94      
95      StemHooksImpl.mostRecentPreUpdateStemExtension = null;
96  
97      stem.setDisplayExtension("the test9");
98      stem.store();
99  
100     assertEquals("test9", stem.getExtension());
101     assertEquals("test9", StemHooksImpl.mostRecentPreUpdateStemExtension);
102 
103     //remove hooks since an update hook is caleld on add
104     overrideHooksRemove();
105     stem = StemHelper.addChildStem(this.edu, "test10", "the test10");
106     overrideHooksAdd();
107     
108     try {
109       stem.setDisplayExtension("the test10");
110       stem.store();
111       fail("Should veto test10");
112     } catch (HookVeto hookVeto) {
113       assertEquals("name cannot be test10", hookVeto.getReason());
114       assertEquals(VetoTypeGrouper.STEM_PRE_UPDATE, hookVeto.getVetoType());
115     }
116     
117   }

276   /**
277    * @throws InsufficientPrivilegeException 
278    * @throws StemDeleteException 
279    * 
280    */
281   public void testStemPostCommitDelete() throws InsufficientPrivilegeException, StemDeleteException {
282     
283     final Stem stem = StemHelper.addChildStem(this.edu, "test4", "the test4");
284     
285     StemHooksImpl.mostRecentPostCommitDeleteStemExtension = null;
286     
287     GrouperTransaction.callbackGrouperTransaction(new GrouperTransactionHandler() {
288       
289       public Object callback(GrouperTransaction grouperTransaction)
290           throws GrouperDAOException {
291         
292         try {
293           stem.delete();
294         } catch (Exception e) {
295           throw new RuntimeException(e);
296         }
297         
298         assertNull("shouldnt fire yet", StemHooksImpl.mostRecentPostCommitDeleteStemExtension);
299         grouperTransaction.commit(GrouperCommitType.COMMIT_NOW);
300         
301         assertEquals("test4", StemHooksImpl.mostRecentPostCommitDeleteStemExtension);
302         return null;
303       }
304     });    
305   }
