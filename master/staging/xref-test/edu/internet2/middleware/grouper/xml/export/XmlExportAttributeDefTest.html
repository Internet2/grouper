87    public void testConvertToXml() {

89      XmlExportAttributeDef xmlExportAttributeDef = new XmlExportAttributeDef();

92      xmlExportAttributeDef.setAssignToAttributeDefAssn("assignToAttributeDefAssn");
93      xmlExportAttributeDef.setAssignToEffMembership("assignToEffMembership");
94      xmlExportAttributeDef.setAssignToEffMembershipAssn("assignToEffMembershipAssn");
95      xmlExportAttributeDef.setAssignToGroup("assignToGroup");
96      xmlExportAttributeDef.setAssignToGroupAssn("assignToGroupAssn");
97      xmlExportAttributeDef.setAssignToImmMembership("assignToImmMembership");
98      xmlExportAttributeDef.setAssignToImmMembershipAssn("assignToImmMembershipAssn");
99      xmlExportAttributeDef.setAssignToMember("assignToMember");
100     xmlExportAttributeDef.setAssignToMemberAssn("assignToMemberAssn");
101     xmlExportAttributeDef.setAssignToStem("assignToStem");
102     xmlExportAttributeDef.setAssignToStemAssn("assignToStemAssn");
103     xmlExportAttributeDef.setAttributeDefPublic("attributeDefPublic");
104     xmlExportAttributeDef.setAttributeDefType("attributeDefType");
105     xmlExportAttributeDef.setContextId("contextId");
106     xmlExportAttributeDef.setCreateTime("createTime");
107     xmlExportAttributeDef.setCreatorId("creatorId");
108     xmlExportAttributeDef.setDescription("description");
109     xmlExportAttributeDef.setExtension("extension");
110     xmlExportAttributeDef.setHibernateVersionNumber(3L);
111     xmlExportAttributeDef.setIdIndex(12345L);
112     xmlExportAttributeDef.setModifierTime("modifierTime");
113     xmlExportAttributeDef.setMultiAssignable("multiAssignable");
114     xmlExportAttributeDef.setMultiValued("multiValued");
115     xmlExportAttributeDef.setName("name");
116     xmlExportAttributeDef.setParentStem("parentStem");
117     xmlExportAttributeDef.setUuid("uuid");
118     xmlExportAttributeDef.setValueType("valueType");
119     
120     String xml = xmlExportAttributeDef.toXml(new GrouperVersion(GrouperVersion.GROUPER_VERSION));
121     
122     xmlExportAttributeDef = XmlExportAttributeDef.fromXml(new GrouperVersion(GrouperVersion.GROUPER_VERSION), xml);
123 
124     assertEquals("assignToAttributeDef", xmlExportAttributeDef.getAssignToAttributeDef());
125     assertEquals("assignToAttributeDefAssn", xmlExportAttributeDef.getAssignToAttributeDefAssn());
126     assertEquals("assignToEffMembership", xmlExportAttributeDef.getAssignToEffMembership());
127     assertEquals("assignToEffMembershipAssn", xmlExportAttributeDef.getAssignToEffMembershipAssn());
128     assertEquals("assignToGroup", xmlExportAttributeDef.getAssignToGroup());
129     assertEquals("assignToGroupAssn", xmlExportAttributeDef.getAssignToGroupAssn());
130     assertEquals("assignToImmMembership", xmlExportAttributeDef.getAssignToImmMembership());
131     assertEquals("assignToImmMembershipAssn", xmlExportAttributeDef.getAssignToImmMembershipAssn());
132     assertEquals("assignToMember", xmlExportAttributeDef.getAssignToMember());
133     assertEquals("assignToMemberAssn", xmlExportAttributeDef.getAssignToMemberAssn());
134     assertEquals("assignToStem", xmlExportAttributeDef.getAssignToStem());
135     assertEquals("assignToStemAssn", xmlExportAttributeDef.getAssignToStemAssn());
136     assertEquals("attributeDefPublic", xmlExportAttributeDef.getAttributeDefPublic());
137     assertEquals("attributeDefType", xmlExportAttributeDef.getAttributeDefType());
138     assertEquals("contextId", xmlExportAttributeDef.getContextId());
139     assertEquals("createTime", xmlExportAttributeDef.getCreateTime());
140     assertEquals("creatorId", xmlExportAttributeDef.getCreatorId());
141     assertEquals("description", xmlExportAttributeDef.getDescription());
142     assertEquals("extension", xmlExportAttributeDef.getExtension());
143     assertEquals(3L, xmlExportAttributeDef.getHibernateVersionNumber());
144     assertEquals(12345L, xmlExportAttributeDef.getIdIndex().longValue());
145     assertEquals("modifierTime", xmlExportAttributeDef.getModifierTime());
146     assertEquals("multiAssignable", xmlExportAttributeDef.getMultiAssignable());
147     assertEquals("multiValued", xmlExportAttributeDef.getMultiValued());

158   public void testConvertToAttributeDef() {
159     AttributeDef attributeDef = AttributeDefTest.exampleAttributeDef();
160     XmlExportAttributeDef xmlExportAttributeDef = attributeDef.xmlToExportAttributeDef(new GrouperVersion(GrouperVersion.GROUPER_VERSION));
161 
162     //now go back
163     attributeDef = xmlExportAttributeDef.toAttributeDef();
164     
165     assertEquals(true, attributeDef.isAssignToAttributeDef());
166     assertEquals(true, attributeDef.isAssignToAttributeDefAssn());
167     assertEquals(true, attributeDef.isAssignToEffMembership());
168     assertEquals(true, attributeDef.isAssignToEffMembershipAssn());
169     assertEquals(true, attributeDef.isAssignToGroup());
170     assertEquals(true, attributeDef.isAssignToGroupAssn());
171     assertEquals(true, attributeDef.isAssignToImmMembership());
172     assertEquals(true, attributeDef.isAssignToImmMembershipAssn());
173     assertEquals(true, attributeDef.isAssignToMember());
174     assertEquals(true, attributeDef.isAssignToMemberAssn());
175     assertEquals(true, attributeDef.isAssignToStem());
176     assertEquals(true, attributeDef.isAssignToStemAssn());
177     assertEquals(true, attributeDef.isAttributeDefPublic());
178     assertEquals(AttributeDefType.attr, attributeDef.getAttributeDefType());
179     assertEquals("contextId", attributeDef.getContextId());
180     assertEquals(new Long(4L), attributeDef.getCreatedOnDb());
181     assertEquals("creatorId", attributeDef.getCreatorId());
182     assertEquals("description", attributeDef.getDescription());
183     assertEquals("extension", attributeDef.getExtension());
184     assertEquals(new Long(5L), attributeDef.getHibernateVersionNumber());
185     assertEquals(new Long(12345L), attributeDef.getIdIndex());
186     assertEquals(new Long(3L), attributeDef.getLastUpdatedDb());
187     assertEquals(true, attributeDef.isMultiAssignable());
188     assertEquals(true, attributeDef.isMultiValued());
189     assertEquals("name", attributeDef.getName());
190     assertEquals("stemId", attributeDef.getStemId());
191     assertEquals("id", attributeDef.getUuid());
192     assertEquals(AttributeDefValueType.floating, attributeDef.getValueType());
193     
194   }
