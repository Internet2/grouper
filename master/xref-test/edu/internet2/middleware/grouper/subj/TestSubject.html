80      GrouperSession.startRootSession();
81      //grouper sys admin
82      Subject grouperSystem = SubjectFinder.findById("GrouperSystem", true);
83      Subject everyEntity = SubjectFinder.findById("GrouperAll", true);
84      
85      Set<Subject> subjects = SubjectFinder.findAll("groupers");
86      
87      assertEquals(1, subjects.size());
88      
89      assertTrue(SubjectHelper.inList(subjects, grouperSystem));
90      
91      subjects = SubjectFinder.findAll("rsyste");
92      
93      assertEquals(1, subjects.size());
94      
95      assertTrue(SubjectHelper.inList(subjects, grouperSystem));
96      
97      subjects = SubjectFinder.findAll("rsysa");
98      
99      assertEquals(1, subjects.size());
100     
101     assertTrue(SubjectHelper.inList(subjects, grouperSystem));    
102     
103     subjects = SubjectFinder.findAll("groupera");
104     
105     assertEquals(1, subjects.size());
106     
107     assertTrue(SubjectHelper.inList(subjects, everyEntity));  
108     
109     subjects = SubjectFinder.findAll("erall");
110     
111     assertEquals(1, subjects.size());
112     
113     assertTrue(SubjectHelper.inList(subjects, everyEntity));  
114     
115     subjects = SubjectFinder.findAll("eryen");
116     
117     assertEquals(1, subjects.size());
118

303   public void testGrouperSubjectEqual() {
304     LOG.info("testGrouperSubjectEqual");
305     try {
306       R               r       = R.populateRegistry(1, 2, 1);
307       Group           gA      = r.getGroup("a", "a");
308       Group           gB      = r.getGroup("a", "b");
309       Subject         subjA   = r.getSubject("a");
310       GrouperSubject  subjGA  = new GrouperSubject( gA);
311       Assert.assertTrue(
312         "gA == gA"    , subjGA.equals(gA.toSubject())
313       );
314       Assert.assertFalse(
315         "gA != gB"    , subjGA.equals(gB.toSubject())
316       );
317       Assert.assertFalse(
318         "gA != subjA" , subjGA.equals(subjA)
319       );
320       r.rs.stop();
321     }
322     catch (Exception e) {
323       T.e(e);
324     }
325   } // public void testGrouperSubjectEqual()
