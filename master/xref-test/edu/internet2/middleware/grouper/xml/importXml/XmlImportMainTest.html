98    public void testImport_v1_6_0() {
99      
100     File importFileXml = GrouperUtil.fileFromResourceName("edu/internet2/middleware/grouper/xml/importXml/xmlImport_v1_6_0.xml");
101     
102     GrouperSession grouperSession = GrouperSession.startRootSession();
103     
104     GrouperCheckConfig.checkGroups();
105 
106     Stem stem = new StemSave(GrouperSession.staticGrouperSession()).assignSaveMode(SaveMode.INSERT_OR_UPDATE)
107       .assignName("etc").save();
108     AttributeDef studentsAttrDef = stem.addChildAttributeDef("students", AttributeDefType.attr);
109     studentsAttrDef.setAssignToAttributeDef(true);
110     studentsAttrDef.store();
111     studentsAttrDef.setAssignToAttributeDef(false);
112     studentsAttrDef.store();
113     studentsAttrDef.setAssignToStem(true);
114     studentsAttrDef.store();
115     String studentsAttrDefUuid = studentsAttrDef.getUuid();
116     studentsAttrDef.getAttributeDefActionDelegate().addAction("someAction");
117     
118     XmlImportMain xmlImportMain = new XmlImportMain();
119     
120     
121     xmlImportMain.setRecordReport(true);
122 
123     xmlImportMain.processXml(importFileXml);
124 
125     File readonlyImportFile = new File(xmlImportMain.getRecordReportFileCanonicalPath());
126     assertTrue(readonlyImportFile.exists() || readonlyImportFile.length() > 0);
127     readonlyImportFile.delete();
128 
129     assertEquals(134,xmlImportMain.getTotalImportFileCount());
130     
131     //probably at least 2 to get started
132     assertTrue(2 < xmlImportMain.getOriginalDbCount());
133 
134     assertsFor_v1_6_0(grouperSession, studentsAttrDefUuid);
135 
136     assertTrue(0 < xmlImportMain.getInsertCount());
137     assertTrue(0 < xmlImportMain.getUpdateCount());
138 
139     //now, lets do it again
140     xmlImportMain = new XmlImportMain();
141 
142     xmlImportMain.processXml(importFileXml);
143 
144     assertEquals(134,xmlImportMain.getTotalImportFileCount());
145     
146     //probably at least 2 to get started
147     assertTrue(2 < xmlImportMain.getOriginalDbCount());
148 
149     assertsFor_v1_6_0(grouperSession, studentsAttrDefUuid);
150 
151     assertEquals(0, xmlImportMain.getInsertCount());
152     assertEquals(0, xmlImportMain.getUpdateCount());
153     assertEquals(134, xmlImportMain.getSkipCount());
154 
155     //############################
156     //try record report again, should not find anything
157     xmlImportMain = new XmlImportMain();
158     
159     xmlImportMain.setRecordReport(true);
160     
161     xmlImportMain.processXml(importFileXml);
162 
163     assertEquals(134,xmlImportMain.getTotalImportFileCount());
164 
165     assertTrue(2 < xmlImportMain.getOriginalDbCount());
166 
167     assertEquals(0, xmlImportMain.getInsertCount());
168     assertEquals(0, xmlImportMain.getUpdateCount());
169 
170     readonlyImportFile = new File(xmlImportMain.getRecordReportFileCanonicalPath());
171     assertTrue(!readonlyImportFile.exists());
172     
173     GrouperSession.stopQuietly(grouperSession);
174     
175   }

275   public void testMembershipAssignments() {
276     GrouperSession grouperSession = GrouperSession.startRootSession();
277 
278     Stem stem = StemFinder.findByName(grouperSession, "etc", true);
279 
280     Group groupA = stem.addChildGroup("groupA", "groupA");
281     groupA.addMember(SubjectTestHelper.SUBJ0);
282 
283     AttributeDef attributeDef = stem.addChildAttributeDef("attributeDef", AttributeDefType.attr);
284     attributeDef.setAssignToStem(true);
285     attributeDef.setAssignToImmMembership(true);
286     attributeDef.setAssignToImmMembershipAssn(true);
287     attributeDef.setAssignToEffMembership(true);
288     attributeDef.setValueType(AttributeDefValueType.string);
289     attributeDef.store();
290     AttributeDefName attributeDefName = stem.addChildAttributeDefName(attributeDef, "attribute", "attribute");
291 
292     Membership immediateMembership = MembershipFinder.findImmediateMembership(grouperSession, groupA, SubjectTestHelper.SUBJ0, Group.getDefaultList(), true);
293     AttributeAssign attributeAssignment1 = immediateMembership.getAttributeValueDelegate().assignValue(attributeDefName.getName(), "value1").getAttributeAssignResult().getAttributeAssign();
294     attributeAssignment1.getAttributeValueDelegate().assignValue(attributeDefName.getName(), "value2").getAttributeAssignResult().getAttributeAssign();
295 
296     // export
297     Writer w = new StringWriter();
298     XmlExportMain xmlExportMain = new XmlExportMain();
299     xmlExportMain.writeAllTables(w, "a string");
300     String xml = w.toString();
301     
302     RegistryReset.reset();
303     GrouperTest.initGroupsAndAttributes();
304     
305     // import
306     XmlImportMain xmlImportMain = new XmlImportMain();
307     xmlImportMain.setRecordReport(true);
308     xmlImportMain.processXml(xml);
309     
310     immediateMembership = MembershipFinder.findImmediateMembership(grouperSession, groupA, SubjectTestHelper.SUBJ0, Group.getDefaultList(), true);
311     assertEquals("value1", immediateMembership.getAttributeValueDelegate().retrieveValueString(attributeDefName.getName()));
312     assertEquals("value2", immediateMembership.getAttributeDelegate().retrieveAssignments().iterator().next().getAttributeValueDelegate().retrieveValueString(attributeDefName.getName()));
313   }
