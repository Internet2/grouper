217   public void testActionDirectedGraphPriority() {
218     
219     //Role<Admin> allows Action<Admin> of Resource<All>
220     this.adminRole.getPermissionRoleDelegate().assignRolePermission(this.adminString, this.all, PermissionAllowed.ALLOWED);
221 
222     //Role<Admin> denies Action<Read / write> of Resource<All>
223     this.adminRole.getPermissionRoleDelegate().assignRolePermission(this.readWriteString, this.all, PermissionAllowed.DISALLOWED);
224 
225     //User subj0 is assigned Role<Admin>
226     this.adminRole.addMember(this.subj0, true);
227 
228     //Result:
229 
230     //User subj0 is denied from Action<Read> and Action<Write> of Resource<Math> since there 
231     //are only inherited assignments and the one with the lower depth (tie in resource, 
232     //Read/Write is lower than Action<Admin>) 
233     List<PermissionEntry> permissionEntries = new ArrayList<PermissionEntry>(new PermissionFinder().addSubject(this.subj0)
234         .addAction(this.readString).addPermissionName(this.math).findPermissions());
235 
236     //there should be two, one should be allow, the other deny
237     assertEquals(2, GrouperUtil.length(permissionEntries));
238     boolean isDisallowed1 = permissionEntries.get(0).isDisallowed();
239     boolean isDisallowed2 = permissionEntries.get(1).isDisallowed();
240     assertTrue(isDisallowed1 != isDisallowed2);
241     
242     //if we filter permissions, should be one
243     permissionEntries = new ArrayList<PermissionEntry>(
244         new PermissionFinder().addSubject(this.subj0)
245         .addAction(this.readString).addPermissionName(this.math)
246         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS).findPermissions());
247     
248     //there should be one, one should be disallow
249     assertEquals(1, GrouperUtil.length(permissionEntries));
250 
251     assertTrue(permissionEntries.get(0).isDisallowed());
252 
253     //if we filter by role, it should find the disallow
254     permissionEntries = new ArrayList<PermissionEntry>(
255         new PermissionFinder().addSubject(this.subj0)
256         .addAction(this.readString).addPermissionName(this.math)
257         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES).findPermissions());
258     
259     //there should be two, one should be allow, the other deny
260     assertEquals(1, GrouperUtil.length(permissionEntries));
261     
262     assertTrue(permissionEntries.get(0).isDisallowed());
263     
264     assertFalse("wrong subject", new PermissionFinder().addSubject(SubjectTestHelper.SUBJ1)
265         .addAction(this.readString).addRole(this.adminRole)
266         .addPermissionName(this.english).hasPermission());
267     assertFalse("tie", new PermissionFinder().addSubject(this.subj0)
268         .addAction(this.readString)
269         .addPermissionName(this.math).hasPermission());
270         
271     assertFalse("tie",new PermissionFinder().addSubject(this.subj0)
272         .addAction(this.readWriteString)
273         .addPermissionName(this.math).hasPermission());
274     
275     assertTrue("tie", new PermissionFinder().addSubject(this.subj0)
276         .addAction(this.adminString)
277         .addPermissionName(this.math).hasPermission());
278         
279     assertFalse("tie", new PermissionFinder().addSubject(this.subj0)
280         .addAction(this.writeString)
281         .addPermissionName(this.math).hasPermission());
282         
283     assertFalse("wrong role", new PermissionFinder().addSubject(this.subj0)
284         .addAction(this.readString).addRole(this.seniorAdmin)
285         .addPermissionName(this.math).hasPermission());
286         
287     assertFalse("wrong role", new PermissionFinder().addSubject(this.subj0)
288         .addAction(this.readString).addRole(this.user)
289         .addPermissionName(this.math).hasPermission());
290         
291     assertFalse("inherits from arts and sciences", new PermissionFinder().addSubject(this.subj0)
292         .addAction(this.readString).addRole(this.adminRole)
293         .addPermissionName(this.english).hasPermission());
294         
295     assertFalse("inherits from engineering", new PermissionFinder().addSubject(this.subj0)
296         .addAction(this.readString).addRole(this.adminRole)
297         .addPermissionName(this.math).hasPermission());
298         
299     assertFalse("inherits from engineering", new PermissionFinder().addSubject(this.subj0)
300         .addAction(this.readString).addRole(this.adminRole)
301         .addPermissionName(this.electricalEngineering).hasPermission());
302         
303     assertFalse("directish", new PermissionFinder().addSubject(this.subj0)
304         .addAction(this.readString).addRole(this.adminRole)
305         .addPermissionName(this.artsAndSciences).hasPermission());
306         
307     assertFalse("is all", new PermissionFinder().addSubject(this.subj0)
308         .addAction(this.readString).addRole(this.adminRole)
309         .addPermissionName(this.all).hasPermission());
310         
311     assertFalse("inherits from engineering", new PermissionFinder().addSubject(this.subj0)
312         .addAction(this.readString).addRole(this.adminRole)
313         .addPermissionName(this.electricalEngineering).hasPermission());
314         
315     assertFalse("wrong action", new PermissionFinder().addSubject(this.subj0)
316         .addAction(this.readWriteString).addRole(this.adminRole)
317         .addPermissionName(this.english).hasPermission());
318     
319     assertFalse("inherits from engineering", new PermissionFinder().addSubject(this.subj0)
320         .addAction(this.readString).addRole(this.adminRole)
321         .addPermissionName(this.math).hasPermission());
