83    /** allowed prefixes for this test */
84    private static Set<String> allowedPrefixes = GrouperUtil.toSet("stem", "stem2", "parent");

86    /**
87     * filter groups for this test
88     * @param groups
89     * @return the same list for chaining
90     */
91    public static Set<Group> filterGroups(Set<Group> groups) {
92      if (groups == null) {
93        return null;
94      }
95      Iterator<Group> iterator = groups.iterator();
96      OUTER: while (iterator.hasNext()) {
97        Group group = iterator.next();
98        for (String prefix : allowedPrefixes) {
99          if (group.getName().startsWith(prefix+":")) {
100           continue OUTER;
101         }
102       }
103       //another group, dont worry about it
104       iterator.remove();
105     }
106     return groups;
107   }
108   
109   /**
110    * filter stems for this test
111    * @param stems
112    * @return the same list for chaining
113    */
114   public static Set<Stem> filterStems(Set<Stem> stems) {
115     if (stems == null) {
116       return null;
117     }
118     Iterator<Stem> iterator = stems.iterator();
119     OUTER: while (iterator.hasNext()) {
120       Stem stem = iterator.next();
121       for (String prefix : allowedPrefixes) {
122         if (stem.getName().startsWith(prefix)) {
123           continue OUTER;
124         }
125       }
126       //another group, dont worry about it
127       iterator.remove();
128     }
129     return stems;
130   }
