101   /**
102    * Asserts that two subjects are equal. If they are not
103    * an AssertionFailedError is thrown with the given message.
104    */
105   static public void assertEquals(String message, Subject expected, Subject actual) {
106     if (expected == null && actual == null)
107       return;
108     if (expected != null && SubjectHelper.eq(expected,actual))
109       return;
110     failNotEquals(message, expected, actual);
111   }

899    * @param memberships
900    */
901   public void printMemberships(Set<Object[]> results) {
902     //print out memberships
903     for (Object[] result : results) {
904       Membership resultMembership = (Membership)result[0];
905       if (!(result[1] instanceof Group)) {
906         System.out.println("Type: " + ((result[1] == null ? null : result[1].getClass().getName())));
907       }
908       String ownerName = null;
909       if (result[1] instanceof Group) {
910         ownerName = ((Group)result[1]).getName();
911       }
912       if (result[1] instanceof Stem) {
913         ownerName = ((Stem)result[1]).getName();
914       }
915       if (result[1] instanceof AttributeDef) {
916         ownerName = ((AttributeDef)result[1]).getName();
917       }
918         
919       Member resultMember = (Member)result[2];
920       
921       String memberString = null;
922       
923       if (StringUtils.equals(resultMember.getSubjectSourceId(), GrouperSourceAdapter.groupSourceId())) {
924         Group theGroup = GroupFinder.findByUuid(
925             GrouperSession.staticGrouperSession().internal_getRootSession(), resultMember.getSubjectId(), false);
926         memberString = theGroup == null ? "null group?" : theGroup.getName();
927       } else {
928         memberString = resultMember.getSubjectId();
929       }
930       
931       System.out.println("Group: " + ownerName + ", Member: " + memberString + ", Field: " + resultMembership.getListName());
932       
933     }

937   /**
938    * make sure a set of groups is similar to another by group name including order
939    * @param set1 expected set
940    * @param set2 received set
941    */
942   public void assertGroupSetsAndOrder(Set<Group> set1, Set<Group> set2) {
943     int set1length = GrouperUtil.length(set1);
944     int set2length = GrouperUtil.length(set2);
945     if (set1length != set2length) {
946       fail("Expecting groups of size: " + set1length + " but received size: " + set2length + ", expecting: "
947           + GrouperUtil.toStringForLog(set1, 200) + ", but received: " + GrouperUtil.toStringForLog(set2, 200));
948     }
949     
950     if (set1length == 0) {
951       return;
952     }
953     
954     List<Group> list2 = new ArrayList<Group>(set2);
955     int i=0;
956     for (Group group : set1) {
957       if (!StringUtils.equals(group.getName(), list2.get(i).getName())) {
958         fail("Expecting index of set: " + i + " to be: " + group.getName() + ", but received: "
959             + list2.get(i).getName() + ", expecting: " 
960             + GrouperUtil.toStringForLog(set1, 200) + ", but received: " + GrouperUtil.toStringForLog(set2, 200));
961       }
962       i++;
963     }
964   }

966   /**
967    * init groups and attributes after reset
968    */
969   public static void initGroupsAndAttributes() {
970     GrouperSession grouperSession = GrouperSession.staticGrouperSession(false);
971     boolean startedRootSession = false;
972     if (grouperSession == null) {
973       grouperSession = GrouperSession.startRootSession();
974       startedRootSession = false;
975     }
976     try {
977       if (!PrivilegeHelper.isWheelOrRoot(grouperSession.getSubject())) {
978         grouperSession = grouperSession.internal_getRootSession();
979       }
980     } catch (Exception e) {
981       //might throw an exception if wheel isnt created yet
982       grouperSession = grouperSession.internal_getRootSession();
983     }
984     try {
985       GrouperSession.callbackGrouperSession(grouperSession, new GrouperSessionHandler() {
986         
987         public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
988   
989           GrouperCheckConfig.checkGroups();
990           GrouperCheckConfig.checkAttributes();
991           return null;
992         }
993       });
994     } finally {
995       if (startedRootSession) {
996         GrouperSession.stopQuietly(grouperSession);
997       }
998     }
999   }
