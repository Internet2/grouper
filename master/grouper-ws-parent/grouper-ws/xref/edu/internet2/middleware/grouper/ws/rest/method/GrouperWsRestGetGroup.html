36  /**
37   * all first level resources on a get request
38   */
39  public enum GrouperWsRestGetGroup {
40  
41    /** group get requests for members */
42    members {
43  
44      /**
45       * handle the incoming request based on HTTP method GET and group as resource, and members as subresource
46       * @param clientVersion version of client, e.g. v1_3_000
47       * @param urlStrings not including the app name or servlet.  
48       * for http://localhost/grouper-ws/servicesRest/xhtml/v3_0_000/groups/a:b/members
49       * the urlStrings would be size three: {"group", "a:b", "member"}
50       * @param groupName in url
51       * @param requestObject is the request body converted to object
52       * @return the return object
53       */
54      @Override
55      public WsResponseBean service(
56          GrouperVersion clientVersion, String groupName, List<String> urlStrings,
57          WsRequestBean requestObject) {
58  
59        if (GrouperUtil.length(urlStrings) == 0 && (requestObject == null ||
60            requestObject instanceof WsRestGetMembersLiteRequest)) {
61          //make sure right type
62          WsRestGetMembersLiteRequest wsRestGetMembersLiteRequest = GrouperUtil.typeCast(
63              requestObject, WsRestGetMembersLiteRequest.class);
64  
65          //url should be: /xhtml/v1_3_000/groups/aStem:aGroup/members
66          return GrouperServiceRest.getMembersLite(
67              clientVersion, groupName,
68              wsRestGetMembersLiteRequest);
69          
70        }
71        
72        if (GrouperUtil.length(urlStrings) == 0 && requestObject instanceof WsRestHasMemberRequest) {
73  
74          //make sure right type
75          WsRestHasMemberRequest wsRestHasMemberRequest = GrouperUtil.typeCast(
76              requestObject, WsRestHasMemberRequest.class);
77  
78          //url should be: /xhtml/v1_3_000/groups/aStem:aGroup/members
79          return GrouperServiceRest.hasMember(
80              clientVersion, groupName,
81              wsRestHasMemberRequest);
82          
83        }
84        
85        //if the member id is on the end, then get that
86        String subjectId = GrouperServiceUtils.extractSubjectInfoFromUrlStrings(urlStrings, 0, false, false);
87        String sourceId = GrouperServiceUtils.extractSubjectInfoFromUrlStrings(urlStrings, 0, true, true);
88        
89        if (requestObject == null || requestObject instanceof WsRestHasMemberLiteRequest) {
90          //make sure right type
91          WsRestHasMemberLiteRequest wsRestHasMemberLiteRequest = GrouperUtil.typeCast(
92              requestObject, WsRestHasMemberLiteRequest.class);
93  
94          //url should be: /xhtml/v1_3_000/groups/aStem:aGroup/members
95          return GrouperServiceRest.hasMemberLite(
96              clientVersion, groupName, subjectId, sourceId,
97              wsRestHasMemberLiteRequest);
98        }
99        
100       throw new RuntimeException("Invalid REST GET Group request: " + clientVersion + " , " + groupName 
101           + ", " + subjectId + ", " + sourceId
102           + ", " + GrouperUtil.toStringForLog(urlStrings));
103     }
104 
105   }, 
106   /** group get requests for memberships */
107   memberships {
108   
109     /**
110      * handle the incoming request based on HTTP method GET and group as resource, and memberships as subresource
111      * @param clientVersion version of client, e.g. v1_3_000
112      * @param urlStrings not including the app name or servlet.  
113      * for http://localhost/grouper-ws/servicesRest/xhtml/v3_0_000/groups/a:b/memberships
114      * the urlStrings would be size three: {"group", "a:b", "member"}
115      * @param groupName in url
116      * @param requestObject is the request body converted to object
117      * @return the return object
118      */
119     @Override
120     public WsResponseBean service(
121         GrouperVersion clientVersion, String groupName, List<String> urlStrings,
122         WsRequestBean requestObject) {
123   
124       if (GrouperUtil.length(urlStrings) == 0 && (requestObject == null || 
125           requestObject instanceof WsRestGetMembershipsLiteRequest)) {
126   
127         WsRestGetMembershipsLiteRequest wsRestGetMembershipsLiteRequest = 
128           (WsRestGetMembershipsLiteRequest)requestObject;
129         
130         //url should be: /xhtml/v1_3_000/groups/a:b/memberships
131         return GrouperServiceRest.getMembershipsLite(clientVersion, groupName, null, 
132             null, wsRestGetMembershipsLiteRequest);
133         
134       }
135   
136       if (GrouperUtil.length(urlStrings) == 0 && requestObject == null || 
137           requestObject instanceof WsRestGetMembershipsRequest) {
138   
139         WsRestGetMembershipsRequest wsRestGetMembershipsRequest = 
140           (WsRestGetMembershipsRequest)requestObject;
141         
142         //url should be: /xhtml/v1_3_000/groups/a:b/memberships
143         return GrouperServiceRest.getMemberships(clientVersion, groupName, null, 
144             null, wsRestGetMembershipsRequest);
145         
146       }
147       
148       throw new RuntimeException("Invalid REST GET Group request: " + clientVersion + " , " + groupName 
149           + ", " + GrouperUtil.toStringForLog(urlStrings));
150     }
151   
152   };
153 
154   /**
155      * handle the incoming request based on HTTP method GET and group as resource
156      * @param clientVersion version of client, e.g. v1_3_000
157      * @param urlStrings not including the app name or servlet.  
158      * for http://localhost/grouper-ws/servicesRest/xhtml/v3_0_000/groups/a:b/members
159      * the urlStrings would be size three: {"group", "a:b", "members"}
160      * @param groupName in url
161      * @param requestObject is the request body converted to object
162      * @return the return object
163    */
164   public abstract WsResponseBean service(
165       GrouperVersion clientVersion, String groupName, List<String> urlStrings,
166       WsRequestBean requestObject);
167 
168   /**
169    * do a case-insensitive matching
170    * 
171    * @param string
172    * @param exceptionOnNotFound true if exception should be thrown on not found
173    * @return the enum or null or exception if not found
174    * @throws GrouperRestInvalidRequest if there is a problem
175    */
176   public static GrouperWsRestGetGroup valueOfIgnoreCase(String string,
177       boolean exceptionOnNotFound) throws GrouperRestInvalidRequest {
178     return GrouperServiceUtils.enumValueOfIgnoreCase(GrouperWsRestGetGroup.class, 
179         string, exceptionOnNotFound);
180   }
