100   /**
101    * compute a url of a resource
102    * @param resourceName
103    * @param canBeNull if cant be null, throw runtime
104    * @return the URL
105    */
106   public static URL computeUrl(String resourceName, boolean canBeNull) {
107     //get the url of the navigation file
108     ClassLoader cl = classLoader();
109 
110     URL url = null;
111 
112     try {
113       url = cl.getResource(resourceName);
114     } catch (NullPointerException npe) {
115       String error = "computeUrl() Could not find resource file: " + resourceName;
116       throw new RuntimeException(error, npe);
117     }
118 
119     if (!canBeNull && url == null) {
120       throw new RuntimeException("Cant find resource: " + resourceName);
121     }
122 
123     return url;
124   }

184   /**
185    * do a case-insensitive matching
186    * @param theEnumClass class of the enum
187    * @param <E> generic type
188    * 
189    * @param string
190    * @param exceptionOnNull true if exception should be thrown on null or blank
191    * @return the enum or null or exception if not found
192    * @throws GrouperRestInvalidRequest if there is a problem
193    */
194   public static <E extends Enum<?>> E enumValueOfIgnoreCase(Class<E> theEnumClass, String string, 
195       boolean exceptionOnNull) throws GrouperRestInvalidRequest {
196     
197     if (!exceptionOnNull && StringUtils.isBlank(string)) {
198       return null;
199     }
200     for (E e : theEnumClass.getEnumConstants()) {
201       if (StringUtils.equalsIgnoreCase(string, e.name())) {
202         return e;
203       }
204     }
205     StringBuilder error = new StringBuilder(
206         "Cant find " + theEnumClass.getSimpleName() + " from string: '").append(string);
207     error.append("', expecting one of: ");
208     for (E e : theEnumClass.getEnumConstants()) {
209       error.append(e.name()).append(", ");
210     }
211     throw new GrouperRestInvalidRequest(error.toString());
212 
213   }

234   /**
235    * <pre>
236    * from url strings, get the subjectId
237    * e.g. if the url is: /groups/aStem:aGroup/members/123412345
238    * then the index should be 3 (0 for group, 1 for group name, etc) 
239    * 
240    * if url is: /groups/aStem:aGroup/members/sourceId/someSource/subjectId/123412345
241    * then index is still 3
242    * </pre>
243    * @param urlStrings
244    * @param startIndex
245    * @param removeSubjectUrlStrings true to remove these url strings after getting data
246    * @param sourceIdOrSubjectId true to get sourceId, false to get subjectId
247    * @return the subjectId if it is found, null if not
248    */
249   public static String extractSubjectInfoFromUrlStrings(List<String> urlStrings, int startIndex, boolean sourceIdOrSubjectId,
250       boolean removeSubjectUrlStrings) {
251     String subjectId = null;
252     String sourceId = null;
253       
254     //url should be: /groups/aStem:aGroup/members/sourceIds/someSource/subjectIds/123412345
255     if (urlStrings.size() >= startIndex + 4 
256         && StringUtils.equalsIgnoreCase("sources", urlStrings.get(startIndex))
257         && StringUtils.equalsIgnoreCase("subjectId", urlStrings.get(startIndex+2))) {
258       subjectId = urlStrings.get(startIndex + 3);
259       sourceId = urlStrings.get(startIndex + 1);
260       if (removeSubjectUrlStrings) {
261         //remove 4
262         for (int i=0;i<4;i++) {
263           //these will collapse so different ones are removed
264           urlStrings.remove(startIndex);
265         }
266       }
267       //no sourceId in this setup
268       return sourceIdOrSubjectId ? sourceId : subjectId;
269     }
270   
271     // /groups/aStem:aGroup/members/123412345
272     if (urlStrings.size() >= startIndex + 1) {
273       subjectId = urlStrings.get(startIndex);
274       if (removeSubjectUrlStrings) {
275         urlStrings.remove(startIndex);
276       }
277       //no sourceId in this setup
278       return sourceIdOrSubjectId ? null : subjectId;
279     }
280     return null;
281   }

283   /**
284    * pick one of the values which is not null or empty.
285    * if both sent in, make sure they are equal.
286    * if neither, then see if ok.  return null if none sent in (blank)
287    * @param first
288    * @param second
289    * @param allowBlank if ok if blank (return null)
290    * @param reasonFieldName if not ok to be blank and both blank, then put this in error.
291    * or if both set to different things, then put this in error
292    * @return one or the other or null or exception
293    * @throws WsInvalidQueryException if there is a problem with the reason text in there
294    */
295   public static String pickOne(String first, String second, boolean allowBlank, String reasonFieldName) {
296     boolean hasFirst = !StringUtils.isBlank(first);
297     boolean hasSecond = !StringUtils.isBlank(second);
298     if (!hasFirst && !hasSecond) {
299       //maybe this is ok
300       if (allowBlank) {
301         return null;
302       }
303       //not ok
304       throw new WsInvalidQueryException("The field '" + reasonFieldName + "' is required");
305     } 
306     //this is the best situation
307     if (!(hasFirst && hasSecond)) {
308       return hasFirst ? first : second;
309     }
310     //has both first and second
311     //if both equal, return one
312     if (first.equals(second)) {
313       return first;
314     }
315     //has both and they arent equal, that is a problem
316     throw new WsInvalidQueryException("The field '" + reasonFieldName + "' is set twice to two different values: '" + first + "', '"
317         + second + "', and should just be set to one value!");
318   }

320   /**
321    * take http params and put into an object (type is specified in the http params object
322    * if there is an object specified)
323    * @param paramMap
324    * @param httpServletRequest if not null, make sure no dupes.  if null, forget it
325    * @param warnings is the warnings, if null, just throw exception
326    * @return the object or null if nothing in query params
327    */
328   public static Object marshalHttpParamsToObject(Map<String,String> paramMap, 
329       HttpServletRequest httpServletRequest, StringBuilder warnings) {
330 
331     String objectTypeSimpleName = GrouperServiceJ2ee.parameterValue(paramMap, httpServletRequest, WS_LITE_OBJECT_TYPE);
332     Set<String> namesForWarnings = new HashSet<String>();
333     //if no object, then set warnings
334     boolean hasObjectType = !StringUtils.isBlank(objectTypeSimpleName);
335     Object object = null;
336     if (!hasObjectType) {
337       namesForWarnings.addAll(paramMap.keySet());
338     } else {
339       
340       Class<?> objectClass = WsRestClassLookup.retrieveClassBySimpleName(objectTypeSimpleName); 
341       object = GrouperUtil.newInstance(objectClass);
342       
343       //lets assign other params
344       for (String key: paramMap.keySet()) {
345         if (StringUtils.equals(key, WS_LITE_OBJECT_TYPE)) {
346           continue;
347         }
348         String value = GrouperServiceJ2ee.parameterValue(paramMap, httpServletRequest, key);
349         Method method = GrouperUtil.setter(objectClass, key, true, false);
350         if (method == null) {
351           namesForWarnings.add(key);
352         } else {
353           //should we see if scalar, or try to assign?  hmmmm... hopefully the
354           //scalar list is up to date
355           if (!GrouperUtil.isScalar(method.getParameterTypes()[0])) {
356             namesForWarnings.add(key);
357           } else {
358             //if null or blank, then forget it
359             if (StringUtils.isBlank(value)) {
360               continue;
361             }
362             try {
363               GrouperUtil.assignSetter(object, key, value, true);
364             } catch (RuntimeException re) {
365               throw new WsInvalidQueryException("Problem assigning object: " + objectClass
366                   + ", property: " + key + ", value: " + value, re); 
367             }
368           }
369         }
370       }
371     }
372     StringBuilder theWarnings = new StringBuilder();
373     if (namesForWarnings.size() > 0) {
374       theWarnings.append("Cant find properties to assign HTTP params: ");
375       for (String property: namesForWarnings) {
376         theWarnings.append(property).append(", ");
377       }
378       theWarnings.append("\n");
379       if (warnings == null) {
380         throw new RuntimeException(theWarnings.toString());
381       }
382       //append the warnings to the stringbuilder passed in
383       warnings.append(theWarnings);
384     }
385     return object;
386 
387   }

389   /**
390    * convert a query string to a map (for testing purposes only, in a real system the
391    * HttpServletRequest would be used.  no dupes allowed
392    * @param queryString
393    * @return the map (null if no query string)
394    */
395   public static Map<String, String> convertQueryStringToMap(String queryString) {
396     
397     if (StringUtils.isBlank(queryString)) {
398       return null;
399     }
400     
401     Map<String, String> paramMap = new HashMap<String, String>();
402     
403     String[] queryStringPairs = GrouperUtil.splitTrim(queryString, "&");
404     
405     for (String queryStringPair : queryStringPairs) {
406       
407       String key = GrouperUtil.prefixOrSuffix(queryStringPair, "=", true);
408       String value = GrouperUtil.prefixOrSuffix(queryStringPair, "=", false);
409       //unescape the value
410       value = GrouperUtil.escapeUrlDecode(value);
411       
412       if (paramMap.containsKey(key)) {
413         throw new RuntimeException("Query string contains two of the same key: " 
414             + key + ", " + queryString);
415         
416       }
417       paramMap.put(key, value);
418     }
419     return paramMap;
420   }

422   /**
423    * convert a bean to a query string (including bean type)
424    * @param object
425    * @param convertNullToEmpty is true to convert null to empty, false to exclude nulls
426    * @param includeClassName true to include the classname in the params
427    * @return the query string
428    */
429   public static String marshalLiteBeanToQueryString(Object object, boolean convertNullToEmpty, boolean includeClassName) {
430     
431     //if no object, then dont worry about it
432     if (object == null) {
433       return "";
434     }
435     
436     StringBuilder queryString = new StringBuilder();
437     
438     //first add classname 
439     if (includeClassName) {
440       queryString.append(WS_LITE_OBJECT_TYPE).append("=").append(
441           GrouperUtil.escapeUrlEncode(object.getClass().getSimpleName()));
442     }
443    
444     Set<Method> getters = GrouperUtil.getters(object.getClass(), Object.class, null, true);
445     
446     for (Method getter : getters) {
447       
448       String propertyName = GrouperUtil.propertyName(getter);
449       //dont do getClass()
450       if (StringUtils.equals(propertyName, "class")) {
451         continue;
452       }
453       //only do scalars
454       if (!GrouperUtil.isScalar(getter.getReturnType())) {
455         throw new RuntimeException("Trying to convert an object " + object.getClass() + " to a query string, but a" +
456         		" property is not a scalar: " + getter.getName() + ": " + getter.getReturnType());
457       }
458       //get the values
459       Object value = GrouperUtil.invokeMethod(getter, object);
460       //if null forget it
461       if (value == null) {
462         if (!convertNullToEmpty) {
463           continue;
464         }
465         value = "";
466       }
467       //convert to string
468       String valueString = GrouperUtil.stringValue(value);
469       String escapedValueString = GrouperUtil.escapeUrlEncode(valueString);
470       //if not doing classname, then there might not be anything before this
471       if (queryString.length() != 0) {
472         queryString.append('&');
473       }
474       queryString.append(propertyName).append("=").append(escapedValueString);
475       
476     }
477     return queryString.toString();
478   }

483   /**
484    * take an http request and format it (assume xml or json)
485    * @param http
486    * @return format
487    */
488   public static String formatHttp(String http) {
489     //is newline correct?  platform independent?
490     http = http.trim();
491     
492     //if there are two newlines, we need to know about it, not have the blank line removed...
493     http = StringUtils.replace(http, "\n\n", "\n \n");
494     
495     String[] lines = GrouperUtil.splitTrim(http, "\n", false);
496     StringBuilder result = new StringBuilder();
497     boolean seenBody = false;
498     boolean seenBlank = false;
499     boolean seenFirstSize = false;
500     for (int i = 0; i < lines.length; i++) {
501       String line = lines[i];
502       if (StringUtils.isBlank(line)) {
503         seenBlank = true;
504         result.append(line).append("\n");
505         continue;
506       }
507       if (seenBlank) {
508         if (!seenFirstSize) {
509           //if there is no chunking, and this is last line, then format
510           if (i == lines.length-1 && lines[i].length() > 6) {
511             String indented = GrouperUtil.indent(lines[i], false);
512             result.append(indented).append("\n");
513             continue;
514           }
515           //maybe xhtml multi-line
516           if (line.startsWith("<")) {
517             StringBuilder body = new StringBuilder();
518             for (; i < lines.length; i++) {
519               body.append(lines[i]).append("\n");
520             }
521             String bodyString = body.toString();
522             String indented = GrouperUtil.indent(bodyString, false);
523             result.append(indented).append("\n");
524             i = lines.length - 1;
525             seenBody = true;
526             continue;
527           }
528           //must be first size, might need a regex here or something
529           seenFirstSize = true;
530           result.append(line).append("\n");
531           continue;
532         }
533       }
534       if (seenBlank && seenFirstSize && !seenBody) {
535         StringBuilder body = new StringBuilder();
536         for (; i < lines.length - 1; i++) {
537           body.append(lines[i]).append("\n");
538         }
539         String bodyString = body.toString();
540         String indented = GrouperUtil.indent(bodyString, false);
541         result.append(indented).append("\n");
542         i = lines.length - 2;
543         seenBody = true;
544         continue;
545       }
546       //format authorization
547       if (line.matches("^Authorization\\: Basic (.+)$")) {
548         line = "Authorization: Basic xxxxxxxxxxxxxxxxx==";
549       }
550       Matcher matcher = cookieResponsePattern.matcher(line);
551       if (matcher.matches()) {
552         line = "Set-Cookie: JSESSIONID=xxxxxxxxxxxxxxxxxxxxxxxx; Path=/" + matcher.group(1);
553       }
554       result.append(line).append("\n");
555     }
556     return result.toString();
557   }

890   /**
891    * if wrong size or a name is blank, then throw descriptive exception.  Assumes all params can be put into a map,
892    * only one name can be sent per request, no blank names.
893    * @param params
894    * @return the map of names to values, will not return null
895    * @throws WsInvalidQueryException if problem with inputs
896    */
897   public static Map<String, String> convertParamsToMap(WsParam[] params) throws WsInvalidQueryException {
898     if (params == null || params.length == 0) {
899       return new HashMap<String,String>();
900     }
901     Map<String, String> result = new LinkedHashMap<String, String>(params.length);
902     int i = 0;
903     for (WsParam param : params) {
904       if (param!=null) {
905         String paramName = param.getParamName();
906         if (StringUtils.isBlank(paramName)) {
907           throw new WsInvalidQueryException("Param name index " + i
908               + " cannot be blank: '" + paramName + "'");
909         }
910         if (result.containsKey(paramName)) {
911           throw new WsInvalidQueryException("Param name index " + i
912               + " cannot be submitted twice: '" + paramName + "'");
913         }
914         result.put(paramName, param.getParamValue());
915       }
916       i++;
917     }
918     return result;
919 
920   }

922   /**
923    * if wrong size or a name is blank, then throw descriptive exception.  Assumes all params can be put into a map,
924    * only one name can be sent per request, no blank names.
925    * @param limitEnvParams
926    * @return the map of names to values, will not return null
927    * @throws WsInvalidQueryException if problem with inputs
928    */
929   public static Map<String, Object> convertLimitsToMap(WsPermissionEnvVar[] limitEnvParams) throws WsInvalidQueryException {
930     if (limitEnvParams == null || limitEnvParams.length == 0) {
931       return new HashMap<String,Object>();
932     }
933     Map<String, Object> result = new LinkedHashMap<String, Object>(limitEnvParams.length);
934     Set<String> paramNames = new HashSet<String>();
935     int i = 0;
936     for (WsPermissionEnvVar limitEnvParam : limitEnvParams) {
937       if (limitEnvParam!=null) {
938         String envParamName = limitEnvParam.getParamName();
939         if (StringUtils.isBlank(envParamName)) {
940           throw new WsInvalidQueryException("Limit env param name index " + i
941               + " cannot be blank: '" + envParamName + "'");
942         }
943         int closeParenIndex = envParamName.indexOf(")");
944         
945         String realName = closeParenIndex == -1 ? envParamName : 
946           GrouperUtil.prefixOrSuffix(envParamName, ")", false);
947 
948         if (paramNames.contains(realName)) {
949           
950           throw new WsInvalidQueryException("Limit env param name index " + i
951               + " cannot be submitted twice: '" + envParamName + "'");
952         }
953         paramNames.add(realName);
954         
955         if (!StringUtils.isBlank(limitEnvParam.getParamType())) {
956           if (envParamName.startsWith("(")) {
957             if (!envParamName.startsWith("(" + limitEnvParam.getParamType() + ")")) {
958               throw new WsInvalidQueryException("Limit env param has typed " +
959               		"name and type which do not match: " + envParamName + ", " + limitEnvParam.getParamType());
960             }
961             //good, just leave it, defined twice, but the same way
962           } else {
963             envParamName = "(" + limitEnvParam.getParamType() + ")" + envParamName;
964           }
965         }
966         
967         result.put(envParamName, limitEnvParam.getParamValue());
968       }
969       i++;
970     }
971     return result;
972 
973   }

1337   /**
1338    * merge one array into another array and return it
1339    * @param <T> is the type that is being handled
1340    * @param toArray
1341    * @param fromArray
1342    * @param propertyNamesForEquality list of property names which are used for equality
1343    * @param theClass 
1344    * @return the new array
1345    */
1346   @SuppressWarnings("unchecked")
1347   public static <T> T[] mergeArrays(T[] toArray, T[] fromArray, String[] propertyNamesForEquality, Class<T> theClass) {
1348     
1349     //if there is nothing coming from, then return the old one (toArray)
1350     if (GrouperUtil.length(fromArray) == 0) {
1351       return toArray;
1352     }
1353     
1354     //if there is nothing in to already, then use the from one
1355     if (GrouperUtil.length(toArray) == 0) {
1356       return fromArray;
1357     }
1358     
1359     //see if they are in there
1360     List<T> itemsToAdd = new ArrayList<T>();
1361     
1362     Set<MultiKey> toArrayNames = new HashSet<MultiKey>();
1363     
1364     //add all the toArrayNames
1365     for (T item : toArray) {
1366       MultiKey multiKey = propertyNameValues(item, propertyNamesForEquality);
1367       toArrayNames.add(multiKey);
1368     }
1369     
1370     Set<MultiKey> itemNamesToAddNames = new HashSet<MultiKey>();
1371     
1372     for (T item : fromArray) {
1373       
1374       MultiKey multiKey = propertyNameValues(item, propertyNamesForEquality);
1375 
1376       if (!toArrayNames.contains(multiKey) && !itemNamesToAddNames.contains(multiKey)) {
1377         itemsToAdd.add(item);
1378         itemNamesToAddNames.add(multiKey);
1379       }
1380       
1381     }
1382     
1383     //if there are any to add, then add them
1384     T[] result = (T[])Array.newInstance(theClass, GrouperUtil.length(toArray) + itemNamesToAddNames.size());
1385 
1386     if (GrouperUtil.length(toArray) > 0) {
1387       System.arraycopy(toArray, 0, result, 0, GrouperUtil.length(toArray));
1388     }
1389     
1390     int index = GrouperUtil.length(toArray);
1391     for (T item : itemsToAdd) {
1392       result[index++] = item;
1393     }
1394     
1395     return result;
1396   }

1404   public static MultiKey propertyNameValues(Object object, String[] propertyNamesForEquality) {
