323   /**
324    * assign the code from the enum
325    * @param deleteMemberResultCode
326    */
327   public void assignResultCode(WsDeleteMemberResultCode deleteMemberResultCode) {
328     this.getResultMetadata().assignResultCode(
329         deleteMemberResultCode == null ? null : deleteMemberResultCode.name());
330     this.getResultMetadata().assignSuccess(
331         GrouperServiceUtils.booleanToStringOneChar(deleteMemberResultCode.isSuccess()));
332     if (LOG.isDebugEnabled()) {
333       LOG.debug("Set result code to : " + this.getResultMetadata().getResultCode() + ", " + this.getResultMetadata().getSuccess());
334     }
335   }

337   /**
338    * assign a resultcode of exception, and process/log the exception
339    * @param e
340    * @param wsSubjectLookup
341    */
342   public void assignResultCodeException(Exception e, WsSubjectLookup wsSubjectLookup) {
343     this.assignResultCode(WsDeleteMemberResultCode.EXCEPTION);
344     this.getResultMetadata().setResultMessage(ExceptionUtils.getFullStackTrace(e));
345     LOG.error(wsSubjectLookup + ", " + e, e);
346   }

348   /**
349    * assign a success based on four situations based on hasImmediate and hasEffective
350    * @param hasImmediate
351    * @param hasEffective
352    * @param canRead if the caller cant read then dont tell them the membership status
353    */
354   public void assignResultCodeSuccess(boolean hasImmediate, boolean hasEffective, boolean canRead) {
355     
356     if (!canRead) {
357       this.assignResultCode(WsDeleteMemberResultCode.SUCCESS);
358       return;
359     }
360     
361     //set success based on scenario
362     if (hasEffective) {
363       this
364           .assignResultCode(hasImmediate ? WsDeleteMemberResultCode.SUCCESS_BUT_HAS_EFFECTIVE
365               : WsDeleteMemberResultCode.SUCCESS_WASNT_IMMEDIATE_BUT_HAS_EFFECTIVE);
366     } else {
367       this.assignResultCode(hasImmediate ? WsDeleteMemberResultCode.SUCCESS
368           : WsDeleteMemberResultCode.SUCCESS_WASNT_IMMEDIATE);
369     }
370   }

382   /**
383    * assign the code from the enum
384    * @param wsSubjectLookup1
385    * @param subjectAttributeNames
386    * @param exceptionIfNotFound 
387    */
388   public void processSubject(WsSubjectLookup wsSubjectLookup1,
389       String[] subjectAttributeNames, boolean exceptionIfNotFound) {
390 
391     this.setWsSubject(new WsSubject(wsSubjectLookup1));
392     try {
393       this.setWsSubject(new WsSubject(wsSubjectLookup1.retrieveSubject("Subject"),
394           subjectAttributeNames, wsSubjectLookup1));
395     } catch (WsInvalidQueryException wsiqe) {
396       if (exceptionIfNotFound) {
397         throw wsiqe;
398       }
399     }
400     SubjectFindResult subjectFindResult = wsSubjectLookup1.retrieveSubjectFindResult();
401 
402     switch (subjectFindResult) {
403       case INVALID_QUERY:
404         this.assignResultCode(WsDeleteMemberResultCode.INVALID_QUERY);
405         break;
406       case SOURCE_UNAVAILABLE:
407         this.assignResultCode(WsDeleteMemberResultCode.EXCEPTION);
408         break;
409       case SUBJECT_DUPLICATE:
410         this.assignResultCode(WsDeleteMemberResultCode.SUBJECT_DUPLICATE);
411         break;
412       case SUBJECT_NOT_FOUND:
413         this.assignResultCode(WsDeleteMemberResultCode.SUBJECT_NOT_FOUND);
414         break;
415       case SUCCESS:
416         return;
417     }
418 
419     this.getResultMetadata().setResultMessage(
420         "Subject: " + wsSubjectLookup1 + " had problems: " + subjectFindResult);
421 
422   }
