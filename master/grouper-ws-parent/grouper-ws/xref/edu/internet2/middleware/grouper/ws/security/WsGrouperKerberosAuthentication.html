42  /**
43   * <pre>
44   * basic kerberos authentication for grouper, settings are specified in grouper-ws.properties
45   * note: this can be used for rest and soap, though it is not a bastion of security:
46   *  1. for soap, ws-security would be better since a ticket is passed instead of user/pass
47   *  2. for rest, Im not sure there is another option
48   *  3. the user/pass is transmitted in basic auth, so make sure SSL is on
49   *  4. passing the user/pass is not how kerberos should work since kerberos passes tickets and not passes
50   *  5. the user is authenticated to the kdc, but an ssl service is not invoked, which would be the next
51   *  level of verification since it might be possible for the kdc to be spoofed to the grouper-ws
52   * 
53   * </pre>
54   */
55  public class WsGrouperKerberosAuthentication implements WsCustomAuthentication {

80    /** logger */
81    private static final Log LOG = LogFactory.getLog(WsGrouperKerberosAuthentication.class);

83    /**
84     * this should be something like: Basic QWxhZGRabcdefGVuIHNlc2FtZQ==
85     * ^\\s*Basic\\s+([^\\s]+)$
86     * "^\\s*       Start of string and optional whitespace
87     * Basic\\s+    Must be basic auth, and have some whitespace after this
88     * ([^\\s]+)$   Capture the next non-whitespace string, then end of input
89     */
90    private static Pattern regexPattern = Pattern.compile("^\\s*Basic\\s+([^\\s]+)$");
91    
92    /**
93     * cache the logins in a hash cache
94     */
95    private static GrouperCache<String, String> loginCache = new GrouperCache<String, String>(
96        WsGrouperKerberosAuthentication.class.getName() + ".userCache", 10000, false, 60*1, 60*1, false);
97  
98    /**
99     * 
100    * @see edu.internet2.middleware.grouper.ws.security.WsCustomAuthentication#retrieveLoggedInSubjectId(javax.servlet.http.HttpServletRequest)
101    */
102   public String retrieveLoggedInSubjectId(HttpServletRequest httpServletRequest)
103       throws RuntimeException {
104     
105     String authHeader = httpServletRequest.getHeader("Authorization");
106 
107     //if not header, we cant go to kerberos
108     if (StringUtils.isBlank(authHeader)) {
109       LOG.error("No authorization header in HTTP");
110       return null;
111     }
112     
113     //hash the authHeader
114     String authHeaderHash = GrouperUtil.encryptSha(authHeader);
115     
116     String cachedLogin = loginCache.get(authHeaderHash);
117     if (!StringUtils.isBlank(cachedLogin)) {
118       LOG.debug("Retrieved cached login");
119       return cachedLogin;
120     }
121     LOG.debug("Login not in cache");
122     
123     Matcher matcher = regexPattern.matcher(authHeader);
124     String authHeaderBase64Part = null;
125     if (matcher.matches()) {
126       authHeaderBase64Part = matcher.group(1);
127     }
128     
129     //either not basic or something else wrong
130     if (StringUtils.isBlank(authHeaderBase64Part)) {
131       //dont log password
132       LOG.error("Cant find base64 part in auth header");
133       return null;
134     }
135     
136     //unencrypt this
137     byte[] base64Bytes = authHeaderBase64Part.getBytes();
138     byte[] unencodedBytes = Base64.decodeBase64(base64Bytes);
139     
140     String unencodedString = new String(unencodedBytes);
141     
142     //split based on user/pass
143     String user = GrouperUtil.prefixOrSuffix(unencodedString, ":", true);
144     String pass = GrouperUtil.prefixOrSuffix(unencodedString, ":", false);
145     
146     if (authenticateKerberos(user, pass)) {
147       
148       loginCache.put(authHeaderHash, user);
149       
150       return user;
151     }
152     
153     LOG.error("Error authenticating user: " + user);
154     return null;
155   }

157   /**
158    * see if a user and pass are correct with berberos
159    * @param principal
160    * @param password
161    * @return true for ok, false for not
162    */
163   public static boolean authenticateKerberos(String principal, String password) {
164 
165     // Obtain a LoginContext, needed for authentication. Tell it 
166     // to use the LoginModule implementation specified by the 
167     // entry named "JaasSample" in the JAAS login configuration 
168     // file and to also use the specified CallbackHandler.
169 
170     File jaasConf = GrouperServiceUtils.fileFromResourceName("jaas.conf");
171 
172     if (jaasConf == null) {
173       throw new RuntimeException("Cant find jaas.conf!");
174     }
175 
176     String krb5Location = GrouperWsConfig.retrieveConfig().propertyValueString("kerberos.krb5.conf.location");
177     
178     File krb5confFile = null;
179     
180     //first look for external central file on OS
181     if (!StringUtils.isBlank(krb5Location)) {
182       krb5confFile = new File(krb5Location);
183       if (!krb5confFile.exists() || !krb5confFile.isFile()) {
184         throw new RuntimeException("krb5 conf file in " + krb5Location + " does not exist or is not a file");
185       }
186     } else {
187          
188        krb5confFile = GrouperUtil.fileFromResourceName("krb5.conf"); 
189     }
190     
191     if (krb5confFile == null) { 
192       LOG.info("Cant find krb5.conf from classpath or config location: kerberos.krb5.conf.location"); 
193       System.setProperty("java.security.krb5.realm", GrouperWsConfig.retrieveConfig().propertyValueStringRequired("kerberos.realm"));
194       System.setProperty("java.security.krb5.kdc", GrouperWsConfig.retrieveConfig().propertyValueStringRequired("kerberos.kdc.address"));
195     } else {
196       
197       System.setProperty("java.security.krb5.conf", krb5confFile.getAbsolutePath()); 
198     }
199  
200     
201     System.setProperty("java.security.auth.login.config", jaasConf.getAbsolutePath());
202     //System.setProperty("sun.security.krb5.debug", "true");
203     
204     LoginContext lc = null;
205     try {
206       lc = new LoginContext("JaasSample", new GrouperWsKerberosHandler(principal, password));
207       
208 
209     } catch (LoginException le) {
210       LOG.error("Cannot create LoginContext. ", le);
211       return false;
212     } catch (SecurityException se) {
213       LOG.error("Cannot create LoginContext. " , se);
214       return false;
215     }
216 
217     try {
218 
219       // attempt authentication
220       lc.login();
221 
222       try {
223         lc.logout();
224       } catch (Exception e) {
225         LOG.warn(e);
226       }
227       return true;
228     } catch (LoginException le) {
229       
230       LOG.warn(le);
231     }
232 
233     return false;
234   }
