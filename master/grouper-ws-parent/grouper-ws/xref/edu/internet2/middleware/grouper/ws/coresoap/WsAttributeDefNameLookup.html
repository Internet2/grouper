142         return WsAttributeDefNameDeleteResultCode.ATTRIBUTE_DEF_NAME_UUID_DOESNT_MATCH_NAME;
143       }
144     },
145 
146     /** cant find the attributeDefName */

245   /**
246    * retrieve the attributeDefName for this lookup if not looked up yet.  pass in a grouper session
247    * @param grouperSession 
248    * @param invalidQueryReason is the text to go in the WsInvalidQueryException
249    * @return the attributeDefName
250    * @throws WsInvalidQueryException if there is a problem, and if the invalidQueryReason is set
251    */
252   public AttributeDefName retrieveAttributeDefNameIfNeeded(GrouperSession grouperSession,
253       String invalidQueryReason) throws WsInvalidQueryException {
254     
255     //see if we already retrieved
256     if (this.attributeDefNameFindResult != null) {
257       return this.attributeDefName;
258     }
259     try {
260       //assume success (set otherwise if there is a problem)
261       this.attributeDefNameFindResult = AttributeDefNameFindResult.SUCCESS;
262 
263       boolean hasUuid = !StringUtils.isBlank(this.uuid);
264 
265       boolean hasName = !StringUtils.isBlank(this.name);
266 
267       boolean hasIdIndex = !StringUtils.isBlank(this.idIndex);
268 
269       //must have a name or uuid
270       if (!hasUuid && !hasName && !hasIdIndex) {
271         this.attributeDefNameFindResult = AttributeDefNameFindResult.INVALID_QUERY;
272         if (!StringUtils.isEmpty(invalidQueryReason)) {
273           throw new WsInvalidQueryException("Invalid attributeDefName query for '"
274               + invalidQueryReason + "', " + this);
275         }
276         String logMessage = "Invalid query: " + this;
277         LOG.warn(logMessage);
278       }
279 
280       AttributeDefName theAttributeDefName = null;
281       
282       if (hasName) {
283         
284         theAttributeDefName = AttributeDefNameFinder.findByName(this.name, true);
285 
286       } else if (hasUuid) {
287         
288         theAttributeDefName = AttributeDefNameFinder.findById(this.uuid, true);
289 
290       } else if (hasIdIndex) {
291         
292         theAttributeDefName = AttributeDefNameFinder.findByIdIndexSecure(GrouperUtil.longValue(this.idIndex), true, 
293             new QueryOptions().secondLevelCache(false));
294 
295       }
296 
297       //make sure uuid matches 
298       if ((hasUuid && !StringUtils.equals(this.uuid, theAttributeDefName.getId()))
299           || (hasName && !StringUtils.equals(this.name, theAttributeDefName.getName()))
300           || (hasIdIndex && !GrouperUtil.equals(GrouperUtil.longValue(this.idIndex), theAttributeDefName.getIdIndex()))){
301         this.attributeDefNameFindResult = AttributeDefNameFindResult.ATTRIBUTE_DEF_NAME_UUID_DOESNT_MATCH_NAME;
302         String error = "AttributeDefName name '" + this.name + "' and uuid '" + this.uuid
303             + "', and idIndex: " + this.idIndex + " do not match";
304         if (!StringUtils.isEmpty(invalidQueryReason)) {
305           throw new WsInvalidQueryException(error + " for '" + invalidQueryReason
306               + "', " + this);
307         }
308         String logMessage = "Invalid query: " + this;
309         LOG.warn(logMessage);
310       }
311 
312       //success
313       this.attributeDefName = theAttributeDefName;
314 
315 
316     } catch (AttributeDefNameNotFoundException anf) {
317       this.attributeDefNameFindResult = AttributeDefNameFindResult.ATTRIBUTE_DEF_NAME_NOT_FOUND;
318       if (!StringUtils.isBlank(invalidQueryReason)) {
319         throw new WsInvalidQueryException("Invalid attributeDefName for '" + invalidQueryReason
320             + "', " + this, anf);
321       }
322     }
323     return this.attributeDefName;
324   }

326   /**
327    * retrieve the pit attribute def names for this lookup if not looked up yet.
328    * @param invalidQueryReason is the text to go in the WsInvalidQueryException
329    * @param pointInTimeFrom 
330    * @param pointInTimeTo 
331    * @return the pit attribute def name
332    * @throws WsInvalidQueryException if there is a problem, and if the invalidQueryReason is set
333    */
334   public Set<PITAttributeDefName> retrievePITAttributeDefNamesIfNeeded(String invalidQueryReason, Timestamp pointInTimeFrom, 
335       Timestamp pointInTimeTo) throws WsInvalidQueryException {
336 
337     //see if we already retrieved
338     if (this.attributeDefNameFindResult != null) {
339       return this.pitAttributeDefNames;
340     }
341 
342     //assume success (set otherwise if there is a problem)
343     this.attributeDefNameFindResult = AttributeDefNameFindResult.SUCCESS;
344     
345     try {
346       boolean hasUuid = !StringUtils.isBlank(this.uuid);
347 
348       boolean hasName = !StringUtils.isBlank(this.name);
349 
350       boolean hasIdIndex = !StringUtils.isBlank(this.idIndex);
351 
352       //must have a name or uuid
353       if (!hasUuid && !hasName) {
354         this.attributeDefNameFindResult = AttributeDefNameFindResult.INVALID_QUERY;
355 
356         String logMessage = "Invalid query: " + this;
357         
358         if (hasIdIndex) {
359           if (!StringUtils.isEmpty(invalidQueryReason)) {
360             throw new WsInvalidQueryException("Invalid attributeDefName query for '"
361                 + invalidQueryReason + "', cant lookup by idIndex: " + this);
362           }
363           logMessage = "Invalid attributeDefName query cant lookup by idIndex: " + this;
364         }
365         if (!StringUtils.isEmpty(invalidQueryReason)) {
366           throw new WsInvalidQueryException("Invalid attributeDefName query for '"
367               + invalidQueryReason + "', " + this);
368         }
369         LOG.warn(logMessage);
370       }
371 
372       if (hasUuid) {        
373         Set<PITAttributeDefName> theAttributeDefNames = PITAttributeDefNameFinder.findBySourceId(this.uuid, pointInTimeFrom, pointInTimeTo, true);
374 
375         //make sure name matches 
376         if (hasName && !StringUtils.equals(this.name, theAttributeDefNames.iterator().next().getName())) {
377           this.attributeDefNameFindResult = AttributeDefNameFindResult.ATTRIBUTE_DEF_NAME_UUID_DOESNT_MATCH_NAME;
378           String error = "AttributeDefName name '" + this.name + "' and uuid '" + this.uuid
379               + "' do not match";
380           if (!StringUtils.isEmpty(invalidQueryReason)) {
381             throw new WsInvalidQueryException(error + " for '" + invalidQueryReason
382                 + "', " + this);
383           }
384           String logMessage = "Invalid query: " + this;
385           LOG.warn(logMessage);
386         }
387 
388         //success
389         this.pitAttributeDefNames = new LinkedHashSet<PITAttributeDefName>(theAttributeDefNames);
390 
391       } else if (hasName) {
392         this.pitAttributeDefNames = PITAttributeDefNameFinder.findByName(this.name, pointInTimeFrom, pointInTimeTo, true, true);
393       }
394 
395     } catch (AttributeDefNameNotFoundException anf) {
396       this.attributeDefNameFindResult = AttributeDefNameFindResult.ATTRIBUTE_DEF_NAME_NOT_FOUND;
397       if (!StringUtils.isBlank(invalidQueryReason)) {
398         throw new WsInvalidQueryException("Invalid attributeDefName for '" + invalidQueryReason
399             + "', " + this, anf);
400       }
401     }
402     return this.pitAttributeDefNames;
403   }

467   public static Set<String> convertToAttributeDefNameIds(GrouperSession grouperSession, 
468       WsAttributeDefNameLookup[] wsAttributeDefNameLookups, StringBuilder errorMessage, AttributeDefType attributeDefType,
469       boolean usePIT, Timestamp pointInTimeFrom, Timestamp pointInTimeTo) {

471         usePIT, pointInTimeFrom, pointInTimeTo, new int[]{0});

474   /**
475    * convert attributeDefName lookups to attributeDefName ids
476    * @param grouperSession
477    * @param wsAttributeDefNameLookups
478    * @param errorMessage
479    * @param attributeDefType 
480    * @param usePIT 
481    * @param pointInTimeFrom 
482    * @param pointInTimeTo 
483    * @param lookupCount is an array of size one int where 1 will be added if there are records, and no change if not
484    * @return the attributeDefName ids
485    */
486   public static Set<String> convertToAttributeDefNameIds(GrouperSession grouperSession, 
487       WsAttributeDefNameLookup[] wsAttributeDefNameLookups, StringBuilder errorMessage, 
488       AttributeDefType attributeDefType, boolean usePIT, Timestamp pointInTimeFrom,
489       Timestamp pointInTimeTo, int[] lookupCount) {
490     //get all the attributeDefNames
491     //we could probably batch these to get better performance.
492     Set<String> attributeDefNameIds = null;
493     if (!GrouperServiceUtils.nullArray(wsAttributeDefNameLookups)) {
494       attributeDefNameIds = new LinkedHashSet<String>();
495       int i=0;
496       
497       boolean foundRecords = false;
498       
499       for (WsAttributeDefNameLookup wsAttributeDefNameLookup : wsAttributeDefNameLookups) {
500         
501         if (wsAttributeDefNameLookup == null || !wsAttributeDefNameLookup.hasData()) {
502           continue;
503         }
504         
505         if (!foundRecords) {
506           lookupCount[0]++;
507           foundRecords = true;
508         }
509         
510         if (!usePIT) {
511           wsAttributeDefNameLookup.retrieveAttributeDefNameIfNeeded(grouperSession);
512         } else {
513           wsAttributeDefNameLookup.retrievePITAttributeDefNamesIfNeeded(null, pointInTimeFrom, pointInTimeTo);
514         }
515         
516         AttributeDefName attributeDefName = wsAttributeDefNameLookup.retrieveAttributeDefName();
517         Set<PITAttributeDefName> pitAttributeDefNames = wsAttributeDefNameLookup.retrievePITAttributeDefNames();
518         
519         if (!usePIT && attributeDefName != null) {
520           if (attributeDefType != null) {
521             AttributeDef attributeDef = attributeDefName.getAttributeDef();
522             if (attributeDefType == attributeDef.getAttributeDefType()) {
523               attributeDefNameIds.add(attributeDefName.getId());
524             } else {
525               
526               if (errorMessage.length() > 0) {
527                 errorMessage.append(", ");
528               }
529               
530               errorMessage.append("Error on attributeDefName index: " + i + ", expecting attributeDefType: " 
531                   + attributeDefType + ", " 
532                   + wsAttributeDefNameLookup.toStringCompact());
533               
534             }
535           } else {
536             attributeDefNameIds.add(attributeDefName.getId());
537           }
538         } else if (usePIT && pitAttributeDefNames != null && pitAttributeDefNames.size() > 0) {
539           for (PITAttributeDefName pitAttributeDefName : pitAttributeDefNames) {
540             if (attributeDefType == null) {
541               attributeDefNameIds.add(pitAttributeDefName.getSourceId());
542             } else {
543               PITAttributeDef pitAttributeDef = PITAttributeDefFinder.findById(pitAttributeDefName.getAttributeDefId(), true);
544               if (attributeDefType.name().equals(pitAttributeDef.getAttributeDefTypeDb())) {
545                 attributeDefNameIds.add(pitAttributeDefName.getSourceId());
546               } else {
547                 if (errorMessage.length() > 0) {
548                   errorMessage.append(", ");
549                 }
550                 
551                 errorMessage.append("Error on attributeDefName index: " + i + ", expecting attributeDefType: " 
552                     + attributeDefType + ", " 
553                     + wsAttributeDefNameLookup.toStringCompact());
554                 
555               }
556             }
557           }
558         } else {
559           
560           if (errorMessage.length() > 0) {
561             errorMessage.append(", ");
562           }
563           
564           errorMessage.append("Error on attributeDefName index: " + i + ", " 
565               + wsAttributeDefNameLookup.retrieveAttributeDefNameFindResult() + ", " 
566               + wsAttributeDefNameLookup.toStringCompact());
567         }
568         
569         i++;
570       }
571       
572     }
573     return attributeDefNameIds;
574   }

578    * @return return a compact to string
579    */
580   public String toStringCompact() {
581     if (!StringUtils.isBlank(this.name)) {
582       return "name: " + this.name;

584     if (!StringUtils.isBlank(this.uuid)) {
585       return "id: " + this.uuid;

588       return "idIndex: " + this.idIndex;
589     }
590     return "blank";
591   }
