32  /**
33   * <pre>
34   * results for the stems save call.
35   * 
36   * result code:
37   * code of the result for this stem overall
38   * SUCCESS: means everything ok
39   * STEM_NOT_FOUND: cant find the stem
40   * STEM_DUPLICATE: found multiple stems
41   * </pre>
42   * @author mchyzer

46    /**
47     * result code of a request
48     */
49    public static enum WsStemSaveResultsCode implements WsResultCode {
50  
51      /** found the stems, saved them (lite http status code 201) (success: T) */
52      SUCCESS(201),
53  
54      /** either overall exception, or one or more stems had exceptions (lite http status code 500) (success: F) */
55      EXCEPTION(500),
56  
57      /** problem deleting existing stems (lite http status code 500) (success: F) */
58      PROBLEM_SAVING_STEMS(500),
59  
60      /** invalid query (e.g. if everything blank) (lite http status code 400) (success: F) */
61      INVALID_QUERY(400);
62  
63      /** get the name label for a certain version of client 
64       * @param clientVersion 
65       * @return */
66      public String nameForVersion(GrouperVersion clientVersion) {
67        return this.name();
68      }

74       * status code for rest/lite e.g. 200

77      private WsStemSaveResultsCode(int statusCode) {
78        this.httpStatusCode = statusCode;
79      }

82       * @see edu.internet2.middleware.grouper.ws.WsResultCode#getHttpStatusCode()

86      }

88      /**
89       * if this is a successful result

162   /**
163    * make sure if there is an error, to record that as an error
164    * @param grouperTransactionType for request
165    * @param theSummary
166    * @param clientVersion 
167    * @return true if success, false if not
168    */
169   public boolean tallyResults(GrouperTransactionType grouperTransactionType,
170       String theSummary, GrouperVersion clientVersion) {
171     //maybe already a failure
172     boolean successOverall = GrouperUtil.booleanValue(this.getResultMetadata()
173         .getSuccess(), true);
174     if (this.getResults() != null) {
175       // check all entries
176       int successes = 0;
177       int failures = 0;
178       for (WsStemSaveResult wsStemSaveResult : this.getResults()) {
179         boolean theSuccess = "T".equalsIgnoreCase(wsStemSaveResult.getResultMetadata()
180             .getSuccess());
181         if (theSuccess) {
182           successes++;
183         } else {
184           failures++;
185         }
186       }
187 
188       //if transaction rolled back all line items, 
189       if ((!successOverall || failures > 0) && grouperTransactionType.isTransactional()
190           && !grouperTransactionType.isReadonly()) {
191         for (WsStemSaveResult wsStemSaveResult : this.getResults()) {
192           if (GrouperUtil.booleanValue(wsStemSaveResult.getResultMetadata().getSuccess(),
193               true)) {
194             wsStemSaveResult
195                 .assignResultCode(WsStemSaveResultCode.TRANSACTION_ROLLED_BACK, clientVersion);
196             failures++;
197           }
198         }
199       }
200 
201       if (failures > 0) {
202         this.getResultMetadata().appendResultMessage(
203             "There were " + successes + " successes and " + failures
204                 + " failures of saving stems.   ");
205         this.assignResultCode(WsStemSaveResultsCode.PROBLEM_SAVING_STEMS, clientVersion);
206         //this might not be a problem
207         LOG.warn(this.getResultMetadata().getResultMessage());
208 
209       } else {
210         this.assignResultCode(WsStemSaveResultsCode.SUCCESS, clientVersion);
211       }
212     } else {
213       //none is not ok
214       this.assignResultCode(WsStemSaveResultsCode.INVALID_QUERY, clientVersion);
215       this.getResultMetadata().setResultMessage("Must pass in at least one stem to save");
216     }
217     //make response descriptive
218     if (GrouperUtil.booleanValue(this.getResultMetadata().getSuccess(), false)) {
219       this.getResultMetadata().appendResultMessage("Success for: " + theSummary);
220       return true;
221     }
222     //false if need rollback
223     return !grouperTransactionType.isTransactional();
224   }

226   /**
227    * prcess an exception, log, etc
228    * @param wsStemSaveResultsCodeOverride
229    * @param theError
230    * @param e
231    * @param clientVersion 
232    */
233   public void assignResultCodeException(
234       WsStemSaveResultsCode wsStemSaveResultsCodeOverride, String theError, 
235       Exception e, GrouperVersion clientVersion) {
236 
237     if (e instanceof WsInvalidQueryException) {
238       wsStemSaveResultsCodeOverride = GrouperUtil.defaultIfNull(
239           wsStemSaveResultsCodeOverride, WsStemSaveResultsCode.INVALID_QUERY);
240       //      if (e.getCause() instanceof StemNotFoundException) {
241       //        wsStemSaveResultsCodeOverride = WsStemSaveResultsCode.STEM_NOT_FOUND;
242       //      }
243       //a helpful exception will probably be in the getMessage()
244       this.assignResultCode(wsStemSaveResultsCodeOverride, clientVersion);
245       this.getResultMetadata().appendResultMessage(e.getMessage());
246       this.getResultMetadata().appendResultMessage(theError);
247       LOG.warn(e);
248 
249     } else {
250       wsStemSaveResultsCodeOverride = GrouperUtil.defaultIfNull(
251           wsStemSaveResultsCodeOverride, WsStemSaveResultsCode.EXCEPTION);
252       LOG.error(theError, e);
253 
254       theError = StringUtils.isBlank(theError) ? "" : (theError + ", ");
255       this.getResultMetadata().appendResultMessage(
256           theError + ExceptionUtils.getFullStackTrace(e));
257       this.assignResultCode(wsStemSaveResultsCodeOverride, clientVersion);
258 
259     }
260   }
