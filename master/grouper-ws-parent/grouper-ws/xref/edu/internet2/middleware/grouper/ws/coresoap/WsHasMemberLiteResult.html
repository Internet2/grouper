32  /**
33   * <pre>
34   * results for the has member call.
35   * 
36   * result code:
37   * code of the result for this group overall
38   * SUCCESS: means everything ok
39   * GROUP_NOT_FOUND: cant find the group
40   * GROUP_DUPLICATE: found multiple groups
41   * </pre>
42   * @author mchyzer
43   */
44  public class WsHasMemberLiteResult implements WsResponseBean, ResultMetadataHolder {
45  
46    /**
47     * logger 
48     */
49    @SuppressWarnings("unused")
50    private static final Log LOG = LogFactory.getLog(WsHasMemberLiteResult.class);
51  
52    /**
53     * result code of a request
54     */
55    public static enum WsHasMemberLiteResultCode implements WsResultCode {
56  
57      /** discovered if each was a member of not (lite http status code 404) (success: F) */
58      GROUP_NOT_FOUND(404),
59  
60      /** had an exception while figuring out if the subjects were members (lite http status code 500) (success: F) */
61      EXCEPTION(500),
62  
63      /** invalid query (e.g. if everything blank) (lite http status code 400) (success: F) */
64      INVALID_QUERY(400), 
65      
66      /** the subject is a member (lite http status code 200) (success = T) */
67      IS_MEMBER(200), 
68      
69      /** the subject was found and is not a member (lite http status code 200) (success = T) */
70      IS_NOT_MEMBER(200), 
71      
72      /** found multiple results (lite http status code 409) (success = F) */
73      SUBJECT_DUPLICATE(409), 
74      
75      /** cant find the subject (lite http status code 404) (success = F) */
76      SUBJECT_NOT_FOUND(404);
77  
78      /** get the name label for a certain version of client 
79       * @param clientVersion 
80       * @return */
81      public String nameForVersion(GrouperVersion clientVersion) {
82        return this.name();
83      }
84  
85      /** http status code for rest/lite e.g. 200 */
86      private int httpStatusCode;
87  
88      /**
89       * status code for rest/lite e.g. 200
90       * @param statusCode
91       */
92      private WsHasMemberLiteResultCode(int statusCode) {
93        this.httpStatusCode = statusCode;
94      }
95  
96      /**
97       * @see edu.internet2.middleware.grouper.ws.WsResultCode#getHttpStatusCode()
98       */
99      public int getHttpStatusCode() {
100       return this.httpStatusCode;
101     }
102 
103     /**
104      * if this is a successful result
105      * @return true if success
106      */
107     public boolean isSuccess() {
108       return this.equals(IS_MEMBER) || this.equals(IS_NOT_MEMBER);
109     }
110   }

230   /**
231    * construct from results of other
232    * @param wsHasMemberResults
233    */
234   public WsHasMemberLiteResult(WsHasMemberResults wsHasMemberResults) {
235 
236     this.getResultMetadata().copyFields(wsHasMemberResults.getResultMetadata());
237     this.setSubjectAttributeNames(wsHasMemberResults.getSubjectAttributeNames());
238     this.setWsGroup(wsHasMemberResults.getWsGroup());
239 
240     WsHasMemberResult wsHasMemberResult = GrouperServiceUtils
241         .firstInArrayOfOne(wsHasMemberResults.getResults());
242     if (wsHasMemberResult != null) {
243       this.getResultMetadata().copyFields(wsHasMemberResult.getResultMetadata());
244       this.getResultMetadata().assignResultCode(
245           wsHasMemberResult.resultCode().convertToLiteCode());
246       this.setWsSubject(wsHasMemberResult.getWsSubject());
247     }
248   }

266   /**
267    * prcess an exception, log, etc
268    * @param wsHasMemberLiteResultCodeOverride
269    * @param theError
270    * @param e
271    */
272   public void assignResultCodeException(
273       WsHasMemberLiteResultCode wsHasMemberLiteResultCodeOverride, 
274       String theError, Exception e) {
275   
276     if (e instanceof WsInvalidQueryException) {
277       wsHasMemberLiteResultCodeOverride = GrouperUtil.defaultIfNull(
278           wsHasMemberLiteResultCodeOverride, WsHasMemberLiteResultCode.INVALID_QUERY);
279       if (e.getCause() instanceof GroupNotFoundException) {
280         wsHasMemberLiteResultCodeOverride = WsHasMemberLiteResultCode.GROUP_NOT_FOUND;
281       }
282       //a helpful exception will probably be in the getMessage()
283       this.assignResultCode(wsHasMemberLiteResultCodeOverride);
284       this.getResultMetadata().appendResultMessage(e.getMessage());
285       this.getResultMetadata().appendResultMessage(theError);
286       LOG.warn(e);
287   
288     } else {
289       wsHasMemberLiteResultCodeOverride = GrouperUtil.defaultIfNull(
290           wsHasMemberLiteResultCodeOverride, WsHasMemberLiteResultCode.EXCEPTION);
291       LOG.error(theError, e);
292   
293       theError = StringUtils.isBlank(theError) ? "" : (theError + ", ");
294       this.getResultMetadata().appendResultMessage(
295           theError + ExceptionUtils.getFullStackTrace(e));
296       this.assignResultCode(wsHasMemberLiteResultCodeOverride);
297   
298     }
299   }
