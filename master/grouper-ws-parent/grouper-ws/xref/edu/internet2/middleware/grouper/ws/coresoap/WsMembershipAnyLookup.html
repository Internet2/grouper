191   /**
192    * retrieve the membership any for this lookup if not looked up yet.  pass in a grouper session
193    * @param grouperSession 
194    * @param invalidQueryReason is the text to go in the WsInvalidQueryException
195    * @return the membership
196    * @throws WsInvalidQueryException if there is a problem, and if the invalidQueryReason is set
197    */
198   public GroupMember retrieveMembershipAnyIfNeeded(GrouperSession grouperSession,
199       String invalidQueryReason) throws WsInvalidQueryException {
200     
201     //see if we already retrieved
202     if (this.membershipAnyFindResult != null) {
203       return this.groupMember;
204     }
205 
206     //assume success (set otherwise if there is a problem)
207     this.membershipAnyFindResult = MembershipAnyFindResult.SUCCESS;
208 
209     //must have a uuid or the other stuff
210     if (!hasData()) {
211       this.membershipAnyFindResult = MembershipAnyFindResult.INVALID_QUERY;
212       if (!StringUtils.isEmpty(invalidQueryReason)) {
213         throw new WsInvalidQueryException("Invalid membership any query (doesnt have data) for '"
214             + invalidQueryReason + "', " + this);
215       }
216       String logMessage = "Invalid query (no data): " + this;
217       LOG.warn(logMessage);
218     } else {
219       
220       this.wsGroupLookup.retrieveGroupIfNeeded(grouperSession);
221       this.wsSubjectLookup.retrieveSubject();
222       
223       if (GroupFindResult.SUCCESS != this.wsGroupLookup.retrieveGroupFindResult()) {
224         
225         this.errorMessage = GrouperUtil.toStringSafe(this.wsGroupLookup.retrieveGroupFindResult());
226         this.membershipAnyFindResult = MembershipAnyFindResult.INVALID_QUERY;
227         
228       } else if (SubjectFindResult.SUCCESS != this.wsSubjectLookup.retrieveSubjectFindResult()) {
229         
230         this.errorMessage = GrouperUtil.toStringSafe(this.wsSubjectLookup.retrieveSubjectFindResult());
231         this.membershipAnyFindResult = MembershipAnyFindResult.INVALID_QUERY;
232 
233       } else if (true) {
234         
235         Group group = this.wsGroupLookup.retrieveGroup();
236         Subject subject = this.wsSubjectLookup.retrieveSubject();
237         Member member = MemberFinder.findBySubject(grouperSession, subject, true);
238         this.groupMember = new GroupMember(group, member);
239       }
240       
241     }
242     return this.groupMember;
243   }

250     this.membershipAnyFindResult = null;

253   /**
254    * convert membership any lookups to membership any ids
255    * @param grouperSession
256    * @param wsMembershipAnyLookups
257    * @param errorMessage
258    * @param typeOfGroup 
259    * @param lookupCount is an array of size one int where 1 will be added if there are records, and no change if not
260    * @return the GroupMember ids
261    */
262   public static Set<MultiKey> convertToGroupMemberIds(GrouperSession grouperSession, 
263       WsMembershipAnyLookup[] wsMembershipAnyLookups, StringBuilder errorMessage, 
264       TypeOfGroup typeOfGroup, int[] lookupCount) {
265     //get all the memberships
266     //we could probably batch these to get better performance.
267     Set<MultiKey> groupMemberIds = null;
268     if (!GrouperServiceUtils.nullArray(wsMembershipAnyLookups)) {
269       
270       groupMemberIds = new LinkedHashSet<MultiKey>();
271       int i=0;
272       
273       boolean foundRecords = false;
274       
275       for (WsMembershipAnyLookup wsMembershipAnyLookup : wsMembershipAnyLookups) {
276         
277         if (wsMembershipAnyLookup == null || !wsMembershipAnyLookup.hasData()) {
278           continue;
279         }
280         
281         if (!foundRecords) {
282           lookupCount[0]++;
283           foundRecords = true;
284         }
285         
286         wsMembershipAnyLookup.retrieveMembershipAnyIfNeeded(grouperSession);
287         GroupMember groupMember = wsMembershipAnyLookup.retrieveGroupMember();
288         if (groupMember != null) {
289           if (typeOfGroup == null || typeOfGroup == groupMember.getGroup().getTypeOfGroup()) {
290             groupMemberIds.add(new MultiKey(groupMember.getGroup().getId(), groupMember.getMember().getUuid()));
291           } else {
292             if (errorMessage.length() > 0) {
293               errorMessage.append(", ");
294             }
295             
296             errorMessage.append("Error on membershipAny index: " + i 
297                 + ", expecting type of group: " + typeOfGroup + ", " 
298                 + wsMembershipAnyLookup.toStringCompact());
299             
300           }
301         } else {
302           
303           if (errorMessage.length() > 0) {
304             errorMessage.append(", ");
305           }
306           
307           errorMessage.append("Error on membershipAny index: " + i 
308               + ", " + wsMembershipAnyLookup.retrieveMembershipAnyFindResult() 
309               + ", " + wsMembershipAnyLookup.toStringCompact());
310         }
311         
312         i++;
313       }
314       
315     }
316     return groupMemberIds;
317   }

319   /**
320    * make sure this is an explicit toString
321    * @return return a compact to string
322    */
323   public String toStringCompact() {
324     StringBuilder result = new StringBuilder();
325     
326     if (this.wsGroupLookup != null) {
327       result.append("Group: ").append(this.wsGroupLookup.toStringCompact());
328     }
329     if (this.wsSubjectLookup != null) {
330       if (result.length() > 0) {
331         result.append(", ");
332       }
333       result.append(this.wsSubjectLookup.toStringCompact());
334     }
335     if (result.length() == 0) {
336       return "blank";
337     }
338     return result.toString();
339   }
