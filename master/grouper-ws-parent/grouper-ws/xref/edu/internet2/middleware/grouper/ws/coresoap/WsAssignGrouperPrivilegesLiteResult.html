56    /**
57     * construct from results of other
58     * @param wsAssignGrouperPrivilegesResults
59     * @param clientVersion 
60     */
61    public WsAssignGrouperPrivilegesLiteResult(WsAssignGrouperPrivilegesResults wsAssignGrouperPrivilegesResults) {
62  
63      this.getResultMetadata().copyFields(wsAssignGrouperPrivilegesResults.getResultMetadata());
64      this.setSubjectAttributeNames(wsAssignGrouperPrivilegesResults.getSubjectAttributeNames());
65      this.setWsGroup(wsAssignGrouperPrivilegesResults.getWsGroup());
66      this.setWsStem(wsAssignGrouperPrivilegesResults.getWsStem());
67  
68      WsAssignGrouperPrivilegesResult wsAssignGrouperPrivilegesResult = GrouperServiceUtils
69          .firstInArrayOfOne(wsAssignGrouperPrivilegesResults.getResults());
70      if (wsAssignGrouperPrivilegesResult != null) {
71        this.getResultMetadata().copyFields(wsAssignGrouperPrivilegesResult.getResultMetadata());
72        this.getResultMetadata().assignResultCode(
73            wsAssignGrouperPrivilegesResult.resultCode().convertToLiteCode());
74        this.setWsSubject(wsAssignGrouperPrivilegesResult.getWsSubject());
75        this.setAllowed(wsAssignGrouperPrivilegesResult.getAllowed());
76        this.setPrivilegeName(wsAssignGrouperPrivilegesResult.getPrivilegeName());
77        this.setPrivilegeType(wsAssignGrouperPrivilegesResult.getPrivilegeType());
78      }
79    }

81    /**
82     * assign the code from the enum
83     * @param wsSubjectLookup1
84     * @param theSubjectAttributeNames
85     */
86    public void processSubject(WsSubjectLookup wsSubjectLookup1,
87        String[] theSubjectAttributeNames) {
88  
89      SubjectFindResult subjectFindResult = wsSubjectLookup1.retrieveSubjectFindResult();
90  
91      switch (subjectFindResult) {
92        case INVALID_QUERY:
93          this.assignResultCode(WsAssignGrouperPrivilegesLiteResultCode.INVALID_QUERY);
94          break;
95        case SOURCE_UNAVAILABLE:
96          this.assignResultCode(WsAssignGrouperPrivilegesLiteResultCode.EXCEPTION);
97          break;
98        case SUBJECT_DUPLICATE:
99          this.assignResultCode(WsAssignGrouperPrivilegesLiteResultCode.SUBJECT_DUPLICATE);
100         break;
101       case SUBJECT_NOT_FOUND:
102         this.assignResultCode(WsAssignGrouperPrivilegesLiteResultCode.SUBJECT_NOT_FOUND);
103         break;
104       case SUCCESS:
105         return;
106     }
107 
108     this.getResultMetadata().setResultMessage(
109         "Subject: " + wsSubjectLookup1 + " had problems: " + subjectFindResult);
110 
111   }

125   /**
126    * prcess an exception, log, etc
127    * @param wsMemberChangeSubjectLiteResultCodeOverride
128    * @param theError
129    * @param e
130    */
131   public void assignResultCodeException(
132       WsAssignGrouperPrivilegesLiteResultCode wsMemberChangeSubjectLiteResultCodeOverride, 
133       String theError, Exception e) {
134 
135     if (e instanceof WsInvalidQueryException) {
136       wsMemberChangeSubjectLiteResultCodeOverride = GrouperUtil.defaultIfNull(
137           wsMemberChangeSubjectLiteResultCodeOverride, WsAssignGrouperPrivilegesLiteResultCode.INVALID_QUERY);
138       if (e.getCause() instanceof StemNotFoundException) {
139         wsMemberChangeSubjectLiteResultCodeOverride = WsAssignGrouperPrivilegesLiteResultCode.STEM_NOT_FOUND;
140       }
141       if (e.getCause() instanceof GroupNotFoundException) {
142         wsMemberChangeSubjectLiteResultCodeOverride = WsAssignGrouperPrivilegesLiteResultCode.GROUP_NOT_FOUND;
143       }
144       if (e.getCause() instanceof SubjectNotFoundException) {
145         wsMemberChangeSubjectLiteResultCodeOverride = WsAssignGrouperPrivilegesLiteResultCode.SUBJECT_NOT_FOUND;
146       }
147       if (e.getCause() instanceof SubjectNotUniqueException) {
148         wsMemberChangeSubjectLiteResultCodeOverride = WsAssignGrouperPrivilegesLiteResultCode.SUBJECT_DUPLICATE;
149       }
150       if (e.getCause() instanceof InsufficientPrivilegeException) {
151         wsMemberChangeSubjectLiteResultCodeOverride = WsAssignGrouperPrivilegesLiteResultCode.INSUFFICIENT_PRIVILEGES;
152       }
153       //a helpful exception will probably be in the getMessage()
154       this.assignResultCode(wsMemberChangeSubjectLiteResultCodeOverride);
155       this.getResultMetadata().appendResultMessage(e.getMessage());
156       this.getResultMetadata().appendResultMessage(theError);
157       LOG.warn(e);
158 
159     } else {
160       wsMemberChangeSubjectLiteResultCodeOverride = GrouperUtil.defaultIfNull(
161           wsMemberChangeSubjectLiteResultCodeOverride, WsAssignGrouperPrivilegesLiteResultCode.EXCEPTION);
162       LOG.error(theError, e);
163 
164       theError = StringUtils.isBlank(theError) ? "" : (theError + ", ");
165       this.getResultMetadata().appendResultMessage(
166           theError + ExceptionUtils.getFullStackTrace(e));
167       this.assignResultCode(wsMemberChangeSubjectLiteResultCodeOverride);
168 
169     }
170   }
