73    private static Pattern xmlHeader = Pattern.compile("^<\\?xml[^>]+>\\s*(.*)$", Pattern.DOTALL | Pattern.MULTILINE);

75    /** doctype pattern */
76    private static Pattern doctypeHeader = Pattern.compile("<!DOCTYPE [^>]+>\\s*(.*)$", Pattern.DOTALL | Pattern.MULTILINE);

78    /**
79     * parse a string to object 
80     * @param string
81     * @param requireHtmlHeader means that the html and body tags are required
82     * @return object
83     */
84    public Object parseXhtmlString(String string) {
85      try {
86        // process xml
87        //getting this error, cant get rid of it
88        //Server returned HTTP response code: 503 for URL: http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd
89        //strip headers
90        //<?xml version='1.0' encoding='iso-8859-1'?>
91        Matcher matcher = xmlHeader.matcher(string);
92        if (matcher.matches()) {
93          string = matcher.group(1);
94        }
95        //<!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Strict//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd">
96        matcher = doctypeHeader.matcher(string);
97        if (matcher.matches()) {
98          string = matcher.group(1);
99        }
100       
101       //lets load this into jdom, since it is xml
102       StringReader xmlReader = new StringReader(string);
103 
104       SAXBuilder saxBuilder = new SAXBuilder();
105 
106       Document document = saxBuilder.build(xmlReader);
107       return parseDocument(document);
108     } catch (Exception e) {
109       throw new RuntimeException("Problems parsing string: " + string, e);
110     }
111   }

113   /**
114    * parse a document
115    * @param document
116    * @return
117    */
118   private Object parseDocument(Document document) {
119     Element rootElement = document.getRootElement();
120 
121     this.htmlNamespace = rootElement.getNamespace();
122 
123     Element bodyElement = null;
124 
125     Element mainDiv = null;
126 
127     //html, body are optional...
128     if (StringUtils.equals(rootElement.getName(), "body")) {
129       bodyElement = rootElement;
130     } else {
131 
132       if (StringUtils.equals(rootElement.getName(), "div")) {
133         mainDiv = rootElement;
134       } else {
135         //must be html
136         //this should be "html"
137         if (!StringUtils.equals("html", rootElement.getName())) {
138           throw new RuntimeException("Expecting html root element, but was: '"
139               + rootElement.getName() + "'");
140         }
141         //get html, body (might be head in there too)
142         bodyElement = retrieveExactlyOneChild(rootElement, "body", true);
143       }
144 
145     }
146     //see if found div yet
147     if (mainDiv == null) {
148       //main div element
149       mainDiv = retrieveExactlyOneChild(bodyElement, "div", false);
150     }
151     //at this point we need a div...
152     Object object = parseObject(null, mainDiv, true, "div", true);
153 
154     return object;
155 
156   }

203   /**
204    * parse an object from a div or li, and assign to parent if there
205    * @param parent is the paren
206    * @param element is the element to parse
207    * @param errorIfProblem is if exception should be thrown if problem with type or parent setter
208    * @param expectedElementName is the element name we are expecting (li or div)
209    * @param rootDiv true if root div since that has different attributes
210    * @return the object
211    */
212   @SuppressWarnings( { "cast", "unchecked" })
213   private Object parseObject(Object parent, Element element, boolean errorIfProblem,
214       String expectedElementName, Boolean rootDiv) {
215 
216     //validate attributes
217     if (StringUtils.equals(element.getName(), "div")
218         && StringUtils.equals("div", expectedElementName)) {
219       if (rootDiv) {
220         if (!validateAttributes(element, rootDivAttributes, errorIfProblem)) {
221           return null;
222         }
223       } else {
224         if (!validateAttributes(element, divAttributes, errorIfProblem)) {
225           return null;
226         }
227       }
228     } else if (StringUtils.equals(element.getName(), "li")
229         && StringUtils.equals("li", expectedElementName)) {
230       if (!validateAttributes(element, liAttributes, errorIfProblem)) {
231         return null;
232       }
233     } else {
234       if (errorIfProblem) {
235         throw new RuntimeException("Expecting '" + expectedElementName + "', but was "
236             + element.getName());
237       }
238       return null;
239     }
240     //get the type of the object, is in title
241     String className = element.getAttributeValue("title");
242 
243     Class<?> type = null;
244 
245     if (!StringUtils.isBlank(className)) {
246       type = retrieveClass(className, errorIfProblem);
247     }
248 
249     boolean hasChildren = validateElements(element, objectSubElements);
250 
251     //if no children, nothing to do
252     if (!hasChildren) {
253       return null;
254     }
255 
256     //instantiate
257     Object object = GrouperUtil.newInstance(type);
258 
259     //now we need to parse all the fields
260     List<Element> childElements = (List<Element>) element.getChildren();
261 
262     for (Element childElement : childElements) {
263       String childElementName = childElement.getName();
264       if (StringUtils.equals("div", childElementName)) {
265         //this is an object, so assign
266         this.parseObject(object, childElement, false, "div", false);
267       } else if (StringUtils.equals("p", childElementName)) {
268         this.parseScalar(object, childElement, true);
269       } else if (StringUtils.equals("ul", childElementName)) {
270         this.parseList(object, childElement);
271       }
272       //note, if none of these, then already warned
273     }
274 
275     if (parent != null) {
276       //get the field name
277       String fieldName = element.getAttributeValue("class");
278 
279       //get setter.  if this fails, that is an error
280       GrouperUtil.assignSetter(parent, fieldName, object, true);
281     }
282 
283     return object;
284   }

286   /**
287    * parse a child list (ul tag).  If no class attribute, just ignore.
288    * If there are child elements, then it is ignored
289    * @param object to assign to
290    * @param element
291    * @param assignField is true for p tags, and false for li tags
292    */
293   @SuppressWarnings("unchecked")
294   private void parseList(Object object, Element element) {
295     //if doesnt have "class", then that is a problem
296     if (!this.validateAttributes(element, ulAttributes, false)) {
297       return;
298     }
299     //validate subelements, shouldnt be any
300     //validate so there are warnings
301 
302     if (!this.validateElements(element, ulSubElements)) {
303       return;
304     }
305     //get the field name
306     String fieldName = element.getAttributeValue("class");
307 
308     //get the children
309     List<Element> liChildren = element.getChildren("li", this.htmlNamespace);
310 
311     Method setter = GrouperUtil.setter(object.getClass(), fieldName, true, true);
312     Class setterParamType = setter.getParameterTypes()[0];
313     Class arrayType = setterParamType.getComponentType();
314     //init array based on how many li's
315     Object array = Array.newInstance(arrayType, liChildren.size());
316 
317     int index = 0;
318     for (Element liElement : liChildren) {
319       Object currentValue = null;
320       //see if scalar or not (has a title attribute?)
321       if (arrayType.getName().startsWith("java") || !arrayType.getName().contains(".")) {
322         currentValue = parseScalar(null, liElement, false);
323       } else {
324         currentValue = parseObject(null, liElement, false, "li", null);
325       }
326       if (currentValue != null) {
327         //type cast
328         currentValue = GrouperUtil.typeCast(currentValue, arrayType);
329         //assign to array
330         Array.set(array, index++, currentValue);
331       }
332     }
333 
334     //get setter.  if this fails, that is an error
335     GrouperUtil.assignSetter(object, fieldName, array, true);
336 
337   }

339   /**
340    * parse a child element (p or li tag).  If no class attribute, just ignore.
341    * however, if the type of the property is not int or string, then 
342    * there is a problem.  If there are child elements, then it is ignored
343    * @param object to assign to
344    * @param element
345    * @param assignField is true for p tags, and false for li tags
346    * @return the value or null if there is a problem
347    */
348   private String parseScalar(Object object, Element element, boolean assignField) {
349     //if doesnt have "class", then that is a problem
350     if (assignField && !this.validateAttributes(element, pAttributes, false)) {
351       return null;
352     }
353     if (!assignField && !this.validateAttributes(element, liScalarAttributes, false)) {
354       //this should never happen, no required attributes
355       return null;
356     }
357     //validate subelements, shouldnt be any
358     if (GrouperUtil.length(element.getChildren()) != 0) {
359       //validate so there are warnings
360       this.validateElements(element, scalarSubElements);
361       return null;
362     }
363     //get the field name
364     String fieldName = null;
365 
366     if (assignField) {
367       fieldName = element.getAttributeValue("class");
368     }
369 
370     //get the value, but trim to null?
371     String value = StringUtils.trimToNull(element.getTextTrim());
372 
373     if (assignField) {
374       //get setter.  if this fails, that is an error
375       GrouperUtil.assignSetter(object, fieldName, value, true);
376     }
377 
378     return value;
379 
380   }

382   /**
383    * make sure that an element has required attributes, and if others, then log
384    * @param element
385    * @param requiredAttributes (dont pass null, pass empty set if none
386    * @param errorIfMissingRequired 
387    * @return if has no problem (true) or if has problem (false)
388    */
389   @SuppressWarnings("unchecked")
390   private boolean validateAttributes(Element element, Set<String> requiredAttributes,
391       boolean errorIfMissingRequired) {
392     for (String requiredAttribute : requiredAttributes) {
393       if (element.getAttribute(requiredAttribute) == null) {
394         String error = "Element '" + element.getName()
395             + "' is missing required attribute: '" + requiredAttribute + "'";
396         if (errorIfMissingRequired) {
397           throw new RuntimeException(error);
398         }
399         this.warnings.append(error).append("\n");
400         return false;
401       }
402     }
403     //now see if any are there which shouldnt and warn
404     for (Attribute attribute : GrouperUtil.nonNull((List<Attribute>) element
405         .getAttributes())) {
406       if (!requiredAttributes.contains(attribute.getName())) {
407         this.warnings.append("Element '").append(element.getName()).append(
408             "' is not expecting attribute: '" + attribute.getName() + "'").append("\n");
409       }
410     }
411     return true;
412   }

414   /**
415    * make sure that an element has only allowed children, if not, log
416    * @param element
417    * @param allowedChildren (dont pass null, pass empty set if none)
418    * @return if this element has any allowed children, or empty
419    */
420   @SuppressWarnings( { "unchecked", "cast" })
421   private boolean validateElements(Element element, Set<String> allowedChildren) {
422 
423     List<Element> childElements = (List<Element>) element.getChildren();
424 
425     //see if empty
426     if (GrouperUtil.length(childElements) == 0) {
427       return false;
428     }
429     int allowedChildrenCount = 0;
430     for (Element childElement : childElements) {
431       if (!allowedChildren.contains(childElement.getName())) {
432         this.warnings.append("Element '").append(element.getName()).append(
433             "' is not expecting child element: '").append(childElement.getName()).append(
434             "'\n");
435       } else {
436         allowedChildrenCount++;
437       }
438     }
439     return allowedChildrenCount > 0;
440   }

442   /**
443    * make sure this element has one and only one child, and it has a certain name.  If more elements,
444    * log warning, but it must have one with this name
445    * @param element
446    * @param childName
447    * @param allowOthers true if others are allowed, not just the one we are looking for
448    * @return the child element
449    */
450   @SuppressWarnings("unchecked")
451   private Element retrieveExactlyOneChild(Element element, String childName,
452       boolean allowOthers) {
453     List<Element> children = element.getChildren();
454 
455     int length = GrouperUtil.length(children);
456 
457     if (length > 1 && !allowOthers) {
458       this.warnings.append("Element '").append(element.getName()).append(
459           "' should have one child '").append(childName).append("', but instead has ")
460           .append(length).append(" children.  \n");
461 
462       Set<String> childSet = new LinkedHashSet<String>();
463       for (Element child : children) {
464         if (!StringUtils.equals(childName, child.getName())) {
465           childSet.add(child.getName());
466         }
467       }
468       this.warnings.append("Element '").append(element.getName()).append(
469           "' should only have children named '").append(childName).append(
470           "' but instead has ");
471       for (String childInvalidName : childSet) {
472         this.warnings.append(childInvalidName).append(", ");
473       }
474       this.warnings.append("\n");
475       //get the first one
476       return element.getChild(childName, this.htmlNamespace);
477     }
478     //if one element, and right name
479     children = element.getChildren(childName, this.htmlNamespace);
480 
481     if (children.size() == 1) {
482       return children.get(0);
483     } else if (children.size() > 1) {
484       throw new RuntimeException("Looking in element: '" + element.getName()
485           + "' for child" + " with name: '" + childName + "', but found multiple ("
486           + children.size() + ".  ");
487     }
488 
489     throw new RuntimeException("Looking in element: '" + element.getName()
490         + "' for child" + " with name: '" + childName + "', but cant find it.  ");
491   }

493   /**
494    * based on className which can be fully qualified or not, 
495    * return the class object (and cache this)
496    * @param className
497    * @param errorIfProblem 
498    * @return the class
499    */
500   public Class<?> retrieveClass(String className, boolean errorIfProblem) {
501     
502     if (this.classLookup.containsKey(className)) {
503       return this.classLookup.get(className);
504     }
505     
506     if (errorIfProblem) {
507       return WsRestClassLookup.retrieveClassBySimpleName(className);
508     }
509     return WsRestClassLookup.retrieveClassBySimpleName(className, this.warnings);
510   }
