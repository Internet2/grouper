41    /**
42     * 
43     * @param source
44     */
45    public SubjectStatusConfig(Source source) {
46  
47      this.sourceId = source.getId();
48      this.statusAllFromUser = source.getInitParam("statusAllFromUser");
49      this.statusDatastoreFieldName = source.getInitParam("statusDatastoreFieldName");
50      
51      //lets make these to lower so they are easy to search for
52      {
53        String statusesFromUser = source.getInitParam("statusesFromUser");
54        if (!StringUtils.isBlank(statusesFromUser)) {
55          for (String statusFromUser : SubjectUtils.splitTrim(statusesFromUser, ",")) {
56            this.statusesFromUser.add(statusFromUser.toLowerCase());
57          }
58        }
59      }
60      
61      this.statusLabel = source.getInitParam("statusLabel");
62      this.statusSearchDefault = source.getInitParam("statusSearchDefault");
63      
64      for (int i=0;i<50;i++) {
65        
66        String statusTranslateUser = source.getInitParam("statusTranslateUser" + i);
67        String statusTranslateDatastore = source.getInitParam("statusTranslateDatastore" + i);
68        
69        if (StringUtils.isBlank(statusTranslateUser)) {
70          break;
71        }
72        
73        if (StringUtils.isBlank(statusTranslateDatastore)) {
74          throw new RuntimeException("Why is statusTranslateDatastore" + i + " blank in sources.xml for source: " + sourceId);
75        }
76        
77        this.statusTranslateUserToDatastore.put(statusTranslateUser, statusTranslateDatastore);
78      }
79      
80  
81    }
82

85     * @return the status
86     */

88      return !StringUtils.isBlank(this.statusDatastoreFieldName);
89    }

111   /**
112    * see if the configuration is ok for this source
113    */
114   public void validate() {
115     
116     boolean hasStatusAllFromUser = !StringUtils.isBlank(this.statusAllFromUser);
117     
118     boolean hasStatusDatastoreFieldName = !StringUtils.isBlank(this.statusDatastoreFieldName);
119 
120     boolean hasStatusesFromUser = SubjectUtils.length(this.statusesFromUser) > 0;
121     
122     boolean hasStatusLabel = !StringUtils.isBlank(this.statusLabel);
123     
124     boolean hasStatusSearchDefault = !StringUtils.isBlank(this.statusLabel);
125     
126     boolean hasStatusTemplateUserToDatastore = SubjectUtils.length(this.statusTranslateUserToDatastore) > 0;
127     
128     //has nothing, thats ok
129     if (!hasStatusAllFromUser && !hasStatusDatastoreFieldName && !hasStatusesFromUser 
130         && !hasStatusLabel && !hasStatusSearchDefault && !hasStatusTemplateUserToDatastore) {
131       return;
132     }
133 
134     //which ones are required?
135     if (!hasStatusDatastoreFieldName) {
136       throw new RuntimeException("'statusDatastoreFieldName' is required in the configuration of source: " + this.getSourceId());
137     }
138 
139     if (!hasStatusLabel) {
140       throw new RuntimeException("'statusLabel' is required in the configuration of source: " + this.getSourceId());
141     }
142     
143     if (!hasStatusAllFromUser) {
144       throw new RuntimeException("'statusAllFromUser' is required in the configuration of source: " + this.getSourceId());
145     }
146     
147     if (!hasStatusSearchDefault) {
148       throw new RuntimeException("'statusSearchDefault' is required in the configuration of source: " + this.getSourceId());
149     }
150     
151     
152     
153     
154   }
