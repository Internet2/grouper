34    /**
35     * append a string to a stringbuilder and a space in between if necessary
36     * @param a builder to append to
37     * @param b string to add
38     */
39    public static void appendWithSpace(StringBuilder a, String b) {
40      if (StringUtils.isBlank(b)) {
41        return;
42      }
43      
44      if (StringUtils.isBlank(a.toString())) {
45        a.setLength(0);
46        
47      } else {
48        a.append(" ");
49      }
50      
51      a.append(StringUtils.trimToEmpty(b));
52      
53    }

175   /**
176    * process the search string.  set the fields in this object or status value,
177    * isequals, and stripped query
178    * @param firstPass true if this is the first pass through the process
179    */
180   private void processOriginalQueryHelper(boolean firstPass) {
181 
182     //operate on the stripped query, since it will be evolving
183     String queryToOperateOn = firstPass ? StringUtils.defaultString(this.originalQuery) : StringUtils.defaultString(this.queryWithDefault);
184     
185     boolean hasStatus = this.subjectStatusConfig.isStatusConfigured();
186     
187     if (hasStatus && StringUtils.isBlank(this.subjectStatusConfig.getStatusLabel())) {
188       throw new RuntimeException("Cant find status label for source");
189     }
190     
191     boolean foundStatus = false;
192     
193     for (Pattern pattern : new Pattern[]{statusPattern}) {
194        
195       //lets look for a status part
196       Matcher matcher = pattern.matcher(queryToOperateOn);
197       
198       StringBuilder queryToOperateOnBuilder = new StringBuilder();
199       
200       //current index of query
201       int currentIndex = 0;
202       
203       while (matcher.find(currentIndex)) {
204         
205         String operation = matcher.group(1).toLowerCase();
206         String equalsOperator = matcher.group(2);
207         String value = matcher.group(3);
208         
209         boolean valueValid = true;
210         
211         if (this.subjectStatusConfig.getStatusesFromUser().size() > 0) {
212           valueValid = this.subjectStatusConfig.getStatusesFromUser().contains(value.toLowerCase());
213         }
214         
215         //if this is a match for this operator
216         if (StringUtils.equalsIgnoreCase(operation, this.subjectStatusConfig.getStatusLabel()) && valueValid) {
217           
218           //just use the first status
219           if (!foundStatus) {
220             
221             this.equalsFromUser = StringUtils.equals("=", equalsOperator);
222             this.statusValueFromUser = value;
223             foundStatus = true;
224             
225           }
226           
227           //strip it out
228           appendWithSpace(queryToOperateOnBuilder, queryToOperateOn.substring(currentIndex, matcher.start()));
229           
230         } else {
231           //see if it is a status label in another source
232           if (SourceManager.getInstance().getSourceManagerStatusBean().getStatusLabels().contains(operation.toLowerCase())) {
233 
234             //strip it out since it is applicable to another source... right?  We want it stripped out??? 
235             appendWithSpace(queryToOperateOnBuilder, queryToOperateOn.substring(currentIndex, matcher.start()));
236 
237           } else {
238 
239             //leave it in, wasnt a status call
240             appendWithSpace(queryToOperateOnBuilder, queryToOperateOn.substring(currentIndex, matcher.end()));
241             
242           }
243         }
244         
245         currentIndex = matcher.end();
246       }
247       
248       appendWithSpace(queryToOperateOnBuilder, queryToOperateOn.substring(currentIndex, queryToOperateOn.length()));
249       
250       queryToOperateOn = queryToOperateOnBuilder.toString();
251       
252     }
253 
254     if (firstPass && !foundStatus && hasStatus && !StringUtils.isBlank(this.subjectStatusConfig.getStatusSearchDefault())) {
255       
256       this.queryWithDefault = this.originalQuery + " " + this.subjectStatusConfig.getStatusSearchDefault();
257       this.processOriginalQueryHelper(false);
258       return;
259       
260     }
261     
262     if (firstPass) {
263       
264       this.queryWithDefault = this.originalQuery;
265     
266     }
267     
268     this.strippedQuery = queryToOperateOn;
269 
270   }

272   /**
273    * take in a search string, and return the search status result
274    * @param searchString
275    * @return the search status result
276    */
277   public SubjectStatusResult processSearch() {
278     
279     SubjectStatusResult subjectStatusResult = new SubjectStatusResult();
280 
281     //this should only be null if things arent configured properly...
282     if (this.subjectStatusConfig == null) {
283       log.info("Why is config null???");
284       subjectStatusResult.setAll(true);
285       subjectStatusResult.setStrippedQuery(this.originalQuery);
286       return subjectStatusResult;
287     }
288         
289     this.processOriginalQuery();
290 
291     subjectStatusResult.setEquals(this.equalsFromUser);
292     subjectStatusResult.setStrippedQuery(this.strippedQuery);
293 
294     if (!StringUtils.isBlank(this.statusValueFromUser)) {
295 
296       //see if all
297       if (StringUtils.equalsIgnoreCase(this.statusValueFromUser, this.subjectStatusConfig.getStatusAllFromUser())) {
298         subjectStatusResult.setAll(true);
299       }
300       
301       subjectStatusResult.setDatastoreFieldName(this.subjectStatusConfig.getStatusDatastoreFieldName());
302      
303       String translatedStatus = this.subjectStatusConfig.getStatusTranslateUserToDatastore().get(this.statusValueFromUser);
304       
305       //default to what was typed in
306       translatedStatus = StringUtils.defaultString(translatedStatus, this.statusValueFromUser);
307       
308       subjectStatusResult.setDatastoreValue(translatedStatus);
309       
310     }
311     
312     return subjectStatusResult;
313   }
