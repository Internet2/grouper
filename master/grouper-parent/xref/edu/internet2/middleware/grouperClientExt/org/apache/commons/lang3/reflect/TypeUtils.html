69      /**
70       * <p> Checks if the subject type may be implicitly cast to the target type
71       * following the Java generics rules. If both types are {@link Class}
72       * objects, the method returns the result of
73       * {@link ClassUtils#isAssignable(Class, Class)}. </p>
74       *
75       * @param type the subject type to be assigned to the target type
76       * @param toType the target type
77       * @return <code>true</code> if <code>type</code> is assignable to <code>toType</code>.
78       */
79      public static boolean isAssignable(Type type, Type toType) {
80          return isAssignable(type, toType, null);
81      }

83      /**
84       * <p> Checks if the subject type may be implicitly cast to the target type
85       * following the Java generics rules. </p>
86       *
87       * @param type the subject type to be assigned to the target type
88       * @param toType the target type
89       * @param typeVarAssigns optional map of type variable assignments
90       * @return <code>true</code> if <code>type</code> is assignable to <code>toType</code>.
91       */
92      private static boolean isAssignable(Type type, Type toType,
93              Map<TypeVariable<?>, Type> typeVarAssigns) {
94          if (toType == null || toType instanceof Class<?>) {
95              return isAssignable(type, (Class<?>) toType);
96          }
97  
98          if (toType instanceof ParameterizedType) {
99              return isAssignable(type, (ParameterizedType) toType, typeVarAssigns);
100         }
101 
102         if (toType instanceof GenericArrayType) {
103             return isAssignable(type, (GenericArrayType) toType, typeVarAssigns);
104         }
105 
106         if (toType instanceof WildcardType) {
107             return isAssignable(type, (WildcardType) toType, typeVarAssigns);
108         }
109 
110         // *
111         if (toType instanceof TypeVariable<?>) {
112             return isAssignable(type, (TypeVariable<?>) toType, typeVarAssigns);
113         }
114         // */
115 
116         throw new IllegalStateException("found an unhandled type: " + toType);
117     }

119     /**
120      * <p> Checks if the subject type may be implicitly cast to the target class
121      * following the Java generics rules. </p>
122      *
123      * @param type the subject type to be assigned to the target type
124      * @param toClass the target class
125      * @return true if <code>type</code> is assignable to <code>toClass</code>.
126      */
127     private static boolean isAssignable(Type type, Class<?> toClass) {
128         if (type == null) {
129             // consistency with ClassUtils.isAssignable() behavior
130             return toClass == null || !toClass.isPrimitive();
131         }
132 
133         // only a null type can be assigned to null type which
134         // would have cause the previous to return true
135         if (toClass == null) {
136             return false;
137         }
138 
139         // all types are assignable to themselves
140         if (toClass.equals(type)) {
141             return true;
142         }
143 
144         if (type instanceof Class<?>) {
145             // just comparing two classes
146             return ClassUtils.isAssignable((Class<?>) type, toClass);
147         }
148 
149         if (type instanceof ParameterizedType) {
150             // only have to compare the raw type to the class
151             return isAssignable(getRawType((ParameterizedType) type), toClass);
152         }
153 
154         // *
155         if (type instanceof TypeVariable<?>) {
156             // if any of the bounds are assignable to the class, then the
157             // type is assignable to the class.
158             for (Type bound : ((TypeVariable<?>) type).getBounds()) {
159                 if (isAssignable(bound, toClass)) {
160                     return true;
161                 }
162             }
163 
164             return false;
165         }
166 
167         // the only classes to which a generic array type can be assigned
168         // are class Object and array classes
169         if (type instanceof GenericArrayType) {
170             return toClass.equals(Object.class)
171                     || toClass.isArray()
172                     && isAssignable(((GenericArrayType) type).getGenericComponentType(), toClass
173                             .getComponentType());
174         }
175 
176         // wildcard types are not assignable to a class (though one would think
177         // "? super Object" would be assignable to Object)
178         if (type instanceof WildcardType) {
179             return false;
180         }
181 
182         throw new IllegalStateException("found an unhandled type: " + type);
183     }

185     /**
186      * <p> Checks if the subject type may be implicitly cast to the target
187      * parameterized type following the Java generics rules. </p>
188      *
189      * @param type the subject type to be assigned to the target type
190      * @param toParameterizedType the target parameterized type
191      * @param typeVarAssigns a map with type variables
192      * @return true if <code>type</code> is assignable to <code>toType</code>.
193      */
194     private static boolean isAssignable(Type type, ParameterizedType toParameterizedType,
195             Map<TypeVariable<?>, Type> typeVarAssigns) {
196         if (type == null) {
197             return true;
198         }
199 
200         // only a null type can be assigned to null type which
201         // would have cause the previous to return true
202         if (toParameterizedType == null) {
203             return false;
204         }
205 
206         // all types are assignable to themselves
207         if (toParameterizedType.equals(type)) {
208             return true;
209         }
210 
211         // get the target type's raw type
212         Class<?> toClass = getRawType(toParameterizedType);
213         // get the subject type's type arguments including owner type arguments
214         // and supertype arguments up to and including the target class.
215         Map<TypeVariable<?>, Type> fromTypeVarAssigns = getTypeArguments(type, toClass, null);
216 
217         // null means the two types are not compatible
218         if (fromTypeVarAssigns == null) {
219             return false;
220         }
221 
222         // compatible types, but there's no type arguments. this is equivalent
223         // to comparing Map< ?, ? > to Map, and raw types are always assignable
224         // to parameterized types.
225         if (fromTypeVarAssigns.isEmpty()) {
226             return true;
227         }
228 
229         // get the target type's type arguments including owner type arguments
230         Map<TypeVariable<?>, Type> toTypeVarAssigns = getTypeArguments(toParameterizedType,
231                 toClass, typeVarAssigns);
232 
233         // now to check each type argument
234         for (Map.Entry<TypeVariable<?>, Type> entry : toTypeVarAssigns.entrySet()) {
235             Type toTypeArg = entry.getValue();
236             Type fromTypeArg = fromTypeVarAssigns.get(entry.getKey());
237 
238             // parameters must either be absent from the subject type, within
239             // the bounds of the wildcard type, or be an exact match to the
240             // parameters of the target type.
241             if (fromTypeArg != null
242                     && !toTypeArg.equals(fromTypeArg)
243                     && !(toTypeArg instanceof WildcardType && isAssignable(fromTypeArg, toTypeArg,
244                             typeVarAssigns))) {
245                 return false;
246             }
247         }
248 
249         return true;
250     }

252     /**
253      * <p> Checks if the subject type may be implicitly cast to the target
254      * generic array type following the Java generics rules. </p>
255      *
256      * @param type the subject type to be assigned to the target type
257      * @param toGenericArrayType the target generic array type
258      * @param typeVarAssigns a map with type variables
259      * @return true if <code>type</code> is assignable to
260      * <code>toGenericArrayType</code>.
261      */
262     private static boolean isAssignable(Type type, GenericArrayType toGenericArrayType,
263             Map<TypeVariable<?>, Type> typeVarAssigns) {
264         if (type == null) {
265             return true;
266         }
267 
268         // only a null type can be assigned to null type which
269         // would have cause the previous to return true
270         if (toGenericArrayType == null) {
271             return false;
272         }
273 
274         // all types are assignable to themselves
275         if (toGenericArrayType.equals(type)) {
276             return true;
277         }
278 
279         Type toComponentType = toGenericArrayType.getGenericComponentType();
280 
281         if (type instanceof Class<?>) {
282             Class<?> cls = (Class<?>) type;
283 
284             // compare the component types
285             return cls.isArray()
286                     && isAssignable(cls.getComponentType(), toComponentType, typeVarAssigns);
287         }
288 
289         if (type instanceof GenericArrayType) {
290             // compare the component types
291             return isAssignable(((GenericArrayType) type).getGenericComponentType(),
292                     toComponentType, typeVarAssigns);
293         }
294 
295         if (type instanceof WildcardType) {
296             // so long as one of the upper bounds is assignable, it's good
297             for (Type bound : getImplicitUpperBounds((WildcardType) type)) {
298                 if (isAssignable(bound, toGenericArrayType)) {
299                     return true;
300                 }
301             }
302 
303             return false;
304         }
305 
306         if (type instanceof TypeVariable<?>) {
307             // probably should remove the following logic and just return false.
308             // type variables cannot specify arrays as bounds.
309             for (Type bound : getImplicitBounds((TypeVariable<?>) type)) {
310                 if (isAssignable(bound, toGenericArrayType)) {
311                     return true;
312                 }
313             }
314 
315             return false;
316         }
317 
318         if (type instanceof ParameterizedType) {
319             // the raw type of a parameterized type is never an array or
320             // generic array, otherwise the declaration would look like this:
321             // Collection[]< ? extends String > collection;
322             return false;
323         }
324 
325         throw new IllegalStateException("found an unhandled type: " + type);
326     }

328     /**
329      * <p> Checks if the subject type may be implicitly cast to the target
330      * wildcard type following the Java generics rules. </p>
331      *
332      * @param type the subject type to be assigned to the target type
333      * @param toWildcardType the target wildcard type
334      * @param typeVarAssigns a map with type variables
335      * @return true if <code>type</code> is assignable to
336      * <code>toWildcardType</code>.
337      */
338     private static boolean isAssignable(Type type, WildcardType toWildcardType,
339             Map<TypeVariable<?>, Type> typeVarAssigns) {
340         if (type == null) {
341             return true;
342         }
343 
344         // only a null type can be assigned to null type which
345         // would have cause the previous to return true
346         if (toWildcardType == null) {
347             return false;
348         }
349 
350         // all types are assignable to themselves
351         if (toWildcardType.equals(type)) {
352             return true;
353         }
354 
355         Type[] toUpperBounds = getImplicitUpperBounds(toWildcardType);
356         Type[] toLowerBounds = getImplicitLowerBounds(toWildcardType);
357 
358         if (type instanceof WildcardType) {
359             WildcardType wildcardType = (WildcardType) type;
360             Type[] upperBounds = getImplicitUpperBounds(wildcardType);
361             Type[] lowerBounds = getImplicitLowerBounds(wildcardType);
362 
363             for (Type toBound : toUpperBounds) {
364                 // if there are assignments for unresolved type variables,
365                 // now's the time to substitute them.
366                 toBound = substituteTypeVariables(toBound, typeVarAssigns);
367 
368                 // each upper bound of the subject type has to be assignable to
369                 // each
370                 // upper bound of the target type
371                 for (Type bound : upperBounds) {
372                     if (!isAssignable(bound, toBound, typeVarAssigns)) {
373                         return false;
374                     }
375                 }
376             }
377 
378             for (Type toBound : toLowerBounds) {
379                 // if there are assignments for unresolved type variables,
380                 // now's the time to substitute them.
381                 toBound = substituteTypeVariables(toBound, typeVarAssigns);
382 
383                 // each lower bound of the target type has to be assignable to
384                 // each
385                 // lower bound of the subject type
386                 for (Type bound : lowerBounds) {
387                     if (!isAssignable(toBound, bound, typeVarAssigns)) {
388                         return false;
389                     }
390                 }
391             }
392 
393             return true;
394         }
395 
396         for (Type toBound : toUpperBounds) {
397             // if there are assignments for unresolved type variables,
398             // now's the time to substitute them.
399             if (!isAssignable(type, substituteTypeVariables(toBound, typeVarAssigns),
400                     typeVarAssigns)) {
401                 return false;
402             }
403         }
404 
405         for (Type toBound : toLowerBounds) {
406             // if there are assignments for unresolved type variables,
407             // now's the time to substitute them.
408             if (!isAssignable(substituteTypeVariables(toBound, typeVarAssigns), type,
409                     typeVarAssigns)) {
410                 return false;
411             }
412         }
413 
414         return true;
415     }

417     /**
418      * <p> Checks if the subject type may be implicitly cast to the target type
419      * variable following the Java generics rules. </p>
420      *
421      * @param type the subject type to be assigned to the target type
422      * @param toTypeVariable the target type variable
423      * @param typeVarAssigns a map with type variables
424      * @return true if <code>type</code> is assignable to
425      * <code>toTypeVariable</code>.
426      */
427     private static boolean isAssignable(Type type, TypeVariable<?> toTypeVariable,
428             Map<TypeVariable<?>, Type> typeVarAssigns) {
429         if (type == null) {
430             return true;
431         }
432 
433         // only a null type can be assigned to null type which
434         // would have cause the previous to return true
435         if (toTypeVariable == null) {
436             return false;
437         }
438 
439         // all types are assignable to themselves
440         if (toTypeVariable.equals(type)) {
441             return true;
442         }
443 
444         if (type instanceof TypeVariable<?>) {
445             // a type variable is assignable to another type variable, if
446             // and only if the former is the latter, extends the latter, or
447             // is otherwise a descendant of the latter.
448             Type[] bounds = getImplicitBounds((TypeVariable<?>) type);
449 
450             for (Type bound : bounds) {
451                 if (isAssignable(bound, toTypeVariable, typeVarAssigns)) {
452                     return true;
453                 }
454             }
455         }
456 
457         if (type instanceof Class<?> || type instanceof ParameterizedType
458                 || type instanceof GenericArrayType || type instanceof WildcardType) {
459             return false;
460         }
461 
462         throw new IllegalStateException("found an unhandled type: " + type);
463     }

465     /**
466      * <p> </p>
467      *
468      * @param type the type to be replaced
469      * @param typeVarAssigns the map with type variables
470      * @return the replaced type
471      * @throws IllegalArgumentException if the type cannot be substituted
472      */
473     private static Type substituteTypeVariables(Type type, Map<TypeVariable<?>, Type> typeVarAssigns) {
474         if (type instanceof TypeVariable<?> && typeVarAssigns != null) {
475             Type replacementType = typeVarAssigns.get(type);
476 
477             if (replacementType == null) {
478                 throw new IllegalArgumentException("missing assignment type for type variable "
479                         + type);
480             }
481 
482             return replacementType;
483         }
484 
485         return type;
486     }

503     /**
504      * <p> Gets the type arguments of a class/interface based on a subtype. For
505      * instance, this method will determine that both of the parameters for the
506      * interface {@link Map} are {@link Object} for the subtype
507      * {@link java.util.Properties Properties} even though the subtype does not
508      * directly implement the <code>Map</code> interface. <p> </p> This method
509      * returns <code>null</code> if <code>type</code> is not assignable to
510      * <code>toClass</code>. It returns an empty map if none of the classes or
511      * interfaces in its inheritance hierarchy specify any type arguments. </p>
512      * <p> A side-effect of this method is that it also retrieves the type
513      * arguments for the classes and interfaces that are part of the hierarchy
514      * between <code>type</code> and <code>toClass</code>. So with the above
515      * example, this method will also determine that the type arguments for
516      * {@link java.util.Hashtable Hashtable} are also both <code>Object</code>.
517      * In cases where the interface specified by <code>toClass</code> is
518      * (indirectly) implemented more than once (e.g. where <code>toClass</code>
519      * specifies the interface {@link java.lang.Iterable Iterable} and
520      * <code>type</code> specifies a parameterized type that implements both
521      * {@link java.util.Set Set} and {@link java.util.Collection Collection}),
522      * this method will look at the inheritance hierarchy of only one of the
523      * implementations/subclasses; the first interface encountered that isn't a
524      * subinterface to one of the others in the <code>type</code> to
525      * <code>toClass</code> hierarchy. </p>
526      *
527      * @param type the type from which to determine the type parameters of
528      * <code>toClass</code>
529      * @param toClass the class whose type parameters are to be determined based
530      * on the subtype <code>type</code>
531      * @return a map of the type assignments for the type variables in each type
532      * in the inheritance hierarchy from <code>type</code> to
533      * <code>toClass</code> inclusive.
534      */
535     public static Map<TypeVariable<?>, Type> getTypeArguments(Type type, Class<?> toClass) {
536         return getTypeArguments(type, toClass, null);
537     }

539     /**
540      * <p> Return a map of the type arguments of <code>type</code> in the context of <code>toClass</code>. </p>
541      *
542      * @param type the type in question
543      * @param toClass the class
544      * @param subtypeVarAssigns a map with type variables
545      * @return the map with type arguments
546      */
547     private static Map<TypeVariable<?>, Type> getTypeArguments(Type type, Class<?> toClass,
548             Map<TypeVariable<?>, Type> subtypeVarAssigns) {
549         if (type instanceof Class<?>) {
550             return getTypeArguments((Class<?>) type, toClass, subtypeVarAssigns);
551         }
552 
553         if (type instanceof ParameterizedType) {
554             return getTypeArguments((ParameterizedType) type, toClass, subtypeVarAssigns);
555         }
556 
557         if (type instanceof GenericArrayType) {
558             return getTypeArguments(((GenericArrayType) type).getGenericComponentType(), toClass
559                     .isArray() ? toClass.getComponentType() : toClass, subtypeVarAssigns);
560         }
561 
562         // since wildcard types are not assignable to classes, should this just
563         // return null?
564         if (type instanceof WildcardType) {
565             for (Type bound : getImplicitUpperBounds((WildcardType) type)) {
566                 // find the first bound that is assignable to the target class
567                 if (isAssignable(bound, toClass)) {
568                     return getTypeArguments(bound, toClass, subtypeVarAssigns);
569                 }
570             }
571 
572             return null;
573         }
574 
575         // *
576         if (type instanceof TypeVariable<?>) {
577             for (Type bound : getImplicitBounds((TypeVariable<?>) type)) {
578                 // find the first bound that is assignable to the target class
579                 if (isAssignable(bound, toClass)) {
580                     return getTypeArguments(bound, toClass, subtypeVarAssigns);
581                 }
582             }
583 
584             return null;
585         }
586         // */
587 
588         throw new IllegalStateException("found an unhandled type: " + type);
589     }

591     /**
592      * <p> Return a map of the type arguments of a parameterized type in the context of <code>toClass</code>. </p>
593      *
594      * @param parameterizedType the parameterized type
595      * @param toClass the class
596      * @param subtypeVarAssigns a map with type variables
597      * @return the map with type arguments
598      */
599     private static Map<TypeVariable<?>, Type> getTypeArguments(
600             ParameterizedType parameterizedType, Class<?> toClass,
601             Map<TypeVariable<?>, Type> subtypeVarAssigns) {
602         Class<?> cls = getRawType(parameterizedType);
603 
604         // make sure they're assignable
605         if (!isAssignable(cls, toClass)) {
606             return null;
607         }
608 
609         Type ownerType = parameterizedType.getOwnerType();
610         Map<TypeVariable<?>, Type> typeVarAssigns;
611 
612         if (ownerType instanceof ParameterizedType) {
613             // get the owner type arguments first
614             ParameterizedType parameterizedOwnerType = (ParameterizedType) ownerType;
615             typeVarAssigns = getTypeArguments(parameterizedOwnerType,
616                     getRawType(parameterizedOwnerType), subtypeVarAssigns);
617         } else {
618             // no owner, prep the type variable assignments map
619             typeVarAssigns = subtypeVarAssigns == null ? new HashMap<TypeVariable<?>, Type>()
620                     : new HashMap<TypeVariable<?>, Type>(subtypeVarAssigns);
621         }
622 
623         // get the subject parameterized type's arguments
624         Type[] typeArgs = parameterizedType.getActualTypeArguments();
625         // and get the corresponding type variables from the raw class
626         TypeVariable<?>[] typeParams = cls.getTypeParameters();
627 
628         // map the arguments to their respective type variables
629         for (int i = 0; i < typeParams.length; i++) {
630             Type typeArg = typeArgs[i];
631             typeVarAssigns.put(typeParams[i], typeVarAssigns.containsKey(typeArg) ? typeVarAssigns
632                     .get(typeArg) : typeArg);
633         }
634 
635         if (toClass.equals(cls)) {
636             // target class has been reached. Done.
637             return typeVarAssigns;
638         }
639 
640         // walk the inheritance hierarchy until the target class is reached
641         return getTypeArguments(getClosestParentType(cls, toClass), toClass, typeVarAssigns);
642     }

644     /**
645      * <p> Return a map of the type arguments of a class in the context of <code>toClass</code>. </p>
646      *
647      * @param cls the class in question
648      * @param toClass the context class
649      * @param subtypeVarAssigns a map with type variables
650      * @return the map with type arguments
651      */
652     private static Map<TypeVariable<?>, Type> getTypeArguments(Class<?> cls, Class<?> toClass,
653             Map<TypeVariable<?>, Type> subtypeVarAssigns) {
654         // make sure they're assignable
655         if (!isAssignable(cls, toClass)) {
656             return null;
657         }
658 
659         // can't work with primitives
660         if (cls.isPrimitive()) {
661             // both classes are primitives?
662             if (toClass.isPrimitive()) {
663                 // dealing with widening here. No type arguments to be
664                 // harvested with these two types.
665                 return new HashMap<TypeVariable<?>, Type>();
666             }
667 
668             // work with wrapper the wrapper class instead of the primitive
669             cls = ClassUtils.primitiveToWrapper(cls);
670         }
671 
672         // create a copy of the incoming map, or an empty one if it's null
673         HashMap<TypeVariable<?>, Type> typeVarAssigns = subtypeVarAssigns == null ? new HashMap<TypeVariable<?>, Type>()
674                 : new HashMap<TypeVariable<?>, Type>(subtypeVarAssigns);
675 
676         // no arguments for the parameters, or target class has been reached
677         if (cls.getTypeParameters().length > 0 || toClass.equals(cls)) {
678             return typeVarAssigns;
679         }
680 
681         // walk the inheritance hierarchy until the target class is reached
682         return getTypeArguments(getClosestParentType(cls, toClass), toClass, typeVarAssigns);
683     }

685     /**
686      * <p> Tries to determine the type arguments of a class/interface based on a
687      * super parameterized type's type arguments. This method is the inverse of
688      * {@link #getTypeArguments(Type, Class)} which gets a class/interface's
689      * type arguments based on a subtype. It is far more limited in determining
690      * the type arguments for the subject class's type variables in that it can
691      * only determine those parameters that map from the subject {@link Class}
692      * object to the supertype. </p> <p> Example: {@link java.util.TreeSet
693      * TreeSet} sets its parameter as the parameter for
694      * {@link java.util.NavigableSet NavigableSet}, which in turn sets the
695      * parameter of {@link java.util.SortedSet}, which in turn sets the
696      * parameter of {@link Set}, which in turn sets the parameter of
697      * {@link java.util.Collection}, which in turn sets the parameter of
698      * {@link java.lang.Iterable}. Since <code>TreeSet</code>'s parameter maps
699      * (indirectly) to <code>Iterable</code>'s parameter, it will be able to
700      * determine that based on the super type <code>Iterable<? extends
701      * Map<Integer,? extends Collection<?>>></code>, the parameter of
702      * <code>TreeSet</code> is <code>? extends Map<Integer,? extends
703      * Collection<?>></code>. </p>
704      *
705      * @param cls the class whose type parameters are to be determined
706      * @param superType the super type from which <code>cls</code>'s type
707      * arguments are to be determined
708      * @return a map of the type assignments that could be determined for the
709      * type variables in each type in the inheritance hierarchy from
710      * <code>type</code> to <code>toClass</code> inclusive.
711      */
712     public static Map<TypeVariable<?>, Type> determineTypeArguments(Class<?> cls,
713             ParameterizedType superType) {
714         Class<?> superClass = getRawType(superType);
715 
716         // compatibility check
717         if (!isAssignable(cls, superClass)) {
718             return null;
719         }
720 
721         if (cls.equals(superClass)) {
722             return getTypeArguments(superType, superClass, null);
723         }
724 
725         // get the next class in the inheritance hierarchy
726         Type midType = getClosestParentType(cls, superClass);
727 
728         // can only be a class or a parameterized type
729         if (midType instanceof Class<?>) {
730             return determineTypeArguments((Class<?>) midType, superType);
731         }
732 
733         ParameterizedType midParameterizedType = (ParameterizedType) midType;
734         Class<?> midClass = getRawType(midParameterizedType);
735         // get the type variables of the mid class that map to the type
736         // arguments of the super class
737         Map<TypeVariable<?>, Type> typeVarAssigns = determineTypeArguments(midClass, superType);
738         // map the arguments of the mid type to the class type variables
739         mapTypeVariablesToArguments(cls, midParameterizedType, typeVarAssigns);
740 
741         return typeVarAssigns;
742     }

744     /**
745      * <p>Performs a mapping of type variables.</p>
746      *
747      * @param <T> the generic type of the class in question
748      * @param cls the class in question
749      * @param parameterizedType the parameterized type
750      * @param typeVarAssigns the map to be filled
751      */
752     private static <T> void mapTypeVariablesToArguments(Class<T> cls,
753             ParameterizedType parameterizedType, Map<TypeVariable<?>, Type> typeVarAssigns) {
754         // capture the type variables from the owner type that have assignments
755         Type ownerType = parameterizedType.getOwnerType();
756 
757         if (ownerType instanceof ParameterizedType) {
758             // recursion to make sure the owner's owner type gets processed
759             mapTypeVariablesToArguments(cls, (ParameterizedType) ownerType, typeVarAssigns);
760         }
761 
762         // parameterizedType is a generic interface/class (or it's in the owner
763         // hierarchy of said interface/class) implemented/extended by the class
764         // cls. Find out which type variables of cls are type arguments of
765         // parameterizedType:
766         Type[] typeArgs = parameterizedType.getActualTypeArguments();
767 
768         // of the cls's type variables that are arguments of parameterizedType,
769         // find out which ones can be determined from the super type's arguments
770         TypeVariable<?>[] typeVars = getRawType(parameterizedType).getTypeParameters();
771 
772         // use List view of type parameters of cls so the contains() method can be used:
773         List<TypeVariable<Class<T>>> typeVarList = Arrays.asList(cls
774                 .getTypeParameters());
775 
776         for (int i = 0; i < typeArgs.length; i++) {
777             TypeVariable<?> typeVar = typeVars[i];
778             Type typeArg = typeArgs[i];
779 
780             // argument of parameterizedType is a type variable of cls
781             if (typeVarList.contains(typeArg)
782             // type variable of parameterizedType has an assignment in
783                     // the super type.
784                     && typeVarAssigns.containsKey(typeVar)) {
785                 // map the assignment to the cls's type variable
786                 typeVarAssigns.put((TypeVariable<?>) typeArg, typeVarAssigns.get(typeVar));
787             }
788         }
789     }

791     /**
792      * <p> Closest parent type? Closest to what? The closest parent type to the
793      * super class specified by <code>superClass</code>. </p>
794      *
795      * @param cls the class in question
796      * @param superClass the super class
797      * @return the closes parent type
798      */
799     private static Type getClosestParentType(Class<?> cls, Class<?> superClass) {
800         // only look at the interfaces if the super class is also an interface
801         if (superClass.isInterface()) {
802             // get the generic interfaces of the subject class
803             Type[] interfaceTypes = cls.getGenericInterfaces();
804             // will hold the best generic interface match found
805             Type genericInterface = null;
806 
807             // find the interface closest to the super class
808             for (Type midType : interfaceTypes) {
809                 Class<?> midClass = null;
810 
811                 if (midType instanceof ParameterizedType) {
812                     midClass = getRawType((ParameterizedType) midType);
813                 } else if (midType instanceof Class<?>) {
814                     midClass = (Class<?>) midType;
815                 } else {
816                     throw new IllegalStateException("Unexpected generic"
817                             + " interface type found: " + midType);
818                 }
819 
820                 // check if this interface is further up the inheritance chain
821                 // than the previously found match
822                 if (isAssignable(midClass, superClass)
823                         && isAssignable(genericInterface, (Type) midClass)) {
824                     genericInterface = midType;
825                 }
826             }
827 
828             // found a match?
829             if (genericInterface != null) {
830                 return genericInterface;
831             }
832         }
833 
834         // none of the interfaces were descendants of the target class, so the
835         // super class has to be one, instead
836         return cls.getGenericSuperclass();
837     }

840      * <p> Checks if the given value can be assigned to the target type
841      * following the Java generics rules. </p>
842      *
843      * @param value the value to be checked
844      * @param type the target type
845      * @return true of <code>value</code> is an instance of <code>type</code>.
846      */
847     public static boolean isInstance(Object value, Type type) {
848         if (type == null) {
849             return false;
850         }
851 
852         return value == null ? !(type instanceof Class<?>) || !((Class<?>) type).isPrimitive()
853                 : isAssignable(value.getClass(), type, null);
854     }

856     /**
857      * <p> This method strips out the redundant upper bound types in type
858      * variable types and wildcard types (or it would with wildcard types if
859      * multiple upper bounds were allowed). </p> <p> Example: with the variable
860      * type declaration:
861      *
862      * <pre> &lt;K extends java.util.Collection&lt;String&gt; &amp;
863      * java.util.List&lt;String&gt;&gt; </pre>
864      *
865      * since <code>List</code> is a subinterface of <code>Collection</code>,
866      * this method will return the bounds as if the declaration had been:
867      *
868      * <pre> &lt;K extends java.util.List&lt;String&gt;&gt; </pre>
869      *
870      * </p>
871      *
872      * @param bounds an array of types representing the upper bounds of either
873      * <code>WildcardType</code> or <code>TypeVariable</code>.
874      * @return an array containing the values from <code>bounds</code> minus the
875      * redundant types.
876      */
877     public static Type[] normalizeUpperBounds(Type[] bounds) {
878         // don't bother if there's only one (or none) type
879         if (bounds.length < 2) {
880             return bounds;
881         }
882 
883         Set<Type> types = new HashSet<Type>(bounds.length);
884 
885         for (Type type1 : bounds) {
886             boolean subtypeFound = false;
887 
888             for (Type type2 : bounds) {
889                 if (type1 != type2 && isAssignable(type2, type1, null)) {
890                     subtypeFound = true;
891                     break;
892                 }
893             }
894 
895             if (!subtypeFound) {
896                 types.add(type1);
897             }
898         }
899 
900         return types.toArray(new Type[types.size()]);
901     }

949     /**
950      * <p> Determines whether or not specified types satisfy the bounds of their
951      * mapped type variables. When a type parameter extends another (such as
952      * <code><T, S extends T></code>), uses another as a type parameter (such as
953      * <code><T, S extends Comparable<T></code>), or otherwise depends on
954      * another type variable to be specified, the dependencies must be included
955      * in <code>typeVarAssigns</code>. </p>
956      *
957      * @param typeVarAssigns specifies the potential types to be assigned to the
958      * type variables.
959      * @return whether or not the types can be assigned to their respective type
960      * variables.
961      */
962     public static boolean typesSatisfyVariables(Map<TypeVariable<?>, Type> typeVarAssigns) {
963         // all types must be assignable to all the bounds of the their mapped
964         // type variable.
965         for (Map.Entry<TypeVariable<?>, Type> entry : typeVarAssigns.entrySet()) {
966             TypeVariable<?> typeVar = entry.getKey();
967             Type type = entry.getValue();
968 
969             for (Type bound : getImplicitBounds(typeVar)) {
970                 if (!isAssignable(type, substituteTypeVariables(bound, typeVarAssigns),
971                         typeVarAssigns)) {
972                     return false;
973                 }
974             }
975         }
976 
977         return true;
978     }

981      * <p> Transforms the passed in type to a {@code Class} object. Type-checking method of convenience. </p>
982      *
983      * @param parameterizedType the type to be converted
984      * @return the corresponding {@code Class} object
985      * @throws IllegalStateException if the conversion fails
986      */
987     private static Class<?> getRawType(ParameterizedType parameterizedType) {
988         Type rawType = parameterizedType.getRawType();
989 
990         // check if raw type is a Class object
991         // not currently necessary, but since the return type is Type instead of
992         // Class, there's enough reason to believe that future versions of Java
993         // may return other Type implementations. And type-safety checking is
994         // rarely a bad idea.
995         if (!(rawType instanceof Class<?>)) {
996             throw new IllegalStateException("Wait... What!? Type of rawType: " + rawType);
997         }
998 
999         return (Class<?>) rawType;
1000     }

1002     /**
1003      * <p> Get the raw type of a Java type, given its context. Primarily for use
1004      * with {@link TypeVariable}s and {@link GenericArrayType}s, or when you do
1005      * not know the runtime type of <code>type</code>: if you know you have a
1006      * {@link Class} instance, it is already raw; if you know you have a
1007      * {@link ParameterizedType}, its raw type is only a method call away. </p>
1008      *
1009      * @param type to resolve
1010      * @param assigningType type to be resolved against
1011      * @return the resolved <code>Class</code> object or <code>null</code> if
1012      * the type could not be resolved
1013      */
1014     public static Class<?> getRawType(Type type, Type assigningType) {
1015         if (type instanceof Class<?>) {
1016             // it is raw, no problem
1017             return (Class<?>) type;
1018         }
1019 
1020         if (type instanceof ParameterizedType) {
1021             // simple enough to get the raw type of a ParameterizedType
1022             return getRawType((ParameterizedType) type);
1023         }
1024 
1025         if (type instanceof TypeVariable<?>) {
1026             if (assigningType == null) {
1027                 return null;
1028             }
1029 
1030             // get the entity declaring this type variable
1031             Object genericDeclaration = ((TypeVariable<?>) type).getGenericDeclaration();
1032 
1033             // can't get the raw type of a method- or constructor-declared type
1034             // variable
1035             if (!(genericDeclaration instanceof Class<?>)) {
1036                 return null;
1037             }
1038 
1039             // get the type arguments for the declaring class/interface based
1040             // on the enclosing type
1041             Map<TypeVariable<?>, Type> typeVarAssigns = getTypeArguments(assigningType,
1042                     (Class<?>) genericDeclaration);
1043 
1044             // enclosingType has to be a subclass (or subinterface) of the
1045             // declaring type
1046             if (typeVarAssigns == null) {
1047                 return null;
1048             }
1049 
1050             // get the argument assigned to this type variable
1051             Type typeArgument = typeVarAssigns.get(type);
1052 
1053             if (typeArgument == null) {
1054                 return null;
1055             }
1056 
1057             // get the argument for this type variable
1058             return getRawType(typeArgument, assigningType);
1059         }
1060 
1061         if (type instanceof GenericArrayType) {
1062             // get raw component type
1063             Class<?> rawComponentType = getRawType(((GenericArrayType) type)
1064                     .getGenericComponentType(), assigningType);
1065 
1066             // create array type from raw component type and return its class
1067             return Array.newInstance(rawComponentType, 0).getClass();
1068         }
1069 
1070         // (hand-waving) this is not the method you're looking for
1071         if (type instanceof WildcardType) {
1072             return null;
1073         }
1074 
1075         throw new IllegalArgumentException("unknown type: " + type);
1076     }
