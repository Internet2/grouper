236     /**
237      * Interpret the given script/expression.
238      * <p>
239      * If the underlying JEXL engine is silent, errors will be logged through its logger as info.
240      * </p>
241      * @param node the script or expression to interpret.
242      * @return the result of the interpretation.
243      * @throws JexlException if any error occurs during interpretation.
244      */
245     public Object interpret(JexlNode node) {
246         try {
247             return node.jjtAccept(this, null);
248         } catch (JexlException.Return xreturn) {
249             Object value = xreturn.getValue();
250             return value;
251         } catch (JexlException xjexl) {
252             if (silent) {
253                 logger.warn(xjexl.getMessage(), xjexl.getCause());
254                 return null;
255             }
256             throw xjexl;
257         } finally {
258             functors = null;
259             parameters = null;
260             registers = null;
261         }
262     }

314      * Finds the node causing a NPE for diadic operators.
315      * @param xrt the RuntimeException
316      * @param node the parent node
317      * @param left the left argument
318      * @param right the right argument
319      * @return the left, right or parent node
320      */
321     protected JexlNode findNullOperand(RuntimeException xrt, JexlNode node, Object left, Object right) {
322         if (xrt instanceof ArithmeticException
323                 && JexlException.NULL_OPERAND == xrt.getMessage()) {
324             if (left == null) {
325                 return node.jjtGetChild(0);
326             }
327             if (right == null) {
328                 return node.jjtGetChild(1);
329             }
330         }
331         return node;
332     }

376     /**
377      * Resolves a namespace, eventually allocating an instance using context as constructor argument.
378      * The lifetime of such instances span the current expression or script evaluation.
379      *
380      * @param prefix the prefix name (may be null for global namespace)
381      * @param node the AST node
382      * @return the namespace instance
383      */
384     protected Object resolveNamespace(String prefix, JexlNode node) {
385         Object namespace = null;
386         // check whether this namespace is a functor
387         if (functors != null) {
388             namespace = functors.get(prefix);
389             if (namespace != null) {
390                 return namespace;
391             }
392         }
393         // check if namespace if a resolver
394         if (context instanceof NamespaceResolver) {
395             namespace = ((NamespaceResolver) context).resolveNamespace(prefix);
396         }
397         if (namespace == null) {
398             namespace = functions.get(prefix);
399             if (prefix != null && namespace == null) {
400                 throw new JexlException(node, "no such function namespace " + prefix);
401             }
402         }
403         // allow namespace to be instantiated as functor with context if possible, not an error otherwise
404         if (namespace instanceof Class<?>) {
405             Object[] args = new Object[]{context};
406             JexlMethod ctor = uberspect.getConstructorMethod(namespace, args, node);
407             if (ctor != null) {
408                 try {
409                     namespace = ctor.invoke(namespace, args);
410                     if (functors == null) {
411                         functors = new HashMap<String, Object>();
412                     }
413                     functors.put(prefix, namespace);
414                 } catch (Exception xinst) {
415                     throw new JexlException(node, "unable to instantiate namespace " + prefix, xinst);
416                 }
417             }
418         }
419         return namespace;
420     }

422     /** {@inheritDoc} */
423     public Object visit(ASTAdditiveNode node, Object data) {
424         /**
425          * The pattern for exception mgmt is to let the child*.jjtAccept
426          * out of the try/catch loop so that if one fails, the ex will
427          * traverse up to the interpreter.
428          * In cases where this is not convenient/possible, JexlException must
429          * be caught explicitly and rethrown.
430          */
431         Object left = node.jjtGetChild(0).jjtAccept(this, data);
432         for (int c = 2, size = node.jjtGetNumChildren(); c < size; c += 2) {
433             Object right = node.jjtGetChild(c).jjtAccept(this, data);
434             try {
435                 JexlNode op = node.jjtGetChild(c - 1);
436                 if (op instanceof ASTAdditiveOperator) {
437                     String which = op.image;
438                     if ("+".equals(which)) {
439                         left = arithmetic.add(left, right);
440                         continue;
441                     }
442                     if ("-".equals(which)) {
443                         left = arithmetic.subtract(left, right);
444                         continue;
445                     }
446                     throw new UnsupportedOperationException("unknown operator " + which);
447                 }
448                 throw new IllegalArgumentException("unknown operator " + op);
449             } catch (ArithmeticException xrt) {
450                 JexlNode xnode = findNullOperand(xrt, node, left, right);
451                 throw new JexlException(xnode, "+/- error", xrt);
452             }
453         }
454         return left;
455     }

462     /** {@inheritDoc} */
463     public Object visit(ASTAndNode node, Object data) {
464         Object left = node.jjtGetChild(0).jjtAccept(this, data);
465         try {
466             boolean leftValue = arithmetic.toBoolean(left);
467             if (!leftValue) {
468                 return Boolean.FALSE;
469             }
470         } catch (RuntimeException xrt) {
471             throw new JexlException(node.jjtGetChild(0), "boolean coercion error", xrt);
472         }
473         Object right = node.jjtGetChild(1).jjtAccept(this, data);
474         try {
475             boolean rightValue = arithmetic.toBoolean(right);
476             if (!rightValue) {
477                 return Boolean.FALSE;
478             }
479         } catch (ArithmeticException xrt) {
480             throw new JexlException(node.jjtGetChild(1), "boolean coercion error", xrt);
481         }
482         return Boolean.TRUE;
483     }

485     /** {@inheritDoc} */
486     public Object visit(ASTArrayAccess node, Object data) {
487         // first objectNode is the identifier
488         Object object = node.jjtGetChild(0).jjtAccept(this, data);
489         // can have multiple nodes - either an expression, integer literal or reference
490         int numChildren = node.jjtGetNumChildren();
491         for (int i = 1; i < numChildren; i++) {
492             JexlNode nindex = node.jjtGetChild(i);
493             if (nindex instanceof JexlNode.Literal<?>) {
494                 object = nindex.jjtAccept(this, object);
495             } else {
496                 Object index = nindex.jjtAccept(this, null);
497                 object = getAttribute(object, index, nindex);
498             }
499         }
500 
501         return object;
502     }

504     /** {@inheritDoc} */
505     public Object visit(ASTArrayLiteral node, Object data) {
506         Object literal = node.getLiteral();
507         if (literal == null) {
508             int childCount = node.jjtGetNumChildren();
509             Object[] array = new Object[childCount];
510             for (int i = 0; i < childCount; i++) {
511                 Object entry = node.jjtGetChild(i).jjtAccept(this, data);
512                 array[i] = entry;
513             }
514             literal = arithmetic.narrowArrayType(array);
515             node.setLiteral(literal);
516         }
517         return literal;
518     }

520     /** {@inheritDoc} */
521     public Object visit(ASTAssignment node, Object data) {
522         // left contains the reference to assign to
523         int register = -1;
524         JexlNode left = node.jjtGetChild(0);
525         if (left instanceof ASTIdentifier) {
526             ASTIdentifier var = (ASTIdentifier) left;
527             register = var.getRegister();
528             if (register < 0) {
529                 throw new JexlException(left, "unknown variable " + left.image);
530             }
531         } else if (!(left instanceof ASTReference)) {
532             throw new JexlException(left, "illegal assignment form 0");
533         }
534         // right is the value expression to assign
535         Object right = node.jjtGetChild(1).jjtAccept(this, data);
536 
537         // determine initial object & property:
538         JexlNode objectNode = null;
539         Object object = register >= 0 ? registers[register] : null;
540         JexlNode propertyNode = null;
541         Object property = null;
542         boolean isVariable = true;
543         int v = 0;
544         StringBuilder variableName = null;
545         // 1: follow children till penultimate, resolve dot/array
546         int last = left.jjtGetNumChildren() - 1;
547         // check we are not assigning a register itself
548         boolean isRegister = last < 0 && register >= 0;
549         // start at 1 if register
550         for (int c = register >= 0 ? 1 : 0; c < last; ++c) {
551             objectNode = left.jjtGetChild(c);
552             // evaluate the property within the object
553             object = objectNode.jjtAccept(this, object);
554             if (object != null) {
555                 continue;
556             }
557             isVariable &= objectNode instanceof ASTIdentifier
558                     || (objectNode instanceof ASTNumberLiteral && ((ASTNumberLiteral) objectNode).isInteger());
559             // if we get null back as a result, check for an ant variable
560             if (isVariable) {
561                 if (v == 0) {
562                     variableName = new StringBuilder(left.jjtGetChild(0).image);
563                     v = 1;
564                 }
565                 for (; v <= c; ++v) {
566                     variableName.append('.');
567                     variableName.append(left.jjtGetChild(v).image);
568                 }
569                 object = context.get(variableName.toString());
570                 // disallow mixing ant & bean with same root; avoid ambiguity
571                 if (object != null) {
572                     isVariable = false;
573                 }
574             } else {
575                 throw new JexlException(objectNode, "illegal assignment form");
576             }
577         }
578         // 2: last objectNode will perform assignement in all cases
579         propertyNode = isRegister ? null : left.jjtGetChild(last);
580         boolean antVar = false;
581         if (propertyNode instanceof ASTIdentifier) {
582             ASTIdentifier identifier = (ASTIdentifier) propertyNode;
583             register = identifier.getRegister();
584             if (register >= 0) {
585                 isRegister = true;
586             } else {
587                 property = identifier.image;
588                 antVar = true;
589             }
590         } else if (propertyNode instanceof ASTNumberLiteral && ((ASTNumberLiteral) propertyNode).isInteger()) {
591             property = ((ASTNumberLiteral) propertyNode).getLiteral();
592             antVar = true;
593         } else if (propertyNode instanceof ASTArrayAccess) {
594             // first objectNode is the identifier
595             objectNode = propertyNode;
596             ASTArrayAccess narray = (ASTArrayAccess) objectNode;
597             Object nobject = narray.jjtGetChild(0).jjtAccept(this, object);
598             if (nobject == null) {
599                 throw new JexlException(objectNode, "array element is null");
600             } else {
601                 object = nobject;
602             }
603             // can have multiple nodes - either an expression, integer literal or
604             // reference
605             last = narray.jjtGetNumChildren() - 1;
606             for (int i = 1; i < last; i++) {
607                 objectNode = narray.jjtGetChild(i);
608                 if (objectNode instanceof JexlNode.Literal<?>) {
609                     object = objectNode.jjtAccept(this, object);
610                 } else {
611                     Object index = objectNode.jjtAccept(this, null);
612                     object = getAttribute(object, index, objectNode);
613                 }
614             }
615             property = narray.jjtGetChild(last).jjtAccept(this, null);
616         } else if (!isRegister) {
617             throw new JexlException(objectNode, "illegal assignment form");
618         }
619         // deal with ant variable; set context
620         if (isRegister) {
621             registers[register] = right;
622             return right;
623         } else if (antVar) {
624             if (isVariable && object == null) {
625                 if (variableName != null) {
626                     if (last > 0) {
627                         variableName.append('.');
628                     }
629                     variableName.append(property);
630                     property = variableName.toString();
631                 }
632                 try {
633                     context.set(String.valueOf(property), right);
634                 } catch (UnsupportedOperationException xsupport) {
635                     throw new JexlException(node, "context is readonly", xsupport);
636                 }
637                 return right;
638             }
639         }
640         if (property == null) {
641             // no property, we fail
642             throw new JexlException(propertyNode, "property is null");
643         }
644         if (object == null) {
645             // no object, we fail
646             throw new JexlException(objectNode, "bean is null");
647         }
648         // one before last, assign
649         setAttribute(object, property, right, propertyNode);
650         return right;
651     }

831             // try contains on map key

833                 return ((Map<?, ?>) right).containsKey(left) ? Boolean.TRUE : Boolean.FALSE;
834             }
835             // try contains on collection
836             if (right instanceof Collection<?>) {
837                 return ((Collection<?>) right).contains(left) ? Boolean.TRUE : Boolean.FALSE;

839             // try a contains method (duck type set)
840             try {

843                 if (vm != null) {
844                     return arithmetic.toBoolean(vm.invoke(right, argv)) ? Boolean.TRUE : Boolean.FALSE;

848                         return arithmetic.toBoolean(vm.invoke(right, argv)) ? Boolean.TRUE : Boolean.FALSE;

851             } catch (InvocationTargetException e) {
852                 throw new JexlException(node, "=~ invocation error", e.getCause());
853             } catch (Exception e) {
854                 throw new JexlException(node, "=~ error", e);
855             }
856             // try iterative comparison
857             Iterator<?> it = uberspect.getIterator(right, node);
858             if (it != null) {
859                 while (it.hasNext()) {
860                     Object next = it.next();
861                     if (next == left || (next != null && next.equals(left))) {
862                         return Boolean.TRUE;
863                     }
864                 }
865                 return Boolean.FALSE;
866             }
867             // defaults to equal
868             return arithmetic.equals(left, right) ? Boolean.TRUE : Boolean.FALSE;
869         } catch (ArithmeticException xrt) {
870             throw new JexlException(node, "=~ error", xrt);
871         }

1008     /**
1009      * Calls a method (or function).
1010      * <p>
1011      * Method resolution is a follows:
1012      * 1 - attempt to find a method in the bean passed as parameter;
1013      * 2 - if this fails, narrow the arguments and try again
1014      * 3 - if this still fails, seeks a Script or JexlMethod as a property of that bean.
1015      * </p>
1016      * @param node the method node
1017      * @param bean the bean this method should be invoked upon
1018      * @param methodNode the node carrying the method name
1019      * @param argb the first argument index, child of the method node
1020      * @return the result of the method invocation
1021      */
1022     private Object call(JexlNode node, Object bean, ASTIdentifier methodNode, int argb) {
1023         if (isCancelled()) {
1024             throw new JexlException.Cancel(node);
1025         }
1026         String methodName = methodNode.image;
1027         // evaluate the arguments
1028         int argc = node.jjtGetNumChildren() - argb;
1029         Object[] argv = new Object[argc];
1030         for (int i = 0; i < argc; i++) {
1031             argv[i] = node.jjtGetChild(i + argb).jjtAccept(this, null);
1032         }
1033 
1034         JexlException xjexl = null;
1035         try {
1036             // attempt to reuse last executor cached in volatile JexlNode.value
1037             if (cache) {
1038                 Object cached = node.jjtGetValue();
1039                 if (cached instanceof JexlMethod) {
1040                     JexlMethod me = (JexlMethod) cached;
1041                     Object eval = me.tryInvoke(methodName, bean, argv);
1042                     if (!me.tryFailed(eval)) {
1043                         return eval;
1044                     }
1045                 }
1046             }
1047             boolean cacheable = cache;
1048             JexlMethod vm = uberspect.getMethod(bean, methodName, argv, node);
1049             // DG: If we can't find an exact match, narrow the parameters and try again
1050             if (vm == null) {
1051                 if (arithmetic.narrowArguments(argv)) {
1052                     vm = uberspect.getMethod(bean, methodName, argv, node);
1053                 }
1054                 if (vm == null) {
1055                     Object functor = null;
1056                     // could not find a method, try as a var
1057                     if (bean == context) {
1058                         int register = methodNode.getRegister();
1059                         if (register >= 0) {
1060                             functor = registers[register];
1061                         } else {
1062                             functor = context.get(methodName);
1063                         }
1064                     } else {
1065                         JexlPropertyGet gfunctor = uberspect.getPropertyGet(bean, methodName, node);
1066                         if (gfunctor != null) {
1067                             functor = gfunctor.tryInvoke(bean, methodName);
1068                         }
1069                     }
1070                     // script of jexl method will do
1071                     if (functor instanceof Script) {
1072                         return ((Script) functor).execute(context, argv.length > 0 ? argv : null);
1073                     } else if (functor instanceof JexlMethod) {
1074                         vm = (JexlMethod) functor;
1075                         cacheable = false;
1076                     } else {
1077                         xjexl = new JexlException.Method(node, methodName);
1078                     }
1079                 }
1080             }
1081             if (xjexl == null) {
1082                 // vm cannot be null if xjexl is null
1083                 Object eval = vm.invoke(bean, argv);
1084                 // cache executor in volatile JexlNode.value
1085                 if (cacheable && vm.isCacheable()) {
1086                     node.jjtSetValue(vm);
1087                 }
1088                 return eval;
1089             }
1090         } catch (InvocationTargetException e) {
1091             xjexl = new JexlException(node, "method invocation error", e.getCause());
1092         } catch (Exception e) {
1093             xjexl = new JexlException(node, "method error", e);
1094         }
1095         return invocationFailed(xjexl);
1096     }

1221     public Object visit(ASTNRNode node, Object data) {
1222         Object left = node.jjtGetChild(0).jjtAccept(this, data);
1223         Object right = node.jjtGetChild(1).jjtAccept(this, data);
1224         try {
1225             if (right instanceof java.util.regex.Pattern || right instanceof String) {
1226                 // use arithmetic / pattern matching
1227                 return arithmetic.matches(left, right) ? Boolean.FALSE : Boolean.TRUE;
1228             }
1229             // try contains on collection
1230             if (right instanceof Set<?>) {
1231                 return ((Set<?>) right).contains(left) ? Boolean.FALSE : Boolean.TRUE;
1232             }
1233             // try contains on map key
1234             if (right instanceof Map<?, ?>) {
1235                 return ((Map<?, ?>) right).containsKey(left) ? Boolean.FALSE : Boolean.TRUE;
1236             }
1237             // try contains on collection
1238             if (right instanceof Collection<?>) {
1239                 return ((Collection<?>) right).contains(left) ? Boolean.FALSE : Boolean.TRUE;
1240             }
1241             // try a contains method (duck type set)
1242             try {
1243                 Object[] argv = {left};
1244                 JexlMethod vm = uberspect.getMethod(right, "contains", argv, node);
1245                 if (vm != null) {
1246                     return arithmetic.toBoolean(vm.invoke(right, argv)) ? Boolean.FALSE : Boolean.TRUE;
1247                 } else if (arithmetic.narrowArguments(argv)) {
1248                     vm = uberspect.getMethod(right, "contains", argv, node);
1249                     if (vm != null) {
1250                         return arithmetic.toBoolean(vm.invoke(right, argv)) ? Boolean.FALSE : Boolean.TRUE;
1251                     }
1252                 }
1253             } catch (InvocationTargetException e) {
1254                 throw new JexlException(node, "!~ invocation error", e.getCause());
1255             } catch (Exception e) {
1256                 throw new JexlException(node, "!~ error", e);
1257             }
1258             // try iterative comparison
1259             Iterator<?> it = uberspect.getIterator(right, node.jjtGetChild(1));
1260             if (it != null) {
1261                 while (it.hasNext()) {
1262                     Object next = it.next();
1263                     if (next == left || (next != null && next.equals(left))) {
1264                         return Boolean.FALSE;
1265                     }
1266                 }
1267                 return Boolean.TRUE;
1268             }
1269             // defaults to not equal
1270             return arithmetic.equals(left, right) ? Boolean.FALSE : Boolean.TRUE;
1271         } catch (ArithmeticException xrt) {
1272             throw new JexlException(node, "!~ error", xrt);
1273         }
1274     }

1310     /** {@inheritDoc} */
1311     public Object visit(ASTReference node, Object data) {
1312         // could be array access, identifier or map literal
1313         // followed by zero or more ("." and array access, method, size,
1314         // identifier or integer literal)
1315         int numChildren = node.jjtGetNumChildren();
1316         // pass first piece of data in and loop through children
1317         Object result = null;
1318         StringBuilder variableName = null;
1319         String propertyName = null;
1320         boolean isVariable = true;
1321         int v = 0;
1322         for (int c = 0; c < numChildren; c++) {
1323             if (isCancelled()) {
1324                 throw new JexlException.Cancel(node);
1325             }
1326             JexlNode theNode = node.jjtGetChild(c);
1327             // integer literals may be part of an antish var name only if no bean was found so far
1328             if (result == null && theNode instanceof ASTNumberLiteral && ((ASTNumberLiteral) theNode).isInteger()) {
1329                 isVariable &= v > 0;
1330             } else {
1331                 isVariable &= (theNode instanceof ASTIdentifier);
1332                 result = theNode.jjtAccept(this, result);
1333             }
1334             // if we get null back a result, check for an ant variable
1335             if (result == null && isVariable) {
1336                 if (v == 0) {
1337                     variableName = new StringBuilder(node.jjtGetChild(0).image);
1338                     v = 1;
1339                 }
1340                 for (; v <= c; ++v) {
1341                     variableName.append('.');
1342                     variableName.append(node.jjtGetChild(v).image);
1343                 }
1344                 result = context.get(variableName.toString());
1345             } else {
1346                 propertyName = theNode.image;
1347             }
1348         }
1349         if (result == null) {
1350             if (isVariable && !isTernaryProtected(node)
1351                     // variable unknow in context and not (from) a register
1352                     && !(context.has(variableName.toString())
1353                     || (numChildren == 1
1354                     && node.jjtGetChild(0) instanceof ASTIdentifier
1355                     && ((ASTIdentifier) node.jjtGetChild(0)).getRegister() >= 0))) {
1356                 JexlException xjexl = propertyName != null
1357                                       ? new JexlException.Property(node, propertyName)
1358                                       : new JexlException.Variable(node, variableName.toString());
1359                 return unknownVariable(xjexl);
1360             }
1361         }
1362         return result;
1363     }

1479     /**
1480      * Calculate the <code>size</code> of various types: Collection, Array,
1481      * Map, String, and anything that has a int size() method.
1482      * @param node the node that gave the value to size
1483      * @param val the object to get the size of.
1484      * @return the size of val
1485      */
1486     private int sizeOf(JexlNode node, Object val) {
1487         if (val instanceof Collection<?>) {
1488             return ((Collection<?>) val).size();
1489         } else if (val.getClass().isArray()) {
1490             return Array.getLength(val);
1491         } else if (val instanceof Map<?, ?>) {
1492             return ((Map<?, ?>) val).size();
1493         } else if (val instanceof String) {
1494             return ((String) val).length();
1495         } else {
1496             // check if there is a size method on the object that returns an
1497             // integer and if so, just use it
1498             Object[] params = new Object[0];
1499             JexlMethod vm = uberspect.getMethod(val, "size", EMPTY_PARAMS, node);
1500             if (vm != null && vm.getReturnType() == Integer.TYPE) {
1501                 Integer result;
1502                 try {
1503                     result = (Integer) vm.invoke(val, params);
1504                 } catch (Exception e) {
1505                     throw new JexlException(node, "size() : error executing", e);
1506                 }
1507                 return result.intValue();
1508             }
1509             throw new JexlException(node, "size() : unsupported type : " + val.getClass(), null);
1510         }
1511     }

1595      *            key for a map
1596      * @param value the value to assign to the object's attribute
1597      * @param node the node that evaluated as the object
1598      */
1599     protected void setAttribute(Object object, Object attribute, Object value, JexlNode node) {
1600         if (isCancelled()) {
1601             throw new JexlException.Cancel(node);
1602         }
1603         // attempt to reuse last executor cached in volatile JexlNode.value
1604         if (node != null && cache) {
1605             Object cached = node.jjtGetValue();
1606             if (cached instanceof JexlPropertySet) {
1607                 JexlPropertySet setter = (JexlPropertySet) cached;
1608                 Object eval = setter.tryInvoke(object, attribute, value);
1609                 if (!setter.tryFailed(eval)) {
1610                     return;
1611                 }
1612             }
1613         }
1614         JexlException xjexl = null;
1615         JexlPropertySet vs = uberspect.getPropertySet(object, attribute, value, node);
1616         // if we can't find an exact match, narrow the value argument and try again
1617         if (vs == null) {
1618             // replace all numbers with the smallest type that will fit
1619             Object[] narrow = {value};
1620             if (arithmetic.narrowArguments(narrow)) {
1621                 vs = uberspect.getPropertySet(object, attribute, narrow[0], node);
1622             }
1623         }
1624         if (vs != null) {
1625             try {
1626                 // cache executor in volatile JexlNode.value
1627                 vs.invoke(object, value);
1628                 if (node != null && cache && vs.isCacheable()) {
1629                     node.jjtSetValue(vs);
1630                 }
1631                 return;
1632             } catch (RuntimeException xrt) {
1633                 if (node == null) {
1634                     throw xrt;
1635                 }
1636                 xjexl = new JexlException(node, "set object property error", xrt);
1637             } catch (Exception xany) {
1638                 if (node == null) {
1639                     throw new RuntimeException(xany);
1640                 }
1641                 xjexl = new JexlException(node, "set object property error", xany);
1642             }
1643         }
1644         if (xjexl == null) {
1645             if (node == null) {
1646                 String error = "unable to set object property"
1647                         + ", class: " + object.getClass().getName()
1648                         + ", property: " + attribute
1649                         + ", argument: " + value.getClass().getSimpleName();
1650                 throw new UnsupportedOperationException(error);
1651             }
1652             xjexl = new JexlException.Property(node, attribute.toString());
1653         }
1654         if (strict) {
1655             throw xjexl;
1656         }
1657         if (!silent) {
1658             logger.warn(xjexl.getMessage());
