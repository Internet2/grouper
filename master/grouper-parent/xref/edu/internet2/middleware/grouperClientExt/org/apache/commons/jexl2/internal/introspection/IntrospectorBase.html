45  /**
46   * This basic function of this class is to return a Method object for a
47   * particular class given the name of a method and the parameters to the method
48   * in the form of an Object[]
49   * <p/>
50   * The first time the Introspector sees a class it creates a class method map
51   * for the class in question. Basically the class method map is a Hastable where
52   * Method objects are keyed by a concatenation of the method name and the names
53   * of classes that make up the parameters.
54   *
55   * For example, a method with the following signature:
56   *
57   * public void method(String a, StringBuffer b)
58   *
59   * would be mapped by the key:
60   *
61   * "method" + "java.lang.String" + "java.lang.StringBuffer"
62   *
63   * This mapping is performed for all the methods in a class and stored.
64   * @since 1.0
65   */
66  public class IntrospectorBase {
67      /** the logger. */
68      protected final Log rlog;
69      /**
70       * Holds the method maps for the classes we know about, keyed by Class.
71       */
72      private final Map<Class<?>, ClassMap> classMethodMaps = new HashMap<Class<?>, ClassMap>();
73      /**
74       * The class loader used to solve constructors if needed.
75       */
76      private ClassLoader loader;
77      /**
78       * Holds the map of classes ctors we know about as well as unknown ones.
79       */
80      private final Map<MethodKey, Constructor<?>> constructorsMap = new HashMap<MethodKey, Constructor<?>>();
81      /**
82       * Holds the set of classes we have introspected.
83       */
84      private final Map<String, Class<?>> constructibleClasses = new HashMap<String, Class<?>>();
85  
86      /**
87       * Create the introspector.
88       * @param log the logger to use
89       */
90      public IntrospectorBase(Log log) {
91          this.rlog = log;
92          loader = getClass().getClassLoader();
93      }
94  
95      /**
96       * Gets a class by name through this introspector class loader.
97       * @param className the class name
98       * @return the class instance or null if it could not be found
99       */
100     public Class<?> getClassByName(String className) {
101         try {
102             return Class.forName(className, false, loader);
103         } catch (ClassNotFoundException xignore) {
104             return null;
105         }
106     }
107 
108     /**
109      * Gets the method defined by the <code>MethodKey</code> for the class <code>c</code>.
110      *
111      * @param c     Class in which the method search is taking place
112      * @param key   Key of the method being searched for
113      * @return The desired method object
114      * @throws MethodKey.AmbiguousException if no unambiguous method could be found through introspection
115      */
116     public Method getMethod(Class<?> c, MethodKey key) {
117         try {
118             ClassMap classMap = getMap(c);
119             return classMap.findMethod(key);
120         } catch (MethodKey.AmbiguousException xambiguous) {
121             // whoops.  Ambiguous.  Make a nice log message and return null...
122             if (rlog != null && rlog.isInfoEnabled()) {
123                 rlog.info("ambiguous method invocation: "
124                         + c.getName() + "."
125                         + key.debugString(), xambiguous);
126             }
127             return null;
128         }
129     }

136      * @return the desired field or null if it does not exist or is not accessible

138     public Field getField(Class<?> c, String key) {
139         ClassMap classMap = getMap(c);

169     /**
170      * Gets the array of accessible method known for a given class.
171      * @param c the class
172      * @param methodName the method name
173      * @return the array of methods (null or not empty)
174      */
175     public Method[] getMethods(Class<?> c, String methodName) {
176         if (c == null) {
177             return null;
178         }
179         ClassMap classMap = getMap(c);
180         return classMap.get(methodName);
181     }

183     /**
184      * A Constructor get cache-miss.
185      */
186     private static class CacheMiss {
187         /** The constructor used as cache-miss. */
188         @SuppressWarnings("unused")
189         public CacheMiss() {}
190     }
191     
192     /** The cache-miss marker for the constructors map. */
193     private static final Constructor<?> CTOR_MISS = CacheMiss.class.getConstructors()[0];
194 
195     /**
196      * Sets the class loader used to solve constructors.
197      * <p>Also cleans the constructors and methods caches.</p>
198      * @param cloader the class loader; if null, use this instance class loader
199      */
200     public void setLoader(ClassLoader cloader) {
201         ClassLoader previous = loader;
202         if (cloader == null) {
203             cloader = getClass().getClassLoader();
204         }
205         if (!cloader.equals(loader)) {
206             // clean up constructor and class maps
207             synchronized (constructorsMap) {
208                 Iterator<Map.Entry<MethodKey, Constructor<?>>> entries = constructorsMap.entrySet().iterator();
209                 while (entries.hasNext()) {
210                     Map.Entry<MethodKey, Constructor<?>> entry = entries.next();
211                     Class<?> clazz = entry.getValue().getDeclaringClass();
212                     if (isLoadedBy(previous, clazz)) {
213                         entries.remove();
214                         // the method name is the name of the class
215                         constructibleClasses.remove(entry.getKey().getMethod());
216                     }
217                 }
218             }
219             // clean up method maps
220             synchronized (classMethodMaps) {
221                 Iterator<Map.Entry<Class<?>, ClassMap>> entries = classMethodMaps.entrySet().iterator();
222                 while (entries.hasNext()) {
223                     Map.Entry<Class<?>, ClassMap> entry = entries.next();
224                     Class<?> clazz = entry.getKey();
225                     if (isLoadedBy(previous, clazz)) {
226                         entries.remove();
227                     }
228                 }
229             }
230             loader = cloader;
231         }
232     }

234     /**
235      * Checks whether a class is loaded through a given class loader or one of its ascendants.
236      * @param loader the class loader
237      * @param clazz the class to check
238      * @return true if clazz was loaded through the loader, false otherwise
239      */
240     private static boolean isLoadedBy(ClassLoader loader, Class<?> clazz) {
241         if (loader != null) {
242             ClassLoader cloader = clazz.getClassLoader();
243             while (cloader != null) {
244                 if (cloader.equals(loader)) {
245                     return true;
246                 } else {
247                     cloader = cloader.getParent();
248                 }
249             }
250         }
251         return false;
252     }

254     /**
255      * Gets the constructor defined by the <code>MethodKey</code>.
256      *
257      * @param key   Key of the constructor being searched for
258      * @return The desired constructor object
259      * or null if no unambiguous constructor could be found through introspection.
260      */
261     public Constructor<?> getConstructor(final MethodKey key) {
262         return getConstructor(null, key);
263     }
264 
265     /**
266      * Gets the constructor defined by the <code>MethodKey</code>.
267      * @param c the class we want to instantiate
268      * @param key   Key of the constructor being searched for
269      * @return The desired constructor object
270      * or null if no unambiguous constructor could be found through introspection.
271      */
272     public Constructor<?> getConstructor(final Class<?> c, final MethodKey key) {
273         Constructor<?> ctor = null;
274         synchronized (constructorsMap) {
275             ctor = constructorsMap.get(key);
276             // that's a clear miss
277             if (CTOR_MISS.equals(ctor)) {
278                 return null;
279             }
280             // let's introspect...
281             if (ctor == null) {
282                 final String cname = key.getMethod();
283                 // do we know about this class?
284                 Class<?> clazz = constructibleClasses.get(cname);
285                 try {
286                     // do find the most specific ctor
287                     if (clazz == null) {
288                         if (c != null && c.getName().equals(key.getMethod())) {
289                             clazz = c;
290                         } else {
291                             clazz = loader.loadClass(cname);
292                         }
293                         // add it to list of known loaded classes
294                         constructibleClasses.put(cname, clazz);
295                     }
296                     List<Constructor<?>> l = new LinkedList<Constructor<?>>();
297                     for (Constructor<?> ictor : clazz.getConstructors()) {
298                         l.add(ictor);
299                     }
300                     // try to find one
301                     ctor = key.getMostSpecificConstructor(l);
302                     if (ctor != null) {
303                         constructorsMap.put(key, ctor);
304                     } else {
305                         constructorsMap.put(key, CTOR_MISS);
306                     }
307                 } catch (ClassNotFoundException xnotfound) {
308                     if (rlog != null && rlog.isInfoEnabled()) {
309                         rlog.info("unable to find class: "
310                                 + cname + "."
311                                 + key.debugString(), xnotfound);
312                     }
313                     ctor = null;
314                 } catch (MethodKey.AmbiguousException xambiguous) {
315                     if (rlog != null && rlog.isInfoEnabled()) {
316                         rlog.info("ambiguous constructor invocation: "
317                                 + cname + "."
318                                 + key.debugString(), xambiguous);
319                     }
320                     ctor = null;
321                 }
322             }
323             return ctor;
324         }
325     }
326 
327     /**
328      * Gets the ClassMap for a given class.
329      * @param c the class
330      * @return the class map
331      */
332     private ClassMap getMap(Class<?> c) {
333         synchronized (classMethodMaps) {
334             ClassMap classMap = classMethodMaps.get(c);
335             if (classMap == null) {
336                 classMap = new ClassMap(c, rlog);
337                 classMethodMaps.put(c, classMap);
338             }
339             return classMap;
340         }
341     }
