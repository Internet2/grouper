32  package edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3;

34  import java.io.ByteArrayInputStream;
35  import java.io.ByteArrayOutputStream;
36  import java.io.IOException;
37  import java.io.InputStream;
38  import java.io.ObjectInputStream;
39  import java.io.ObjectOutputStream;
40  import java.io.ObjectStreamClass;
41  import java.io.OutputStream;
42  import java.io.Serializable;
43  
44  /**
45   * <p>Assists with the serialization process and performs additional functionality based
46   * on serialization.</p>
47   * <p>
48   * <ul>
49   * <li>Deep clone using serialization
50   * <li>Serialize managing finally and IOException
51   * <li>Deserialize managing finally and IOException
52   * </ul>
53   *
54   * <p>This class throws exceptions for invalid {@code null} inputs.
55   * Each method documents its behaviour in more detail.</p>
56   *
57   * <p>#ThreadSafe#</p>
58   * @since 1.0
59   * @version $Id: SerializationUtils.java 1199718 2011-11-09 12:43:20Z sebb $
60   */
61  public class SerializationUtils {
62  
63      /**
64       * <p>SerializationUtils instances should NOT be constructed in standard programming.
65       * Instead, the class should be used as {@code SerializationUtils.clone(object)}.</p>
66       *
67       * <p>This constructor is public to permit tools that require a JavaBean instance
68       * to operate.</p>
69       * @since 2.0
70       */
71      public SerializationUtils() {
72          super();
73      }
74  
75      // Clone
76      //-----------------------------------------------------------------------
77      /**
78       * <p>Deep clone an {@code Object} using serialization.</p>
79       *
80       * <p>This is many times slower than writing clone methods by hand
81       * on all objects in your object graph. However, for complex object
82       * graphs, or for those that don't support deep cloning this can
83       * be a simple alternative implementation. Of course all the objects
84       * must be {@code Serializable}.</p>
85       *
86       * @param <T> the type of the object involved
87       * @param object  the {@code Serializable} object to clone
88       * @return the cloned object
89       * @throws SerializationException (runtime) if the serialization fails
90       */
91      public static <T extends Serializable> T clone(T object) {
92          if (object == null) {
93              return null;
94          }
95          byte[] objectData = serialize(object);
96          ByteArrayInputStream bais = new ByteArrayInputStream(objectData);
97  
98          ClassLoaderAwareObjectInputStream in = null;
99          try {
100             // stream closed in the finally
101             in = new ClassLoaderAwareObjectInputStream(bais, object.getClass().getClassLoader());
102             /*
103              * when we serialize and deserialize an object,
104              * it is reasonable to assume the deserialized object
105              * is of the same type as the original serialized object
106              */
107             @SuppressWarnings("unchecked") // see above
108             T readObject = (T) in.readObject();
109             return readObject;
110 
111         } catch (ClassNotFoundException ex) {
112             throw new SerializationException("ClassNotFoundException while reading cloned object data", ex);
113         } catch (IOException ex) {
114             throw new SerializationException("IOException while reading cloned object data", ex);
115         } finally {
116             try {
117                 if (in != null) {
118                     in.close();
119                 }
120             } catch (IOException ex) {
121                 throw new SerializationException("IOException on closing cloned object data InputStream.", ex);
122             }
123         }
124     }
125 
126     // Serialize
127     //-----------------------------------------------------------------------
128     /**
129      * <p>Serializes an {@code Object} to the specified stream.</p>
130      *
131      * <p>The stream will be closed once the object is written.
132      * This avoids the need for a finally clause, and maybe also exception
133      * handling, in the application code.</p>
134      *
135      * <p>The stream passed in is not buffered internally within this method.
136      * This is the responsibility of your application if desired.</p>
137      *
138      * @param obj  the object to serialize to bytes, may be null
139      * @param outputStream  the stream to write to, must not be null
140      * @throws IllegalArgumentException if {@code outputStream} is {@code null}
141      * @throws SerializationException (runtime) if the serialization fails
142      */
143     public static void serialize(Serializable obj, OutputStream outputStream) {
144         if (outputStream == null) {
145             throw new IllegalArgumentException("The OutputStream must not be null");
146         }
147         ObjectOutputStream out = null;
148         try {
149             // stream closed in the finally
150             out = new ObjectOutputStream(outputStream);
151             out.writeObject(obj);
152 
153         } catch (IOException ex) {
154             throw new SerializationException(ex);
155         } finally {
156             try {
157                 if (out != null) {
158                     out.close();
159                 }
160             } catch (IOException ex) { // NOPMD
161                 // ignore close exception
162             }
163         }
164     }
165 
166     /**
167      * <p>Serializes an {@code Object} to a byte array for
168      * storage/serialization.</p>
169      *
170      * @param obj  the object to serialize to bytes
171      * @return a byte[] with the converted Serializable
172      * @throws SerializationException (runtime) if the serialization fails
173      */
174     public static byte[] serialize(Serializable obj) {
175         ByteArrayOutputStream baos = new ByteArrayOutputStream(512);
176         serialize(obj, baos);
177         return baos.toByteArray();
178     }
179 
180     // Deserialize
181     //-----------------------------------------------------------------------
182     /**
183      * <p>Deserializes an {@code Object} from the specified stream.</p>
184      *
185      * <p>The stream will be closed once the object is written. This
186      * avoids the need for a finally clause, and maybe also exception
187      * handling, in the application code.</p>
188      *
189      * <p>The stream passed in is not buffered internally within this method.
190      * This is the responsibility of your application if desired.</p>
191      *
192      * @param inputStream  the serialized object input stream, must not be null
193      * @return the deserialized object
194      * @throws IllegalArgumentException if {@code inputStream} is {@code null}
195      * @throws SerializationException (runtime) if the serialization fails
196      */
197     public static Object deserialize(InputStream inputStream) {
198         if (inputStream == null) {
199             throw new IllegalArgumentException("The InputStream must not be null");
200         }
201         ObjectInputStream in = null;
202         try {
203             // stream closed in the finally
204             in = new ObjectInputStream(inputStream);
205             return in.readObject();
206 
207         } catch (ClassNotFoundException ex) {
208             throw new SerializationException(ex);
209         } catch (IOException ex) {
210             throw new SerializationException(ex);
211         } finally {
212             try {
213                 if (in != null) {
214                     in.close();
215                 }
216             } catch (IOException ex) { // NOPMD
217                 // ignore close exception
218             }
219         }
220     }
221 
222     /**
223      * <p>Deserializes a single {@code Object} from an array of bytes.</p>
224      *
225      * @param objectData  the serialized object, must not be null
226      * @return the deserialized object
227      * @throws IllegalArgumentException if {@code objectData} is {@code null}
228      * @throws SerializationException (runtime) if the serialization fails
229      */
230     public static Object deserialize(byte[] objectData) {
231         if (objectData == null) {
232             throw new IllegalArgumentException("The byte[] must not be null");
233         }
234         ByteArrayInputStream bais = new ByteArrayInputStream(objectData);
235         return deserialize(bais);
236     }
237 
238     /**
239      * <p>Custom specialization of the standard JDK {@link java.io.ObjectInputStream}
240      * that uses a custom  <code>ClassLoader</code> to resolve a class.
241      * If the specified <code>ClassLoader</code> is not able to resolve the class,
242      * the context classloader of the current thread will be used.
243      * This way, the standard deserialization work also in web-application
244      * containers and application servers, no matter in which of the
245      * <code>ClassLoader</code> the particular class that encapsulates
246      * serialization/deserialization lives. </p>
247      * 
248      * <p>For more in-depth information about the problem for which this
249      * class here is a workaround, see the JIRA issue LANG-626. </p>
250      */
251      static class ClassLoaderAwareObjectInputStream extends ObjectInputStream {
252         private ClassLoader classLoader;
253 
254         /**
255          * Constructor.
256          * @param in The <code>InputStream</code>.
257          * @param classLoader classloader to use
258          * @throws IOException if an I/O error occurs while reading stream header.
259          * @see java.io.ObjectInputStream
260          */
261         public ClassLoaderAwareObjectInputStream(InputStream in, ClassLoader classLoader) throws IOException {
262             super(in);
263             this.classLoader = classLoader;
264         }
265 
266         /**
267          * Overriden version that uses the parametrized <code>ClassLoader</code> or the <code>ClassLoader</code>
268          * of the current <code>Thread</code> to resolve the class.
269          * @param desc An instance of class <code>ObjectStreamClass</code>.
270          * @return A <code>Class</code> object corresponding to <code>desc</code>.
271          * @throws IOException Any of the usual Input/Output exceptions.
272          * @throws ClassNotFoundException If class of a serialized object cannot be found.
273          */
274         @Override
275         protected Class<?> resolveClass(ObjectStreamClass desc) throws IOException, ClassNotFoundException {
276             String name = desc.getName();
277             try {
278                 return Class.forName(name, false, classLoader);
279             } catch (ClassNotFoundException ex) {
280                 return Class.forName(name, false, Thread.currentThread().getContextClassLoader());
281             }
282         }
283 
284     }
