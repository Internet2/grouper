32  package edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.concurrent;
33  
34  import java.util.concurrent.Executors;
35  import java.util.concurrent.ThreadFactory;
36  import java.util.concurrent.atomic.AtomicLong;
37  
38  /**
39   * <p>
40   * An implementation of the {@code ThreadFactory} interface that provides some
41   * configuration options for the threads it creates.
42   * </p>
43   * <p>
44   * A {@code ThreadFactory} is used for instance by an {@code ExecutorService} to
45   * create the threads it uses for executing tasks. In many cases users do not
46   * have to care about a {@code ThreadFactory} because the default one used by an
47   * {@code ExecutorService} will do. However, if there are special requirements
48   * for the threads, a custom {@code ThreadFactory} has to be created.
49   * </p>
50   * <p>
51   * This class provides some frequently needed configuration options for the
52   * threads it creates. These are the following:
53   * <ul>
54   * <li>A name pattern for the threads created by this factory can be specified.
55   * This is often useful if an application uses multiple executor services for
56   * different purposes. If the names of the threads used by these services have
57   * meaningful names, log output or exception traces can be much easier to read.
58   * Naming patterns are <em>format strings</em> as used by the {@code
59   * String.format()} method. The string can contain the place holder {@code %d}
60   * which will be replaced by the number of the current thread ({@code
61   * ThreadFactoryImpl} keeps a counter of the threads it has already created).
62   * For instance, the naming pattern {@code "My %d. worker thread"} will result
63   * in thread names like {@code "My 1. worker thread"}, {@code
64   * "My 2. worker thread"} and so on.</li>
65   * <li>A flag whether the threads created by this factory should be daemon
66   * threads. This can impact the exit behavior of the current Java application
67   * because the JVM shuts down if there are only daemon threads running.</li>
68   * <li>The priority of the thread. Here an integer value can be provided. The
69   * {@code java.lang.Thread} class defines constants for valid ranges of priority
70   * values.</li>
71   * <li>The {@code UncaughtExceptionHandler} for the thread. This handler is
72   * called if an uncaught exception occurs within the thread.</li>
73   * </ul>
74   * </p>
75   * <p>
76   * {@code BasicThreadFactory} wraps another thread factory which actually
77   * creates new threads. The configuration options are set on the threads created
78   * by the wrapped thread factory. On construction time the factory to be wrapped
79   * can be specified. If none is provided, a default {@code ThreadFactory} is
80   * used.
81   * </p>
82   * <p>
83   * Instances of {@code BasicThreadFactory} are not created directly, but the
84   * nested {@code Builder} class is used for this purpose. Using the builder only
85   * the configuration options an application is interested in need to be set. The
86   * following example shows how a {@code BasicThreadFactory} is created and
87   * installed in an {@code ExecutorService}:
88   *
89   * <pre>
90   * // Create a factory that produces daemon threads with a naming pattern and
91   * // a priority
92   * BasicThreadFactory factory = new BasicThreadFactory.Builder()
93   *     .namingPattern(&quot;workerthread-%d&quot;)
94   *     .daemon(true)
95   *     .priority(Thread.MAX_PRIORITY)
96   *     .build();
97   * // Create an executor service for single-threaded execution
98   * ExecutorService exec = Executors.newSingleThreadExecutor(factory);
99   * </pre>
100  * </p>
101  *
102  * @since 3.0
103  * @version $Id: BasicThreadFactory.java 1079423 2011-03-08 16:38:09Z sebb $
104  */
105 public class BasicThreadFactory implements ThreadFactory {
106     /** A counter for the threads created by this factory. */
107     private final AtomicLong threadCounter;
108 
109     /** Stores the wrapped factory. */
110     private final ThreadFactory wrappedFactory;
111 
112     /** Stores the uncaught exception handler. */
113     private final Thread.UncaughtExceptionHandler uncaughtExceptionHandler;
114 
115     /** Stores the naming pattern for newly created threads. */
116     private final String namingPattern;
117 
118     /** Stores the priority. */
119     private final Integer priority;
120 
121     /** Stores the daemon status flag. */
122     private final Boolean daemonFlag;
123 
124     /**
125      * Creates a new instance of {@code ThreadFactoryImpl} and configures it
126      * from the specified {@code Builder} object.
127      *
128      * @param builder the {@code Builder} object
129      */
130     private BasicThreadFactory(Builder builder) {
131         if (builder.wrappedFactory == null) {
132             wrappedFactory = Executors.defaultThreadFactory();
133         } else {
134             wrappedFactory = builder.wrappedFactory;
135         }
136 
137         namingPattern = builder.namingPattern;
138         priority = builder.priority;
139         daemonFlag = builder.daemonFlag;
140         uncaughtExceptionHandler = builder.exceptionHandler;
141 
142         threadCounter = new AtomicLong();
143     }
144 
145     /**
146      * Returns the wrapped {@code ThreadFactory}. This factory is used for
147      * actually creating threads. This method never returns <b>null</b>. If no
148      * {@code ThreadFactory} was passed when this object was created, a default
149      * thread factory is returned.
150      *
151      * @return the wrapped {@code ThreadFactory}
152      */
153     public final ThreadFactory getWrappedFactory() {
154         return wrappedFactory;
155     }
156 
157     /**
158      * Returns the naming pattern for naming newly created threads. Result can
159      * be <b>null</b> if no naming pattern was provided.
160      *
161      * @return the naming pattern
162      */
163     public final String getNamingPattern() {
164         return namingPattern;
165     }
166 
167     /**
168      * Returns the daemon flag. This flag determines whether newly created
169      * threads should be daemon threads. If <b>true</b>, this factory object
170      * calls {@code setDaemon(true)} on the newly created threads. Result can be
171      * <b>null</b> if no daemon flag was provided at creation time.
172      *
173      * @return the daemon flag
174      */
175     public final Boolean getDaemonFlag() {
176         return daemonFlag;
177     }
178 
179     /**
180      * Returns the priority of the threads created by this factory. Result can
181      * be <b>null</b> if no priority was specified.
182      *
183      * @return the priority for newly created threads
184      */
185     public final Integer getPriority() {
186         return priority;
187     }
188 
189     /**
190      * Returns the {@code UncaughtExceptionHandler} for the threads created by
191      * this factory. Result can be <b>null</b> if no handler was provided.
192      *
193      * @return the {@code UncaughtExceptionHandler}
194      */
195     public final Thread.UncaughtExceptionHandler getUncaughtExceptionHandler() {
196         return uncaughtExceptionHandler;
197     }
198 
199     /**
200      * Returns the number of threads this factory has already created. This
201      * class maintains an internal counter that is incremented each time the
202      * {@link #newThread(Runnable)} method is invoked.
203      *
204      * @return the number of threads created by this factory
205      */
206     public long getThreadCount() {
207         return threadCounter.get();
208     }
209 
210     /**
211      * Creates a new thread. This implementation delegates to the wrapped
212      * factory for creating the thread. Then, on the newly created thread the
213      * corresponding configuration options are set.
214      *
215      * @param r the {@code Runnable} to be executed by the new thread
216      * @return the newly created thread
217      */
218     public Thread newThread(Runnable r) {
219         Thread t = getWrappedFactory().newThread(r);
220         initializeThread(t);
221 
222         return t;
223     }
224 
225     /**
226      * Initializes the specified thread. This method is called by
227      * {@link #newThread(Runnable)} after a new thread has been obtained from
228      * the wrapped thread factory. It initializes the thread according to the
229      * options set for this factory.
230      *
231      * @param t the thread to be initialized
232      */
233     private void initializeThread(Thread t) {
234 
235         if (getNamingPattern() != null) {
236             Long count = Long.valueOf(threadCounter.incrementAndGet());
237             t.setName(String.format(getNamingPattern(), count));
238         }
239 
240         if (getUncaughtExceptionHandler() != null) {
241             t.setUncaughtExceptionHandler(getUncaughtExceptionHandler());
242         }
243 
244         if (getPriority() != null) {
245             t.setPriority(getPriority().intValue());
246         }
247 
248         if (getDaemonFlag() != null) {
249             t.setDaemon(getDaemonFlag().booleanValue());
250         }
251     }
252 
253     /**
254      * <p>
255      * A <em>builder</em> class for creating instances of {@code
256      * BasicThreadFactory}.
257      * </p>
258      * <p>
259      * Using this builder class instances of {@code BasicThreadFactory} can be
260      * created and initialized. The class provides methods that correspond to
261      * the configuration options supported by {@code BasicThreadFactory}. Method
262      * chaining is supported. Refer to the documentation of {@code
263      * BasicThreadFactory} for a usage example.
264      * </p>
265      *
266      * @version $Id: BasicThreadFactory.java 1079423 2011-03-08 16:38:09Z sebb $
267      */
268     public static class Builder 
269         implements edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.builder.Builder<BasicThreadFactory> {
270         
271         /** The wrapped factory. */
272         private ThreadFactory wrappedFactory;
273 
274         /** The uncaught exception handler. */
275         private Thread.UncaughtExceptionHandler exceptionHandler;
276 
277         /** The naming pattern. */
278         private String namingPattern;
279 
280         /** The priority. */
281         private Integer priority;
282 
283         /** The daemon flag. */
284         private Boolean daemonFlag;
285 
286         /**
287          * Sets the {@code ThreadFactory} to be wrapped by the new {@code
288          * BasicThreadFactory}.
289          *
290          * @param factory the wrapped {@code ThreadFactory} (must not be
291          * <b>null</b>)
292          * @return a reference to this {@code Builder}
293          * @throws NullPointerException if the passed in {@code ThreadFactory}
294          * is <b>null</b>
295          */
296         public Builder wrappedFactory(ThreadFactory factory) {
297             if (factory == null) {
298                 throw new NullPointerException(
299                         "Wrapped ThreadFactory must not be null!");
300             }
301 
302             wrappedFactory = factory;
303             return this;
304         }
305 
306         /**
307          * Sets the naming pattern to be used by the new {@code
308          * BasicThreadFactory}.
309          *
310          * @param pattern the naming pattern (must not be <b>null</b>)
311          * @return a reference to this {@code Builder}
312          * @throws NullPointerException if the naming pattern is <b>null</b>
313          */
314         public Builder namingPattern(String pattern) {
315             if (pattern == null) {
316                 throw new NullPointerException(
317                         "Naming pattern must not be null!");
318             }
319 
320             namingPattern = pattern;
321             return this;
322         }
323 
324         /**
325          * Sets the daemon flag for the new {@code BasicThreadFactory}. If this
326          * flag is set to <b>true</b> the new thread factory will create daemon
327          * threads.
328          *
329          * @param f the value of the daemon flag
330          * @return a reference to this {@code Builder}
331          */
332         public Builder daemon(boolean f) {
333             daemonFlag = Boolean.valueOf(f);
334             return this;
335         }
336 
337         /**
338          * Sets the priority for the threads created by the new {@code
339          * BasicThreadFactory}.
340          *
341          * @param prio the priority
342          * @return a reference to this {@code Builder}
343          */
344         public Builder priority(int prio) {
345             priority = Integer.valueOf(prio);
346             return this;
347         }
348 
349         /**
350          * Sets the uncaught exception handler for the threads created by the
351          * new {@code BasicThreadFactory}.
352          *
353          * @param handler the {@code UncaughtExceptionHandler} (must not be
354          * <b>null</b>)
355          * @return a reference to this {@code Builder}
356          * @throws NullPointerException if the exception handler is <b>null</b>
357          */
358         public Builder uncaughtExceptionHandler(
359                 Thread.UncaughtExceptionHandler handler) {
360             if (handler == null) {
361                 throw new NullPointerException(
362                         "Uncaught exception handler must not be null!");
363             }
364 
365             exceptionHandler = handler;
366             return this;
367         }
368 
369         /**
370          * Resets this builder. All configuration options are set to default
371          * values. Note: If the {@link #build()} method was called, it is not
372          * necessary to call {@code reset()} explicitly because this is done
373          * automatically.
374          */
375         public void reset() {
376             wrappedFactory = null;
377             exceptionHandler = null;
378             namingPattern = null;
379             priority = null;
380             daemonFlag = null;
381         }
382 
383         /**
384          * Creates a new {@code BasicThreadFactory} with all configuration
385          * options that have been specified by calling methods on this builder.
386          * After creating the factory {@link #reset()} is called.
387          *
388          * @return the new {@code BasicThreadFactory}
389          */
390         public BasicThreadFactory build() {
391             BasicThreadFactory factory = new BasicThreadFactory(this);
392             reset();
393             return factory;
394         }
395     }
