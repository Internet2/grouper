46  package edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.methods;
47  
48  import edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.HttpMethodBase;
49  import edu.internet2.middleware.grouperClientExt.org.apache.commons.logging.Log;
50  import edu.internet2.middleware.grouperClientExt.org.apache.commons.logging.LogFactory;
51  
52  /**
53   * Implements the HTTP GET method.
54   * <p>
55   * The HTTP GET method is defined in section 9.3 of
56   * <a href="http://www.ietf.org/rfc/rfc2616.txt">RFC2616</a>:
57   * <blockquote>
58   * The GET method means retrieve whatever information (in the form of an
59   * entity) is identified by the Request-URI. If the Request-URI refers
60   * to a data-producing process, it is the produced data which shall be
61   * returned as the entity in the response and not the source text of the
62   * process, unless that text happens to be the output of the process.
63   * </blockquote>
64   * </p>
65   * <p>
66   * GetMethods will follow redirect requests from the http server by default.
67   * This behavour can be disabled by calling setFollowRedirects(false).</p>
68   *
69   * @author <a href="mailto:remm@apache.org">Remy Maucherat</a>
70   * @author Sung-Gu Park
71   * @author Sean C. Sullivan
72   * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
73   * @author <a href="mailto:jsdever@apache.org">Jeff Dever</a>
74   * 
75   * @version $Revision: 1.1 $
76   * @since 1.0
77   */
78  public class GetMethod extends HttpMethodBase {
79  
80      // -------------------------------------------------------------- Constants
81  
82      /** Log object for this class. */
83      private static final Log LOG = LogFactory.getLog(GetMethod.class);
84  
85      // ----------------------------------------------------------- Constructors
86  
87      /**
88       * No-arg constructor.
89       * 
90       * @since 1.0
91       */
92      public GetMethod() {
93          setFollowRedirects(true);
94      }
95  
96      /**
97       * Constructor specifying a URI.
98       *
99       * @param uri either an absolute or relative URI
100      * 
101      * @since 1.0
102      */
103     public GetMethod(String uri) {
104         super(uri);
105         LOG.trace("enter GetMethod(String)");
106         setFollowRedirects(true);
107     }
108 
109     // --------------------------------------------------------- Public Methods
110 
111     /**
112      * Returns <tt>"GET"</tt>.
113      * 
114      * @return <tt>"GET"</tt>
115      * 
116      * @since 2.0
117      */
118     public String getName() {
119         return "GET";
120     }
121 
122     // ------------------------------------------------------------- Properties
123 
124     /**
125      * Recycles the HTTP method so that it can be used again.
126      * Note that all of the instance variables will be reset
127      * once this method has been called. This method will also
128      * release the connection being used by this HTTP method.
129      * 
130      * @see #releaseConnection()
131      * 
132      * @since 1.0
133      * 
134      * @deprecated no longer supported and will be removed in the future
135      *             version of HttpClient
136      */
137     public void recycle() {
138         LOG.trace("enter GetMethod.recycle()");
139 
140         super.recycle();
141         setFollowRedirects(true);
142     }
143 
144 }
