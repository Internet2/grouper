426     /**
427      * Used to invert an escape array into an unescape array
428      * @param array String[][] to be inverted
429      * @return String[][] inverted array
430      */
431     public static String[][] invert(String[][] array) {
432         String[][] newarray = new String[array.length][2];
433         for(int i = 0; i<array.length; i++) {
434             newarray[i][0] = array[i][1];
435             newarray[i][1] = array[i][0];
436         }
437         return newarray;
438     }
