27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io;
28  
29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.ErrorWriter;
30  
31  import java.util.Iterator;
32  
33  /**
34   * @author Joe Walnes
35   */
36  public interface HierarchicalStreamReader {
37  
38      /**
39       * Does the node have any more children remaining that have not yet been read?
40       */
41      boolean hasMoreChildren();
42  
43      /**
44       * Select the current child as current node.
45       * A call to this function must be balanced with a call to {@link #moveUp()}.
46       */
47      void moveDown();
48  
49      /**
50       * Select the parent node as current node.
51       */
52      void moveUp();
53  
54      /**
55       * Get the name of the current node.
56       */
57      String getNodeName();
58  
59      /**
60       * Get the value (text content) of the current node.
61       */
62      String getValue();
63  
64      /**
65       * Get the value of an attribute of the current node.
66       */
67      String getAttribute(String name);
68  
69      /**
70       * Get the value of an attribute of the current node, by index.
71       */
72      String getAttribute(int index);
73      
74      /**
75       * Number of attributes in current node.
76       */
77      int getAttributeCount();
78  
79      /**
80       * Name of attribute in current node.
81       */
82      String getAttributeName(int index);
83  
84      /**
85       * Names of attributes (as Strings). 
86       */
87      Iterator getAttributeNames();
88  
89      /**
90       * If any errors are detected, allow the reader to add any additional information that can aid debugging
91       * (such as line numbers, XPath expressions, etc).
92       */
93      void appendErrors(ErrorWriter errorWriter);
94  
95      /**
96       * Close the reader, if necessary.
97       */
98      void close();
99  
100     /**
101      * Return the underlying HierarchicalStreamReader implementation.
102      *
103      * <p>If a Converter needs to access methods of a specific HierarchicalStreamReader implementation that are not
104      * defined in the HierarchicalStreamReader interface, it should call this method before casting. This is because
105      * the reader passed to the Converter is often wrapped/decorated by another implementation to provide additional
106      * functionality (such as XPath tracking).</p>
107      *
108      * <p>For example:</p>
109      * <pre>MySpecificReader mySpecificReader = (MySpecificReader)reader; <b>// INCORRECT!</b>
110      * mySpecificReader.doSomethingSpecific();</pre>
111 
112      * <pre>MySpecificReader mySpecificReader = (MySpecificReader)reader.underlyingReader();  <b>// CORRECT!</b>
113      * mySpecificReader.doSomethingSpecific();</pre>
114      *
115      * <p>Implementations of HierarchicalStreamReader should return 'this', unless they are a decorator, in which case
116      * they should delegate to whatever they are wrapping.</p>
117      */
118     HierarchicalStreamReader underlyingReader();
119 
120 }
