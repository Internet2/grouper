27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters;
28  
29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamReader;
30  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamWriter;
31  
32  /**
33   * Converter implementations are responsible marshalling Java objects
34   * to/from textual data.
35   * <p/>
36   * <p>If an exception occurs during processing, a {@link ConversionException}
37   * should be thrown.</p>
38   * <p/>
39   * <p>If working with the high level {@link edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.XStream} facade,
40   * you can register new converters using the XStream.registerConverter() method.</p>
41   * <p/>
42   * <p>If working with the lower level API, the
43   * {@link edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.ConverterLookup} implementation is
44   * responsible for looking up the appropriate converter.</p>
45   * <p/>
46   * <p>Converters for object that can store all information in a single value
47   * should implement {@link edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.SingleValueConverter}.
48   * <p>{@link edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.basic.AbstractSingleValueConverter}
49   * provides a starting point.</p>
50   * <p/>
51   * <p>{@link edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.collections.AbstractCollectionConverter}
52   * provides a starting point for objects that hold a collection of other objects
53   * (such as Lists and Maps).</p>
54   *
55   * @author Joe Walnes
56   * @see edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.XStream
57   * @see edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.ConverterLookup
58   * @see edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.basic.AbstractSingleValueConverter
59   * @see edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.collections.AbstractCollectionConverter
60   */
61  public interface Converter extends ConverterMatcher {
62  
63      /**
64       * Convert an object to textual data.
65       *
66       * @param source  The object to be marshalled.
67       * @param writer  A stream to write to.
68       * @param context A context that allows nested objects to be processed by XStream.
69       */
70      void marshal(Object source, HierarchicalStreamWriter writer, MarshallingContext context);
71  
72      /**
73       * Convert textual data back into an object.
74       *
75       * @param reader  The stream to read the text from.
76       * @param context
77       * @return The resulting object.
78       */
79      Object unmarshal(HierarchicalStreamReader reader, UnmarshallingContext context);
80  
81  }
