27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.xml;
28  
29  import java.io.BufferedReader;
30  import java.io.IOException;
31  import java.io.Reader;
32  
33  import org.xmlpull.mxp1.MXParser;
34  import org.xmlpull.v1.XmlPullParser;
35  import org.xmlpull.v1.XmlPullParserException;
36  
37  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.ErrorWriter;
38  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.StreamException;
39  
40  /**
41   * XStream reader that pulls directly from the stream using the XmlPullParser API.
42   *
43   * @author Joe Walnes
44   */
45  public class XppReader extends AbstractPullReader {
46  
47      private final XmlPullParser parser;
48      private final BufferedReader reader;
49  
50      public XppReader(Reader reader) {
51          this(reader, new XmlFriendlyReplacer());
52      }
53  
54      /**
55       * @since 1.2
56       */
57      public XppReader(Reader reader, XmlFriendlyReplacer replacer) {
58          super(replacer);
59          try {
60              parser = createParser();
61              this.reader = new BufferedReader(reader);
62              parser.setInput(this.reader);
63              moveDown();
64          } catch (XmlPullParserException e) {
65              throw new StreamException(e);
66          }
67      }
68      
69      /**
70       * To use another implementation of org.xmlpull.v1.XmlPullParser, override this method.
71       */
72      protected XmlPullParser createParser() {
73          return new MXParser();
74      }
75  
76      protected int pullNextEvent() {
77          try {
78              switch(parser.next()) {
79                  case XmlPullParser.START_DOCUMENT:
80                  case XmlPullParser.START_TAG:
81                      return START_NODE;
82                  case XmlPullParser.END_DOCUMENT:
83                  case XmlPullParser.END_TAG:
84                      return END_NODE;
85                  case XmlPullParser.TEXT:
86                      return TEXT;
87                  case XmlPullParser.COMMENT:
88                      return COMMENT;
89                  default:
90                      return OTHER;
91              }
92          } catch (XmlPullParserException e) {
93              throw new StreamException(e);
94          } catch (IOException e) {
95              throw new StreamException(e);
96          }
97      }
98  
99      protected String pullElementName() {
100         return parser.getName();
101     }
102 
103     protected String pullText() {
104         return parser.getText();
105     }
106 
107     public String getAttribute(String name) {
108         return parser.getAttributeValue(null, name);
109     }
110 
111     public String getAttribute(int index) {
112         return parser.getAttributeValue(index);
113     }
114 
115     public int getAttributeCount() {
116         return parser.getAttributeCount();
117     }
118 
119     public String getAttributeName(int index) {
120         return unescapeXmlName(parser.getAttributeName(index));
121     }
122 
123     public void appendErrors(ErrorWriter errorWriter) {
124         errorWriter.add("line number", String.valueOf(parser.getLineNumber()));
125     }
126 
127     public void close() {
128         try {
129             reader.close();
130         } catch (IOException e) {
131             throw new StreamException(e);
132         }
133     }
134 
135 }
