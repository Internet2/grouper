36  /**
37   * A simple writer that outputs XML in a pretty-printed indented stream.
38   * <p>
39   * By default, the chars <code><pre>
40   * &amp; &lt; &gt; &quot; ' \r
41   * </pre></code> are escaped and replaced with a suitable XML entity. To alter this behavior, override
42   * the the {@link #writeText(edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.core.util.QuickWriter, String)} and
43   * {@link #writeAttributeValue(edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.core.util.QuickWriter, String)} methods.
44   * </p>
45   * <p>
46   * Note: Depending on the XML version some characters cannot be written. Especially a 0
47   * character is never valid in XML, neither directly nor as entity nor within CDATA. However, this writer
48   * works by default in a quirks mode, where it will write any character at least as character entity (even
49   * a null character). You may switch into XML_1_1 mode (which supports most characters) or XML_1_0
50   * that does only support a very limited number of control characters. See XML specification for version
51   * <a href="http://www.w3.org/TR/2006/REC-xml-20060816/#charsets">1.0</a> or
52   * <a href="http://www.w3.org/TR/2006/REC-xml11-20060816/#charsets">1.1</a>. If a character is 
53   * not supported, a {@link StreamException} is thrown. Select a proper parser implementation that
54   * respects the version in the XML header (the Xpp3 parser will also read character entities of normally
55   * invalid characters).
56   * </p>
57   * 
58   * @author Joe Walnes
59   * @author J&ouml;rg Schaible
60   */
61  public class PrettyPrintWriter extends AbstractXmlWriter {
62  
63      public static int XML_QUIRKS = -1;
64      public static int XML_1_0 = 0;
65      public static int XML_1_1 = 1;
66  
67      private final QuickWriter writer;
68      private final FastStack elementStack = new FastStack(16);
69      private final char[] lineIndenter;
70      private final int mode;
71  
72      private boolean tagInProgress;
73      protected int depth;
74      private boolean readyForNewLine;
75      private boolean tagIsEmpty;
76      private String newLine;
77  
78      private static final char[] NULL = "&#x0;".toCharArray();
79      private static final char[] AMP = "&amp;".toCharArray();
80      private static final char[] LT = "&lt;".toCharArray();
81      private static final char[] GT = "&gt;".toCharArray();
82      private static final char[] CR = "&#xd;".toCharArray();
83      private static final char[] QUOT = "&quot;".toCharArray();
84      private static final char[] APOS = "&apos;".toCharArray();
85      private static final char[] CLOSE = "</".toCharArray();
86  
87      private PrettyPrintWriter(
88          Writer writer, int mode, char[] lineIndenter, XmlFriendlyReplacer replacer,
89          String newLine) {
90          super(replacer);
91          this.writer = new QuickWriter(writer);
92          this.lineIndenter = lineIndenter;
93          this.newLine = newLine;
94          this.mode = mode;
95          if (mode < XML_QUIRKS || mode > XML_1_1) {
96              throw new IllegalArgumentException("Not a valid XML mode");
97          }
98      }

175     public void startNode(String name) {
176         String escapedName = escapeXmlName(name);
177         tagIsEmpty = false;
178         finishTag();
179         writer.write('<');
180         writer.write(escapedName);
181         elementStack.push(escapedName);
182         tagInProgress = true;
183         depth++ ;
184         readyForNewLine = true;
185         tagIsEmpty = true;
186     }
187 
188     public void startNode(String name, Class clazz) {
189         startNode(name);
190     }
191 
192     public void setValue(String text) {
193         readyForNewLine = false;
194         tagIsEmpty = false;
195         finishTag();
196 
197         writeText(writer, text);
198     }
199 
200     public void addAttribute(String key, String value) {
201         writer.write(' ');
202         writer.write(escapeXmlName(key));
203         writer.write('=');
204         writer.write('\"');
205         writeAttributeValue(writer, value);
206         writer.write('\"');
207     }
208 
209     protected void writeAttributeValue(QuickWriter writer, String text) {
210         writeText(text);
211     }
212 
213     protected void writeText(QuickWriter writer, String text) {
214         writeText(text);
215     }
216 
217     private void writeText(String text) {
218         int length = text.length();
219         for (int i = 0; i < length; i++ ) {
220             char c = text.charAt(i);
221             switch (c) {
222             case '\0':
223                 if (mode == XML_QUIRKS) {
224                     this.writer.write(NULL);
225                 } else {
226                     throw new StreamException("Invalid character 0x0 in XML stream");
227                 }
228                 break;
229             case '&':
230                 this.writer.write(AMP);
231                 break;
232             case '<':
233                 this.writer.write(LT);
234                 break;
235             case '>':
236                 this.writer.write(GT);
237                 break;
238             case '"':
239                 this.writer.write(QUOT);
240                 break;
241             case '\'':
242                 this.writer.write(APOS);
243                 break;
244             case '\r':
245                 this.writer.write(CR);
246                 break;
247             case '\t':
248             case '\n':
249                 this.writer.write(c);
250                 break;
251             default:
252                 if (Character.isDefined(c) && !Character.isISOControl(c)) {
253                     if (mode != XML_QUIRKS) {
254                         if (c > '\ud7ff' && c < '\ue000') {
255                             throw new StreamException("Invalid character 0x"
256                                 + Integer.toHexString(c)
257                                 + " in XML stream");
258                         }
259                     }
260                     this.writer.write(c);
261                 } else {
262                     if (mode == XML_1_0) {
263                         if (c < 9
264                             || c == '\u000b'
265                             || c == '\u000c'
266                             || c == '\u000e'
267                             || c == '\u000f') {
268                             throw new StreamException("Invalid character 0x"
269                                 + Integer.toHexString(c)
270                                 + " in XML 1.0 stream");
271                         }
272                     }
273                     if (mode != XML_QUIRKS) {
274                         if (c == '\ufffe' || c == '\uffff') {
275                             throw new StreamException("Invalid character 0x"
276                                 + Integer.toHexString(c)
277                                 + " in XML stream");
278                         }
279                     }
280                     this.writer.write("&#x");
281                     this.writer.write(Integer.toHexString(c));
282                     this.writer.write(';');
283                 }
284             }
285         }
286     }
287 
288     public void endNode() {
289         depth-- ;
290         if (tagIsEmpty) {
291             writer.write('/');
292             readyForNewLine = false;
293             finishTag();
294             elementStack.popSilently();
295         } else {
296             finishTag();
297             writer.write(CLOSE);
298             writer.write((String)elementStack.pop());
299             writer.write('>');
300         }
301         readyForNewLine = true;
302         if (depth == 0) {
303             writer.flush();
304         }
305     }
306 
307     private void finishTag() {
308         if (tagInProgress) {
309             writer.write('>');
310         }
311         tagInProgress = false;
312         if (readyForNewLine) {
313             endOfLine();
314         }
315         readyForNewLine = false;
316         tagIsEmpty = false;
317     }
318 
319     protected void endOfLine() {
320         writer.write(getNewLine());
321         for (int i = 0; i < depth; i++ ) {
322             writer.write(lineIndenter);
323         }
324     }
325 
326     public void flush() {
327         writer.flush();
328     }
329 
330     public void close() {
331         writer.close();
332     }
333 
334     protected String getNewLine() {
335         return newLine;
336     }
