51  /**
52   * A SAX {@link org.xml.sax.XMLReader parser} that acts as an XStream
53   * {@link edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamWriter} to enable direct generation of a
54   * SAX event flow from the XStream serialization of a list of list of Java objects. <p/> As a
55   * custom SAX parser, this class ignores the arguments of the two standard parse methods ({@link #parse(java.lang.String)}
56   * and {@link #parse(org.xml.sax.InputSource)}) but relies on a proprietary SAX property
57   * {@link #SOURCE_OBJECT_LIST_PROPERTY} to define the list of objects to serialize.
58   * </p>
59   * <p/> Configuration of this SAX parser is achieved through the standard
60   * {@link #setProperty SAX property mechanism}. While specific setter methods require direct
61   * access to the parser instance, SAX properties support configuration settings to be propagated
62   * through a chain of {@link org.xml.sax.XMLFilter filters} down to the underlying parser
63   * object.
64   * </p>
65   * <p/> This mechanism shall be used to configure the
66   * {@link #SOURCE_OBJECT_LIST_PROPERTY objects to be serialized} as well as the
67   * {@link #CONFIGURED_XSTREAM_PROPERTY XStream facade}.
68   * </p>
69   * 
70   * @author Laurent Bihanic
71   */
72  public final class SaxWriter extends AbstractXmlWriter implements XMLReader {
73      /**
74       * The {@link #setProperty SAX property} to configure the XStream
75       * facade to be used for object serialization.  If the property
76       * is not set, a new XStream facade will be allocated for each
77       * parse.
78       */
79      public final static String CONFIGURED_XSTREAM_PROPERTY =
80              "http://com.thoughtworks.xstream/sax/property/configured-xstream";
81  
82      /**
83       * The {@link #setProperty SAX property} to configure the list of
84       * Java objects to serialize.  Setting this property prior
85       * invoking one of the parse() methods is mandatory.
86       *
87       * @see #parse(java.lang.String)
88       * @see #parse(org.xml.sax.InputSource)
89       */
90      public final static String SOURCE_OBJECT_LIST_PROPERTY =
91              "http://com.thoughtworks.xstream/sax/property/source-object-list";

157     /**
158      * Sets the state of a feature.
159      * <p/>
160      * The feature name is any fully-qualified URI.</p>
161      * <p/>
162      * All XMLReaders are required to support setting
163      * <code>http://xml.org/sax/features/namespaces</code> to
164      * <code>true</code> and
165      * <code>http://xml.org/sax/features/namespace-prefixes</code> to
166      * <code>false</code>.</p>
167      * <p/>
168      * Some feature values may be immutable or mutable only
169      * in specific contexts, such as before, during, or after
170      * a parse.</p>
171      * <p/>
172      * <strong>Note</strong>: This implementation only supports the two
173      * mandatory SAX features.</p>
174      *
175      * @param name  the feature name, which is a fully-qualified URI.
176      * @param value the requested state of the feature (true or false).
177      * @throws SAXNotRecognizedException when the XMLReader does not
178      *                                   recognize the feature name.
179      * @see #getFeature
180      */
181     public void setFeature(String name, boolean value)
182             throws SAXNotRecognizedException {
183         if ((name.equals("http://xml.org/sax/features/namespaces")) ||
184                 (name.equals("http://xml.org/sax/features/namespace-prefixes"))) {
185             this.features.put(name, value ? Boolean.TRUE : Boolean.FALSE); // JDK 1.3 friendly
186         } else {
187             throw new SAXNotRecognizedException(name);
188         }
189     }
190 
191     /**
192      * Looks up the value of a feature.
193      * <p/>
194      * The feature name is any fully-qualified URI.  It is
195      * possible for an XMLReader to recognize a feature name but
196      * to be unable to return its value; this is especially true
197      * in the case of an adapter for a SAX1 Parser, which has
198      * no way of knowing whether the underlying parser is
199      * performing validation or expanding external entities.</p>
200      * <p/>
201      * All XMLReaders are required to recognize the
202      * <code>http://xml.org/sax/features/namespaces</code> and the
203      * <code>http://xml.org/sax/features/namespace-prefixes</code> feature
204      * names.</p>
205      * <p/>
206      * Some feature values may be available only in specific
207      * contexts, such as before, during, or after a parse.</p>
208      * <p/>
209      * Implementors are free (and encouraged) to invent their own
210      * features, using names built on their own URIs.</p>
211      *
212      * @param name the feature name, which is a fully-qualified URI.
213      * @return the current state of the feature (true or false).
214      * @throws SAXNotRecognizedException when the XMLReader does not
215      *                                   recognize the feature name.
216      * @see #setFeature
217      */
218     public boolean getFeature(String name)
219             throws SAXNotRecognizedException {
220         if ((name.equals("http://xml.org/sax/features/namespaces")) ||
221                 (name.equals("http://xml.org/sax/features/namespace-prefixes"))) {
222             Boolean value = (Boolean) (this.features.get(name));
223 
224             if (value == null) {
225                 value = Boolean.FALSE;
226             }
227             return value.booleanValue();
228         } else {
229             throw new SAXNotRecognizedException(name);
230         }
231     }

233     /**
234      * Sets the value of a property.
235      * <p/>
236      * The property name is any fully-qualified URI.  It is
237      * possible for an XMLReader to recognize a property name but
238      * to be unable to set its value.</p>
239      * <p/>
240      * XMLReaders are not required to recognize setting any
241      * specific property names, though a core set is provided with
242      * SAX2.</p>
243      * <p/>
244      * Some property values may be immutable or mutable only
245      * in specific contexts, such as before, during, or after
246      * a parse.</p>
247      * <p/>
248      * This method is also the standard mechanism for setting
249      * extended handlers.</p>
250      * <p/>
251      * <strong>Note</strong>: This implementation only supports two
252      * (proprietary) properties: {@link #CONFIGURED_XSTREAM_PROPERTY}
253      * and {@link #SOURCE_OBJECT_LIST_PROPERTY}.</p>
254      *
255      * @param name  the property name, which is a fully-qualified URI.
256      * @param value the requested value for the property.
257      * @throws SAXNotRecognizedException when the XMLReader does not
258      *                                   recognize the property name.
259      * @throws SAXNotSupportedException  when the XMLReader recognizes
260      *                                   the property name but cannot set
261      *                                   the requested value.
262      * @see #getProperty
263      */
264     public void setProperty(String name, Object value)
265             throws SAXNotRecognizedException,
266             SAXNotSupportedException {
267         if (name.equals(CONFIGURED_XSTREAM_PROPERTY)) {
268             if (!(value instanceof XStream)) {
269                 throw new SAXNotSupportedException("Value for property \"" +
270                         CONFIGURED_XSTREAM_PROPERTY +
271                         "\" must be a non-null XStream object");
272             }
273         } else if (name.equals(SOURCE_OBJECT_LIST_PROPERTY)) {
274             if (value instanceof List) {
275                 List list = (List) value;
276 
277                 if (list.isEmpty()) {
278                     throw new SAXNotSupportedException("Value for property \"" +
279                             SOURCE_OBJECT_LIST_PROPERTY +
280                             "\" shall not be an empty list");
281                 } else {
282                     // Perform a copy of the list to prevent the application to
283                     // modify its content while the parse is being performed.
284                     value = Collections.unmodifiableList(new ArrayList(list));
285                 }
286             } else {
287                 throw new SAXNotSupportedException("Value for property \"" +
288                         SOURCE_OBJECT_LIST_PROPERTY +
289                         "\" must be a non-null List object");
290             }
291         } else {
292             throw new SAXNotRecognizedException(name);
293         }
294         this.properties.put(name, value);
295     }

298      * Looks up the value of a property.
299      * <p/>
300      * The property name is any fully-qualified URI.  It is
301      * possible for an XMLReader to recognize a property name but
302      * to be unable to return its state.</p>
303      * <p/>
304      * XMLReaders are not required to recognize any specific
305      * property names, though an initial core set is documented for
306      * SAX2.</p>
307      * <p/>
308      * Some property values may be available only in specific
309      * contexts, such as before, during, or after a parse.</p>
310      * <p/>
311      * Implementors are free (and encouraged) to invent their own properties,
312      * using names built on their own URIs.</p>
313      *
314      * @param name the property name, which is a fully-qualified URI.
315      * @return the current value of the property.
316      * @throws SAXNotRecognizedException when the XMLReader does not
317      *                                   recognize the property name.
318      * @see #getProperty
319      */
320     public Object getProperty(String name)

323                 (name.equals(SOURCE_OBJECT_LIST_PROPERTY))) {
324             return this.properties.get(name);
325         } else {
326             throw new SAXNotRecognizedException(name);
327         }
328     }

508     /**
509      * Parse an XML document.
510      * <p/>
511      * The application can use this method to instruct the XML
512      * reader to begin parsing an XML document from any valid input
513      * source (a character stream, a byte stream, or a URI).</p>
514      * <p/>
515      * Applications may not invoke this method while a parse is in
516      * progress (they should create a new XMLReader instead for each
517      * nested XML document).  Once a parse is complete, an
518      * application may reuse the same XMLReader object, possibly
519      * with a different input source.</p>
520      * <p/>
521      * During the parse, the XMLReader will provide information
522      * about the XML document through the registered event
523      * handlers.</p>
524      * <p/>
525      * This method is synchronous: it will not return until parsing
526      * has ended.  If a client application wants to terminate
527      * parsing early, it should throw an exception.</p>
528      * <p/>
529      * <strong>Note</strong>: As a custom SAX parser, this class
530      * ignores the <code>source</code> argument of this method
531      * and relies on the proprietary SAX property
532      * {@link #SOURCE_OBJECT_LIST_PROPERTY}) to define the list of
533      * objects to serialize.</p>
534      *
535      * @param input The input source for the top-level of the
536      *              XML document.
537      * @throws SAXException Any SAX exception, possibly wrapping
538      *                      another exception.
539      * @see org.xml.sax.InputSource
540      * @see #parse(java.lang.String)
541      * @see #setEntityResolver
542      * @see #setDTDHandler
543      * @see #setContentHandler
544      * @see #setErrorHandler
545      */
546     public void parse(InputSource input) throws SAXException {
547         this.parse();
548     }

550     /**
551      * Serializes the Java objects of the configured list into a flow
552      * of SAX events.
553      *
554      * @throws SAXException if the configured object list is invalid
555      *                      or object serialization failed.
556      */
557     private void parse() throws SAXException {
558         XStream xstream = (XStream) (this.properties.get(CONFIGURED_XSTREAM_PROPERTY));
559         if (xstream == null) {
560             xstream = new XStream();
561         }
562 
563         List source = (List) (this.properties.get(SOURCE_OBJECT_LIST_PROPERTY));
564         if ((source == null) || (source.isEmpty())) {
565             throw new SAXException("Missing or empty source object list. Setting property \"" +
566                     SOURCE_OBJECT_LIST_PROPERTY + "\" is mandatory");
567         }
568 
569         try {
570             this.startDocument(true);
571             for (Iterator i = source.iterator(); i.hasNext();) {
572                 xstream.marshal(i.next(), this);
573             }
574             this.endDocument(true);
575         } catch (StreamException e) {
576             if (e.getCause() instanceof SAXException) {
577                 throw (SAXException) (e.getCause());
578             } else {
579                 throw new SAXException(e);
580             }
581         }
582     }

589     private int depth = 0;
590     private List elementStack = new LinkedList();
591     private char[] buffer = new char[128];
592     private boolean startTagInProgress = false;
593     private final AttributesImpl attributeList = new AttributesImpl();
594 
595     public void startNode(String name) {
596         try {
597             if (this.depth != 0) {
598                 this.flushStartTag();
599             } else if (includeEnclosingDocument) {
600                 this.startDocument(false);
601             }
602             this.elementStack.add(0, escapeXmlName(name));
603 
604             this.startTagInProgress = true;
605             this.depth++;
606         } catch (SAXException e) {
607             throw new StreamException(e);
608         }
609     }
610 
611     public void addAttribute(String name, String value) {
612         if (this.startTagInProgress) {
613             String escapedName = escapeXmlName(name);
614             this.attributeList.addAttribute("", escapedName, escapedName, "CDATA", value);
615         } else {
616             throw new StreamException(new IllegalStateException("No startElement being processed"));
617         }
618     }
619 
620     public void setValue(String text) {
621         try {
622             this.flushStartTag();
623 
624             int lg = text.length();
625             if (lg > buffer.length) {
626                 buffer = new char[lg];
627             }
628             text.getChars(0, lg, buffer, 0);
629 
630             this.contentHandler.characters(buffer, 0, lg);
631         } catch (SAXException e) {
632             throw new StreamException(e);
633         }
634     }
635 
636     public void endNode() {
637         try {
638             this.flushStartTag();
639 
640             String tagName = (String) (this.elementStack.remove(0));
641 
642             this.contentHandler.endElement("", tagName, tagName);
643 
644             this.depth--;
645             if (this.depth == 0 && includeEnclosingDocument) {
646                 this.endDocument(false);
647             }
648         } catch (SAXException e) {
649             throw new StreamException(e);
650         }
651     }

653     /**
654      * Fires the SAX startDocument event towards the configured
655      * ContentHandler.
656      *
657      * @param multiObjectMode whether serialization of several
658      *                        object will be merge into a single
659      *                        SAX document.
660      * @throws SAXException if thrown by the ContentHandler.
661      */
662     private void startDocument(boolean multiObjectMode) throws SAXException {
663         if (this.depth == 0) {
664             // Notify contentHandler of document start.
665             this.contentHandler.startDocument();
666 
667             if (multiObjectMode) {
668                 // Prevent marshalling of each object to fire its own
669                 // start/endDocument events.
670                 this.depth++;
671             }
672         }
673     }
674 
675     /**
676      * Fires the SAX endDocument event towards the configured
677      * ContentHandler.
678      *
679      * @param multiObjectMode whether serialization of several
680      *                        object will be merge into a single
681      *                        SAX document.
682      * @throws SAXException if thrown by the ContentHandler.
683      */
684     private void endDocument(boolean multiObjectMode) throws SAXException {
685         if ((this.depth == 0) || ((this.depth == 1) && (multiObjectMode))) {
686             this.contentHandler.endDocument();
687             this.depth = 0;
688         }
689     }
690 
691     /**
692      * Fires any pending SAX startElement event towards the
693      * configured ContentHandler.
694      *
695      * @throws SAXException if thrown by the ContentHandler.
696      */
697     private void flushStartTag() throws SAXException {
698         if (this.startTagInProgress) {
699             String tagName = (String) (this.elementStack.get(0));
700 
701             this.contentHandler.startElement("", tagName,
702                     tagName, this.attributeList);
703             this.attributeList.clear();
704             this.startTagInProgress = false;
705         }
