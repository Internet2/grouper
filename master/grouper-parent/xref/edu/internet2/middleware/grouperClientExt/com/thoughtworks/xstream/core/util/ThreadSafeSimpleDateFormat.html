27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.core.util;
28  
29  import java.text.DateFormat;
30  import java.text.ParseException;
31  import java.text.SimpleDateFormat;
32  import java.util.Date;
33  import java.util.Locale;
34  import java.util.TimeZone;
35  
36  /**
37   * Wrapper around java.text.SimpleDateFormat that can
38   * be called by multiple threads concurrently.
39   * <p>SimpleDateFormat has a high overhead in creating
40   * and is not thread safe. To make best use of resources,
41   * the ThreadSafeSimpleDateFormat provides a dynamically
42   * sizing pool of instances, each of which will only
43   * be called by a single thread at a time.</p>
44   * <p>The pool has a maximum capacity, to limit overhead.
45   * If all instances in the pool are in use and another is
46   * required, it shall block until one becomes available.</p>
47   *
48   * @author Joe Walnes
49   * @author J&ouml;rg Schaible
50   */
51  public class ThreadSafeSimpleDateFormat {
52  
53      private final String formatString;
54      private final Pool pool;
55  
56      public ThreadSafeSimpleDateFormat(String format, int initialPoolSize, int maxPoolSize, final boolean lenient) {
57          formatString = format;
58          pool = new Pool(initialPoolSize, maxPoolSize, new Pool.Factory() {
59              public Object newInstance() {
60                  SimpleDateFormat dateFormat = new SimpleDateFormat(formatString, Locale.ENGLISH);
61                  dateFormat.setLenient(lenient);
62                  return dateFormat;
63              }
64              
65          });
66      }
67  
68      public String format(Date date) {
69          DateFormat format = fetchFromPool();
70          try {
71              return format.format(date);
72          } finally {
73              pool.putInPool(format);
74          }
75      }
76  
77      public Date parse(String date) throws ParseException {
78          DateFormat format = fetchFromPool();
79          try {
80              return format.parse(date);
81          } finally {
82              pool.putInPool(format);
83          }
84      }
85  
86      private DateFormat fetchFromPool() {
87          TimeZone tz = TimeZone.getDefault();
88          DateFormat format = (DateFormat)pool.fetchFromPool();
89          if (!tz.equals(format.getTimeZone())) {
90              format.setTimeZone(tz);
91          }
92          return format;
93      }
94  }
