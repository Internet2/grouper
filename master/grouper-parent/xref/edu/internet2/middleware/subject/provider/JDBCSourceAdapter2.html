137   public void checkConfig() {
138 
139     Properties props = this.getInitParams();
140     String error = "problem with sources.xml source id: " + this.getId() + ", ";
141 
142     //TODO encapsulate this stuff from the superclass into one method
143     //see if has jdbc in provider
144     if (this.jdbcConnectionProvider.requiresJdbcConfigInSourcesXml()) {
145 
146       String dbUrl = props.getProperty("dbUrl");
147       if (StringUtils.isBlank(dbUrl)) {
148         System.err.println("Subject API error: " + error + ", dbUrl param is required");
149         log.error(error + ", dbUrl param is required");
150         return;
151       }
152       String driver = SubjectApiUtils.convertUrlToDriverClassIfNeeded(dbUrl, props.getProperty("dbDriver"));
153       if (StringUtils.isBlank(driver)) {
154         System.err.println("Subject API error: " + error + ", driver param is required");
155         log.error(error + ", driver param is required");
156         return;
157       }
158       String dbUser = props.getProperty("dbUser");
159       if (StringUtils.isBlank(dbUser)) {
160         System.err.println("Subject API error: " + error + ", dbUser param is required");
161         log.error(error + ", dbUser param is required");
162         return;
163       }
164       String dbPwd = StringUtils.defaultString(props.getProperty("dbPwd"));
165       //      if (StringUtils.isBlank(dbPwd)) {
166       //        System.err.println("Subject API error: " + error + ", dbPwd param is required");
167       //        log.error(error + ", dbPwd param is required");
168       //        return;
169       //      }
170       dbPwd = Morph.decryptIfFile(dbPwd);
171 
172       try {
173 
174         Class driverClass = null;
175         try {
176           driverClass = Class.forName(driver);
177         } catch (Exception e) {
178           String theError = error
179               + "Error finding database driver class: "
180               + driver
181               + ", perhaps you did not put the database driver jar in the lib/custom dir or lib dir, "
182               + "or you have the wrong driver listed";
183           System.err.println("Subject API error: " + theError + ": "
184               + ExceptionUtils.getFullStackTrace(e));
185           log.error(theError, e);
186           return;
187         }
188 
189         //check out P6Spy
190         String spyInsert = "";
191         //dont load class here
192         if (driverClass.getName().equals("com.p6spy.engine.spy.P6SpyDriver")) {
193           spyInsert = " and spy.properties";
194           if (!SubjectCheckConfig.checkConfig("spy.properties")) {
195             return;
196           }
197           Properties spyProperties = SubjectUtils
198               .propertiesFromResourceName("spy.properties");
199           driver = spyProperties.getProperty("realdriver");
200           try {
201             driverClass = SubjectUtils.forName(driver);
202           } catch (Exception e) {
203             String theError = error
204                 + "Error finding database driver class from spy.properties: "
205                 + driver
206                 + ", perhaps you did not put the database driver jar in the lib/custom dir or lib dir, "
207                 + "or you have the wrong driver listed";
208             System.err.println("Subject API error: " + theError + ": "
209                 + ExceptionUtils.getFullStackTrace(e));
210             log.error(theError, e);
211             return;
212           }
213         }
214 
215         //lets make a db connection
216         Connection dbConnection = null;
217         try {
218           dbConnection = DriverManager.getConnection(dbUrl, dbUser, dbPwd);
219           @SuppressWarnings("unused")
220           String version = dbConnection.getMetaData().getDatabaseProductVersion();
221         } catch (SQLException sqlException) {
222           String theError = error
223               + "Error connecting to the database with credentials from sources.xml"
224               + spyInsert + ", url: " + dbUrl + ", driver: " + driver + ", user: "
225               + dbUser;
226           System.out.println("Subject API error: " + theError + ", "
227               + ExceptionUtils.getFullStackTrace(sqlException));
228           log.error(theError, sqlException);
229           return;
230         } finally {
231           SubjectUtils.closeQuietly(dbConnection);
232         }
233 
234       } catch (Exception e) {
235         String theError = error + "Error verifying sources.xml database configuration: ";
236         System.err.println("Subject API error: " + theError
237             + ExceptionUtils.getFullStackTrace(e));
238         log.error(theError, e);
239       }
240 
241     }
242 
243     //TODO encapsulate this in its own method
244     {
245       //    <param-name>dbTableOrView</param-name>
246       String dbTableOrView = props.getProperty("dbTableOrView");
247       if (StringUtils.isBlank(dbTableOrView)) {
248         String errorDetail = "dbTableOrView is required";
249         System.err.println(error + errorDetail);
250         log.error(error + errorDetail);
251         return;
252       }
253     }
254 
255     {
256       //    <param-name>subjectIdCol</param-name>
257       String subjectIdCol = props.getProperty("subjectIdCol");
258       if (StringUtils.isBlank(subjectIdCol)) {
259         String errorDetail = "subjectIdCol is required";
260         System.err.println(error + errorDetail);
261         log.error(error + errorDetail);
262         return;
263       }
264     }
265 
266     {
267       //    <param-name>nameCol</param-name>
268       String nameCol = props.getProperty("nameCol");
269       if (StringUtils.isBlank(nameCol)) {
270         String errorDetail = "nameCol is required";
271         System.err.println(error + errorDetail);
272         log.error(error + errorDetail);
273         return;
274       }
275     }
276 
277     {
278       //    <param-name>lowerSearchCol</param-name>
279       String lowerSearchCol = props.getProperty("lowerSearchCol");
280       if (StringUtils.isBlank(lowerSearchCol)) {
281         String errorDetail = "lowerSearchCol is required";
282         System.err.println(error + errorDetail);
283         log.error(error + errorDetail);
284         return;
285       }
286     }
287     {
288       String maxResultsString = props.getProperty("maxResults");
289       if (!StringUtils.isBlank(maxResultsString)) {
290         try {
291           this.maxResults = Integer.parseInt(maxResultsString);
292         } catch (NumberFormatException nfe) {
293           throw new SourceUnavailableException("Cant parse maxResults: " + maxResultsString, nfe);
294         }
295       }
296     }

300    * @see edu.internet2.middleware.subject.provider.BaseSourceAdapter#getSubject(java.lang.String, boolean, java.lang.String)
301    */
302   @Override
303   public Subject getSubject(String id1, boolean exceptionIfNull, String realm)
304       throws SubjectNotFoundException, SubjectNotUniqueException {
305     
306     boolean needsRealm = threadLocalRealm.get() == null;
307     if (needsRealm) {
308       threadLocalRealm.set(StringUtils.defaultString(realm));
309     }
310     try {
311       Map<String, Subject> subjectMap = getSubjectsByIds(SubjectApiUtils.toSet(id1));
312       
313       if (SubjectApiUtils.length(subjectMap) > 1) {
314         throw new RuntimeException("Why are there more than one result??? " + id1 + ", " + SubjectApiUtils.length(subjectMap));
315       }
316       
317       Subject subject = null;
318       
319       if (SubjectApiUtils.length(subjectMap) == 1) {
320         subject = subjectMap.values().iterator().next();
321       }
322       
323       if (subject == null && exceptionIfNull) {
324         throw new SubjectNotFoundException("Subject not found by id: " + id1);
325       }
326       return subject;
327     } finally {
328       if (needsRealm) {
329         threadLocalRealm.remove();
330       }
331     }
332   }

399   /**
400    * @see edu.internet2.middleware.subject.provider.BaseSourceAdapter#getSubjectsByIdentifiers(java.util.Collection, java.lang.String)
401    */
402   @Override
403   public Map<String, Subject> getSubjectsByIdentifiers(Collection<String> identifiers,
404       String realm) {
405 
406     boolean needsRealm = threadLocalRealm.get() == null;
407     if (needsRealm) {
408       threadLocalRealm.set(StringUtils.defaultString(realm));
409     }
410     try {
411       if (identifiers == null) {
412         return null;
413       }
414   
415       Map<String, Subject> results = new HashMap<String, Subject>();
416       
417       if (SubjectApiUtils.length(identifiers) > 0) {
418   
419         //if no identifier col, just get by id
420         if (this.subjectIdentifierCols.size() == 0) {
421           return this.getSubjectsByIds(identifiers);
422         }
423   
424         StringBuilder theSelectCols = new StringBuilder(StringUtils.join(this.selectCols.iterator(), ","));
425         
426         for (String subjectIdentifier : this.subjectIdentifierCols) {
427           
428           if (!this.selectCols.contains(subjectIdentifier)) {
429             theSelectCols.append(", ").append(subjectIdentifier);
430           }
431           
432         }
433   
434         int batchSize = 180 / this.subjectIdentifierCols.size();
435         int numberOfBatches = SubjectApiUtils.batchNumberOfBatches(identifiers, batchSize);
436         
437         List<String> identifiersList = new ArrayList<String>(identifiers);
438         
439         for (int i=0;i<numberOfBatches;i++) {
440           
441           List<String> identifiersBatch = SubjectApiUtils.batchList(identifiersList, batchSize, i);        
442   
443         
444           //we need the select cols, and the identifier cols so we can match the results with the identifiers
445           StringBuilder query = new StringBuilder("select "
446               + theSelectCols.toString() + " from "
447               + dbTableOrView() + " where ");
448     
449           List<String> args = new ArrayList<String>();
450     
451           int identifierIndex = 0;
452           for (String identifier : identifiersBatch) {
453             
454             if (identifierIndex > 0) {
455               
456               query.append(" or ");
457               
458             }
459             
460             query.append(" ( ");
461             
462             int index = 0;
463             for (String subjectIdentifierCol : this.subjectIdentifierCols) {
464               query.append(subjectIdentifierCol + " = ?");
465               if (index != this.subjectIdentifierCols.size() - 1) {
466                 query.append(" or ");
467               }
468               args.add(identifier);
469               index++;
470             }
471             
472             query.append(" ) ");
473             identifierIndex++;
474           }
475           
476           
477           this.search(query.toString(), args, false, false, false, null, identifiersBatch, results);
478           
479         }      
480         
481       }
482       return results;
483     } finally {
484       if (needsRealm) {
485         threadLocalRealm.remove();
486       }
487     }
488   }

490   /**
491    * @see edu.internet2.middleware.subject.provider.BaseSourceAdapter#getSubjectsByIds(java.util.Collection, java.lang.String)
492    */
493   @Override
494   public Map<String, Subject> getSubjectsByIds(Collection<String> ids, String realm) {
495 
496     boolean needsRealm = threadLocalRealm.get() == null;
497     if (needsRealm) {
498       threadLocalRealm.set(StringUtils.defaultString(realm));
499     }
500     try {
501   
502       if (ids == null) {
503         return null;
504       }
505       
506       Map<String, Subject> results = new HashMap<String, Subject>();
507       
508       if (ids.size() > 0) {
509         
510         int batchSize = 180;
511         int numberOfBatches = SubjectApiUtils.batchNumberOfBatches(ids, batchSize);
512         
513         List<String> idsList = new ArrayList<String>(ids);
514         
515         for (int i=0;i<numberOfBatches;i++) {
516   
517           List<String> idsBatch = SubjectApiUtils.batchList(idsList, batchSize, i);        
518   
519           String query = "select " + StringUtils.join(this.selectCols.iterator(), ",")
520             + " from " + dbTableOrView() + " where " + this.subjectIdCol + " in ("
521             + SubjectApiUtils.convertToInClauseForSqlStatic(idsBatch) + ")";
522       
523           List<String> args = new ArrayList<String>(idsBatch);
524     
525           Set<Subject> subjects = this.search(query.toString(), args, false, false, false, null, null, null);
526           
527           for (Subject subject : SubjectApiUtils.nonNull(subjects)) {
528             results.put(subject.getId(), subject);
529           }
530         }
531       }
532       return results;
533     } finally {
534       if (needsRealm) {
535         threadLocalRealm.remove();
536       }
537     }
538   }

604   /**
605    * DataSource connection pool setup.
606    * @param props 
607    * @throws SourceUnavailableException
608    */
609   @SuppressWarnings("unchecked")
610   protected void setupDataSource(Properties props) throws SourceUnavailableException {
611 
612     //TODO encapsulate generic connection stuff like this into one method
613     String driver = props.getProperty("dbDriver");
614 
615     //default is 2
616     String maxActiveString = props.getProperty("maxActive");
617     Integer maxActive = StringUtils.isBlank(maxActiveString) ? null : Integer
618         .parseInt(maxActiveString);
619 
620     //default is 2
621     String maxIdleString = props.getProperty("maxIdle");
622     Integer maxIdle = StringUtils.isBlank(maxIdleString) ? null : Integer
623         .parseInt(maxIdleString);
624 
625     //default is 5
626     String maxWaitString = props.getProperty("maxWait");
627     Integer maxWaitSeconds = StringUtils.isBlank(maxWaitString) ? null : Integer
628         .parseInt(maxWaitString);
629 
630     String dbUrl = null;
631     log.debug("Initializing connection factory.");
632     dbUrl = props.getProperty("dbUrl");
633     String dbUser = props.getProperty("dbUser");
634     String dbPwd = props.getProperty("dbPwd");
635     dbPwd = Morph.decryptIfFile(dbPwd);
636 
637     //defaults to true
638     Boolean readOnly = SubjectUtils.booleanObjectValue(props.getProperty("readOnly"));
639 
640     String jdbcConnectionProviderString = SubjectUtils.defaultIfBlank(props
641         .getProperty("jdbcConnectionProvider"), C3p0JdbcConnectionProvider.class
642         .getName());
643     Class<JdbcConnectionProvider> jdbcConnectionProviderClass = null;
644     try {
645       jdbcConnectionProviderClass = SubjectUtils.forName(jdbcConnectionProviderString);
646     } catch (RuntimeException re) {
647       SubjectUtils
648           .injectInException(
649               re,
650               "Valid built-in options are: "
651                   + C3p0JdbcConnectionProvider.class.getName()
652                   + " (default) [note: its a zero, not a capital O], "
653                   // + DbcpJdbcConnectionProvider.class.getName()
654                   + ", edu.internet2.middleware.grouper.subj.GrouperJdbcConnectionProvider (if using Grouper).  "
655                   + "Note, these are the built-ins for the Subject API or Grouper, there might be other valid choices.");
656       throw re;
657     }
658 
659     this.jdbcConnectionProvider = SubjectUtils.newInstance(jdbcConnectionProviderClass);
660     this.jdbcConnectionProvider.init(props, this.getId(), driver, maxActive, 2, maxIdle, 2,
661         maxWaitSeconds, 5, dbUrl, dbUser, dbPwd, readOnly, true);
662 
663     log.info("Data Source initialized.");
664 
665     //    <param-name>dbTableOrView</param-name>
666     this.dbTableOrView = props.getProperty("dbTableOrView");
667     if (StringUtils.isBlank(this.dbTableOrView)) {
668       throw new SourceUnavailableException("dbTableOrView not defined, source: "
669           + this.getId());
670     }
671 
672     //    <param-name>subjectIdCol</param-name>
673     this.subjectIdCol = props.getProperty("subjectIdCol");
674     if (StringUtils.isBlank(this.subjectIdCol)) {
675       throw new SourceUnavailableException("subjectIdCol not defined, source: "
676           + this.getId());
677     }
678     this.selectCols.add(this.subjectIdCol);
679 
680     //    <param-name>nameCol</param-name>
681     this.nameCol = props.getProperty("nameCol");
682     if (StringUtils.isBlank(this.nameCol)) {
683       throw new SourceUnavailableException("nameCol not defined, source: " + this.getId());
684     }
685     this.selectCols.add(this.nameCol);
686 
687     //    <param-name>descriptionCol</param-name>
688     this.descriptionCol = props.getProperty("descriptionCol");
689     if (!StringUtils.isBlank(this.descriptionCol)) {
690       this.selectCols.add(this.descriptionCol);
691     }
692 
693     //    <param-name>lowerSearchCol</param-name>

718         this.selectCols.add(subjectIdentifierCol);
719

721       }
722     }
723 
724     {
725       int index = 0;
726       while (true) {

741

823   /**
824    * @param searchValue 
825    * @param firstPageOnly 
826    * @return result
827    * 
828    */
829   private SearchPageResult searchHelper(String searchValue, boolean firstPageOnly) {
830 
831     SubjectStatusResult subjectStatusResult = null;
832 
833     {
834       //see if we are doing status
835       SubjectStatusProcessor subjectStatusProcessor = new SubjectStatusProcessor(searchValue, this.getSubjectStatusConfig());
836       subjectStatusResult = subjectStatusProcessor.processSearch();
837 
838       //strip out status parts
839       searchValue = subjectStatusResult.getStrippedQuery();
840     }      
841     
842     Set<Subject> results = new LinkedHashSet<Subject>();
843     boolean tooManyResults = false;
844 
845     //if there is nothing, what are we searching on?
846     if (StringUtils.isBlank(searchValue)) {
847       throw new RuntimeException("You need to supply a search value: '" + searchValue
848           + "'");
849     }
850 
851     //lets split by any whitespace space
852     String[] terms = searchValue.split("\\s+");
853 
854     StringBuilder query = new StringBuilder("select "
855         + StringUtils.join(this.selectCols.iterator(), ",") + " from "
856         + dbTableOrView() + " where ");
857 
858     List<String> args = new ArrayList<String>();
859 
860     boolean addedArg = false;
861     for (int i = 0; i < terms.length; i++) {
862       addedArg = true;
863       query.append(this.lowerSearchCol + " like ?");
864       if (i != terms.length - 1) {
865         query.append(" and ");
866       }
867       args.add("%" + terms[i].toLowerCase() + "%");
868     }
869     
870     //add status?
871     if (!subjectStatusResult.isAll() && !StringUtils.isBlank(subjectStatusResult.getDatastoreFieldName())) {
872       
873       if (addedArg) {
874         query.append(" and ");
875       }
876       addedArg = true;
877       query.append(" " + subjectStatusResult.getDatastoreFieldName() + " " + (subjectStatusResult.isEquals()?"=":"<>") + " ? ");
878       args.add(subjectStatusResult.getDatastoreValue());
879     }
880 
881     
882     if (!StringUtils.isBlank(this.defaultSortCol)) {
883       query.append(" order by ").append(this.defaultSortCol);
884     }
885     
886     String throwErrorOnFindAllFailureString = this.getInitParam("throwErrorOnFindAllFailure");
887     boolean throwErrorOnFindAllFailure = SubjectUtils.booleanValue(throwErrorOnFindAllFailureString, true);
888 
889     try {
890       boolean[] tooManyResultHelper = new boolean[]{false};
891       results = this.search(query.toString(), args, false, true, firstPageOnly, tooManyResultHelper, null, null);
892       tooManyResults = tooManyResultHelper[0];
893     } catch (Exception ex) {
894       if (ex instanceof SubjectTooManyResults) {
895         throw (SubjectTooManyResults)ex;
896       }
897       if (!throwErrorOnFindAllFailure) {
898         log.error(ex.getMessage() + ", source: " + this.getId() + ", searchValue: "
899           + searchValue, ex);
900       } else {
901         throw new SourceUnavailableException(ex.getMessage() + ", source: " + this.getId() + ", searchValue: "
902             + searchValue, ex);
903       }
904     }
905 
906     return new SearchPageResult(tooManyResults, results);
907 
908   }

922   /**
923    * Perform a search for subjects
924    * 
925    * @param query is query to run, prepared statement args should be question marks
926    * @param args are the prepared statement args
927    * @param expectSingle true if expecting one answer
928    * @param exceptionIfNull 
929    * @param firstPageOnly if we should only get first page
930    * @param tooManyResults flag to return for too many results
931    * @param identifiersForIdentifierToMap optional, if we want the resultIdentifierMap back, then pass in the identifiers
932    * @param resultIdentifierToSubject optional, if we want the resultIdentifierMap back, then pass in the map,
933    * and this will populate it
934    * @return subjects or empty set if none or null if expect one and no results and not exception on null
935    * @throws SubjectNotFoundException if expecting one and not found
936    * @throws SubjectNotUniqueException
937    * @throws InvalidQueryRuntimeException 
938    */
939   private Set<Subject> search(String query, List<String> args, boolean expectSingle, 
940       boolean exceptionIfNull, boolean firstPageOnly, boolean[] tooManyResults, 
941       Collection<String> identifiersForIdentifierToMap, Map<String, Subject> resultIdentifierToSubject)
942       throws SubjectNotFoundException, SubjectNotUniqueException,
943       InvalidQueryRuntimeException {
944 
945     //no npes
946     if (tooManyResults == null || tooManyResults.length == 0) {
947       tooManyResults = new boolean[1];
948     }
949     
950     if (resultIdentifierToSubject != null) {
951       if (SubjectApiUtils.length(identifiersForIdentifierToMap) == 0) {
952         throw new RuntimeException("Why is there no identifiersForIdentifierToMap???");
953       }
954     }
955 
956     Connection conn = null;
957     PreparedStatement stmt = null;
958     JdbcConnectionBean jdbcConnectionBean = null;
959     Set<Subject> results = new LinkedHashSet<Subject>();
960 
961     try {
962       jdbcConnectionBean = this.jdbcConnectionProvider.connectionBean();
963       conn = jdbcConnectionBean.connection();
964       
965       Integer resultSetLimit = resultSetLimit(firstPageOnly, this.getMaxPage(), this.maxResults);
966       
967       if (resultSetLimit != null && this.isChangeSearchQueryForMaxResults()) {
968         query = tryToChangeQuery(query, conn, resultSetLimit);
969       }
970       
971       stmt = conn.prepareStatement(query);
972       ResultSet rs = null;
973 
974       for (int i = 0; i < SubjectUtils.length(args); i++) {
975 
976         String arg = args.get(i);
977 
978         try {
979           stmt.setString(i + 1, arg);
980         } catch (SQLException e) {
981           SubjectUtils.injectInException(e, "Error setting param: " + i
982               + " (zero indexed) in source: " + this.getId() + ", in query: " + query
983               + ", " + e.getMessage());
984           throw e;
985         }
986       }
987 
988       rs = stmt.executeQuery();
989 
990       while (rs.next()) {
991 
992         Subject subject = createSubject(rs, query, identifiersForIdentifierToMap, resultIdentifierToSubject);
993         results.add(subject);
994 
995         //if we are at the end of the page
996         if (firstPageOnly && this.getMaxPage() != null && results.size() >= this.getMaxPage()) {
997           tooManyResults[0] = true;
998           break;
999         }
1000 
1001         if (this.maxResults != null && results.size() > this.maxResults) {
1002           throw new SubjectTooManyResults(
1003               "More results than allowed: " + this.maxResults 
1004               + " for search '" + query + "'");
1005         }
1006         
1007 
1008       }
1009 
1010       jdbcConnectionBean.doneWithConnection();
1011     } catch (SQLException ex) {
1012       String error = "problem in sources.xml source: " + this.getId() + ", sql: " + query;
1013       try {
1014         jdbcConnectionBean.doneWithConnectionError(ex);
1015       } catch (RuntimeException e) {
1016         log.error(error, e);
1017       }
1018       throw new SourceUnavailableException(error, ex);
1019     } finally {
1020       
1021       if (log.isDebugEnabled()) {
1022         log.debug("Query returned " + results.size() + ", " + query + ", " + SubjectUtils.toStringForLog(args));
1023       }
1024       
1025       closeStatement(stmt);
1026       if (jdbcConnectionBean != null) {
1027         jdbcConnectionBean.doneWithConnectionFinally();
1028       }
1029     }
1030 
1031     if (expectSingle) {
1032       if (results.size() > 1) {
1033         throw new SubjectNotUniqueException("Multiple subjects exist: " + query + ", "
1034             + StringUtils.join(args.iterator(), ","));
1035       }
1036       if (results.size() == 0) {
1037         if (exceptionIfNull) {
1038           throw new SubjectNotFoundException("Subject not found: " + query + ", "
1039               + StringUtils.join(args.iterator(), ","));
1040         }
1041         results = null;
1042       }
1043     }
1044 
1045     return results;
1046   }

1048   /**
1049    * Create a subject from the current row in the resultSet
1050    * 
1051    * @param resultSet
1052    * @param query 
1053    * @param identifiersForIdentifierToMap
1054    * @param resultIdentifierToSubject
1055    * @return subject
1056    * @throws SQLException 
1057    */
1058   private Subject createSubject(ResultSet resultSet, String query, 
1059       Collection<String> identifiersForIdentifierToMap, Map<String, Subject> resultIdentifierToSubject) throws SQLException {
1060 
1061     String name = "";
1062     String subjectID = "";
1063     String description = "";
1064     SubjectImpl subject = null;
1065     //lets do this through metadata so caps dont matter
1066     ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
1067 
1068     subjectID = retrieveString(resultSet, this.subjectIdCol, "subjectIdCol", query,
1069         resultSetMetaData);
1070     name = retrieveString(resultSet, this.nameCol, "nameCol", query, resultSetMetaData);
1071     if (!StringUtils.isBlank(this.descriptionCol)) {
1072       description = retrieveString(resultSet, this.descriptionCol, "descriptionCol",
1073           query, resultSetMetaData);
1074     }
1075     Map attributes = loadAttributes(resultSet, query, resultSetMetaData);
1076     subject = new SubjectImpl(subjectID, name, description, this.getSubjectType().getName(), this.getId(),
1077         attributes);
1078     
1079     if (resultIdentifierToSubject != null) {
1080       boolean foundValue = false;
1081       
1082       if (identifiersForIdentifierToMap.contains(subject.getId())) {
1083         resultIdentifierToSubject.put(subject.getId(), subject);
1084         foundValue = true;
1085       } else {
1086       
1087         for (String identifierCol : this.subjectIdentifierCols) {
1088           
1089           String identifierValue = retrieveString(resultSet, identifierCol, identifierCol, query, resultSetMetaData);
1090           if (!StringUtils.isBlank(identifierValue)) {
1091             
1092             if (identifiersForIdentifierToMap.contains(identifierValue)) {
1093               resultIdentifierToSubject.put(identifierValue, subject);
1094               foundValue = true;
1095               break;
1096             }
1097             
1098           }
1099         }
1100         
1101       }
1102       if (!foundValue) {
1103         throw new RuntimeException("Why did a query by identifier return a subject " +
1104         		"which cant be found by identifier??? " + SubjectApiUtils.subjectToString(subject)
1105         		+ ", " + query);
1106       }
1107     }
1108     
1109     return subject;
1110   }

1141    * retrieve a param from resultset
1142    * @param resultSet
1143    * @param name
1144    * @param varName 
1145    * @param query
1146    * @param resultSetMetaData
1147    * @return the value
1148    * @throws SQLException 
1149    */
1150   private String retrieveString(ResultSet resultSet, String name, String varName,
1151       String query, ResultSetMetaData resultSetMetaData) throws SQLException {
1152     try {
1153       int columnCount = resultSetMetaData.getColumnCount();
1154 
1155       //this will be case-insensitive
1156       for (int i = 0; i < columnCount; i++) {
1157         if (StringUtils.equalsIgnoreCase(name, resultSetMetaData.getColumnName(i + 1))
1158             || StringUtils
1159                 .equalsIgnoreCase(name, resultSetMetaData.getColumnLabel(i + 1))) {
1160           return resultSet.getString(i + 1);
1161         }
1162       }
1163       String result = resultSet.getString(name);
1164       return result;
1165     } catch (SQLException se) {
1166       SubjectUtils.injectInException(se, "Error retrieving column name: '" + name
1167           + "' in source: " + this.getId() + ", in query: " + query + ", "
1168           + se.getMessage() + ", maybe the column configured in " + varName
1169           + " does not exist as a query column");
1170       throw se;

1172     }
