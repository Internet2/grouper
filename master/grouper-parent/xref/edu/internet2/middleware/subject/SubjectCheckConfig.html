20  package edu.internet2.middleware.subject;
21  
22  import java.net.URL;
23  import java.util.Collection;
24  
25  import org.apache.commons.lang.exception.ExceptionUtils;
26  import org.apache.commons.logging.Log;
27  import org.apache.commons.logging.LogFactory;
28  
29  import edu.internet2.middleware.subject.provider.SourceManager;
30  
31  /**
32   * make sure the subject.xml config is correct
33   */
34  public class SubjectCheckConfig {
35  
36    /** */
37    public static final String GROUPER_TEST_SUBJECT_BY_ID = "grouperTestSubjectByIdOnStartupASDFGHJ";
38  
39    /** */
40    public static final String SUBJECT_ID_TO_FIND_ON_CHECK_CONFIG = "subjectIdToFindOnCheckConfig";
41  
42    /** */
43    public static final String FIND_SUBJECT_BY_ID_ON_CHECK_CONFIG = "findSubjectByIdOnCheckConfig";
44    
45    /** logger */
46    private static Log log = LogFactory.getLog(SubjectCheckConfig.class);
47  
48    /**
49     * make surce classpath file is there
50     * @param resourceName
51     * @return true if ok
52     */
53    public static boolean checkConfig(String resourceName) {
54      boolean foundResource = false;
55      //first, there must be a sources.xml
56      try {
57        //get the url of the navigation file
58        ClassLoader cl = SubjectCheckConfig.class.getClassLoader();
59  
60        URL url = cl.getResource(resourceName);
61  
62        foundResource = url != null;
63      } catch (Exception e) {
64        //this means it cant be found
65        log.info("Exception looking for " + resourceName, e);
66      }
67      if (!foundResource) {
68        String error = "Cant find required resource on classpath: " + resourceName;
69        //this is serious, lets go out and error
70        System.err.println("Subject API error: " + error);
71        log.error(error);
72      }
73      return foundResource;
74    }
75    
76    /**
77     * check the subject config
78     */
79    public static void checkConfig() {
80      
81  //    if (!checkConfig("sources.xml")) {
82  //      return;
83  //    }
84      
85      //at this point, we have a sources.xml...  now check it out
86      Collection<Source> sources = null;
87      
88      try {
89        sources = SourceManager.getInstance().getSources();
90      } catch (Exception e) {
91        String error = "problem initting sources from sources.xml";
92        System.err.println("Subject API error: " + error + ", " + ExceptionUtils.getFullStackTrace(e));
93        log.error(error, e);
94        return;
95      }
96      int sourceCount = 0;
97      for (Source source: sources) {
98        
99        sourceCount++;
100       String error = "error with subject source id: " + source.getId() + ", name: " + source.getName()
101         + ", ";
102       try {
103         source.checkConfig();
104         
105         String findSubjectOnCheckConfigString = source.getInitParam(FIND_SUBJECT_BY_ID_ON_CHECK_CONFIG);
106         boolean findSubjectOnCheckConfig = SubjectUtils.booleanValue(findSubjectOnCheckConfigString, true);
107         
108         if (findSubjectOnCheckConfig) {
109           String subjectToFindOnCheckConfig = source.getInitParam(SUBJECT_ID_TO_FIND_ON_CHECK_CONFIG);
110           subjectToFindOnCheckConfig = SubjectUtils.defaultIfBlank(subjectToFindOnCheckConfig, GROUPER_TEST_SUBJECT_BY_ID);
111           source.getSubject(subjectToFindOnCheckConfig, false);
112         }
113 
114       } catch (Exception e) {
115         String theError = error + "problem with getSubject by id, in sources.xml: search searchSubject: ";
116         System.err.println("Subject API error: " + theError + ", " + ExceptionUtils.getFullStackTrace(e));
117         log.error(theError, e);
118         continue;
119       }
120       
121       try {
122         
123         String findSubjectOnCheckConfigString = source.getInitParam("findSubjectByIdentifiedOnCheckConfig");
124         boolean findSubjectOnCheckConfig = SubjectUtils.booleanValue(findSubjectOnCheckConfigString, true);
125         
126         if (findSubjectOnCheckConfig) {
127           String subjectIdentifierToFindOnCheckConfig = source.getInitParam("subjectIdentifierToFindOnCheckConfig");
128           subjectIdentifierToFindOnCheckConfig = SubjectUtils.defaultIfBlank(subjectIdentifierToFindOnCheckConfig, "grouperTestSubjectByIdentifierOnStartupASDFGHJ");
129           source.getSubjectByIdentifier(subjectIdentifierToFindOnCheckConfig, false);
130         }
131 
132       } catch (SubjectNotFoundException snfe) {
133         //good!
134       } catch (Exception e) {
135         String theError = error + "problem with getSubject by identifier, in sources.xml: serachType searchSubjectByIdentifier: ";
136         System.err.println("Subject API error: " + theError + ", " + ExceptionUtils.getFullStackTrace(e));
137         log.error(theError, e);
138         continue;
139       }
140     
141       try {
142         
143         String findSubjectOnCheckConfigString = source.getInitParam("findSubjectByStringOnCheckConfig");
144         boolean findSubjectOnCheckConfig = SubjectUtils.booleanValue(findSubjectOnCheckConfigString, true);
145         
146         if (findSubjectOnCheckConfig) {
147           String stringToFindOnCheckConfig = source.getInitParam("stringToFindOnCheckConfig");
148           stringToFindOnCheckConfig = SubjectUtils.defaultIfBlank(stringToFindOnCheckConfig, "grouperTestStringOnStartupASDFGHJ");
149           source.search(stringToFindOnCheckConfig);
150         }
151         
152       } catch (Exception e) {
153         String theError = error + "problem with search, in sources.xml: serachType search: ";
154         System.err.println("Subject API error: " + theError + ", " + ExceptionUtils.getFullStackTrace(e));
155         log.error(theError, e);
156         continue;
157       }
158     }
159     if (sourceCount == 0) {
160       System.err.println("Subject API warning: there are no sources available from sources.xml");
161       log.warn("there are no sources available from sources.xml");
162     }
163 
164   }
