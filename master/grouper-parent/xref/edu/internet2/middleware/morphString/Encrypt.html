39      //allow this in case the other way messes up for some reason

44        MorphPropertyFileUtils.retrieveProperties();
45      } catch (Exception e) {
46        //probably cant find them, prompt to find config file
47        System.out.print("Enter the location of morphString.properties: ");
48        
49        //  open up standard input

51  
52        //  read the username from the command-line; need to use try/catch with the 
53        //  readLine() method 
54        String configLocation = null;
55        try { 
56          configLocation = br.readLine();

58           System.out.println("IO error trying to read config file location! " + MorphStringUtils.getFullStackTrace(ioe));

60        }

62        
63      }
64      
65      encryptInput(dontMask);
66  
67    }

69    /**
70     * encrypt user input, perhaps dont mask
71     * @param dontMask
72     */
73    public static void encryptInput(boolean dontMask) {
74      String passwordString = null;
75      String prompt = "Type the string to encrypt (note: pasting might echo it back): ";
76  
77      if (dontMask) {
78  
79        System.out.print(prompt);
80        //  open up standard input 
81        BufferedReader br = new BufferedReader(new InputStreamReader(System.in)); 
82  
83        //  read the username from the command-line; need to use try/catch with the 
84        //  readLine() method 
85        try { 
86           passwordString = br.readLine(); 
87        } catch (IOException ioe) { 
88           System.out.println("IO error! " + MorphStringUtils.getFullStackTrace(ioe));
89           System.exit(1); 
90        } 
91  
92      } else {
93        char password[] = null;
94        try {
95          password = password(System.in, prompt);
96        } catch (IOException ioe) {
97          ioe.printStackTrace();
98        }
99        passwordString = String.valueOf(password);
100     }
101     if (MorphStringUtils.isBlank(passwordString)) {
102       System.out.println("No input entered");
103     } else {
104       System.out.println("The encrypted string is: " + Morph.encrypt(passwordString));
105     }
106     
107   }

109   /**
110    * @param in stream to be used (e.g. System.in)
111    *@param prompt The prompt to display to the user.
112    *@return The password as entered by the user.
113    * @throws IOException 
114    */
115 
116   public static final char[] password(InputStream in, String prompt) throws IOException {
117     MaskingThread maskingthread = new MaskingThread(prompt);
118 
119     Thread thread = new Thread(maskingthread);
120     thread.start();
121 
122     char[] lineBuffer;
123     char[] buf;
124 
125     buf = lineBuffer = new char[128];
126 
127     int room = buf.length;
128     int offset = 0;
129     int c;
130 
131     loop: while (true) {
132       switch (c = in.read()) {
133         case -1:
134         case '\n':
135           break loop;
136 
137         case '\r':
138           int c2 = in.read();
139           if ((c2 != '\n') && (c2 != -1)) {
140             if (!(in instanceof PushbackInputStream)) {
141               in = new PushbackInputStream(in);
142             }
143             ((PushbackInputStream) in).unread(c2);
144           } else {
145             break loop;
146           }
147 
148         default:
149           if (--room < 0) {
150             buf = new char[offset + 128];
151             room = buf.length - offset - 1;
152             System.arraycopy(lineBuffer, 0, buf, 0, offset);
153             Arrays.fill(lineBuffer, ' ');
154             lineBuffer = buf;
155           }
156           buf[offset++] = (char) c;
157           break;
158       }
159     }
160     maskingthread.stopMasking();
161     if (offset == 0) {
162       return null;
163     }
164     char[] ret = new char[offset];
165     System.arraycopy(buf, 0, ret, 0, offset);
166     Arrays.fill(buf, ' ');
167     return ret;
168   }

170   /**
171    * thread to mask input
172    */
173   static class MaskingThread extends Thread {
174 
175     /** stop */
176     private volatile boolean stop;
177 
178     /** echo char, this doesnt work correctly, so make a space so people dont notice...  
179      * prints out too many */
180     private char echochar = ' ';
181 
182     /**
183      *@param prompt The prompt displayed to the user
184      */
185     public MaskingThread(String prompt) {
186       System.out.print(prompt);
187     }
188 
189     /**
190      * Begin masking until asked to stop.
191      */
192     @Override
193     public void run() {
194 
195       int priority = Thread.currentThread().getPriority();
196       Thread.currentThread().setPriority(Thread.MAX_PRIORITY);
197 
198       try {
199         this.stop = true;
200         while (this.stop) {
201           System.out.print("\010" + this.echochar);
202           try {
203             // attempt masking at this rate
204             Thread.sleep(1);
205           } catch (InterruptedException iex) {
206             Thread.currentThread().interrupt();
207             return;
208           }
209         }
210       } finally { // restore the original priority
211         Thread.currentThread().setPriority(priority);
212       }
213     }
214 
215     /**
216      * Instruct the thread to stop masking.
217      */
218     public void stopMasking() {
219       this.stop = false;
220     }
221   }
