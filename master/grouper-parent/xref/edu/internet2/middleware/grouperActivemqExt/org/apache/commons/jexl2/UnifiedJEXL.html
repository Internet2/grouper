51  /**
52   * An evaluator similar to the Unified EL evaluator used in JSP/JSF based on JEXL.
53   * It is intended to be used in configuration modules, XML based frameworks or JSP taglibs
54   * and facilitate the implementation of expression evaluation.
55   * <p>
56   * An expression can mix immediate, deferred and nested sub-expressions as well as string constants;
57   * <ul>
58   * <li>The "immediate" syntax is of the form <code>"...${jexl-expr}..."</code></li>
59   * <li>The "deferred" syntax is of the form <code>"...#{jexl-expr}..."</code></li>
60   * <li>The "nested" syntax is of the form <code>"...#{...${jexl-expr0}...}..."</code></li>
61   * <li>The "composite" syntax is of the form <code>"...${jexl-expr0}... #{jexl-expr1}..."</code></li>
62   * </ul>
63   * </p>
64   * <p>
65   * Deferred & immediate expression carry different intentions:
66   * <ul>
67   * <li>An immediate expression indicate that evaluation is intended to be performed close to
68   * the definition/parsing point.</li>
69   * <li>A deferred expression indicate that evaluation is intended to occur at a later stage.</li>
70   * </ul>
71   * </p>
72   * <p>
73   * For instance: <code>"Hello ${name}, now is #{time}"</code> is a composite "deferred" expression since one
74   * of its subexpressions is deferred. Furthermore, this (composite) expression intent is
75   * to perform two evaluations; one close to its definition and another one in a later
76   * phase.
77   * </p>
78   * <p>
79   * The API reflects this feature in 2 methods, prepare and evaluate. The prepare method
80   * will evaluate the immediate subexpression and return an expression that contains only
81   * the deferred subexpressions (& constants), a prepared expression. Such a prepared expression
82   * is suitable for a later phase evaluation that may occur with a different JexlContext.
83   * Note that it is valid to call evaluate without prepare in which case the same JexlContext
84   * is used for the 2 evaluation phases.
85   * </p>
86   * <p>
87   * In the most common use-case where deferred expressions are to be kept around as properties of objects,
88   * one should parse & prepare an expression before storing it and evaluate it each time
89   * the property storing it is accessed.
90   * </p>
91   * <p>
92   * Note that nested expression use the JEXL syntax as in:
93   * <code>"#{${bar}+'.charAt(2)'}"</code>
94   * The most common mistake leading to an invalid expression being the following:
95   * <code>"#{${bar}charAt(2)}"</code>
96   * </p>
97   * <p>Also note that methods that parse evaluate expressions may throw <em>unchecked</em> exceptions;
98   * The {@link UnifiedJEXL.Exception} are thrown when the engine instance is in "non-silent" mode
99   * but since these are RuntimeException, user-code <em>should</em> catch them where appropriate.
100  * </p>
101  * @since 2.0
102  */

133     /**
134      * Types of expressions.
135      * Each instance carries a counter index per (composite sub-) expression type.
136      * @see ExpressionBuilder
137      */
138     private static enum ExpressionType {
139         /** Constant expression, count index 0. */
140         CONSTANT(0),
141         /** Immediate expression, count index 1. */
142         IMMEDIATE(1),
143         /** Deferred expression, count index 2. */
144         DEFERRED(2),
145         /** Nested (which are deferred) expressions, count index 2. */
146         NESTED(2),
147         /** Composite expressions are not counted, index -1. */
148         COMPOSITE(-1);
149         /** The index in arrays of expression counters for composite expressions. */
150         private final int index;
151 
152         /**
153          * Creates an ExpressionType.
154          * @param idx the index for this type in counters arrays.
155          */
156         ExpressionType(int idx) {
157             this.index = idx;
158         }
159     }

161     /**
162      * A helper class to build expressions.
163      * Keeps count of sub-expressions by type.
164      */
165     private static class ExpressionBuilder {
166         /** Per expression type counters. */
167         private final int[] counts;
168         /** The list of expressions. */
169         private final ArrayList<Expression> expressions;
170 
171         /**
172          * Creates a builder.
173          * @param size the initial expression array size
174          */
175         ExpressionBuilder(int size) {
176             counts = new int[]{0, 0, 0};
177             expressions = new ArrayList<Expression>(size <= 0 ? 3 : size);
178         }
179 
180         /**
181          * Adds an expression to the list of expressions, maintain per-type counts.
182          * @param expr the expression to add
183          */
184         void add(Expression expr) {
185             counts[expr.getType().index] += 1;
186             expressions.add(expr);
187         }
188 
189         /**
190          * Builds an expression from a source, performs checks.
191          * @param el the unified el instance
192          * @param source the source expression
193          * @return an expression
194          */
195         Expression build(UnifiedJEXL el, Expression source) {
196             int sum = 0;
197             for (int count : counts) {
198                 sum += count;
199             }
200             if (expressions.size() != sum) {
201                 StringBuilder error = new StringBuilder("parsing algorithm error, exprs: ");
202                 error.append(expressions.size());
203                 error.append(", constant:");
204                 error.append(counts[ExpressionType.CONSTANT.index]);
205                 error.append(", immediate:");
206                 error.append(counts[ExpressionType.IMMEDIATE.index]);
207                 error.append(", deferred:");
208                 error.append(counts[ExpressionType.DEFERRED.index]);
209                 throw new IllegalStateException(error.toString());
210             }
211             // if only one sub-expr, no need to create a composite
212             if (expressions.size() == 1) {
213                 return expressions.get(0);
214             } else {
215                 return el.new CompositeExpression(counts, expressions, source);
216             }
217         }
218     }

255     /**
256      * The abstract base class for all expressions, immediate '${...}' and deferred '#{...}'.
257      */
258     public abstract class Expression {
259         /** The source of this expression (see {@link UnifiedJEXL.Expression#prepare}). */
260         protected final Expression source;
261 
262         /**
263          * Creates an expression.
264          * @param src the source expression if any
265          */
266         Expression(Expression src) {
267             this.source = src != null ? src : this;
268         }
269 
270         /**
271          * Checks whether this expression is immediate.
272          * @return true if immediate, false otherwise
273          */
274         public boolean isImmediate() {
275             return true;
276         }
277 
278         /**
279          * Checks whether this expression is deferred.
280          * @return true if deferred, false otherwise
281          */
282         public final boolean isDeferred() {
283             return !isImmediate();
284         }
285 
286         /**
287          * Gets this expression type.
288          * @return its type
289          */
290         abstract ExpressionType getType();
291 
292         /**
293          * Formats this expression, adding its source string representation in
294          * comments if available: 'expression /*= source *\/'' .
295          * <b>Note:</b> do not override; will be made final in a future release.
296          * @return the formatted expression string
297          */
298         @Override
299         public String toString() {
300             StringBuilder strb = new StringBuilder();
301             asString(strb);
302             if (source != this) {
303                 strb.append(" /*= ");
304                 strb.append(source.toString());
305                 strb.append(" */");
306             }
307             return strb.toString();
308         }
309 
310         /**
311          * Generates this expression's string representation.
312          * @return the string representation
313          */
314         public String asString() {
315             StringBuilder strb = new StringBuilder();
316             asString(strb);
317             return strb.toString();
318         }
319 
320         /**
321          * Adds this expression's string representation to a StringBuilder.
322          * @param strb the builder to fill
323          * @return the builder argument
324          */
325         public abstract StringBuilder asString(StringBuilder strb);
326 
327         /**
328          * Gets the list of variables accessed by this expression.
329          * <p>This method will visit all nodes of the sub-expressions and extract all variables whether they
330          * are written in 'dot' or 'bracketed' notation. (a.b is equivalent to a['b']).</p>
331          * @return the set of variables, each as a list of strings (ant-ish variables use more than 1 string)
332          *         or the empty set if no variables are used
333          * @since 2.1
334          */
335         public Set<List<String>> getVariables() {
336             return Collections.emptySet();
337         }
338 
339         /**
340          * Fills up the list of variables accessed by this expression.
341          * @param refs the set of variable being filled
342          * @since 2.1
343          */
344         protected void getVariables(Set<List<String>> refs) {
345             // nothing to do
346         }
347 
348         /**
349          * Evaluates the immediate sub-expressions.
350          * <p>
351          * When the expression is dependant upon immediate and deferred sub-expressions,
352          * evaluates the immediate sub-expressions with the context passed as parameter
353          * and returns this expression deferred form.
354          * </p>
355          * <p>
356          * In effect, this binds the result of the immediate sub-expressions evaluation in the
357          * context, allowing to differ evaluation of the remaining (deferred) expression within another context.
358          * This only has an effect to nested & composite expressions that contain differed & immediate sub-expressions.
359          * </p>
360          * <p>
361          * If the underlying JEXL engine is silent, errors will be logged through its logger as warning.
362          * </p>
363          * <b>Note:</b> do not override; will be made final in a future release.
364          * @param context the context to use for immediate expression evaluations
365          * @return an expression or null if an error occurs and the {@link JexlEngine} is running in silent mode
366          * @throws UnifiedJEXL.Exception if an error occurs and the {@link JexlEngine} is not in silent mode
367          */
368         public Expression prepare(JexlContext context) {
369             try {
370                 Interpreter interpreter = new Interpreter(jexl, context, !jexl.isLenient(), jexl.isSilent());
371                 if (context instanceof TemplateContext) {
372                     interpreter.setFrame(((TemplateContext) context).getFrame());
373                 }
374                 return prepare(interpreter);
375             } catch (JexlException xjexl) {
376                 Exception xuel = createException("prepare", this, xjexl);
377                 if (jexl.isSilent()) {
378                     jexl.logger.warn(xuel.getMessage(), xuel.getCause());
379                     return null;
380                 }
381                 throw xuel;
382             }
383         }
384 
385         /**
386          * Evaluates this expression.
387          * <p>
388          * If the underlying JEXL engine is silent, errors will be logged through its logger as warning.
389          * </p>
390          * <b>Note:</b> do not override; will be made final in a future release.
391          * @param context the variable context
392          * @return the result of this expression evaluation or null if an error occurs and the {@link JexlEngine} is
393          * running in silent mode
394          * @throws UnifiedJEXL.Exception if an error occurs and the {@link JexlEngine} is not silent
395          */
396         public Object evaluate(JexlContext context) {
397             try {
398                 Interpreter interpreter = new Interpreter(jexl, context, !jexl.isLenient(), jexl.isSilent());
399                 if (context instanceof TemplateContext) {
400                     interpreter.setFrame(((TemplateContext) context).getFrame());
401                 }
402                 return evaluate(interpreter);
403             } catch (JexlException xjexl) {
404                 Exception xuel = createException("prepare", this, xjexl);
405                 if (jexl.isSilent()) {
406                     jexl.logger.warn(xuel.getMessage(), xuel.getCause());
407                     return null;
408                 }
409                 throw xuel;
410             }
411         }
412 
413         /**
414          * Retrieves this expression's source expression.
415          * If this expression was prepared, this allows to retrieve the
416          * original expression that lead to it.
417          * Other expressions return themselves.
418          * @return the source expression
419          */
420         public final Expression getSource() {
421             return source;
422         }
423 
424         /**
425          * Prepares a sub-expression for interpretation.
426          * @param interpreter a JEXL interpreter
427          * @return a prepared expression
428          * @throws JexlException (only for nested & composite)
429          */
430         protected Expression prepare(Interpreter interpreter) {
431             return this;
432         }
433 
434         /**
435          * Intreprets a sub-expression.
436          * @param interpreter a JEXL interpreter
437          * @return the result of interpretation
438          * @throws JexlException (only for nested & composite)
439          */
440         protected abstract Object evaluate(Interpreter interpreter);
441     }

443     /** A constant expression. */
444     private class ConstantExpression extends Expression {
445         /** The constant held by this expression. */
446         private final Object value;
447 
448         /**
449          * Creates a constant expression.
450          * <p>
451          * If the wrapped constant is a string, it is treated
452          * as a JEXL strings with respect to escaping.
453          * </p>
454          * @param val the constant value
455          * @param source the source expression if any
456          */
457         ConstantExpression(Object val, Expression source) {
458             super(source);
459             if (val == null) {
460                 throw new NullPointerException("constant can not be null");
461             }
462             if (val instanceof String) {
463                 val = StringParser.buildString((String) val, false);
464             }
465             this.value = val;
466         }
467 
468         /** {@inheritDoc} */
469         @Override
470         ExpressionType getType() {
471             return ExpressionType.CONSTANT;
472         }
473 
474         /** {@inheritDoc} */
475         @Override
476         public StringBuilder asString(StringBuilder strb) {
477             if (value != null) {
478                 strb.append(value.toString());
479             }
480             return strb;
481         }
482 
483         /** {@inheritDoc} */
484         @Override
485         protected Object evaluate(Interpreter interpreter) {
486             return value;
487         }
488     }

490     /** The base for Jexl based expressions. */
491     private abstract class JexlBasedExpression extends Expression {
492         /** The JEXL string for this expression. */
493         protected final CharSequence expr;
494         /** The JEXL node for this expression. */
495         protected final JexlNode node;
496 
497         /**
498          * Creates a JEXL interpretable expression.
499          * @param theExpr the expression as a string
500          * @param theNode the expression as an AST
501          * @param theSource the source expression if any
502          */
503         protected JexlBasedExpression(CharSequence theExpr, JexlNode theNode, Expression theSource) {
504             super(theSource);
505             this.expr = theExpr;
506             this.node = theNode;
507         }
508 
509         /** {@inheritDoc} */
510         @Override
511         public StringBuilder asString(StringBuilder strb) {
512             strb.append(isImmediate() ? IMM_CHAR : DEF_CHAR);
513             strb.append("{");
514             strb.append(expr);
515             strb.append("}");
516             return strb;
517         }
518 
519         /** {@inheritDoc} */
520         @Override
521         protected Object evaluate(Interpreter interpreter) {
522             return interpreter.interpret(node);
523         }
524 
525         /** {@inheritDoc} */
526         @Override
527         public Set<List<String>> getVariables() {
528             Set<List<String>> refs = new LinkedHashSet<List<String>>();
529             getVariables(refs);
530             return refs;
531         }
532 
533         /** {@inheritDoc} */
534         @Override
535         protected void getVariables(Set<List<String>> refs) {
536             jexl.getVariables(node, refs, null);
537         }
538     }

604     /**
605      * An immediate expression nested into a deferred expression.
606      * #{...${jexl}...}
607      * Note that the deferred syntax is JEXL's, not UnifiedJEXL.
608      */
609     private class NestedExpression extends JexlBasedExpression {
610         /**
611          * Creates a nested expression.
612          * @param expr the expression as a string
613          * @param node the expression as an AST
614          * @param source the source expression if any
615          */
616         NestedExpression(CharSequence expr, JexlNode node, Expression source) {
617             super(expr, node, source);
618             if (this.source != this) {
619                 throw new IllegalArgumentException("Nested expression can not have a source");
620             }
621         }
622 
623         @Override
624         public StringBuilder asString(StringBuilder strb) {
625             strb.append(expr);
626             return strb;
627         }
628 
629         /** {@inheritDoc} */
630         @Override
631         public boolean isImmediate() {
632             return false;
633         }
634 
635         /** {@inheritDoc} */
636         @Override
637         ExpressionType getType() {
638             return ExpressionType.NESTED;
639         }
640 
641         /** {@inheritDoc} */
642         @Override
643         protected Expression prepare(Interpreter interpreter) {
644             String value = interpreter.interpret(node).toString();
645             JexlNode dnode = jexl.parse(value, jexl.isDebug() ? node.debugInfo() : null, null);
646             return new ImmediateExpression(value, dnode, this);
647         }
648 
649         /** {@inheritDoc} */
650         @Override
651         protected Object evaluate(Interpreter interpreter) {
652             return prepare(interpreter).evaluate(interpreter);
653         }
654     }

656     /** A composite expression: "... ${...} ... #{...} ...". */
657     private class CompositeExpression extends Expression {
658         /** Bit encoded (deferred count > 0) bit 1, (immediate count > 0) bit 0. */
659         private final int meta;
660         /** The list of sub-expression resulting from parsing. */
661         protected final Expression[] exprs;
662 
663         /**
664          * Creates a composite expression.
665          * @param counters counters of expression per type
666          * @param list the sub-expressions
667          * @param src the source for this expresion if any
668          */
669         CompositeExpression(int[] counters, ArrayList<Expression> list, Expression src) {
670             super(src);
671             this.exprs = list.toArray(new Expression[list.size()]);
672             this.meta = (counters[ExpressionType.DEFERRED.index] > 0 ? 2 : 0)
673                     | (counters[ExpressionType.IMMEDIATE.index] > 0 ? 1 : 0);
674         }
675 
676         /** {@inheritDoc} */
677         @Override
678         public boolean isImmediate() {
679             // immediate if no deferred
680             return (meta & 2) == 0;
681         }
682 
683         /** {@inheritDoc} */
684         @Override
685         ExpressionType getType() {
686             return ExpressionType.COMPOSITE;
687         }
688 
689         /** {@inheritDoc} */
690         @Override
691         public StringBuilder asString(StringBuilder strb) {
692             for (Expression e : exprs) {
693                 e.asString(strb);
694             }
695             return strb;
696         }
697 
698         /** {@inheritDoc} */
699         @Override
700         public Set<List<String>> getVariables() {
701             Set<List<String>> refs = new LinkedHashSet<List<String>>();
702             for (Expression expr : exprs) {
703                 expr.getVariables(refs);
704             }
705             return refs;
706         }
707 
708         /** {@inheritDoc} */
709         @Override
710         protected Expression prepare(Interpreter interpreter) {
711             // if this composite is not its own source, it is already prepared
712             if (source != this) {
713                 return this;
714             }
715             // we need to prepare all sub-expressions
716             final int size = exprs.length;
717             final ExpressionBuilder builder = new ExpressionBuilder(size);
718             // tracking whether prepare will return a different expression
719             boolean eq = true;
720             for (int e = 0; e < size; ++e) {
721                 Expression expr = exprs[e];
722                 Expression prepared = expr.prepare(interpreter);
723                 // add it if not null
724                 if (prepared != null) {
725                     builder.add(prepared);
726                 }
727                 // keep track of expression equivalence
728                 eq &= expr == prepared;
729             }
730             Expression ready = eq ? this : builder.build(UnifiedJEXL.this, this);
731             return ready;
732         }
733 
734         /** {@inheritDoc} */
735         @Override
736         protected Object evaluate(Interpreter interpreter) {
737             final int size = exprs.length;
738             Object value = null;
739             // common case: evaluate all expressions & concatenate them as a string
740             StringBuilder strb = new StringBuilder();
741             for (int e = 0; e < size; ++e) {
742                 value = exprs[e].evaluate(interpreter);
743                 if (value != null) {
744                     strb.append(value.toString());
745                 }
746             }
747             value = strb.toString();
748             return value;
749         }

752     /** Creates a a {@link UnifiedJEXL.Expression} from an expression string.
753      *  Uses & fills up the expression cache if any.
754      * <p>
755      * If the underlying JEXL engine is silent, errors will be logged through its logger as warnings.
756      * </p>
757      * @param expression the UnifiedJEXL string expression
758      * @return the UnifiedJEXL object expression, null if silent and an error occured
759      * @throws UnifiedJEXL.Exception if an error occurs and the {@link JexlEngine} is not silent
760      */
761     public Expression parse(String expression) {
762         Exception xuel = null;
763         Expression stmt = null;
764         try {
765             if (cache == null) {
766                 stmt = parseExpression(expression, null);
767             } else {
768                 synchronized (cache) {
769                     stmt = cache.get(expression);
770                     if (stmt == null) {
771                         stmt = parseExpression(expression, null);
772                         cache.put(expression, stmt);
773                     }
774                 }
775             }
776         } catch (JexlException xjexl) {
777             xuel = new Exception("failed to parse '" + expression + "'", xjexl);
778         } catch (Exception xany) {
779             xuel = xany;
780         } finally {
781             if (xuel != null) {
782                 if (jexl.isSilent()) {
783                     jexl.logger.warn(xuel.getMessage(), xuel.getCause());
784                     return null;
785                 }
786                 throw xuel;
787             }
788         }
789         return stmt;
790     }

792     /**
793      * Creates a UnifiedJEXL.Exception from a JexlException.
794      * @param action parse, prepare, evaluate
795      * @param expr the expression
796      * @param xany the exception
797      * @return an exception containing an explicit error message
798      */
799     private Exception createException(String action, Expression expr, java.lang.Exception xany) {
800         StringBuilder strb = new StringBuilder("failed to ");
801         strb.append(action);
802         if (expr != null) {
803             strb.append(" '");
804             strb.append(expr.toString());
805             strb.append("'");
806         }
807         Throwable cause = xany.getCause();
808         if (cause != null) {
809             String causeMsg = cause.getMessage();
810             if (causeMsg != null) {
811                 strb.append(", ");
812                 strb.append(causeMsg);
813             }
814         }
815         return new Exception(strb.toString(), xany);
816     }

818     /** The different parsing states. */
819     private static enum ParseState {
820         /** Parsing a constant. */
821         CONST,
822         /** Parsing after $ .*/
823         IMMEDIATE0,
824         /** Parsing after # .*/
825         DEFERRED0,
826         /** Parsing after ${ .*/
827         IMMEDIATE1,
828         /** Parsing after #{ .*/
829         DEFERRED1,
830         /** Parsing after \ .*/
831         ESCAPE
832     }
833 
834     /**
835      * Parses a unified expression.
836      * @param expr the string expression
837      * @param scope the expression scope
838      * @return the expression instance
839      * @throws JexlException if an error occur during parsing
840      */
841     private Expression parseExpression(String expr, JexlEngine.Scope scope) {
842         final int size = expr.length();
843         ExpressionBuilder builder = new ExpressionBuilder(0);
844         StringBuilder strb = new StringBuilder(size);
845         ParseState state = ParseState.CONST;
846         int inner = 0;
847         boolean nested = false;
848         int inested = -1;
849         for (int i = 0; i < size; ++i) {
850             char c = expr.charAt(i);
851             switch (state) {
852                 default: // in case we ever add new expression type
853                     throw new UnsupportedOperationException("unexpected expression type");
854                 case CONST:
855                     if (c == IMM_CHAR) {
856                         state = ParseState.IMMEDIATE0;
857                     } else if (c == DEF_CHAR) {
858                         inested = i;
859                         state = ParseState.DEFERRED0;
860                     } else if (c == '\\') {
861                         state = ParseState.ESCAPE;
862                     } else {
863                         // do buildup expr
864                         strb.append(c);
865                     }
866                     break;
867                 case IMMEDIATE0: // $
868                     if (c == '{') {
869                         state = ParseState.IMMEDIATE1;
870                         // if chars in buffer, create constant
871                         if (strb.length() > 0) {
872                             Expression cexpr = new ConstantExpression(strb.toString(), null);
873                             builder.add(cexpr);
874                             strb.delete(0, Integer.MAX_VALUE);
875                         }
876                     } else {
877                         // revert to CONST
878                         strb.append(IMM_CHAR);
879                         strb.append(c);
880                         state = ParseState.CONST;
881                     }
882                     break;
883                 case DEFERRED0: // #
884                     if (c == '{') {
885                         state = ParseState.DEFERRED1;
886                         // if chars in buffer, create constant
887                         if (strb.length() > 0) {
888                             Expression cexpr = new ConstantExpression(strb.toString(), null);
889                             builder.add(cexpr);
890                             strb.delete(0, Integer.MAX_VALUE);
891                         }
892                     } else {
893                         // revert to CONST
894                         strb.append(DEF_CHAR);
895                         strb.append(c);
896                         state = ParseState.CONST;
897                     }
898                     break;
899                 case IMMEDIATE1: // ${...
900                     if (c == '}') {
901                         // materialize the immediate expr
902                         Expression iexpr = new ImmediateExpression(
903                                 strb.toString(),
904                                 jexl.parse(strb, null, scope),
905                                 null);
906                         builder.add(iexpr);
907                         strb.delete(0, Integer.MAX_VALUE);
908                         state = ParseState.CONST;
909                     } else {
910                         // do buildup expr
911                         strb.append(c);
912                     }
913                     break;
914                 case DEFERRED1: // #{...
915                     // skip inner strings (for '}')
916                     if (c == '"' || c == '\'') {
917                         strb.append(c);
918                         i = StringParser.readString(strb, expr, i + 1, c);
919                         continue;
920                     }
921                     // nested immediate in deferred; need to balance count of '{' & '}'
922                     if (c == '{') {
923                         if (expr.charAt(i - 1) == IMM_CHAR) {
924                             inner += 1;
925                             strb.deleteCharAt(strb.length() - 1);
926                             nested = true;
927                         }
928                         continue;
929                     }
930                     // closing '}'
931                     if (c == '}') {
932                         // balance nested immediate
933                         if (inner > 0) {
934                             inner -= 1;
935                         } else {
936                             // materialize the nested/deferred expr
937                             Expression dexpr = null;
938                             if (nested) {
939                                 dexpr = new NestedExpression(
940                                         expr.substring(inested, i + 1),
941                                         jexl.parse(strb, null, scope),
942                                         null);
943                             } else {
944                                 dexpr = new DeferredExpression(
945                                         strb.toString(),
946                                         jexl.parse(strb, null, scope),
947                                         null);
948                             }
949                             builder.add(dexpr);
950                             strb.delete(0, Integer.MAX_VALUE);
951                             nested = false;
952                             state = ParseState.CONST;
953                         }
954                     } else {
955                         // do buildup expr
956                         strb.append(c);
957                     }
958                     break;
959                 case ESCAPE:
960                     if (c == DEF_CHAR) {
961                         strb.append(DEF_CHAR);
962                     } else if (c == IMM_CHAR) {
963                         strb.append(IMM_CHAR);
964                     } else {
965                         strb.append('\\');
966                         strb.append(c);
967                     }
968                     state = ParseState.CONST;
969             }
970         }
971         // we should be in that state
972         if (state != ParseState.CONST) {
973             throw new Exception("malformed expression: " + expr, null);
974         }
975         // if any chars were buffered, add them as a constant
976         if (strb.length() > 0) {
977             Expression cexpr = new ConstantExpression(strb.toString(), null);
978             builder.add(cexpr);
979         }
980         return builder.build(this, null);
981     }

1019     /**
1020      * A Template is a script that evaluates by writing its content through a Writer.
1021      * This is a simplified replacement for Velocity that uses JEXL (instead of OGNL/VTL) as the scripting
1022      * language.
1023      * <p>
1024      * The source text is parsed considering each line beginning with '$$' (as default pattern) as JEXL script code
1025      * and all others as Unified JEXL expressions; those expressions will be invoked from the script during
1026      * evaluation and their output gathered through a writer. 
1027      * It is thus possible to use looping or conditional construct "around" expressions generating output.
1028      * </p>
1029      * For instance:
1030      * <p><blockquote><pre>
1031      * $$ for(var x : [1, 3, 5, 42, 169]) {
1032      * $$   if (x == 42) {
1033      * Life, the universe, and everything
1034      * $$   } else if (x > 42) {
1035      * The value $(x} is over fourty-two
1036      * $$   } else {
1037      * The value ${x} is under fourty-two
1038      * $$   }
1039      * $$ }
1040      * </pre></blockquote>
1041      * Will evaluate as:
1042      * <p><blockquote><pre>
1043      * The value 1 is under fourty-two
1044      * The value 3 is under fourty-two
1045      * The value 5 is under fourty-two
1046      * Life, the universe, and everything
1047      * The value 169 is over fourty-two
1048      * </pre></blockquote>
1049      * <p>
1050      * During evaluation, the template context exposes its writer as '$jexl' which is safe to use in this case.
1051      * This allows writing directly through the writer without adding new-lines as in:
1052      * <p><blockquote><pre>
1053      * $$ for(var cell : cells) { $jexl.print(cell); $jexl.print(';') }
1054      * </pre></blockquote>
1055      * </p>
1056      * <p>
1057      * A template is expanded as one JEXL script and a list of UnifiedJEXL expressions; each UnifiedJEXL expression
1058      * being replace in the script by a call to jexl:print(expr) (the expr is in fact the expr number in the template).
1059      * This integration uses a specialized JexlContext (TemplateContext) that serves as a namespace (for jexl:)
1060      * and stores the expression array and the writer (java.io.Writer) that the 'jexl:print(...)'
1061      * delegates the output generation to.
1062      * </p>
1063      * @since 2.1
1064      */
1065     public final class Template {
1066         /** The prefix marker. */
1067         private final String prefix;
1068         /** The array of source blocks. */
1069         private final TemplateBlock[] source;
1070         /** The resulting script. */
1071         private final ASTJexlScript script;
1072         /** The UnifiedJEXL expressions called by the script. */
1073         private final Expression[] exprs;
1074 
1075         /**
1076          * Creates a new template from an input.
1077          * @param directive the prefix for lines of code; can not be "$", "${", "#" or "#{"
1078          * since this would preclude being able to differentiate directives and UnifiedJEXL expressions
1079          * @param reader the input reader
1080          * @param parms the parameter names
1081          * @throws NullPointerException if either the directive prefix or input is null
1082          * @throws IllegalArgumentException if the directive prefix is invalid
1083          */
1084         public Template(String directive, Reader reader, String... parms) {
1085             if (directive == null) {
1086                 throw new NullPointerException("null prefix");
1087             }
1088             if ("$".equals(directive)
1089                     || "${".equals(directive)
1090                     || "#".equals(directive)
1091                     || "#{".equals(directive)) {
1092                 throw new IllegalArgumentException(directive + ": is not a valid directive pattern");
1093             }
1094             if (reader == null) {
1095                 throw new NullPointerException("null input");
1096             }
1097             JexlEngine.Scope scope = new JexlEngine.Scope(parms);
1098             prefix = directive;
1099             List<TemplateBlock> blocks = readTemplate(prefix, reader);
1100             List<Expression> uexprs = new ArrayList<Expression>();
1101             StringBuilder strb = new StringBuilder();
1102             int nuexpr = 0;
1103             int codeStart = -1;
1104             for (int b = 0; b < blocks.size(); ++b) {
1105                 TemplateBlock block = blocks.get(b);
1106                 if (block.type == BlockType.VERBATIM) {
1107                     strb.append("jexl:print(");
1108                     strb.append(nuexpr++);
1109                     strb.append(");");
1110                 } else {
1111                     // keep track of first block of code, the frame creator
1112                     if (codeStart < 0) {
1113                         codeStart = b;
1114                     }
1115                     strb.append(block.body);
1116                 }
1117             }
1118             // parse the script
1119             script = getEngine().parse(strb.toString(), null, scope);
1120             scope = script.getScope();
1121             // parse the exprs using the code frame for those appearing after the first block of code
1122             for (int b = 0; b < blocks.size(); ++b) {
1123                 TemplateBlock block = blocks.get(b);
1124                 if (block.type == BlockType.VERBATIM) {
1125                     uexprs.add(UnifiedJEXL.this.parseExpression(block.body, b > codeStart ? scope : null));
1126                 }
1127             }
1128             source = blocks.toArray(new TemplateBlock[blocks.size()]);
1129             exprs = uexprs.toArray(new Expression[uexprs.size()]);
1130         }

1327         /**
1328          * Prints to output.
1329          * <p>This will dynamically try to find the best suitable method in the writer through uberspection.
1330          * Subclassing Writer by adding 'print' methods should be the preferred way to specialize output.
1331          * </p>
1332          * @param arg the argument to print out
1333          */
1334         private void doPrint(Object arg) {
1335             try {
1336                 if (arg instanceof CharSequence) {
1337                     writer.write(arg.toString());
1338                 } else if (arg != null) {
1339                     Object[] value = {arg};
1340                     Uberspect uber = getEngine().getUberspect();
1341                     JexlMethod method = uber.getMethod(writer, "print", value, null);
1342                     if (method != null) {
1343                         method.invoke(writer, value);
1344                     } else {
1345                         writer.write(arg.toString());
1346                     }
1347                 }
1348             } catch (java.io.IOException xio) {
1349                 throw createException("call print", null, xio);
1350             } catch (java.lang.Exception xany) {
1351                 throw createException("invoke print", null, xany);
1352             }
1353         }

1356     /**
1357      * Whether a sequence starts with a given set of characters (following spaces).
1358      * <p>Space characters at beginning of line before the pattern are discarded.</p>
1359      * @param sequence the sequence
1360      * @param pattern the pattern to match at start of sequence
1361      * @return the first position after end of pattern if it matches, -1 otherwise
1362      * @since 2.1
1363      */
1364     protected int startsWith(CharSequence sequence, CharSequence pattern) {
1365         int s = 0;
1366         while (Character.isSpaceChar(sequence.charAt(s))) {
1367             s += 1;
1368         }
1369         sequence = sequence.subSequence(s, sequence.length());
1370         if (pattern.length() <= sequence.length()
1371                 && sequence.subSequence(0, pattern.length()).equals(pattern)) {
1372             return s + pattern.length();
1373         } else {
1374             return -1;
1375         }
1376     }

1378     /**
1379      * Reads lines of a template grouping them by typed blocks.
1380      * @param prefix the directive prefix
1381      * @param source the source reader
1382      * @return the list of blocks
1383      * @since 2.1
1384      */
1385     protected List<TemplateBlock> readTemplate(final String prefix, Reader source) {
1386         try {
1387             int prefixLen = prefix.length();
1388             List<TemplateBlock> blocks = new ArrayList<TemplateBlock>();
1389             BufferedReader reader;
1390             if (source instanceof BufferedReader) {
1391                 reader = (BufferedReader) source;
1392             } else {
1393                 reader = new BufferedReader(source);
1394             }
1395             StringBuilder strb = new StringBuilder();
1396             BlockType type = null;
1397             while (true) {
1398                 CharSequence line = reader.readLine();
1399                 if (line == null) {
1400                     // at end
1401                     TemplateBlock block = new TemplateBlock(type, strb.toString());
1402                     blocks.add(block);
1403                     break;
1404                 } else if (type == null) {
1405                     // determine starting type if not known yet
1406                     prefixLen = startsWith(line, prefix);
1407                     if (prefixLen >= 0) {
1408                         type = BlockType.DIRECTIVE;
1409                         strb.append(line.subSequence(prefixLen, line.length()));
1410                     } else {
1411                         type = BlockType.VERBATIM;
1412                         strb.append(line.subSequence(0, line.length()));
1413                         strb.append('\n');
1414                     }
1415                 } else if (type == BlockType.DIRECTIVE) {
1416                     // switch to verbatim if necessary
1417                     prefixLen = startsWith(line, prefix);
1418                     if (prefixLen < 0) {
1419                         TemplateBlock code = new TemplateBlock(BlockType.DIRECTIVE, strb.toString());
1420                         strb.delete(0, Integer.MAX_VALUE);
1421                         blocks.add(code);
1422                         type = BlockType.VERBATIM;
1423                         strb.append(line.subSequence(0, line.length()));
1424                     } else {
1425                         strb.append(line.subSequence(prefixLen, line.length()));
1426                     }
1427                 } else if (type == BlockType.VERBATIM) {
1428                     // switch to code if necessary(
1429                     prefixLen = startsWith(line, prefix);
1430                     if (prefixLen >= 0) {
1431                         strb.append('\n');
1432                         TemplateBlock verbatim = new TemplateBlock(BlockType.VERBATIM, strb.toString());
1433                         strb.delete(0, Integer.MAX_VALUE);
1434                         blocks.add(verbatim);
1435                         type = BlockType.DIRECTIVE;
1436                         strb.append(line.subSequence(prefixLen, line.length()));
1437                     } else {
1438                         strb.append(line.subSequence(0, line.length()));
1439                     }
1440                 }
1441             }
1442             return blocks;
1443         } catch (IOException xio) {
1444             return null;
1445         }
1446     }
