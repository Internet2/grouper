46  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.cookie;
47  
48  import java.util.Comparator;
49  
50  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.Cookie;
51  
52  /**
53   * This cookie comparator ensures that multiple cookies satisfying 
54   * a common criteria are ordered in the <tt>Cookie</tt> header such
55   * that those with more specific Path attributes precede those with
56   * less specific.
57   *  
58   * <p>
59   * This comparator assumes that Path attributes of two cookies 
60   * path-match a commmon request-URI. Otherwise, the result of the
61   * comparison is undefined.
62   * </p>
63   * 
64   * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
65   * 
66   * @since 3.1
67   */
68  public class CookiePathComparator implements Comparator {
69  
70      private String normalizePath(final Cookie cookie) {
71          String path = cookie.getPath();
72          if (path == null) {
73              path = "/";
74          }
75          if (!path.endsWith("/")) {
76              path = path + "/";
77          }
78          return path;
79      }
80      
81      public int compare(final Object o1, final Object o2) {
82          Cookie c1 = (Cookie) o1;
83          Cookie c2 = (Cookie) o2;
84          String path1 = normalizePath(c1);
85          String path2 = normalizePath(c2);
86          if (path1.equals(path2)) {
87              return 0;
88          } else if (path1.startsWith(path2)) {
89              return -1;
90          } else if (path2.startsWith(path1)) {
91              return 1;
92          } else {
93              // Does not really matter
94              return 0;
95          }
96      }
97  
98  }
