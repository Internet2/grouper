52    /**
53     * make sure there is a user authenticated
54     * @param context
55     */
56    static void assertAuthenticatedAndHasPermissions(ConnectionContext context) {
57      
58      assertAuthenticated(context);
59      
60      if (!context.getSecurityContext().isBrokerContext()) {
61        //lets make sure the user at least has some permissions...
62        String userName = context.getSecurityContext().getUserName();
63        if (!GrouperActivemqPermissionsEngine.hasAnyPermission(userName)) {
64          throw new SecurityException("User does not have any permissions: " + userName);
65        }
66      }
67    }
68  
69    static {
70      try {
71        GrouperActivemqPermissionsEngine.startupOnce();
72      } catch (RuntimeException e) {
73        LOG.error("Error starting up", e);
74        throw e;
75      }
76    }

78    /**
79     * make sure there is a user authenticated
80     * @param context
81     */
82    static void assertAuthenticated(ConnectionContext context) {
83      
84      GrouperActivemqPermissionsEngine.startupOnce();
85  
86      final SecurityContext securityContext = context.getSecurityContext();
87      if (securityContext == null) {
88        throw new SecurityException("User is not authenticated.");
89      }
90  
91    }

213   /**
214    * regex for converting a destination to 
215    */
216   private static Pattern destinationPattern = Pattern.compile("^(topic|queue)://(.*)$");
217   
218   /**
219    * destination is queue://something.something.something
220    * @param destination
221    * @return the destination: something.something.something
222    */
223   static String destinationStringFromDestination(String destination) {
224     
225     Matcher matcher = destinationPattern.matcher(destination);
226     if (!matcher.matches()) {
227       throw new RuntimeException("Why does destination not match pattern? " + destination);
228     }
229     
230     return matcher.group(2);
231   }

263   /**
264    * make sure the user has the permission
265    * @param context
266    * @param destination
267    * @param action
268    */
269   static void assertHasPermission(ConnectionContext context, ActiveMQDestination destination, GrouperActivemqPermissionAction action) {
270 
271     if (!context.getSecurityContext().isBrokerContext() && destination != null) {
272 
273       String qualifiedName = destination.getQualifiedName();
274   
275       //could be comma separated???
276       //ActiveMQ.Advisory.TempQueue,topic://ActiveMQ.Advisory.TempTopic
277 
278       //note, destinations cant have commas
279       String[] qualifiedNames = GrouperClientUtils.splitTrim(qualifiedName, ",");
280       
281       for (String individualQualifiedName : qualifiedNames) {
282       
283         individualQualifiedName = destinationStringFromDestination(individualQualifiedName);
284         
285         String userName = context.getSecurityContext().getUserName();
286   
287         if (!GrouperActivemqPermissionsEngine.hasPermission(userName, action, individualQualifiedName)) {
288           throw new SecurityException("User: " + userName + " does not have permission to : " + action + " on " + individualQualifiedName);
289         }
290       }
291     }
292   }
