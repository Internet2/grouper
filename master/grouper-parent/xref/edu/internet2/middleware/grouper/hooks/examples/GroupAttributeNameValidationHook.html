44  /**
45   * <pre>
46   * built in hook to grouper, which is turned on when it is configured in the grouper.properties.
47   * 
48   * you can retrict certain attributes of a group to be within a certain regex
49   * 
50   * </pre>
51   */
52  public class GroupAttributeNameValidationHook extends GroupHooks {
53    
54    /**
55     * for unit tests
56     */
57    public static final String TEST_ATTRIBUTE_NAME = "testAttribute123";
58  
59    /**
60     * for unit tests
61     */
62    public static final String TEST_PATTERN = "qwertyuiop";
63  
64    /**
65     * only register once
66     */
67    private static boolean registered = false;
68    
69    /**
70     * see if this is configured in the grouper.properties, if so, register this hook
71     * @param addTestValidation if a test validation should be added
72     */
73    public static void registerHookIfNecessary(boolean addTestValidation) {
74      
75      if (registered) {
76        return;
77      }
78      
79      //see if there are config entries
80      Properties grouperProperties = GrouperConfig.retrieveConfig().properties();
81  
82      
83      //group.attribute.validator.attributeName.0=extension
84      //group.attribute.validator.regex.0=^[a-zA-Z0-9]+$
85      //group.attribute.validator.vetoMessage.0=Group ID '$attributeValue$' is invalid since it must contain only alpha-numerics
86      
87      int index = 0;
88  
89      while (true) {
90        String attributeName = StringUtils.trim(grouperProperties.getProperty(
91            "group.attribute.validator.attributeName." + index));
92        String regex = StringUtils.trim(grouperProperties.getProperty(
93            "group.attribute.validator.regex." + index));
94        String vetoMessage = StringUtils.trim(grouperProperties.getProperty(
95            "group.attribute.validator.vetoMessage." + index));
96        
97        //if we are done checking
98        if (StringUtils.isBlank(attributeName)) {
99          if (!StringUtils.isBlank(regex) || !StringUtils.isBlank(vetoMessage)) {
100           throw new RuntimeException("Dont configure a regex or vetoMessage without an attribute name! index: " + index
101               + ", check the grouper.properties");
102         }
103         break;
104       }
105       //these are required
106       if (StringUtils.isBlank(regex) || StringUtils.isBlank(vetoMessage)) {
107         throw new RuntimeException("Regex and vetoMessage are required for attribute: '" + attributeName
108             + "' index: " + index + ", check the grouper.properties file");
109       }
110       //see if already exists
111       if (attributeNamePatterns.containsKey(attributeName)) {
112         throw new RuntimeException("Attribute name already exists (duplicate): '" + attributeName
113             + "' index: " + index + ", check the grouper.properties file");
114       }
115       //add all configs
116       attributeNamePatterns.put(attributeName, Pattern.compile(regex));
117       attributeNameRegexes.put(attributeName, regex);
118       attributeNameVetoMessages.put(attributeName, vetoMessage);
119       
120       index++;
121     }
122 
123     if (addTestValidation) {
124       
125       attributeNamePatterns.put(TEST_ATTRIBUTE_NAME, Pattern.compile("^" + TEST_PATTERN + "$"));
126       attributeNameRegexes.put(TEST_ATTRIBUTE_NAME, "^" + TEST_PATTERN + "$");
127       attributeNameVetoMessages.put(TEST_ATTRIBUTE_NAME, "Attribute testAttribute123 cannot have the value: '$attributeValue$'");
128       
129       index++;
130     }
131     
132     //register the hook
133     if (index > 0) {
134       //register this hook
135       GrouperHooksUtils.addHookManual(GrouperHookType.GROUP.getPropertyFileKey(), GroupAttributeNameValidationHook.class);
136       GrouperHooksUtils.addHookManual(GrouperHookType.ATTRIBUTE.getPropertyFileKey(), GroupAttributeNameValidationAttrHook.class);
137     }
138     
139     registered = true;
140 
141   }

143   /**
144    * 
145    * @see edu.internet2.middleware.grouper.hooks.GroupHooks#groupPreInsert(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksGroupBean)
146    */
147   @Override
148   public void groupPreInsert(HooksContext hooksContext, HooksGroupBean preInsertBean) {
149     Group group = preInsertBean.getGroup();
150     Map<String, Attribute> attributesMap = GrouperUtil.nonNull(group.getAttributesMap(false));
151     Set<String> attributesChanged = new HashSet<String>(attributesMap.keySet());
152     
153     //also add name, extension, etc
154     attributesChanged.addAll(Group.INTERNAL_FIELD_ATTRIBUTES);
155     
156     groupPreChangeAttribute(group, attributesChanged);
157   }
158 
159   /**
160    * check that a new attribute value is ok
161    * @param group 
162    * @param attributeNamesToCheck 
163    */
164   private static void groupPreChangeAttribute(Group group, Set<String> attributeNamesToCheck) {
165 
166     //loop through fields, look for attributes
167     for (String fieldName : GrouperUtil.nonNull(attributeNamesToCheck)) {
168 
169       //if attribute or built-in field then process
170       if (Group.INTERNAL_FIELD_ATTRIBUTES.contains(fieldName)) {
171         String value = (String)group.fieldValue(fieldName);
172         groupPreChangeAttribute(fieldName, value);
173       }
174     }
175   }
176 
177   /** cache of attribute names to patterns */
178   static Map<String, Pattern> attributeNamePatterns = new HashMap<String, Pattern>();
179 
180   /** cache of attribute names to regexs */
181   static Map<String, String> attributeNameRegexes = new HashMap<String, String>();
182   
183   /** cache of attribute names to error messages when a name doesnt match */
184   static Map<String, String> attributeNameVetoMessages = new HashMap<String, String>();
185   
186   /**
187    * check that a new attribute value is ok (either a group field, or an attribute)
188    * @param attributeName 
189    * @param attributeValue 
190    */
191   static void groupPreChangeAttribute(String attributeName, String attributeValue) {
192     //see if there is a configuration about this attribute
193     if (attributeNamePatterns.containsKey(attributeName)) {
194       Pattern pattern = attributeNamePatterns.get(attributeName);
195       if (pattern == null) {
196         throw new RuntimeException("Regex pattern '" + attributeNameRegexes.get(attributeName) 
197             + "'probably didnt compile for attribute: '" 
198             + attributeName + "', check logs or grouper.properties");
199       }
200       Matcher matcher = pattern.matcher(StringUtils.defaultString(attributeValue));
201       if (!matcher.matches()) {
202         
203         String attributeNameErrorMessage = attributeNameVetoMessages.get(attributeName);
204 
205         //substitute the attribute name
206         attributeNameErrorMessage = StringUtils.replace(attributeNameErrorMessage, "$attributeValue$", attributeValue);
207         
208         throw new HookVeto("veto.group.attribute.name.regex." + attributeName, attributeNameErrorMessage);
209       }
210     }
211   }
212   
213   /**
214    * @see edu.internet2.middleware.grouper.hooks.GroupHooks#groupPreUpdate(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksGroupBean)
215    */
216   @Override
217   public void groupPreUpdate(HooksContext hooksContext, HooksGroupBean preUpdateBean) {
218     Group group = preUpdateBean.getGroup();
219     Set<String> fieldNames = group.dbVersionDifferentFields(false);
220     
221     groupPreChangeAttribute(group, fieldNames);
222   }
