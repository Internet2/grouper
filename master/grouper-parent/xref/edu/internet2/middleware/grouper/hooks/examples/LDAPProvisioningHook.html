31  /**
32   * hook to prevent ldap provisioning by setting an attribute on the group.
33   */
34  
35  public class LDAPProvisioningHook extends GroupHooks {
36  
37   /**
38    * @see edu.internet2.middleware.grouper.hooks.GroupHooks#groupPostInsert(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksGroupBean)
39    */
40  
41    @SuppressWarnings("deprecation")
42    @Override
43    public void groupPostInsert(HooksContext hooksContext, HooksGroupBean postInsertBean) {
44  
45      final Group group = postInsertBean.getGroup();
46      String name = group.getName();
47  
48      boolean excludeMatches = false;
49      int count = 0;
50      while (true) {
51        String property = "LDAPProvisioningHook.exclude.regex." + count;
52        String regex = GrouperConfig.retrieveConfig().propertyValueString(property);
53        if (regex == null) {
54          break;
55        }
56  
57        if (name.matches(regex)) {
58          excludeMatches = true;
59          break;
60        }
61  
62        count++;
63      }
64  
65      if (excludeMatches) {
66  
67        GrouperSession.callbackGrouperSession(GrouperSession.staticGrouperSession().internal_getRootSession(), new GrouperSessionHandler() {
68  
69          @Override
70          public Object callback(GrouperSession grouperSession)
71              throws GrouperSessionException {
72  
73            GroupType groupType = GroupTypeFinder.find("LDAPProvisioning", false);
74            if (groupType == null) {
75              groupType = GroupType.createType(grouperSession, "LDAPProvisioning");
76              groupType.addAttribute(grouperSession, "LDAPProvisioningExclude");
77              groupType.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_READ, false);
78              groupType.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_UPDATE, false);
79              groupType.internal_getAttributeDefForAttributes().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_READ, false);
80              groupType.internal_getAttributeDefForAttributes().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_UPDATE, false);
81            }
82  
83            group.addType(groupType);
84            group.setAttribute("LDAPProvisioningExclude", "true");
85            return null;
86          }
87        });
88      }
89    }
90  }
