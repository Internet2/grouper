134   /**
135    * cache if stem name has the membership one attribute
136    */
137   private static GrouperCache<String, Boolean> stemHasMembershipOneAttribute = new GrouperCache(
138       MembershipOneInFolderMaxHook.class.getName() + ".membershipOneAttribute", 5000, false, 60, 60, false);
139   //TODO remove defaults in 2.3+
140   
141   /**
142    * @see edu.internet2.middleware.grouper.hooks.MembershipHooks#membershipPreAddMember(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksMembershipChangeBean)
143    */
144   @Override
145   public void membershipPreAddMember(HooksContext hooksContext,
146       final HooksMembershipChangeBean preAddMemberBean) {
147 
148     if (!FieldType.LIST.equals(preAddMemberBean.getField().getType())) {
149       return;
150     }
151 
152     if (GrouperCheckConfig.inCheckConfig || !GrouperStartup.isFinishedStartupSuccessfully()) {
153       return;
154     }
155 
156     initObjectsOnce(false);
157 
158     final Set<Group> problemGroups = new HashSet<Group>();
159 
160     GrouperSession.internal_callbackRootGrouperSession(new GrouperSessionHandler() {
161       
162       public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
163         
164         
165         //lets see if we are limiting for this membership
166         Group group = preAddMemberBean.getGroup();
167         
168         String parentStemName = group.getParentStemName();
169         
170         Boolean membershipOne = stemHasMembershipOneAttribute.get(parentStemName);
171 
172         //if checking, see what memberships are in this folder
173         Stem parentStem = null;
174 
175         if (membershipOne == null) {
176 
177           if (parentStem == null) {
178             parentStem = group.getParentStem();
179           }
180           
181           //if not in cache that we check this stem, add it
182           AttributeDefName membershipOneAttributeDefName = membershipOneInFolderAttributeDefName();
183           
184           membershipOne = parentStem.getAttributeDelegate().hasAttribute(membershipOneAttributeDefName);
185 
186           stemHasMembershipOneAttribute.put(parentStemName, membershipOne);
187         }
188           
189         if (membershipOne) {
190 
191           if (parentStem == null) {
192             parentStem = group.getParentStem();
193           }
194 
195           Set<MembershipSubjectContainer> membershipSubjectContainers = new MembershipFinder()
196             .assignStem(parentStem).assignStemScope(Scope.ONE).assignField(Group.getDefaultList())
197             .assignMemberIds(GrouperUtil.toSet(preAddMemberBean.getMember().getId()))
198             .assignEnabled(true).findMembershipResult().getMembershipSubjectContainers();
199           
200           for (MembershipSubjectContainer membershipSubjectContainer : GrouperUtil.nonNull(membershipSubjectContainers)) {
201 
202             Group currentGroup = membershipSubjectContainer.getGroupOwner();
203             if (!StringUtils.equals(currentGroup.getName(), group.getName())) {
204               problemGroups.add(currentGroup);
205             }
206             
207           }
208           
209         }
210         return null;
211       }
212     });
213     
214 
215     //now we are out of the GrouperSystem, and we need to remove memberships
216     //do this with the privileges of the calling user.  if they cant, throw a veto
217     for (Group currentGroup : problemGroups) {
218       try {
219         currentGroup.deleteMember(preAddMemberBean.getMember());
220       } catch (Exception e) {
221         if (LOG.isDebugEnabled()) {
222           LOG.debug("Problem auto removing from " + currentGroup.getName() + ", " + GrouperUtil.subjectToString(GrouperSession.staticGrouperSession().getSubject()), e);
223         }
224         throw new HookVeto(HOOK_VETO_MEMBERSHIP_ONE_IN_FOLDER_CANT_DELETE_MEMBER, "This folder only allows one membership in any group, "
225             + "but there was a problem removing from " + currentGroup.getName() + ", do you have privileges to remove memberships there?");
226       }
227     }
228     
229   }

231   /**
232    * @return attribute def name for this hook
233    */
234   public static AttributeDefName membershipOneInFolderAttributeDefName() {
235     initObjectsOnce(false);
236     return AttributeDefNameFinder.findByNameCache(membershipOneFolderStemName() + ":" + membershipOneFolderExtensionOfAttributeDefName, true);
237   }

239   /**
240    * pass in the stem and assign attribute and clear cache
241    * @param stem
242    */
243   public static void assignMembershipOneInFolderAttributeDefName(Stem stem) {
244     
245     stem.getAttributeDelegate().assignAttribute(MembershipOneInFolderMaxHook.membershipOneInFolderAttributeDefName());
246     stemHasMembershipOneAttribute.clear();
247   }
