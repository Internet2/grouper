140   /**
141    * retrieve the attributeAssign for this lookup if not looked up yet.  pass in a grouper session
142    * @param grouperSession 
143    * @param invalidQueryReason is the text to go in the WsInvalidQueryException
144    * @param backReferenceAttributeAssignIds are attribute assign ids  if in batch and needs backreferences
145    * @return the attributeAssign
146    * @throws WsInvalidQueryException if there is a problem, and if the invalidQueryReason is set
147    */
148   public AttributeAssign retrieveAttributeAssignIfNeeded(GrouperSession grouperSession,
149       String invalidQueryReason, String[] backReferenceAttributeAssignIds) throws WsInvalidQueryException {
150     
151     //see if we already retrieved
152     if (this.attributeAssignFindResult != null) {
153       return this.attributeAssign;
154     }
155     try {
156       //assume success (set otherwise if there is a problem)
157       this.attributeAssignFindResult = AttributeAssignResult.SUCCESS;
158 
159       String theUuid = this.uuid;
160       if (StringUtils.isBlank(theUuid) && !StringUtils.isBlank(this.batchIndex)) {
161         int batchIndexInt = GrouperUtil.intValue(this.batchIndex);
162         if (batchIndexInt >= GrouperUtil.length(backReferenceAttributeAssignIds)) {
163           this.attributeAssignFindResult = AttributeAssignResult.INVALID_QUERY;
164           throw new WsInvalidQueryException("Batch index is " + batchIndexInt 
165               + ", but max index is: " + (GrouperUtil.length(backReferenceAttributeAssignIds) -1));
166         }
167         theUuid = backReferenceAttributeAssignIds[batchIndexInt];        
168       }
169 
170       //must have a name or uuid
171       if (StringUtils.isBlank(theUuid)) {
172         this.attributeAssignFindResult = AttributeAssignResult.INVALID_QUERY;
173         if (!StringUtils.isEmpty(invalidQueryReason)) {
174           throw new WsInvalidQueryException("Invalid attributeAssign query for '"
175               + invalidQueryReason + "', " + this);
176         }
177         String logMessage = "Invalid query: " + this;
178         LOG.warn(logMessage);
179       } else {
180         this.attributeAssign = AttributeAssignFinder.findById(theUuid, true);
181       }
182 
183     } catch (AttributeAssignNotFoundException anf) {
184       this.attributeAssignFindResult = AttributeAssignResult.ATTRIBUTE_ASSIGN_NOT_FOUND;
185       if (!StringUtils.isBlank(invalidQueryReason)) {
186         throw new WsInvalidQueryException("Invalid attributeAssign for '" + invalidQueryReason
187             + "', " + this, anf);
188       }
189     }
190     return this.attributeAssign;
191   }

250   /**
251    * convert attributeAssign lookups to attributeAssign ids
252    * @param grouperSession
253    * @param wsAttributeAssignLookups
254    * @param errorMessage
255    * @param lookupCount is an array of size one int where 1 will be added if there are records, and no change if not
256    * @param backReferenceAttributeAssignIds are attribute assign ids  if in batch and needs backreferences
257    * @return the membership ids
258    */
259   public static Set<String> convertToAttributeAssignIds(GrouperSession grouperSession, 
260       WsAttributeAssignLookup[] wsAttributeAssignLookups, StringBuilder errorMessage, int[] lookupCount, String[] backReferenceAttributeAssignIds) {
261     //get all the attributeAssigns
262     //we could probably batch these to get better performance.
263     Set<String> attributeAssignIds = null;
264     if (GrouperUtil.length(wsAttributeAssignLookups) > 0) {
265       
266       attributeAssignIds = new LinkedHashSet<String>();
267       int i=0;
268       
269       boolean foundRecords = false;
270       
271       for (WsAttributeAssignLookup wsAttributeAssignLookup : wsAttributeAssignLookups) {
272         
273         if (wsAttributeAssignLookup == null || !wsAttributeAssignLookup.hasData()) {
274           continue;
275         }
276         
277         if (!foundRecords) {
278           lookupCount[0]++;
279           foundRecords = true;
280         }
281         
282         wsAttributeAssignLookup.retrieveAttributeAssignIfNeeded(grouperSession, null, backReferenceAttributeAssignIds);
283         AttributeAssign attributeAssign = wsAttributeAssignLookup.retrieveAttributeAssign();
284         if (attributeAssign != null) {
285           attributeAssignIds.add(attributeAssign.getId());
286         } else {
287           
288           if (errorMessage.length() > 0) {
289             errorMessage.append(", ");
290           }
291           
292           errorMessage.append("Error on attributeAssign index: " + i + ", " 
293               + wsAttributeAssignLookup.retrieveAttributeAssignFindResult() + ", " + wsAttributeAssignLookup.toStringCompact());
294         }
295         
296         i++;
297       }
298       
299     }
300     return attributeAssignIds;
301   }

305    * make sure this is an explicit toString
306    * @return return a compact to string
307    */
308   public String toStringCompact() {
309     if (!StringUtils.isBlank(this.uuid)) {
310       return "id: " + this.uuid;
311     }
312     if (!StringUtils.isBlank(this.batchIndex)) {
313       return "batchIndex: " + this.batchIndex;
314     }
315     return "blank";
316   }
