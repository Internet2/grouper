35  /**

45    /**
46     * see if blank
47     * @return true if blank
48     */
49    public boolean blank() {
50      return StringUtils.isBlank(this.identifier)
51        && this.externalSubject == null && this.externalSubjectFindResult == null;
52    }
53  
54    
55    /**
56     * see if this attributeDef lookup has data
57     * @return true if it has data
58     */
59    public boolean hasData() {
60      return !StringUtils.isBlank(this.identifier);
61    }

63    /**
64     * logger 
65     */
66    private static final Log LOG = LogFactory.getLog(WsExternalSubjectLookup.class);

68    /** find the external subject */
69    @XStreamOmitField
70    private ExternalSubject externalSubject = null;
71    
72    /** result of extneral subject find */
73    public static enum ExternalSubjectFindResult {
74  
75      /** found the attributeDef */
76      SUCCESS {
77  
78        @Override
79        public WsExternalSubjectDeleteResultCode convertToExternalSubjectDeleteResultCode() {
80          return WsExternalSubjectDeleteResultCode.SUCCESS;
81        }
82      },
83  
84      /** cant find the externalSubject */
85      EXTERNAL_SUBJECT_NOT_FOUND {
86  
87        @Override
88        public WsExternalSubjectDeleteResultCode convertToExternalSubjectDeleteResultCode() {
89          return WsExternalSubjectDeleteResultCode.SUCCESS_EXTERNAL_SUBJECT_NOT_FOUND;
90        }
91      },
92  
93      /** invalid query (e.g. if everything blank) */
94      INVALID_QUERY {
95  
96        @Override
97        public WsExternalSubjectDeleteResultCode convertToExternalSubjectDeleteResultCode() {
98          return WsExternalSubjectDeleteResultCode.INVALID_QUERY;
99        }
100     };
101 
102     /**
103      * if this is a successful result
104      * @return true if success
105      */
106     public boolean isSuccess() {
107       return this == SUCCESS;
108     }
109 
110     /**
111      * convert this code to a delete code
112      * @return the code
113      */
114     public abstract WsExternalSubjectDeleteResultCode convertToExternalSubjectDeleteResultCode();
115 
116     /**
117      * null safe equivalent to convertToDeleteCode
118      * @param externalSubjectFindResult to convert
119      * @return the code
120      */
121     public static WsExternalSubjectDeleteResultCode convertToExternalSubjectDeleteCodeStatic(
122         ExternalSubjectFindResult externalSubjectFindResult) {
123       return externalSubjectFindResult == null ? WsExternalSubjectDeleteResultCode.EXCEPTION
124           : externalSubjectFindResult.convertToExternalSubjectDeleteResultCode();
125     }

167   }

169   /**
170    * retrieve the external subject for this lookup if not looked up yet.  pass in a grouper session
171    * @param grouperSession 
172    * @param invalidQueryReason is the text to go in the WsInvalidQueryException
173    * @return the attributeDef
174    * @throws WsInvalidQueryException if there is a problem, and if the invalidQueryReason is set
175    */
176   public ExternalSubject retrieveExternalSubjectIfNeeded(GrouperSession grouperSession,
177       String invalidQueryReason) throws WsInvalidQueryException {
178     
179     //see if we already retrieved
180     if (this.externalSubjectFindResult != null) {
181       return this.externalSubject;
182     }
183     try {
184       //assume success (set otherwise if there is a problem)
185       this.externalSubjectFindResult = ExternalSubjectFindResult.SUCCESS;
186 
187       boolean hasIdentifier = !StringUtils.isBlank(this.identifier);
188 
189       //must have a name or uuid
190       if (!hasIdentifier) {
191         this.externalSubjectFindResult = ExternalSubjectFindResult.INVALID_QUERY;
192         if (!StringUtils.isEmpty(invalidQueryReason)) {
193           throw new WsInvalidQueryException("Invalid externalSubject query for '"
194               + invalidQueryReason + "', " + this);
195         }
196         String logMessage = "Invalid query: " + this;
197         LOG.warn(logMessage);
198       }
199 
200       ExternalSubject theExternalSubject = null;
201       
202       if (hasIdentifier) {
203         
204         theExternalSubject = GrouperDAOFactory.getFactory().getExternalSubject().findByIdentifier(this.identifier, true, null);
205             //new QueryOptions().secondLevelCache(false));
206 
207       }
208 
209       //success
210       this.externalSubject = theExternalSubject;
211 
212     } catch (Exception e) {
213       this.externalSubjectFindResult = ExternalSubjectFindResult.EXTERNAL_SUBJECT_NOT_FOUND;
214       if (!StringUtils.isBlank(invalidQueryReason)) {
215         throw new WsInvalidQueryException("Invalid externalSubject for '" + invalidQueryReason
216             + "', " + this, e);
217       }
218     }
219     return this.externalSubject;
220   }
221   
222   /**
223    * clear the attributeDef if a setter is called
224    */
225   private void clearExternalSubject() {
226     this.externalSubject = null;
227   }

230   private String identifier;

234   private ExternalSubjectFindResult externalSubjectFindResult = null;

236   /**
237    * name of the attributeDef to find (includes stems, e.g. stem1:stem2:attributeDef
238    * @return the theName
239    */
240   public String getIdentifier() {
241     return this.identifier;
242   }
243 
244   /**
245    * name of the attributeDef to find (includes stems, e.g. stem1:stem2:attributeDef
246    * @param theName the theName to set
247    */
248   public void setIdentifier(String theName) {
249     this.identifier = theName;
250     this.clearExternalSubject();
251   }

253   /**
254    * make sure this is an explicit toString
255    * @return return a compact to string
256    */
257   public String toStringCompact() {
258     if (!StringUtils.isBlank(this.identifier)) {
259       return "name: " + this.identifier;
260     }
261     return "blank";
262   }
