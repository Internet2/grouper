114   /**
115    * validate the settings (e.g. that booleans are set correctly)
116    */
117   public void validate() {
118     try {
119       if (!StringUtils.isBlank(this.saveMode)) {
120         //make sure it exists
121         SaveMode.valueOfIgnoreCase(this.saveMode);
122       }
123       if (!this.getWsAttributeDef().areThereAnyAssignables()) {
124         throw new WsInvalidQueryException("select at least one object type that this attribute type can be assigned to "); 
125       }
126     } catch (RuntimeException e) {
127       throw new WsInvalidQueryException("Problem with save mode: " + e.getMessage()
128           + ", " + this, e);
129     }
130   }

132   /**
133    * save this attributeDef
134    * 
135    * @param grouperSession
136    *            to save
137    * @return the attributeDef that was inserted or updated
138    * @throws StemNotFoundException 
139    * @throws InsufficientPrivilegeException
140    */
141   public AttributeDef save(GrouperSession grouperSession) {
142 
143     AttributeDef attributeDef = null;
144 
145     SaveMode theSaveMode = SaveMode.valueOfIgnoreCase(this.saveMode);
146 
147     if (this.getWsAttributeDef() == null) {
148       throw new WsInvalidQueryException(
149           "getWsAttributeDef is required to save an attributeDef");
150     }
151     if (!this.getWsAttributeDef().areThereAnyAssignables()) {
152       throw new WsInvalidQueryException(
153           "atleast one assignable to is required.");
154     }
155 
156     if (this.getWsAttributeDefLookup() == null
157         || this.getWsAttributeDefLookup().blank()) {
158       WsAttributeDefLookup theWsAttributeDefLookup = this
159           .getWsAttributeDefLookup() == null ? new WsAttributeDefLookup()
160               : this.getWsAttributeDefLookup();
161 
162       if (!StringUtils.isBlank(this.getWsAttributeDef().getUuid())) {
163         theWsAttributeDefLookup.setUuid(this.getWsAttributeDef().getUuid());
164       } else if (!StringUtils.isBlank(this.getWsAttributeDef().getName())) {
165         theWsAttributeDefLookup.setName(this.getWsAttributeDef().getName());
166       }
167 
168       if (SaveMode.INSERT != theSaveMode || !theWsAttributeDefLookup.blank()) {
169         this.setWsAttributeDefLookup(theWsAttributeDefLookup);
170       }
171 
172     }
173 
174     if (SaveMode.INSERT != theSaveMode && this.getWsAttributeDefLookup() == null) {
175       throw new WsInvalidQueryException(
176           "wsAttributeDefLookup is required to save an attributeDef (probably just put the name in it)");
177     }
178 
179     this.getWsAttributeDefLookup().retrieveAttributeDefIfNeeded(grouperSession);
180 
181     AttributeDef attributeDefLookedup = this.getWsAttributeDefLookup()
182         .retrieveAttributeDef();
183 
184     String attributeDefLookup = attributeDefLookedup == null ? null
185         : attributeDefLookedup.getName();
186 
187     AttributeDefType attributeDefType = AttributeDefType.valueOfIgnoreCase(this
188         .getWsAttributeDef().getAttributeDefType(), true);
189     AttributeDefValueType attributeDefValueType = AttributeDefValueType
190         .valueOfIgnoreCase(this.getWsAttributeDef().getValueType(), true);
191     boolean multiAssignable = GrouperUtil.booleanValue(this.getWsAttributeDef()
192         .getMultiAssignable(), false);
193     boolean multiValued = GrouperUtil.booleanValue(this.getWsAttributeDef()
194         .getMultiValued(), false);
195 
196     AttributeDefSave attributeDefSave = new AttributeDefSave(grouperSession);
197     attributeDefSave.assignAttributeDefNameToEdit(attributeDefLookup);
198     attributeDefSave.assignId(this.getWsAttributeDef().getUuid()).assignName(
199         this.getWsAttributeDef().getName());
200     attributeDefSave.assignDescription(this.getWsAttributeDef().getDescription());
201     attributeDefSave.assignSaveMode(theSaveMode);
202     attributeDefSave.assignCreateParentStemsIfNotExist(GrouperUtil.booleanValue(
203         this.getCreateParentStemsIfNotExist(), false));
204     attributeDefSave.assignAttributeDefType(attributeDefType);
205     attributeDefSave.assignValueType(attributeDefValueType);
206     attributeDefSave.assignMultiAssignable(multiAssignable);
207     attributeDefSave.assignMultiValued(multiValued);
208     attributeDefSave.assignToAttributeDef(this.getWsAttributeDef().getAssignToAttributeDef().equals("T"));
209     attributeDefSave.assignToAttributeDefAssn(this.getWsAttributeDef().getAssignToAttributeDefAssignment().equals("T"));
210     attributeDefSave.assignToStem(this.getWsAttributeDef().getAssignToStem().equals("T"));
211     attributeDefSave.assignToStemAssn(this.getWsAttributeDef().getAssignToStemAssignment().equals("T"));
212     attributeDefSave.assignToGroup(this.getWsAttributeDef().getAssignToGroup().equals("T"));
213     attributeDefSave.assignToGroupAssn(this.getWsAttributeDef().getAssignToGroupAssignment().equals("T"));
214     attributeDefSave.assignToMember(this.getWsAttributeDef().getAssignToMember().equals("T"));
215     attributeDefSave.assignToMemberAssn(this.getWsAttributeDef().getAssignToMemberAssignment().equals("T"));
216     attributeDefSave.assignToEffMembership(this.getWsAttributeDef().getAssignToEffectiveMembership().equals("T"));
217     attributeDefSave.assignToEffMembershipAssn(this.getWsAttributeDef().getAssignToEffectiveMembershipAssignment().equals("T"));
218     attributeDefSave.assignToImmMembership(this.getWsAttributeDef().getAssignToImmediateMembership().equals("T"));
219     attributeDefSave.assignToImmMembershipAssn(this.getWsAttributeDef().getAssignToImmediateMembershipAssignment().equals("T"));
220 
221     if (!StringUtils.isBlank(this.getWsAttributeDef().getIdIndex())) {
222       attributeDefSave.assignIdIndex(GrouperUtil.longValue(this.getWsAttributeDef()
223           .getIdIndex()));
224     }
225 
226     attributeDef = attributeDefSave.save();
227 
228     this.saveResultType = attributeDefSave.getSaveResultType();
229 
230     return attributeDef;
231   }
