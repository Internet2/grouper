57    /**
58     * assign the code from the enum
59     * @param wsSubjectLookup1
60     * @param theSubjectAttributeNames
61     */
62    public void processSubject(WsSubjectLookup wsSubjectLookup1,
63        String[] theSubjectAttributeNames) {
64  
65      SubjectFindResult subjectFindResult = wsSubjectLookup1.retrieveSubjectFindResult();
66  
67      switch (subjectFindResult) {
68        case INVALID_QUERY:
69          this.assignResultCode(WsGetGrouperPrivilegesLiteResultCode.INVALID_QUERY);
70          break;
71        case SOURCE_UNAVAILABLE:
72          this.assignResultCode(WsGetGrouperPrivilegesLiteResultCode.EXCEPTION);
73          break;
74        case SUBJECT_DUPLICATE:
75          this.assignResultCode(WsGetGrouperPrivilegesLiteResultCode.SUBJECT_DUPLICATE);
76          break;
77        case SUBJECT_NOT_FOUND:
78          this.assignResultCode(WsGetGrouperPrivilegesLiteResultCode.SUBJECT_NOT_FOUND);
79          break;
80        case SUCCESS:
81          return;
82      }
83  
84      this.getResultMetadata().setResultMessage(
85          "Subject: " + wsSubjectLookup1 + " had problems: " + subjectFindResult);
86  
87    }

101   /**
102    * prcess an exception, log, etc
103    * @param wsMemberChangeSubjectLiteResultCodeOverride
104    * @param theError
105    * @param e
106    */
107   public void assignResultCodeException(
108       WsGetGrouperPrivilegesLiteResultCode wsMemberChangeSubjectLiteResultCodeOverride, 
109       String theError, Exception e) {
110 
111     if (e instanceof WsInvalidQueryException) {
112       wsMemberChangeSubjectLiteResultCodeOverride = GrouperUtil.defaultIfNull(
113           wsMemberChangeSubjectLiteResultCodeOverride, WsGetGrouperPrivilegesLiteResultCode.INVALID_QUERY);
114       if (e.getCause() instanceof StemNotFoundException) {
115         wsMemberChangeSubjectLiteResultCodeOverride = WsGetGrouperPrivilegesLiteResultCode.STEM_NOT_FOUND;
116       }
117       if (e.getCause() instanceof GroupNotFoundException) {
118         wsMemberChangeSubjectLiteResultCodeOverride = WsGetGrouperPrivilegesLiteResultCode.GROUP_NOT_FOUND;
119       }
120       if (e.getCause() instanceof SubjectNotFoundException) {
121         wsMemberChangeSubjectLiteResultCodeOverride = WsGetGrouperPrivilegesLiteResultCode.SUBJECT_NOT_FOUND;
122       }
123       if (e.getCause() instanceof SubjectNotUniqueException) {
124         wsMemberChangeSubjectLiteResultCodeOverride = WsGetGrouperPrivilegesLiteResultCode.SUBJECT_DUPLICATE;
125       }
126       if (e.getCause() instanceof InsufficientPrivilegeException) {
127         wsMemberChangeSubjectLiteResultCodeOverride = WsGetGrouperPrivilegesLiteResultCode.INSUFFICIENT_PRIVILEGES;
128       }
129       //a helpful exception will probably be in the getMessage()
130       this.assignResultCode(wsMemberChangeSubjectLiteResultCodeOverride);
131       this.getResultMetadata().appendResultMessage(e.getMessage());
132       this.getResultMetadata().appendResultMessage(theError);
133       LOG.warn(e);
134 
135     } else {
136       wsMemberChangeSubjectLiteResultCodeOverride = GrouperUtil.defaultIfNull(
137           wsMemberChangeSubjectLiteResultCodeOverride, WsGetGrouperPrivilegesLiteResultCode.EXCEPTION);
138       LOG.error(theError, e);
139 
140       theError = StringUtils.isBlank(theError) ? "" : (theError + ", ");
141       this.getResultMetadata().appendResultMessage(
142           theError + ExceptionUtils.getFullStackTrace(e));
143       this.assignResultCode(wsMemberChangeSubjectLiteResultCodeOverride);
144 
145     }
146   }
