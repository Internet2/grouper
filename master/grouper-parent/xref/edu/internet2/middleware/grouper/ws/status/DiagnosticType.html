54  /**
55   * type of diagnostics to run (trivial, deep, etc)
56   * 
57   * @author mchyzer
58   */
59  public enum DiagnosticType {
60  
61    /**
62     * just do a trivial memory only test
63     */
64    TRIVIAL {
65  
66      /**
67       * @see DiagnosticType#appendDiagnostics(List)
68       */
69      @Override
70      public void appendDiagnostics(Set<DiagnosticTask> diagnosticsTasks) {
71        diagnosticsTasks.add(new DiagnosticMemoryTest());
72      }
73    },
74    
75    /**
76     * just do the trivial plus the database check
77     */
78    DB {
79  
80      /**
81       * @see DiagnosticType#appendDiagnostics(List)
82       */
83      @Override
84      public void appendDiagnostics(Set<DiagnosticTask> diagnosticsTasks) {
85        TRIVIAL.appendDiagnostics(diagnosticsTasks);
86        diagnosticsTasks.add(new DiagnosticDbTest());
87      }
88    },
89    
90    /**
91     * do the DB test plus check the sources
92     */
93    SOURCES {
94  
95      /**
96       * @see DiagnosticType#appendDiagnostics(List)
97       */
98      @Override
99      public void appendDiagnostics(Set<DiagnosticTask> diagnosticsTasks) {
100       DB.appendDiagnostics(diagnosticsTasks);
101 
102       Collection<Source> sources = SourceManager.getInstance().getSources();
103       
104       for (Source source : sources) {
105         
106         diagnosticsTasks.add(new DiagnosticSourceTest(source.getId()));
107         
108       }
109       
110     }
111   },
112   
113   /**
114    * do the sources test plus the jobs
115    */
116   ALL {
117 
118     /**
119      * @see DiagnosticType#appendDiagnostics(List)
120      */
121     @SuppressWarnings("unchecked")
122     @Override
123     public void appendDiagnostics(Set<DiagnosticTask> diagnosticsTasks) {
124       SOURCES.appendDiagnostics(diagnosticsTasks);
125       
126       diagnosticsTasks.add(new DiagnosticLoaderJobTest("CHANGE_LOG_changeLogTempToChangeLog", GrouperLoaderType.CHANGE_LOG));
127 
128       String emailTo = GrouperLoaderConfig.getPropertyString("daily.report.emailTo");
129       String reportDirectory = GrouperLoaderConfig.getPropertyString("daily.report.saveInDirectory");
130       
131       if (!StringUtils.isBlank(emailTo) ||  !StringUtils.isBlank(reportDirectory)) {
132         diagnosticsTasks.add(new DiagnosticLoaderJobTest("MAINTENANCE__grouperReport", GrouperLoaderType.MAINTENANCE));
133       }
134       diagnosticsTasks.add(new DiagnosticLoaderJobTest("MAINTENANCE_cleanLogs", GrouperLoaderType.MAINTENANCE));
135 
136       {
137         //expand these out
138         Map<String, String> consumerMap = GrouperLoaderConfig.retrieveConfig().propertiesMap( 
139             GrouperCheckConfig.grouperLoaderConsumerPattern);
140         
141         for (String consumerKey : consumerMap.keySet()) {
142 
143           //get the name
144           Matcher matcher = GrouperCheckConfig.grouperLoaderConsumerPattern.matcher(consumerKey);
145           matcher.matches();
146           String consumerName = matcher.group(1);
147 
148           //diagnosticsTasks.add(new DiagnosticLoaderJobTest("CHANGE_LOG_consumer_xmpp", GrouperLoaderType.CHANGE_LOG));
149           diagnosticsTasks.add(new DiagnosticLoaderJobTest("CHANGE_LOG_consumer_" + consumerName, GrouperLoaderType.CHANGE_LOG));
150 
151         }
152       }
153 
154       {
155         GroupType groupType = GroupTypeFinder.find("grouperLoader", false);
156         if (groupType != null) {
157           Set<Group> groupSet = sourceCache.get(GrouperLoaderType.SQL_SIMPLE);
158           
159           if (groupSet == null) {
160 
161             groupSet = GroupFinder.findAllByType(GrouperSession.staticGrouperSession(), groupType);
162 
163             sourceCache.put(GrouperLoaderType.SQL_SIMPLE, groupSet);
164             
165           }
166           
167           for (Group group : groupSet) {
168             
169             String grouperLoaderType = group.getAttributeValue(GrouperLoader.GROUPER_LOADER_TYPE, false, false);
170             
171             GrouperLoaderType grouperLoaderTypeEnum = GrouperLoaderType.valueOfIgnoreCase(grouperLoaderType, true);
172     
173             String jobName = grouperLoaderTypeEnum.name() + "__" + group.getName() + "__" + group.getUuid();
174             
175             //diagnosticsTasks.add(new DiagnosticLoaderJobTest("SQL_SIMPLE__penn:sas:query:faculty:adjunct_faculty__707d94fa3aaa4ef88051a144b74bac77", GrouperLoaderType.SQL_SIMPLE));
176             //diagnosticsTasks.add(new DiagnosticLoaderJobTest("SQL_GROUP_LIST__penn:community:student:loader:studentPrimaryGroups__93750690c3474b41b349bbd196167d3e", GrouperLoaderType.SQL_GROUP_LIST));
177             
178             diagnosticsTasks.add(new DiagnosticLoaderJobTest(jobName, grouperLoaderTypeEnum));
179             
180           }
181           
182         }
183         
184       }
185       
186       {
187         String attrRootStem = GrouperConfig.getProperty("grouper.attribute.rootStem");
188         if (!StringUtils.isBlank(attrRootStem)) {
189           AttributeDefName attributeDefName = AttributeDefNameFinder.findByName(
190               GrouperCheckConfig.attributeLoaderStemName() + ":attributeLoader", false);
191           
192           //see if attributeDef
193           if (attributeDefName != null) {
194             
195             //lets get the attributeDefs which have this type
196             Set<AttributeDef> attributeDefs = sourceCache.get(GrouperLoaderType.ATTR_SQL_SIMPLE);
197             
198             if (attributeDefs == null) {
199     
200               attributeDefs = GrouperDAOFactory.getFactory().getAttributeAssign()
201                 .findAttributeDefsByAttributeDefNameId(attributeDefName.getId());
202     
203               sourceCache.put(GrouperLoaderType.ATTR_SQL_SIMPLE, attributeDefs);
204               
205             }
206     
207             for (AttributeDef attributeDef : attributeDefs) {
208               
209                //lets get all attribute values
210                String grouperLoaderType = attributeDef.getAttributeValueDelegate().retrieveValueString(GrouperCheckConfig.attributeLoaderStemName() + ":" + GrouperLoader.ATTRIBUTE_LOADER_TYPE);
211                 
212                GrouperLoaderType grouperLoaderTypeEnum = GrouperLoaderType.valueOfIgnoreCase(grouperLoaderType, true);
213         
214                String jobName = grouperLoaderTypeEnum.name() + "__" + attributeDef.getName() + "__" + attributeDef.getUuid();
215     
216                //diagnosticsTasks.add(new DiagnosticLoaderJobTest("ATTR_SQL_SIMPLE__penn:community:employee:orgPermissions:orgs__092bd6259d814b5db665f2f0f4ca7dc6", GrouperLoaderType.ATTR_SQL_SIMPLE));
217                diagnosticsTasks.add(new DiagnosticLoaderJobTest(jobName, grouperLoaderTypeEnum));
218             }
219           }
220         }
221       }
222       
223       {
224         AttributeDefName ldapLoaderAttributeDefName = LoaderLdapUtils.grouperLoaderLdapAttributeDefName(false);
225         if (ldapLoaderAttributeDefName != null) {
226           
227           //lets get the groups which have this type
228           Set<DiagnosticLoaderJobTest> diagnosticLoaderJobTests = sourceCache.get(GrouperLoaderType.LDAP_SIMPLE);
229           
230           if (diagnosticLoaderJobTests == null) {
231   
232             diagnosticLoaderJobTests = new LinkedHashSet<DiagnosticLoaderJobTest>();
233             Set<AttributeAssign> attributeAssigns = GrouperDAOFactory.getFactory().getAttributeAssign().findGroupAttributeAssignments(null, null, GrouperUtil.toSet(ldapLoaderAttributeDefName.getId()), null, null, true, false);
234   
235             for (AttributeAssign attributeAssign : attributeAssigns) {
236               
237               Group group = attributeAssign.getOwnerGroup();
238 
239               //lets get all attribute values
240               String grouperLoaderType = attributeAssign.getAttributeValueDelegate().retrieveValueString(LoaderLdapUtils.grouperLoaderLdapTypeName());
241                
242               GrouperLoaderType grouperLoaderTypeEnum = GrouperLoaderType.valueOfIgnoreCase(grouperLoaderType, true);
243        
244               String jobName = grouperLoaderTypeEnum.name() + "__" + group.getName() + "__" + group.getUuid();
245               
246               DiagnosticLoaderJobTest diagnosticLoaderJobTest = new DiagnosticLoaderJobTest(jobName, grouperLoaderTypeEnum);
247               
248               diagnosticLoaderJobTests.add(diagnosticLoaderJobTest);
249               
250             }
251  
252             //diagnosticsTasks.add(new DiagnosticLoaderJobTest("ATTR_SQL_SIMPLE__penn:community:employee:orgPermissions:orgs__092bd6259d814b5db665f2f0f4ca7dc6", GrouperLoaderType.ATTR_SQL_SIMPLE));
253             sourceCache.put(GrouperLoaderType.LDAP_SIMPLE, diagnosticLoaderJobTests);
254             
255           }
256   
257           for (DiagnosticLoaderJobTest diagnosticLoaderJobTest : diagnosticLoaderJobTests) {
258             
259              diagnosticsTasks.add(diagnosticLoaderJobTest);
260           }
261         }
262       }
263       
264       {
265         //do min size groups
266         Pattern groupNamePattern = Pattern.compile("^ws\\.diagnostic\\.checkGroupSize\\.(.+)\\.groupName$");
267         
268         Properties properties = GrouperWsConfig.retrieveConfig().properties();
269         for (String key : (Set<String>)(Object)properties.keySet()) {
270           Matcher groupNameMatcher = groupNamePattern.matcher(key);
271           if (groupNameMatcher.matches()) {
272             String configName = groupNameMatcher.group(1);
273             int minSize = Integer.parseInt(GrouperWsConfig.retrieveConfig().propertyValueString(
274                 "ws.diagnostic.checkGroupSize." + configName + ".minSize"));
275             String groupName = properties.getProperty(key);
276             diagnosticsTasks.add(new DiagnosticMinGroupSize(groupName, minSize));
277           }
278         }
279       }
280       
281     }
282   };

288   private static GrouperCache<GrouperLoaderType, Set> sourceCache = new GrouperCache<GrouperLoaderType, Set>("loaderSets", 100, false, 1200, 1200, false);

294   public abstract void appendDiagnostics(Set<DiagnosticTask> diagnosticsTasks);

296   /**
297    * do a case-insensitive matching
298    * 
299    * @param string
300    * @param exceptionOnNotFound true to throw exception if method not found
301    * @return the enum or null or exception if not found
302    * @throws GrouperRestInvalidRequest if there is a problem
303    */
304   public static DiagnosticType valueOfIgnoreCase(String string,
305       boolean exceptionOnNotFound) throws GrouperRestInvalidRequest {
306     return GrouperServiceUtils.enumValueOfIgnoreCase(DiagnosticType.class, 
307         string, exceptionOnNotFound);
308   }
