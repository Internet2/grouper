52       */
53    public void executeSample(WsSampleGeneratedType wsSampleGeneratedType) {
54      acknowledgeMessage(wsSampleGeneratedType);
55    }
56  
57    /**
58       * @param wsSampleGeneratedType can run as soap or xml/http
59       */
60    public static void acknowledgeMessage(WsSampleGeneratedType wsSampleGeneratedType) {
61      try {
62        //URL, e.g. http://localhost:8091/grouper-ws/services/GrouperService
63        GrouperServiceStub stub = new GrouperServiceStub(GeneratedClientSettings.URL);
64        Options options = stub._getServiceClient().getOptions();
65        HttpTransportProperties.Authenticator auth = new HttpTransportProperties.Authenticator();
66        auth.setUsername(GeneratedClientSettings.USER);
67        auth.setPassword(GeneratedClientSettings.PASS);
68        auth.setPreemptiveAuthentication(true);
69  
70        options.setProperty(HTTPConstants.AUTHENTICATE, auth);
71        options.setProperty(HTTPConstants.SO_TIMEOUT, new Integer(3600000));
72        options.setProperty(HTTPConstants.CONNECTION_TIMEOUT,
73            new Integer(3600000));
74  
75        GrouperSession.startRootSession();
76        
77        GrouperSession.start(SubjectFinder.findById("test.subject.0", true));
78  
79        GrouperMessagingEngine
80            .send(new GrouperMessageSendParam().assignQueueOrTopicName("def")
81                .addMessageBody("message body")
82                .assignQueueType(GrouperMessageQueueType.queue));
83  
84        GrouperMessageReceiveResult grouperMessageReceiveResult = GrouperMessagingEngine
85            .receive(new GrouperMessageReceiveParam().assignQueueName("def"));
86  
87        GrouperMessage grouperMessage = grouperMessageReceiveResult.getGrouperMessages()
88            .iterator().next();
89  
90        Acknowledge acknowledgeMessage = Acknowledge.class.newInstance();
91  
92        //version, e.g. v1_3_000
93        acknowledgeMessage.setClientVersion(GeneratedClientSettings.VERSION);
94        WsSubjectLookup wsSubjectLookup = new WsSubjectLookup();
95        wsSubjectLookup.setSubjectId("test.subject.0");
96        acknowledgeMessage.setActAsSubjectLookup(wsSubjectLookup);
97        acknowledgeMessage.setQueueOrTopicName("def");
98        acknowledgeMessage.setAcknowledgeType("mark_as_processed");
99        acknowledgeMessage.setMessageIds(new String[] { grouperMessage.getId() });
100 
101       AcknowledgeResponse acknowledgeResponse = stub.acknowledge(acknowledgeMessage);
102 
103       System.out.println(ToStringBuilder.reflectionToString(
104           acknowledgeResponse));
105 
106       WsMessageAcknowledgeResults wsMessageAcknowledgeResults = acknowledgeResponse
107           .get_return();
108 
109       if (wsMessageAcknowledgeResults.getQueueOrTopicName().equals("def")) {
110         throw new RuntimeException("didnt get correct queue or topic name! ");
111       }
112 
113       if (!StringUtils.equals("T",
114           wsMessageAcknowledgeResults.getResultMetadata().getSuccess())) {
115         throw new RuntimeException("didnt get success! ");
116       }
117 
118     } catch (Exception e) {
119       throw new RuntimeException(e);
120     }
121   }
