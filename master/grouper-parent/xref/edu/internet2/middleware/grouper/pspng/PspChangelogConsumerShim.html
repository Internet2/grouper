19  import java.util.ArrayList;
20  import java.util.List;
21  
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import org.slf4j.MDC;
25  
26  import edu.internet2.middleware.grouper.changeLog.ChangeLogConsumerBase;
27  import edu.internet2.middleware.grouper.changeLog.ChangeLogEntry;
28  import edu.internet2.middleware.grouper.changeLog.ChangeLogProcessorMetadata;
29

31  /** 
32   * This class connects a PSPNG provsioner with the changelog. This is only necessary
33   * until the PSPNG provisioners are written as daemons that pull events from queues.
34   * 
35   * @author Bert Bee-Lindgren
36   *
37   */
38  public class PspChangelogConsumerShim extends ChangeLogConsumerBase {
39    final private static Logger LOG = LoggerFactory.getLogger(PspChangelogConsumerShim.class);
40    
41    public PspChangelogConsumerShim() {
42      LOG.debug("Constructing PspngChangelogConsumerShim");
43    }
44    
45    @Override
46    public long processChangeLogEntries(List<ChangeLogEntry> changeLogEntryList,
47        ChangeLogProcessorMetadata changeLogProcessorMetadata) {
48      try {
49        String consumerName = changeLogProcessorMetadata.getConsumerName();
50        MDC.put("why", "CLog/");
51        MDC.put("who", consumerName+"/");
52        
53        Provisioner provisioner;
54        try {
55          provisioner = ProvisionerFactory.getProvisioner(consumerName);
56          
57          // Make sure the full syncer is also created and running
58          FullSyncProvisionerFactory.getFullSyncer(consumerName);
59        } catch (PspException e1) {
60          throw new RuntimeException("provisioner could not be created: " + e1.getMessage());
61        }
62        
63        List<ProvisioningWorkItem> workItems = new ArrayList<ProvisioningWorkItem>();
64        
65        LOG.warn("Processing changelog entries: " + changeLogEntryList.size() );
66        
67        for ( ChangeLogEntry entry : changeLogEntryList ) {
68          workItems.add(new ProvisioningWorkItem(entry));
69        }
70        
71        provisioner.provisionBatchOfItems(workItems);
72        
73        // This determines which changelog entries to acknowledge: The highest
74        // number entry before we have a failure
75        long lastSuccessfulChangelogEntry = -1;
76        int numSuccessfulWorkItems = 0;
77        int numFailedWorkItems = 0;
78        int numSuccessfulWorkItems_thatWillBeRetried=0;
79        
80        String firstErrorMessage = null;
81        
82        for( ProvisioningWorkItem workItem : workItems ) {
83          if ( workItem.wasSuccessful() ) {
84            numSuccessfulWorkItems++;
85            
86            // If we haven't seen a failure yet, then keep counting up the successes
87            if ( numFailedWorkItems == 0 )
88              lastSuccessfulChangelogEntry = workItem.getChangelogEntry().getSequenceNumber();
89            else
90              numSuccessfulWorkItems_thatWillBeRetried++;
91          }
92          else if ( firstErrorMessage == null ) {
93            numFailedWorkItems++;
94            firstErrorMessage = workItem.getStatusMessage();
95          }
96        }
97        
98        StringBuilder summary = new StringBuilder();
99        if ( firstErrorMessage != null ) 
100         summary.append(String.format("Summary: %d successes/%d failures.  ", numSuccessfulWorkItems, numFailedWorkItems));
101       
102       if ( numSuccessfulWorkItems_thatWillBeRetried > 0 )
103         summary.append(String.format("(%d successful entries will be retried because they follow a failure in the queue.) ",
104             numSuccessfulWorkItems_thatWillBeRetried));
105       
106       if ( firstErrorMessage != null )
107         summary.append(String.format("First error was: %s", firstErrorMessage));
108       
109       
110       if ( numFailedWorkItems > 0 )
111         LOG.warn("Provisioning summary: {}", summary);
112       else
113         LOG.info("Provisioning summary: {}", summary);
114       
115       changeLogProcessorMetadata.getHib3GrouperLoaderLog().appendJobMessage(summary.toString());
116       return lastSuccessfulChangelogEntry;
117     }
118     finally {
119       MDC.remove("why");
120       MDC.remove("who");
121     }
122   }
123 }
