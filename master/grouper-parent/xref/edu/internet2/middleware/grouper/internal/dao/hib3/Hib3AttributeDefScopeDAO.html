31  /**
32   * Data Access Object for attribute def scope
33   * @author  mchyzer
34   * @version $Id: Hib3AttributeDefScopeDAO.java,v 1.1 2009-06-29 15:58:24 mchyzer Exp $

38    /**
39     * 
40     */
41    @SuppressWarnings("unused")
42    private static final String KLASS = Hib3AttributeDefScopeDAO.class.getName();

44    /**
45     * reset the attribute def scopes
46     * @param hibernateSession
47     */
48    static void reset(HibernateSession hibernateSession) {
49      hibernateSession.byHql().createQuery("delete from AttributeDefScope").executeUpdate();
50    }
51  
52    /**
53     * retrieve by id
54     * @param id 
55     * @param exceptionIfNotFound 
56     * @return  attribute def scope
57     */
58    public AttributeDefScope findById(String id, boolean exceptionIfNotFound) {
59      AttributeDefScope attributeDefScope = HibernateSession.byHqlStatic().createQuery(
60          "from AttributeDefScope where id = :theId")
61        .setString("theId", id).uniqueResult(AttributeDefScope.class);
62      if (attributeDefScope == null && exceptionIfNotFound) {
63        throw new AttributeDefScopeNotFoundException("Cant find attribute def scope by id: " + id);
64     }
65  
66      return attributeDefScope;
67    }
68  
69    /**
70     * save or update
71     * @param attributeDefScope 
72     */
73    public void saveOrUpdate(AttributeDefScope attributeDefScope) {
74      HibernateSession.byObjectStatic().saveOrUpdate(attributeDefScope);
75    }

77    /**
78     * 
79     * @see edu.internet2.middleware.grouper.internal.dao.AttributeDefScopeDAO#findByUuidOrKey(java.util.Collection, java.lang.String, java.lang.String, java.lang.String, boolean, String)
80     */
81    public AttributeDefScope findByUuidOrKey(Collection<String> idsToIgnore, String id, String attributeDefId,
82        String attributeDefScopeType, boolean exceptionIfNull, String scopeString) throws GrouperDAOException {
83      try {
84        Set<AttributeDefScope> attributeDefScopes = HibernateSession.byHqlStatic()
85          .createQuery("from AttributeDefScope as theAttributeDefScope where " +
86          		"theAttributeDefScope.id = :theId or (theAttributeDefScope.attributeDefId = :theAttributeDefId and " +
87          		"theAttributeDefScope.attributeDefScopeTypeDb = :theAttributeDefScopeTypeDb)")
88          .setCacheable(true)
89          .setCacheRegion(KLASS + ".FindByUuidOrName")
90          .setString("theId", id)
91          .setString("theAttributeDefId", attributeDefId)
92          .setString("theAttributeDefScopeTypeDb", attributeDefScopeType)
93          .listSet(AttributeDefScope.class);
94        if (GrouperUtil.length(attributeDefScopes) == 0) {
95          if (exceptionIfNull) {
96            throw new RuntimeException("Can't find attributeDefScope by id: '" + id + "' or attributeDefId '" + attributeDefId 
97                + "', attributeDefScopeType: " + attributeDefScopeType);
98          }
99          return null;
100       }
101       
102       idsToIgnore = GrouperUtil.nonNull(idsToIgnore);
103       
104       //lets remove ones we have already processed or will process
105       Iterator<AttributeDefScope> iterator = attributeDefScopes.iterator();
106       while (iterator.hasNext()) {
107         
108         AttributeDefScope attributeDefScope = iterator.next();
109         if (idsToIgnore.contains(attributeDefScope.getId())) {
110           iterator.remove();
111         }
112       }
113       
114       //first case, the ID matches
115       iterator = attributeDefScopes.iterator();
116       while (iterator.hasNext()) {
117         
118         AttributeDefScope attributeDefScope = iterator.next();
119         if (StringUtils.equals(id, attributeDefScope.getId())) {
120           return attributeDefScope;
121         }
122       }
123 
124       //second case, the value matches
125       iterator = attributeDefScopes.iterator();
126       while (iterator.hasNext()) {
127         
128         AttributeDefScope attributeDefScope = iterator.next();
129         if (StringUtils.equals(scopeString, attributeDefScope.getScopeString())) {
130           return attributeDefScope;
131         }
132       }
133       
134       //ok, if there is one left, return it
135       if (attributeDefScopes.size() > 0) {
136         return attributeDefScopes.iterator().next();
137       }
138       
139       //cant find one
140       return null;
141     }
142     catch (GrouperDAOException e) {
143       String error = "Problem find attributeDefScope by id: '" + id + "' or attributeDefId '" + attributeDefId 
144             + "', attributeDefScopeType: " + attributeDefScopeType 
145             + "', scopeString: " + scopeString + ", " + e.getMessage();
146       throw new GrouperDAOException( error, e );
147     }
148 
149   }

151   /**
152    * @see edu.internet2.middleware.grouper.internal.dao.AttributeDefScopeDAO#saveUpdateProperties(edu.internet2.middleware.grouper.attr.AttributeDefScope)
153    */
154   public void saveUpdateProperties(AttributeDefScope attributeDefScope) {
155     //run an update statement since the business methods affect these properties
156     HibernateSession.byHqlStatic().createQuery("update AttributeDefScope " +
157         "set hibernateVersionNumber = :theHibernateVersionNumber, " +
158         "contextId = :theContextId, " +
159         "createdOnDb = :theCreatedOnDb, " +
160         "lastUpdatedDb = :theLastUpdatedDb " +
161         "where id = :theId")
162         .setLong("theHibernateVersionNumber", attributeDefScope.getHibernateVersionNumber())
163         .setLong("theCreatedOnDb", attributeDefScope.getCreatedOnDb())
164         .setLong("theLastUpdatedDb", attributeDefScope.getLastUpdatedDb())
165         .setString("theContextId", attributeDefScope.getContextId())
166         .setString("theId", attributeDefScope.getId()).executeUpdate();
167   }

169   /**
170    * @see edu.internet2.middleware.grouper.internal.dao.AttributeDefScopeDAO#findByAttributeDefId(java.lang.String, QueryOptions)
171    */
172   public Set<AttributeDefScope> findByAttributeDefId(String attributeDefId, QueryOptions queryOptions) {
173     Set<AttributeDefScope> attributeDefScopes = HibernateSession.byHqlStatic()
174       .createQuery("from AttributeDefScope as theAttributeDefScope where " +
175           "theAttributeDefScope.attributeDefId = :theAttributeDefId")
176       .options(queryOptions)
177       .setCacheable(true)
178       .setCacheRegion(KLASS + ".FindByUuidOrName")
179       .setString("theAttributeDefId", attributeDefId)
180       .listSet(AttributeDefScope.class);
181     return attributeDefScopes;
182   }
