273   /**
274    * @see edu.internet2.middleware.grouper.GrouperAPI#onPreUpdate(edu.internet2.middleware.grouper.hibernate.HibernateSession)
275    */
276   @Override
277   public void onPreUpdate(HibernateSession hibernateSession) {
278     super.onPreUpdate(hibernateSession);
279     
280     // add change log entry for permissions
281     if (!this.isActive() && this.dbVersion().isActive() && this.getDepth() == 1 && this.getNotificationsForRolesWithPermissionChangesOnSaveOrUpdate()) {
282       Set<ChangeLogEntry> changeLogEntryBatch = new LinkedHashSet<ChangeLogEntry>();
283       int batchSize = GrouperConfig.getHibernatePropertyInt("hibernate.jdbc.batch_size", 20);
284       if (batchSize <= 0) {
285         batchSize = 1;
286       }
287 
288       Set<PITGroup> roles = GrouperDAOFactory.getFactory().getPITGroup().findRolesWithPermissionsContainingObject(this);
289 
290       for (PITGroup role : roles) {
291         ChangeLogEntry changeLogEntry = new ChangeLogEntry(false, ChangeLogTypeBuiltin.PERMISSION_CHANGE_ON_ROLE,
292             ChangeLogLabels.PERMISSION_CHANGE_ON_ROLE.roleId.name(), role.getSourceId(),
293             ChangeLogLabels.PERMISSION_CHANGE_ON_ROLE.roleName.name(), role.getName());
294             
295         changeLogEntry.setContextId(this.getContextId());
296         changeLogEntry.setCreatedOnDb(this.getStartTimeDb());
297         changeLogEntryBatch.add(changeLogEntry);
298         if (changeLogEntryBatch.size() % batchSize == 0) {
299           GrouperDAOFactory.getFactory().getChangeLogEntry().saveBatch(changeLogEntryBatch, false);
300           changeLogEntryBatch.clear();
301         }
302       }
303       
304       // make sure all changes get made      
305       if (changeLogEntryBatch.size() > 0) {
306         GrouperDAOFactory.getFactory().getChangeLogEntry().saveBatch(changeLogEntryBatch, false);
307         changeLogEntryBatch.clear();
308       }
309     }
310   }

386   /**
387    * @see edu.internet2.middleware.grouper.GrouperAPI#onPreDelete(edu.internet2.middleware.grouper.hibernate.HibernateSession)
388    */
389   @Override
390   public void onPreDelete(HibernateSession hibernateSession) {
391     super.onPreDelete(hibernateSession);
392 
393     if (this.isActive()) {
394       throw new RuntimeException("Cannot delete active point in time action set object with id=" + this.getId());
395     }
396     
397     // Note that not all action sets that exist because of this action set are deleted by this.
398     // We're assuming that action sets only get deleted when the action is deleted.  
399     // So some action sets get deleted separately.
400 
401     Set<PITAttributeAssignActionSet> childResults = GrouperDAOFactory.getFactory().getPITAttributeAssignActionSet().findImmediateChildren(this);
402 
403     for (PITAttributeAssignActionSet child : childResults) {
404       GrouperDAOFactory.getFactory().getPITAttributeAssignActionSet().delete(child);
405     }
406   }
