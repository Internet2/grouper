93     * get the fields that this field implies by inheritance
94     * @return the fields, note, dont change the list after you get it
95     */
96    public Collection<Field> getImpliedFields() {
97  
98      if (this.isPrivilege()) {
99        Privilege privilege = Privilege.listToPriv(this.name, true);
100       return Privilege.convertPrivilegesToFields(privilege.getImpliedPrivileges());
101     }
102     
103     throw new RuntimeException("Not expecting field: " + this.name);
104   }

126   /**
127    * see if there are inherited privileges to also include
128    * @return the inherited fields
129    */
130   public static Collection<Field> calculateInheritedPrivileges(Collection<Field> fields, boolean includeInheritedPrivileges) {
131     if (!includeInheritedPrivileges || GrouperUtil.length(fields) == 0) {
132       return fields;
133     }
134     
135     Set<Field> additionalFields = new HashSet<Field>();
136     
137     for (Field field : GrouperUtil.nonNull(fields)) {
138       
139       if (field.isAttributeDefListField() || field.isGroupAccessField() || field.isStemListField()) {
140         Privilege privilege = Privilege.listToPriv(field.getName(), true);
141         Collection<Privilege> privileges = privilege.getInheritedPrivileges();
142         Collection<Field> theFields = Privilege.convertPrivilegesToFields(privileges);
143         additionalFields.addAll(theFields);
144       }
145       
146     }
147     
148     return additionalFields;
149   }

232   /**
233    * print out a collection of fields
234    * @param collection
235    * @return the field names comma separated
236    */
237   public static String fieldNames(Collection<Field> collection) {
238     StringBuilder result = new StringBuilder();
239     for (Field field : GrouperUtil.nonNull(collection)) {
240       result.append(field.getName()).append(", ");
241     }
242     if (result.length() >= 2) {
243       //take off the last comma and space
244       result.delete(result.length()-2, result.length());
245     }
246     return result.toString();
247   }

584   /**
585    * @see edu.internet2.middleware.grouper.GrouperAPI#onPostSave(edu.internet2.middleware.grouper.hibernate.HibernateSession)
586    */
587   @Override
588   public void onPostSave(HibernateSession hibernateSession) {
589     super.onPostSave(hibernateSession);
590     
591     // add group sets
592     if (this.type.equals("list") && !Group.getDefaultList().getUuid().equals(this.getUuid())) {
593       Set<Group> groups = GrouperDAOFactory.getFactory().getGroup().findAllByType(this.getGroupType());
594       for (Group group : groups) {
595         if (group.getTypeOfGroup() != null && group.getTypeOfGroup().supportsField(this)) {
596           GroupSet groupSet = new GroupSet();
597           groupSet.setId(GrouperUuid.getUuid());
598           groupSet.setCreatorId(GrouperSession.staticGrouperSession().getMemberUuid());
599           groupSet.setDepth(0);
600           groupSet.setMemberGroupId(group.getUuid());
601           groupSet.setOwnerGroupId(group.getUuid());
602           groupSet.setParentId(groupSet.getId());
603           groupSet.setFieldId(this.getUuid());
604           GrouperDAOFactory.getFactory().getGroupSet().save(groupSet);
605         }
606       }
607     }
608     
609     GrouperHooksUtils.callHooksIfRegistered(this, GrouperHookType.FIELD, 
610         FieldHooks.METHOD_FIELD_POST_INSERT, HooksFieldBean.class, 
611         this, Field.class, VetoTypeGrouper.FIELD_POST_INSERT, true, false);
612 
613     //do these second so the right object version is set, and dbVersion is ok
614     GrouperHooksUtils.schedulePostCommitHooksIfRegistered(GrouperHookType.FIELD, 
615         FieldHooks.METHOD_FIELD_POST_COMMIT_INSERT, HooksFieldBean.class, 
616         this, Field.class);
617 
618 
619   }

639    */
640   @Override
641   public void onPreDelete(HibernateSession hibernateSession) {
642     super.onPreDelete(hibernateSession);
643     
644     // remove group sets
645     if (this.type.equals("list") && !Group.getDefaultList().getUuid().equals(this.getUuid())) {
646       Set<Group> groups = GrouperDAOFactory.getFactory().getGroup().findAllByType(this.getGroupType());
647       for (Group group : groups) {
648         GrouperDAOFactory.getFactory().getGroupSet().deleteSelfByOwnerGroupAndField(group.getUuid(), this.getUuid());
649       }
650     }
651     
652     GrouperHooksUtils.callHooksIfRegistered(this, GrouperHookType.FIELD, 
653         FieldHooks.METHOD_FIELD_PRE_DELETE, HooksFieldBean.class, 
654         this, Field.class, VetoTypeGrouper.FIELD_PRE_DELETE, false, false);
655   
656     //change log into temp table
657     new ChangeLogEntry(true, ChangeLogTypeBuiltin.GROUP_FIELD_DELETE, 
658         ChangeLogLabels.GROUP_FIELD_DELETE.id.name(), 
659         this.getUuid(), ChangeLogLabels.GROUP_FIELD_DELETE.name.name(), 
660         this.getName(), null, 
661         null,
662         null, 
663         null,
664         ChangeLogLabels.GROUP_FIELD_DELETE.type.name(), this.getTypeString()
665     ).save();
666   }

936   /**
937    * add a field if it is not already there
938    * @param s
939    * @param name
940    * @param type
941    * @param read
942    * @param write
943    * @param exceptionIfExists
944    * @param updateIfExists 
945    * @param changedArray is an array of 1 if you want to know if this method changed anything, else null
946    * @param uuid 
947    * @return the field
948    * @throws InsufficientPrivilegeException
949    * @throws SchemaException
950    */
951   public static Field internal_addField(
952     final GrouperSession s, final String name, final FieldType type, final Privilege read, 
953     final Privilege write, final boolean exceptionIfExists, final boolean updateIfExists,
954     final boolean[] changedArray, String uuid) throws  InsufficientPrivilegeException, SchemaException {
955 
956     //these are reserved words:
957     if (Group.INTERNAL_FIELD_ATTRIBUTES.contains(name)) {
958       throw new RuntimeException("You cannot add a field which is a reserved word '" 
959           + name + "', reserved words are : " + GrouperUtil.toStringForLog(Group.INTERNAL_FIELD_ATTRIBUTES));
960     }
961     
962     final String UUID = StringUtils.isBlank(uuid) ? GrouperUuid.getUuid() : uuid;
963     
964     return (Field)HibernateSession.callbackHibernateSession(
965         GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, AuditControl.WILL_AUDIT, new HibernateHandler() {
966 
967       public Object callback(HibernateHandlerBean hibernateHandlerBean)
968           throws GrouperDAOException {
969         
970         try {
971           //note, no need for GrouperSession inverse of control
972           StopWatch sw  = new StopWatch();
973           sw.start();
974           AddFieldToGroupTypeValidator v = AddFieldToGroupTypeValidator.validate(name, !exceptionIfExists);
975           if (v.isInvalid()) {
976             throw new SchemaException( v.getErrorMessage() );
977           }
978           Field field = FieldFinder.find(name, false);
979 
980           if (field != null) {
981             boolean changed = false;
982             if (!type.equals(field.getType())) {
983               //dont want to change types, that could be bad!
984               throw new SchemaException("field '" + name + "' does not have type: " + type + ", it has: " + field.getType());
985             }
986             if (!read.equals(field.getReadPriv())) {
987               if (exceptionIfExists) {
988                 throw new SchemaException("field '" + name + "' does not have read privilege: " + read + ", it has: " + field.getReadPrivilege());
989               }
990               if (updateIfExists) {
991                 changed = true;
992                 field.setReadPrivilege(read);
993               }
994             }
995             if (!write.equals(field.getWritePriv())) {
996               if (exceptionIfExists) {
997                 throw new SchemaException("field '" + name + "' does not have write privilege: " + write + ", it has: " + field.getWritePrivilege());
998               }
999               if (updateIfExists) {
1000                 changed = true;
1001                 field.setWritePrivilege(write);
1002               }
1003             }
1004             if (exceptionIfExists) {
1005               throw new SchemaException("field exists: '" + name + "'");
1006             }
1007             //store minor changes to db
1008             if (changed && updateIfExists) {
1009               changed = true;
1010               
1011               String differences = GrouperUtil.dbVersionDescribeDifferences(field.dbVersion(), 
1012                   field, field.dbVersion() != null ? field.dbVersionDifferentFields() : Field.CLONE_FIELDS);
1013               
1014               GrouperDAOFactory.getFactory().getField().createOrUpdate(field);
1015   
1016               if (!hibernateHandlerBean.isCallerWillCreateAudit()) {
1017                 //audit the update
1018                 AuditEntry auditEntry = new AuditEntry(AuditTypeBuiltin.GROUP_FIELD_UPDATE, "id", 
1019                     field.getUuid(), "name", field.getName(), "groupTypeId", null, 
1020                     "groupTypeName", null, "type", type.getType());
1021                 
1022                 String description = "Updated group field: " + name + ", id: " + field.getUuid() 
1023                     + ", type: " + type + ".\n" + differences;
1024                 auditEntry.setDescription(description);
1025                 
1026                 auditEntry.saveOrUpdate(true);
1027               }
1028               
1029               if (GrouperUtil.length(changedArray) > 0) {
1030                 changedArray[0] = true;
1031               }
1032             } else {
1033               if (GrouperUtil.length(changedArray) > 0) {
1034                 changedArray[0] = false;
1035               }
1036             }
1037             FieldFinder.internal_updateKnownFields();
1038             return field;
1039           }
1040           if (GrouperUtil.length(changedArray) > 0) {
1041             changedArray[0] = true;
1042           }
1043           try {
1044             field = new Field();
1045             field.setName(name);
1046             field.setReadPrivilege(read);
1047             field.setType(type);
1048             field.setUuid(UUID);
1049             field.setWritePrivilege(write);
1050               
1051             GrouperDAOFactory.getFactory().getField().createOrUpdate(field);
1052             
1053             sw.stop();
1054             EventLog.info(
1055               s, 
1056               M.GROUPTYPE_ADDFIELD + Quote.single(field.getName()) + " ftype=" + Quote.single(type.toString()),
1057               sw
1058             );
1059             FieldFinder.internal_updateKnownFields();
1060           }
1061           catch (GrouperDAOException eDAO) {
1062             String msg = E.GROUPTYPE_FIELDADD + name + ": " + eDAO.getMessage();
1063             LOG.error( msg);
1064             throw new SchemaException(msg, eDAO);
1065           }
1066           
1067           
1068           //only audit if actually changed the type
1069           AuditEntry auditEntry = new AuditEntry(AuditTypeBuiltin.GROUP_FIELD_ADD, "id", 
1070               field.getUuid(), "name", field.getName(), "groupTypeId", null, "groupTypeName", null, "type", type.getType());
1071           auditEntry.setDescription("Added group field: " + name + ", id: " + field.getUuid() + ", type: " + type);
1072           auditEntry.saveOrUpdate(true);
1073           
1074           return field;
1075         } catch (GrouperDAOException eDAO) {
1076           String msg = E.GROUPTYPE_FIELDADD + name + ": " + eDAO.getMessage();
1077           LOG.error( msg);
1078           throw new SchemaException(msg, eDAO);
1079         }
1080       }
1081       
1082     });
