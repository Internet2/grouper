59      group_addGroup {
60  
61        public void process(ChangeLogEntry changeLogEntry,
62            ChangeLogConsumerBaseImpl consumer) {
63          // does this event pertain to us? was the group or one of its parent folders marked for sync
64          final String groupName = changeLogEntry
65              .retrieveValueForLabel(ChangeLogLabels.GROUP_ADD.name);
66          final Group group = GroupFinder.findByName(
67              GrouperSession.staticGrouperSession(false), groupName, false);
68          if (group != null) {
69            if (consumer.isGroupMarkedForSync(groupName)) {
70              if (consumer.debugLog != null) {
71                consumer.debugLog.put("callingAddGroup", true);
72              }
73              consumer.addGroup(group, changeLogEntry);
74            } else {
75              if (consumer.debugLog != null) {
76                consumer.debugLog.put("skippingAddGroupNotMarked", true);
77                consumer.debugLog.put("notMarked", true);
78              }
79            }
80          } else {
81            if (consumer.debugLog != null) {
82              consumer.debugLog.put("skippingAddGroup", true);
83              consumer.debugLog.put("groupDeleted", true);
84            }
85          }
86        }

88      group_updateGroup {
89  
90        public void process(ChangeLogEntry changeLogEntry,
91            ChangeLogConsumerBaseImpl consumer) {
92          final String propertyChanged = changeLogEntry
93              .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.propertyChanged);
94          if (consumer.debugLog != null) {
95            consumer.debugLog.put("propertyChanged", propertyChanged);
96          }
97          // a group "move" is a change in group name and shows up in change log as a group update
98          if ("name".equals(changeLogEntry
99              .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.propertyChanged))) {
100           // group was moved to a new folder, get the new name and the old name
101           final String newGroupName = changeLogEntry
102               .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.propertyNewValue);
103           final String oldGroupName = changeLogEntry
104               .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.propertyOldValue);
105           if (consumer.debugLog != null) {
106             consumer.debugLog.put("oldGroupName", oldGroupName);
107             consumer.debugLog.put("newGroupName", newGroupName);
108           }
109 
110           // if newGroupName is marked for sync then rename or add group and membership at target
111           // this covers three cases:
112           //  1) directly marked group moved
113           //  2) indirectly marked group moved and is still indirectly marked
114           //  3) unmarked group move and is now marked (indirectly)
115           if (consumer.isGroupMarkedForSync(newGroupName)) {
116             if (consumer.debugLog != null) {
117               consumer.debugLog.put("callingRenameGroup", true);
118               consumer.debugLog.put("oldGroupName", oldGroupName);
119               consumer.debugLog.put("newGroupName", newGroupName);
120             }
121             consumer.renameGroup(oldGroupName, newGroupName, changeLogEntry);
122           } else {
123             // newGroupName not marked, check oldGroupName folders for indirect mark, if so delete at target, otherwise no-op
124 
125             // get the parent folder path name from some:parent:path:oldGroupName
126             final int lastPathSeparator = oldGroupName.lastIndexOf(":");
127             final String oldGroupNameFolderPath = oldGroupName.substring(0,
128                 lastPathSeparator);
129             final Stem oldGroupParentFolder = StemFinder
130                 .findByName(GrouperSession.staticGrouperSession(false),
131                     oldGroupNameFolderPath, false);
132             if (oldGroupParentFolder != null
133                 && oldGroupParentFolder.getAttributeDelegate()
134                     .hasAttributeOrAncestorHasAttribute(consumer.syncAttribute.getName(),
135                         false)) {
136               // oldGroupName previously marked, so remove from target
137               if (consumer.debugLog != null) {
138                 consumer.debugLog.put("callingRemoveMovedGroup", true);
139               }
140               consumer.removeMovedGroup(oldGroupName, changeLogEntry);
141             } else {
142               // could find moved parent folder (already deleted?) or not previously marked, no-op in either case
143               if (consumer.debugLog != null) {
144                 consumer.debugLog.put("skippingUpdateGroup", true);
145                 consumer.debugLog.put("noParentOrNotMarked", true);
146               }
147             }
148           }
149         } else {
150           // some other group property update; description, parentStemId, displayName, etc
151           final String groupName = changeLogEntry
152               .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.name);
153           if (consumer.debugLog != null) {
154             consumer.debugLog.put("skipingUpdateGroup", true);
155             consumer.debugLog.put("invalidPropertyChanged", true);
156             consumer.debugLog.put("groupName", groupName);
157           }
158         }
159       }

298       /**
299        * On assignment of the syncAttribute marker, create all the groups or group (if directly assigned)
300        * and any current memberships at the target.
301        */
302       public void process(ChangeLogEntry changeLogEntry,
303           ChangeLogConsumerBaseImpl consumer) {
304         // check if this is our syncAttribute, otherwise nothing to do.
305         final String attributeDefNameName = changeLogEntry
306             .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_ADD.attributeDefNameName);
307         if (consumer.syncAttribute.getName().equals(attributeDefNameName)) {
308           // syncAttribute applied to a group, then create the group at the target
309           String assignType = changeLogEntry
310               .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_ADD.assignType);
311           String ownerId1 = changeLogEntry
312               .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_ADD.ownerId1);
313           if (consumer.debugLog != null) {
314             consumer.debugLog.put("assignType", assignType);
315             consumer.debugLog.put("ownerId", ownerId1);
316           }
317           if (GROUP.equals(assignType)) {
318             Group markedGroup = GroupFinder.findByUuid(
319                 GrouperSession.staticGrouperSession(false), ownerId1, false);
320             if (markedGroup != null) {
321               if (consumer.debugLog != null) {
322                 consumer.debugLog.put("callingAddGroupAndMemberships", true);
323                 consumer.debugLog.put("groupName", markedGroup.getName());
324               }
325               consumer.addGroupAndMemberships(markedGroup, changeLogEntry);
326             } else { // couldn't find group, already deleted?
327               if (consumer.debugLog != null) {
328                 consumer.debugLog.put("skippingAddGroup", true);
329                 consumer.debugLog.put("cantFindGroup", true);
330               }
331             }
332           } else if (FOLDER.equals(assignType)) {
333             // syncAttribute applied to a folder, get all the groups below this folder and sub folders and create them at the target
334             Stem markedFolder = StemFinder.findByUuid(
335                 GrouperSession.staticGrouperSession(false), ownerId1, false);
336             if (markedFolder != null) {
337               Set<Group> markedGroups = markedFolder.getChildGroups(Stem.Scope.SUB);
338               for (Group group : markedGroups) {
339                 if (consumer.debugLog != null) {
340                   consumer.debugLog.put("folderName", markedFolder.getName());
341                   consumer.debugLog.put("callingAddGroupAndMemberships", true);
342                   consumer.debugLog.put("groupName", group.getName());
343                 }
344                 consumer.addGroupAndMemberships(group, changeLogEntry);
345               }
346             } else { // couldn't find folder, already deleted?
347               if (consumer.debugLog != null) {
348                 consumer.debugLog.put("folderId", ownerId1);
349                 consumer.debugLog.put("skippingAddGroupAndMemberships", true);
350                 consumer.debugLog.put("cantFindFolder", true);
351               }
352             }
353           }
354         } else {
355           // not our syncAttribute, so nothing to do
356           if (consumer.debugLog != null) {
357             consumer.debugLog.put("skippingAddAttribute", true);
358           }
359         }
360       }

364       /**
365        * On removal of the syncAttribute marker, delete all the groups or group (if directly assigned) at the target, unless
366        * otherwise still marked by direct assignment or a parent folder.
367        */
368       public void process(ChangeLogEntry changeLogEntry,
369           ChangeLogConsumerBaseImpl consumer) {
370         // is this our syncAttribute? otherwise nothing to do.
371         final String attributeDefNameName = changeLogEntry
372             .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_DELETE.attributeDefNameName);
373         if (consumer.syncAttribute.getName().equals(attributeDefNameName)) {
374           String assignType = changeLogEntry
375               .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_DELETE.assignType);
376           String ownerId1 = changeLogEntry
377               .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_DELETE.ownerId1);
378           // is it for a group? then delete the group at the target, unless a parent folder is also marked.
379           if (consumer.debugLog != null) {
380             consumer.debugLog.put("assignType", assignType);
381             consumer.debugLog.put("ownerId", ownerId1);
382           }
383           if (GROUP.equals(assignType)) {
384             // get the group and then check for a marked parent folder before deleting at target.
385             Group group = GroupFinder.findByUuid(
386                 GrouperSession.staticGrouperSession(false), ownerId1, false);
387             if (group != null) {
388               consumer.debugLog.put("groupName", group.getName());
389               // case when group had a direct syncAttribute marker assignment removed, does it still have a parent marker?
390               // need to flush edu.internet2.middleware.grouper.attr.assign.AttributeAssignBaseDelegate.objectHasAttributeCache to make sure syncAttribute is cleared
391               EhcacheController
392                   .ehcacheController()
393                   .getGrouperCache(
394                       "edu.internet2.middleware.grouper.attr.assign.AttributeAssignBaseDelegate.objectHasAttributeCache")
395                   .clear();
396               if (group.getAttributeDelegate().hasAttributeOrAncestorHasAttribute(
397                   consumer.syncAttribute.getName(), false)) {
398                 if (consumer.debugLog != null) {
399                   consumer.debugLog.put("skippingDeleteAttribute", true);
400                   consumer.debugLog.put("markedByParentFolder", true);
401                 }
402               } else {
403                 // marker syncAttribute removed from group and no other parent folder marked so delete at target
404                 if (consumer.debugLog != null) {
405                   consumer.debugLog.put("callingRemoveGroup", true);
406                   consumer.debugLog.put("markedByGroup", true);
407                 }
408                 consumer.removeGroup(group, changeLogEntry);
409               }
410             } else {
411               // case when a group which had a direct syncAttribute marker was deleted, always delete at target
412               PITGroup pitGroup = PITGroupFinder.findBySourceId(ownerId1, false)
413                   .iterator().next();
414               if (pitGroup != null) {
415                 String pitGroupName = pitGroup.getName();
416                 // marker syncAttribute removed when deleting a group, always delete at target
417                 if (consumer.debugLog != null) {
418                   consumer.debugLog.put("callingRemoveGroup", true);
419                   consumer.debugLog.put("deletedGroup", true);
420                   consumer.debugLog.put("pitGroupName", pitGroupName);
421                 }
422                 consumer.removeDeletedGroup(pitGroup, changeLogEntry);
423               } else {
424                 // couldn't find group anywhere? so can't determine its name.
425                 if (consumer.debugLog != null) {
426                   consumer.debugLog.put("skippingRemoveGroup", true);
427                   consumer.debugLog.put("cantFindGroup", true);
428                 }
429               }
430             }
431           } else if (FOLDER.equals(assignType)) {
432             // is it a folder, then delete all the containing groups at the target, unless they are still marked otherwise (direct or indirect)
433             Stem unMarkedFolder = StemFinder.findByUuid(
434                 GrouperSession.staticGrouperSession(false), ownerId1, false);
435             if (unMarkedFolder != null) {
436               if (consumer.debugLog != null) {
437                 consumer.debugLog.put("folderName", unMarkedFolder.getName());
438               }
439               // get all the groups below this folder and sub folders and to see if they are still marked, otherwise delete them at the target
440               Set<Group> unMarkedGroups = unMarkedFolder.getChildGroups(Stem.Scope.SUB);
441               // need to flush edu.internet2.middleware.grouper.attr.assign.AttributeAssignBaseDelegate.objectHasAttributeCache to make sure syncAttribute is cleared
442               EhcacheController
443                   .ehcacheController()
444                   .getGrouperCache(
445                       "edu.internet2.middleware.grouper.attr.assign.AttributeAssignBaseDelegate.objectHasAttributeCache")
446                   .clear();
447               for (Group group : unMarkedGroups) {
448                 // check that the group isn't marked directly or from some other parent folder
449                 if (consumer.isGroupMarkedForSync(group.getName())) {
450                   if (consumer.debugLog != null) {
451                     consumer.debugLog.put("skippingRemoveGroup", true);
452                     consumer.debugLog.put("foundMarkForGroup", true);
453                   }
454                 } else {
455                   if (consumer.debugLog != null) {
456                     consumer.debugLog.put("callingRemoveGroup", true);
457                     consumer.debugLog.put("groupName", group.getName());
458                   }
459                   consumer.removeGroup(group, changeLogEntry);
460                 }
461               }
462             } else {
463               // can't delete a folder without first deleting child objects, so there should be nothing to delete at the target
464               // deleting a folder with an attribute assign will cause a deleteAttributeAssign to come after folder already deleted.
465               if (consumer.debugLog != null) {
466                 consumer.debugLog.put("skippingRemoveGroup", true);
467                 consumer.debugLog.put("deletedFolder", true);
468                 consumer.debugLog.put("folderId", true);
469               }
470             }
471           }
472         }
473       }

555   /**
556    * If syncAttribute was applied to the group or one of the parent folders return true
557    * Method keeps an internal cache of results per run in markedFoldersAndGroups
558    * Will also check the PIT for recently deleted groups
559    */
560   private boolean isGroupMarkedForSync(String groupName) {
561 
562     boolean markedForSync = false;
563 
564     String markedForSyncString = markedFoldersAndGroups.get(groupName);
565     // have we seen this group already in this run
566     if (!StringUtils.isBlank(markedForSyncString)) {
567       markedForSync = markedForSyncString.equals(MARKED);
568       if (this.debugLog != null) {
569         this.debugLog.put("foundGroupInMarkedCache", true);
570         this.debugLog.put("markedForSync", markedForSync);
571       }
572       return markedForSync;
573     }
574 
575     // looking for the group
576     final Group group = GroupFinder.findByCurrentName(
577         GrouperSession.staticGrouperSession(false), groupName, false);
578 
579     if (group != null) {
580       // is it marked with the syncAttribute?
581       markedForSync = group.getAttributeDelegate().hasAttributeOrAncestorHasAttribute(
582           syncAttribute.getName(), false);
583       if (this.debugLog != null) {
584         this.debugLog.put("foundGroup", true);
585       }
586     } else {
587       // looking for the deleted group in the PIT
588       PITGroup pitGroup = PITGroupFinder.findMostRecentByName(groupName, false);
589       if (pitGroup != null) {
590         if (this.debugLog != null) {
591           this.debugLog.put("foundPitGroup", true);
592         }
593         // looking for syncAttribute assignment in the PIT
594         Set<PITAttributeDefName> pitSyncAttributes = PITAttributeDefNameFinder
595             .findByName(syncAttribute.getName(), false, true);
596         PITAttributeDefName pitSyncAttribute = pitSyncAttributes.iterator().next();
597         //System.out.println(pitGroup.getStartTime() + " -> " + pitGroup.getEndTime());
598         Set<PITAttributeAssign> pitAttributeAssigns = PITAttributeAssignFinder
599             .findByOwnerPITGroupAndPITAttributeDefName(pitGroup, pitSyncAttribute,
600                 pitGroup.getEndTime(), pitGroup.getEndTime());
601         if (!pitAttributeAssigns.isEmpty()) {
602           markedForSync = true;
603         } else {
604           // only gets direct group assignments, check folder next if no direct mark
605           // check for folders for mark
606           String parentStemName = GrouperUtil.parentStemNameFromName(pitGroup.getName());
607           Stem folder = StemFinder.findByName(GrouperSession.staticGrouperSession(false),
608               parentStemName, false);
609           if (folder != null) {
610             markedForSync = folder.getAttributeDelegate()
611                 .hasAttributeOrAncestorHasAttribute(syncAttribute.getName(), false);
612             if (this.debugLog != null) {
613               this.debugLog.put("foundPitGroupFolder", true);
614             }
615           }
616         }
617       } else {
618         // couldn't find group anywhere including the PIT
619         if (this.debugLog != null) {
620           this.debugLog.put("couldntFindGroup", true);
621         }
622         markedForSync = false;
623       }
624     }
625 
626     if (this.debugLog != null) {
627       debugLog.put("markedForSync", markedForSync);
628     }
629 
630     // remember this for next time
631     if (markedForSync) {
632       markedFoldersAndGroups.put(groupName, MARKED);
633       return true;
634     } 
635     markedFoldersAndGroups.put(groupName, NOT_MARKED);
636     return false;
637   }

820   /**
821    * Call the method of the {@link ChangeLogEventType} enum which matches the {@link ChangeLogEntry} category and action (i.e. the change log event type).
822    *
823    * @param changeLogEntry the change log entry
824    */
825   private void processChangeLogEntry(ChangeLogEntry changeLogEntry) {
826 
827     if (LOG.isDebugEnabled()) {
828        this.debugLog = new LinkedHashMap<String, Object>();
829        this.debugLog.put("consumerName", consumerName);
830        this.debugLog.put("startingSequenceNumber", changeLogEntry.getSequenceNumber());
831        this.debugLog.put("method", "processChangeLogEntry");
832     }
833 
834     // construct key from changeLogEntry in the form of <category>_<action>
835     final String changeLogEventTypeKey = changeLogEntry.getChangeLogType()
836         .getChangeLogCategory() + "_"
837         + changeLogEntry.getChangeLogType().getActionName();
838 
839     if (this.debugLog != null) {
840       this.debugLog.put("changeLogEventTypeKey", changeLogEventTypeKey);
841     }
842     
843     ChangeLogEventType changeLogEventType = null;
844 
845     // look up method to map to call it
846     try {
847       try {
848         changeLogEventType = ChangeLogEventType.valueOf(changeLogEventTypeKey);
849       } catch (IllegalArgumentException e) {
850         if (this.debugLog != null) {
851           this.debugLog.put("invalidEvent", true);
852         }      
853       }
854       if (changeLogEventType != null) {
855         changeLogEventType.process(changeLogEntry, this);
856       }
857     } catch (RuntimeException e) {
858       if (this.debugLog != null) {
859         this.debugLog.put("error", GrouperUtil.getFullStackTrace(e));
860       }
861       throw e;
862     } finally {
863       if (this.debugLog != null) {
864         LOG.debug(GrouperUtil.mapToString(this.debugLog));
865       }
866     }
