54  
55    /**
56     * @see edu.internet2.middleware.grouper.privs.AccessResolver#stop()
57     */
58    public void stop() {
59      //not sure why this is here...
60      super.getDecoratedResolver().flushCache();
61    }

66    public void flushCache() {
67      super.getDecoratedResolver().flushCache();
68    }

72    /**
73     * 
74     */
75    private Subject root;
76  
77    /**
78     * @param resolver 
79     * @since   1.2.1
80     */
81    public GrouperSystemAccessResolver(AccessResolver resolver) {
82      super(resolver);
83      this.root = SubjectFinder.findRootSubject();
84    }
85  
86    /**
87     * @see     AccessResolver#getPrivileges(Group, Subject)
88     * @since   1.2.1
89     */
90    public Set<AccessPrivilege> getPrivileges(Group group, Subject subject)
91        throws IllegalArgumentException {
92      //2007-11-02 Gary Brown
93      //If you are the root user you automatically have
94      //all Access privileges exception OPTIN / OPTOUT
95      if (SubjectHelper.eq(this.root, subject)) {
96        Set<Privilege> privs = Privilege.getAccessPrivs();
97        Set<AccessPrivilege> accessPrivs = new HashSet<AccessPrivilege>();
98        AccessPrivilege ap = null;
99        for (Privilege p : privs) {
100         //Not happy about the klass but will do for now in the absence of a GrouperSession
101         if (!p.equals(AccessPrivilege.OPTIN) && !p.equals(AccessPrivilege.OPTOUT)) {
102           ap = new AccessPrivilege(group, subject, subject, p, GrouperAccessAdapter.class.getName(), false, null);
103           accessPrivs.add(ap);
104         }
105       }
106 
107       return accessPrivs;
108     }
109     return super.getDecoratedResolver().getPrivileges(group, subject);
110   }
111 
112   /**
113    * @see     AccessResolver#hasPrivilege(Group, Subject, Privilege)
114    * @since   1.2.1
115    */
116   public boolean hasPrivilege(Group group, Subject subject, Privilege privilege)
117       throws IllegalArgumentException {
118     if (SubjectHelper.eq(this.root, subject)) {
119       if (!privilege.equals(AccessPrivilege.OPTIN)
120           && !privilege.equals(AccessPrivilege.OPTOUT)) {
121         return true;
122       }
123       return false;
124     }
125     return super.getDecoratedResolver().hasPrivilege(group, subject, privilege);
126   }

151   /**
152    * @see edu.internet2.middleware.grouper.privs.AccessResolver#hqlFilterGroupsWhereClause(edu.internet2.middleware.subject.Subject, edu.internet2.middleware.grouper.hibernate.HqlQuery, java.lang.StringBuilder, String, Set)
153    */
154   public boolean hqlFilterGroupsWhereClause(Subject subject, HqlQuery hqlQuery,
155       StringBuilder hql, String groupColumn, Set<Privilege> privInSet) {
156 
157     if (SubjectHelper.eq(this.root, subject)) {
158       return false;
159     }
160 
161     AccessResolver decoratedResolver = super.getDecoratedResolver();
162     //System.out.println(decoratedResolver.getClass().getName());
163     //CachingAccessResolver
164     return decoratedResolver.hqlFilterGroupsWhereClause(subject, hqlQuery, hql,
165         groupColumn, privInSet);
166   }
167   /**
168    * @see edu.internet2.middleware.grouper.privs.AccessResolver#hqlFilterGroupsNotWithPrivWhereClause(edu.internet2.middleware.subject.Subject, edu.internet2.middleware.grouper.hibernate.HqlQuery, java.lang.StringBuilder, String, Privilege, boolean)
169    */
170   public boolean hqlFilterGroupsNotWithPrivWhereClause(Subject subject, HqlQuery hqlQuery,
171       StringBuilder hql, String groupColumn, Privilege privilege, boolean considerAllSubject) {
172 
173     if (SubjectHelper.eq(this.root, subject)) {
174       return false;
175     }
176 
177     AccessResolver decoratedResolver = super.getDecoratedResolver();
178     //System.out.println(decoratedResolver.getClass().getName());
179     //CachingAccessResolver
180     return decoratedResolver.hqlFilterGroupsNotWithPrivWhereClause(subject, hqlQuery, hql,
181         groupColumn, privilege, considerAllSubject);
182   }

199   }
