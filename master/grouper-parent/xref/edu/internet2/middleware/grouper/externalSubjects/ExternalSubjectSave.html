205   /**
206    * <pre>
207    * create or update a external subject.
208    * 
209    * Steps:
210    * 
211    * 1. Find the external subject by identifierToEdit
212    * 2. Internally set all the fields of the stem (no need to reset if already the same)
213    * 3. Store the external subject (insert or update) if needed
214    * 4. Return the external subject object
215    * 
216    * This runs in a tx so that if part of it fails the whole thing fails, and potentially the outer
217    * transaction too
218    * </pre>
219    * @return the external subject
220    * @throws InsufficientPrivilegeException
221    */
222   public ExternalSubject save() {
223 
224     //help with incomplete entries
225     if (StringUtils.isBlank(this.identifier)) {
226       this.identifier = this.identifierToEdit;
227     }
228 
229     //get from uuid since could be a rename
230     if (StringUtils.isBlank(this.identifierToEdit) && !StringUtils.isBlank(this.uuid)) {
231       ExternalSubject externalSubject = HibernateSession.byObjectStatic().load(ExternalSubject.class, this.uuid);
232       this.identifierToEdit = externalSubject.getUuid();
233     }
234     
235     if (StringUtils.isBlank(this.identifierToEdit)) {
236       this.identifierToEdit = this.identifier;
237     }
238     
239     //default to insert or update
240     ExternalSubjectSave.this.saveMode = (SaveMode)ObjectUtils.defaultIfNull(ExternalSubjectSave.this.saveMode, SaveMode.INSERT_OR_UPDATE);
241     final SaveMode SAVE_MODE = ExternalSubjectSave.this.saveMode;
242 
243     try {
244       //do this in a transaction
245       ExternalSubject externalSubject = (ExternalSubject)GrouperTransaction.callbackGrouperTransaction(new GrouperTransactionHandler() {
246   
247         @SuppressWarnings("cast")
248         @Override
249         public Object callback(GrouperTransaction grouperTransaction)
250             throws GrouperDAOException {
251           
252           grouperTransaction.setCachingEnabled(false);
253           
254           return (ExternalSubject)GrouperSession.callbackGrouperSession(ExternalSubjectSave.this.grouperSession, new GrouperSessionHandler() {
255 
256               @Override
257               public Object callback(GrouperSession theGrouperSession)
258                   throws GrouperSessionException {
259                 
260                 String identifierForError = GrouperUtil.defaultIfBlank(ExternalSubjectSave.this.identifierToEdit, ExternalSubjectSave.this.identifier);
261                 
262                 ExternalSubject theExternalSubject = null;
263 
264                 //see if update
265                 boolean isUpdate = SAVE_MODE.isUpdate(ExternalSubjectSave.this.identifierToEdit, ExternalSubjectSave.this.identifier);
266 
267                 theExternalSubject = GrouperDAOFactory.getFactory().getExternalSubject().findByIdentifier(ExternalSubjectSave.this.identifierToEdit, false, null);
268 
269                 if (theExternalSubject != null) {
270                   //while we are here, make sure uuid's match if passed in
271                   if (!StringUtils.isBlank(ExternalSubjectSave.this.uuid) && !StringUtils.equals(ExternalSubjectSave.this.uuid, theExternalSubject.getUuid())) {
272                     throw new RuntimeException("UUID external subject changes are not supported: new: " + ExternalSubjectSave.this.uuid + ", old: " 
273                         + theExternalSubject.getUuid() + ", " + identifierForError);
274                   }
275                   
276                 } else {
277                   
278                   if (SAVE_MODE.equals(SaveMode.INSERT_OR_UPDATE) || SAVE_MODE.equals(SaveMode.INSERT)) {
279                     isUpdate = false;
280                   } else {
281                     throw new RuntimeException("Cant update an external subject which doesnt already exist! " + ExternalSubjectSave.this.identifierToEdit);
282                   }
283                 }
284 
285                 //default
286                 ExternalSubjectSave.this.saveResultType = SaveResultType.NO_CHANGE;
287                 boolean needsSave = false;
288                 //if inserting
289                 if (!isUpdate) {
290 
291                   ExternalSubjectSave.this.saveResultType = SaveResultType.INSERT;
292                   needsSave = true;
293                   theExternalSubject = new ExternalSubject();
294                   theExternalSubject.setIdentifier(ExternalSubjectSave.this.identifierToEdit);
295 
296                 }
297                 
298                 //now compare and put all attributes (then store if needed)
299                 if (GrouperUtil.booleanValue(ExternalSubjectSave.this.enabled) != theExternalSubject.isEnabled()) {
300                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
301                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
302                   }
303                   theExternalSubject.setEnabled(ExternalSubjectSave.this.enabled);
304                   needsSave = true;
305                 }
306 
307                 if (!StringUtils.equals(StringUtils.trimToEmpty(ExternalSubjectSave.this.email), StringUtils.trimToEmpty(theExternalSubject.getEmail()))) {
308                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
309                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
310                   }
311                   theExternalSubject.setEmail(ExternalSubjectSave.this.email);
312                   needsSave = true;
313                 }
314 
315                 if (!StringUtils.equals(StringUtils.trimToEmpty(ExternalSubjectSave.this.institution), StringUtils.trimToEmpty(theExternalSubject.getInstitution()))) {
316                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
317                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
318                   }
319                   theExternalSubject.setInstitution(ExternalSubjectSave.this.institution);
320                   needsSave = true;
321                 }
322 
323                 if (!StringUtils.equals(StringUtils.trimToEmpty(ExternalSubjectSave.this.name), StringUtils.trimToEmpty(theExternalSubject.getName()))) {
324                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
325                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
326                   }
327                   theExternalSubject.setName(ExternalSubjectSave.this.name);
328                   needsSave = true;
329                 }
330                 if (!StringUtils.equals(StringUtils.trimToEmpty(ExternalSubjectSave.this.vettedEmailAddresses), StringUtils.trimToEmpty(theExternalSubject.getVettedEmailAddresses()))) {
331                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
332                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
333                   }
334                   theExternalSubject.setVettedEmailAddresses(ExternalSubjectSave.this.vettedEmailAddresses);
335                   needsSave = true;
336                 }
337 
338                 Map<String, ExternalSubjectAttribute> mapNewAttributeNameToObject = new HashMap<String, ExternalSubjectAttribute>();
339                 Map<String, ExternalSubjectAttribute> mapOldAttributeNameToObject = new HashMap<String, ExternalSubjectAttribute>();
340 
341                 if (!StringUtils.isBlank(theExternalSubject.getUuid())) {
342                   Set<ExternalSubjectAttribute> setOldAttributes = new HashSet<ExternalSubjectAttribute>();
343                   setOldAttributes.addAll(GrouperUtil.nonNull(theExternalSubject.retrieveAttributes()));
344                   for (ExternalSubjectAttribute externalSubjectAttribute : setOldAttributes) {
345                     mapOldAttributeNameToObject.put(externalSubjectAttribute.getAttributeSystemName(), externalSubjectAttribute);
346                   }
347                 }
348                 for (String attributeName : GrouperUtil.nonNull(ExternalSubjectSave.this.attributes).keySet()) {
349                   String newValue = ExternalSubjectSave.this.attributes.get(attributeName);
350                   {
351                     ExternalSubjectAttribute oldExternalSubjectAttribute = mapOldAttributeNameToObject.get(attributeName);
352                     if (oldExternalSubjectAttribute != null) {
353                       mapOldAttributeNameToObject.remove(attributeName);
354                       mapNewAttributeNameToObject.put(attributeName, oldExternalSubjectAttribute);
355                       if (StringUtils.equals(newValue, oldExternalSubjectAttribute.getAttributeValue())) {
356                         continue;
357                       }
358                       oldExternalSubjectAttribute.setAttributeValue(newValue);
359                       if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
360                         ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
361                       }
362                       needsSave = true;
363                       continue;
364                     }
365                   }
366                   ExternalSubjectAttribute externalSubjectAttribute = new ExternalSubjectAttribute();
367                   externalSubjectAttribute.setAttributeSystemName(attributeName);
368                   externalSubjectAttribute.setAttributeValue(newValue);
369                   mapNewAttributeNameToObject.put(attributeName, externalSubjectAttribute);
370                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
371                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
372                   }
373                   needsSave = true;
374                 }
375                 
376                 if (mapOldAttributeNameToObject.size() > 0) {
377                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
378                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
379                   }
380                   needsSave = true;
381                   for (ExternalSubjectAttribute externalSubjectAttribute : mapOldAttributeNameToObject.values()) {
382                     externalSubjectAttribute.setAttributeValue(null);
383                     mapNewAttributeNameToObject.put(externalSubjectAttribute.getAttributeSystemName(), externalSubjectAttribute);
384                   }
385                 }
386                 
387                 //only store once, include attributes
388                 if (needsSave) {
389                   theExternalSubject.store(new HashSet<ExternalSubjectAttribute>(mapNewAttributeNameToObject.values()), null, true, true, false);
390                 }
391                 
392 			          return theExternalSubject;
393               }
394           });
395         }
396       });
397       return externalSubject;
398     } catch (RuntimeException re) {
399       
400       GrouperUtil.injectInException(re, "Problem saving externalSubject: " + this.identifier + ", thread: " + Integer.toHexString(Thread.currentThread().hashCode()));
401       
402       Throwable throwable = re.getCause();
403 //      if (throwable instanceof StemNotFoundException) {
404 //        throw (StemNotFoundException)throwable;
405 //      }
406       if (throwable instanceof InsufficientPrivilegeException) {
407         throw (InsufficientPrivilegeException)throwable;
408       }
409       //must just be runtime
410       throw re;
411     }
