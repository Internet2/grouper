99    /**
100    * print where config is read from, to sys out and log warn
101    */
102   private static void printConfigOnce() {
103     
104     if (printedConfigLocation) {
105       return;
106     }
107 
108     printedConfigLocation = true;
109 
110     Properties properties = GrouperUtil.propertiesFromResourceName("grouper.properties");
111     
112     String displayMessageString = GrouperUtil.propertiesValue(properties, "configuration.display.startup.message");
113     
114     String grouperStartup = "Grouper starting up: " + versionTimestamp();
115     if (!GrouperUtil.booleanValue(displayMessageString, true)) {
116       //just log this to make sure we can
117       try {
118         LOG.warn(grouperStartup);
119       } catch (RuntimeException re) {
120         //this is bad, print to stderr rightaway (though might dupe)
121         System.err.println(GrouperUtil.LOG_ERROR);
122         re.printStackTrace();
123         throw new RuntimeException(GrouperUtil.LOG_ERROR, re);
124       }
125       
126       return;
127     }
128 
129     StringBuilder resultString = new StringBuilder();
130     resultString.append(grouperStartup + "\n");
131     File grouperPropertiesFile = GrouperUtil.fileFromResourceName("grouper.properties");
132     String propertiesFileLocation = grouperPropertiesFile == null ? "not found" 
133         : GrouperUtil.fileCanonicalPath(grouperPropertiesFile); 
134     resultString.append("grouper.properties read from: " + propertiesFileLocation + "\n");
135 
136     if (GrouperConfig.retrieveConfig().propertyValueBoolean("grouper.api.readonly", false)) {
137       resultString.append("grouper.api.readonly:         true\n");
138     }
139     
140     resultString.append("Grouper current directory is: " + new File("").getAbsolutePath() + "\n");
141     //get log4j file
142     File log4jFile = GrouperUtil.fileFromResourceName("log4j.properties");
143     String log4jFileLocation = log4jFile == null ? " [cant find log4j.properties]" :
144       GrouperUtil.fileCanonicalPath(log4jFile);
145     resultString.append("log4j.properties read from:   " + log4jFileLocation + "\n");
146     
147     resultString.append(GrouperUtil.logDirPrint());    
148     File hibPropertiesFile = GrouperUtil.fileFromResourceName("grouper.hibernate.properties");
149     String hibPropertiesFileLocation = hibPropertiesFile == null ? " [cant find grouper.hibernate.properties]" :
150       GrouperUtil.fileCanonicalPath(hibPropertiesFile);
151     resultString.append("grouper.hibernate.properties: " + hibPropertiesFileLocation + "\n");
152     
153     Properties grouperHibernateProperties = GrouperUtil.propertiesFromResourceName("grouper.hibernate.properties");
154     String url = StringUtils.trim(grouperHibernateProperties.getProperty("hibernate.connection.url"));
155     String user = StringUtils.trim(grouperHibernateProperties.getProperty("hibernate.connection.username"));
156     resultString.append("grouper.hibernate.properties: " + user + "@" + url + "\n");
157     String sourcesString = "problem with sources";
158     try {
159       sourcesString = SourceManager.getInstance().printConfig();
160     } catch (Exception e) {
161       LOG.error("problem with sources", e);
162     }
163     resultString.append(sourcesString);
164     System.out.println(resultString);
165     try {
166       if (!GrouperUtil.isPrintGrouperLogsToConsole()) {
167         LOG.warn(resultString);
168       } else {
169         //print something to log to make sure we can
170         LOG.warn(grouperStartup);
171       }
172     } catch (RuntimeException re) {
173       //this is bad, print to stderr rightaway (though might dupe)
174       System.err.println(GrouperUtil.LOG_ERROR);
175       re.printStackTrace();
176       throw new RuntimeException(GrouperUtil.LOG_ERROR, re);
177     }
178   }

203    * @return false if already started, true if this started it
204    */

206     try {
207       if (started) {
208         return false;
209       }
210       started = true;
211   
212       printConfigOnce();
213   
214       //check java version
215 //      if (GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.checkJavaVersion", true)) {
216 //        String javaVersion = System.getProperty("java.version");
217 //        if (javaVersion != null && !javaVersion.startsWith("1.6") && !javaVersion.startsWith("1.7")) {
218 //          String error = "Error: Java should be version 6 or 7 (1.6 or 1.7), but is detected as: " + javaVersion;
219 //          LOG.error(error);
220 //          System.out.println(error);
221 //        }
222 //      }
223 
224       //add in custom sources
225       SourceManager.getInstance().loadSource(SubjectFinder.internal_getGSA());
226       SourceManager.getInstance().loadSource(InternalSourceAdapter.instance());
227       
228       if (GrouperConfig.retrieveConfig().propertyValueBoolean("externalSubjects.autoCreateSource", true)) {
229         
230         SourceManager.getInstance().loadSource(ExternalSubjectAutoSourceAdapter.instance());
231         
232       }
233       
234 //      if (GrouperConfig.retrieveConfig().propertyValueBoolean("entities.autoCreateSource", true)) {
235 //        
236 //        SourceManager.getInstance().loadSource(EntitySourceAdapter.instance());
237 //        
238 //      }
239       
240       //dont print big classname, dont print nulls
241       ToStringBuilder.setDefaultStyle(new GrouperToStringStyle());
242 
243       //first check databases
244       
245       if (!ignoreCheckConfig) {
246         GrouperCheckConfig.checkGrouperDb();
247       }
248       
249       if (runDdlBootstrap) {
250         //first make sure the DB ddl is up to date
251         GrouperDdlUtils.bootstrap(false, false, false);
252       }
253 
254       if (!ignoreCheckConfig) {
255         //make sure configuration is ok
256         GrouperCheckConfig.checkConfig();
257       }
258       
259       //startup hooks
260       GrouperHooksUtils.fireGrouperStartupHooksIfNotFiredAlready();
261   
262       //register hib objects
263       Hib3DAO.initHibernateIfNotInitted();
264       
265       initData(true);
266       
267       boolean legacyAttributeMigrationIncomplete = GrouperDdlUtils.getTableCount("grouper_types_legacy", false) > 0;
268       
269       // skip for now if legacy attribute migration is not done
270       if (!legacyAttributeMigrationIncomplete) {
271         //init include exclude type
272         initIncludeExcludeType();
273         
274         //init loader types and attributes if configured
275         initLoaderType();
276         
277         //init membership lite config type
278         initMembershipLiteConfigType();
279       }
280       
281       // verify member search and sort config
282       verifyMemberSortAndSearchConfig();
283       
284       // verify id indexes
285       verifyTableIdIndexes();
286 
287       verifyUtf8andTransactions();
288       
289       finishedStartupSuccessfully = true;
290       
291       return true;
292     } catch (RuntimeException re) {
293       if (logErrorStatic) {
294         //NOTE, the caller might not handle this exception, so print now. 
295         //ALSO, the logger might not work, so print to stderr first
296         String error = "Couldnt startup grouper: " + re.getMessage();
297         System.err.println(error);
298         re.printStackTrace();
299         LOG.error(error, re);
300       }
301       throw re;
302     }
303   }

333   /**
334    * make sure grouper can handle utf8
335    */
336   public static void verifyUtf8andTransactionsHelper() {
337 
338     //hibernate not ok yet, try next time starting up
339     if (!GrouperDdlUtils.okToUseHibernate()) {
340       return;
341     }
342     
343     boolean detectTransactionProblems = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.detect.db.transaction.problems", true);
344 
345     boolean detectUtf8Problems = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.detect.utf8.problems", true);
346 
347     boolean detectUtf8FileProblems = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.detect.utf8.file.problems", true);
348     
349     boolean detectCaseSensitiveProblems = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.detect.db.caseSensitive.problems", true);
350 
351     boolean utfProblems = false;
352     
353     if (!detectUtf8Problems && !detectTransactionProblems && !detectCaseSensitiveProblems) {
354       return;
355     }
356 
357     final String someUtfString = "ٹٺٻټكلل";
358 
359     final String name = "grouperUtf_" + GrouperUuid.getUuid();
360     final String id = GrouperUuid.getUuid();
361     
362     if (detectUtf8FileProblems && !utfProblems) {
363       String theStringFromFile = null;
364       try {
365         theStringFromFile = GrouperUtil.readResourceIntoString("grouperUtf8.txt", false);
366       } catch (Exception e) {
367         String error = "Error: Cannot read string from resource grouperUtf8.txt";
368         LOG.error(error, e);
369         System.out.println(error);
370         e.printStackTrace();
371         utfProblems = true;
372       }
373       if (!utfProblems && !StringUtils.equals(theStringFromFile, someUtfString)) {
374         String error = "Error: Cannot properly read UTF string from resource: grouperUtf8.txt: '" + theStringFromFile
375             + "'";
376 
377         String fileEncoding = System.getProperty("file.encoding");
378         if (fileEncoding == null || !fileEncoding.toLowerCase().startsWith("utf")) {
379           error += ", make sure you pass in the JVM switch -Dfile.encoding=utf-8 (currently is '" 
380               + fileEncoding + "')";
381         }
382 
383         fileEncoding = GrouperConfig.retrieveConfig().propertyValueString("grouper.default.fileEncoding");
384         if (fileEncoding == null || !fileEncoding.toLowerCase().startsWith("utf")) {
385           error += ", make sure you have grouper.default.fileEncoding set to UTF-8 in the grouper.properties (or leave it out since the default should be UTF-8)";
386         }
387 
388         LOG.error(error);
389         System.out.println(error);
390         utfProblems = true;
391       }
392       if (!utfProblems & GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.utf8.success.message", false)) {
393         System.out.println("Grouper can read UTF characters correctly from files");
394       }
395     }
396 
397     //lets check case sensitive
398     if (detectCaseSensitiveProblems) {
399       //load the grouper object in app upper
400 
401       Hib3GrouperDdl grouperDdl = GrouperDdlUtils.retrieveDdlByNameFromDatabase("GROUPER");
402 
403       if (grouperDdl != null) {
404 
405         String error = "Error: Queries in your database seem to be case insensitive, "
406             + "this can be a problem for Grouper, if you are using MySQL you should use a bin collation";
407         LOG.error(error);
408         System.out.println(error);
409 
410       } else if (GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.db.caseSensitive.success.message", false)) {
411 
412         System.out.println("Your database can handle case sensitive queries correctly");
413 
414       }
415     }
416 
417     //this shouldnt exist, just make sure
418     GrouperDdlUtils.deleteUtfDdls();
419 
420     Hib3GrouperDdl grouperDdl = GrouperDdlUtils.storeAndReadUtfString(someUtfString, id, name);
421 
422     //lets check transactions
423     if (detectTransactionProblems) {
424       Hib3GrouperDdl grouperDdlNew = GrouperDdlUtils.retrieveDdlByIdFromDatabase(id);
425       if (grouperDdlNew != null) {
426 
427         String error = "Error: Your database does not seem to support transactions, Grouper requires a transactional database";
428         LOG.error(error);
429         System.out.println(error);
430 
431         //delete it again
432         GrouperDdlUtils.deleteDdlById(id);
433 
434       } else if (GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.transaction.success.message", false)) {
435         System.out.println("Your database can handle transactions correctly");
436       }
437     }
438 
439     if (grouperDdl == null) {
440       String error = "Error: Why is grouperDdl utf null???";
441       LOG.error(error);
442       System.out.println(error);
443       utfProblems = true;
444     } else if (!utfProblems) {
445 
446       if (!StringUtils.equals(grouperDdl.getHistory(), someUtfString)) {
447         String error = "Error: Cannot properly read UTF string from database: '" + grouperDdl.getHistory()
448             + "', make sure your database has UTF tables and perhaps a hibernate.connection.url in grouper.hibernate.properties";
449         LOG.error(error);
450         System.out.println(error);
451         utfProblems = true;
452       }
453     }
454     if (!utfProblems & GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.utf8.success.message", false)) {
455       System.out.println("Grouper and the database can handle UTF characters correctly");
456     }
457   }

474   /**
475    * verify that table id indexes
476    */
477   public static void verifyTableIdIndexes() {
478     if (!GrouperConfig.retrieveConfig().propertyValueBoolean("grouper.tableIndex.verifyOnStartup", true)) {
479       return;
480     }
481         
482     //lets see if there are any nulls
483     for (TableIndexType tableIndexType : TableIndexType.values()) {
484       
485       List<String> ids = HibernateSession.bySqlStatic().listSelect(
486           String.class, "select id from " + tableIndexType.tableName() + " where id_index is null order by id" , null);
487 
488       if (GrouperUtil.length(ids) > 0) {
489         LOG.error("Found " + GrouperUtil.length(ids) + " " + tableIndexType.name() + " records with null id_index... correcting... please wait...");
490         for (int i=0;i<GrouperUtil.length(ids); i++) {
491           HibernateSession.bySqlStatic().executeSql("update " + tableIndexType.tableName() 
492               + " set id_index = ? where id = ?", GrouperUtil.toListObject(TableIndex.reserveId(tableIndexType), ids.get(i)));
493           if (i+1 % 1000 == 0) {
494             LOG.warn("Updated " + (i-1) + "/" + GrouperUtil.length(ids) + " " + tableIndexType + " id_index records...");
495           }
496         }
497         LOG.warn("Finished " + GrouperUtil.length(ids) + " " + tableIndexType + " id_index records...");
498       }
499     }
500   }

601   /**
602    * init the include/exclude type if configured in the grouper.properties
603    */
604   public static void initIncludeExcludeType() {
605     
606     final boolean useGrouperIncludeExclude = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.use", false);
607     final boolean useGrouperRequireGroups = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.requireGroups.use", false);
608     
609     final String includeExcludeGroupTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.type.name");
610     final String requireGroupsTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.type.name");
611 
612     GrouperSession grouperSession = null;
613 
614     try {
615 
616       grouperSession = GrouperSession.startRootSession(false);
617 
618       GrouperSession.callbackGrouperSession(grouperSession, new GrouperSessionHandler() {
619 
620         public Object callback(GrouperSession grouperSession)
621             throws GrouperSessionException {
622           try {
623             
624             @SuppressWarnings("unused")
625             GroupType includeExcludeGroupType = useGrouperIncludeExclude ? 
626                 GroupType.createType(grouperSession, includeExcludeGroupTypeName, false) : null;
627 
628             GroupType requireGroupsType = useGrouperRequireGroups ? 
629                 GroupType.createType(grouperSession, requireGroupsTypeName, false) : null;
630 
631             //first the requireGroups
632             String attributeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.attributeName");
633 
634             if (useGrouperRequireGroups && !StringUtils.isBlank(attributeName)) {
635               requireGroupsType.addAttribute(grouperSession,attributeName, false);
636             }
637 
638             if (useGrouperRequireGroups) {
639               //add types/attributes from grouper.properties
640               int i=0;
641               while (true) {
642                 String propertyName = "grouperIncludeExclude.requireGroup.name." + i;
643                 String attributeOrTypePropertyName = "grouperIncludeExclude.requireGroup.attributeOrType." + i;
644 
645                 String propertyValue = GrouperConfig.retrieveConfig().propertyValueString(propertyName);
646                 if (StringUtils.isBlank(propertyValue)) {
647                   break;
648                 }
649                 String attributeOrTypeValue = GrouperConfig.retrieveConfig().propertyValueString(attributeOrTypePropertyName);
650                 boolean attributeOrType = StringUtils.equals("attribute", attributeOrTypeValue);
651                 if (attributeOrType) {
652                   requireGroupsType.addAttribute(grouperSession, propertyValue, false);
653                 } else {
654                   GroupType.createType(grouperSession, propertyValue, false);
655                 }
656                 i++;
657               }
658             }
659             
660           } catch (Exception e) {
661             throw new RuntimeException(e.getMessage(), e);
662           } finally {
663             GrouperSession.stopQuietly(grouperSession);
664           }
665           return null;
666         }
667         
668       });
669       
670       //register the hook if not already
671       GroupTypeTupleIncludeExcludeHook.registerHookIfNecessary(true);
672       
673     } catch (Exception e) {
674       throw new RuntimeException("Problem adding include/exclude type: " + includeExcludeGroupTypeName, e);
675     }
676 
677   }

679   /**
680    * init data
681    * @param logError
682    */
683   public static void initData(boolean logError) {
684     try {
685       //lets see if we need to
686       boolean needsInit;
687       GrouperSession grouperSession = null;
688       try {
689         grouperSession = GrouperSession.start(SubjectFinder.findRootSubject());
690       } catch (SessionException se) {
691         throw new RuntimeException(se);
692       }
693       try {
694         needsInit = StemFinder.findRootStem(grouperSession) == null;
695         needsInit = needsInit || FieldFinder.find(Field.FIELD_NAME_ADMINS, true) == null ;
696       } catch (Exception e) {
697         if (logError && logErrorStatic) {
698           LOG.error("Error initializing data, might just need to auto-create some data to fix...", e);
699         }
700         needsInit = true;
701       } finally {
702         GrouperSession.stopQuietly(grouperSession);
703       }
704       if (needsInit) {
705         if (GrouperConfig.retrieveConfig().propertyValueBoolean("registry.autoinit", true)) {
706           try {
707             
708             RegistryInstall.install();
709             
710           } catch (Exception e) {
711             if (logError && logErrorStatic) {
712               String error = "Couldnt auto-create data: " + e.getMessage();
713               LOG.fatal(error, e);
714             }
715           }
716         } else {
717           
718           if (logError && logErrorStatic) {
719             LOG.fatal("grouper.properties registry.autoinit is false, so not auto initting.  " +
720               "But the registry needs to be auto-initted.  Please init the registry with GSH: registryInstall()  " +
721               "Initting means adding some default data like the root stem, built in fields, etc.");
722           }
723         }
724       }
725     } catch (Exception e) {
726       if (logError && logErrorStatic) {
727         LOG.error("Error initting data", e);
728       }
729     }
