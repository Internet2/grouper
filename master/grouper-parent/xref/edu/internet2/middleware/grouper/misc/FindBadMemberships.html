69   * Find bad memberships in the Grouper memberships table.
70   *
71   * This script is used to find bad memberships in Grouper.  If a bad membership is found, a GSH script will be created to 
72   * resolve the issue.

74   * @since   1.3.1

88    public static StringWriter gshScript = null;

91    private static final String gshScriptFilename = "findbadmemberships.gsh";

93    /** Whether to print memberships errors to standard out. */
94    private static boolean printErrorsToSTOUT = false;

100     gshScript = null;

103   /**
104    * @param args 
105    * @since   1.3.1
106    */
107   public static void main(String[] args) {
108     Options options = new Options();
109     OptionGroup optionGroup = new OptionGroup();
110     optionGroup.addOption(new Option("all", false, 
111       "Find bad memberships."));
112     optionGroup.setRequired(true);
113     options.addOptionGroup(optionGroup);
114 
115     if (args.length == 0) {
116       printUsage(options);
117       System.exit(0);
118     }
119 
120     CommandLineParser parser = new GnuParser();
121     CommandLine line = null;
122     try {
123       line = parser.parse(options, args);
124     } catch (ParseException e) {
125       System.err.println(e.getMessage());
126       printUsage(options);
127       System.exit(1);
128     }
129     clearResults();
130     // maybe this should go to a log file instead?
131     printErrorsToSTOUT = true;
132     
133     GrouperSession grouperSession = null;
134     
135     try {
136       grouperSession = GrouperSession.startRootSession();
137 
138       if (line.hasOption("all")) {
139         checkAll(out);
140       } else {
141         printUsage(options);
142         System.exit(0);
143       }
144 
145     } catch (Exception e) {
146       System.err.println(e.getMessage());
147       System.exit(1);
148     } finally {
149       GrouperSession.stopQuietly(grouperSession);
150     }
151 
152     out.println();
153     out.println();
154     if (gshScript == null) {
155       out.println("No membership errors found.");
156     } else {
157       out.println("Membership errors have been found.  Do the following to resolve the errors:");
158       writeFile(gshScript, gshScriptFilename);
159       out.println(" - Review the GSH script before applying any changes to your database.");
160       out.println(" - Execute the GSH Script " + gshScriptFilename);
161       out.println(" - Re-run the bad membership finder utility to verify that bad memberships have been fixed.");
162     }
163     System.exit(0);
164   }

166   /**
167    * @param printStream 
168    * @throws SessionException
169    */
170   public static void checkAll(PrintStream printStream) throws SessionException {
171     PrintStream oldPrintStream = out;
172     out = printStream;
173     try {  
174       checkAll();
175     } finally {
176       out = oldPrintStream;
177     }
178   }

180   /**
181    * @return count of bad and missing memberships
182    */
183   public static long checkAll() {
184     long errors = 0;
185     
186     if (printErrorsToSTOUT) {
187       out.println();
188       out.println("Checking Composite Memberships");
189     }
190     
191     errors += checkComposites();
192     
193     if (errors == 0) {
194       // only check group sets if composites are okay
195       if (printErrorsToSTOUT) {
196         out.println("Checking Group Sets");
197       }
198       errors += checkGroupSets();
199     } else {
200       if (printErrorsToSTOUT) {
201         out.println("*** Skipping group set check since there were composite issues. Re-run after fixing composites. ***\n");
202       }
203     }
204     
205     if (printErrorsToSTOUT) {
206       out.println("Checking memberships where member is a deleted group");
207     }
208     errors += checkDeletedGroupAsMember();
209     
210     
211     if (printErrorsToSTOUT) {
212       out.println("Checking GrouperAll");
213     }
214     errors += checkGrouperAll();
215     
216     return errors;
217   }

251   /**
252    * Note that this isn't a complete check of the group set table..
253    * @return count of bad and missing group sets
254    */
255   public static long checkGroupSets() {
256     Set<GroupSet> badType = GrouperDAOFactory.getFactory().getGroupSet().findTypeMismatch();
257 
258     for (GroupSet gs : badType) {
259       if (gs.getDepth() != 0) {
260         throw new RuntimeException("Unexpected depth of " + gs.getDepth());
261       }
262       
263       Group g = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getOwnerGroupId(), true);
264       Composite c = GrouperDAOFactory.getFactory().getComposite().findAsOwner(g, false);
265       String currentType = gs.getType();
266       String newType = c == null ? "immediate" : "composite";
267       if (printErrorsToSTOUT) {
268         out.println("Bad group set type: owner groupId=" + g.getUuid() + ", owner group name=" + g.getName() + ", currentType=" + currentType + ", newType=" + newType + ".");
269       }
270       
271       logGshScript("sqlRun(\"update grouper_group_set set mship_type='" + newType + "' where id='" + gs.getId() + "'\");\n");
272     }
273     
274     Set<GroupSet> badCompositeGroupSets = GrouperDAOFactory.getFactory().getGroupSet().findBadGroupSetsForCompositeGroups();
275     for (GroupSet gs : badCompositeGroupSets) {
276       Group ownerGroup = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getOwnerGroupId(), true);
277       Group memberGroup = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getMemberGroupId(), true);
278 
279       if (printErrorsToSTOUT) {
280         out.println("Bad group set for composite: owner groupId=" + ownerGroup.getUuid() + ", owner group name=" + ownerGroup.getName() + ", member groupId=" + memberGroup.getUuid() + ", member group name=" + memberGroup.getName() + ".");
281       }
282       
283       logGshScript("GrouperDAOFactory.getFactory().getGroupSet().findById(\"" + gs.getId() + "\").delete(true);\n");
284     }
285     
286     Set<GroupSet> immediateGroupSetsWithMissingEffective = getImmediateGroupSetsWithMissingEffective(GrouperDAOFactory.getFactory().getGroupSet().findMissingEffectiveGroupSets());
287     for (GroupSet gs : immediateGroupSetsWithMissingEffective) {
288       if (gs.getOwnerGroupId() == null || gs.getMemberGroupId() == null || gs.getDepth() != 1 || !gs.getFieldId().equals(Group.getDefaultList().getUuid())) {
289         throw new RuntimeException("Excepted an immediate group set with an ownerGroup, memberGroup, and member field.  id=" + gs.getId());
290       }
291       
292       Group ownerGroup = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getOwnerGroupId(), true);
293       Group memberGroup = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getMemberGroupId(), true);
294       
295       if (printErrorsToSTOUT) {
296         out.println("Incomplete group set hierarchy (GSH script will attempt to delete and recreate it): owner groupId=" + ownerGroup.getUuid() + ", owner group name=" + ownerGroup.getName() + ", member groupId=" + memberGroup.getUuid() + ", member group name=" + memberGroup.getName() + ".");
297       }
298       
299       // TODO: this should be improved to fix group sets directly without having to delete and recreate the membership
300       // simplifying this right now to avoid timing issues with point in time.
301       logGshScript("delMember(\"" + ownerGroup.getName() + "\", \"" + memberGroup.getId() + "\");\n");      
302       logGshScript("addMember(\"" + ownerGroup.getName() + "\", \"" + memberGroup.getId() + "\");\n");      
303     }
304     
305     return badType.size() + badCompositeGroupSets.size() + immediateGroupSetsWithMissingEffective.size();
306   }

308   /**
309    * GrouperAll shouldn't have manage privileges or be a member of a group
310    * @return count of bad memberships
311    */
312   public static long checkGrouperAll() {
313     
314     Member grouperAll = GrouperDAOFactory.getFactory().getMember().findBySubject(SubjectFinder.findAllSubject().getId(), SubjectFinder.findAllSubject().getSourceId(), SubjectFinder.findAllSubject().getTypeName(), false);
315     if (grouperAll == null) {
316       return 0;
317     }
318     
319     Set<Membership> badMemberships = new LinkedHashSet<Membership>();
320     
321     for (Privilege privilege : AccessPrivilege.MANAGE_PRIVILEGES) {
322       Field field = privilege.getField();
323       badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findAllImmediateByMemberAndField(grouperAll.getId(), field, false));
324     }
325     
326     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findAllImmediateByMemberAndField(grouperAll.getId(), Group.getDefaultList(), false));
327     
328     for (Membership ms : badMemberships) {
329       if (printErrorsToSTOUT) {
330         out.println("Bad GrouperAll membership: groupId=" + ms.getOwnerGroupId() + ", group name=" + ms.getOwnerGroup().getName() + ", field=" + ms.getField().getName() + ".");
331       }
332       
333       logGshScript("GrouperDAOFactory.getFactory().getMembership().findByImmediateUuid(\"" + ms.getImmediateMembershipId() + "\", true).delete();\n");
334     }
335     
336     return badMemberships.size();
337   }

339   /**
340    * GrouperAll shouldn't have manage privileges or be a member of a group.  Use this to force a cleanup without outputting a GSH script.
341    */
342   public static void checkAndFixGrouperAll() {
343     
344     Member grouperAll = GrouperDAOFactory.getFactory().getMember().findBySubject(SubjectFinder.findAllSubject().getId(), SubjectFinder.findAllSubject().getSourceId(), SubjectFinder.findAllSubject().getTypeName(), false);
345     if (grouperAll == null) {
346       System.out.println("Finished running successfully.  0 changes made.");
347       return;
348     }
349     
350     Set<Membership> badMemberships = new LinkedHashSet<Membership>();
351     
352     for (Privilege privilege : AccessPrivilege.MANAGE_PRIVILEGES) {
353       Field field = privilege.getField();
354       badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findAllImmediateByMemberAndField(grouperAll.getId(), field, false));
355     }
356     
357     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findAllImmediateByMemberAndField(grouperAll.getId(), Group.getDefaultList(), false));
358     
359     for (Membership ms : badMemberships) {
360       System.out.println("Removing GrouperAll membership: groupId=" + ms.getOwnerGroupId() + ", group name=" + ms.getOwnerGroup().getName() + ", field=" + ms.getField().getName() + ".");
361       
362       ms.delete();
363     }
364     
365     System.out.println("Finished running successfully.  " + badMemberships.size() + " changes made.");
366   }

368   /**
369    * @return count of bad and missing memberships
370    */
371   public static long checkComposites() {
372     
373     Set<Membership> badMemberships = new LinkedHashSet<Membership>();
374     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadMembershipsOnCompositeGroup());
375     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadCompositeMembershipsOnNonCompositeGroup());
376     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadComplementMemberships());
377     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadUnionMemberships());
378     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadIntersectionMemberships());
379     
380     Set<Object[]> missingMemberships = new LinkedHashSet<Object[]>();
381     missingMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findMissingComplementMemberships());
382     missingMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findMissingUnionMemberships());
383     missingMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findMissingIntersectionMemberships());
384 
385     for (Membership ms : badMemberships) {
386       if (printErrorsToSTOUT) {
387         out.println("Bad composite membership: groupId=" + ms.getOwnerGroupId() + ", group name=" + ms.getOwnerGroup().getName() + ", subjectId=" + ms.getMember().getSubjectId() + ".");
388       }
389       
390       logGshScript("GrouperDAOFactory.getFactory().getMembership().findByImmediateUuid(\"" + ms.getImmediateMembershipId() + "\", true).delete();\n");
391     }
392     
393     for (Object[] ownerAndCompositeAndMember : missingMemberships) {
394       String ownerGroupId = (String)ownerAndCompositeAndMember[0];
395       String compositeId = (String)ownerAndCompositeAndMember[1];
396       String memberId = (String)ownerAndCompositeAndMember[2];
397       
398       if (printErrorsToSTOUT) {
399         Group group = GrouperDAOFactory.getFactory().getGroup().findByUuid(ownerGroupId, true);
400         Member member = GrouperDAOFactory.getFactory().getMember().findByUuid(memberId, true);
401         out.println("Missing composite membership: groupId=" + group.getId() + ", group name=" + group.getName() + ", subjectId=" + member.getSubjectId() + ".");
402       }
403       
404       logGshScript("GrouperDAOFactory.getFactory().getMembership().save(Composite.internal_createNewCompositeMembershipObject(\"" + ownerGroupId + "\", \"" + memberId + "\", \"" + compositeId + "\"));\n");
405     }
406     
407     return badMemberships.size() + missingMemberships.size();
408   }

411    * @return count of bad memberships
412    */
413   public static long checkDeletedGroupAsMember() {
414     
415     Set<Membership> badMemberships = GrouperDAOFactory.getFactory().getMembership().findBadMembershipsDeletedGroupAsMember();
416 
417     for (Membership ms : badMemberships) {
418       if (printErrorsToSTOUT) {
419         out.println("Bad membership where member is a deleted group: groupId=" + ms.getOwnerGroupId() + ", group name=" + ms.getOwnerGroup().getName() + ", subjectId=" + ms.getMember().getSubjectId() + ".");
420       }
421       
422       logGshScript("sqlRun(\"delete from grouper_memberships where id='" + ms.getImmediateMembershipId() + "'\");\n");
423     }
424     
425     return badMemberships.size();
426   }

428   /**
429    * @param parentAndImmediateSet
430    * @return immediates that need to be recreated
431    */
432   private static Set<GroupSet> getImmediateGroupSetsWithMissingEffective(Set<Object[]> parentAndImmediateSet) {
433     Set<GroupSet> missing = new LinkedHashSet<GroupSet>();
434     
435     for (Object[] parentAndImmediate : parentAndImmediateSet) {
436       GroupSet parent = (GroupSet)parentAndImmediate[0];
437       GroupSet immediate = (GroupSet)parentAndImmediate[1];
438       
439       // note that these objects aren't actually saved, we're just creating them for convenience..
440       GroupSet groupSet = new GroupSet();
441       groupSet.setId(GrouperUuid.getUuid());
442       groupSet.setDepth(parent.getDepth() + 1);
443       groupSet.setParentId(parent.getId());
444       groupSet.setFieldId(parent.getFieldId());
445       groupSet.setMemberGroupId(immediate.getMemberId());
446       groupSet.setOwnerGroupId(parent.getOwnerGroupId());
447       groupSet.setOwnerAttrDefId(parent.getOwnerAttrDefId());
448       groupSet.setOwnerStemId(parent.getOwnerStemId());
449       groupSet.setType(MembershipType.EFFECTIVE.getTypeString());
450       
451       if (!immediate.internal_isCircular(groupSet, parent)) {
452         missing.add(immediate);
453       }
454     }
455     
456     return missing;
457   }

459   /**
460    * Keep the GSH script in memory until we're done with this utility.
461    *
462    * @param script to log
463    */
464   private static void logGshScript(String script) {
465     if (gshScript == null) {
466       gshScript = new StringWriter();
467       gshScript.write("GrouperSession.startRootSession();\n");
468     }
469 
470     gshScript.write(script);
471   }

482   /**
483    * Write data to a file.
484    *
485    * @param data to write
486    * @param filename to write data to.
487    */
488   private static void writeFile(StringWriter data, String filename) {
489     FileWriter fw = null;
490     try {
491       out.println("Writing file: " + GrouperUtil.fileCanonicalPath(new File(filename)));
492       fw = new FileWriter(filename, false);
493       fw.write(data.toString());
494     } catch (IOException e) {
495       out.println("Exception while writing out to file " + filename + ": " + e.toString());
496       out.println("Writing data out here instead: ");
497       out.println(data.toString());
498     } finally {
499       try {
500         if (fw != null) {
501           fw.close();
502         }
503       } catch (IOException e) {
504         // do nothing
505       }
506     }
507   }
