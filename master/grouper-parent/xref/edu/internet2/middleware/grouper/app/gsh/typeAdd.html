23  package edu.internet2.middleware.grouper.app.gsh;
24  import bsh.CallStack;
25  import bsh.Interpreter;
26  import edu.internet2.middleware.grouper.GroupType;
27  import edu.internet2.middleware.grouper.exception.InsufficientPrivilegeException;
28  import edu.internet2.middleware.grouper.exception.SchemaException;
29  
30  /**
31   * Add a {@link GroupType}.
32   * <p/>
33   * @author  blair christensen.
34   * @version $Id: typeAdd.java,v 1.2 2008-09-29 03:38:28 mchyzer Exp $
35   * @since   0.1.0
36   */
37  public class typeAdd {
38  
39    // PUBLIC CLASS METHODS //
40  
41    /**
42     * Add a {@link GroupType}.
43     * <p/>
44     * @param   i           BeanShell interpreter.
45     * @param   stack       BeanShell call stack.
46     * @param   name        Name of {@link GroupType} to add.
47     * @return  {@link GroupType}
48     * @throws  GrouperShellException
49     * @since   0.1.0
50     */
51    public static GroupType invoke(Interpreter i, CallStack stack, String name) 
52      throws  GrouperShellException
53    {
54      GrouperShell.setOurCommand(i, true);
55      try {
56        return GroupType.createType(
57          GrouperShell.getSession(i), name
58        );
59      }
60      catch (InsufficientPrivilegeException eIP)  {
61        GrouperShell.error(i, eIP);
62      }
63      catch (SchemaException eS)                  {
64        GrouperShell.error(i, eS);
65      }
66      throw new GrouperShellException(GshErrorMessages.TYPE_ADD + name);
67    } // public static GroupType invoke(i, stack, name)
68  
69  } // public class typeAdd
