61  /**
62   * class which will run a loader job
63   * implements StatefulJob so multiple dont run at once
64   */
65  @PersistJobDataAfterExecution
66  @DisallowConcurrentExecution
67  public class GrouperLoaderJob implements Job {
68  
69    /**
70     * logger 
71     */
72    private static final Log LOG = GrouperUtil.getLog(GrouperLoaderJob.class);
73  
74    /**
75     * @see org.quartz.Job#execute(org.quartz.JobExecutionContext)
76     */
77    public void execute(JobExecutionContext context) throws JobExecutionException {
78      long startTime = System.currentTimeMillis();
79  
80      Hib3GrouperLoaderLog hib3GrouploaderLog = new Hib3GrouperLoaderLog();
81      
82      Group group = null;
83      AttributeDef attributeDef = null;
84      GrouperSession grouperSession = null;
85      try {
86        grouperSession = GrouperSession.startRootSession();
87        String jobName = context.getJobDetail().getKey().getName();
88    
89        hib3GrouploaderLog.setJobName(jobName);
90        
91        String grouperLoaderQuartzCronFromOwner = null;
92        String grouperLoaderTypeFromOwner = null;
93        String grouperLoaderScheduleTypeFromOwner = null;
94        Integer grouperLoaderPriorityFromOwner = null;
95        Integer grouperLoaderIntervalSecondsFromOwner = null;
96        
97        //job name is GrouperLoaderType__groupname__uuid
98        GrouperLoaderType grouperLoaderType = GrouperLoaderType.typeForThisName(jobName);
99        if (grouperLoaderType.equals(GrouperLoaderType.SQL_GROUP_LIST) || 
100           grouperLoaderType.equals(GrouperLoaderType.SQL_SIMPLE)) {
101         
102         int uuidIndexStart = jobName.lastIndexOf("__");
103         
104         if (uuidIndexStart >= 0) {
105           String grouperLoaderGroupUuid = null;
106           grouperLoaderGroupUuid = jobName.substring(uuidIndexStart+2, jobName.length());
107           hib3GrouploaderLog.setGroupUuid(grouperLoaderGroupUuid);
108 
109           
110           group = GroupFinder.findByUuid(grouperSession, grouperLoaderGroupUuid, true);
111           grouperLoaderQuartzCronFromOwner = GrouperLoaderType.attributeValueOrDefaultOrNull(group, 
112               GrouperLoader.GROUPER_LOADER_QUARTZ_CRON);
113           grouperLoaderScheduleTypeFromOwner = GrouperLoaderType.attributeValueOrDefaultOrNull(group, 
114               GrouperLoader.GROUPER_LOADER_SCHEDULE_TYPE);
115           grouperLoaderTypeFromOwner = GrouperLoaderType.attributeValueOrDefaultOrNull(group, 
116               GrouperLoader.GROUPER_LOADER_TYPE);
117           grouperLoaderPriorityFromOwner = GrouperUtil.intObjectValue(
118               GrouperLoaderType.attributeValueOrDefaultOrNull(group,
119               GrouperLoader.GROUPER_LOADER_PRIORITY), true);
120           
121           //lets reset the job name in case the name has changed
122           jobName = grouperLoaderTypeFromOwner + "__" + group.getName() + "__" + group.getUuid();
123           hib3GrouploaderLog.setJobName(jobName);
124           
125         }
126       }
127       
128       if (grouperLoaderType.equals(GrouperLoaderType.ATTR_SQL_SIMPLE)) {
129         
130         int uuidIndexStart = jobName.lastIndexOf("__");
131         
132         if (uuidIndexStart >= 0) {
133           String grouperLoaderAttrDefUuid = null;
134           grouperLoaderAttrDefUuid = jobName.substring(uuidIndexStart+2, jobName.length());
135           hib3GrouploaderLog.setGroupUuid(grouperLoaderAttrDefUuid);
136 
137           
138           attributeDef = AttributeDefFinder.findById(grouperLoaderAttrDefUuid, true);
139           
140           grouperLoaderQuartzCronFromOwner = GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(attributeDef, 
141               GrouperLoader.ATTRIBUTE_LOADER_QUARTZ_CRON);
142           grouperLoaderScheduleTypeFromOwner = GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(attributeDef, 
143               GrouperLoader.ATTRIBUTE_LOADER_SCHEDULE_TYPE);
144           grouperLoaderTypeFromOwner = GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(attributeDef, 
145               GrouperLoader.ATTRIBUTE_LOADER_TYPE);
146           grouperLoaderIntervalSecondsFromOwner = GrouperUtil.intObjectValue(
147               GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(attributeDef,
148               GrouperLoader.ATTRIBUTE_LOADER_INTERVAL_SECONDS), true);
149           grouperLoaderPriorityFromOwner = GrouperUtil.intObjectValue(
150               GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(attributeDef,
151               GrouperLoader.ATTRIBUTE_LOADER_PRIORITY), true);
152           
153           //lets reset the job name in case the name has changed
154           jobName = grouperLoaderTypeFromOwner + "__" + attributeDef.getName() + "__" + attributeDef.getId();
155           hib3GrouploaderLog.setJobName(jobName);
156           
157         }
158       }
159       
160       if (grouperLoaderType.equals(GrouperLoaderType.LDAP_SIMPLE)
161           || grouperLoaderType.equals(GrouperLoaderType.LDAP_GROUP_LIST)
162           || grouperLoaderType.equals(GrouperLoaderType.LDAP_GROUPS_FROM_ATTRIBUTES)) {
163         
164         int uuidIndexStart = jobName.lastIndexOf("__");
165         
166         if (uuidIndexStart >= 0) {
167           String grouperLoaderGroupUuid = null;
168           grouperLoaderGroupUuid = jobName.substring(uuidIndexStart+2, jobName.length());
169           hib3GrouploaderLog.setGroupUuid(grouperLoaderGroupUuid);
170 
171           
172           group = GroupFinder.findByUuid(grouperSession, grouperLoaderGroupUuid, true);
173           
174           AttributeDefName attributeDefName = AttributeDefNameFinder.findByName(LoaderLdapUtils.grouperLoaderLdapName(), false);
175           
176           AttributeAssign attributeAssign = group.getAttributeDelegate().retrieveAssignment(AttributeDef.ACTION_DEFAULT, attributeDefName, false, true);
177           
178           grouperLoaderQuartzCronFromOwner = attributeAssign.getAttributeValueDelegate().retrieveValueString(LoaderLdapUtils.grouperLoaderLdapQuartzCronName());
179           
180           grouperLoaderTypeFromOwner = attributeAssign.getAttributeValueDelegate().retrieveValueString(LoaderLdapUtils.grouperLoaderLdapTypeName());
181           grouperLoaderPriorityFromOwner = GrouperUtil.intObjectValue(attributeAssign.getAttributeValueDelegate()
182               .retrieveValueInteger(LoaderLdapUtils.grouperLoaderLdapPriorityName()), true);
183           
184           //lets reset the job name in case the name has changed
185           jobName = grouperLoaderTypeFromOwner + "__" + group.getName() + "__" + group.getId();
186           hib3GrouploaderLog.setJobName(jobName);
187           
188         }
189       }
190       
191       //switch the job type?
192       if (!StringUtils.isBlank(grouperLoaderTypeFromOwner)) {
193         GrouperLoaderType grouperLoaderTypeFromGroupEnum = GrouperLoaderType.valueOfIgnoreCase(grouperLoaderTypeFromOwner, true);
194         if (!grouperLoaderTypeFromGroupEnum.equals(grouperLoaderType)) {
195           LOG.debug("Grouper loader type has changed to " + grouperLoaderTypeFromGroupEnum
196               + " from " + grouperLoaderType + ", for job: " + jobName);
197           grouperLoaderType = grouperLoaderTypeFromGroupEnum;
198         }
199       }
200       
201       Trigger trigger = context.getTrigger();
202       String grouperLoaderQuartzCron = null;
203       String grouperLoaderScheduleType = null;
204       if (trigger instanceof CronTrigger) {
205         grouperLoaderQuartzCron = ((CronTrigger)trigger).getCronExpression();
206         grouperLoaderScheduleType = GrouperLoaderScheduleType.CRON.name();
207       }
208       Integer grouperLoaderIntervalSeconds = null;
209       if (trigger instanceof SimpleTrigger) {
210         grouperLoaderIntervalSeconds = (int)(((SimpleTrigger)trigger).getRepeatInterval()/1000);
211         grouperLoaderScheduleType = GrouperLoaderScheduleType.START_TO_START_INTERVAL.name();
212       }
213       
214       boolean scheduleChange = false;
215       
216       if (!StringUtils.isBlank(grouperLoaderScheduleTypeFromOwner)
217           && !StringUtils.equalsIgnoreCase(grouperLoaderScheduleTypeFromOwner, grouperLoaderScheduleType)) {
218         LOG.warn("Detected a grouper loader schedule change in job: " + jobName 
219             + ", scheduleType from: " + grouperLoaderScheduleTypeFromOwner + ", to: " + grouperLoaderScheduleType);
220 
221         scheduleChange = true;
222       }
223       if (!StringUtils.isBlank(grouperLoaderQuartzCronFromOwner)
224           && !StringUtils.equals(grouperLoaderQuartzCronFromOwner, grouperLoaderQuartzCron)) {
225 
226         LOG.warn("Detected a grouper loader schedule change in job: " + jobName 
227             + ", quartzCron from: " + grouperLoaderQuartzCronFromOwner + ", to: " + grouperLoaderQuartzCron);
228 
229         scheduleChange = true;
230       }
231       if (grouperLoaderIntervalSecondsFromOwner != null
232           && !ObjectUtils.equals(grouperLoaderIntervalSecondsFromOwner, grouperLoaderIntervalSeconds)) {
233 
234         LOG.warn("Detected a grouper loader schedule change in job: " + jobName 
235             + ", intervalSeconds from: " + grouperLoaderIntervalSecondsFromOwner + ", to: " + grouperLoaderIntervalSeconds);
236 
237         scheduleChange = true;
238       }
239       if (grouperLoaderPriorityFromOwner != null && 
240           !ObjectUtils.equals(grouperLoaderPriorityFromOwner, trigger.getPriority())) {
241 
242         LOG.warn("Detected a grouper loader schedule change in job: " + jobName 
243             + ", priority from: " + grouperLoaderPriorityFromOwner + ", to: " + trigger.getPriority());
244 
245         scheduleChange = true;
246       }
247       
248       //see if the runtime settings have changed
249       if (scheduleChange) {
250         
251         GrouperLoaderScheduleType grouperLoaderScheduleTypeEnumFromOwner = null;
252         
253         //if there is a cron string, then it must be cron
254         if (StringUtils.isBlank(grouperLoaderScheduleTypeFromOwner) && !StringUtils.isBlank(grouperLoaderQuartzCronFromOwner)) {
255           grouperLoaderScheduleTypeEnumFromOwner = GrouperLoaderScheduleType.CRON;
256           
257           //if it is an LDAP job, then it must be cron
258         } else if (StringUtils.isBlank(grouperLoaderQuartzCronFromOwner) && (grouperLoaderType.equals(GrouperLoaderType.LDAP_SIMPLE)
259             || grouperLoaderType.equals(GrouperLoaderType.LDAP_GROUP_LIST)
260             || grouperLoaderType.equals(GrouperLoaderType.LDAP_GROUPS_FROM_ATTRIBUTES))) {
261           grouperLoaderScheduleTypeEnumFromOwner = GrouperLoaderScheduleType.CRON;
262           
263           //else parse the schedule type, and it is required
264         } else {
265           grouperLoaderScheduleTypeEnumFromOwner = GrouperLoaderScheduleType
266               .valueOfIgnoreCase(grouperLoaderScheduleTypeFromOwner, true);
267         }
268         
269         if (grouperLoaderScheduleTypeEnumFromOwner.equals(GrouperLoaderScheduleType.START_TO_START_INTERVAL)) {
270           if (grouperLoaderIntervalSecondsFromOwner == null) {
271             grouperLoaderIntervalSecondsFromOwner = 60*60*24;
272           }
273         }
274         if (grouperLoaderScheduleTypeEnumFromOwner.equals(GrouperLoaderScheduleType.CRON)) {
275           if (StringUtils.isBlank(grouperLoaderQuartzCronFromOwner)) {
276             throw new RuntimeException("Cron cant be blank if cron schedule: " + jobName);
277           }
278         }
279       }
280       
281       if (grouperLoaderType != null) {
282         hib3GrouploaderLog.setJobType(grouperLoaderType.name());
283       }
284       if (!StringUtils.isBlank(grouperLoaderScheduleTypeFromOwner)) {
285         hib3GrouploaderLog.setJobScheduleType(grouperLoaderScheduleTypeFromOwner);
286       }
287       hib3GrouploaderLog.setJobScheduleIntervalSeconds(grouperLoaderIntervalSecondsFromOwner);
288       
289       if (grouperLoaderPriorityFromOwner != null || trigger != null) {
290         
291         hib3GrouploaderLog.setJobSchedulePriority(grouperLoaderPriorityFromOwner != null ? grouperLoaderPriorityFromOwner 
292             : trigger.getPriority());
293       }
294 
295       hib3GrouploaderLog.setJobScheduleQuartzCron(!StringUtils.isBlank(grouperLoaderQuartzCronFromOwner) ? 
296           grouperLoaderQuartzCronFromOwner : grouperLoaderQuartzCron );
297       
298       if (scheduleChange) {
299         LOG.warn("Detected a grouper loader schedule change in job: " + jobName + ", to: " 
300             + grouperLoaderScheduleTypeFromOwner + ", cron: " + grouperLoaderQuartzCronFromOwner
301             + ", interval: " + grouperLoaderIntervalSecondsFromOwner);
302         
303         GrouperLoaderType.scheduleJob(jobName, true, grouperLoaderScheduleTypeFromOwner, grouperLoaderQuartzCronFromOwner,
304             grouperLoaderIntervalSecondsFromOwner, grouperLoaderPriorityFromOwner);
305         
306       }
307       
308       if (grouperLoaderType.equals(GrouperLoaderType.ATTR_SQL_SIMPLE)) {
309         
310         runJobAttrDef(hib3GrouploaderLog, attributeDef, grouperSession);
311       } else if (grouperLoaderType.equals(GrouperLoaderType.LDAP_SIMPLE)
312         || grouperLoaderType.equals(GrouperLoaderType.LDAP_GROUP_LIST)
313         || grouperLoaderType.equals(GrouperLoaderType.LDAP_GROUPS_FROM_ATTRIBUTES)) {
314         
315         runJobLdap(hib3GrouploaderLog, group, grouperSession);
316       } else {
317         //all other jobs go through here
318         runJob(hib3GrouploaderLog, group, grouperSession);
319       }
320     } catch (Exception e) {
321       LOG.error("Error running up job", e);
322       if (!(e instanceof JobExecutionException)) {
323         e = new JobExecutionException(e);
324       }
325       JobExecutionException jobExecutionException = (JobExecutionException)e;
326       storeLogInDb(hib3GrouploaderLog, false, startTime);
327       throw jobExecutionException;
328     } finally {
329       GrouperSession.stopQuietly(grouperSession);
330     }
331     
332   }

334   /**
335    * run a job (either from quartz or outside)
336    * @param hib3GrouploaderLog will get information, most importantly the job name
337    * @param jobGroup if a group job, this is the group object
338    * @param grouperSession 
339    */
340   public static void runJob(Hib3GrouperLoaderLog hib3GrouploaderLog, Group jobGroup, GrouperSession grouperSession) {
341     long startTime = System.currentTimeMillis();
342     boolean throwExceptionsInFinally = true;
343     String jobName = null;
344     try {
345       
346       jobName = hib3GrouploaderLog.getJobName();
347       
348       GrouperLoaderType grouperLoaderTypeEnum = GrouperLoaderType.typeForThisName(jobName);
349       
350       List<Group> andGroups = new ArrayList<Group>();
351   
352       //find the groups whose membership we "and" with the dynamic group
353       //CH 20110925: why is this not retrieved from the jobGroup???
354       String grouperLoaderAndGroupNames = hib3GrouploaderLog.getAndGroupNames();
355       if (!StringUtils.isBlank(grouperLoaderAndGroupNames)) {
356         
357         hib3GrouploaderLog.setAndGroupNames(grouperLoaderAndGroupNames);
358         
359         //there are groups to and with, get the list
360         String[] groupNames = GrouperUtil.splitTrim(grouperLoaderAndGroupNames, ",");
361         
362         
363         for (String groupName : groupNames) {
364           Group group = GroupFinder.findByName(grouperSession, groupName, true);
365           andGroups.add(group);
366         }
367       }
368       
369       //log that we are starting a job
370       hib3GrouploaderLog.setHost(GrouperUtil.hostname());
371       hib3GrouploaderLog.setStartedTime(new Timestamp(System.currentTimeMillis()));
372       hib3GrouploaderLog.setStatus(GrouperLoaderStatus.STARTED.name());
373       
374       hib3GrouploaderLog.store();
375       
376       String grouperLoaderDbName = "grouper";
377       
378       String grouperLoaderQuery = null;
379       String groupName = null;
380       
381       String groupTypesString = null;
382       String groupLikeString = null;
383       String groupQuery = null;
384       
385       if (jobGroup != null) {
386         grouperLoaderDbName = GrouperLoaderType.attributeValueOrDefaultOrNull(jobGroup, GrouperLoader.GROUPER_LOADER_DB_NAME);
387         grouperLoaderQuery = GrouperLoaderType.attributeValueOrDefaultOrNull(jobGroup, GrouperLoader.GROUPER_LOADER_QUERY);
388         groupTypesString = GrouperLoaderType.attributeValueOrDefaultOrNull(jobGroup, GrouperLoader.GROUPER_LOADER_GROUP_TYPES);
389         groupLikeString = GrouperLoaderType.attributeValueOrDefaultOrNull(jobGroup, GrouperLoader.GROUPER_LOADER_GROUPS_LIKE);
390         groupQuery = GrouperLoaderType.attributeValueOrDefaultOrNull(jobGroup, GrouperLoader.GROUPER_LOADER_GROUP_QUERY);
391         groupName = jobGroup.getName();
392         hib3GrouploaderLog.setGroupUuid(jobGroup.getUuid());
393       }
394       
395       GrouperLoaderDb grouperLoaderDb = GrouperLoaderConfig.retrieveDbProfile(grouperLoaderDbName);
396       
397       List<GroupType> groupTypes = null;
398       if (!StringUtils.isBlank(groupTypesString)) {
399         String[] groupTypeArray = GrouperUtil.splitTrim(groupTypesString, ",");
400         groupTypes = new ArrayList<GroupType>();
401         for (String groupType : groupTypeArray) {
402           //this better find the type!
403           groupTypes.add(GroupTypeFinder.find(groupType, true));
404         }
405       }
406       
407       LoaderJobBean loaderJobBean = new LoaderJobBean(grouperLoaderTypeEnum, groupName, grouperLoaderDb, grouperLoaderQuery, 
408           hib3GrouploaderLog, grouperSession, andGroups, groupTypes, groupLikeString, groupQuery, startTime);
409       
410       //call hooks if registered
411       GrouperHooksUtils.callHooksIfRegistered(GrouperHookType.LOADER, 
412           LoaderHooks.METHOD_LOADER_PRE_RUN, HooksLoaderBean.class, loaderJobBean, 
413           LoaderJobBean.class, VetoTypeGrouper.LOADER_PRE_RUN);
414   
415       //based on type, run query from the db and sync members
416       grouperLoaderTypeEnum.runJob(loaderJobBean);
417   
418       if (GrouperLoaderConfig.retrieveConfig().propertyValueBoolean("grouperLoader.throwExceptionsv2_2_1.onJobErrors", true)) {
419         GrouperLoaderStatus grouperLoaderStatus = StringUtils.isBlank(hib3GrouploaderLog.getStatus()) ? null : 
420           GrouperLoaderStatus.valueOfIgnoreCase(hib3GrouploaderLog.getStatus(), true);
421         
422         if (grouperLoaderStatus == null || grouperLoaderStatus == GrouperLoaderStatus.ERROR 
423             || grouperLoaderStatus == GrouperLoaderStatus.CONFIG_ERROR) {
424           
425           throw new RuntimeException("Error in loader job: " + groupName + ", check logs: " + StringUtils.trimToEmpty(hib3GrouploaderLog.getJobMessage()));
426           
427         }
428       }
429       
430       //call hooks if registered
431       GrouperHooksUtils.callHooksIfRegistered(GrouperHookType.LOADER, 
432           LoaderHooks.METHOD_LOADER_POST_RUN, HooksLoaderBean.class, loaderJobBean, 
433           LoaderJobBean.class, VetoTypeGrouper.LOADER_POST_RUN);
434       
435     } catch (Exception t) {
436       LOG.error("Error on job: " + jobName, t);
437       
438       hib3GrouploaderLog.setStatus(GrouperLoaderStatus.ERROR.name());
439       hib3GrouploaderLog.appendJobMessage(ExceptionUtils.getFullStackTrace(t));
440       throwExceptionsInFinally = false;
441       if (t instanceof RuntimeException) {
442         throw (RuntimeException)t;
443       }
444       throw new RuntimeException(t.getMessage(), t);
445     } finally {
446       
447       storeLogInDb(hib3GrouploaderLog, throwExceptionsInFinally, startTime);
448     }
449     
450   }

452   /**
453    * run a job (either from quartz or outside)
454    * @param hib3GrouploaderLog will get information, most importantly the job name
455    * @param jobGroup group that this ldap job is about
456    * @param grouperSession 
457    */
458   public static void runJobLdap(Hib3GrouperLoaderLog hib3GrouploaderLog, Group jobGroup, GrouperSession grouperSession) {
459     long startTime = System.currentTimeMillis();
460     boolean throwExceptionsInFinally = true;
461     String jobName = null;
462     try {
463       
464       jobName = hib3GrouploaderLog.getJobName();
465       
466       GrouperLoaderType grouperLoaderTypeEnum = GrouperLoaderType.typeForThisName(jobName);
467       
468       //log that we are starting a job
469       hib3GrouploaderLog.setHost(GrouperUtil.hostname());
470       hib3GrouploaderLog.setStartedTime(new Timestamp(System.currentTimeMillis()));
471       hib3GrouploaderLog.setStatus(GrouperLoaderStatus.STARTED.name());
472       
473       hib3GrouploaderLog.store();
474       
475       String grouperLoaderLdapType = null;
476       String grouperLoaderLdapServerId = null;
477       String grouperLoaderLdapFilter = null;
478       String grouperLoaderLdapSubjectAttribute = null;
479       String grouperLoaderLdapSearchDn = null;
480       String grouperLoaderLdapSourceId = null;
481       String grouperLoaderLdapSubjectIdType = null;
482       String grouperLoaderLdapSearchScope = null;
483       String grouperLoaderLdapAndGroups = null;
484       String grouperLoaderLdapGroupAttributeName = null;
485       String grouperLoaderLdapAttributeFilterExpression = null;
486       String grouperLoaderLdapExtraAttributes = null;
487       String grouperLoaderLdapGroupNameExpression = null;
488       String grouperLoaderLdapGroupDisplayExtensionExpression = null;
489       String grouperLoaderLdapGroupDescriptionExpression = null;
490       String grouperLoaderLdapSubjectExpression = null;
491       String groupTypesString = null;
492       String grouperLoaderLdapGroupReaders = null;
493       String grouperLoaderLdapGroupViewers = null;
494       String grouperLoaderLdapGroupAdmins = null;
495       String grouperLoaderLdapGroupUpdaters = null;
496       String grouperLoaderLdapGroupOptins = null;
497       String grouperLoaderLdapGroupOptouts = null;
498       String grouperLoaderLdapGroupAttrReaders = null;
499       String grouperLoaderLdapGroupAttrUpdaters = null;
500       String grouperLoaderLdapGroupsLike = null;
501       
502       AttributeDefName grouperLoaderLdapTypeAttributeDefName = GrouperDAOFactory.getFactory()
503       .getAttributeDefName().findByNameSecure(LoaderLdapUtils.grouperLoaderLdapName(), false);
504       AttributeAssign attributeAssign = grouperLoaderLdapTypeAttributeDefName == null ? null : 
505         jobGroup.getAttributeDelegate().retrieveAssignment(
506           null, grouperLoaderLdapTypeAttributeDefName, false, false);
507 
508       grouperLoaderLdapType = GrouperLoaderType.attributeValueOrDefaultOrNull(attributeAssign, LoaderLdapUtils.grouperLoaderLdapTypeName());

550           andGroups.add(group);
551         }
552       }
553       
554       List<GroupType> groupTypes = null;
555       if (!StringUtils.isBlank(groupTypesString)) {
556         String[] groupTypeArray = GrouperUtil.splitTrim(groupTypesString, ",");
557         groupTypes = new ArrayList<GroupType>();
558         for (String groupType : groupTypeArray) {
559           //this better find the type!
560           groupTypes.add(GroupTypeFinder.find(groupType, true));
561         }
562       }

588           GrouperLoaderStatus.valueOfIgnoreCase(hib3GrouploaderLog.getStatus(), true);
589         
590         if (grouperLoaderStatus == null || grouperLoaderStatus == GrouperLoaderStatus.ERROR

608       throwExceptionsInFinally = false;
609       if (t instanceof RuntimeException) {

611       }
612       throw new RuntimeException(t.getMessage(), t);
613     } finally {
614       
615       storeLogInDb(hib3GrouploaderLog, throwExceptionsInFinally, startTime);
616     }

619   /**
620    * run a job (either from quartz or outside)
621    * @param hib3GrouploaderLog will get information, most importantly the job name
622    * @param jobAttributeDef if a attributeDef job, this is the attributeDef object
623    * @param grouperSession 
624    */
625   public static void runJobAttrDef(Hib3GrouperLoaderLog hib3GrouploaderLog, AttributeDef jobAttributeDef, GrouperSession grouperSession) {
626     long startTime = System.currentTimeMillis();
627     boolean throwExceptionsInFinally = true;
628     String jobName = null;
629     try {
630       
631       jobName = hib3GrouploaderLog.getJobName();
632       
633       GrouperLoaderType grouperLoaderTypeEnum = GrouperLoaderType.typeForThisName(jobName);
634       
635       //log that we are starting a job
636       hib3GrouploaderLog.setHost(GrouperUtil.hostname());
637       hib3GrouploaderLog.setStartedTime(new Timestamp(System.currentTimeMillis()));
638       hib3GrouploaderLog.setStatus(GrouperLoaderStatus.STARTED.name());
639       
640       hib3GrouploaderLog.store();
641       
642       String attributeLoaderDbName = "grouper";
643       
644       String attributeLoaderActionQuery = null;
645       String attributeDefName = null;
646       
647       String attributeLoaderAttrsLike = null;
648       String attributeLoaderActionSetQuery = null;
649       String attributeLoaderAttrQuery = null;
650       String attributeLoaderAttrSetQuery = null;
651       
652       //why are we checking for null?
653       if (jobAttributeDef != null) {
654         attributeLoaderActionQuery = GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(jobAttributeDef, GrouperLoader.ATTRIBUTE_LOADER_ACTION_QUERY);
655         attributeLoaderActionSetQuery = GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(jobAttributeDef, GrouperLoader.ATTRIBUTE_LOADER_ACTION_SET_QUERY);
656         attributeLoaderAttrQuery = GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(jobAttributeDef, GrouperLoader.ATTRIBUTE_LOADER_ATTR_QUERY);

659         attributeLoaderDbName = GrouperLoaderType.attributeValueOrDefaultOrNullAttrDef(jobAttributeDef, GrouperLoader.ATTRIBUTE_LOADER_DB_NAME);
660         attributeDefName = jobAttributeDef.getName();
661         hib3GrouploaderLog.setGroupUuid(jobAttributeDef.getUuid());
662       }
663       
664       GrouperLoaderDb grouperLoaderDb = GrouperLoaderConfig.retrieveDbProfile(attributeLoaderDbName);
665

667           hib3GrouploaderLog, grouperSession, attributeLoaderAttrQuery, attributeLoaderAttrSetQuery, 
668           attributeLoaderAttrsLike, attributeLoaderActionQuery, attributeLoaderActionSetQuery, startTime);
669       
670       //call hooks if registered

673           LoaderJobBean.class, VetoTypeGrouper.LOADER_PRE_RUN);

680           LoaderHooks.METHOD_LOADER_POST_RUN, HooksLoaderBean.class, loaderJobBean, 
681           LoaderJobBean.class, VetoTypeGrouper.LOADER_POST_RUN);

683     } catch (Exception t) {
684       LOG.error("Error on job: " + jobName, t);
685       
686       hib3GrouploaderLog.setStatus(GrouperLoaderStatus.ERROR.name());
687       hib3GrouploaderLog.appendJobMessage(ExceptionUtils.getFullStackTrace(t));
688       throwExceptionsInFinally = false;
689       if (t instanceof RuntimeException) {
690         throw (RuntimeException)t;
691       }
692       throw new RuntimeException(t.getMessage(), t);
693     } finally {
694

696     }

700   /**
701    * @param hib3GrouploaderLog
702    * @param throwException 
703    * @param startTime
704    */
705   private static void storeLogInDb(Hib3GrouperLoaderLog hib3GrouploaderLog,
706       boolean throwException, long startTime) {
707     //store this safely
708     try {
709       
710       long endTime = System.currentTimeMillis();
711       hib3GrouploaderLog.setEndedTime(new Timestamp(endTime));
712       hib3GrouploaderLog.setMillis((int)(endTime-startTime));
713       
714       hib3GrouploaderLog.store();
715       
716     } catch (RuntimeException e) {
717       LOG.error("Problem storing final log", e);
718       //dont preempt an existing exception
719       if (throwException) {
720         throw e;
721       }
722     }
723   }
