423   /**
424    * based on what you are querying for, see if has permission.
425    * Note, you should be looking for one subject, 
426    * one action, one resource, one role or multiple roles, etc
427    * If you are looking for multiple, it will see if anyone has that permission
428    * @return true if has permission, false if not
429    */
430   public boolean hasPermission() {
431     
432     //there needs to be a subject if looking by subject
433     if (this.permissionType == PermissionType.role_subject) {
434       if (GrouperUtil.length(this.memberIds) != 1) {
435         throw new RuntimeException("You need to search for 1 and only 1 subject when using hasPermission for subject permissions: " + this);
436       }
437     } else if (this.permissionType == PermissionType.role) {
438       if (GrouperUtil.length(this.roleIds) != 1) {
439         throw new RuntimeException("You need to search for 1 and only 1 role when using hasPermission for role permissions: " + this);
440       }
441     }
442     
443     if (this.permissionProcessor == null) {
444       //get all the permissions for this user in these roles
445       this.permissionProcessor = limitEnvVars == null ? PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES
446           : PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS ;
447     } else if (this.permissionProcessor != PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES 
448         && this.permissionProcessor != PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS){
449       throw new RuntimeException("permissionProcessor must be FILTER_REDUNDANT_PERMISSIONS_AND_ROLES " +
450       		"or FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS");
451     }
452     
453     Set<PermissionEntry> permissionEntriesSet = findPermissions();
454 
455     if (GrouperUtil.length(permissionEntriesSet) > 1) {
456       throw new RuntimeException("Why is there more than one permission entry? " + GrouperUtil.stringValue(permissionEntriesSet));
457     }
458     
459     if (GrouperUtil.length(permissionEntriesSet) == 0) {
460       return false;
461     }
462     
463     if (pointInTimeTo != null || pointInTimeFrom != null) {
464       // we're not taking into consideration limits here...
465       return !permissionEntriesSet.iterator().next().isDisallowed();
466     }
467 
468     return permissionEntriesSet.iterator().next().isAllowedOverall();
469   }

542   /**
543    * find a list of permissions
544    * @return the set of permissions never null
545    */
546   public Set<PermissionEntry> findPermissions() {
547 
548     validateProcessor();
549 
550     Set<PermissionEntry> permissionEntries = null;
551     
552     if (pointInTimeFrom == null && pointInTimeTo == null) {
553       if (this.permissionType == PermissionType.role_subject) {
554         permissionEntries = GrouperDAOFactory.getFactory().getPermissionEntry().findPermissions(
555             this.permissionDefIds, this.permissionNameIds, this.roleIds, this.actions, this.enabled, 
556             this.memberIds, false, this.permissionNameFolder, this.permissionNameFolderScope);
557       } else if (this.permissionType == PermissionType.role) {
558         permissionEntries = GrouperDAOFactory.getFactory().getPermissionEntry().findRolePermissions(
559             this.permissionDefIds, this.permissionNameIds, this.roleIds, this.actions, 
560             this.enabled, false, this.permissionNameFolder, this.permissionNameFolderScope);
561       } else {
562         throw new RuntimeException("Not expecting permission type: " + this.permissionType);
563       }
564     } else {
565       if (this.permissionType == PermissionType.role_subject) {
566         if (this.permissionNameFolder != null) {
567           throw new RuntimeException("Not implemented looking for permissions by folder and point in time");
568         }
569         permissionEntries = GrouperDAOFactory.getFactory().getPITPermissionAllView().findPermissions(
570             permissionDefIds, permissionNameIds, roleIds, actions, memberIds, pointInTimeFrom, pointInTimeTo);
571       } else {
572         throw new RuntimeException("Not expecting permission type: " + this.permissionType);
573       }
574     }
575     
576     //if size is one, there arent redundancies to process
577     if (this.permissionProcessor != null) {
578       this.permissionProcessor.processPermissions(permissionEntries, this.limitEnvVars);
579     }
580     
581     //if immediate only, do this after processing since it might affect the best decision
582     if (this.immediateOnly) {
583       //see if we are doing immediate only
584       Iterator<PermissionEntry> iterator = GrouperUtil.nonNull(permissionEntries).iterator();
585       while (iterator.hasNext()) {
586         PermissionEntry permissionEntry = iterator.next();
587         if (!permissionEntry.isImmediate(this.permissionType)) {
588           iterator.remove();
589         }
590       }
591     }
592     
593     return permissionEntries;
594     
595   }

597   /**
598    * validate that the processor dosent conflict with anything...
599    */
600   private void validateProcessor() {
601     if (this.permissionProcessor != null && (this.enabled != null && !this.enabled)) {      
602       throw new RuntimeException("You cannot process the permissions " +
603       		"(FILTER_REDUNDANT_PERMISSIONS || FILTER_REUNDANT_PERMISSIONS_AND_ROLES) " +
604       		"without looking for enabled permissions only");
605     }
606     
607     //if processing permissions, just look at enabled
608     if (this.permissionProcessor != null && this.enabled == null) {
609       this.enabled = true;
610     }
611     
612     // verify options for point in time queries
613     if (pointInTimeFrom != null || pointInTimeTo != null) {
614       if (limitEnvVars != null && limitEnvVars.size() > 0) {
615         throw new RuntimeException("Cannot use limits for point in time queries.");
616       }
617       
618       if (immediateOnly) {
619         throw new RuntimeException("immediateOnly is not supported for point in time queries.");
620       }
621       
622       if (enabled == null || !enabled) {
623         throw new RuntimeException("Cannot search for disabled permissions for point in time queries.");
624       }
625       
626       if (permissionType == PermissionType.role) {
627         throw new RuntimeException("Permission type " + PermissionType.role.getName() + " is not supported for point in time queries.");
628       }
629       
630       if (permissionProcessor != null) {
631         if (permissionProcessor.isLimitProcessor()) {
632           throw new RuntimeException("limit processors are not supported for point in time queries.");
633         }
634         
635         if (pointInTimeFrom == null || pointInTimeTo == null || pointInTimeFrom.getTime() != pointInTimeTo.getTime()) {
636           throw new RuntimeException("When using permission processors with point in time queries, queries have to be at a single point in time.");
637         }
638       }
639     }
640   }

642   /**
643    * find a permission
644    * @param exceptionIfNotFound true if exception should be thrown if permission not found
645    * @return the permission or null
646    */
647   public PermissionEntry findPermission(boolean exceptionIfNotFound) {
648 
649     Set<PermissionEntry> permissions = findPermissions();
650     
651     //this should find one if it is there...
652     PermissionEntry permissionEntry = null;
653     
654     if (GrouperUtil.length(permissions) > 1) {
655       throw new RuntimeException("Why is there more than one permission found? " + this);
656     }
657     
658     if (GrouperUtil.length(permissions) == 1) {
659       permissionEntry = permissions.iterator().next();
660     }
661     
662     if (permissionEntry == null && exceptionIfNotFound) {
663       throw new RuntimeException("could not find permission: " 
664           + this);
665     }
666     return permissionEntry;
667     
668   }

670   /**
671    * @see Object#toString()
672    */
673   @Override
674   public String toString() {
675     StringBuilder result = new StringBuilder();
676     if (GrouperUtil.length(this.actions) > 0) {
677       result.append("actions: ").append(GrouperUtil.toStringForLog(this.actions, 100));
678     }
679     if (GrouperUtil.length(this.permissionDefIds) > 0) {
680       result.append("attributeDefIds: ").append(GrouperUtil.toStringForLog(this.permissionDefIds, 100));
681     }
682     if (GrouperUtil.length(this.permissionNameIds) > 0) {
683       result.append("attributeDefNameIds: ").append(GrouperUtil.toStringForLog(this.permissionNameIds, 100));
684     }
685     if (enabled != null) {
686       result.append("enabled: ").append(this.enabled);
687     }
688     if (this.immediateOnly) {
689       result.append("immediateOnly: ").append(this.immediateOnly);
690     }
691     if (GrouperUtil.length(this.limitEnvVars) > 0) {
692       result.append("limitEnvVars: ").append(GrouperUtil.toStringForLog(this.limitEnvVars, 100));
693     }
694     if (GrouperUtil.length(this.memberIds) > 0) {
695       result.append("memberIds: ").append(GrouperUtil.toStringForLog(this.memberIds, 100));
696     }
697     if (this.permissionProcessor != null) {
698       result.append("permissionProcessor: ").append(this.permissionProcessor);
