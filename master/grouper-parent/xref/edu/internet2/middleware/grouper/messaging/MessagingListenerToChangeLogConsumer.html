29  /**
30   * convert a messaging listener to a change log consumer
31   */
32  public class MessagingListenerToChangeLogConsumer extends MessagingListenerBase {
33  
34    /**
35     * logger 
36     */
37    private static final Log LOG = GrouperUtil.getLog(MessagingListenerToChangeLogConsumer.class);
38  
39    /**
40     * 
41     */
42    public MessagingListenerToChangeLogConsumer() {
43    }
44  
45    /**
46     * @see edu.internet2.middleware.grouper.messaging.MessagingListenerBase#processMessages(java.lang.String, java.lang.String, java.util.Collection, edu.internet2.middleware.grouper.messaging.MessagingListenerMetadata)
47     */
48    @Override
49    public void processMessages(String messageSystemName, String queue,
50        Collection<GrouperMessage> grouperMessageList,
51        MessagingListenerMetadata messagingListenerMetadata) {
52      
53      String changeLogConsumerClassName = GrouperLoaderConfig.retrieveConfig().propertyValueString("messaging.listener." 
54          + messagingListenerMetadata.getConsumerName() + ".changeLogConsumerClass");
55  
56      Class<ChangeLogConsumerBase> changeLogConsumerClass = GrouperUtil.forName(changeLogConsumerClassName);
57      ChangeLogConsumerBase changeLogConsumerBase = GrouperUtil.newInstance(changeLogConsumerClass);
58      
59      int size = GrouperUtil.length(grouperMessageList);
60      
61      List<ChangeLogEntry> changeLogEntries = new ArrayList<ChangeLogEntry>();
62      
63      Iterator<GrouperMessage> grouperMessageIterator = grouperMessageList.iterator();
64  
65      Map<Long, GrouperMessage> changeLogIdToGrouperMessage = new LinkedHashMap<Long, GrouperMessage>();
66      
67      for (int i=0;i<size;i++) {
68        
69        GrouperMessage grouperMessage = grouperMessageIterator.next();
70        
71        try {
72          Collection<ChangeLogEntry> changeLogEntriesForMessage = ChangeLogEntry.fromJsonToCollection(grouperMessage.getMessageBody());
73          
74          for (ChangeLogEntry changeLogEntry : GrouperUtil.nonNull(changeLogEntriesForMessage)) {
75            changeLogIdToGrouperMessage.put(changeLogEntry.getSequenceNumber(), grouperMessage);
76          }
77          
78          changeLogEntries.addAll(GrouperUtil.nonNull(changeLogEntriesForMessage));
79          
80        } catch (Exception e) {
81          LOG.error("Cannot convert message body to change log entry: '" 
82              + messagingListenerMetadata.getConsumerName() + "''" + grouperMessage.getMessageBody() + "'", e);
83          //dont get this message again
84          GrouperMessagingEngine.acknowledge(new GrouperMessageAcknowledgeParam().assignGrouperMessageSystemName(messageSystemName)
85              .assignQueueName(queue).addGrouperMessage(grouperMessage).assignAcknowledgeType(GrouperMessageAcknowledgeType.mark_as_processed));
86        }
87      }
88      
89      if (GrouperUtil.length(changeLogEntries) == 0) {
90        return;
91      }
92      
93      ChangeLogProcessorMetadata changeLogProcessorMetadata = new ChangeLogProcessorMetadata();
94      changeLogProcessorMetadata.setConsumerName(messagingListenerMetadata.getConsumerName());
95      changeLogProcessorMetadata.setHib3GrouperLoaderLog(messagingListenerMetadata.getHib3GrouperLoaderLog());
96  
97      long lastProcessed = -1;
98      
99      try {
100       lastProcessed = changeLogConsumerBase.processChangeLogEntries(changeLogEntries, changeLogProcessorMetadata);
101 
102       messagingListenerMetadata.setHadProblem(changeLogProcessorMetadata.isHadProblem());
103       messagingListenerMetadata.setRecordException(changeLogProcessorMetadata.getRecordException());
104       messagingListenerMetadata.setRecordProblemText(changeLogProcessorMetadata.getRecordProblemText());
105 
106     } catch (Exception e) {
107       messagingListenerMetadata.setHadProblem(true);
108       messagingListenerMetadata.setRecordException(e);
109       messagingListenerMetadata.setRecordProblemText(e.getLocalizedMessage());
110       lastProcessed = changeLogProcessorMetadata.getRecordExceptionSequence() -1;
111     }
112     
113     if (lastProcessed > -1) {
114       Set<String> messageIdAcked = new HashSet<String>();
115       for (Long changeLogId : changeLogIdToGrouperMessage.keySet()) {
116         GrouperMessage grouperMessage = changeLogIdToGrouperMessage.get(changeLogId);
117 
118         // multiple change log entries can be in one message
119         if (messageIdAcked.contains(grouperMessage.getId())) {
120            continue;
121         }
122         messageIdAcked.add(grouperMessage.getId());
123         
124         boolean firstError = true;
125         
126         if (changeLogId <= lastProcessed) {
127 
128           //all good, ack
129           GrouperMessagingEngine.acknowledge(new GrouperMessageAcknowledgeParam().assignGrouperMessageSystemName(messageSystemName)
130               .assignQueueName(queue).addGrouperMessage(grouperMessage).assignAcknowledgeType(GrouperMessageAcknowledgeType.mark_as_processed));
131           
132         } else {
133 
134           if (firstError) {
135             messagingListenerMetadata.setRecordExceptionId(grouperMessage.getId());
136 
137             firstError = false;
138           }
139           
140           changeLogProcessorMetadata.setHadProblem(true);
141 
142           //not processed correctly
143           GrouperMessagingEngine.acknowledge(new GrouperMessageAcknowledgeParam().assignGrouperMessageSystemName(messageSystemName)
144               .assignQueueName(queue).addGrouperMessage(grouperMessage).assignAcknowledgeType(GrouperMessageAcknowledgeType.return_to_queue));
145         }
146       }
147     }
148     
149   }
