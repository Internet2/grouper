453   /**
454    * 
455    * @param writer
456    * @param xmlExportMain 
457    */
458   public static void exportAuditTypes(final Writer writer, final XmlExportMain xmlExportMain) {
459     //get the audittypes
460     HibernateSession.callbackHibernateSession(GrouperTransactionType.READONLY_OR_USE_EXISTING, AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
461       
462       public Object callback(HibernateHandlerBean hibernateHandlerBean)
463           throws GrouperDAOException {
464   
465         Session session = hibernateHandlerBean.getHibernateSession().getSession();
466   
467         //select all auditTypes in order
468         Query query = session.createQuery(
469             "select theAuditType from AuditType as theAuditType order by theAuditType.auditCategory, theAuditType.actionName");
470   
471         GrouperVersion grouperVersion = new GrouperVersion(GrouperVersion.GROUPER_VERSION);
472         try {
473           writer.write("  <auditTypes>\n");
474   
475           //this is an efficient low-memory way to iterate through a resultset
476           ScrollableResults results = null;
477           try {
478             results = query.scroll();
479             while(results.next()) {
480               Object object = results.get(0);
481               AuditType auditType = (AuditType)object;
482               XmlExportAuditType xmlExportAuditType = auditType.xmlToExportAuditType(grouperVersion);
483               writer.write("    ");
484               xmlExportAuditType.toXml(grouperVersion, writer);
485               writer.write("\n");
486               xmlExportMain.incrementRecordCount();
487             }
488           } finally {
489             HibUtils.closeQuietly(results);
490           }
491           
492           //end the auditTypes element 
493           writer.write("  </auditTypes>\n");
494         } catch (IOException ioe) {
495           throw new RuntimeException("Problem with streaming auditTypes", ioe);
496         }
497         return null;
498       }
499     });
500   }

532   /**
533    * parse the xml file for groups
534    * @param xmlImportMain
535    */
536   public static void processXmlFirstPass(final XmlImportMain xmlImportMain) {
537     xmlImportMain.getReader().addHandler( AUDIT_TYPES_XPATH, 
538         new ElementHandler() {
539             public void onStart(ElementPath path) {
540             }
541             public void onEnd(ElementPath path) {
542                 // process a ROW element
543                 Element row = path.getCurrent();
544   
545                 // prune the tree
546                 row.detach();
547             }
548         }
549     );
550   
551     xmlImportMain.getReader().addHandler( XML_EXPORT_AUDIT_TYPE_XPATH, 
552         new ElementHandler() {
553             public void onStart(ElementPath path) {
554                 // do nothing here...    
555             }
556             public void onEnd(ElementPath path) {
557                 // process a ROW element
558                 Element row = path.getCurrent();
559   
560                 // prune the tree
561                 row.detach();
562   
563                 xmlImportMain.incrementTotalImportFileCount();
564             }
565         }
566     );
567   
568   }

570   /**
571    * parse the xml file for groups
572    * @param xmlImportMain
573    */
574   public static void processXmlSecondPass(final XmlImportMain xmlImportMain) {
575     xmlImportMain.getReader().addHandler( AUDIT_TYPES_XPATH, 
576         new ElementHandler() {
577             public void onStart(ElementPath path) {
578             }
579             public void onEnd(ElementPath path) {
580                 // process a ROW element
581                 Element row = path.getCurrent();
582   
583                 // prune the tree
584                 row.detach();
585             }
586         }
587     );
588   
589     xmlImportMain.getReader().addHandler( XML_EXPORT_AUDIT_TYPE_XPATH, 
590         new ElementHandler() {
591             public void onStart(ElementPath path) {
592                 // do nothing here...    
593             }
594             public void onEnd(ElementPath path) {
595   
596               Element row = null;
597               try {
598                 // process a ROW element
599                 row = path.getCurrent();
600   
601                 // prune the tree
602                 row.detach();
603   
604                 XmlExportAuditType xmlExportAuditTypeFromFile = (XmlExportAuditType)xmlImportMain.getXstream().unmarshal(new Dom4JReader(row));
605                 
606                 AuditType auditType = xmlExportAuditTypeFromFile.toAuditType();
607                 
608                 XmlExportUtils.syncImportable(auditType, xmlImportMain);
609                 
610                 xmlImportMain.incrementCurrentCount();
611               } catch (RuntimeException re) {
612                 LOG.error("Problem importing audittype: " + XmlExportUtils.toString(row), re);
613                 throw re;
614               }
615             }
616         }
617     );
618   
619   }

621   /**
622    * convert to audit type
623    * @return the audit type
624    */
625   public AuditType toAuditType() {
626     AuditType auditType = new AuditType();
627     
628     auditType.setActionName(this.getActionName());
629     auditType.setAuditCategory(this.getAuditCategory());
630     auditType.setContextId(this.contextId);
631     auditType.setCreatedOnDb(GrouperUtil.dateLongValue(this.createdOn));
632     auditType.setHibernateVersionNumber(this.hibernateVersionNumber);
633     auditType.setId(this.id);
634     auditType.setLabelInt01(this.labelInt01);
635     auditType.setLabelInt02(this.labelInt02);
636     auditType.setLabelInt03(this.labelInt03);
637     auditType.setLabelInt04(this.labelInt04);
638     auditType.setLabelInt05(this.labelInt05);
639     auditType.setLabelString01(this.labelString01);
640     auditType.setLabelString02(this.labelString02);
641     auditType.setLabelString03(this.labelString03);
642     auditType.setLabelString04(this.labelString04);
643     auditType.setLabelString05(this.labelString05);
644     auditType.setLabelString06(this.labelString06);
645     auditType.setLabelString07(this.labelString07);
646     auditType.setLabelString08(this.labelString08);
647     auditType.setLastUpdatedDb(GrouperUtil.dateLongValue(this.lastUpdated));
648     return auditType;
649   }
