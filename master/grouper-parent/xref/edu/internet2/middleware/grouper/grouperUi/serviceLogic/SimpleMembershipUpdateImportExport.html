129   /**
130    * export all fields of a group
131    * @param group
132    * @param headersCommaSeparated
133    * @param exportAllSortField
134    * @param simpleMembershipUpdateContainer
135    * @param immediateOnly
136    * @throws IOException
137    */
138   public static void exportGroupAllFieldsToBrowser(Group group, String headersCommaSeparated, String exportAllSortField, boolean immediateOnly) {
139     
140     try {
141       Set<Member> members = immediateOnly ? group.getImmediateMembers() : group.getMembers();
142       
143       Member.resolveSubjects(members, true);
144       
145       HttpServletResponse response = GrouperUiFilter.retrieveHttpServletResponse(); 
146       
147       String[] headers = GrouperUtil.splitTrim(headersCommaSeparated, ",");
148       
149       //note: isError is second to last col, error is the last column
150       boolean[] isAttribute = new boolean[headers.length];
151       int sortCol = 0;
152       int sourceIdCol = -1;
153       for (int i=0;i<headers.length;i++) {
154         isAttribute[i] = !nonAttributeCols.contains(headers[i].toLowerCase());
155   
156         if (StringUtils.equalsIgnoreCase(headers[i], exportAllSortField)) {
157           sortCol = i;
158         } else if (StringUtils.equalsIgnoreCase("sourceId", headers[i])) {
159           sourceIdCol = i;
160         }
161       }
162       
163       List<String[]> memberData = new ArrayList<String[]>(); 
164       for (Member member : members) {
165         // feed in your array (or convert your data to an array)
166         String[] entries = exportAllStringArray(member, headers, isAttribute);
167         memberData.add(entries);
168       }      
169    
170       final int SOURCE_ID_COL = sourceIdCol;
171       final int SORT_COL = sortCol;
172       //sort
173       Collections.sort(memberData, new Comparator() {
174    
175         /**
176          * 
177          * @param o1
178          * @param o2
179          * @return 1, -1, 0
180          */
181         public int compare(Object o1, Object o2) {
182           String[] first = (String[])o1;
183           String[] second = (String[])o2;
184           if (SOURCE_ID_COL != -1 && !StringUtils.equals(first[SOURCE_ID_COL], second[SOURCE_ID_COL])) {
185             return GrouperUiUtils.compare(first[SOURCE_ID_COL], second[SOURCE_ID_COL], true);
186           }
187           return GrouperUiUtils.compare(first[SORT_COL], second[SORT_COL], true);
188         }
189       });
190       
191       //say it is CSV
192       response.setContentType(HttpContentType.TEXT_CSV.getContentType());
193     
194       String groupExtensionFileName = GuiGroup.getExportAllFileNameStatic(group);
195       
196       response.setHeader ("Content-Disposition", "inline;filename=\"" + groupExtensionFileName + "\"");
197       
198       //just write some stuff
199       PrintWriter out = null;
200     
201       try {
202         out = response.getWriter();
203       } catch (Exception e) {
204         throw new RuntimeException("Cant get response.getWriter: ", e);
205       }
206       
207       CSVWriter writer = new CSVWriter(out);
208       String[] headersNew = new String[headers.length+2];
209       System.arraycopy(headers, 0, headersNew, 0, headers.length);
210       headersNew[headers.length] = "success";
211       headersNew[headers.length+1] = "errorMessage";
212       writer.writeNext(headersNew);
213       for (String[] entries: memberData) {
214         // feed in your array (or convert your data to an array)
215         writer.writeNext(entries);
216       }      
217       writer.close();
218       auditExport(group.getUuid(), group.getName(), memberData.size(), groupExtensionFileName);
219  
220     } catch (NoSessionException se) {
221       throw se;
222     } catch (Exception se) {
223       throw new RuntimeException("Error exporting all members from group: " + group.getName() 
224           + ", " + se.getMessage(), se);
225     }
226     throw new ControllerDone();
227   }

229   /**
230    * export all members
231    * @param member
232    * @param headers 
233    * @param isAttribute which indexes are attributes
234    * @return the stirng array for csv
235    */
236   public static String[] exportAllStringArray(Member member, String[] headers, boolean[] isAttribute) {
237     String[] result = new String[headers.length+2];
238     
239     //lets see what we can get from the member
240     for (int i=0;i<headers.length;i++) {
241       String header = headers[i];
242       if ("subjectId".equalsIgnoreCase(header)) {
243         result[i] = member.getSubjectId();
244       } else if ("entityId".equalsIgnoreCase(header)) {
245         result[i] = member.getSubjectId();
246       } else if ("sourceId".equalsIgnoreCase(header)) {
247         result[i] = member.getSubjectSourceId();
248       } else if ("memberId".equalsIgnoreCase(header)) {
249         result[i] = member.getUuid();
250       }
251     }
252     
253     
254     try {
255       
256       Subject subject = member.getSubject();
257       
258       //lets see what we can get from the subject
259       for (int i=0;i<headers.length;i++) {
260         String header = headers[i];
261         if ("name".equalsIgnoreCase(header)) {
262           result[i] = subject.getName();
263         } else if ("description".equalsIgnoreCase(header)) {
264           result[i] = subject.getDescription();
265         } else if ("screenLabel".equalsIgnoreCase(header)) {
266           result[i] = GrouperUiUtils.convertSubjectToLabelConfigured(subject);
267         } else if (isAttribute[i]) {
268           result[i] = subject.getAttributeValueOrCommaSeparated(header);
269         }
270       }
271       
272       result[headers.length] = "T";
273     } catch (NoSessionException se) {
274       throw se;
275     } catch (Exception e) {
276       result[headers.length] = "F";
277       String error = "error with memberId: " + member.getUuid() + ", subjectId: " + member.getSubjectId()
278         + ", " + ExceptionUtils.getFullStackTrace(e);
279       LOG.error(error);
280       result[headers.length + 1] = error;
281     }
282     return result;
283   }

285   /**
286    * export group subject ids
287    * @param group
288    * @param immediateOnly
289    */
290   public static void exportGroupSubjectIdsCsv(Group group, boolean immediateOnly) {
291     
292     try {
293   
294       Set<Member> members = immediateOnly ? group.getImmediateMembers() : group.getMembers();
295       
296       HttpServletResponse response = GrouperUiFilter.retrieveHttpServletResponse(); 
297       
298       List<String[]> memberData = new ArrayList<String[]>(); 
299       for (Member member : members) {
300         // feed in your array (or convert your data to an array)
301         String[] entries = new String[]{member.getSubjectSourceId(), member.getSubjectId()};
302         memberData.add(entries);
303       }      
304   
305       //sort
306       Collections.sort(memberData, new Comparator() {
307   
308         /**
309          * 
310          * @param o1
311          * @param o2
312          * @return 1, -1, 0
313          */
314         public int compare(Object o1, Object o2) {
315           String[] first = (String[])o1;
316           String[] second = (String[])o2;
317           if (!StringUtils.equals(first[0], second[0])) {
318             return GrouperUiUtils.compare(first[0], second[0], true);
319           }
320           return GrouperUiUtils.compare(first[1], second[1], true);
321         }
322       });
323       
324       //say it is CSV
325       response.setContentType("text/csv");
326     
327       String groupExtensionFileName = GuiGroup.getExportSubjectIdsFileNameStatic(group);
328       
329       response.setHeader ("Content-Disposition", "inline;filename=\"" + groupExtensionFileName + "\"");
330       
331       //just write some stuff
332       PrintWriter out = null;
333     
334       try {
335         out = response.getWriter();
336       } catch (Exception e) {
337         throw new RuntimeException("Cant get response.getWriter: ", e);
338       }
339       
340       CSVWriter writer = new CSVWriter(out);
341       writer.writeNext(new String[]{"sourceId", "entityId"});
342       for (String[] entries: memberData) {
343         // feed in your array (or convert your data to an array)
344         writer.writeNext(entries);
345       }      
346       writer.close();
347       
348       auditExport(group.getUuid(), group.getName(), memberData.size(), groupExtensionFileName);
349       throw new ControllerDone();
350     } catch (NoSessionException se) {
351       throw se;
352     } catch (ControllerDone cd) {
353       throw cd;
354     } catch (Exception se) {
355       throw new RuntimeException("Error exporting all members from group: " + group.getName()
356           + ", " + se.getMessage(), se);
357     }
358     
359   }

361 	private static void auditExport(final String groupId, final String groupName, final int exportSize, final String groupExtensionFileName) {
362 		HibernateSession.callbackHibernateSession(
363 	          GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, AuditControl.WILL_AUDIT,
364 	    	      new HibernateHandler() {
365 	    	          public Object callback(HibernateHandlerBean hibernateHandlerBean)
366 	    	              throws GrouperDAOException {
367 	    	        	  
368 	    	        	  AuditEntry auditEntry = new AuditEntry(AuditTypeBuiltin.MEMBERSHIP_GROUP_EXPORT, 
369 	    	        			  "exportSize", String.valueOf(exportSize), "groupId", groupId,
370 	    	        			  "groupName", groupName, "file", groupExtensionFileName);
371 	    	                    
372 	    	              String description = "exported : " + exportSize + " subjects from "+groupName + " in : " + groupExtensionFileName 
373 	    	                  + " file.";
374 	    	              auditEntry.setDescription(description);
375 	    	              auditEntry.saveOrUpdate(true);
376 	    	        	  
377 	    	        	  return null;
378 	    	          }
379 	      });
380 	}

411   /**
412    * import a CSV file
413    * @param httpServletRequest
414    * @param httpServletResponse
415    */
416   @SuppressWarnings("unchecked")
417   public void importCsv(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
418   
419     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
420     guiResponseJs.setAddTextAreaTag(true);
421     guiResponseJs.addAction(GuiScreenAction.newCloseModal());
422   
423     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
424     
425     GrouperSession grouperSession = null;
426   
427     Group group = null;
428     String groupName = null;
429   
430     try {
431       SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
432       grouperSession = GrouperSession.start(loggedInSubject);
433   
434       group = new SimpleMembershipUpdate().retrieveGroup(grouperSession);
435       groupName = group.getName();
436       
437       List<Member> existingMembers = new ArrayList<Member>(GrouperUtil.nonNull(group.getImmediateMembers()));
438       
439       int existingCount = GrouperUtil.length(existingMembers);
440   
441       GrouperRequestWrapper grouperRequestWrapper = (GrouperRequestWrapper)httpServletRequest;
442       
443       Reader reader = null;
444       String fileName = null;
445       
446       GuiHideShow membershipLiteImportFileHideShow = GuiHideShow.retrieveHideShow("membershipLiteImportFile", true);
447       
448       if (membershipLiteImportFileHideShow.isShowing()) {
449         
450         FileItem importCsvFile = grouperRequestWrapper.getParameterFileItem("importCsvFile");
451         
452         fileName = StringUtils.defaultString(importCsvFile == null ? "" : importCsvFile.getName());
453     
454         String fileSize = importCsvFile == null ? "" : (" : " + FileUtils.byteCountToDisplaySize(importCsvFile.getSize()));
455         
456         fileName += fileSize;
457         
458         //validate the inputs, file is required
459         if (importCsvFile == null || importCsvFile.getSize() == 0 || 
460             importCsvFile.getName() == null 
461             || (!importCsvFile.getName().toLowerCase().endsWith(".csv")
462               &&  !importCsvFile.getName().toLowerCase().endsWith(".txt"))) {
463           
464           guiResponseJs.addAction(GuiScreenAction.newAlert("<pre>"  + simpleMembershipUpdateContainer.getText().getImportErrorNoWrongFile()
465               + fileName + "</pre>"));
466           return;
467         }
468         
469         reader = new InputStreamReader(importCsvFile.getInputStream());
470         
471       } else {
472         
473         //textarea import
474         String importCsvTextarea = grouperRequestWrapper.getParameter("importCsvTextarea");
475         
476         if (StringUtils.isBlank(importCsvTextarea)) {
477           guiResponseJs.addAction(GuiScreenAction.newAlert("<pre>" 
478               + simpleMembershipUpdateContainer.getText().getImportErrorBlankTextarea()
479               + "</pre>"));
480           return;
481           
482         }
483         
484         fileName = "textareaInput";
485         
486         reader = new StringReader(importCsvTextarea);
487       }
488       
489       
490       boolean importReplaceMembers = grouperRequestWrapper.getParameterBoolean("importReplaceMembers", false);
491       
492       //convert the import file to subjects
493       List<String> subjectErrors = new ArrayList<String>();
494       
495       List<Subject> importedSubjectWrappers = parseCsvImportFile(
496           reader, fileName, subjectErrors, new LinkedHashMap<String, Integer>(),
497           membershipLiteImportFileHideShow.isShowing());
498 
499       final List<String> addErrors = new ArrayList<String>();
500 
501       //filter out the require group
502       final String requireGroup = simpleMembershipUpdateContainer.configValue(
503           "simpleMembershipUpdate.subjectSearchRequireGroup", false);
504       
505       if (!StringUtils.isBlank(requireGroup)) {
506 
507         final List<Subject> SUBJECTS = importedSubjectWrappers;
508         
509         GrouperSession.callbackGrouperSession(
510             grouperSession.internal_getRootSession(), new GrouperSessionHandler() {
511           
512           @Override
513           public Object callback(GrouperSession rootGrouperSession) throws GrouperSessionException {
514             
515             Group groupFilter = GroupFinder.findByName(rootGrouperSession, requireGroup, true);
516             
517             Iterator<Subject> iterator = SUBJECTS.iterator();
518             
519             while (iterator.hasNext()) {
520               
521               Subject subject = iterator.next();
522               
523               if (!groupFilter.hasMember(subject)) {
524                 iterator.remove();
525                 String error = "Subject " + ((ImportSubjectWrapper)subject).errorLabelForError() 
526                   + " not in required group " + groupFilter.getName();
527                 addErrors.add(error);
528               }
529               
530               
531             }
532             return null;
533           }
534         });
535       }
536       
537       GrouperUiUtils.removeOverlappingSubjects(existingMembers, importedSubjectWrappers);
538       
539       int addedCount = 0;
540       
541       //first lets add some members
542       for (int i=0;i<importedSubjectWrappers.size();i++) {
543         
544         ImportSubjectWrapper importedSubjectWrapper = (ImportSubjectWrapper)importedSubjectWrappers.get(i);
545         Subject wrappedSubject = null;
546         try {
547           wrappedSubject = importedSubjectWrapper.wrappedSubject();
548         } catch (Exception e) {
549           //ignore
550         }
551         try {
552           if (wrappedSubject != null) {
553             
554             group.addMember(wrappedSubject, false);
555           } else {
556             
557             group.addMember(importedSubjectWrapper, false);
558           }
559           
560           addedCount++;
561         } catch (Exception e) {
562           String error = "Error adding subject from " + importedSubjectWrapper.errorLabelForError() + ", " + e.getMessage();
563           LOG.warn(error, e);
564           addErrors.add(error);
565         }
566   
567       }
568   
569       boolean didntImportDueToSubjects = false;
570       int deletedCount = 0;
571       List<String> deleteErrors = new ArrayList<String>();
572   
573       //remove the ones which are already there
574       if (importReplaceMembers) {
575         
576         if (GrouperUtil.length(subjectErrors) == 0) {
577           
578           for (Member existingMember : existingMembers) {
579             
580             try {
581               group.deleteMember(existingMember, false);
582               deletedCount++;
583             } catch (Exception e) {
584               String error = "Error deleting subject " + SubjectHelper.getPretty(existingMember) + e.getMessage();
585               LOG.warn(error, e);
586               deleteErrors.add(error);
587             }
588           }
589         } else {
590           didntImportDueToSubjects = true;
591         }
592         
593         
594       }
595       
596       //this might be a little wasteful, but I think it is a good sanity check
597       int newSize = group.getImmediateMembers().size();
598       
599       StringBuilder result = new StringBuilder();
600       //result.append("File: " + importCsvFile.getName() + ", " + importCsvFile.getFieldName() 
601       //    + ", isFormField: " + importCsvFile.isFormField() + ", inMemory: " 
602       //    + importCsvFile.isInMemory() + "\n\n" + importCsvFile.getString() + "\n\n");
603       //result.append("importReplaceMembers: " + importReplaceMembers + "\n\n");
604       
605       //first of all, was it successful?
606       int errorSize = GrouperUtil.length(subjectErrors) + GrouperUtil.length(addErrors) 
607         + GrouperUtil.length(deleteErrors);
608       boolean hasError = errorSize > 0;
609       if (!hasError) {
610         
611         result.append("<b>").append(simpleMembershipUpdateContainer.getText().getImportSuccessSummary()).append("</b><br /><br />\n");
612         
613       } else {
614         
615         result.append("<b>").append(simpleMembershipUpdateContainer.getText().getImportErrorSummary(errorSize)).append("</b><br /><br />\n");
616         
617       }
618       
619       //give general summary
620       result.append(simpleMembershipUpdateContainer.getText().getImportSizeSummary(existingCount, newSize)).append("<br />\n");
621   
622       if (didntImportDueToSubjects) {
623         result.append(simpleMembershipUpdateContainer.getText().getImportErrorSubjectProblems()).append("<br />\n");
624       }
625       
626       //adds, deletes
627       result.append(simpleMembershipUpdateContainer.getText().getImportAddsDeletesSummary(addedCount, deletedCount)).append("<br />\n");
628       
629       if (GrouperUtil.length(subjectErrors) > 0) {
630         result.append("<br /><b>").append(simpleMembershipUpdateContainer.getText().getImportSubjectErrorsLabel())
631             .append("</b><br />\n");
632         for (String error: subjectErrors) {
633           result.append(error).append("<br />\n");
634         }
635       }
636       
637       if (GrouperUtil.length(addErrors) > 0) {
638         result.append("<br /><b>").append(simpleMembershipUpdateContainer.getText().getImportAddErrorsLabel())
639             .append("</b><br />\n");
640         for (String error: addErrors) {
641           result.append(error).append("<br />\n");
642         }
643       }
644   
645       if (GrouperUtil.length(deleteErrors) > 0) {
646         result.append("<br /><b>").append(simpleMembershipUpdateContainer.getText().getImportRemoveErrorsLabel())
647           .append("</b><br />\n");
648         for (String error: deleteErrors) {
649           result.append(error).append("<br />\n");
650         }
651       }
652       
653       guiResponseJs.addAction(GuiScreenAction.newAlert(result.toString(), false));
654   
655     } catch (NoSessionException se) {
656       throw se;
657     } catch (ControllerDone cd) {
658       throw cd;
659     } catch (Exception se) {
660       throw new RuntimeException("Error importing members to group: " + groupName 
661           + ", " + se.getMessage(), se);
662     } finally {
663       GrouperSession.stopQuietly(grouperSession); 
664     }
665     
666     //refresh list... since it probably changed
667     new SimpleMembershipUpdate().retrieveMembers(httpServletRequest, httpServletResponse);
668     
669   }

709   /**
710    * Note, this will close the reader passed in
711    * @param originalReader
712    * @param fileName
713    * @param subjectErrors pass in a list and errors will be put in here
714    * @param errorSubjectIdsOnRow is a map where the key is subjectId and the value if the line number
715    * @param isFileUpload true if file upload, as opposed to a textarea
716    * @return the list, never null
717    * @throws GrouperImportException for messages to the screen
718    */
719   @SuppressWarnings("unchecked")
720   public static List<Subject> parseCsvImportFile(Reader originalReader, String fileName, List<String> subjectErrors, 
721       Map<String, Integer> errorSubjectIdsOnRow, boolean isFileUpload) throws GrouperImportException {
722     
723     //convert from CSV to 
724     CSVReader reader = null;
725     
726     try {
727       //note, the first row is the title
728       List<String[]> csvEntries = null;
729     
730       try {
731         reader = new CSVReader(originalReader);
732         csvEntries = reader.readAll();
733       } catch (IOException ioe) {
734         throw new GrouperImportException("Error processing file: " + fileName, ioe);
735       }
736       
737       List<Subject> uploadedSubjects = new ArrayList<Subject>();
738       
739       //lets get the headers
740       int sourceIdColumn = -1;
741       int subjectIdColumn = -1;
742       int subjectIdentifierColumn = -1;
743       int subjectIdOrIdentifierColumn = -1;
744       
745       //must have lines
746       if (GrouperUtil.length(csvEntries) == 0) {
747         if (isFileUpload) {
748           throw new GrouperImportException(GrouperUiUtils.message("simpleMembershipUpdate.importErrorNoWrongFile"));
749         }
750         throw new GrouperImportException(GrouperUiUtils.message("simpleMembershipUpdate.importErrorBlankTextarea"));
751       }
752       
753       //lets go through the headers
754       String[] headers = csvEntries.get(0);
755       int headerSize = headers.length;
756       boolean foundHeader = false;
757       for (int i=0;i<headerSize;i++) {
758         if ("sourceId".equalsIgnoreCase(headers[i])) {
759           foundHeader = true;
760           sourceIdColumn = i;
761         }
762         if ("subjectId".equalsIgnoreCase(headers[i]) || "entityId".equalsIgnoreCase(headers[i])) {
763           foundHeader = true;
764           subjectIdColumn = i;
765         }
766         if ("subjectIdentifier".equalsIgnoreCase(headers[i]) || "entityIdentifier".equalsIgnoreCase(headers[i])) {
767           foundHeader = true;
768           subjectIdentifierColumn = i;
769         }
770         if ("subjectIdOrIdentifier".equalsIgnoreCase(headers[i]) || "entityIdOrIdentifier".equalsIgnoreCase(headers[i])) {
771           foundHeader = true;
772           subjectIdOrIdentifierColumn = i;
773         }
774       }
775       
776       //normally start on index 1, if the first row is header
777       int startIndex = 1;
778       
779       //must pass in an id
780       if (subjectIdColumn == -1 && subjectIdentifierColumn == -1 && subjectIdOrIdentifierColumn == -1) {
781         if (!foundHeader && headerSize == 1) {
782           //there was no header, so pretend like it was subjectIdOrIdentifier
783           subjectIdOrIdentifierColumn = 0;
784           startIndex = 0;
785         } else {
786           throw new GrouperImportException(TextContainer.retrieveFromRequest().getText().get("simpleMembershipUpdate.importErrorNoIdCol"));
787         }
788       }
789       
790       //ok, lets go through the rows, start after the headers
791       for (int i=startIndex;i<csvEntries.size();i++) {
792         String[] csvEntry = csvEntries.get(i);
793         int row = i+1;
794         
795         //try catch each one and see where we get
796         String subjectId = null;
797         String subjectIdentifier = null;
798         String subjectIdOrIdentifier = null;
799         try {
800           String sourceId = null;
801     
802           sourceId = sourceIdColumn == -1 ? null : csvEntry[sourceIdColumn]; 
803           subjectId = subjectIdColumn == -1 ? null : csvEntry[subjectIdColumn]; 
804           subjectIdentifier = subjectIdentifierColumn == -1 ? null : csvEntry[subjectIdentifierColumn]; 
805           subjectIdOrIdentifier = subjectIdOrIdentifierColumn == -1 ? null : csvEntry[subjectIdOrIdentifierColumn]; 
806           
807           ImportSubjectWrapper importSubjectWrapper = 
808             new ImportSubjectWrapper(row, sourceId, subjectId, subjectIdentifier, subjectIdOrIdentifier, csvEntry);
809           uploadedSubjects.add(importSubjectWrapper);
810           
811         } catch (Exception e) {
812           LOG.info(e);
813           subjectErrors.add("Error on " + ImportSubjectWrapper.errorLabelForRowStatic(row, csvEntry) + ": " +    e.getMessage());
814           
815           String errorSubjectId = StringUtils.defaultIfEmpty(subjectId, subjectIdentifier);
816           errorSubjectId = StringUtils.defaultIfEmpty(errorSubjectId, subjectIdOrIdentifier);
817           
818           errorSubjectIdsOnRow.put(errorSubjectId, row);
819         }
820       
821       }
822       
823       return uploadedSubjects;
824       
825     } finally {
826       if (reader != null) {
827         try {
828           reader.close();
829         } catch (Exception e) {
830           LOG.warn("error", e);
831         }
832       }
833     }
834     
835   }
