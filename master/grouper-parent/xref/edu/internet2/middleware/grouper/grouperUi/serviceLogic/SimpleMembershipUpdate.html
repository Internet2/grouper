78     * submit the enabled / disabled dates popup
79     * @param request
80     * @param response

82    public void enabledDisabledSubmit(HttpServletRequest request, HttpServletResponse response) {
83      
84      final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
85  
86      GrouperSession grouperSession = null;
87      try {
88  
89        grouperSession = GrouperSession.start(loggedInSubject);
90  
91        String enabledDate = request.getParameter("enabledDate");
92        String disabledDate = request.getParameter("disabledDate");
93        SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
94        
95        Membership membership = simpleMembershipUpdateContainer.getEnabledDisabledMember().getMembership();
96        
97        if (StringUtils.isBlank(enabledDate) ) {
98          membership.setEnabledTime(null);
99        } else {
100         //must be yyyy/mm/dd
101         Timestamp enabledTimestamp = GrouperUtil.toTimestamp(enabledDate);
102         membership.setEnabledTime(enabledTimestamp);
103       }
104       
105       if (StringUtils.isBlank(disabledDate) ) {
106         membership.setDisabledTime(null);
107       } else {
108         //must be yyyy/mm/dd
109         Timestamp disabledTimestamp = GrouperUtil.toTimestamp(disabledDate);
110         membership.setDisabledTime(disabledTimestamp);
111       }
112       
113       membership.update();
114       
115       
116       GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
117       guiResponseJs.addAction(GuiScreenAction.newCloseModal());
118       guiResponseJs.addAction(GuiScreenAction.newAlert(simpleMembershipUpdateContainer.getText().getEnabledDisabledSuccess()));
119   
120       //refresh this list
121       retrieveMembers(request, response);
122 
123     } finally {
124       GrouperSession.stopQuietly(grouperSession); 
125     }
126

240   /**
241    * get a group from app state
242    * @param grouperSession
243    * @return the group
244    */
245   private Group retrieveGroupHelper(GrouperSession grouperSession) {
246 
247     final SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
248     
249     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
250 
251     AppState appState = AppState.retrieveFromRequest();
252 
253     //lets get the group
254     String id = null;
255     String name = null;
256 
257     HttpServletResponse response = GrouperUiFilter.retrieveHttpServletResponse();
258     HttpServletRequest request = GrouperUiFilter.retrieveHttpServletRequest();
259 
260     if (appState.getUrlArgObjects() != null) {
261       id = appState.getUrlArgObjects().get("groupId");
262       if (StringUtils.isBlank(id)) {
263         id = request.getParameter("groupId");
264       }
265       name = appState.getUrlArgObjects().get("groupName");
266       if (StringUtils.isBlank(name)) {
267         name = request.getParameter("groupName");
268       }
269     }
270 
271     GuiGroup guiGroup = simpleMembershipUpdateContainer.getGuiGroup();
272     if (guiGroup != null && guiGroup.getGroup() != null) {
273 
274       //make sure two browser windows in the same session arent editing the same group
275       if ((!StringUtils.isBlank(id) && !StringUtils.equals(id, guiGroup.getGroup().getId()))
276           || (!StringUtils.isBlank(name) && !StringUtils.equals(name, guiGroup.getGroup().getName()))) {
277         guiResponseJs.addAction(GuiScreenAction.newAlert(simpleMembershipUpdateContainer.getText().getErrorTooManyBrowsers()));
278         init(request, response);
279         throw new ControllerDone(true);
280       }
281       
282       return guiGroup.getGroup();
283     }
284     
285     if (StringUtils.isBlank(id) && StringUtils.isBlank(name)) {
286       //make sure the URL is ok
287       guiResponseJs.addAction(GuiScreenAction.newScript("location.href = 'grouper.html#operation=SimpleMembershipUpdate.index'"));
288 
289       if ("".equals(id)) {
290         //if the id is there, but empty, maybe they didnt enter anything...
291         guiResponseJs.addAction(GuiScreenAction.newAlert(GrouperUiUtils.message("simpleMembershipUpdate.errorGroupSearchNothingEntered", false)));
292         
293       } else {
294         guiResponseJs.addAction(GuiScreenAction.newAlert(GrouperUiUtils.message("simpleMembershipUpdate.errorGroupSearchNoParams", false)));
295       }
296       index(request, response);
297       throw new ControllerDone(true);
298     }
299     Group group = null;
300     try {
301       if (!StringUtils.isBlank(id)) {
302         group = GroupFinder.findByUuid(grouperSession, id, true);
303       } else if (!StringUtils.isBlank(name)) {
304         group = GroupFinder.findByName(grouperSession, name, false);
305       }
306     } catch (GroupNotFoundException gnfe) {
307       //ignore
308     } catch (Exception e) {
309       throw new RuntimeException("Problem in simpleMembershipUpdateInit : " + GrouperUiUtils.escapeHtml(id, true) 
310           + ", " + GrouperUiUtils.escapeHtml(name, true) + ", " + e.getMessage(), e);
311     }
312     
313     if (group == null) {
314       guiResponseJs.addAction(GuiScreenAction.newScript("location.href = 'grouper.html#operation=SimpleMembershipUpdate.index'"));
315       guiResponseJs.addAction(GuiScreenAction.newAlert(GrouperUiUtils.message("simpleMembershipUpdate.errorGroupSearchCantFindGroup", false)));
316       index(request, response);
317       throw new ControllerDone(true);
318     }
319     
320     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
321 
322     final Group GROUP = group;
323     
324     //do this as root
325     GrouperSession.callbackGrouperSession(grouperSession.internal_getRootSession(), new GrouperSessionHandler() {
326 
327       public Object callback(GrouperSession theGrouperSession)
328           throws GrouperSessionException {
329         boolean hasRead = GROUP.hasRead(loggedInSubject);
330         boolean hasUpdate = GROUP.hasUpdate(loggedInSubject);
331         boolean hasAdmin = GROUP.hasAdmin(loggedInSubject);
332         
333         simpleMembershipUpdateContainer.setCanReadGroup(hasAdmin || hasRead);
334         simpleMembershipUpdateContainer.setCanUpdateGroup(hasAdmin || hasUpdate);
335         return null;
336       }
337       
338     });
339 
340     if (!simpleMembershipUpdateContainer.isCanReadGroup() && !simpleMembershipUpdateContainer.isCanUpdateGroup()) {
341       guiResponseJs.addAction(GuiScreenAction.newScript("location.href = 'grouper.html#operation=SimpleMembershipUpdate.index'"));
342       guiResponseJs.addAction(GuiScreenAction.newAlert(GrouperUiUtils.message("simpleMembershipUpdate.errorGroupSearchPermissions", false)));
343       index(request, response);
344       throw new ControllerDone(true);
345     }
346 
347     
348     if (group.hasComposite()) {
349       guiResponseJs.addAction(GuiScreenAction.newScript("location.href = 'grouper.html#operation=SimpleMembershipUpdate.index'"));
350       guiResponseJs.addAction(GuiScreenAction.newAlert(GrouperUiUtils.message("simpleMembershipUpdate.errorGroupComposite", false)));
351       index(request, response);
352       throw new ControllerDone(true);
353       
354     }        
355     //store this in session
356     simpleMembershipUpdateContainer.setGuiGroup(new GuiGroup(group));
357     return group;
358   }

361    * retrieve members
362    * @param request
363    * @param response
364    */
365   @SuppressWarnings("unchecked")
366   public void retrieveMembers(HttpServletRequest request, HttpServletResponse response) {
367     
368     final SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
369     
370     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
371     
372     //init the pager
373     GuiPaging guiPaging = GuiPaging.retrievePaging("simpleMemberUpdateMembers", true);
374     
375     GrouperSession grouperSession = null;
376     Group group = null;
377     String groupName = null;
378     try {
379       grouperSession = GrouperSession.start(loggedInSubject);
380 
381       group = this.retrieveGroup(grouperSession);
382       groupName = group.getName();
383 
384       String simpleMembershipFilterMember = simpleMembershipUpdateContainer.getMemberFilter();
385 
386       //we have the group, now get the members
387       Set<Member> members = null;
388       if (StringUtils.isBlank(simpleMembershipFilterMember)) {
389         simpleMembershipUpdateContainer.setMemberFilterForScreen(null);
390         members = retrieveMembersNoFilter(guiPaging, group);
391         
392       } else {
393         members = new SimpleMembershipUpdateFilter().retrieveMembersFilter(guiPaging, group, simpleMembershipFilterMember, null); 
394       }      
395         
396       Set<Membership> memberships = group.getImmediateMemberships(Group.getDefaultList(), members, true);
397       Map<String, Membership> membershipMapByMemberId = new HashMap<String, Membership>();
398       for (Membership membership : memberships) {
399         //only add due to disabled date
400         if (membership.getDisabledTime() != null ) {
401           membershipMapByMemberId.put(membership.getMemberUuid(), membership);
402         }
403       }
404       
405       GuiMember[] guiMembers = new GuiMember[members.size()];
406       int i=0;
407       Membership membership = null;
408       for (Member member : members) {
409         guiMembers[i] = new GuiMember(member);
410         //TODO update this
411         guiMembers[i].setDeletable(true);
412         membership = membershipMapByMemberId.get(member.getUuid());
413         if (membership != null) {
414           guiMembers[i].setMembership(membership);
415         }
416         
417         i++;
418         
419       }
420       simpleMembershipUpdateContainer.setGuiMembers(guiMembers);
421     
422 //    Set<edu.internet2.middleware.grouper.Membership> memberships  = MembershipFinder.internal_findAllByGroupAndFieldAndPage(
423 //        group, Group.getDefaultList(), 0, 100, 200, numberOfRecords);
424 
425       GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
426       guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#simpleMembershipResultsList", 
427           "/WEB-INF/grouperUi/templates/simpleMembershipUpdate/simpleMembershipMembershipList.jsp"));
428     } catch (NoSessionException se) {
429       throw se;
430     } catch (ControllerDone cd) {
431       throw cd;
432     } catch (Exception e) {
433       throw new RuntimeException("Error with group: " + GrouperUiUtils.escapeHtml(groupName, false) + ", " + e.getMessage(), e);
434     } finally {
435       GrouperSession.stopQuietly(grouperSession); 
436     }
437       
438   
439   }

441   /**
442    * @param guiPaging
443    * @param group
444    * @return the members
445    * @throws SchemaException
446    */
447   @SuppressWarnings("unchecked")
448   Set<Member> retrieveMembersNoFilter(GuiPaging guiPaging, Group group)
449       throws SchemaException {
450     final SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
451 
452     Set<Member> members;
453     //get the size
454     QueryOptions queryOptions = new QueryOptions().retrieveCount(true).retrieveResults(false);
455     group.getImmediateMembers(Group.getDefaultList(), queryOptions);
456     int totalSize = queryOptions.getCount() == null ? 0 : queryOptions.getCount().intValue();
457     
458     int pageSize = guiPaging.getPageSize();
459     
460     guiPaging.setTotalRecordCount(totalSize);
461     guiPaging.setPageSize(pageSize);
462     
463     //if there are less than the sort limit, then just get all, no problem 
464     //  -- applies only if we're not doing sorting based on member attributes
465     int sortLimit = GrouperUiConfig.retrieveConfig().propertyValueInt("comparator.sort.limit", 200);
466     SortStringEnum sortStringEnum = simpleMembershipUpdateContainer.getSelectedSortStringEnum();
467     
468     QueryPaging queryPaging = new QueryPaging();
469     queryPaging.setPageSize(pageSize);
470     queryPaging.setPageNumber(guiPaging.getPageNumber());
471     
472     queryOptions = new QueryOptions().paging(queryPaging);
473     
474     if (sortStringEnum != null) {
475       members = group.getImmediateMembers(Group.getDefaultList(), null, queryOptions, sortStringEnum, null, null);
476     } else if (totalSize <= sortLimit) {
477       members = group.getImmediateMembers();
478       members = GrouperUiUtils.membersSortedPaged(members, queryPaging);
479     } else {
480       members = group.getMembers(Group.getDefaultList(), queryOptions);
481     }
482     Member.resolveSubjects(members, false);
483     guiPaging.setPageNumber(queryPaging.getPageNumber());
484     return members;
485   }

532   /**
533    * 
534    * @param httpServletRequest
535    * @param httpServletResponse
536    */
537   public void addMember(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
538     
539     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
540 
541     GrouperSession grouperSession = null;
542 
543     String comboValue = httpServletRequest.getParameter("simpleMembershipUpdateAddMember");
544 
545     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
546     SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
547     
548     //maybe they dont know how to use it
549     if (StringUtils.isBlank(comboValue)) {
550       guiResponseJs.addAction(GuiScreenAction.newAlert(
551           simpleMembershipUpdateContainer.getText().getErrorUserSearchNothingEntered()));
552       return;
553 
554     }
555     
556     Subject subject = null;
557     Group group = null;
558     String groupName = null;
559 
560     String subjectLabel = comboValue;
561 
562     try {
563 
564       grouperSession = GrouperSession.start(loggedInSubject);
565 
566       group = this.retrieveGroup(grouperSession);
567       groupName = group.getName();
568       
569       subject = GrouperUiUtils.findSubject(comboValue, true);
570       
571       final String requireGroup = simpleMembershipUpdateContainer.configValue(
572           "simpleMembershipUpdate.subjectSearchRequireGroup", false);
573       
574       final String requireSources = simpleMembershipUpdateContainer.configValue(
575           "simpleMembershipUpdate.subjectSearchRequireSources", false);
576       
577       if (!StringUtils.isBlank(requireSources)) {
578         Set<Source> sources = GrouperUtil.convertSources(requireSources);
579         String subjectSourceId = subject.getSourceId();
580         boolean containsSource = false;
581         for (Source source : sources) {
582           if (StringUtils.equals(source.getId(), subjectSourceId)) {
583             
584             containsSource = true;
585             
586           }
587         }
588         if (!containsSource) {
589           guiResponseJs.addAction(
590               
591               GuiScreenAction.newAlert(simpleMembershipUpdateContainer.getText().getErrorSubjectNotFound(subjectLabel)));
592         }
593       }
594 
595       //this would only happen if they didnt select one from the drop down...
596       if (!StringUtils.isBlank(requireGroup)) {
597 
598         final Subject SUBJECT = subject;
599         
600         boolean hasMember = (Boolean)GrouperSession.callbackGrouperSession(
601             grouperSession.internal_getRootSession(), new GrouperSessionHandler() {
602           
603           @Override
604           public Object callback(GrouperSession rootGrouperSession) throws GrouperSessionException {
605             
606             Group groupFilter = GroupFinder.findByName(rootGrouperSession, requireGroup, true);
607             return groupFilter.hasMember(SUBJECT);
608           }});
609         
610         if (!hasMember) {
611 
612           guiResponseJs.addAction(
613               
614               GuiScreenAction.newAlert(simpleMembershipUpdateContainer.getText().getErrorSubjectNotFound(subjectLabel)));
615 
616           return;
617           
618         }
619       }
620       
621       
622       subjectLabel = GrouperUiUtils.convertSubjectToLabel(subject);
623       Membership membership = MembershipFinder.findImmediateMembership(GrouperSession.staticGrouperSession(), group, subject, Group.getDefaultList(), false);
624       if (membership != null) {
625         
626         //lets clear out the combobox:
627         guiResponseJs.addAction(GuiScreenAction.newFormFieldValue("simpleMembershipUpdateAddMember", null));
628         
629         guiResponseJs.addAction(
630             
631             GuiScreenAction.newAlert(simpleMembershipUpdateContainer.getText().getWarningSubjectAlreadyMember(subjectLabel)));
632 
633         return;
634       }
635       
636       group.addMember(subject);
637       
638 
639       //lets clear out the combobox:
640       guiResponseJs.addAction(GuiScreenAction.newFormFieldValue("simpleMembershipUpdateAddMember", null));
641       
642       guiResponseJs.addAction(GuiScreenAction.newAlert(
643           simpleMembershipUpdateContainer.getText().getSuccessMemberAdded(subjectLabel)));
644       
645     } catch (NoSessionException se) {
646       throw se;
647     } catch (ControllerDone cd) {
648       throw cd;
649     } catch (SubjectNotFoundException snfe) {
650       
651       guiResponseJs.addAction(GuiScreenAction.newAlert(
652           simpleMembershipUpdateContainer.getText().getErrorSubjectNotFound(subjectLabel)));
653 
654       return;
655     } catch (SourceUnavailableException sue) {
656       guiResponseJs.addAction(GuiScreenAction.newAlert(
657           simpleMembershipUpdateContainer.getText().getErrorSourceUnavailable()));
658       
659       return;
660 
661     } catch (SubjectNotUniqueException snue) {
662       
663       guiResponseJs.addAction(GuiScreenAction.newAlert(
664           simpleMembershipUpdateContainer.getText().getErrorSubjectNotUnique(subjectLabel)));
665       return;
666 
667     } catch (Exception se) {
668       if (ExceptionUtils.getFullStackTrace(se).contains("membership cannot be circular")) {
669         LOG.warn("Error adding member to group: " + groupName + ", " + subjectLabel + ", " + se.getMessage(), se);
670         guiResponseJs.addAction(GuiScreenAction.newAlert(TagUtils.navResourceString("simpleMembershipUpdate.errorCircularReference")));
671         return;
672       } else {
673         throw new RuntimeException("Error adding member to group: " + groupName + ", " + subjectLabel + ", " + se.getMessage(), se);
674       }
675     } finally {
676       GrouperSession.stopQuietly(grouperSession); 
677     }
678 
679     retrieveMembers(httpServletRequest, httpServletResponse);
680     
681   }

752    * @param httpServletRequest
753    * @param httpServletResponse
754    */
755   @SuppressWarnings("unchecked")
756   public void deleteAll(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
757     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
758     
759     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
760 
761     GrouperSession grouperSession = null;
762 
763     Group group = null;
764     String groupName = null;
765 
766     String currentMemberUuid = null;
767     
768     try {
769 
770       grouperSession = GrouperSession.start(loggedInSubject);
771 
772       group = this.retrieveGroup(grouperSession);
773       groupName = group.getName();
774       
775       int deleteCount = 0;
776 
777       Set<Member> members = group.getImmediateMembers();
778       
779       for (Member member : members) {

781         
782         if (group.deleteMember(member, false)) {
783           deleteCount++;
784         }
785       }      
786       
787       guiResponseJs.addAction(GuiScreenAction.newAlert(GrouperUiUtils.message("simpleMembershipUpdate.successAllMembersDeleted",

789       
790     } catch (NoSessionException se) {
791       throw se;
792     } catch (ControllerDone cd) {
793       throw cd;
794     } catch (Exception se) {
795       throw new RuntimeException("Error deleting all members from group: " + groupName 
796           + ", " + currentMemberUuid + ", " + se.getMessage(), se);
797     } finally {
798       GrouperSession.stopQuietly(grouperSession); 
799     }
800 
801     retrieveMembers(httpServletRequest, httpServletResponse);
802 
803 
804   }
