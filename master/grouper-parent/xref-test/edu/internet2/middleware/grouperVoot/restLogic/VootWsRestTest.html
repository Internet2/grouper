53    /**
54     * note: running this will delete all data in the registry!
55     * /groups/test.subject.0
56     */
57    public void testMemberInOneGroup() {
58  
59      //setup data as root user
60      GrouperSession grouperSession = GrouperSession.startRootSession();
61      
62      //group where the user is a member and they can read it (only one returned)
63      Group group = new GroupSave(grouperSession).assignName("aStem:aGroup")
64          .assignDescription("some description").assignCreateParentStemsIfNotExist(true).save();
65      group.grantPriv(SubjectTestHelper.SUBJ0, AccessPrivilege.READ, false);
66      group.addMember(SubjectTestHelper.SUBJ0, false);
67  
68      //group where user isnt a member and cant read
69      group = new GroupSave(grouperSession).assignName("aStem:aGroup2")
70          .assignDescription("some description2").assignCreateParentStemsIfNotExist(true).save();
71      group.grantPriv(SubjectTestHelper.SUBJ1, AccessPrivilege.READ, false);
72      group.addMember(SubjectTestHelper.SUBJ1, false);
73  
74      //group where user is a member and cant read
75      group = new GroupSave(grouperSession).assignName("aStem:aGroup3")
76          .assignDescription("some description3").assignCreateParentStemsIfNotExist(true).save();
77      group.grantPriv(SubjectTestHelper.SUBJ0, AccessPrivilege.READ, false);
78      group.addMember(SubjectTestHelper.SUBJ1, false);
79      
80      //group where user isnt a member and can read
81      group = new GroupSave(grouperSession).assignName("aStem:aGroup4")
82          .assignDescription("some description4").assignCreateParentStemsIfNotExist(true).save();
83      group.grantPriv(SubjectTestHelper.SUBJ0, AccessPrivilege.READ, false);
84      group.addMember(SubjectTestHelper.SUBJ1, false);
85      
86      GrouperSession.stopQuietly(grouperSession);
87  
88      
89      //start session as logged in user to web service
90      grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ0);
91      
92      // URL: /groups/test.subject.0
93      List<String> urlStrings = GrouperUtil.toList("groups", "test.subject.0");
94  
95      String resource = GrouperServiceUtils.popUrlString(urlStrings);
96      VootRestHttpMethod vootRestHttpMethod = VootRestHttpMethod.valueOfIgnoreCase("GET", true);
97  
98      // validate and get the operation
99      VootWsRest vootWsRest = VootWsRest.valueOfIgnoreCase(resource, false);
100 
101     Map<String, String[]> urlParamMap = new HashMap<String, String[]>();
102     
103     //add params here
104     
105     // main business logic method
106     Object resultObject = vootWsRest.service(urlStrings, vootRestHttpMethod, urlParamMap);
107 
108     //analyze the result
109     assertNotNull(resultObject);
110     
111     assertTrue(resultObject.getClass().toString(), resultObject instanceof VootGetGroupsResponse);
112     
113     VootGetGroupsResponse vootGetGroupsResponse = (VootGetGroupsResponse)resultObject;
114     
115     assertEquals(1, vootGetGroupsResponse.getTotalResults().intValue());
116 
117     assertEquals(1, GrouperUtil.length(vootGetGroupsResponse.getEntry()));
118     
119     VootGroup vootGroup = vootGetGroupsResponse.getEntry()[0];
120     
121     assertEquals("aStem:aGroup", vootGroup.getId());
122     assertEquals("aStem:aGroup", vootGroup.getName());
123     assertEquals("some description", vootGroup.getDescription());
124     assertEquals("member", vootGroup.getVoot_membership_role());
125 
126 
127     GrouperSession.stopQuietly(grouperSession);
128     
129   }
