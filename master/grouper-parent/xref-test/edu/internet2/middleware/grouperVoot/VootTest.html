105   /**
106    * Method the realize the call to VOOT groups api.
107    * @param grouperSession the grouper session to be used for calling
108    * @param userUrl the url to call
109    * @param sortBy the field name to be used for sorting or null of no sorting.
110    * @param start the first element in the result set (0 means start from beginning).
111    * @param count the number of elements in the result set (-1 or 0 means find all).
112    * @return the object obtained from VOOT call
113    */
114   protected Object callGroupsAPI(String userUrl, String search, String sortBy, int start, int count) {
115     // URL: /groups/test.subject.0
116     List<String> urlStrings = (userUrl != null) ? GrouperUtil.toList("groups", userUrl) : GrouperUtil.toList("groups");
117 
118     String resource = GrouperServiceUtils.popUrlString(urlStrings);
119     VootRestHttpMethod vootRestHttpMethod = VootRestHttpMethod.valueOfIgnoreCase("GET", true);
120 
121     // validate and get the operation
122     VootWsRest vootWsRest = VootWsRest.valueOfIgnoreCase(resource, false);
123 
124     Map<String, String[]> urlParamMap = new HashMap<String, String[]>();
125     
126     //add params here
127     if (search != null) urlParamMap.put("search", new String[]{ search });
128     if (sortBy != null) urlParamMap.put("sortBy", new String[]{ sortBy });
129     if (start != 0) urlParamMap.put("startIndex", new String[]{ Integer.toString(start) });
130     if (count != -1) urlParamMap.put("count", new String[]{ Integer.toString(count) });
131     
132     // main business logic method
133     return vootWsRest.service(urlStrings, vootRestHttpMethod, urlParamMap);
134   }

175   /**
176    * Method to validate groups result from VOOT call.
177    * @param resultObject the object returned from VOOT call.
178    * @param count the number of groups expected to be in the output object.
179    * @param groups the groups id (in this class definition) to be searched in the result.
180    */
181   protected void validateGroups(Object resultObject, int total, String orderBy, int start, int count, int[] groups, String[] roles) {
182     assertNotNull(resultObject);
183     assertTrue(resultObject.getClass().toString(), resultObject instanceof VootGetGroupsResponse);
184     
185     VootGetGroupsResponse vootGetGroupsResponse = (VootGetGroupsResponse) resultObject;
186     assertEquals(start, vootGetGroupsResponse.getStartIndex().intValue());
187     assertEquals(count, vootGetGroupsResponse.getItemsPerPage().intValue());
188     assertEquals(total, vootGetGroupsResponse.getTotalResults().intValue());
189     assertEquals(count, GrouperUtil.length(vootGetGroupsResponse.getEntry()));
190     
191     if (groups == null) {
192       assertNull(vootGetGroupsResponse.getEntry());
193     }
194     else {
195       assertNotNull(vootGetGroupsResponse.getEntry());
196       
197       // Check if all groups are present in the response, without considering their order
198       int groupCount = 0;
199       @SuppressWarnings("unchecked")
200       List<VootGroup> groupsList = Arrays.asList(vootGetGroupsResponse.getEntry());
201       for (int i = 0; i < total; ++i) {
202         VootGroup vootGroup = new VootGroup();
203         vootGroup.setId(GROUP_NAMES[groups[i]]);
204         vootGroup.setName(GROUP_NAMES[groups[i]]);
205         vootGroup.setDescription(GROUP_DESCRIPTIONS[groups[i]]);
206         vootGroup.setVoot_membership_role(roles[i]);
207         
208         if (groupsList.contains(vootGroup)) groupCount++;
209       }
210       
211       assertEquals(count, groupCount);
212       
213       // Check if all groups are present in the response, also considering their order
214       if (orderBy != null) {
215         try {
216           // Set up introspection for the field specified by orderBy
217           Field f = VootGroup.class.getDeclaredField(orderBy);
218           f.setAccessible(true);
219           
220           // Verify sorting
221           for (int i = 0; i < total-1; ++i) {
222             String value1 = f.get(vootGetGroupsResponse.getEntry()[i]).toString();
223             String value2 = f.get(vootGetGroupsResponse.getEntry()[i+1]).toString();
224             
225             assertTrue(value1.compareTo(value2) <= 0);
226           }
227         }
228         catch (Exception e) {
229           assertTrue("Exception in checking sort", false);
230         }
231       }
232     }
233   }

235   /**
236    * Method to validate subject result from VOOT call.
237    * @param resultObject the object returned from VOOT call.
238    * @param subjects the subjects expected to be in the output.
239    * @param roles the roles of the subjects in the output.
240    */
241   protected void validateMembers(Object resultObject, int total, String orderBy, int start, int count, Subject[] subjects, String[] roles) {
242     assertNotNull(resultObject);
243     assertTrue(resultObject.getClass().toString(), resultObject instanceof VootGetMembersResponse);
244     
245     int subjectCount = (subjects == null) ? 0 : subjects.length;
246     VootGetMembersResponse vootGetMembersResponse = (VootGetMembersResponse) resultObject;
247     assertNotNull(vootGetMembersResponse.getEntry());
248     
249     // Some group may contain GrouperSystem between its members.
250     // In case just ignore this user from the validation.
251     boolean isGrouperSystemPresent = false;
252     for (int i = 0; i < vootGetMembersResponse.getEntry().length; ++i) {
253       if ("GrouperSystem".equals(vootGetMembersResponse.getEntry()[i].getId())) {
254         isGrouperSystemPresent = true;
255       }
256     }
257     
258     if (isGrouperSystemPresent) {
259       assertEquals(start, vootGetMembersResponse.getStartIndex().intValue());
260       assertEquals(count, vootGetMembersResponse.getItemsPerPage().intValue());
261       assertEquals(subjectCount+1, vootGetMembersResponse.getTotalResults().intValue());
262       assertEquals(count+1, GrouperUtil.length(vootGetMembersResponse.getEntry()));
263     }
264     else {
265       assertEquals(start, vootGetMembersResponse.getStartIndex().intValue());
266       assertEquals(count, vootGetMembersResponse.getItemsPerPage().intValue());
267       assertEquals(subjectCount, vootGetMembersResponse.getTotalResults().intValue());
268       assertEquals(count, GrouperUtil.length(vootGetMembersResponse.getEntry()));
269     }
270     
271     if (subjects != null) {
272       // Check if all groups are present in the response, without considering their order
273       int memberCount = 0;
274       @SuppressWarnings("unchecked")
275       List<VootGroup> membersList = Arrays.asList(vootGetMembersResponse.getEntry());
276       for (int i = 0; i < subjectCount; ++i) {
277         VootPerson vootPerson = new VootPerson(subjects[i]);
278         vootPerson.setVoot_membership_role(roles[i]);
279         
280         if (membersList.contains(vootPerson)) memberCount++;
281       }
282       
283       assertEquals(count, memberCount);
284       
285       // Check if all groups are present in the response, also considering their order
286       if (orderBy != null) {
287         try {
288           // Set up introspection for the field specified by orderBy
289           Field f = VootPerson.class.getDeclaredField(orderBy);
290           f.setAccessible(true);
291           
292           // Verify sorting
293           for (int i = 0; i < total-1; ++i) {
294             String value1 = f.get(vootGetMembersResponse.getEntry()[i]).toString();
295             String value2 = f.get(vootGetMembersResponse.getEntry()[i+1]).toString();
296             
297             assertTrue(value1.compareTo(value2) <= 0);
298           }
299         }
300         catch (Exception e) {
301           assertTrue("Exception in checking sort", false);
302         }
303       }
304     }
305   }
