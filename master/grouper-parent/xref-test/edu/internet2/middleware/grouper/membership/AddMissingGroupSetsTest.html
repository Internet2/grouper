100     GrouperSession session = GrouperSession.startRootSession();

115     Group composite1Right = top.addChildGroup("composite1Right", "composite1Right");

117     Group composite1Subj = top.addChildGroup("composite1Subj", "composite1Subj");

119     Group updateFieldTest = top.addChildGroup("updateFieldTest", "updateFieldTest");

128     Group six = top.addChildGroup("six", "six");

130     composite1Owner.addCompositeMember(CompositeType.UNION, composite1Left, composite1Right);
131     composite2Owner.addCompositeMember(CompositeType.COMPLEMENT, composite2Left, composite2Right);
132     composite1Left.addMember(composite1Subj.toSubject());
133     
134     one.addMember(two.toSubject());
135     two.addMember(three.toSubject());
136     three.addMember(four.toSubject());

139

141     updateFieldTest.grantPriv(four.toSubject(), AccessPrivilege.UPDATE);
142     top.grantPriv(four.toSubject(), NamingPrivilege.CREATE);
143 
144     // get number of groupSets before proceeding..
145     Set<GroupSet> originalGroupSets = GrouperDAOFactory.getFactory().getGroupSet().findAllByCreator(rootMember);
146     int size = originalGroupSets.size();
147 
148     // delete all groupSets
149     reset();
150     Set<GroupSet> currentGroupSets = GrouperDAOFactory.getFactory().getGroupSet().findAllByCreator(rootMember);
151     T.amount("Total groupSets", 0, currentGroupSets.size());
152

155     currentGroupSets = GrouperDAOFactory.getFactory().getGroupSet().findAllByCreator(rootMember);
156     T.amount("Total groupSets", size, currentGroupSets.size());
157     Set<GroupSet> currentGroupSetsAfterMod = convertCurrentGroupSets(originalGroupSets, currentGroupSets);
158     Set<GroupSet> checkGroupSets = new HashSet<GroupSet>(originalGroupSets);
159     checkGroupSets.removeAll(currentGroupSetsAfterMod);
160     T.amount("Check groupSet difference between original and current", 0, checkGroupSets.size());
161     
162     // now let's remove some group sets
163     GroupSet gs1 = GrouperDAOFactory.getFactory().getGroupSet().findSelfGroup(composite1Owner.getUuid(), Group.getDefaultList().getUuid());
164     GroupSet gs2 = GrouperDAOFactory.getFactory().getGroupSet().findSelfGroup(customFieldTest.getUuid(), FieldFinder.find("testList", true).getUuid());
165     GroupSet gs3 = GrouperDAOFactory.getFactory().getGroupSet().findSelfStem(top.getUuid(), FieldFinder.find(Field.FIELD_NAME_STEMMERS, true).getUuid());
166     GroupSet gs4 = GrouperDAOFactory.getFactory().getGroupSet().findSelfStem(top.getUuid(), FieldFinder.find(Field.FIELD_NAME_CREATORS, true).getUuid());
167     GrouperDAOFactory.getFactory().getGroupSet().delete(GrouperDAOFactory.getFactory().getGroupSet().findAllChildren(gs1));
168     GrouperDAOFactory.getFactory().getGroupSet().delete(GrouperDAOFactory.getFactory().getGroupSet().findAllChildren(gs2));
169     GrouperDAOFactory.getFactory().getGroupSet().delete(GrouperDAOFactory.getFactory().getGroupSet().findAllChildren(gs3));
170     GrouperDAOFactory.getFactory().getGroupSet().delete(GrouperDAOFactory.getFactory().getGroupSet().findAllChildren(gs4));
171     GrouperDAOFactory.getFactory().getGroupSet().deleteSelfByOwnerGroupAndField(composite1Owner.getUuid(), Group.getDefaultList().getUuid());
172     GrouperDAOFactory.getFactory().getGroupSet().deleteSelfByOwnerGroupAndField(customFieldTest.getUuid(), FieldFinder.find("testList", true).getUuid());
173     GrouperDAOFactory.getFactory().getGroupSet().deleteSelfByOwnerStem(top.getUuid());
174 
175     // now add them back and verify again
176     new AddMissingGroupSets().showResults(false).addAllMissingGroupSets();
177     currentGroupSets = GrouperDAOFactory.getFactory().getGroupSet().findAllByCreator(rootMember);
178     T.amount("Total groupSets", size, currentGroupSets.size());
179     currentGroupSetsAfterMod = convertCurrentGroupSets(originalGroupSets, currentGroupSets);
180     checkGroupSets = new HashSet<GroupSet>(originalGroupSets);
181     checkGroupSets.removeAll(currentGroupSetsAfterMod);
182     T.amount("Check groupSet difference between original and current", 0, checkGroupSets.size());
183   }

185   private Set<GroupSet> convertCurrentGroupSets(Set<GroupSet> originalGroupSets, Set<GroupSet> currentGroupSets) {
186     // This is a bit of a hack, but here's what I'm trying to do...
187     // In order to verify that the currentGroupSets equal the originalGroupSets, we need to update the parentIds.
188     // Based on the groupSets that were added, I'm using the following as a key for lookups.  In reality, this isn't a unique constraint.
189     //      owner_group_id_null, owner_stem_id_null, member_id, field_id
190     
191     Set<GroupSet> currentGroupSetsAfterMod = new HashSet<GroupSet>();
192     
193     Map<String, String> lookupOrigIdByKey = new HashMap<String, String>();
194     Map<String, String> lookupOrigIdByCurrentId = new HashMap<String, String>();
195     
196     Iterator<GroupSet> originalGroupSetsIter = originalGroupSets.iterator();
197     while (originalGroupSetsIter.hasNext()) {
198       GroupSet gs = originalGroupSetsIter.next();
199       String key = gs.getOwnerGroupIdNull() + ":" + gs.getOwnerStemIdNull() + ":" + gs.getMemberId() + ":" + 
200           gs.getFieldId();
201       lookupOrigIdByKey.put(key, gs.getId());
202     }
203     
204     Iterator<GroupSet> currentGroupSetsIter = currentGroupSets.iterator();
205     while (currentGroupSetsIter.hasNext()) {
206       GroupSet gs = currentGroupSetsIter.next();
207       String key = gs.getOwnerGroupIdNull() + ":" + gs.getOwnerStemIdNull() + ":" + gs.getMemberId() + ":" + 
208           gs.getFieldId();
209       lookupOrigIdByCurrentId.put(gs.getId(), lookupOrigIdByKey.get(key));
210     }
211     
212     currentGroupSetsIter = currentGroupSets.iterator();
213     while (currentGroupSetsIter.hasNext()) {
214       GroupSet gs = currentGroupSetsIter.next();
215       gs.setParentId(lookupOrigIdByCurrentId.get(gs.getParentId()));
216       gs.setId(lookupOrigIdByCurrentId.get(gs.getId()));
217       currentGroupSetsAfterMod.add(gs);
218     }
219     
220     return currentGroupSetsAfterMod;
221   }

223   private void reset() {
224     HibernateSession.callbackHibernateSession(
225         GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, AuditControl.WILL_NOT_AUDIT,
226         new HibernateHandler() {
227 
228           public Object callback(HibernateHandlerBean hibernateHandlerBean)
229               throws GrouperDAOException {
230             HibernateSession hibernateSession = hibernateHandlerBean
231                 .getHibernateSession();
232 
233             // this will delete everything except the self group sets for the root stem
234             Hib3GroupSetDAO.reset(hibernateSession);
235             
236             // set parent id to null first to avoid mysql bug -- http://bugs.mysql.com/bug.php?id=15746
237             HibernateSession.byHqlStatic().createQuery("update GroupSet set parentId = null").executeUpdate();
238             HibernateSession.byHqlStatic().createQuery("delete from GroupSet").executeUpdate();
239             
240             return null;
241           }
242 
243         });
244
