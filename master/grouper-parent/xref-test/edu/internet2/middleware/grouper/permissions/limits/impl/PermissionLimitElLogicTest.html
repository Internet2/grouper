195   public void testAmount() {
196     
197     //
198     //User subj0 is assigned Role<Admin>
199     this.adminRole.addMember(this.subj0, true);
200   
201     //
202     //User subj0 is assigned permission Deny, Action<Read>, Resource<Arts and sciences>, in the context of Role<Admin>
203     AttributeAssignResult attributeAssignResult = this.adminRole.getPermissionRoleDelegate().assignSubjectRolePermission(
204         this.readString, this.artsAndSciences, this.subj0, PermissionAllowed.ALLOWED);
205   
206     
207     //Result:
208     //
209     //subj0 can read except for the limit
210     
211     //lets get all of the permission assignments
212     List<PermissionEntry> permissionEntries = new ArrayList<PermissionEntry>(
213         new PermissionFinder().addSubject(this.subj0)
214         .addAction(this.readString).addPermissionName(this.english)
215         .assignPermissionProcessor(PermissionProcessor.PROCESS_LIMITS)
216         .addLimitEnvVar("(int)amount", "49000").findPermissions());
217         
218     //there should be two, one should be allow, the other deny
219     assertEquals(1, GrouperUtil.length(permissionEntries));
220     assertFalse(permissionEntries.get(0).isDisallowed());
221     assertTrue(permissionEntries.get(0).isAllowedOverall());
222   
223     AttributeAssign attributeAssign = attributeAssignResult.getAttributeAssign();
224     
225     attributeAssign.getAttributeValueDelegate().assignValue(PermissionLimitUtils.limitElAttributeDefName().getName(), "amount < 50000");
226     
227     permissionEntries = new ArrayList<PermissionEntry>(new PermissionFinder().addSubject(this.subj0)
228         .addAction(this.readString).addPermissionName(this.english)
229         .assignPermissionProcessor(PermissionProcessor.PROCESS_LIMITS)
230         .addLimitEnvVar("(int)amount", "49000").findPermissions());
231         
232     //there should be two, one should be allow, the other deny
233     assertEquals(1, GrouperUtil.length(permissionEntries));
234     assertFalse(permissionEntries.get(0).isDisallowed());
235     assertTrue(permissionEntries.get(0).isAllowedOverall());
236   
237     permissionEntries = new ArrayList<PermissionEntry>(new PermissionFinder().addSubject(this.subj0)
238         .addAction(this.readString).addPermissionName(this.english)
239         .assignPermissionProcessor(PermissionProcessor.PROCESS_LIMITS)
240         .addLimitEnvVar("(int)amount", "51000").findPermissions());
241         
242     //there should be two, one should be allow, the other deny
243     assertEquals(1, GrouperUtil.length(permissionEntries));
244     assertFalse(permissionEntries.get(0).isDisallowed());
245     assertFalse(permissionEntries.get(0).isAllowedOverall());
246 
247     attributeAssign.getAttributeValueDelegate().assignValue(PermissionLimitUtils.limitElAttributeDefName().getName(), "amount < 50000 && amount2 < 23");
248 
249     //what if you check without the variable, should be an error
250     try {
251       new PermissionFinder().addSubject(this.subj0).addAction(this.readString).addPermissionName(this.english)
252         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS).hasPermission();
253       fail("Shouldnt get here");
254     } catch (ExpressionLanguageMissingVariableException e) {
255       assertEquals("variable 'amount' is not defined in script: 'amount < 50000 && amount2 < 23'", e.getMessage());
256       //good
257     }
258   }

264     
265     //
266     //User subj0 is assigned Role<Admin>

273

278     
279     //there should be two, one should be allow, the other deny
280     assertTrue(new PermissionFinder().addSubject(this.subj0)
281         .addAction(this.readString).addPermissionName(this.english)
282         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS)
283         .addLimitEnvVar("(int)hourOfDay", "10").hasPermission());
284     assertTrue(new PermissionFinder().addSubject(this.subj0)
285         .addAction(this.readString).addPermissionName(this.english)
286         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS)
287         .addLimitEnvVar("(int)hourOfDay", "9").hasPermission());
288   
289     assertFalse(new PermissionFinder().addSubject(this.subj0)
290         .addAction(this.readString).addPermissionName(this.english)
291         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS)
292         .addLimitEnvVar("(int)hourOfDay", "8").hasPermission());
293   
294     assertTrue(new PermissionFinder().addSubject(this.subj0)
295         .addAction(this.readString).addPermissionName(this.english)
296         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS)
297         .addLimitEnvVar("(int)hourOfDay", "16").hasPermission());
298     assertTrue(new PermissionFinder().addSubject(this.subj0)
299         .addAction(this.readString).addPermissionName(this.english)
300         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS)
301         .addLimitEnvVar("(int)hourOfDay", "17").hasPermission());
302     assertFalse(new PermissionFinder().addSubject(this.subj0)
303         .addAction(this.readString).addPermissionName(this.english)
304         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS)
305         .addLimitEnvVar("(int)hourOfDay", "18").hasPermission());
306     
307     //this should not throw an exception since hourOfDay is built in
308     Calendar calendar = new GregorianCalendar();
309     int hourOfDay = calendar.get(Calendar.HOUR_OF_DAY);
310     assertEquals(hourOfDay >= 9 && hourOfDay <= 17, new PermissionFinder().addSubject(this.subj0)
311         .addAction(this.readString).addPermissionName(this.english)
312         .assignPermissionProcessor(PermissionProcessor.FILTER_REDUNDANT_PERMISSIONS_AND_ROLES_AND_PROCESS_LIMITS)
313         .hasPermission());
