124   public void testGrantStemToGroup() {
125     try {
126       // Setup
127       GrouperSession  s     = GrouperSession.startRootSession();
128       Stem            root  = StemFinder.findRootStem(s);
129       Stem            edu   = root.addChildStem("edu", "educational");
130       Group           i2    = edu.addChildGroup("i2", "internet2");
131       Subject         subj0 = SubjectFinder.findById("test.subject.0", true);
132       i2.addMember(subj0);
133       Assert.assertTrue("i2 has mem subj0", i2.hasMember(subj0));
134       Assert.assertTrue("i2 has imm mem subj0", i2.hasImmediateMember(subj0));
135       // Test
136       Stem            ns    = StemFinder.findByName(s, edu.getName(), true);
137       Assert.assertNotNull("ns !null", ns);
138       Group           g     = GroupFinder.findByName(s, i2.getName(), true);
139       Assert.assertNotNull("g !null", g);
140   
141       // Without the pre-granting of CREATE, the later granting of STEM
142       // is fine.
143       ns.grantPriv(
144         SubjectFinder.findById(g.getUuid(), true),
145          Privilege.getInstance("create")
146        );
147       Assert.assertTrue("g (ns) has CREATE", g.toMember().hasCreate(ns));
148       Assert.assertTrue("g (m) has CREATE",  ns.hasCreate(g.toSubject()));
149   
150       ns.grantPriv(
151         SubjectFinder.findById(g.getUuid(), true),
152         Privilege.getInstance("stem")
153       );
154       Assert.assertTrue("g (ns) has STEM", g.toMember().hasStem(ns));
155       Assert.assertTrue("g (m) has STEM",  ns.hasStem(g.toSubject()));
156   
157       s.stop();
158     }
159     catch (Exception e) {
160       Assert.fail("exception: " + e.getMessage());
161     }
162   } // public void testGrantStemToGroup()

164   // @source  Gary Brown, 20051216, <39C7E27B4A5BE4494B39A93F@cse-gwb.cse.bris.ac.uk>>
165   // @status  fixed
166   public void testHasAccessPrivButCannotReadPriv() 
167     throws  GroupAddException,
168             GroupNotFoundException,
169             InsufficientPrivilegeException,
170             MemberNotFoundException,
171             SessionException,
172             StemAddException
173   {
174 
175     Subject         subj  = SubjectTestHelper.SUBJ0;
176     GrouperSession  nrs   = GrouperSession.start(subj);
177     Member          m     = MemberFinder.findBySubject(nrs, subj, true);
178 
179     int baseLineViewSize = m.hasView().size();
180     int baseLineReadSize = m.hasRead().size();
181     
182     GrouperConfig.retrieveConfig().propertiesOverrideMap().put("groups.create.grant.all.read", "true");
183     GrouperConfig.retrieveConfig().propertiesOverrideMap().put("groups.create.grant.all.view", "true");
184     
185     LOG.info("testHasAccessPrivButCannotReadPriv");
186     // Setup
187     Subject         all   = SubjectFinder.findAllSubject();
188     GrouperSession  s     = SessionHelper.getRootSession();
189     Stem            root  = StemFinder.findRootStem(s);
190     Stem            edu   = root.addChildStem("edu", "edu");
191     Group           a     = edu.addChildGroup("a", "a");
192     Group           b     = edu.addChildGroup("b", "b");
193     Group           c     = edu.addChildGroup("c", "c");
194     edu.addChildGroup("d", "d");
195     PrivHelper.grantPriv(s, a, all, AccessPrivilege.OPTIN);
196     PrivHelper.grantPriv(s, b, all, AccessPrivilege.OPTOUT);
197     s.stop();
198   
199     // Test
200     nrs   = GrouperSession.start(subj);
201     m     = MemberFinder.findBySubject(nrs, subj, true);
202     Group           A     = GroupFinder.findByName(nrs, "edu:a", true);
203     Group           B     = GroupFinder.findByName(nrs, "edu:b", true);
204     Group           C     = GroupFinder.findByName(nrs, "edu:c", true);
205     Group           D     = GroupFinder.findByName(nrs, "edu:d", true);
206   
207     Set view    = m.hasView();
208     
209     assertEquals(
210       "VIEW/4 ("    + view.size()   + ")", 4+baseLineViewSize, view.size() 
211     );
212     Assert.assertTrue("VIEW/A/HAS",   A.hasView(subj)   );
213     Assert.assertTrue("VIEW/B/HAS",   B.hasView(subj)   );
214     Assert.assertTrue("VIEW/C/HAS",   C.hasView(subj)   );
215     Assert.assertTrue("VIEW/D/HAS",   D.hasView(subj)   );
216     Assert.assertTrue("VIEW/A/IS" ,   m.hasView(A)      );
217     Assert.assertTrue("VIEW/B/IS" ,   m.hasView(B)      );
218     Assert.assertTrue("VIEW/C/IS" ,   m.hasView(C)      );
219     Assert.assertTrue("VIEW/D/IS" ,   m.hasView(D)      );
220   
221     Set read    = m.hasRead();
222     Assert.assertEquals(
223       "READ/4 ("    + read.size()   + ")", 4 + baseLineReadSize, read.size()
224     );
225     Assert.assertTrue("READ/A/HAS",   A.hasRead(subj)   );
226     Assert.assertTrue("READ/B/HAS",   B.hasRead(subj)   );
227     Assert.assertTrue("READ/C/HAS",   C.hasRead(subj)   );
228     Assert.assertTrue("READ/D/HAS",   D.hasRead(subj)   );
229     Assert.assertTrue("READ/A/IS" ,   m.hasRead(A)      );
230     Assert.assertTrue("READ/B/IS" ,   m.hasRead(B)      );
231     Assert.assertTrue("READ/C/IS" ,   m.hasRead(C)      );
232     Assert.assertTrue("READ/D/IS" ,   m.hasRead(D)      );
233   
234     Set optin   = m.hasOptin();
235     Assert.assertTrue(
236       "OPTIN/1 ("   + optin.size()  + ")", optin.size()   == 1
237     );
238     Assert.assertTrue("OPTIN/A/HAS",  A.hasOptin(subj)  );
239     Assert.assertTrue("OPTIN/B/HAS",  !B.hasOptin(subj) );
240     Assert.assertTrue("OPTIN/C/HAS",  !C.hasOptin(subj) );
241     Assert.assertTrue("OPTIN/D/HAS",  !D.hasOptin(subj) );
242     Assert.assertTrue("OPTIN/A/IS" ,  m.hasOptin(A)     );
243     Assert.assertTrue("OPTIN/B/IS" ,  !m.hasOptin(B)    );
244     Assert.assertTrue("OPTIN/C/IS" ,  !m.hasOptin(C)    );
245     Assert.assertTrue("OPTIN/D/IS" ,  !m.hasOptin(D)    );
246   
247     Set optout  = m.hasOptout();
248     Assert.assertTrue(
249       "OPTOUT/1 ("  + optout.size() + ")", optout.size()  == 1
250     );
251     Assert.assertTrue("OPTOUT/A/HAS", !A.hasOptout(subj));
252     Assert.assertTrue("OPTOUT/B/HAS", B.hasOptout(subj) );
253     Assert.assertTrue("OPTOUT/C/HAS", !C.hasOptout(subj));
254     Assert.assertTrue("OPTOUT/D/HAS", !D.hasOptout(subj));
255     Assert.assertTrue("OPTOUT/A/IS" , !m.hasOptout(A)   );
256     Assert.assertTrue("OPTOUT/B/IS" , m.hasOptout(B)    );
257     Assert.assertTrue("OPTOUT/C/IS" , !m.hasOptout(C)   );
258     Assert.assertTrue("OPTOUT/D/IS" , !m.hasOptout(D)   );
259   
260     Set update  = m.hasUpdate();
261     Assert.assertTrue(
262       "UPDATE/1 ("  + update.size() + ")", update.size()  == 0
263     );
264     Assert.assertTrue("UPDATE/A/HAS", !A.hasUpdate(subj));
265     Assert.assertTrue("UPDATE/B/HAS", !B.hasUpdate(subj));
266     Assert.assertTrue("UPDATE/C/HAS", !C.hasUpdate(subj) );
267     Assert.assertTrue("UPDATE/D/HAS", !D.hasUpdate(subj));
268     Assert.assertTrue("UPDATE/A/IS" , !m.hasUpdate(A)   );
269     Assert.assertTrue("UPDATE/B/IS" , !m.hasUpdate(B)   );
270     Assert.assertTrue("UPDATE/C/IS" , !m.hasUpdate(C)    );
271     Assert.assertTrue("UPDATE/D/IS" , !m.hasUpdate(D)   );
272   
273     Set admin   = m.hasAdmin();
274     Assert.assertTrue(
275       "ADMIN/0 ("   + admin.size()  + ")", admin.size()   == 0
276     );
277     Assert.assertTrue("ADMIN/A/HAS",  !A.hasAdmin(subj) );
278     Assert.assertTrue("ADMIN/B/HAS",  !B.hasAdmin(subj) );
279     Assert.assertTrue("ADMIN/C/HAS",  !C.hasAdmin(subj) );
280     Assert.assertTrue("ADMIN/D/HAS",  !D.hasAdmin(subj) );
281     Assert.assertTrue("ADMIN/A/IS" ,  !m.hasAdmin(A)    );
282     Assert.assertTrue("ADMIN/B/IS" ,  !m.hasAdmin(B)    );
283     Assert.assertTrue("ADMIN/C/IS" ,  !m.hasAdmin(C)    );
284     Assert.assertTrue("ADMIN/D/IS" ,  !m.hasAdmin(D)    );
285   
286     nrs.stop();
287   
288   }

290   // @source  Gary Brown, 20051216, <39C7E27B4A5BE4494B39A93F@cse-gwb.cse.bris.ac.uk>>
291   // @status  fixed
292   public void testHasNamingPrivButCannotReadPriv() 
293     throws  InsufficientPrivilegeException,
294             MemberNotFoundException,
295             SessionException,
296             StemAddException,
297             StemNotFoundException
298   {
299     LOG.info("testHasNamingPrivButCannotReadPriv");
300   
301     // Setup
302     Subject         all   = SubjectFinder.findAllSubject();
303     GrouperSession  s     = SessionHelper.getRootSession();
304     Stem            root  = StemFinder.findRootStem(s);
305     Stem            edu   = root.addChildStem("edu", "edu");
306     Stem            a     = edu.addChildStem("a", "a");
307     Stem            b     = edu.addChildStem("b", "b");
308     edu.addChildStem("c", "c");
309     PrivHelper.grantPriv(s, a, all, NamingPrivilege.CREATE);
310     PrivHelper.grantPriv(s, b, all, NamingPrivilege.STEM);
311     s.stop();
312   
313     // Test
314     Subject         subj  = SubjectTestHelper.SUBJ0;
315     GrouperSession  nrs   = GrouperSession.start(subj);
316     Member          m     = MemberFinder.findBySubject(nrs, subj, true);
317     Stem            A     = StemFinder.findByName(nrs, "edu:a", true);
318     Stem            B     = StemFinder.findByName(nrs, "edu:b", true);
319     Stem            C     = StemFinder.findByName(nrs, "edu:c", true);
320   
321     Set create  = m.hasCreate();
322     T.amount( "CREATE", 1, create.size() );
323     Assert.assertTrue("CREATE/A/HAS", A.hasCreate(subj) );
324     Assert.assertTrue("CREATE/B/HAS", !B.hasCreate(subj));
325     Assert.assertTrue("CREATE/C/HAS", !C.hasCreate(subj));
326     Assert.assertTrue("CREATE/A/IS" , m.hasCreate(A)    );
327     Assert.assertTrue("CREATE/B/IS" , !m.hasCreate(B)   );
328     Assert.assertTrue("CREATE/C/IS" , !m.hasCreate(C)   );
329   
330     Set stem    = m.hasStem();
331     T.amount( "STEM", 1, stem.size() );
332     Assert.assertTrue("STEM/A/HAS"  , !A.hasStem(subj)  );
333     Assert.assertTrue("STEM/B/HAS"  , B.hasStem(subj)   );
334     Assert.assertTrue("STEM/C/HAS"  , !C.hasStem(subj)  );
335     Assert.assertTrue("STEM/A/IS"   , !m.hasStem(A)     );
336     Assert.assertTrue("STEM/B/IS"   , m.hasStem(B)      );
337     Assert.assertTrue("STEM/C/IS"   , !m.hasStem(C)     );
338   
339     nrs.stop();
340   }
