358   /**
359    * make sure the caching flag propagates appropriately
360    */
361   public void testCachingPropagate() {
362     
363     HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, 
364         AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
365       
366       public Object callback(HibernateHandlerBean hibernateHandlerBean)
367           throws GrouperDAOException {
368         
369         HibernateSession hibernateSession1 = hibernateHandlerBean.getHibernateSession();
370         
371         assertTrue("Should default to true", hibernateSession1.isCachingEnabled());
372 
373         //do an inner callback, see that it is false, set to true
374         HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_OR_USE_EXISTING,
375             AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
376               
377               public Object callback(HibernateHandlerBean hibernateHandlerBean)
378                   throws GrouperDAOException {
379                 
380                 HibernateSession hibernateSession2 = hibernateHandlerBean.getHibernateSession();
381                 assertTrue(hibernateSession2.isCachingEnabled());
382                 hibernateSession2.setCachingEnabled(false);
383                 
384                 //see that it propagates to inner
385                 HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_NEW,
386                     AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
387                       
388                       public Object callback(HibernateHandlerBean hibernateHandlerBean)
389                           throws GrouperDAOException {
390                         HibernateSession hibernateSession3 = hibernateHandlerBean.getHibernateSession();
391                         
392                         assertFalse("should propagate from parent", hibernateSession3.isCachingEnabled());
393 
394                         //set to false should not affect parent
395                         hibernateSession3.setCachingEnabled(true);
396 
397                         return null;
398                       }
399                     });
400                 
401                 assertFalse("child setting should not affect this", hibernateSession2.isCachingEnabled());
402                 return null;
403               }
404             });
405         
406         assertTrue("make sure the inner doesnt affect this", hibernateSession1.isCachingEnabled());
407         return null;
408       }
409     });
410     
411   }

493     Group child = StemHelper.addChildGroup(edu, "child", "child");
494     
495     //lets count the memberships (all view)
496     int initialCountMembershipsAllV = HibernateSession.bySqlStatic().select(
497         int.class, "select count(1) from grouper_memberships_all_v");
498 
499     int initialCountGroupSet = HibernateSession.bySqlStatic().select(
500         int.class, "select count(1) from grouper_group_set");
501 
502     //##################################################
503     parent.addMember(child.toSubject());
504     child.addMember(SubjectTestHelper.SUBJ1);
505     
506     int currentGroupCountMembershipsAllV = HibernateSession.bySqlStatic().select(
507         int.class, "select count(1) from grouper_memberships_all_v");
508 
509     int currentGroupCountGroupSet = HibernateSession.bySqlStatic().select(
510         int.class, "select count(1) from grouper_group_set");
511     
512     assertEquals("3 memberships, 1 for subj1 in child, 1 for subj1 in parent, and 1 for child in parent", 
513         initialCountMembershipsAllV + 3, currentGroupCountMembershipsAllV);
514     assertEquals("1 more group set for child in parent", 
515         initialCountGroupSet + 1, currentGroupCountGroupSet);
516     
517     //####################################################
518     //disable
519     Membership membership = parent.getImmediateMembership(Group.getDefaultList(), child.toSubject(), true, true);
520     membership.setDisabledTime(new Timestamp(System.currentTimeMillis() - 10000));
521     membership.update();
522     
523     currentGroupCountMembershipsAllV = HibernateSession.bySqlStatic().select(
524         int.class, "select count(1) from grouper_memberships_all_v");
525 
526     currentGroupCountGroupSet = HibernateSession.bySqlStatic().select(
527         int.class, "select count(1) from grouper_group_set");
528     
529     assertEquals("2 memberships, 1 for subj1 in child, 1 for disabled child in parent, none for subj1 in parent", 
530         initialCountMembershipsAllV + 2, currentGroupCountMembershipsAllV);
531     assertEquals("same as original group sets", 
532         initialCountGroupSet, currentGroupCountGroupSet);
533   }
