194   /**
195    * retrieve example audit entry from db for testing
196    * @param id 
197    * @param exceptionIfNull 
198    * @return an example audit entry
199    */
200   public static AuditEntry exampleRetrieveAuditEntryDb(String id, boolean exceptionIfNull) {
201     AuditType auditType = AuditTypeFinder.find(AuditTypeBuiltin.GROUP_ADD.getAuditCategory(), 
202         AuditTypeBuiltin.GROUP_ADD.getActionName(), true);
203     AuditEntry auditEntry = HibernateSession.byHqlStatic().createQuery("from AuditEntry as theAuditEntry " +
204     		"where theAuditEntry.auditTypeId = :theAuditTypeId and theAuditEntry.string01 = :theId ")
205     		.setString("theAuditTypeId", auditType.getId())
206         .setString("theId", id)
207     		.uniqueResult(AuditEntry.class);
208     if (auditEntry == null && exceptionIfNull) {
209       throw new RuntimeException("Cant find audit entry");
210     }
211     return auditEntry;
212   }

218   public void testXmlInsert() {
219     
220     GrouperSession.startRootSession();
221     
222     AuditEntry auditEntryOriginal = exampleAuditEntryDb("exampleInsert");
223     
224     //do this because last membership update isnt there, only in db
225     auditEntryOriginal = exampleRetrieveAuditEntryDb("exampleInsert", true);
226     AuditEntry auditEntryCopy = exampleRetrieveAuditEntryDb("exampleInsert", true);
227     AuditEntry auditEntryCopy2 = exampleRetrieveAuditEntryDb("exampleInsert", true);
228     HibernateSession.byObjectStatic().delete(auditEntryCopy);
229     
230     //lets insert the original
231     auditEntryCopy2.xmlSaveBusinessProperties(null);
232     auditEntryCopy2.xmlSaveUpdateProperties();
233 
234     //refresh from DB
235     auditEntryCopy = exampleRetrieveAuditEntryDb("exampleInsert", true);
236     
237     assertFalse(auditEntryCopy == auditEntryOriginal);
238     assertFalse(auditEntryCopy.xmlDifferentBusinessProperties(auditEntryOriginal));
239     assertFalse(auditEntryCopy.xmlDifferentUpdateProperties(auditEntryOriginal));
240
