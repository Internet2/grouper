85    public void testConvertToXml() {
86      
87      XmlExportAuditEntry xmlExportAuditEntry = new XmlExportAuditEntry();
88      
89      xmlExportAuditEntry.setActAsMemberId("actAsMemberId");
90      xmlExportAuditEntry.setAuditTypeId("auditTypeId");
91      xmlExportAuditEntry.setContextId("contextId");
92      xmlExportAuditEntry.setCreatedOn("createdOn");
93      xmlExportAuditEntry.setDescription("description");
94      xmlExportAuditEntry.setDurationMicroseconds(3L);
95      xmlExportAuditEntry.setEnvName("envName");
96      xmlExportAuditEntry.setGrouperEngine("grouperEngine");
97      xmlExportAuditEntry.setGrouperVersion("grouperVersion");
98      xmlExportAuditEntry.setHibernateVersionNumber(4L);
99      xmlExportAuditEntry.setId("id");
100     xmlExportAuditEntry.setInt01(11L);
101     xmlExportAuditEntry.setInt02(12L);
102     xmlExportAuditEntry.setInt03(13L);
103     xmlExportAuditEntry.setInt04(14L);
104     xmlExportAuditEntry.setInt05(15L);
105     xmlExportAuditEntry.setLastUpdated("lastUpdated");
106     xmlExportAuditEntry.setLoggedInMemberId("loggedInMemberId");
107     xmlExportAuditEntry.setQueryCount(5);
108     xmlExportAuditEntry.setServerHost("serverHost");
109     xmlExportAuditEntry.setServerUserName("serverUserName");
110     xmlExportAuditEntry.setString01("string01");
111     xmlExportAuditEntry.setString02("string02");
112     xmlExportAuditEntry.setString03("string03");
113     xmlExportAuditEntry.setString04("string04");
114     xmlExportAuditEntry.setString05("string05");
115     xmlExportAuditEntry.setString06("string06");
116     xmlExportAuditEntry.setString07("string07");
117     xmlExportAuditEntry.setString08("string08");
118     xmlExportAuditEntry.setUserIpAddress("userIpAddress");
119 
120     
121     String xml = xmlExportAuditEntry.toXml(new GrouperVersion(GrouperVersion.GROUPER_VERSION));
122     
123     xmlExportAuditEntry = XmlExportAuditEntry.fromXml(new GrouperVersion(GrouperVersion.GROUPER_VERSION), xml);
124 
125     assertEquals("actAsMemberId", xmlExportAuditEntry.getActAsMemberId());
126     assertEquals("auditTypeId", xmlExportAuditEntry.getAuditTypeId());
127     assertEquals("contextId", xmlExportAuditEntry.getContextId());
128     assertEquals("createdOn", xmlExportAuditEntry.getCreatedOn());
129     assertEquals("description", xmlExportAuditEntry.getDescription());
130     assertEquals(3L, xmlExportAuditEntry.getDurationMicroseconds());
131     assertEquals("envName", xmlExportAuditEntry.getEnvName());
132     assertEquals("grouperEngine", xmlExportAuditEntry.getGrouperEngine());
133     assertEquals("grouperVersion", xmlExportAuditEntry.getGrouperVersion());
134     assertEquals(4L, xmlExportAuditEntry.getHibernateVersionNumber());
135     assertEquals("id", xmlExportAuditEntry.getId());
136     assertEquals(new Long(11), xmlExportAuditEntry.getInt01());
137     assertEquals(new Long(12), xmlExportAuditEntry.getInt02());
138     assertEquals(new Long(13), xmlExportAuditEntry.getInt03());
139     assertEquals(new Long(14), xmlExportAuditEntry.getInt04());
140     assertEquals(new Long(15), xmlExportAuditEntry.getInt05());
141     assertEquals("lastUpdated", xmlExportAuditEntry.getLastUpdated());
142     assertEquals("loggedInMemberId", xmlExportAuditEntry.getLoggedInMemberId());
143     assertEquals(5, xmlExportAuditEntry.getQueryCount());
144     assertEquals("serverHost", xmlExportAuditEntry.getServerHost());
145     assertEquals("serverUserName", xmlExportAuditEntry.getServerUserName());
146     assertEquals("string01", xmlExportAuditEntry.getString01());
147     assertEquals("string02", xmlExportAuditEntry.getString02());
148     assertEquals("string03", xmlExportAuditEntry.getString03());
149     assertEquals("string04", xmlExportAuditEntry.getString04());
150     assertEquals("string05", xmlExportAuditEntry.getString05());
151     assertEquals("string06", xmlExportAuditEntry.getString06());
152     assertEquals("string07", xmlExportAuditEntry.getString07());
153     assertEquals("string08", xmlExportAuditEntry.getString08());
154     assertEquals("userIpAddress", xmlExportAuditEntry.getUserIpAddress());
