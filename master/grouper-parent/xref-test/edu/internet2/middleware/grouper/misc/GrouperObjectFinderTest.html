108   public void testFindFoldersAllowed() {
109     GrouperSession grouperSession = GrouperSession.startRootSession();
110     
111     Stem test = new StemSave(grouperSession).assignName("test").save();
112     
113     Stem stem1 = new StemSave(grouperSession).assignName("test:stem1").save();
114     Group group1 = new GroupSave(grouperSession).assignName("test:stem1:group1").save();
115     group1.grantPriv(SubjectTestHelper.SUBJ0, AccessPrivilege.READ);
116     
117     Stem stem2 = new StemSave(grouperSession).assignName("test:stem2").save();
118     Group group2 = new GroupSave(grouperSession).assignName("test:stem2:group2").save();
119     group2.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTIN);
120     
121     Stem stem3 = new StemSave(grouperSession).assignName("test:stem3").save();
122     Group group3 = new GroupSave(grouperSession).assignName("test:stem3:group3").save();
123     group3.grantPriv(SubjectTestHelper.SUBJ1, AccessPrivilege.VIEW);
124     
125     Stem stem4 = new StemSave(grouperSession).assignName("test:stem4").save();
126     Stem stem4sub = new StemSave(grouperSession).assignName("test:stem4:stem4sub").save();
127     stem4sub.grantPriv(SubjectTestHelper.SUBJ0, NamingPrivilege.CREATE);
128     
129     
130     Stem stem5 = new StemSave(grouperSession).assignName("test:stem5").save();
131     AttributeDef attributeDef5 = new AttributeDefSave(grouperSession).assignName("test:stem5:attributeDef5")
132         .assignAttributeDefType(AttributeDefType.attr).assignValueType(AttributeDefValueType.marker).save();
133     attributeDef5.getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_DEF_ATTR_READ, false);
134 
135     Stem stem6 = new StemSave(grouperSession).assignName("test:stem6").save();
136     Stem stem6sub = new StemSave(grouperSession).assignName("test:stem6:stem6sub").save();
137     Group group6 = new GroupSave(grouperSession).assignName("test:stem6:stem6sub:group6").save();
138     group6.grantPriv(SubjectTestHelper.SUBJ0, AccessPrivilege.VIEW);
139 
140     Stem stem7 = new StemSave(grouperSession).assignName("test:stem7").save();
141     stem7.grantPriv(SubjectTestHelper.SUBJ1, NamingPrivilege.CREATE);
142 
143     Stem stem8 = new StemSave(grouperSession).assignName("test:stem8").save();
144     Stem stem8sub = new StemSave(grouperSession).assignName("test:stem8:stem8sub").save();
145     Stem stem8subsub = new StemSave(grouperSession).assignName("test:stem8:stem8sub:stem8subsub").save();
146     stem8subsub.grantPriv(SubjectTestHelper.SUBJ0, NamingPrivilege.STEM_ATTR_READ);
147 
148     Stem stem9 = new StemSave(grouperSession).assignName("test:stem9").save();
149     Stem stem9sub = new StemSave(grouperSession).assignName("test:stem9:stem9sub").save();
150     AttributeDef attributeDef9sub = new AttributeDefSave(grouperSession).assignName("test:stem9:stem9sub:attributeDef9")
151         .assignAttributeDefType(AttributeDefType.attr).assignValueType(AttributeDefValueType.marker).save();
152     attributeDef9sub.getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_ADMIN, false);
153 
154     GrouperSession.stopQuietly(grouperSession);
155 
156     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ0);
157 
158     QueryOptions queryOptions = QueryOptions.create("displayExtension", true, 1, 50);
159 
160     GrouperObjectFinder grouperObjectFinder = new GrouperObjectFinder()
161       .assignObjectPrivilege(ObjectPrivilege.view)
162       .assignParentStemId(test.getId())
163       .assignQueryOptions(queryOptions)
164       .assignSplitScope(true).assignStemScope(Scope.ONE)
165       .assignSubject(GrouperSession.staticGrouperSession().getSubject());
166 
167 //    if (!StringUtils.isBlank(filterText)) {
168 //      grouperObjectFinder.assignFilterText(filterText);
169 //    }
170 
171     List<GrouperObject> results = new ArrayList<GrouperObject>(grouperObjectFinder.findGrouperObjects());
172     
173     assertEquals(7, results.size());
174     assertEquals(stem1.getName(), results.get(0).getName());
175     assertEquals(stem2.getName(), results.get(1).getName());
176     assertEquals(stem4.getName(), results.get(2).getName());
177     assertEquals(stem5.getName(), results.get(3).getName());
178     assertEquals(stem6.getName(), results.get(4).getName());
179     assertEquals(stem8.getName(), results.get(5).getName());
180     assertEquals(stem9.getName(), results.get(6).getName());
181     
182     GrouperSession.stopQuietly(grouperSession);
183 
184   
185     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ1);
186     
187     grouperObjectFinder = new GrouperObjectFinder()
188       .assignObjectPrivilege(ObjectPrivilege.view)
189       .assignParentStemId(test.getId())
190       .assignQueryOptions(queryOptions)
191       .assignSplitScope(true).assignStemScope(Scope.ONE)
192       .assignSubject(GrouperSession.staticGrouperSession().getSubject());
193 
194     results = new ArrayList<GrouperObject>(grouperObjectFinder.findGrouperObjects());
195     
196     assertEquals(3, results.size());
197     assertEquals(stem2.getName(), results.get(0).getName());
198     assertEquals(stem3.getName(), results.get(1).getName());
199     assertEquals(stem7.getName(), results.get(2).getName());
200     
201     GrouperSession.stopQuietly(grouperSession);
202 
203   
204   }
