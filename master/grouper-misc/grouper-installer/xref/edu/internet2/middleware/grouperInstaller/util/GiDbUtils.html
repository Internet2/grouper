89    /** 
90     * oracle types
91     */
92    public static enum DbType {
93  
94      /** string type */
95      STRING {
96  
97        /**
98         * 
99         * @see edu.internet2.middleware.GiDbUtils.util.GcDbUtils.DbType#processResultSet(java.sql.ResultSet, int)
100        */
101       @Override
102       public Object processResultSet(ResultSet resultSet, int indexZeroIndexed) {
103         try {
104           return resultSet.getString(indexZeroIndexed+1);
105         } catch (SQLException e) {
106           throw new RuntimeException("Error reading col (zero indexed) " + indexZeroIndexed, e);
107         }
108       }
109 
110       /**
111        * @see edu.internet2.middleware.GiDbUtils.util.GcDbUtils.DbType#attachParam(PreparedStatement, Object, int)
112        */
113       @Override
114       public void attachParam(PreparedStatement preparedStatement, Object arg,
115           int indexZeroIndexed) {
116         String argString = GrouperInstallerUtils.stringValue(arg);
117         try {
118           preparedStatement.setString(indexZeroIndexed+1, argString);
119         } catch (SQLException e) {
120           throw new RuntimeException("Error setting param (zero indexed) " + indexZeroIndexed + ": " + argString, e);
121         }
122       }
123     };
124 
125     /**
126      * get the data
127      * @param resultSet
128      * @param indexZeroIndexed
129      * @return the object or null
130      */
131     public abstract Object processResultSet(ResultSet resultSet, int indexZeroIndexed);
132     
133     /**
134      * attach a param to a prepared statement
135      * @param preparedStatement
136      * @param arg
137      * @param indexZeroIndexed
138      */
139     public abstract void attachParam(PreparedStatement preparedStatement, Object arg, int indexZeroIndexed);
140     
141     /**
142      * convert an object arg to a dbtype
143      * @param object
144      * @return dbtype
145      */
146     public static DbType fromObject(Object object) {
147       
148       if (object == null || object instanceof String) {
149         return DbType.STRING;
150       }
151       throw new RuntimeException("Unsupported type: " + GrouperInstallerUtils.className(object) + ", " + object);
152     }
153     
154     /**
155      * convert an object arg to a dbtype
156      * @param theClass
157      * @return dbtype
158      */
159     public static DbType fromClass(Class<?> theClass) {
160       
161       if (theClass instanceof Class<?> && String.class.equals(theClass)) {
162         return DbType.STRING;
163       }
164       
165       throw new RuntimeException("Unsupported type: " + theClass);
166     }

256   /**
257    * if there is no driver class specified, then try to derive it from the URL
258    * @param connectionUrl
259    * @param driverClassName
260    * @return the driver class
261    */
262   public static String convertUrlToDriverClassIfNeeded(String connectionUrl, String driverClassName) {
263     //default some of the stuff
264     if (GrouperInstallerUtils.isBlank(driverClassName)) {
265       
266       if (isHsql(connectionUrl)) {
267         driverClassName = "org.hsqldb.jdbcDriver";
268       } else if (isMysql(connectionUrl)) {
269         driverClassName = "com.mysql.jdbc.Driver";
270       } else if (isOracle(connectionUrl)) {
271         driverClassName = "oracle.jdbc.driver.OracleDriver";
272       } else if (isPostgres(connectionUrl)) { 
273         driverClassName = "org.postgresql.Driver";
274       } else if (isSQLServer(connectionUrl)) {
275         driverClassName = "com.microsoft.sqlserver.jdbc.SQLServerDriver";
276       } else {
277         
278         //if this is blank we will figure it out later
279         if (!GrouperInstallerUtils.isBlank(connectionUrl)) {
280         
281           String error = "Cannot determine the driver class from database URL: " + connectionUrl;
282           System.err.println(error);
283           LOG.error(error);
284           return null;
285         }
286       }
287     }
288     return driverClassName;
289 
290   }

292   /**
293    * get a connection to the oracle DB
294    * @return a connection
295    */
296   public Connection connection() {
297     
298     String oracleUrl = null;
299     
300     try {
301       
302       String driverClass = convertUrlToDriverClassIfNeeded(this.url, null);
303   
304       GrouperInstallerUtils.forName(driverClass);
305       
306       Connection conn = DriverManager.getConnection(this.url, this.user, this.pass);
307       
308       return conn;
309     } catch (Exception e) {
310       throw new RuntimeException("Error connecting to: " + oracleUrl, e);
311     }
312   }

326   /**
327    * select a value from the db
328    * @param <T>
329    * @param colType
330    * @param query
331    * @return the data
332    */
333   public <T> T select(Class<T> colType, String query) {
334     List<T> rows = listSelect(colType, query, GrouperInstallerUtils.toList(DbType.fromClass(colType)));
335     
336     T data = GrouperInstallerUtils.listPopOne(rows);
337     
338     return data;
339   }

341   /**
342    * run a query and get rows back
343    * @param rowType type of each row returned, e.g. String.class or Object[]
344    * @param <T> generic type
345    * @param query
346    * @param returnColTypes
347    * @param args preparedstatement arguments
348    * @return the list of objects
349    */
350   @SuppressWarnings("unchecked")
351   public <T> List<T> listSelect(Class<T> rowType, String query, List<DbType> returnColTypes, List<Object> args) {
352     Connection connection = null;
353     PreparedStatement preparedStatement = null;
354     ResultSet resultSet = null;
355     List<T> results = new ArrayList<T>();
356     
357     try {
358       
359       int returnColTypesLength = GrouperInstallerUtils.length(returnColTypes);
360       if (returnColTypesLength==0) {
361         throw new RuntimeException("Why is returnColTypesLength == 0???");
362       }
363       
364       connection = this.connection();
365       preparedStatement = connection.prepareStatement(query);
366       
367       //attach arguments
368       int argsLength = GrouperInstallerUtils.length(args);
369 
370       for (int i=0; i<argsLength;i++) {
371         Object arg = args.get(i);
372         DbType dbType = DbType.fromObject(arg);
373         dbType.attachParam(preparedStatement, arg, i);
374       }
375       
376       resultSet = preparedStatement.executeQuery();
377 
378       if (rowType.isArray() != returnColTypesLength>1) {
379         throw new RuntimeException("If returnColTypesLength > 1 (" + returnColTypesLength + ") then you must pass in an array as the return type");
380       }
381       
382       while (resultSet.next()) {
383         
384         //if array of cols
385         if (rowType.isArray()) {
386         
387           Object[] result = new Object[returnColTypesLength];
388           results.add((T)result);
389           
390           for (int i=0; i<returnColTypesLength;i++) {
391             result[i] = returnColTypes.get(i).processResultSet(resultSet, i);
392           }
393         } else {
394           //if one col
395           results.add((T)returnColTypes.get(0).processResultSet(resultSet, 0));
396         }
397         
398       }
399     } catch (Exception e) {
400       throw new RuntimeException("Error with query: " + query, e);
401     } 
402     finally {
403       rollbackQuietly(connection);
404       GrouperInstallerUtils.closeQuietly(resultSet);
405       GrouperInstallerUtils.closeQuietly(preparedStatement);
406       GrouperInstallerUtils.closeQuietly(connection);
407     }
408     return results;
409 
410   }

481   /**
482    * check connection to the db
483    * @return the query to check connection with
484    */
485   public String checkConnectionQuery() {
486     if (this.isHsql()) {
487       return "SELECT 1 FROM INFORMATION_SCHEMA.SYSTEM_USERS";
488     }
489     if (this.isMysql()) {
490       return "select 1";
491     }
492     if (this.isOracle()) {
493       return "select 1 from dual";
494     }
495     if (this.isPostgres()) {
496       return "select 1";
497     }
498     if (this.isSQLServer()) {
499       return "select 1";
500     }
501     throw new RuntimeException("Cant find which database type from URL: " + this.url);
502 
503   }

505   /**
506    * execute a query (insert/update/delete/etc)
507    * @param query query to execute
508    * @param args preparedstatement arguments
509    * @param commit if we should commit
510    * @return either (1) the row count for SQL Data Manipulation Language (DML) statements
511      *         or (2) 0 for SQL statements that return nothing
512    */
513   public int executeUpdate(String query, List<Object> args, boolean commit) {
514     Connection connection = null;
515     PreparedStatement preparedStatement = null;
516 
517     try {
518       connection = this.connection();
519       preparedStatement = connection.prepareStatement(query);
520       
521       //attach arguments
522       int argsLength = GrouperInstallerUtils.length(args);
523 
524       for (int i=0; i<argsLength;i++) {
525         Object arg = args.get(i);
526         DbType dbType = DbType.fromObject(arg);
527         dbType.attachParam(preparedStatement, arg, i);
528       }
529       
530       int result = preparedStatement.executeUpdate();
531       
532       if (commit) {
533         connection.commit();
534       }
535       
536       return result;
537       
538     } catch (Exception e) {
539       rollbackQuietly(connection);
540       throw new RuntimeException("Error with query: " + query, e);
541     } 
542     finally {
543       GrouperInstallerUtils.closeQuietly(preparedStatement);
544       GrouperInstallerUtils.closeQuietly(connection);
545     }
