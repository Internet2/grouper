20  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip;
21  
22  import java.io.DataInputStream;
23  import java.io.IOException;
24  import java.io.InputStream;
25  import java.util.Arrays;
26  
27  /**
28   * Binary tree of positive values.
29   * 
30   * @author Emmanuel Bourg
31   * @since 1.7
32   */
33  class BinaryTree {
34  
35      /** Value in the array indicating an undefined node */
36      private static final int UNDEFINED = -1;
37      
38      /** Value in the array indicating a non leaf node */
39      private static final int NODE = -2;
40  
41      /** 
42       * The array representing the binary tree. The root is at index 0,
43       * the left children are at 2*i+1 and the right children at 2*i+2.
44       */
45      private final int[] tree;
46  
47      public BinaryTree(int depth) {
48          tree = new int[(1 << (depth + 1)) - 1];
49          Arrays.fill(tree, UNDEFINED);
50      }
51  
52      /**
53       * Adds a leaf to the tree.
54       * 
55       * @param node   the index of the node where the path is appended
56       * @param path   the path to the leaf (bits are parsed from the right to the left)
57       * @param depth  the number of nodes in the path
58       * @param value  the value of the leaf (must be positive)
59       */
60      public void addLeaf(int node, int path, int depth, int value) {
61          if (depth == 0) {
62              // end of the path reached, add the value to the current node
63              if (tree[node] == UNDEFINED) {
64                  tree[node] = value;
65              } else {
66                  throw new IllegalArgumentException("Tree value at index " + node + " has already been assigned (" + tree[node] + ")");
67              }
68          } else {
69              // mark the current node as a non leaf node
70              tree[node] = NODE;
71              
72              // move down the path recursively
73              int nextChild = 2 * node + 1 + (path & 1);
74              addLeaf(nextChild, path >>> 1, depth - 1, value);
75          }
76      }
77  
78      /**
79       * Reads a value from the specified bit stream.
80       * 
81       * @param stream
82       * @return the value decoded, or -1 if the end of the stream is reached
83       */
84      public int read(BitStream stream) throws IOException {
85          int currentIndex = 0;
86  
87          while (true) {
88              int bit = stream.nextBit();
89              if (bit == -1) {
90                  return -1;
91              }
92  
93              int childIndex = 2 * currentIndex + 1 + bit;
94              int value = tree[childIndex];
95              if (value == NODE) {
96                  // consume the next bit
97                  currentIndex = childIndex;
98              } else if (value != UNDEFINED) {
99                  return value;
100             } else {
101                 throw new IOException("The child " + bit + " of node at index " + currentIndex + " is not defined");
102             }
103         }
104     }
105     
106 
107     /**
108      * Decodes the packed binary tree from the specified stream.
109      */
110     static BinaryTree decode(InputStream in, final int totalNumberOfValues) throws IOException {
111         // the first byte contains the size of the structure minus one
112         int size = in.read() + 1;
113         if (size == 0) {
114             throw new IOException("Cannot read the size of the encoded tree, unexpected end of stream");
115         }
116 
117         byte[] encodedTree = new byte[size];
118         new DataInputStream(in).readFully(encodedTree);
119 
120         /** The maximum bit length for a value (16 or lower) */
121         int maxLength = 0;
122         
123         int[] originalBitLengths = new int[totalNumberOfValues];
124         int pos = 0;
125         for (byte b : encodedTree) {
126             // each byte encodes the number of values (upper 4 bits) for a bit length (lower 4 bits)
127             int numberOfValues = ((b & 0xF0) >> 4) + 1;
128             int bitLength = (b & 0x0F) + 1;
129 
130             for (int j = 0; j < numberOfValues; j++) {
131                 originalBitLengths[pos++] = bitLength;
132             }
133             
134             maxLength = Math.max(maxLength, bitLength);
135         }
136 
137         // sort the array of bit lengths and memorize the permutation used to restore the order of the codes
138         int[] permutation = new int[originalBitLengths.length];
139         for (int k = 0; k < permutation.length; k++) {
140             permutation[k] = k;
141         }
142         
143         int c = 0;
144         int[] sortedBitLengths = new int[originalBitLengths.length];
145         for (int k = 0; k < originalBitLengths.length; k++) {
146             // iterate over the values
147             for (int l = 0; l < originalBitLengths.length; l++) {
148                 // look for the value in the original array
149                 if (originalBitLengths[l] == k) {
150                     // put the value at the current position in the sorted array...
151                     sortedBitLengths[c] = k;
152                     
153                     // ...and memorize the permutation
154                     permutation[c] = l; 
155                     
156                     c++;
157                 }
158             }
159         }
160 
161         // decode the values of the tree
162         int code = 0;
163         int codeIncrement = 0;
164         int lastBitLength = 0;
165 
166         int[] codes = new int[totalNumberOfValues];
167 
168         for (int i = totalNumberOfValues - 1; i >= 0; i--) {
169             code = code + codeIncrement;
170             if (sortedBitLengths[i] != lastBitLength) {
171                 lastBitLength = sortedBitLengths[i];
172                 codeIncrement = 1 << (16 - lastBitLength);
173             }
174             codes[permutation[i]] = code;
175         }
176         
177         // build the tree
178         BinaryTree tree = new BinaryTree(maxLength);
179         
180         for (int k = 0; k < codes.length; k++) {
181             int bitLength = originalBitLengths[k];
182             if (bitLength > 0) {
183                 tree.addLeaf(0, Integer.reverse(codes[k] << 16), bitLength, k);
184             }
185         }
186 
187         return tree;
188     }
189 }
