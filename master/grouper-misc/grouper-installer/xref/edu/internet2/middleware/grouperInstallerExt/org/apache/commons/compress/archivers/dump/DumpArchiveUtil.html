19  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.dump;
20  
21  import java.io.IOException;
22  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip.ZipEncoding;
23  
24  /**
25   * Various utilities for dump archives.
26   */
27  class DumpArchiveUtil {
28      /**
29       * Private constructor to prevent instantiation.
30       */
31      private DumpArchiveUtil() {
32      }
33  
34      /**
35       * Calculate checksum for buffer.
36       *
37       * @param buffer buffer containing tape segment header
38       * @returns checksum
39       */
40      public static int calculateChecksum(byte[] buffer) {
41          int calc = 0;
42  
43          for (int i = 0; i < 256; i++) {
44              calc += DumpArchiveUtil.convert32(buffer, 4 * i);
45          }
46  
47          return DumpArchiveConstants.CHECKSUM -
48          (calc - DumpArchiveUtil.convert32(buffer, 28));
49      }
50  
51      /**
52       * Verify that the buffer contains a tape segment header.
53       *
54       * @param buffer
55       */
56      public static final boolean verify(byte[] buffer) {
57          // verify magic. for now only accept NFS_MAGIC.
58          int magic = convert32(buffer, 24);
59  
60          if (magic != DumpArchiveConstants.NFS_MAGIC) {
61              return false;
62          }
63  
64          //verify checksum...
65          int checksum = convert32(buffer, 28);
66  
67          if (checksum != calculateChecksum(buffer)) {
68              return false;
69          }
70  
71          return true;
72      }
73  
74      /**
75       * Get the ino associated with this buffer.
76       *
77       * @param buffer
78       */
79      public static final int getIno(byte[] buffer) {
80          return convert32(buffer, 20);
81      }
82  
83      /**
84       * Read 8-byte integer from buffer.
85       *
86       * @param buffer
87       * @param offset
88       * @return the 8-byte entry as a long
89       */
90      public static final long convert64(byte[] buffer, int offset) {
91          long i = 0;
92          i += (((long) buffer[offset + 7]) << 56);
93          i += (((long) buffer[offset + 6] << 48) & 0x00FF000000000000L);
94          i += (((long) buffer[offset + 5] << 40) & 0x0000FF0000000000L);
95          i += (((long) buffer[offset + 4] << 32) & 0x000000FF00000000L);
96          i += (((long) buffer[offset + 3] << 24) & 0x00000000FF000000L);
97          i += (((long) buffer[offset + 2] << 16) & 0x0000000000FF0000L);
98          i += (((long) buffer[offset + 1] << 8) & 0x000000000000FF00L);
99          i += (buffer[offset] & 0x00000000000000FFL);
100 
101         return i;
102     }
103 
104     /**
105      * Read 4-byte integer from buffer.
106      *
107      * @param buffer
108      * @param offset
109      * @return the 4-byte entry as an int
110      */
111     public static final int convert32(byte[] buffer, int offset) {
112         int i = 0;
113         i = buffer[offset + 3] << 24;
114         i += (buffer[offset + 2] << 16) & 0x00FF0000;
115         i += (buffer[offset + 1] << 8) & 0x0000FF00;
116         i += buffer[offset] & 0x000000FF;
117 
118         return i;
119     }
120 
121     /**
122      * Read 2-byte integer from buffer.
123      *
124      * @param buffer
125      * @param offset
126      * @return the 2-byte entry as an int
127      */
128     public static final int convert16(byte[] buffer, int offset) {
129         int i = 0;
130         i += (buffer[offset + 1] << 8) & 0x0000FF00;
131         i += buffer[offset] & 0x000000FF;
132 
133         return i;
134     }
135 
136     /**
137      * Decodes a byte array to a string.
138      */
139     static String decode(ZipEncoding encoding, byte[] b, int offset, int len)
140         throws IOException {
141         byte[] copy = new byte[len];
142         System.arraycopy(b, offset, copy, 0, len);
143         return encoding.decode(copy);
144     }
145 }
