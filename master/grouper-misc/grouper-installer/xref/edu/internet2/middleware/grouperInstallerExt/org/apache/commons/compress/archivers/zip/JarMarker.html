18  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip;
19  
20  import java.util.zip.ZipException;
21  
22  /**
23   * If this extra field is added as the very first extra field of the
24   * archive, Solaris will consider it an executable jar file.
25   * @Immutable
26   */
27  public final class JarMarker implements ZipExtraField {
28  
29      private static final ZipShort ID = new ZipShort(0xCAFE);
30      private static final ZipShort NULL = new ZipShort(0);
31      private static final byte[] NO_BYTES = new byte[0];
32      private static final JarMarker DEFAULT = new JarMarker();
33  
34      /** No-arg constructor */
35      public JarMarker() {
36          // empty
37      }
38  
39      /**
40       * Since JarMarker is stateless we can always use the same instance.
41       * @return the DEFAULT jarmaker.
42       */
43      public static JarMarker getInstance() {
44          return DEFAULT;
45      }
46  
47      /**
48       * The Header-ID.
49       * @return the header id
50       */
51      public ZipShort getHeaderId() {
52          return ID;
53      }
54  
55      /**
56       * Length of the extra field in the local file data - without
57       * Header-ID or length specifier.
58       * @return 0
59       */
60      public ZipShort getLocalFileDataLength() {
61          return NULL;
62      }
63  
64      /**
65       * Length of the extra field in the central directory - without
66       * Header-ID or length specifier.
67       * @return 0
68       */
69      public ZipShort getCentralDirectoryLength() {
70          return NULL;
71      }
72  
73      /**
74       * The actual data to put into local file data - without Header-ID
75       * or length specifier.
76       * @return the data
77       */
78      public byte[] getLocalFileDataData() {
79          return NO_BYTES;
80      }
81  
82      /**
83       * The actual data to put central directory - without Header-ID or
84       * length specifier.
85       * @return the data
86       */
87      public byte[] getCentralDirectoryData() {
88          return NO_BYTES;
89      }
90  
91      /**
92       * Populate data from this array as if it was in local file data.
93       * @param data an array of bytes
94       * @param offset the start offset
95       * @param length the number of bytes in the array from offset
96       *
97       * @throws ZipException on error
98       */
99      public void parseFromLocalFileData(byte[] data, int offset, int length)
100         throws ZipException {
101         if (length != 0) {
102             throw new ZipException("JarMarker doesn't expect any data");
103         }
104     }
105 
106     /**
107      * Doesn't do anything special since this class always uses the
108      * same data in central directory and local file data.
109      */
110     public void parseFromCentralDirectoryData(byte[] buffer, int offset,
111                                               int length)
112         throws ZipException {
113         parseFromLocalFileData(buffer, offset, length);
114     }
115 }
