18  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.sevenz;
19  
20  import java.io.IOException;
21  import java.io.InputStream;
22  import java.io.OutputStream;
23  
24  /**
25   * Base Codec class.
26   */
27  abstract class CoderBase {
28      private final Class<?>[] acceptableOptions;
29      private static final byte[] NONE = new byte[0];
30  
31      /**
32       * @param acceptableOptions types that can be used as options for this codec.
33       */
34      protected CoderBase(Class<?>... acceptableOptions) {
35          this.acceptableOptions = acceptableOptions;
36      }
37  
38      /**
39       * @return whether this method can extract options from the given object.
40       */
41      boolean canAcceptOptions(Object opts) {
42          for (Class<?> c : acceptableOptions) {
43              if (c.isInstance(opts)) {
44                  return true;
45              }
46          }
47          return false;
48      }
49  
50      /**
51       * @return property-bytes to write in a Folder block
52       */
53      byte[] getOptionsAsProperties(Object options) {
54          return NONE;
55      }
56  
57      /**
58       * @return configuration options that have been used to create the given InputStream from the given Coder
59       */
60      Object getOptionsFromCoder(Coder coder, InputStream in) {
61          return null;
62      }
63  
64      /**
65       * @return a stream that reads from in using the configured coder and password.
66       */
67      abstract InputStream decode(final InputStream in, long uncomressedLength,
68          final Coder coder, byte[] password) throws IOException;
69  
70      /**
71       * @return a stream that writes to out using the given configuration.
72       */
73      OutputStream encode(final OutputStream out, final Object options) throws IOException {
74          throw new UnsupportedOperationException("method doesn't support writing");
75      }
76  
77      /**
78       * If the option represents a number, return its integer
79       * value, otherwise return the given default value.
80       */
81      protected static int numberOptionOrDefault(Object options, int defaultValue) {
82          return options instanceof Number ? ((Number) options).intValue() : defaultValue;
83      }
84  }
