18  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip;
19  
20  import java.io.Serializable;
21  
22  import static edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip.ZipConstants.BYTE_MASK;
23  
24  /**
25   * Utility class that represents a two byte integer with conversion
26   * rules for the big endian byte order of ZIP files.
27   * @Immutable
28   */
29  public final class ZipShort implements Cloneable, Serializable {
30      private static final long serialVersionUID = 1L;
31  
32      private static final int BYTE_1_MASK = 0xFF00;
33      private static final int BYTE_1_SHIFT = 8;
34  
35      private final int value;
36  
37      /**
38       * Create instance from a number.
39       * @param value the int to store as a ZipShort
40       */
41      public ZipShort (int value) {
42          this.value = value;
43      }
44  
45      /**
46       * Create instance from bytes.
47       * @param bytes the bytes to store as a ZipShort
48       */
49      public ZipShort (byte[] bytes) {
50          this(bytes, 0);
51      }
52  
53      /**
54       * Create instance from the two bytes starting at offset.
55       * @param bytes the bytes to store as a ZipShort
56       * @param offset the offset to start
57       */
58      public ZipShort (byte[] bytes, int offset) {
59          value = ZipShort.getValue(bytes, offset);
60      }
61  
62      /**
63       * Get value as two bytes in big endian byte order.
64       * @return the value as a a two byte array in big endian byte order
65       */
66      public byte[] getBytes() {
67          byte[] result = new byte[2];
68          result[0] = (byte) (value & BYTE_MASK);
69          result[1] = (byte) ((value & BYTE_1_MASK) >> BYTE_1_SHIFT);
70          return result;
71      }
72  
73      /**
74       * Get value as Java int.
75       * @return value as a Java int
76       */
77      public int getValue() {
78          return value;
79      }
80  
81      /**
82       * Get value as two bytes in big endian byte order.
83       * @param value the Java int to convert to bytes
84       * @return the converted int as a byte array in big endian byte order
85       */
86      public static byte[] getBytes(int value) {
87          byte[] result = new byte[2];
88          result[0] = (byte) (value & BYTE_MASK);
89          result[1] = (byte) ((value & BYTE_1_MASK) >> BYTE_1_SHIFT);
90          return result;
91      }
92  
93      /**
94       * Helper method to get the value as a java int from two bytes starting at given array offset
95       * @param bytes the array of bytes
96       * @param offset the offset to start
97       * @return the corresponding java int value
98       */
99      public static int getValue(byte[] bytes, int offset) {
100         int value = (bytes[offset + 1] << BYTE_1_SHIFT) & BYTE_1_MASK;
101         value += (bytes[offset] & BYTE_MASK);
102         return value;
103     }
104 
105     /**
106      * Helper method to get the value as a java int from a two-byte array
107      * @param bytes the array of bytes
108      * @return the corresponding java int value
109      */
110     public static int getValue(byte[] bytes) {
111         return getValue(bytes, 0);
112     }
113 
114     /**
115      * Override to make two instances with same value equal.
116      * @param o an object to compare
117      * @return true if the objects are equal
118      */
119     @Override
120     public boolean equals(Object o) {
121         if (o == null || !(o instanceof ZipShort)) {
122             return false;
123         }
124         return value == ((ZipShort) o).getValue();
125     }
126 
127     /**
128      * Override to make two instances with same value equal.
129      * @return the value stored in the ZipShort
130      */
131     @Override
132     public int hashCode() {
133         return value;
134     }
135 
136     @Override
137     public Object clone() {
138         try {
139             return super.clone();
140         } catch (CloneNotSupportedException cnfe) {
141             // impossible
142             throw new RuntimeException(cnfe);
143         }
144     }
145 
146     @Override
147     public String toString() {
148         return "ZipShort value: " + value;
149     }
150 }
