20  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip;
21  
22  import java.io.IOException;
23  import java.nio.ByteBuffer;
24  import java.util.ArrayList;
25  import java.util.Collections;
26  import java.util.List;
27  
28  /**
29   * This ZipEncoding implementation implements a simple 8bit character
30   * set, which mets the following restrictions:
31   * 
32   * <ul>
33   * <li>Characters 0x0000 to 0x007f are encoded as the corresponding
34   *        byte values 0x00 to 0x7f.</li>
35   * <li>All byte codes from 0x80 to 0xff are mapped to a unique unicode
36   *       character in the range 0x0080 to 0x7fff. (No support for
37   *       UTF-16 surrogates)
38   * </ul>
39   * 
40   * <p>These restrictions most notably apply to the most prominent
41   * omissions of java-1.4's {@link java.nio.charset.Charset Charset}
42   * implementation, Cp437 and Cp850.</p>
43   * 
44   * <p>The methods of this class are reentrant.</p>
45   * @Immutable
46   */
47  class Simple8BitZipEncoding implements ZipEncoding {
48  
49      /**
50       * A character entity, which is put to the reverse mapping table
51       * of a simple encoding.
52       */
53      private static final class Simple8BitChar implements Comparable<Simple8BitChar> {
54          public final char unicode;
55          public final byte code;
56  
57          Simple8BitChar(byte code, char unicode) {
58              this.code = code;
59              this.unicode = unicode;
60          }
61  
62          public int compareTo(Simple8BitChar a) {
63              return this.unicode - a.unicode;
64          }
65  
66          @Override
67          public String toString() {
68              return "0x" + Integer.toHexString(0xffff & unicode)
69                  + "->0x" + Integer.toHexString(0xff & code);
70          }
71  
72          @Override
73          public boolean equals(Object o) {
74              if (o instanceof Simple8BitChar) {
75                  Simple8BitChar other = (Simple8BitChar) o;
76                  return unicode == other.unicode && code == other.code;
77              }
78              return false;
79          }
80  
81          @Override
82          public int hashCode() {
83              return unicode;
84          }
85      }
86  
87      /**
88       * The characters for byte values of 128 to 255 stored as an array of
89       * 128 chars.
90       */
91      private final char[] highChars;
92  
93      /**
94       * A list of {@link Simple8BitChar} objects sorted by the unicode
95       * field.  This list is used to binary search reverse mapping of
96       * unicode characters with a character code greater than 127.
97       */
98      private final List<Simple8BitChar> reverseMapping;
99  
100     /**
101      * @param highChars The characters for byte values of 128 to 255
102      * stored as an array of 128 chars.
103      */
104     public Simple8BitZipEncoding(char[] highChars) {
105         this.highChars = highChars.clone();
106         List<Simple8BitChar> temp =
107             new ArrayList<Simple8BitChar>(this.highChars.length);
108 
109         byte code = 127;
110 
111         for (char highChar : this.highChars) {
112             temp.add(new Simple8BitChar(++code, highChar));
113         }
114 
115         Collections.sort(temp);
116         this.reverseMapping = Collections.unmodifiableList(temp);
117     }
118 
119     /**
120      * Return the character code for a given encoded byte.
121      * 
122      * @param b The byte to decode.
123      * @return The associated character value.
124      */
125     public char decodeByte(byte b) {
126         // code 0-127
127         if (b >= 0) {
128             return (char) b;
129         }
130 
131         // byte is signed, so 128 == -128 and 255 == -1
132         return this.highChars[128 + b];
133     }
134 
135     /**
136      * @param c The character to encode.
137      * @return Whether the given unicode character is covered by this encoding.
138      */
139     public boolean canEncodeChar(char c) {
140 
141         if (c >= 0 && c < 128) {
142             return true;
143         }
144 
145         Simple8BitChar r = this.encodeHighChar(c);
146         return r != null;
147     }
148 
149     /**
150      * Pushes the encoded form of the given character to the given byte buffer.
151      * 
152      * @param bb The byte buffer to write to.
153      * @param c The character to encode.
154      * @return Whether the given unicode character is covered by this encoding.
155      *         If {@code false} is returned, nothing is pushed to the
156      *         byte buffer. 
157      */
158     public boolean pushEncodedChar(ByteBuffer bb, char c) {
159 
160         if (c >= 0 && c < 128) {
161             bb.put((byte) c);
162             return true;
163         }
164 
165         Simple8BitChar r = this.encodeHighChar(c);
166         if (r == null) {
167             return false;
168         }
169         bb.put(r.code);
170         return true;
171     }
172 
173     /**
174      * @param c A unicode character in the range from 0x0080 to 0x7f00
175      * @return A Simple8BitChar, if this character is covered by this encoding.
176      *         A {@code null} value is returned, if this character is not
177      *         covered by this encoding.
178      */
179     private Simple8BitChar encodeHighChar(char c) {
180         // for performance an simplicity, yet another reincarnation of
181         // binary search...
182         int i0 = 0;
183         int i1 = this.reverseMapping.size();
184 
185         while (i1 > i0) {
186 
187             int i = i0 + (i1 - i0) / 2;
188 
189             Simple8BitChar m = this.reverseMapping.get(i);
190 
191             if (m.unicode == c) {
192                 return m;
193             }
194 
195             if (m.unicode < c) {
196                 i0 = i + 1;
197             } else {
198                 i1 = i;
199             }
200         }
201 
202         if (i0 >= this.reverseMapping.size()) {
203             return null;
204         }
205 
206         Simple8BitChar r = this.reverseMapping.get(i0);
207 
208         if (r.unicode != c) {
209             return null;
210         }
211 
212         return r;
213     }
214 
215     /**
216      * @see
217      * edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip.ZipEncoding#canEncode(java.lang.String)
218      */
219     public boolean canEncode(String name) {
220 
221         for (int i=0;i<name.length();++i) {
222 
223             char c = name.charAt(i);
224 
225             if (!this.canEncodeChar(c)) {
226                 return false;
227             }
228         }
229 
230         return true;
231     }
232 
233     /**
234      * @see
235      * edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip.ZipEncoding#encode(java.lang.String)
236      */
237     public ByteBuffer encode(String name) {
238         ByteBuffer out = ByteBuffer.allocate(name.length()
239                                              + 6 + (name.length() + 1) / 2);
240 
241         for (int i=0;i<name.length();++i) {
242 
243             char c = name.charAt(i);
244 
245             if (out.remaining() < 6) {
246                 out = ZipEncodingHelper.growBuffer(out,out.position() + 6);
247             }
248 
249             if (!this.pushEncodedChar(out,c)) {
250 
251                 ZipEncodingHelper.appendSurrogate(out,c);
252             }
253         }
254 
255         out.limit(out.position());
256         out.rewind();
257         return out;
258     }
259 
260     /**
261      * @see
262      * edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.zip.ZipEncoding#decode(byte[])
263      */
264     public String decode(byte[] data) throws IOException {
265         char [] ret = new char[data.length];
266 
267         for (int i=0;i<data.length;++i) {
268             ret[i] = this.decodeByte(data[i]);
269         }
270 
271         return new String(ret);
272     }
