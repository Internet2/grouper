32  package edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.concurrent;
33  
34  import java.util.concurrent.atomic.AtomicReference;
35  
36  /**
37   * <p>
38   * A specialized implementation of the {@code ConcurrentInitializer} interface
39   * based on an {@link AtomicReference} variable.
40   * </p>
41   * <p>
42   * This class maintains a member field of type {@code AtomicReference}. It
43   * implements the following algorithm to create and initialize an object in its
44   * {@link #get()} method:
45   * <ul>
46   * <li>First it is checked whether the {@code AtomicReference} variable contains
47   * already a value. If this is the case, the value is directly returned.</li>
48   * <li>Otherwise the {@link #initialize()} method is called. This method must be
49   * defined in concrete subclasses to actually create the managed object.</li>
50   * <li>After the object was created by {@link #initialize()} it is checked
51   * whether the {@code AtomicReference} variable is still undefined. This has to
52   * be done because in the meantime another thread may have initialized the
53   * object. If the reference is still empty, the newly created object is stored
54   * in it and returned by this method.</li>
55   * <li>Otherwise the value stored in the {@code AtomicReference} is returned.</li>
56   * </ul>
57   * </p>
58   * <p>
59   * Because atomic variables are used this class does not need any
60   * synchronization. So there is no danger of deadlock, and access to the managed
61   * object is efficient. However, if multiple threads access the {@code
62   * AtomicInitializer} object before it has been initialized almost at the same
63   * time, it can happen that {@link #initialize()} is called multiple times. The
64   * algorithm outlined above guarantees that {@link #get()} always returns the
65   * same object though.
66   * </p>
67   * <p>
68   * Compared with the {@link LazyInitializer} class, this class can be more
69   * efficient because it does not need synchronization. The drawback is that the
70   * {@link #initialize()} method can be called multiple times which may be
71   * problematic if the creation of the managed object is expensive. As a rule of
72   * thumb this initializer implementation is preferable if there are not too many
73   * threads involved and the probability that multiple threads access an
74   * uninitialized object is small. If there is high parallelism,
75   * {@link LazyInitializer} is more appropriate.
76   * </p>
77   *
78   * @since 3.0
79   * @version $Id: AtomicInitializer.java 1088899 2011-04-05 05:31:27Z bayard $
80   * @param <T> the type of the object managed by this initializer class
81   */
82  public abstract class AtomicInitializer<T> implements ConcurrentInitializer<T> {
83      /** Holds the reference to the managed object. */
84      private final AtomicReference<T> reference = new AtomicReference<T>();
85  
86      /**
87       * Returns the object managed by this initializer. The object is created if
88       * it is not available yet and stored internally. This method always returns
89       * the same object.
90       *
91       * @return the object created by this {@code AtomicInitializer}
92       * @throws ConcurrentException if an error occurred during initialization of
93       * the object
94       */
95      public T get() throws ConcurrentException {
96          T result = reference.get();
97  
98          if (result == null) {
99              result = initialize();
100             if (!reference.compareAndSet(null, result)) {
101                 // another thread has initialized the reference
102                 result = reference.get();
103             }
104         }
105 
106         return result;
107     }
108 
109     /**
110      * Creates and initializes the object managed by this {@code
111      * AtomicInitializer}. This method is called by {@link #get()} when the
112      * managed object is not available yet. An implementation can focus on the
113      * creation of the object. No synchronization is needed, as this is already
114      * handled by {@code get()}. As stated by the class comment, it is possible
115      * that this method is called multiple times.
116      *
117      * @return the managed data object
118      * @throws ConcurrentException if an error occurs during object creation
119      */
120     protected abstract T initialize() throws ConcurrentException;
121 }
