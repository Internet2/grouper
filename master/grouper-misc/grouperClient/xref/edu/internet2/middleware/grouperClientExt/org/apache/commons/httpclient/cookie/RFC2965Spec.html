124     protected void registerAttribHandler(
125             final String name, final CookieAttributeHandler handler) {
126         if (name == null) {
127             throw new IllegalArgumentException("Attribute name may not be null");
128         }
129         if (handler == null) {
130             throw new IllegalArgumentException("Attribute handler may not be null");
131         }
132         if (!this.attribHandlerList.contains(handler)) {
133             this.attribHandlerList.add(handler);
134         }
135         this.attribHandlerMap.put(name, handler);

172     /**
173      * Parses the Set-Cookie2 value into an array of <tt>Cookie</tt>s.
174      *
175      * <P>The syntax for the Set-Cookie2 response header is:
176      *
177      * <PRE>
178      * set-cookie      =    "Set-Cookie2:" cookies
179      * cookies         =    1#cookie
180      * cookie          =    NAME "=" VALUE * (";" cookie-av)
181      * NAME            =    attr
182      * VALUE           =    value
183      * cookie-av       =    "Comment" "=" value
184      *                 |    "CommentURL" "=" <"> http_URL <">
185      *                 |    "Discard"
186      *                 |    "Domain" "=" value
187      *                 |    "Max-Age" "=" value
188      *                 |    "Path" "=" value
189      *                 |    "Port" [ "=" <"> portlist <"> ]
190      *                 |    "Secure"
191      *                 |    "Version" "=" 1*DIGIT
192      * portlist        =       1#portnum
193      * portnum         =       1*DIGIT
194      * </PRE>
195      *
196      * @param host the host from which the <tt>Set-Cookie2</tt> value was
197      * received
198      * @param port the port from which the <tt>Set-Cookie2</tt> value was
199      * received
200      * @param path the path from which the <tt>Set-Cookie2</tt> value was
201      * received
202      * @param secure <tt>true</tt> when the <tt>Set-Cookie2</tt> value was
203      * received over secure conection
204      * @param header the <tt>Set-Cookie2</tt> <tt>Header</tt> received from the server
205      * @return an array of <tt>Cookie</tt>s parsed from the Set-Cookie2 value
206      * @throws MalformedCookieException if an exception occurs during parsing
207      */
208     public Cookie[] parse(
209             String host, int port, String path, boolean secure, final Header header)
210             throws MalformedCookieException {
211         LOG.trace("enter RFC2965.parse("
212                   + "String, int, String, boolean, Header)");
213 
214         if (header == null) {
215             throw new IllegalArgumentException("Header may not be null.");
216         }
217         if (header.getName() == null) {
218             throw new IllegalArgumentException("Header name may not be null.");
219         }
220 
221         if (header.getName().equalsIgnoreCase(SET_COOKIE2_KEY)) {
222             // parse cookie2 cookies
223             return parse(host, port, path, secure, header.getValue());
224         } else if (header.getName().equalsIgnoreCase(RFC2109Spec.SET_COOKIE_KEY)) {
225             // delegate parsing of old-style cookies to rfc2109Spec
226             return this.rfc2109.parse(host, port, path, secure, header.getValue());
227         } else {
228             throw new MalformedCookieException("Header name is not valid. " +
229                                                "RFC 2965 supports \"set-cookie\" " +
230                                                "and \"set-cookie2\" headers.");
231         }
232     }

234     /**
235      * @see #parse(String, int, String, boolean, edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Header)
236      */
237     public Cookie[] parse(String host, int port, String path,
238                           boolean secure, final String header)
239             throws MalformedCookieException {
240         LOG.trace("enter RFC2965Spec.parse("
241                   + "String, int, String, boolean, String)");
242 
243         // before we do anything, lets check validity of arguments
244         if (host == null) {
245             throw new IllegalArgumentException(
246                     "Host of origin may not be null");
247         }
248         if (host.trim().equals("")) {
249             throw new IllegalArgumentException(
250                     "Host of origin may not be blank");
251         }
252         if (port < 0) {
253             throw new IllegalArgumentException("Invalid port: " + port);
254         }
255         if (path == null) {
256             throw new IllegalArgumentException(
257                     "Path of origin may not be null.");
258         }
259         if (header == null) {
260             throw new IllegalArgumentException("Header may not be null.");
261         }
262 
263         if (path.trim().equals("")) {
264             path = PATH_DELIM;
265         }
266         host = getEffectiveHost(host);
267 
268         HeaderElement[] headerElements =
269                 HeaderElement.parseElements(header.toCharArray());
270 
271         List cookies = new LinkedList();
272         for (int i = 0; i < headerElements.length; i++) {
273             HeaderElement headerelement = headerElements[i];
274             Cookie2 cookie = null;
275             try {
276                 cookie = new Cookie2(host,
277                                     headerelement.getName(),
278                                     headerelement.getValue(),
279                                     path,
280                                     null,
281                                     false,
282                                     new int[] {port});
283             } catch (IllegalArgumentException ex) {
284                 throw new MalformedCookieException(ex.getMessage());
285             }
286             NameValuePair[] parameters = headerelement.getParameters();
287             // could be null. In case only a header element and no parameters.
288             if (parameters != null) {
289                 // Eliminate duplicate attribues. The first occurence takes precedence
290                 Map attribmap = new HashMap(parameters.length); 
291                 for (int j = parameters.length - 1; j >= 0; j--) {
292                     NameValuePair param = parameters[j];
293                     attribmap.put(param.getName().toLowerCase(), param);
294                 }
295                 for (Iterator it = attribmap.entrySet().iterator(); it.hasNext(); ) {
296                     Map.Entry entry = (Map.Entry) it.next();
297                     parseAttribute((NameValuePair) entry.getValue(), cookie);
298                 }
299             }
300             cookies.add(cookie);
301             // cycle through the parameters
302         }
303         return (Cookie[]) cookies.toArray(new Cookie[cookies.size()]);
304     }

306     /**
307      * Parse RFC 2965 specific cookie attribute and update the corresponsing
308      * {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} properties.
309      *
310      * @param attribute {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.NameValuePair} cookie attribute from the
311      * <tt>Set-Cookie2</tt> header.
312      * @param cookie {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} to be updated
313      * @throws MalformedCookieException if an exception occurs during parsing
314      */
315     public void parseAttribute(
316             final NameValuePair attribute, final Cookie cookie)
317             throws MalformedCookieException {
318         if (attribute == null) {
319             throw new IllegalArgumentException("Attribute may not be null.");
320         }
321         if (attribute.getName() == null) {
322             throw new IllegalArgumentException("Attribute Name may not be null.");
323         }
324         if (cookie == null) {
325             throw new IllegalArgumentException("Cookie may not be null.");
326         }
327         final String paramName = attribute.getName().toLowerCase();
328         final String paramValue = attribute.getValue();
329 
330         CookieAttributeHandler handler = findAttribHandler(paramName);
331         if (handler == null) {
332             // ignore unknown attribute-value pairs
333             if (LOG.isDebugEnabled())
334                 LOG.debug("Unrecognized cookie attribute: " +
335                           attribute.toString());
336         } else {
337             handler.parse(cookie, paramValue);
338         }
339     }

341     /**
342      * Performs RFC 2965 compliant {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} validation
343      *
344      * @param host the host from which the {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} was received
345      * @param port the port from which the {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} was received
346      * @param path the path from which the {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} was received
347      * @param secure <tt>true</tt> when the {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} was received using a
348      * secure connection
349      * @param cookie The cookie to validate
350      * @throws MalformedCookieException if an exception occurs during
351      * validation
352      */
353     public void validate(final String host, int port, final String path,
354                          boolean secure, final Cookie cookie)
355             throws MalformedCookieException {
356 
357         LOG.trace("enter RFC2965Spec.validate(String, int, String, "
358                   + "boolean, Cookie)");
359 
360         if (cookie instanceof Cookie2) {
361             if (cookie.getName().indexOf(' ') != -1) {
362                 throw new MalformedCookieException("Cookie name may not contain blanks");
363             }
364             if (cookie.getName().startsWith("$")) {
365                 throw new MalformedCookieException("Cookie name may not start with $");
366             }
367             CookieOrigin origin = new CookieOrigin(getEffectiveHost(host), port, path, secure); 
368             for (Iterator i = getAttribHandlerIterator(); i.hasNext(); ) {
369               CookieAttributeHandler handler = (CookieAttributeHandler) i.next();
370               handler.validate(cookie, origin);
371             }
372         } else {
373             // old-style cookies are validated according to the old rules
374             this.rfc2109.validate(host, port, path, secure, cookie);
375         }
376     }

378     /**
379      * Return <tt>true</tt> if the cookie should be submitted with a request
380      * with given attributes, <tt>false</tt> otherwise.
381      * @param host the host to which the request is being submitted
382      * @param port the port to which the request is being submitted (ignored)
383      * @param path the path to which the request is being submitted
384      * @param secure <tt>true</tt> if the request is using a secure connection
385      * @return true if the cookie matches the criterium
386      */
387     public boolean match(String host, int port, String path,
388                          boolean secure, final Cookie cookie) {
389 
390         LOG.trace("enter RFC2965.match("
391                   + "String, int, String, boolean, Cookie");
392         if (cookie == null) {
393             throw new IllegalArgumentException("Cookie may not be null");
394         }
395         if (cookie instanceof Cookie2) {
396             // check if cookie has expired
397             if (cookie.isPersistent() && cookie.isExpired()) {
398                 return false;
399             }
400             CookieOrigin origin = new CookieOrigin(getEffectiveHost(host), port, path, secure); 
401             for (Iterator i = getAttribHandlerIterator(); i.hasNext(); ) {
402                 CookieAttributeHandler handler = (CookieAttributeHandler) i.next();
403                 if (!handler.match(cookie, origin)) {
404                     return false;
405                 }
406             }
407             return true;
408         } else {
409             // old-style cookies are matched according to the old rules
410             return this.rfc2109.match(host, port, path, secure, cookie);
411         }
412     }

414     private void doFormatCookie2(final Cookie2 cookie, final StringBuffer buffer) {
415         String name = cookie.getName();
416         String value = cookie.getValue();
417         if (value == null) {
418             value = "";
419         }
420         this.formatter.format(buffer, new NameValuePair(name, value));
421         // format domain attribute
422         if (cookie.getDomain() != null && cookie.isDomainAttributeSpecified()) {
423             buffer.append("; ");
424             this.formatter.format(buffer, new NameValuePair("$Domain", cookie.getDomain()));
425         }
426         // format path attribute
427         if ((cookie.getPath() != null) && (cookie.isPathAttributeSpecified())) {
428             buffer.append("; ");
429             this.formatter.format(buffer, new NameValuePair("$Path", cookie.getPath()));
430         }
431         // format port attribute
432         if (cookie.isPortAttributeSpecified()) {
433             String portValue = "";
434             if (!cookie.isPortAttributeBlank()) {
435                 portValue = createPortAttribute(cookie.getPorts());
436             }
437             buffer.append("; ");
438             this.formatter.format(buffer, new NameValuePair("$Port", portValue));
439         }
440     }

442     /**
443      * Return a string suitable for sending in a <tt>"Cookie"</tt> header as
444      * defined in RFC 2965
445      * @param cookie a {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie} to be formatted as string
446      * @return a string suitable for sending in a <tt>"Cookie"</tt> header.
447      */
448     public String formatCookie(final Cookie cookie) {
449         LOG.trace("enter RFC2965Spec.formatCookie(Cookie)");
450 
451         if (cookie == null) {
452             throw new IllegalArgumentException("Cookie may not be null");
453         }
454         if (cookie instanceof Cookie2) {
455             Cookie2 cookie2 = (Cookie2) cookie;
456             int version = cookie2.getVersion();
457             final StringBuffer buffer = new StringBuffer();
458             this.formatter.format(buffer, new NameValuePair("$Version", Integer.toString(version)));
459             buffer.append("; ");
460             doFormatCookie2(cookie2, buffer);
461             return buffer.toString();
462         } else {
463             // old-style cookies are formatted according to the old rules
464             return this.rfc2109.formatCookie(cookie);
465         }
466     }

468     /**
469      * Create a RFC 2965 compliant <tt>"Cookie"</tt> header value containing all
470      * {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie}s suitable for
471      * sending in a <tt>"Cookie"</tt> header
472      * @param cookies an array of {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie}s to be formatted
473      * @return a string suitable for sending in a Cookie header.
474      */
475     public String formatCookies(final Cookie[] cookies) {
476         LOG.trace("enter RFC2965Spec.formatCookieHeader(Cookie[])");
477 
478         if (cookies == null) {
479             throw new IllegalArgumentException("Cookies may not be null");
480         }
481         // check if cookies array contains a set-cookie (old style) cookie
482         boolean hasOldStyleCookie = false;
483         int version = -1;
484         for (int i = 0; i < cookies.length; i++) {
485             Cookie cookie = cookies[i];
486             if (!(cookie instanceof Cookie2)) {
487                 hasOldStyleCookie = true;
488                 break;
489             }
490             if (cookie.getVersion() > version) {
491                 version = cookie.getVersion();
492             }
493         }
494         if (version < 0) {
495             version = 0;
496         }
497         if (hasOldStyleCookie || version < 1) {
498             // delegate old-style cookie formatting to rfc2109Spec
499             return this.rfc2109.formatCookies(cookies);
500         }
501         // Arrange cookies by path
502         Arrays.sort(cookies, PATH_COMPOARATOR);
503         
504         final StringBuffer buffer = new StringBuffer();
505         // format cookie version
506         this.formatter.format(buffer, new NameValuePair("$Version", Integer.toString(version)));
507         for (int i = 0; i < cookies.length; i++) {
508             buffer.append("; ");
509             Cookie2 cookie = (Cookie2) cookies[i];
510             // format cookie attributes
511             doFormatCookie2(cookie, buffer);
512         }
513         return buffer.toString();
514     }

516     /**
517      * Retrieves valid Port attribute value for the given ports array.
518      * e.g. "8000,8001,8002"
519      *
520      * @param ports int array of ports
521      */
522     private String createPortAttribute(int[] ports) {
523         StringBuffer portValue = new StringBuffer();
524         for (int i = 0, len = ports.length; i < len; i++) {
525             if (i > 0) {
526                 portValue.append(",");
527             }
528             portValue.append(ports[i]);
529         }
530         return portValue.toString();
531     }

533     /**
534      * Parses the given Port attribute value (e.g. "8000,8001,8002")
535      * into an array of ports.
536      *
537      * @param portValue port attribute value
538      * @return parsed array of ports
539      * @throws MalformedCookieException if there is a problem in
540      *          parsing due to invalid portValue.
541      */
542     private int[] parsePortAttribute(final String portValue)
543             throws MalformedCookieException {
544         StringTokenizer st = new StringTokenizer(portValue, ",");
545         int[] ports = new int[st.countTokens()];
546         try {
547             int i = 0;
548             while(st.hasMoreTokens()) {
549                 ports[i] = Integer.parseInt(st.nextToken().trim());
550                 if (ports[i] < 0) {
551                   throw new MalformedCookieException ("Invalid Port attribute.");
552                 }
553                 ++i;
554             }
555         } catch (NumberFormatException e) {
556             throw new MalformedCookieException ("Invalid Port "
557                                                 + "attribute: " + e.getMessage());
558         }
559         return ports;
560     }

562     /**
563      * Gets 'effective host name' as defined in RFC 2965.
564      * <p>
565      * If a host name contains no dots, the effective host name is
566      * that name with the string .local appended to it.  Otherwise
567      * the effective host name is the same as the host name.  Note
568      * that all effective host names contain at least one dot.
569      *
570      * @param host host name where cookie is received from or being sent to.
571      * @return
572      */
573     private static String getEffectiveHost(final String host) {
574         String effectiveHost = host.toLowerCase();
575         if (host.indexOf('.') < 0) {
576             effectiveHost += ".local";
577         }
578         return effectiveHost;
579     }

581     /**
582      * Performs domain-match as defined by the RFC2965.
583      * <p>
584      * Host A's name domain-matches host B's if
585      * <ol>
586      *   <ul>their host name strings string-compare equal; or</ul>
587      *   <ul>A is a HDN string and has the form NB, where N is a non-empty
588      *       name string, B has the form .B', and B' is a HDN string.  (So,
589      *       x.y.com domain-matches .Y.com but not Y.com.)</ul>
590      * </ol>
591      *
592      * @param host host name where cookie is received from or being sent to.
593      * @param domain The cookie domain attribute.
594      * @return true if the specified host matches the given domain.
595      */
596     public boolean domainMatch(String host, String domain) {
597         boolean match = host.equals(domain)
598                         || (domain.startsWith(".") && host.endsWith(domain));
599 
600         return match;
601     }

603     /**
604      * Returns <tt>true</tt> if the given port exists in the given
605      * ports list.
606      *
607      * @param port port of host where cookie was received from or being sent to.
608      * @param ports port list
609      * @return true returns <tt>true</tt> if the given port exists in
610      *         the given ports list; <tt>false</tt> otherwise.
611      */
612     private boolean portMatch(int port, int[] ports) {
613         boolean portInList = false;
614         for (int i = 0, len = ports.length; i < len; i++) {
615             if (port == ports[i]) {
616                 portInList = true;
617                 break;
618             }
619         }
620         return portInList;
621     }

629         /**
630          * Parse cookie path attribute.
631          */
632         public void parse(final Cookie cookie, final String path)
633                 throws MalformedCookieException {
634             if (cookie == null) {
635                 throw new IllegalArgumentException("Cookie may not be null");
636             }
637             if (path == null) {
638                 throw new MalformedCookieException(
639                         "Missing value for path attribute");
640             }
641             if (path.trim().equals("")) {
642                 throw new MalformedCookieException(
643                         "Blank value for path attribute");
644             }
645             cookie.setPath(path);
646             cookie.setPathAttributeSpecified(true);
647         }

653         public void validate(final Cookie cookie, final CookieOrigin origin)

659                 throw new IllegalArgumentException("Cookie origin may not be null");
660             }
661             String path = origin.getPath();
662             if (path == null) {
663                 throw new IllegalArgumentException(
664                         "Path of origin host may not be null.");
665             }
666             if (cookie.getPath() == null) {
667                 throw new MalformedCookieException("Invalid cookie state: " +
668                                                    "path attribute is null.");
669             }
670             if (path.trim().equals("")) {
671                 path = PATH_DELIM;
672             }
673 
674             if (!pathMatch(path, cookie.getPath())) {
675                 throw new MalformedCookieException(
676                         "Illegal path attribute \"" + cookie.getPath()
677                         + "\". Path of origin: \"" + path + "\"");
678             }
679         }

683          * prefix of the request-URI (case-sensitive matching).
684          */
685         public boolean match(final Cookie cookie, final CookieOrigin origin) {
686             if (cookie == null) {
687                 throw new IllegalArgumentException("Cookie may not be null");
688             }
689             if (origin == null) {
690                 throw new IllegalArgumentException("Cookie origin may not be null");
691             }
692             String path = origin.getPath();

695                 return false;

698                 path = PATH_DELIM;
699             }

708     /**
709      * <tt>"Domain"</tt> cookie attribute handler for RFC 2965 cookie spec.
710      */
711     private class Cookie2DomainAttributeHandler
712             implements CookieAttributeHandler {
713 
714         /**
715          * Parse cookie domain attribute.
716          */
717         public void parse(final Cookie cookie, String domain)
718                 throws MalformedCookieException {
719             if (cookie == null) {
720                 throw new IllegalArgumentException("Cookie may not be null");
721             }
722             if (domain == null) {
723                 throw new MalformedCookieException(
724                         "Missing value for domain attribute");
725             }
726             if (domain.trim().equals("")) {
727                 throw new MalformedCookieException(
728                         "Blank value for domain attribute");
729             }
730             domain = domain.toLowerCase();
731             if (!domain.startsWith(".")) {
732                 // Per RFC 2965 section 3.2.2
733                 // "... If an explicitly specified value does not start with
734                 // a dot, the user agent supplies a leading dot ..."
735                 // That effectively implies that the domain attribute 
736                 // MAY NOT be an IP address of a host name
737                 domain = "." + domain;
738             }
739             cookie.setDomain(domain);
740             cookie.setDomainAttributeSpecified(true);
741         }
742 
743         /**
744          * Validate cookie domain attribute.
745          */
746         public void validate(final Cookie cookie, final CookieOrigin origin)
747                 throws MalformedCookieException {
748             if (cookie == null) {
749                 throw new IllegalArgumentException("Cookie may not be null");
750             }
751             if (origin == null) {
752                 throw new IllegalArgumentException("Cookie origin may not be null");
753             }
754             String host = origin.getHost().toLowerCase();
755             if (cookie.getDomain() == null) {
756                 throw new MalformedCookieException("Invalid cookie state: " +
757                                                    "domain not specified");
758             }
759             String cookieDomain = cookie.getDomain().toLowerCase();
760 
761             if (cookie.isDomainAttributeSpecified()) {
762                 // Domain attribute must start with a dot
763                 if (!cookieDomain.startsWith(".")) {
764                     throw new MalformedCookieException("Domain attribute \"" +
765                         cookie.getDomain() + "\" violates RFC 2109: domain must start with a dot");
766                 }
767 
768                 // Domain attribute must contain atleast one embedded dot,
769                 // or the value must be equal to .local.
770                 int dotIndex = cookieDomain.indexOf('.', 1);
771                 if (((dotIndex < 0) || (dotIndex == cookieDomain.length() - 1))
772                     && (!cookieDomain.equals(".local"))) {
773                     throw new MalformedCookieException(
774                             "Domain attribute \"" + cookie.getDomain()
775                             + "\" violates RFC 2965: the value contains no embedded dots "
776                             + "and the value is not .local");
777                 }
778 
779                 // The effective host name must domain-match domain attribute.
780                 if (!domainMatch(host, cookieDomain)) {
781                     throw new MalformedCookieException(
782                             "Domain attribute \"" + cookie.getDomain()
783                             + "\" violates RFC 2965: effective host name does not "
784                             + "domain-match domain attribute.");
785                 }
786 
787                 // effective host name minus domain must not contain any dots
788                 String effectiveHostWithoutDomain = host.substring(
789                         0, host.length() - cookieDomain.length());
790                 if (effectiveHostWithoutDomain.indexOf('.') != -1) {
791                     throw new MalformedCookieException("Domain attribute \""
792                                                        + cookie.getDomain() + "\" violates RFC 2965: "
793                                                        + "effective host minus domain may not contain any dots");
794                 }
795             } else {
796                 // Domain was not specified in header. In this case, domain must
797                 // string match request host (case-insensitive).
798                 if (!cookie.getDomain().equals(host)) {
799                     throw new MalformedCookieException("Illegal domain attribute: \""
800                                                        + cookie.getDomain() + "\"."
801                                                        + "Domain of origin: \""
802                                                        + host + "\"");
803                 }
804             }
805         }
806 
807         /**
808          * Match cookie domain attribute.
809          */
810         public boolean match(final Cookie cookie, final CookieOrigin origin) {
811             if (cookie == null) {
812                 throw new IllegalArgumentException("Cookie may not be null");
813             }
814             if (origin == null) {
815                 throw new IllegalArgumentException("Cookie origin may not be null");
816             }
817             String host = origin.getHost().toLowerCase();
818             String cookieDomain = cookie.getDomain();
819 
820             // The effective host name MUST domain-match the Domain
821             // attribute of the cookie.
822             if (!domainMatch(host, cookieDomain)) {
823                 return false;
824             }
825             // effective host name minus domain must not contain any dots
826             String effectiveHostWithoutDomain = host.substring(
827                     0, host.length() - cookieDomain.length());
828             if (effectiveHostWithoutDomain.indexOf('.') != -1) {
829                 return false;
830             }
831             return true;
832         }
833

837      * <tt>"Port"</tt> cookie attribute handler for RFC 2965 cookie spec.
838      */
839     private class Cookie2PortAttributeHandler
840             implements CookieAttributeHandler {
841 
842         /**
843          * Parse cookie port attribute.
844          */
845         public void parse(final Cookie cookie, final String portValue)
846                 throws MalformedCookieException {
847             if (cookie == null) {
848                 throw new IllegalArgumentException("Cookie may not be null");
849             }
850             if (cookie instanceof Cookie2) {
851                 Cookie2 cookie2 = (Cookie2) cookie;
852                 if ((portValue == null) || (portValue.trim().equals(""))) {
853                     // If the Port attribute is present but has no value, the
854                     // cookie can only be sent to the request-port.
855                     // Since the default port list contains only request-port, we don't
856                     // need to do anything here.
857                     cookie2.setPortAttributeBlank(true);
858                 } else {
859                     int[] ports = parsePortAttribute(portValue);
860                     cookie2.setPorts(ports);
861                 }
862                 cookie2.setPortAttributeSpecified(true);
863             }
864         }
865 
866         /**
867          * Validate cookie port attribute. If the Port attribute was specified
868          * in header, the request port must be in cookie's port list.
869          */
870         public void validate(final Cookie cookie, final CookieOrigin origin)
871                 throws MalformedCookieException {
872             if (cookie == null) {
873                 throw new IllegalArgumentException("Cookie may not be null");
874             }
875             if (origin == null) {
876                 throw new IllegalArgumentException("Cookie origin may not be null");
877             }
878             if (cookie instanceof Cookie2) {
879                 Cookie2 cookie2 = (Cookie2) cookie;
880                 int port = origin.getPort();
881                 if (cookie2.isPortAttributeSpecified()) {
882                     if (!portMatch(port, cookie2.getPorts())) {
883                         throw new MalformedCookieException(
884                                 "Port attribute violates RFC 2965: "
885                                 + "Request port not found in cookie's port list.");
886                     }
887                 }
888             }
889         }
890 
891         /**
892          * Match cookie port attribute. If the Port attribute is not specified
893          * in header, the cookie can be sent to any port. Otherwise, the request port
894          * must be in the cookie's port list.
895          */
896         public boolean match(final Cookie cookie, final CookieOrigin origin) {
897             if (cookie == null) {
898                 throw new IllegalArgumentException("Cookie may not be null");
899             }
900             if (origin == null) {
901                 throw new IllegalArgumentException("Cookie origin may not be null");
902             }
903             if (cookie instanceof Cookie2) {
904                 Cookie2 cookie2 = (Cookie2) cookie;
905                 int port = origin.getPort();
906                 if (cookie2.isPortAttributeSpecified()) {
907                     if (cookie2.getPorts() == null) {
908                         LOG.warn("Invalid cookie state: port not specified");
909                         return false;
910                     }
911                     if (!portMatch(port, cookie2.getPorts())) {
912                         return false;
913                     }
914                 }
915                 return true;
916             } else {
917                 return false;
918             }
919         }
