34  package edu.internet2.middleware.grouperClientExt.org.apache.commons.logging.impl;
35  
36  import java.lang.ref.ReferenceQueue;
37  import java.lang.ref.WeakReference;
38  import java.util.Collection;
39  import java.util.Enumeration;
40  import java.util.HashSet;
41  import java.util.Hashtable;
42  import java.util.Iterator;
43  import java.util.Map;
44  import java.util.Set;
45  
46  /**
47   * <p>Implementation of <code>Hashtable</code> that uses <code>WeakReference</code>'s
48   * to hold its keys thus allowing them to be reclaimed by the garbage collector.
49   * The associated values are retained using strong references.</p>
50   *
51   * <p>This class follows the symantics of <code>Hashtable</code> as closely as
52   * possible. It therefore does not accept null values or keys.</p>
53   *
54   * <p><strong>Note:</strong>
55   * This is <em>not</em> intended to be a general purpose hash table replacement.
56   * This implementation is also tuned towards a particular purpose: for use as a replacement
57   * for <code>Hashtable</code> in <code>LogFactory</code>. This application requires
58   * good liveliness for <code>get</code> and <code>put</code>. Various tradeoffs
59   * have been made with this in mind.
60   * </p>
61   * <p>
62   * <strong>Usage:</strong> typical use case is as a drop-in replacement 
63   * for the <code>Hashtable</code> used in <code>LogFactory</code> for J2EE enviroments
64   * running 1.3+ JVMs. Use of this class <i>in most cases</i> (see below) will
65   * allow classloaders to be collected by the garbage collector without the need 
66   * to call {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.logging.LogFactory#release(ClassLoader) LogFactory.release(ClassLoader)}.
67   * </p>
68   *
69   * <p><code>org.apache.commons.logging.LogFactory</code> checks whether this class
70   * can be supported by the current JVM, and if so then uses it to store
71   * references to the <code>LogFactory</code> implementationd it loads 
72   * (rather than using a standard Hashtable instance). 
73   * Having this class used instead of <code>Hashtable</code> solves
74   * certain issues related to dynamic reloading of applications in J2EE-style
75   * environments. However this class requires java 1.3 or later (due to its use
76   * of <code>java.lang.ref.WeakReference</code> and associates).
77   * And by the way, this extends <code>Hashtable</code> rather than <code>HashMap</code>
78   * for backwards compatibility reasons. See the documentation
79   * for method <code>LogFactory.createFactoryStore</code> for more details.</p>
80   *
81   * <p>The reason all this is necessary is due to a issue which
82   * arises during hot deploy in a J2EE-like containers. 
83   * Each component running in the container owns one or more classloaders; when
84   * the component loads a LogFactory instance via the component classloader
85   * a reference to it gets stored in the static LogFactory.factories member,
86   * keyed by the component's classloader so different components don't
87   * stomp on each other. When the component is later unloaded, the container
88   * sets the component's classloader to null with the intent that all the 
89   * component's classes get garbage-collected. However there's still a
90   * reference to the component's classloader from a key in the "global"
91   * <code>LogFactory</code>'s factories member! If <code>LogFactory.release()</code>
92   * is called whenever component is unloaded, the classloaders will be correctly
93   * garbage collected; this <i>should</i> be done by any container that 
94   * bundles commons-logging by default. However, holding the classloader
95   * references weakly ensures that the classloader will be garbage collected
96   * without the container performing this step. </p>
97   *
98   * <p>
99   * <strong>Limitations:</strong>
100  * There is still one (unusual) scenario in which a component will not 
101  * be correctly unloaded without an explicit release. Though weak references
102  * are used for its keys, it is necessary to use strong references for its values.
103  * </p>
104  * 
105  * <p> If the abstract class <code>LogFactory</code> is 
106  * loaded by the container classloader but a subclass of 
107  * <code>LogFactory</code> [LogFactory1] is loaded by the component's 
108  * classloader and an instance stored in the static map associated with the
109  * base LogFactory class, then there is a strong reference from the LogFactory
110  * class to the LogFactory1 instance (as normal) and a strong reference from
111  * the LogFactory1 instance to the component classloader via
112  * <code>getClass().getClassLoader()</code>. This chain of references will prevent 
113  * collection of the child classloader.</p>
114  *
115  * <p>
116  * Such a situation occurs when the commons-logging.jar is
117  * loaded by a parent classloader (e.g. a server level classloader in a
118  * servlet container) and a custom <code>LogFactory</code> implementation is
119  * loaded by a child classloader (e.g. a web app classloader).</p>
120  * 
121  * <p>To avoid this scenario, ensure
122  * that any custom LogFactory subclass is loaded by the same classloader as 
123  * the base <code>LogFactory</code>. Creating custom LogFactory subclasses is,
124  * however, rare. The standard LogFactoryImpl class should be sufficient
125  * for most or all users.</p>
126  *
127  *
128  * @author Brian Stansberry
129  * 
130  * @since 1.1
131  */
132 public final class WeakHashtable extends Hashtable {
133 
134     /** 
135      * The maximum number of times put() or remove() can be called before
136      * the map will be purged of all cleared entries.
137      */
138     private static final int MAX_CHANGES_BEFORE_PURGE = 100;
139     
140     /** 
141      * The maximum number of times put() or remove() can be called before
142      * the map will be purged of one cleared entry.
143      */
144     private static final int PARTIAL_PURGE_COUNT     = 10;
145     
146     /* ReferenceQueue we check for gc'd keys */
147     private ReferenceQueue queue = new ReferenceQueue();
148     /* Counter used to control how often we purge gc'd entries */
149     private int changeCount = 0;
150     
151     /**
152      * Constructs a WeakHashtable with the Hashtable default
153      * capacity and load factor.
154      */
155     public WeakHashtable() {}
156     
157     
158     /**
159      *@see Hashtable
160      */
161     public boolean containsKey(Object key) {
162         // purge should not be required
163         Referenced referenced = new Referenced(key);
164         return super.containsKey(referenced);
165     }
166     
167     /**
168      *@see Hashtable
169      */
170     public Enumeration elements() {
171         purge();
172         return super.elements();
173     }
174     
175     /**
176      *@see Hashtable
177      */
178     public Set entrySet() {
179         purge();
180         Set referencedEntries = super.entrySet();
181         Set unreferencedEntries = new HashSet();
182         for (Iterator it=referencedEntries.iterator(); it.hasNext();) {
183             Map.Entry entry = (Map.Entry) it.next();
184             Referenced referencedKey = (Referenced) entry.getKey();
185             Object key = referencedKey.getValue();
186             Object value = entry.getValue();
187             if (key != null) {
188                 Entry dereferencedEntry = new Entry(key, value);
189                 unreferencedEntries.add(dereferencedEntry);
190             }
191         }
192         return unreferencedEntries;
193     }
194     
195     /**
196      *@see Hashtable
197      */
198     public Object get(Object key) {
199         // for performance reasons, no purge
200         Referenced referenceKey = new Referenced(key);
201         return super.get(referenceKey);
202     }
203     
204     /**
205      *@see Hashtable
206      */
207     public Enumeration keys() {
208         purge();
209         final Enumeration enumer = super.keys();
210         return new Enumeration() {
211             public boolean hasMoreElements() {
212                 return enumer.hasMoreElements();
213             }
214             public Object nextElement() {
215                  Referenced nextReference = (Referenced) enumer.nextElement();
216                  return nextReference.getValue();
217             }
218         };
219     }
220     
221         
222     /**
223      *@see Hashtable
224      */
225     public Set keySet() {
226         purge();
227         Set referencedKeys = super.keySet();
228         Set unreferencedKeys = new HashSet();
229         for (Iterator it=referencedKeys.iterator(); it.hasNext();) {
230             Referenced referenceKey = (Referenced) it.next();
231             Object keyValue = referenceKey.getValue();
232             if (keyValue != null) {
233                 unreferencedKeys.add(keyValue);
234             }
235         }
236         return unreferencedKeys;
237     }
238     
239     /**
240      *@see Hashtable
241      */    
242     public Object put(Object key, Object value) {
243         // check for nulls, ensuring symantics match superclass
244         if (key == null) {
245             throw new NullPointerException("Null keys are not allowed");
246         }
247         if (value == null) {
248             throw new NullPointerException("Null values are not allowed");
249         }
250 
251         // for performance reasons, only purge every 
252         // MAX_CHANGES_BEFORE_PURGE times
253         if (changeCount++ > MAX_CHANGES_BEFORE_PURGE) {
254             purge();
255             changeCount = 0;
256         }
257         // do a partial purge more often
258         else if ((changeCount % PARTIAL_PURGE_COUNT) == 0) {
259             purgeOne();
260         }
261         
262         Referenced keyRef = new Referenced(key, queue);
263         return super.put(keyRef, value);
264     }
265     
266     /**
267      *@see Hashtable
268      */    
269     public void putAll(Map t) {
270         if (t != null) {
271             Set entrySet = t.entrySet();
272             for (Iterator it=entrySet.iterator(); it.hasNext();) {
273                 Map.Entry entry = (Map.Entry) it.next();
274                 put(entry.getKey(), entry.getValue());
275             }
276         }
277     }
278     
279     /**
280      *@see Hashtable
281      */      
282     public Collection values() {
283         purge();
284         return super.values();
285     }
286     
287     /**
288      *@see Hashtable
289      */     
290     public Object remove(Object key) {
291         // for performance reasons, only purge every 
292         // MAX_CHANGES_BEFORE_PURGE times
293         if (changeCount++ > MAX_CHANGES_BEFORE_PURGE) {
294             purge();
295             changeCount = 0;
296         }
297         // do a partial purge more often
298         else if ((changeCount % PARTIAL_PURGE_COUNT) == 0) {
299             purgeOne();
300         }
301         return super.remove(new Referenced(key));
302     }
303     
304     /**
305      *@see Hashtable
306      */    
307     public boolean isEmpty() {
308         purge();
309         return super.isEmpty();
310     }
311     
312     /**
313      *@see Hashtable
314      */    
315     public int size() {
316         purge();
317         return super.size();
318     }
319     
320     /**
321      *@see Hashtable
322      */        
323     public String toString() {
324         purge();
325         return super.toString();
326     }
327     
328     /**
329      * @see Hashtable
330      */
331     protected void rehash() {
332         // purge here to save the effort of rehashing dead entries
333         purge();
334         super.rehash();
335     }
336     
337     /**
338      * Purges all entries whose wrapped keys
339      * have been garbage collected.
340      */
341     private void purge() {
342         synchronized (queue) {
343             WeakKey key;
344             while ((key = (WeakKey) queue.poll()) != null) {
345                 super.remove(key.getReferenced());
346             }
347         }
348     }
349     
350     /**
351      * Purges one entry whose wrapped key 
352      * has been garbage collected.
353      */
354     private void purgeOne() {
355         
356         synchronized (queue) {
357             WeakKey key = (WeakKey) queue.poll();
358             if (key != null) {
359                 super.remove(key.getReferenced());
360             }
361         }
362     }
363     
364     /** Entry implementation */
365     private final static class Entry implements Map.Entry {
366     
367         private final Object key;
368         private final Object value;
369         
370         private Entry(Object key, Object value) {
371             this.key = key;
372             this.value = value;
373         }
374     
375         public boolean equals(Object o) {
376             boolean result = false;
377             if (o != null && o instanceof Map.Entry) {
378                 Map.Entry entry = (Map.Entry) o;
379                 result =    (getKey()==null ?
380                                             entry.getKey() == null : 
381                                             getKey().equals(entry.getKey()))
382                             &&
383                             (getValue()==null ?
384                                             entry.getValue() == null : 
385                                             getValue().equals(entry.getValue()));
386             }
387             return result;
388         } 
389         
390         public int hashCode() {
391 
392             return (getKey()==null ? 0 : getKey().hashCode()) ^
393                 (getValue()==null ? 0 : getValue().hashCode());
394         }
395 
396         public Object setValue(Object value) {
397             throw new UnsupportedOperationException("Entry.setValue is not supported.");
398         }
399         
400         public Object getValue() {
401             return value;
402         }
403         
404         public Object getKey() {
405             return key;
406         }
407     }
408     
409     
410     /** Wrapper giving correct symantics for equals and hashcode */
411     private final static class Referenced {
412         
413         private final WeakReference reference;
414         private final int           hashCode;
415 
416         /**
417          * 
418          * @throws NullPointerException if referant is <code>null</code>
419          */        
420         private Referenced(Object referant) {
421             reference = new WeakReference(referant);
422             // Calc a permanent hashCode so calls to Hashtable.remove()
423             // work if the WeakReference has been cleared
424             hashCode  = referant.hashCode();
425         }
426         
427         /**
428          * 
429          * @throws NullPointerException if key is <code>null</code>
430          */
431         private Referenced(Object key, ReferenceQueue queue) {
432             reference = new WeakKey(key, queue, this);
433             // Calc a permanent hashCode so calls to Hashtable.remove()
434             // work if the WeakReference has been cleared
435             hashCode  = key.hashCode();
436 
437         }
438         
439         public int hashCode() {
440             return hashCode;
441         }
442         
443         private Object getValue() {
444             return reference.get();
445         }
446         
447         public boolean equals(Object o) {
448             boolean result = false;
449             if (o instanceof Referenced) {
450                 Referenced otherKey = (Referenced) o;
451                 Object thisKeyValue = getValue();
452                 Object otherKeyValue = otherKey.getValue();
453                 if (thisKeyValue == null) {                     
454                     result = (otherKeyValue == null);
455                     
456                     // Since our hashcode was calculated from the original
457                     // non-null referant, the above check breaks the 
458                     // hashcode/equals contract, as two cleared Referenced
459                     // objects could test equal but have different hashcodes.
460                     // We can reduce (not eliminate) the chance of this
461                     // happening by comparing hashcodes.
462                     if (result == true) {
463                         result = (this.hashCode() == otherKey.hashCode());
464                     }
465                     // In any case, as our c'tor does not allow null referants
466                     // and Hashtable does not do equality checks between 
467                     // existing keys, normal hashtable operations should never 
468                     // result in an equals comparison between null referants
469                 }
470                 else
471                 {
472                     result = thisKeyValue.equals(otherKeyValue);
473                 }
474             }
475             return result;
476         }
477     }
478     
479     /**
480      * WeakReference subclass that holds a hard reference to an
481      * associated <code>value</code> and also makes accessible
482      * the Referenced object holding it.
483      */
484     private final static class WeakKey extends WeakReference {
485 
486         private final Referenced referenced;
487         
488         private WeakKey(Object key, 
489                         ReferenceQueue queue,
490                         Referenced referenced) {
491             super(key, queue);
492             this.referenced = referenced;
493         }
494         
495         private Referenced getReferenced() {
496             return referenced;
497         }
498      }
