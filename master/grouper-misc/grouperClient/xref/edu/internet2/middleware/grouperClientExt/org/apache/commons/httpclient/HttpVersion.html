46  package edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient;
47  
48  /**
49   *  <p>HTTP version, as specified in RFC 2616.</p>
50   *  <p>
51   *  HTTP uses a "&lt;major&gt;.&lt;minor&gt;" numbering scheme to indicate
52   *  versions of the protocol. The protocol versioning policy is intended to
53   *  allow the sender to indicate the format of a message and its capacity for
54   *  understanding further HTTP communication, rather than the features
55   *  obtained via that communication. No change is made to the version
56   *  number for the addition of message components which do not affect
57   *  communication behavior or which only add to extensible field values.
58   *  The &lt;minor&gt; number is incremented when the changes made to the
59   *  protocol add features which do not change the general message parsing
60   *  algorithm, but which may add to the message semantics and imply
61   *  additional capabilities of the sender. The &lt;major&gt; number is
62   *  incremented when the format of a message within the protocol is
63   *  changed. See RFC 2145 [36] for a fuller explanation.
64   *  </p>
65   *  <p>
66   *  The version of an HTTP message is indicated by an HTTP-Version field
67   *  in the first line of the message.
68   *  </p>
69   *  <pre>
70   *     HTTP-Version   = "HTTP" "/" 1*DIGIT "." 1*DIGIT
71   *  </pre>
72   *  <p>
73   *   Note that the major and minor numbers MUST be treated as separate
74   *   integers and that each MAY be incremented higher than a single digit.
75   *   Thus, HTTP/2.4 is a lower version than HTTP/2.13, which in turn is
76   *   lower than HTTP/12.3. Leading zeros MUST be ignored by recipients and
77   *   MUST NOT be sent.
78   *  </p>
79   * 
80   * @author <a href="mailto:oleg@ural.ru">Oleg Kalnichevski</a>
81   * 
82   * @version $Revision: 1.1 $ $Date: 2008-11-30 10:57:19 $
83   *
84   * @since 3.0
85   */
86  public class HttpVersion implements Comparable {
87  
88      /** Major version number of the HTTP protocol */
89      private int major = 0;
90  
91      /** Minor version number of the HTTP protocol */
92      private int minor = 0;
93  
94      /** HTTP protocol version 0.9 */
95      public static final HttpVersion HTTP_0_9 = new HttpVersion(0, 9);  
96  
97      /** HTTP protocol version 1.0 */
98      public static final HttpVersion HTTP_1_0 = new HttpVersion(1, 0);  
99  
100     /** HTTP protocol version 1.1 */
101     public static final HttpVersion HTTP_1_1 = new HttpVersion(1, 1);  
102     
103     /**
104      * Create an HTTP protocol version designator.
105      *
106      * @param major   the major version number of the HTTP protocol
107      * @param minor   the minor version number of the HTTP protocol
108      * 
109      * @throws IllegalArgumentException if either major or minor version number is negative
110      */
111     public HttpVersion(int major, int minor) {
112         if (major < 0) {
113             throw new IllegalArgumentException("HTTP major version number may not be negative");
114         }
115         this.major = major;
116         if (minor < 0) {
117             throw new IllegalArgumentException("HTTP minor version number may not be negative");
118         }
119         this.minor = minor;
120     }
121     
122     /**
123      * Returns the major version number of the HTTP protocol.
124      * 
125      * @return the major version number.
126      */
127     public int getMajor() {
128         return major;
129     }
130 
131     /**
132      * Returns the minor version number of the HTTP protocol.
133      * 
134      * @return the minor version number.
135      */
136     public int getMinor() {
137         return minor;
138     }
139 
140     /**
141      * @see java.lang.Object#hashCode()
142      */
143     public int hashCode() {
144         return this.major * 100000 + this.minor;
145     }
146         
147     /**
148      * @see java.lang.Object#equals(java.lang.Object)
149      */
150     public boolean equals(Object obj) {
151         if (this == obj) {
152             return true;
153         }
154         if (!(obj instanceof HttpVersion)) {
155             return false;
156         }
157         return equals((HttpVersion)obj);  
158     }
159 
160     /**
161      * Compares this HTTP protocol version with another one.
162      * 
163      * @param anotherVer the version to be compared with.
164      *  
165      * @return a negative integer, zero, or a positive integer as this version is less than, 
166      *    equal to, or greater than the specified version.     
167      */
168     public int compareTo(HttpVersion anotherVer) {
169         if (anotherVer == null) {
170             throw new IllegalArgumentException("Version parameter may not be null"); 
171         }
172         int delta = getMajor() - anotherVer.getMajor();
173         if (delta == 0) {
174             delta = getMinor() - anotherVer.getMinor();
175         }
176         return delta;
177     }
178 
179     /**
180      * @see java.lang.Comparable#compareTo(java.lang.Object)
181      */
182     public int compareTo(Object o) {
183         return compareTo((HttpVersion)o);
184     }
185 
186     /**
187      * Test if the HTTP protocol version is equal to the given number.
188      * 
189      * @return <tt>true</tt> if HTTP protocol version is given to the given number, 
190      *         <tt>false</tt> otherwise.
191      */
192     public boolean equals(HttpVersion version) {
193         return compareTo(version) == 0;  
194     }
195 
196     /**
197      * Test if the HTTP protocol version is greater or equal to the given number.
198      * 
199      * @return <tt>true</tt> if HTTP protocol version is greater or equal given to the 
200      *         given number, <tt>false</tt> otherwise.
201      */
202     public boolean greaterEquals(HttpVersion version) {
203         return compareTo(version) >= 0;
204     }
205 
206     /**
207      * Test if the HTTP protocol version is less or equal to the given number.
208      * 
209      * @return <tt>true</tt> if HTTP protocol version is less or equal to given to the 
210      *         given number, <tt>false</tt> otherwise.
211      */
212     public boolean lessEquals(HttpVersion version) {
213         return compareTo(version) <= 0;
214     }
215 
216     /**
217      * @see java.lang.Object#toString()
218      */
219     public String toString() {
220         StringBuffer buffer = new StringBuffer();
221         buffer.append("HTTP/"); 
222         buffer.append(this.major); 
223         buffer.append('.'); 
224         buffer.append(this.minor); 
225         return buffer.toString();
226     }
227 
228     /**
229      * Parses the textual representation of the given HTTP protocol version.
230      * 
231      * @return HTTP protocol version.
232      * 
233      * @throws ProtocolException if the string is not a valid HTTP protocol version. 
234      */
235     public static HttpVersion parse(final String s) throws ProtocolException {
236         if (s == null) {
237             throw new IllegalArgumentException("String may not be null");
238         }
239         if (!s.startsWith("HTTP/")) {
240             throw new ProtocolException("Invalid HTTP version string: " + s);
241         }
242         int major, minor;
243         
244         int i1 = "HTTP/".length();
245         int i2 = s.indexOf(".", i1);
246         if (i2 == -1) {
247             throw new ProtocolException("Invalid HTTP version number: " + s);
248         }
249         try {
250             major = Integer.parseInt(s.substring(i1, i2)); 
251         } catch (NumberFormatException e) {
252             throw new ProtocolException("Invalid HTTP major version number: " + s);
253         }
254         i1 = i2 + 1;
255         i2 = s.length();
256         try {
257             minor = Integer.parseInt(s.substring(i1, i2)); 
258         } catch (NumberFormatException e) {
259             throw new ProtocolException("Invalid HTTP minor version number: " + s);
260         }
261         return new HttpVersion(major, minor);
262     }
