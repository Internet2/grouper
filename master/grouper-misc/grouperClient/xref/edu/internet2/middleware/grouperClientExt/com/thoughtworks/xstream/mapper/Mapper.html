27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.mapper;
28  
29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.Converter;
30  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.SingleValueConverter;
31  
32  public interface Mapper {
33      /**
34       * Place holder type used for null values.
35       */
36      class Null {}
37  
38      /**
39       * How a class name should be represented in its serialized form.
40       */
41      String serializedClass(Class type);
42  
43      /**
44       * How a serialized class representation should be mapped back to a real class.
45       */
46      Class realClass(String elementName);
47  
48      /**
49       * How a class member should be represented in its serialized form.
50       */
51      String serializedMember(Class type, String memberName);
52  
53      /**
54       * How a serialized member representation should be mapped back to a real member.
55       */
56      String realMember(Class type, String serialized);
57  
58      /**
59       * Whether this type is a simple immutable value (int, boolean, String, URL, etc.
60       * Immutable types will be repeatedly written in the serialized stream, instead of using object references.
61       */
62      boolean isImmutableValueType(Class type);
63  
64      Class defaultImplementationOf(Class type);
65  
66      /**
67       * @deprecated since 1.2, use aliasForAttribute instead.
68       */
69      String attributeForImplementationClass();
70  
71      /**
72       * @deprecated since 1.2, use aliasForAttribute instead.
73       */
74      String attributeForClassDefiningField();
75  
76      /**
77       * @deprecated since 1.2, use aliasForAttribute instead.
78       */
79      String attributeForReadResolveField();
80  
81      /**
82       * @deprecated since 1.2, use aliasForAttribute instead.
83       */
84      String attributeForEnumType();
85  
86      /**
87       * Get the alias for an attrbute's name.
88       * 
89       * @param attribute the attribute
90       * @return the alias
91       * @since 1.2
92       */
93      String aliasForAttribute(String attribute);
94  
95      /**
96       * Get the attribut's name for an alias.
97       * 
98       * @param alias the alias
99       * @return the attribute's name
100      * @since 1.2
101      */
102     String attributeForAlias(String alias);
103 
104     /**
105      * Get the name of the field that acts as the default collection for an object, or return null if there is none.
106      *
107      * @param definedIn     owning type
108      * @param itemType      item type
109      * @param itemFieldName optional item element name
110      */
111     String getFieldNameForItemTypeAndName(Class definedIn, Class itemType, String itemFieldName);
112 
113     Class getItemTypeForItemFieldName(Class definedIn, String itemFieldName);
114 
115     ImplicitCollectionMapping getImplicitCollectionDefForFieldName(Class itemType, String fieldName);
116 
117     /**
118      * Determine whether a specific member should be serialized.
119      *
120      * @since 1.1.3
121      */
122     boolean shouldSerializeMember(Class definedIn, String fieldName);
123 
124     interface ImplicitCollectionMapping {
125         String getFieldName();
126         String getItemFieldName();
127         Class getItemType();
128     }
129 
130     /**
131      * @deprecated since 1.3, use {@link #getConverterFromItemType(String, Class, Class)}
132      */
133     SingleValueConverter getConverterFromItemType(String fieldName, Class type);
134 
135     /**
136      * @deprecated since 1.3, use {@link #getConverterFromItemType(String, Class, Class)}
137      */
138     SingleValueConverter getConverterFromItemType(Class type);
139 
140     /**
141      * @deprecated since 1.3, use {@link #getConverterFromAttribute(Class, String)}
142      */
143     SingleValueConverter getConverterFromAttribute(String name);
144 
145     Converter getLocalConverter(Class definedIn, String fieldName);
146 
147     Mapper lookupMapperOfType(Class type);
148 
149     /**
150      * Returns a single value converter to be used in a specific field.
151      * 
152      * @param fieldName the field name
153      * @param type the field type
154      * @param definedIn the type which defines this field
155      * @return a SingleValueConverter or null if there no such converter should be used for this
156      *         field.
157      * @since 1.2.2
158      */
159     SingleValueConverter getConverterFromItemType(String fieldName, Class type, Class definedIn);
160 
161     /**
162      * Returns an alias for a single field defined in an specific type.
163      * 
164      * @param definedIn the type where the field was defined
165      * @param fieldName the field name
166      * @return the alias for this field or its own name if no alias was defined
167      * @since 1.2.2
168      * @deprecated since 1.3, use combination of {@link #serializedMember(Class, String)} and {@link #getConverterFromItemType(String, Class, Class)} 
169      */
170     String aliasForAttribute(Class definedIn, String fieldName);
171 
172     /**
173      * Returns the field name for an aliased attribute.
174      * 
175      * @param definedIn the type where the field was defined
176      * @param alias the alias
177      * @return the original attribute name
178      * @since 1.2.2
179      * @deprecated since 1.3, use combination of {@link #realMember(Class, String)} and {@link #getConverterFromItemType(String, Class, Class)} 
180      */
181     String attributeForAlias(Class definedIn, String alias);
182 
183     /**
184      * Returns which converter to use for an specific attribute in a type.
185      * 
186      * @param definedIn the field's parent
187      * @param attribute the attribute name
188      * @since 1.2.2
189      */
190     SingleValueConverter getConverterFromAttribute(Class definedIn, String attribute);
191 }
