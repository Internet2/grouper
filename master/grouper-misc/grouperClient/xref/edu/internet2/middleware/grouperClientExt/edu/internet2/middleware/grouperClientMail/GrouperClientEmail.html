38  /**
39   * use this chaining utility to send email
40   */
41  public class GrouperClientEmail {
42  
43    /** who this email is going to (comma separated) */
44    private String to;
45    
46    /** subject of email */
47    private String subject;
48    
49    /** email address this is from */
50    private String from;
51    
52    /** body of email (currently HTML is not supported, only plain text) */
53    private String body;
54    
55    /**
56     * 
57     */
58    public GrouperClientEmail() {
59      //empty 
60    }
61    
62    /**
63     * set the to address
64     * @param theToAddress 
65     * @return this for chaining
66     */
67    public GrouperClientEmail setTo(String theToAddress) {
68      this.to = theToAddress;
69      return this;
70    }
71    
72    /**
73     * set subject
74     * @param theSubject
75     * @return this for chaining
76     */
77    public GrouperClientEmail setSubject(String theSubject) {
78      this.subject = theSubject;
79      return this;
80    }
81    
82    /**
83     * set the body
84     * @param theBody
85     * @return this for chaining
86     */
87    public GrouperClientEmail setBody(String theBody) {
88      this.body = theBody;
89      return this;
90    }
91   
92    /**
93     * set the from address
94     * @param theFrom
95     * @return the from address
96     */
97    public GrouperClientEmail setFrom(String theFrom) {
98      this.from = theFrom;
99      return this;
100   }
101 
102   /**
103    * try an email
104    * @param args
105    */
106   public static void main(String[] args) {
107     new GrouperClientEmail().setBody("hey").setSubject("subject").setTo("mchyzer@yahoo.com,mchyzer@isc.upenn.edu").send();
108   }
109   
110   /**
111    * send the email
112    */
113   public void send() {
114     
115     try {
116       //mail.smtp.server = whatever.school.edu
117       //#mail.from.address = noreply@school.edu
118       String theFrom = GrouperClientUtils.defaultIfEmpty(this.from, GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.mail.from.address"));
119       if (GrouperClientUtils.isBlank(theFrom)) {
120         throw new RuntimeException("You need to specify the from email address mail.from.address in grouper.client.properties");
121       }
122       
123       String smtpServer = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.mail.smtp.server");
124       if (GrouperClientUtils.isBlank(smtpServer)) {
125         throw new RuntimeException("You need to specify the from smtp server mail.smtp.server in grouper.client.properties");
126       }
127       
128       String subjectPrefix = GrouperClientUtils.defaultString(GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.mail.subject.prefix"));
129       
130       final String SMTP_USER = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.mail.smtp.user"); 
131       
132       String smtpPass = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.mail.smtp.pass"); 
133       
134       final String SMTP_PASS = GrouperClientUtils.isBlank(smtpPass) ? null : Morph.decryptIfFile(smtpPass);
135       
136       Properties properties = new Properties();
137       
138       properties.put("mail.host", smtpServer);
139       properties.put("mail.transport.protocol", "smtp");
140       
141       Authenticator authenticator = null;
142   
143       //this has never been tested... :)
144       if (!GrouperClientUtils.isBlank(SMTP_USER)) {
145         properties.setProperty("mail.smtp.submitter", SMTP_USER);
146         properties.setProperty("mail.smtp.auth", "true");
147         
148         authenticator = new Authenticator() {
149           /**
150            * 
151            * @see javax.mail.Authenticator#getPasswordAuthentication()
152            */
153           @Override
154           protected PasswordAuthentication getPasswordAuthentication() {
155             return new PasswordAuthentication(SMTP_USER, SMTP_PASS);
156           }
157         };
158       }
159       
160       Session session = Session.getInstance(properties, authenticator);
161       Message message = new MimeMessage(session);
162       
163       String theTo = this.to;
164       
165       if (!GrouperClientUtils.isBlank(theTo)) {
166         
167         theTo = GrouperClientUtils.replace(theTo, ";", ",");
168         String[] theTos = GrouperClientUtils.splitTrim(theTo, ",");
169         for (String aTo : theTos) {
170           message.addRecipient(RecipientType.TO, new InternetAddress(aTo));
171         }
172         
173       }
174       message.addFrom(new InternetAddress[] { new InternetAddress(theFrom) });
175   
176       String theSubject = GrouperClientUtils.defaultString(subjectPrefix) + this.subject;
177       message.setSubject(theSubject);
178       
179       message.setContent(this.body, "text/plain");
180       Transport.send(message);
181     } catch (Exception e) {
182       throw new RuntimeException(e);
183     }
184   }
