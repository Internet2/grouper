79    /**
80    * see if the esb event matches an EL filter. Note the available objects are
81    * event for the EsbEvent, and grouperUtil for the GrouperUtil class which has
82    * a lot of utility methods
83    * @param filterString
84    * @param esbEvent
85    * @return true if matches, false if doesnt
86    */
87    public static boolean matchesFilter(EsbEvent esbEvent, String filterString) {
88      try {
89        Map<String, Object> variableMap = new HashMap<String, Object>();
90        variableMap.put("event", esbEvent);
91        variableMap.put("grouperClientUtils", new GrouperClientUtils());
92        
93        String elResultString = GrouperClientUtils.substituteExpressionLanguage(filterString, variableMap, true, false, false, true);
94        return GrouperClientUtils.booleanValue(elResultString);
95        
96      } catch (Exception e) {
97        throw new RuntimeException("Problem seeing if matches filter for sequence: "
98            + (esbEvent == null ? null : esbEvent.getSequenceNumber()) + ", '" + filterString + "'", e);
99      }
100   }

105   /**
106   * get an xmpp connection
107   * @return xmpp connection
108   */
109   private static synchronized XMPPConnection xmppConnection() {
110     if (xmppConnection == null || !xmppConnection.isAuthenticated() || !xmppConnection.isConnected()) {
111       String user = null;
112       String pass = null;
113       String resource = null;
114       String server = null;
115       int port = -1;
116       try {
117         if (xmppConnection != null) {
118           try {
119             xmppConnection.disconnect();
120           } catch (Exception e) {
121             //this is ok
122           }
123         }
124         server = xmppServer();
125         port = xmppPort();
126         ConnectionConfiguration config = new ConnectionConfiguration(server, port);
127 
128         boolean debuggerEnabled = GrouperClientConfig.retrieveConfig().propertyValueBoolean("grouperClient.xmpp.debuggerEnabled", false);
129 
130         config.setDebuggerEnabled(debuggerEnabled);
131         config.setReconnectionAllowed(true);
132 
133         config.setSASLAuthenticationEnabled(true);
134         SASLAuthentication.supportSASLMechanism("PLAIN");
135         config.setSecurityMode(ConnectionConfiguration.SecurityMode.enabled);
136         xmppConnection = new XMPPConnection(config);
137         xmppConnection.connect();
138 
139         user = xmppUser();
140         pass = xmppPass();
141         resource = xmppResource();
142         xmppConnection.login(user, pass, resource);
143       } catch (XMPPException xe) {
144         throw new RuntimeException("Problem connecting: server: " + server + ", port: "
145             + port + ", user: " + user + ", pass not included, "
146             //+ GrouperClientUtils.repeat("*", GrouperClientUtils.defaultString(pass).length())
147             + ", resource: " + resource, xe);
148       }
149     }
150     return xmppConnection;
151   }

153   /**
154   * xmpp pass (decrypted if file)
155   * @return the pass
156   */
157   public static String xmppPass() {
158 
159     boolean disableExternalFileLookup = GrouperClientConfig.retrieveConfig().propertyValueBooleanRequired(
160         "encrypt.disableExternalFileLookup");
161 
162     //lets lookup if file
163     String pass = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.xmpp.pass");
164     String passFromFile = GrouperClientUtils.readFromFileIfFile(pass, disableExternalFileLookup);
165     
166     if (GrouperClientConfig.retrieveConfig().propertyValueBoolean("encrypt.encryptLikeServer", false)) {
167       if (!GrouperClientUtils.equals(pass, passFromFile)) {
168   
169         String encryptKey = GrouperClientUtils.encryptKey();
170         pass = new Crypto(encryptKey).decrypt(passFromFile);
171         
172       }
173 
174       return pass;
175     }
176     if (!GrouperClientUtils.equals(pass, passFromFile)) {
177 
178       String encryptKey = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("encrypt.key");
179       encryptKey = GrouperClientUtils.readFromFileIfFile(encryptKey, disableExternalFileLookup);
180       passFromFile = new Crypto(encryptKey).decrypt(passFromFile);
181 
182     }
183 
184     return passFromFile;
185   }

219   /**
220   * connect to xmpp
221    * @param grouperClientXmppMessageHandler the handler for the message
222    */
223    public static void xmppConnect(final GrouperClientXmppMessageHandler grouperClientXmppMessageHandler) {
224      
225      XMPPConnection theXmppConnection = xmppConnection();
226      theXmppConnection.addPacketListener(new PacketListener() {
227 
228        // @Override
229        public void processPacket(Packet packet) {
230          Message message = null;
231          try {
232            message = (Message) packet;
233            if (log.isDebugEnabled()) {
234              log.debug(message == null ? null : message.toXML());
235            }
236            grouperClientXmppMessageHandler.handleMessage(message);
237 
238          } catch (Throwable re) {
239            String messageXml = message == null ? null : message.toXML();
240            log.error("Problem with message: " + messageXml, re);
241            throw new RuntimeException(re);
242          }
243 
244        }
245      }, new PacketFilter() {
246 
247        // @Override
248        public boolean accept(Packet packet) {
249          if (packet instanceof Message) {
250            Message message = (Message) packet;
251            Type type = message.getType();
252            if (type == Type.chat && !GrouperClientUtils.isBlank(message.getBody())) {
253              if (allowFromJabberIds().contains(message.getFrom())) {
254                return true;
255              }
256              if (log.isDebugEnabled()) {
257                log.debug("Not expecting message from: " + message.getFrom());
258              }
259            }
260          }
261          return false;
262        }
263      });
264    }

266   /**
267   * connect to xmpp
268   */
269   private static void xmppLoopForGroups() {
270     xmppLoop(new GrouperClientXmppMessageHandler() {
271       
272       @Override
273       public void handleMessage(Message message) {
274         
275         String body = message.getBody();
276         EsbEvents esbEvents = (EsbEvents)JsonUtils.jsonConvertFrom(body, EsbEvents.class);
277         for (EsbEvent esbEvent : GrouperClientUtils.nonNull(esbEvents.getEsbEvent(), EsbEvent.class)) {
278           //loop through jobs and see what matches
279           for (GrouperClientXmppJob grouperClientXmppJob : GrouperClientXmppJob.retrieveXmppJobs()) {
280             String elfilter = grouperClientXmppJob.getElfilter();
281             Boolean matches = null;
282 
283             String groupName = esbEvent.getGroupName();
284             if (GrouperClientUtils.isBlank(groupName) && (
285                 "GROUP_ADD".equals(esbEvent.getEventType()) || 
286                 "GROUP_DELETE".equals(esbEvent.getEventType()) ||
287                 "GROUP_UPDATE".equals(esbEvent.getEventType()))) {
288               groupName = esbEvent.getName();
289             }
290 
291             if (!GrouperClientUtils.isBlank(elfilter)) {
292               if (!matchesFilter(esbEvent, elfilter)) {
293 
294                 matches = false;
295 
296                 if (log.isDebugEnabled()) {
297                   log.debug("skipping event to not match filter, sequence: " + (esbEvent == null ? null : esbEvent.getSequenceNumber()) + ", '" + elfilter + "', " + grouperClientXmppJob.getJobName());
298                 }
299               } else {
300                 matches = true;
301               }
302 
303             }
304             if (XmppJobEventAction.incremental == grouperClientXmppJob.getEventAction()
305                 && grouperClientXmppJob.isAllowIncrementalNotInGroupNamesList()) {
306               if (log.isDebugEnabled()) {
307                 log.debug("including since incremental and allowIncrementalNotInGroupNamesList is true: " + grouperClientXmppJob.getJobName());
308               }
309               matches = true;
310             } else {
311               if (GrouperClientUtils.nonNull(grouperClientXmppJob.getGroupNames()).size() > 0) {
312                 if (grouperClientXmppJob.getGroupNames().contains(groupName)) {
313                   if (matches == null) {
314                     matches = true;
315                   }
316                 } else {
317                   if (log.isDebugEnabled()) {
318                     log.debug("skipping event to not match group name list, sequence: " + (esbEvent == null ? null : esbEvent.getSequenceNumber()) + ", " + grouperClientXmppJob.getJobName());
319                   }
320                   matches = false;
321                 }
322               }
323             }
324             if (matches != null && !matches) {
325               continue;
326             }
327 
328             //see what type
329             if (XmppJobEventAction.reload_group == grouperClientXmppJob.getEventAction()) {
330 
331               if (log.isDebugEnabled()) {
332                 log.debug("performing a full reload on group: " + groupName
333                     + " for job: " + grouperClientXmppJob.getJobName() + ", subject: " + esbEvent.getSubjectId());
334               }
335               fullRefreshGroup(grouperClientXmppJob, groupName);
336 
337             } else if (XmppJobEventAction.incremental == grouperClientXmppJob.getEventAction()) {
338 
339               if (log.isDebugEnabled()) {
340                 log.debug("performing an incremental reload on group: " + groupName
341                     + " for job: " + grouperClientXmppJob.getJobName() + ", subject: " + esbEvent.getSubjectId());
342               }
343 
344               GrouperClientXmppSubject grouperClientXmppSubject = new GrouperClientXmppSubject(esbEvent);
345 
346               incrementalRefreshGroup(grouperClientXmppJob, groupName,
347                   grouperClientXmppSubject, esbEvent.getEventType());
348 
349             } else {
350               throw new RuntimeException("Not expecting event action: " + grouperClientXmppJob.getEventAction());
351             }
352 
353           }
354         }
355       }
356     });
357     }

421   /** mape of group name to list of subjects in group */
422   private static Map<String, List<GrouperClientXmppSubject>> groupMemberships = new HashMap<String, List<GrouperClientXmppSubject>>();

424   /**
425   *
426   * @param grouperClientXmppJob
427   * @param groupName
428   */
429   @SuppressWarnings("unchecked")
430   public static void fullRefreshGroup(GrouperClientXmppJob grouperClientXmppJob, String groupName) {
431     GcGetMembers gcGetMembers = new GcGetMembers();
432 
433     List<String> subjectAttributeNames = grouperClientXmppJob.getSubjectAttributeNames();
434 
435     if (GrouperClientUtils.length(subjectAttributeNames) > 0) {
436       for (String subjectAttributeName : subjectAttributeNames) {
437         gcGetMembers.addSubjectAttributeName(subjectAttributeName);
438       }
439     }
440 
441     WsGetMembersResults wsGetMembersResults = gcGetMembers.addGroupName(groupName).execute();
442     WsGetMembersResult wsGetMembersResult = wsGetMembersResults.getResults()[0];
443 
444     List<GrouperClientXmppSubject> xmppSubjects = new ArrayList<GrouperClientXmppSubject>();
445 
446     for (WsSubject wsSubject : GrouperClientUtils.nonNull(wsGetMembersResult.getWsSubjects(), WsSubject.class)) {
447       GrouperClientXmppSubject xmppSubject = new GrouperClientXmppSubject(wsSubject, wsGetMembersResults.getSubjectAttributeNames());
448 
449       xmppSubjects.add(xmppSubject);
450     }
451 
452     if (log.isDebugEnabled()) {
453       log.debug("Refreshing all for " + groupName + " found " + GrouperClientUtils.length(xmppSubjects) + " subjects");
454     }
455 
456     Class<GrouperClientXmppHandler> handlerClass = GrouperClientUtils.forName(grouperClientXmppJob.getHandlerClass());
457     GrouperClientXmppHandler grouperClientXmppHandler = GrouperClientUtils.newInstance(handlerClass);
458     groupMemberships.put(groupName, xmppSubjects);
459     grouperClientXmppHandler.handleAll(grouperClientXmppJob, groupName, GrouperClientUtils.extensionFromName(groupName), xmppSubjects);
460 
461   }

493   /**
494    * note, this doesnt return, and you should only call this once...
495    * @param grouperClientXmppMessageHandler handle the message
496    */
497    public static void xmppLoop(GrouperClientXmppMessageHandler grouperClientXmppMessageHandler) {
498      //are we already in this loop?
499      if (xmppConnection != null) {
500        return;
501      }
502      while (true) {
503        try {
504          if (xmppConnection == null || !xmppConnection.isConnected() || !xmppConnection.isAuthenticated()) {
505 
506            //if not starting, this is a problem
507            if (xmppConnection != null) {
508              log.error("xmpp connection is not connected");
509              try {
510                xmppConnection.disconnect();
511              } catch (Exception e) {
512                log.error("error", e);
513              }
514              xmppConnection = null;
515            }
516            xmppConnect(grouperClientXmppMessageHandler);
517          }
518        } catch (Exception e) {
519          log.error("Problem with xmpp", e);
520        }
521        GrouperClientUtils.sleep(60000);
522      }
523    }

525    /**
526     * schedule a cron job
527     * @param jobName something unique and descriptive
528     * @param quartzCronString
529     * @param jobClass
530     */
531    public static void scheduleJob(String jobName, String quartzCronString, Class<? extends Job> jobClass) {
532      //no cron string, dont run the cron
533      if (GrouperClientUtils.isBlank(quartzCronString)) {
534        return;
535      }
536      
537      String jobGroup = Scheduler.DEFAULT_GROUP;
538      JobDetail jobDetail = new JobDetail(jobName,
539          jobGroup, jobClass);
540 
541      Scheduler scheduler = GrouperClientXmppMain.scheduler();
542 
543      //atlassian requires durable jobs...
544      jobDetail.setDurability(true);
545 
546      boolean uniqueTriggerNames = GrouperClientConfig.retrieveConfig().propertyValueBoolean("grouperClient.xmpp.uniqueQuartzTriggerNames", false);
547 
548      //in old versions of quartz, the trigger group cannot be null
549      String triggerName = "trigger_" + jobName;
550 
551      if (uniqueTriggerNames) {
552        triggerName += GrouperClientUtils.uniqueId();
553      }
554 
555      CronTrigger cronTrigger = null;
556      if (!uniqueTriggerNames) {
557        try {
558          cronTrigger = (CronTrigger) scheduler.getTrigger(triggerName, jobGroup);
559        } catch (SchedulerException se) {
560          throw new RuntimeException("Problem with trigger: " + jobName, se);
561        }
562      }
563      if (cronTrigger == null) {
564        cronTrigger = new CronTrigger(triggerName, jobGroup);
565      }
566 
567      try {
568        cronTrigger.setCronExpression(quartzCronString);
569      } catch (ParseException pe) {
570        throw new RuntimeException("Problems parsing: '" + quartzCronString + "'", pe);
571      }
572 
573      try {
574        scheduler.scheduleJob(jobDetail, cronTrigger);
575      } catch (SchedulerException se) {
576        throw new RuntimeException("Problem with job: " + jobName, se);
577      }
578 
579 
580    }

631   /**
632   * Mutate the membership list depending on the action.
633   * Cache the current membership in a field of this class.
634   * Pass the previous and current membership list off to the handler class for the {@link GrouperClientXmppJob}
635   * @param grouperClientXmppJob
636   * @param groupName
637   * @param grouperClientXmppSubject
638   * @param eventType
639   */
640   @SuppressWarnings("unchecked")
641   private static void incrementalRefreshGroup(GrouperClientXmppJob grouperClientXmppJob,
642       String groupName, GrouperClientXmppSubject grouperClientXmppSubject, String eventType) {
643 
644     if (!SUPPORTED_EVENT_TYPES.contains(eventType)) {
645       throw new RuntimeException("Not expecting action: '" + eventType + "'");
646     }
647 
648     List<GrouperClientXmppSubject> oldList = java.util.Collections.EMPTY_LIST;
649 
650     if (eventType.equals("MEMBERSHIP_ADD") || eventType.equals("MEMBERSHIP_DELETE")) {
651       oldList = groupMemberships.get(groupName);
652       if (oldList == null) {
653         // Do a full refresh and then ask for the group membership again
654         fullRefreshGroup(grouperClientXmppJob, groupName);
655         oldList = groupMemberships.get(groupName);
656       }
657       if (oldList == null) {
658         throw new NullPointerException("Why is old list null????");
659       }
660     }
661 
662     List<GrouperClientXmppSubject> newList = new ArrayList<GrouperClientXmppSubject>(oldList);
663 
664     if (GrouperClientUtils.equals(eventType, "MEMBERSHIP_ADD")) {
665 
666       boolean changed = newList.add(grouperClientXmppSubject);
667       if (changed && log.isDebugEnabled()) {
668         log.debug("Group " + groupName + " already contains subject: " + grouperClientXmppSubject.getSubjectId());
669       }
670 
671     } else if (GrouperClientUtils.equals(eventType, "MEMBERSHIP_DELETE")) {
672 
673       Collection<GrouperClientXmppSubject> toRemove = GrouperClientUtils.toList(grouperClientXmppSubject);
674       boolean changed = newList.removeAll(toRemove);
675       if (!changed && log.isDebugEnabled()) {
676         log.debug("Group " + groupName + " already doesnt contain subject: " + grouperClientXmppSubject.getSubjectId());
677       }
678     }
679 
680     groupMemberships.put(groupName, newList);
681 
682     if (log.isDebugEnabled()) {
683       log.debug("Event: " + eventType + " for " + groupName + ", memberships list was "
684           + GrouperClientUtils.length(oldList)
685           + " and is now " + GrouperClientUtils.length(newList) + " subjects");
686     }
687 
688     // Instantiate the handler for this job and call the handleIncremental method
689     Class<GrouperClientXmppHandler> handlerClass = GrouperClientUtils
690         .forName(grouperClientXmppJob.getHandlerClass());
691     GrouperClientXmppHandler grouperClientXmppHandler = GrouperClientUtils
692         .newInstance(handlerClass);
693     grouperClientXmppHandler.handleIncremental(grouperClientXmppJob,
694         groupName, GrouperClientUtils.extensionFromName(groupName),
695          newList, oldList, grouperClientXmppSubject, eventType);
696   }
