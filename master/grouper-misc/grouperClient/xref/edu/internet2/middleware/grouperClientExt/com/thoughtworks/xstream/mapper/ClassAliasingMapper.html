27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.mapper;
28  
29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.alias.ClassMapper;
30  
31  import java.util.HashMap;
32  import java.util.Iterator;
33  import java.util.Map;
34  
35  /**
36   * Mapper that allows a fully qualified class name to be replaced with a shorter alias.
37   *
38   * @author Joe Walnes
39   * @author J&ouml;rg Schaible
40   */
41  public class ClassAliasingMapper extends MapperWrapper {
42  
43      protected final Map typeToName = new HashMap();
44      protected final Map classToName = new HashMap();
45      protected transient Map nameToType = new HashMap();
46  
47      public ClassAliasingMapper(Mapper wrapped) {
48          super(wrapped);
49      }
50  
51      /**
52       * @deprecated since 1.2, use {@link #ClassAliasingMapper(Mapper)}
53       */
54      public ClassAliasingMapper(ClassMapper wrapped) {
55          this((Mapper)wrapped);
56      }
57  
58      public void addClassAlias(String name, Class type) {
59          nameToType.put(name, type.getName());
60          classToName.put(type.getName(), name);
61      }
62  
63      /**
64       * @deprecated since 1.3, method was a leftover of an old implementation
65       */
66      public void addClassAttributeAlias(String name, Class type) {
67          addClassAlias(name, type);
68      }
69  
70      public void addTypeAlias(String name, Class type) {
71          nameToType.put(name, type.getName());
72          typeToName.put(type, name);
73      }
74  
75      public String serializedClass(Class type) {
76          String alias = (String) classToName.get(type.getName());
77          if (alias != null) {
78              return alias;
79          } else {
80              for (final Iterator iter = typeToName.keySet().iterator(); iter.hasNext();) {
81                  final Class compatibleType = (Class)iter.next();
82                  if (compatibleType.isAssignableFrom(type)) {
83                      return (String)typeToName.get(compatibleType);
84                  }
85              }
86              return super.serializedClass(type);
87          }
88      }
89  
90      public Class realClass(String elementName) {
91          String mappedName = (String) nameToType.get(elementName);
92  
93          if (mappedName != null) {
94              Class type = primitiveClassNamed(mappedName);
95              if (type != null) {
96                  return type;
97              }
98              elementName = mappedName;
99          }
100 
101         return super.realClass(elementName);
102     }
103 
104     public boolean itemTypeAsAttribute(Class clazz) {
105         return classToName.containsKey(clazz);
106     }
107 
108     public boolean aliasIsAttribute(String name) {
109         return nameToType.containsKey(name);
110     }
111     
112     private Object readResolve() {
113         nameToType = new HashMap();
114         for (final Iterator iter = classToName.keySet().iterator(); iter.hasNext();) {
115             final Object type = iter.next();
116             nameToType.put(classToName.get(type), type);
117         }
118         for (final Iterator iter = typeToName.keySet().iterator(); iter.hasNext();) {
119             final Class type = (Class)iter.next();
120             nameToType.put(typeToName.get(type), type.getName());
121         }
122         return this;
123     }
124     
125     private Class primitiveClassNamed(String name) {
126         return
127                 name.equals("void") ? Void.TYPE :
128                 name.equals("boolean") ? Boolean.TYPE :
129                 name.equals("byte") ? Byte.TYPE :
130                 name.equals("char") ? Character.TYPE :
131                 name.equals("short") ? Short.TYPE :
132                 name.equals("int") ? Integer.TYPE :
133                 name.equals("long") ? Long.TYPE :
134                 name.equals("float") ? Float.TYPE :
135                 name.equals("double") ? Double.TYPE :
136                 null;
137     }
138 }
