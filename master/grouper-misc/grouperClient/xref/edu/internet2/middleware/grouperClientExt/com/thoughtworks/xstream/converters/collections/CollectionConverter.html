43  /**
44   * Converts most common Collections (Lists and Sets) to XML, specifying a nested
45   * element for each item.
46   * <p/>
47   * <p>Supports java.util.ArrayList, java.util.HashSet,
48   * java.util.LinkedList, java.util.Vector and java.util.LinkedHashSet.</p>
49   *
50   * @author Joe Walnes
51   */
52  public class CollectionConverter extends AbstractCollectionConverter {
53  
54      public CollectionConverter(Mapper mapper) {
55          super(mapper);
56      }
57  
58      public boolean canConvert(Class type) {
59          return type.equals(ArrayList.class)
60                  || type.equals(HashSet.class)
61                  || type.equals(LinkedList.class)
62                  || type.equals(Vector.class)
63                  || (JVM.is14() && type.getName().equals("java.util.LinkedHashSet"));
64      }
65  
66      public void marshal(Object source, HierarchicalStreamWriter writer, MarshallingContext context) {
67          Collection collection = (Collection) source;
68          for (Iterator iterator = collection.iterator(); iterator.hasNext();) {
69              Object item = iterator.next();
70              writeItem(item, context, writer);
71          }
72      }
73  
74      public Object unmarshal(HierarchicalStreamReader reader, UnmarshallingContext context) {
75          Collection collection = (Collection) createCollection(context.getRequiredType());
76          populateCollection(reader, context, collection);
77          return collection;
78      }
79  
80      protected void populateCollection(HierarchicalStreamReader reader, UnmarshallingContext context, Collection collection) {
81          while (reader.hasMoreChildren()) {
82              reader.moveDown();
83              Object item = readItem(reader, context, collection);
84              collection.add(item);
85              reader.moveUp();
86          }
87      }
88  
89  }
