96    /**
97     * @param args
98     */
99    public static void main(String[] args) {
100     
101     long startTime = System.currentTimeMillis();
102 
103     String operation = null;
104     try {
105       if (GrouperClientUtils.length(args) == 0) {
106         usage();
107         return;
108       }
109       
110       //map of all command line args
111       Map<String, String> argMap = GrouperClientUtils.argMap(args);
112       
113       Map<String, String> argMapNotUsed = new LinkedHashMap<String, String>(argMap);
114 
115       boolean debugMode = GrouperClientUtils.argMapBoolean(argMap, argMapNotUsed, "debug", false, false);
116       
117       GrouperClientLog.assignDebugToConsole(debugMode);
118       
119       //init if not already
120       GrouperClientConfig.retrieveConfig().properties();
121       
122       //see where log file came from
123       StringBuilder callingLog = new StringBuilder();
124       GrouperClientUtils.propertiesFromResourceName("grouper.client.properties", 
125           false, true, GrouperClientCommonUtils.class, callingLog);
126       
127       //see if the message about where it came from is
128       //log.debug(callingLog.toString());
129       
130       operation = GrouperClientUtils.argMapString(argMap, argMapNotUsed, "operation", true);
131       
132       //where results should go if file
133       String saveResultsToFile = GrouperClientUtils.argMapString(argMap, argMapNotUsed, "saveResultsToFile", false);
134       boolean shouldSaveResultsToFile = !GrouperClientUtils.isBlank(saveResultsToFile);
135       
136       if (shouldSaveResultsToFile) {
137         log.debug("Will save results to file: " + GrouperClientUtils.fileCanonicalPath(new File(saveResultsToFile)));
138       }
139       
140       String result = null;
141       
142       if (GrouperClientUtils.equals(operation, "getFile")) {
143         
144         String fileName = GrouperClientUtils.argMapString(argMap, argMapNotUsed, "fileName", true);
145         
146         File file = retrieveFile(fileName, true);
147         if (file == null) {
148           
149           throw new RuntimeException("Discovery service did not retrieve a success for file: " + fileName);
150         }
151         result = file.getAbsolutePath();
152         
153       } else {
154         System.err.println("Error: invalid operation: '" + operation + "', for usage help, run: java -cp grouperClient.jar edu.internet2.middleware.grouperClient.discovery.DiscoveryClient" );
155         if (GrouperClient.exitOnError) {
156           System.exit(1);
157         }
158         throw new RuntimeException("Invalid usage");
159       }
160       
161       //this already has a newline on it
162       if (shouldSaveResultsToFile) {
163         GrouperClientUtils.saveStringIntoFile(new File(saveResultsToFile), result);
164       } else {
165         System.out.print(result);
166       }
167 
168       GrouperClient.failOnArgsNotUsed(argMapNotUsed);
169       
170     } catch (Exception e) {
171       System.err.println("Error with grouper client, check the logs: " + e.getMessage());
172       log.fatal(e.getMessage(), e);
173       if (GrouperClient.exitOnError) {
174         System.exit(1);
175       }
176       if (e instanceof RuntimeException) {
177         throw (RuntimeException)e;
178       }
179       throw new RuntimeException(e.getMessage(), e);
180     } finally {
181       try {
182         log.debug("Elapsed time: " + (System.currentTimeMillis() - startTime) + "ms");
183       } catch (Exception e) {}
184       GrouperClientLog.assignDebugToConsole(false);
185     }
186     
187   }

247   /**
248    * retrieve a file from the discovery server
249    * @param fileName file name on the server
250    * @param throwExceptionIfNotFound true if should throw an exception if not found
251    * @return the file or throw an exception if not found if supposed to.  If not configured to use
252    * discovery, return null
253    */
254   public static File retrieveFile(String fileName, boolean throwExceptionIfNotFound) {
255 
256     Map<String, Object> logMap = log.isDebugEnabled() ? new LinkedHashMap<String, Object>() : null;
257 
258     if (logMap != null) {
259       logMap.put("method", "DiscoveryClient.retrieveFile");
260     }
261 
262     File file = null;
263     try {
264 
265       //see if not doing discovery
266       if (!hasDiscovery()) {
267         if (logMap != null) {
268           logMap.put("configuredToUseDiscovery", false);
269         }
270         return null;
271       }
272       String localFileName = convertFileNameToLocalFileName(fileName);
273       
274       
275       //check discovery cache
276       file = discoveryFileCache().get(localFileName);
277 
278       //if we got it from cache, dont put it back into cache
279       boolean retrievedFromCache = file != null;
280       
281       if (logMap != null) {
282         logMap.put("existsInDiscoveryFilecache", file != null);
283       }
284 
285       if (file == null) {
286         //lets see what the most recent file is on the file system
287         File discoveryLocalFile = mostRecentFileFromFileSystem(localFileName);
288         if (discoveryLocalFile != null && discoveryLocalFile.exists()) {
289           
290           //file.ext would become
291           //file_20120102_132414_123_sd43sdf.ext
292           Matcher matcher = localCacheDatePattern.matcher(discoveryLocalFile.getName());
293 
294           if (!matcher.matches()) {
295             throw new RuntimeException("Why does matcher not match???? " + discoveryLocalFile.getAbsolutePath());
296           }
297 
298           String datePart = matcher.group(2);
299 
300           DateFormat dateFormat = new SimpleDateFormat(TEMP_FILE_DATE_FORMAT);
301           Date date = null;
302 
303           try {
304             date = dateFormat.parse(datePart);
305           } catch (ParseException pe) {
306             
307             throw new RuntimeException("Why date format exception???? " + file.getAbsolutePath());
308             
309           }
310 
311           //see if date in range
312           if ((System.currentTimeMillis() - date.getTime()) / 1000 < GrouperClientConfig.retrieveConfig().propertyValueInt("grouperClient.cacheDiscoveryPropertiesForSeconds", 120)) {
313             file = discoveryLocalFile;
314           }
315           
316           
317         }
318         //if this is less than however old, then use it
319         if (logMap != null) {
320           logMap.put("existsInFilecache", discoveryLocalFile != null);
321         }
322         if (logMap != null) {
323           logMap.put("fileIsYoungEnough", file != null);
324         }
325 
326       }
327       
328       if (file == null) {
329         
330         //lets get it from discovery again, this will return null if problem
331         try {
332           file = retrieveFileFromDiscoveryServer(fileName, localFileName);
333         } catch (Exception e) {
334           log.error("Problem retrieving file from discovery server: " + fileName, e);
335         }
336         if (logMap != null) {
337           logMap.put("fileFromServer", file != null);
338         }
339       }
340 
341       if (file == null) {
342         //just get whatever we have in the filesystem
343         file = mostRecentFileFromFileSystem(localFileName);
344         if (logMap != null) {
345           logMap.put("fileFromFailsafeLocalSystem", file != null);
346         }
347       }
348 
349       //add back to cache if didnt retrieve from cache
350       if (!retrievedFromCache && file != null) {
351         synchronized (DiscoveryClient.class) {
352           discoveryFileCache().put(localFileName, file);
353         }
354       }
355 
356       
357       //end
358       if (logMap != null) {
359         logMap.put("fileFound", file!=null);
360         if (file != null) {
361           logMap.put("fileSizeBytes", file.length());
362           logMap.put("lastModified", new Date(file.lastModified()));
363         }
364       }
365 
366       if (file == null && throwExceptionIfNotFound) {
367         
368         throw new RuntimeException("Cant find file from discovery: '" + fileName + "'");
369       }
370 
371     } finally {
372 
373       if (log.isDebugEnabled()) {
374         log.debug(GrouperClientUtils.mapToString(logMap));
375       }
376 
377     }
378     return file;
379   }

384   static boolean failoverClientConfigured = false;

386   /**
387    * read the grouper.client.properties and configure the failover client once
388    */
389   private static void configureFailoverClientOnce() {
390     if (failoverClientConfigured) {
391       return;
392     }
393     synchronized (DiscoveryClient.class) {
394       if (!failoverClientConfigured) {
395         
396 
397         List<String> discoveryUrls = new ArrayList<String>();
398         
399         for (int i=0;i<100;i++) {
400           String discoveryUrl = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.urlOfDiscovery." + i);
401           if (GrouperClientUtils.isBlank(discoveryUrl)) {
402             break;
403           }
404           discoveryUrls.add(discoveryUrl);
405         }
406         
407         //there should be failover servers at this point...
408         if (discoveryUrls.size() > 0) {
409 
410           //configure it
411           FailoverConfig failoverConfig = new FailoverConfig();
412           failoverConfig.setConnectionNames(discoveryUrls);
413           failoverConfig.setConnectionType(DISCOVERY_CLIENT_CONNECTION_TYPE);
414           failoverConfig.setFailoverStrategy(FailoverStrategy.activeActive);
415           FailoverClient.initFailoverClient(failoverConfig);
416         
417           //remember that we just configured it
418           failoverClientConfigured = true;
419 
420         } else {
421           log.error("There are no discovery URLs in grouper.client.properties");
422         }
423       }
424     }
425   }

433   /**
434    * synchronized object based on name or create if not there

437   private static String synchronizedObjectBasedOnName(String name) {
438     String synchronizedObject = synchronizedObjectBasedOnName.get(name);
439     if (synchronizedObject == null) {
440       synchronized (DiscoveryClient.class) {
441 
442         //make sure another thread didnt just grab it
443         synchronizedObject = synchronizedObjectBasedOnName.get(name);
444         if (synchronizedObject == null) {
445           synchronizedObject = new String(name);
446           synchronizedObjectBasedOnName.put(name, synchronizedObject);
447         }
448       }
449     }
450     return synchronizedObject;
451   }

453   /**
454    * get the most recent file from the file system for a given file name
455    * @param localFileName
456    * @return the file
457    */
458   static File mostRecentFileFromFileSystem(String localFileName) {
459 
460     File cacheDirectory = new File(GrouperClientUtils.cacheDirectoryName());
461     
462     File[] files = cacheDirectory.listFiles();
463 
464     // non tmp file catalog to see if we can delete non tmp file yet
465     Map<Date, File> mapOfDateToFile = new TreeMap<Date, File>();
466     
467     //lets look for tmp files that are 20 minutes old or older
468     for (File file : files) {
469       
470       if (file.getAbsolutePath().endsWith(DISCOVERYTMP)) {
471         continue;
472       }
473         
474       //the filename is something like this:
475       String fileName = file.getName();
476 
477       //file.ext would become
478       //file_20120102_132414_123_sd43sdf.ext
479       Matcher matcher = localCacheDatePattern.matcher(fileName);
480 
481       if (!matcher.matches()) {
482         continue;
483       }
484 
485       //lets catalog the non tmp files
486       String prefixFileName = matcher.group(1);
487       String suffixFileName = matcher.group(3);
488       
489       //see if fileName matches
490       if (!GrouperClientCommonUtils.equals(prefixFileName + suffixFileName, localFileName)) {
491         continue; 
492       }
493       
494       String datePart = matcher.group(2);
495 
496       DateFormat dateFormat = new SimpleDateFormat(TEMP_FILE_DATE_FORMAT);
497       Date date = null;
498 
499       try {
500         date = dateFormat.parse(datePart);
501       } catch (ParseException pe) {
502         
503         log.error("Why date format exception???? " + file.getAbsolutePath());
504         continue;
505         
506       }
507 
508       //add an entry for this
509       mapOfDateToFile.put(date, file);
510       
511     }
512     
513     //the last one should be the most recent
514     if (mapOfDateToFile.size() == 0) {
515       return null;
516     }
517     
518     return new ArrayList<File>(mapOfDateToFile.values()).get(mapOfDateToFile.size()-1);
519 
520   }

522   /**
523    * get the discovery local file with unique extension.  if file exists, try again
524    * @param localFileName
525    * @param isTmpFile true if .discoverytmp should be added to extension
526    * @return the local file
527    */
528   private static File discoveryLocalFileUnique(String localFileName, boolean isTmpFile) {
529     File file = null;
530     //get a new file if the file already exists (unlikely)
531     for (int i=0;i<100;i++) {
532       file = discoveryLocalFileUniqueHelper(localFileName, isTmpFile);
533       if (!file.exists()) {
534         break;
535       }
536     }
537     if (file.exists()) {
538       throw new RuntimeException("Why does the file exist?");
539     }
540     return file;
541   }

544   /**
545    * get the discovery local file with unique extension
546    * @param localFileName
547    * @param isTmpFile true if .discoverytmp should be added to extension
548    * @return the local file
549    */
550   private static File discoveryLocalFileUniqueHelper(String localFileName, boolean isTmpFile) {
551 
552     String directoryName = GrouperClientUtils.cacheDirectoryName();
553 
554     if (localFileName.contains("/")) {
555       throw new RuntimeException("Local file cannot contain / : " + localFileName);
556     }
557 
558     if (localFileName.contains("\\")) {
559       throw new RuntimeException("Local file cannot contain \\ : " + localFileName);
560     }
561     
562     //lets change the local file name
563     int dotIndex = localFileName.lastIndexOf('.');
564 
565     if (dotIndex == -1) {
566       throw new RuntimeException("Local filename must have a dot in it! " + localFileName);
567     }
568 
569     String fileNamePrefix = localFileName.substring(0, dotIndex);
570     String fileNameSuffix = localFileName.substring(dotIndex, localFileName.length());
571 
572     DateFormat dateFormat = new SimpleDateFormat(TEMP_FILE_DATE_FORMAT);
573     
574     //file.ext would become
575     //file_20120102_132414_123_sd43sdf.ext
576     
577     String pathname = directoryName + File.separator + fileNamePrefix + "_" + dateFormat.format(new Date()) + "_" + GrouperClientUtils.uniqueId() + fileNameSuffix;
578     
579     if (isTmpFile) {
580       //file.ext would become
581       //file_20120102_132414_123_sd43sdf.ext.discoverytmp
582       pathname += DISCOVERYTMP;
583     }
584     return new File(pathname);
585 
586   }

588   /**
589    * contact the discovery server(s) to get the file
590    * @param fileNameOnServer
591    * @param localFileName should contain no slashes, just the filename locally which should conflict with no other filenames
592    * @return the file or null if problem (and log problem)
593    */
594   static File retrieveFileFromDiscoveryServer(final String fileNameOnServer, final String localFileName) {
595 
596     if (!hasDiscovery()) {
597       return null;
598     }
599     
600     //make sure the failover client is configured
601     configureFailoverClientOnce();
602     
603     File file = FailoverClient.failoverLogic(DiscoveryClient.DISCOVERY_CLIENT_CONNECTION_TYPE, new FailoverLogic<File>() {
604 
605       /**
606        * logic to get the discovery file
607        */
608       // @Override
609       public File logic(FailoverLogicBean failoverLogicBean) {
610         
611         Map<String, Object> logMap = log.isDebugEnabled() ? new LinkedHashMap<String, Object>() : null;
612 
613         if (logMap != null) {
614           logMap.put("method", "DiscoveryClient.retrieveFileFromDiscoveryServer.logic");
615         }
616 
617         //this is the URL
618         String url = failoverLogicBean.getConnectionName();
619                 
620         //strip last slash
621         url = GrouperClientUtils.stripEnd(url, "/");
622         
623         String fileNameNoSlash = GrouperClientUtils.stripStart(fileNameOnServer, "/");
624         
625         String fullUrl = url + "/" + fileNameNoSlash;
626 
627         if (logMap != null) {
628           logMap.put("fullUrl", fullUrl);
629         }
630 
631         //see if invalid SSL
632         String httpsSocketFactoryName = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.https.customSocketFactory");
633         
634         //is there overhead here?  should only do this once?
635         //perhaps give a custom factory
636         if (!GrouperClientUtils.isBlank(httpsSocketFactoryName)) {
637           Class<? extends SecureProtocolSocketFactory> httpsSocketFactoryClass = GrouperClientUtils.forName(httpsSocketFactoryName);
638           SecureProtocolSocketFactory httpsSocketFactoryInstance = GrouperClientUtils.newInstance(httpsSocketFactoryClass);
639           Protocol easyhttps = new Protocol("https", httpsSocketFactoryInstance, 443);
640           Protocol.registerProtocol("https", easyhttps);
641         }
642 
643         HttpClient httpClient = new HttpClient();
644 
645         DefaultHttpParams.getDefaultParams().setParameter(
646             HttpMethodParams.RETRY_HANDLER, new DefaultHttpMethodRetryHandler(0, false));
647 
648         int soTimeoutMillis = GrouperClientConfig.retrieveConfig().propertyValueInt(
649             "grouperClient.discovery.httpSocketTimeoutMillis", 90000);
650 
651         httpClient.getParams().setSoTimeout(soTimeoutMillis);
652         httpClient.getParams().setParameter(HttpMethodParams.HEAD_BODY_CHECK_TIMEOUT, soTimeoutMillis);
653 
654         int connectionManagerMillis = GrouperClientConfig.retrieveConfig().propertyValueInt(
655             "grouperClient.discovery.httpConnectionManagerTimeoutMillis", 90000);
656 
657         httpClient.getParams().setConnectionManagerTimeout(connectionManagerMillis);
658 
659         String user = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.discover.user");
660         
661         if (logMap != null) {
662           logMap.put("user", user);
663         }
664         
665         if (!GrouperClientUtils.isBlank(user)) {
666         
667           httpClient.getParams().setAuthenticationPreemptive(true);
668   
669           boolean disableExternalFileLookup = GrouperClientConfig.retrieveConfig().propertyValueBooleanRequired(
670               "encrypt.disableExternalFileLookup");
671           
672           //lets lookup if file
673           String pass = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.discovery.password");
674           String passFromFile = GrouperClientUtils.readFromFileIfFile(pass, disableExternalFileLookup);
675   
676           String passPrefix = null;
677   
678           if (!GrouperClientUtils.equals(pass, passFromFile)) {
679   
680             passPrefix = "Discovery pass: reading encrypted value from file: " + pass;
681   
682             String encryptKey = GrouperClientUtils.encryptKey();
683             pass = new Crypto(encryptKey).decrypt(passFromFile);
684             
685           } else {
686             passPrefix = "Discovery pass: reading scalar value from grouper.client.properties";
687           }
688           
689           if (GrouperClientConfig.retrieveConfig().propertyValueBoolean("grouperClient.logging.logMaskedPassword", false)) {
690             if (logMap != null) {
691               logMap.put(passPrefix, GrouperClientUtils.repeat("*", pass.length()));
692             }
693           }
694   
695           Credentials defaultcreds = new UsernamePasswordCredentials(user, pass);
696   
697           //set auth scope to null and negative so it applies to all hosts and ports
698           httpClient.getState().setCredentials(new AuthScope(null, -1), defaultcreds);
699         }
700 
701         GetMethod getMethod = new GetMethod(fullUrl);
702 
703         getMethod.setRequestHeader("Connection", "close");
704         
705         File file = discoveryLocalFileUnique(localFileName, true);
706         
707         try {
708 
709           int responseCodeInt = httpClient.executeMethod(getMethod);
710 
711           if (responseCodeInt != 200) {
712             throw new RuntimeException("Expected 200, but received response code: " + responseCodeInt);
713           }
714 
715           synchronized (synchronizedObjectBasedOnName(localFileName)) {
716             InputStream inputStream = null;
717             OutputStream outputStream = null;
718             try {
719 
720               inputStream = getMethod.getResponseBodyAsStream();
721               GrouperClientUtils.deleteFile(file);
722 
723               outputStream = new FileOutputStream(file);
724 
725               GrouperClientUtils.copy(inputStream, outputStream);
726 
727             } catch (RuntimeException re) {
728               try {
729                 GrouperClientUtils.deleteFile(file);
730               } catch (Exception e) {
731                 //dont pre-empt the other exception
732                 log.error("Cant delete file: " + file.getAbsolutePath(), e);
733               }
734               throw re;
735             } finally {
736               GrouperClientUtils.closeQuietly(inputStream);
737               GrouperClientUtils.closeQuietly(outputStream);
738             }
739 
740           }
741 
742         } catch (Exception exception) {
743 
744           throw new RuntimeException("Problem with url: " + fullUrl + ", and local file: " + file.getAbsolutePath(), exception);
745           
746         }
747         
748         return file;
749       }
750       
751       
752     });
753     
754     //lets get the non tmp version
755     File realFile = new File(file.getAbsolutePath().substring(0, file.getAbsolutePath().length() - DISCOVERYTMP.length()));
756 
757     //we made it, lets delete the real file if it is there, and copy the new one to the old one
758     //note, if using command line, this synchronization will not work, but try anyways...
759     synchronized (synchronizedObjectBasedOnName(localFileName)) {
760 
761       GrouperClientUtils.renameTo(file, realFile);
762     }
763     
764     cleanoutOldFiles();
765     
766     return realFile;
767 
768   }

770   /** local cache date pattern */
771   private static Pattern localCacheDatePattern = Pattern.compile("(.*)_(\\d{8}_\\d{6}_\\d{3})_.*(\\..*)");
772   
773   /**
774    * if the file is a tmp file and more than 20 minutes old, or if the file is a non temp file, and more than 20
775    * min old and there one that is newer (and the one that is newer is 20 minutes old), then delete it.  
776    * only log exceptions since there could be two processes doing the same thing and step on toes
777    * 
778    */
779   static void cleanoutOldFiles() {
780     
781     try {
782       File cacheDirectory = new File(GrouperClientUtils.cacheDirectoryName());
783       
784       File[] files = cacheDirectory.listFiles();
785 
786       // non tmp file catalog to see if we can delete non tmp file yet
787       Map<String, Map<Date, File>> nonTmpFileCatalog = new HashMap<String, Map<Date, File>>();
788       
789       //lets look for tmp files that are 20 minutes old or older
790       for (File file : files) {
791 
792         //the filename is something like this:
793         String fileName = file.getName();
794 
795         //file.ext would become
796         //file_20120102_132414_123_sd43sdf.ext.discoverytmp or file_20120102_132414_123_sd43sdf.ext
797         Matcher matcher = localCacheDatePattern.matcher(fileName);
798 
799         if (!matcher.matches()) {
800           continue;
801         }
802         String datePart = matcher.group(2);
803 
804         DateFormat dateFormat = new SimpleDateFormat(TEMP_FILE_DATE_FORMAT);
805         Date date = dateFormat.parse(datePart);
806 
807         //see if more than 20 minutes old
808         if (System.currentTimeMillis() - date.getTime() < (long)20 * 60 * 1000) {
809           continue;
810         }
811         
812         if (file.getAbsolutePath().endsWith(DISCOVERYTMP)) {
813 
814           try {
815             GrouperClientUtils.deleteFile(file);
816           } catch (Exception e) {
817             log.error("Cant delete file: " + file.getAbsolutePath(), e);
818           }
819 
820         } else {
821 
822           //lets catalog the non tmp files
823           String prefixFileName = matcher.group(1);
824           String suffixFileName = matcher.group(3);
825           String nonTmpFileName = prefixFileName + suffixFileName;
826           
827           //get the map of date to file in tree map so it is sorted
828           Map<Date, File> mapOfDateToFile = nonTmpFileCatalog.get(nonTmpFileName);
829           
830           if (mapOfDateToFile == null) {
831             mapOfDateToFile = new TreeMap<Date, File>();
832             nonTmpFileCatalog.put(nonTmpFileName, mapOfDateToFile);
833           }
834 
835           //not sure why it would exist, but if it does, delete it
836           if (mapOfDateToFile.containsKey(date)) {
837             try {
838               GrouperClientUtils.deleteFile(file);
839             } catch (Exception e) {
840               log.error("Cant delete file: " + file.getAbsolutePath(), e);
841             }
842             
843           }
844           
845           //add an entry for this
846           mapOfDateToFile.put(date, file);
847           
848         }
849       }
850       
851       //now go through the map and look for things to delete
852       for (Map<Date, File> mapOfDateToFile : nonTmpFileCatalog.values()) {
853         
854         if (mapOfDateToFile.size() <= 1) {
855           continue;
856         }
857         
858         //dont delete the last one, that one is newest
859         //these files are ordered by download date newest last
860         int index = 0;
861         Collection<File> filesToDelete = mapOfDateToFile.values();
862         for (File file : filesToDelete) {
863           if (index < filesToDelete.size()-1) {
864             try {
865               GrouperClientCommonUtils.deleteFile(file);
866             } catch (Exception e) {
867               log.error("Cant delete file: " + file.getAbsolutePath(), e);
868             }
869           }
870             
871           index++;
872         }
873       }
874       
875     } catch (Exception e) {
876       log.error("Cant clean log directory!", e);
877     }
878   }
