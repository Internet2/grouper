294   /**
295    * execute the call and return the results.  If there is a problem calling the service, an
296    * exception will be thrown
297    * 
298    * @return the results
299    */
300   public WsAssignPermissionsResults execute() {
301     this.validate();
302     WsAssignPermissionsResults wsAssignPermissionsResults = null;
303     try {
304       //Make the body of the request, in this case with beans and marshaling, but you can make
305       //your request document in whatever language or way you want
306       WsRestAssignPermissionsRequest assignPermissions = new WsRestAssignPermissionsRequest();
307 
308       assignPermissions.setActAsSubjectLookup(this.actAsSubject);
309 
310       //########### ATTRIBUTE DEF NAMES
311       List<WsAttributeDefNameLookup> permissionDefNameLookups = new ArrayList<WsAttributeDefNameLookup>();
312       //add names and/or uuids
313       for (String attributeDefNameName : this.permissionDefNameNames) {
314         permissionDefNameLookups.add(new WsAttributeDefNameLookup(attributeDefNameName, null));
315       }
316       for (String attributeDefNameUuid : this.permissionDefNameUuids) {
317         permissionDefNameLookups.add(new WsAttributeDefNameLookup(null, attributeDefNameUuid));
318       }
319       for (Long attributeDefNameIdIndex : this.permissionDefNameIdIndexes) {
320         permissionDefNameLookups.add(new WsAttributeDefNameLookup(null, null, attributeDefNameIdIndex.toString()));
321       }
322       if (GrouperClientUtils.length(permissionDefNameLookups) > 0) {
323         assignPermissions.setPermissionDefNameLookups(GrouperClientUtils.toArray(permissionDefNameLookups, WsAttributeDefNameLookup.class));
324       }
325 
326       //########### GROUPS
327       List<WsGroupLookup> roleLookups = new ArrayList<WsGroupLookup>();
328       //add names and/or uuids
329       for (String ownerGroupName : this.roleNames) {
330         roleLookups.add(new WsGroupLookup(ownerGroupName, null));
331       }
332       for (String ownerGroupUuid : this.roleUuids) {
333         roleLookups.add(new WsGroupLookup(null, ownerGroupUuid));
334       }
335       for (Long ownerGroupIdIndex : this.roleIdIndexes) {
336         roleLookups.add(new WsGroupLookup(null, null, ownerGroupIdIndex.toString()));
337       }
338       if (GrouperClientUtils.length(roleLookups) > 0) {
339         assignPermissions.setRoleLookups(GrouperClientUtils.toArray(roleLookups, WsGroupLookup.class));
340       }
341 
342       //############# MEMBERSHIP ANY LOOKUPS
343       if (GrouperClientUtils.length(this.subjectRoleLookups) > 0) {
344         assignPermissions.setSubjectRoleLookups(GrouperClientUtils.toArray(this.subjectRoleLookups, WsMembershipAnyLookup.class));
345       }
346       
347       //############# REPLACE STUFF
348       if (GrouperClientUtils.length(this.actionsToReplace) > 0) {
349         assignPermissions.setActionsToReplace(GrouperClientUtils.toArray(this.actionsToReplace, String.class));
350       }
351       List<WsAttributeDefLookup> attributeDefLookupsToReplace = new ArrayList<WsAttributeDefLookup>();
352       //add names and/or uuids
353       for (String attributeDefNameToReplace : this.attributeDefNamesToReplace) {
354         attributeDefLookupsToReplace.add(new WsAttributeDefLookup(attributeDefNameToReplace, null));
355       }
356       for (String attributeDefUuidToReplace : this.attributeDefUuidsToReplace) {
357         attributeDefLookupsToReplace.add(new WsAttributeDefLookup(null, attributeDefUuidToReplace));
358       }
359       for (Long attributeDefIdIndexToReplace : this.attributeDefIdIndexesToReplace) {
360         attributeDefLookupsToReplace.add(new WsAttributeDefLookup(null, null, attributeDefIdIndexToReplace.toString()));
361       }
362       if (GrouperClientUtils.length(attributeDefLookupsToReplace) > 0) {
363         assignPermissions.setAttributeDefsToReplace(GrouperClientUtils.toArray(attributeDefLookupsToReplace, WsAttributeDefLookup.class));
364       }
365 
366       
367       if (this.includeGroupDetail != null) {
368         assignPermissions.setIncludeGroupDetail(this.includeGroupDetail ? "T" : "F");
369       }
370 
371       if (this.includeSubjectDetail != null) {
372         assignPermissions.setIncludeSubjectDetail(this.includeSubjectDetail ? "T" : "F");
373       }
374       
375       if (this.assignmentDisabledTime != null) {
376         String disabledTime = GrouperClientUtils.dateToString(this.assignmentDisabledTime);
377         assignPermissions.setAssignmentDisabledTime(disabledTime);
378       }
379 
380       if (this.assignmentEnabledTime != null) {
381         String enabledTime = GrouperClientUtils.dateToString(this.assignmentEnabledTime);
382         assignPermissions.setAssignmentEnabledTime(enabledTime);
383       }
384 
385       assignPermissions.setAssignmentNotes(this.assignmentNotes);
386       assignPermissions.setPermissionAssignOperation(this.permissionAssignOperation);
387       assignPermissions.setDelegatable(this.delegatable);
388 
389       if (this.disallowed != null) {
390         assignPermissions.setDisallowed(this.disallowed ? "T" : "F");
391       }
392 
393       if (GrouperClientUtils.length(this.attributeAssignLookups) > 0) {
394         assignPermissions.setWsAttributeAssignLookups(GrouperClientUtils.toArray(
395             this.attributeAssignLookups, WsAttributeAssignLookup.class));
396       }
397 
398       assignPermissions.setPermissionType(this.permissionType);
399 
400       //add params if there are any
401       if (this.params.size() > 0) {
402         assignPermissions.setParams(GrouperClientUtils.toArray(this.params, WsParam.class));
403       }
404 
405       if (this.subjectAttributeNames.size() > 0) {
406         assignPermissions.setSubjectAttributeNames(
407             GrouperClientUtils.toArray(this.subjectAttributeNames, String.class));
408       }
409 
410       if (GrouperClientUtils.length(this.actions) > 0) {
411         assignPermissions.setActions(GrouperClientUtils.toArray(this.actions, String.class));
412       }
413 
414       GrouperClientWs grouperClientWs = new GrouperClientWs();
415 
416       //kick off the web service
417       wsAssignPermissionsResults = (WsAssignPermissionsResults)
418         grouperClientWs.executeService("permissionAssignments", 
419             assignPermissions, "getPermissionAssignments", this.clientVersion, false);
420 
421       String resultMessage = wsAssignPermissionsResults.getResultMetadata().getResultMessage();
422       grouperClientWs.handleFailure(wsAssignPermissionsResults, null, resultMessage);
423       
424     } catch (Exception e) {
425       GrouperClientUtils.convertToRuntimeException(e);
426     }
427     return wsAssignPermissionsResults;
428     
429   }
