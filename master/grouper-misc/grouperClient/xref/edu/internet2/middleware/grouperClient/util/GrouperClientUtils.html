71    /**
72     * @param theClass
73     * @return the log
74     */
75    public static Log retrieveLog(Class<?> theClass) {
76  
77      Log theLog = LogFactory.getLog(theClass);
78      
79      //if this isnt here, dont configure yet
80      if (isBlank(GrouperClientConfig.retrieveConfig().propertyValueString("encrypt.disableExternalFileLookup"))
81          || theClass.equals(GrouperClientCommonUtils.class)) {
82        return new GrouperClientLog(theLog);
83      }
84      
85      if (!configuredLogs) {
86        String logLevel = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.logging.logLevel");
87        String logFile = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.logging.logFile");
88        String grouperClientLogLevel = GrouperClientConfig.retrieveConfig().propertyValueString(
89            "grouperClient.logging.grouperClientOnly.logLevel");
90        
91        boolean hasLogLevel = !isBlank(logLevel);
92        boolean hasLogFile = !isBlank(logFile);
93        boolean hasGrouperClientLogLevel = !isBlank(grouperClientLogLevel);
94        
95        if (hasLogLevel || hasLogFile) {
96          if (theLog instanceof Jdk14Logger) {
97            Jdk14Logger jdkLogger = (Jdk14Logger) theLog;
98            Logger logger = jdkLogger.getLogger();
99            long timeToLive = 60;
100           while (logger.getParent() != null && timeToLive-- > 0) {
101             //this should be root logger
102             logger = logger.getParent();
103           }
104   
105           if (length(logger.getHandlers()) == 1) {
106   
107             //remove console appender if only one
108             if (logger.getHandlers()[0].getClass() == ConsoleHandler.class) {
109               logger.removeHandler(logger.getHandlers()[0]);
110             }
111           }
112   
113           if (length(logger.getHandlers()) == 0) {
114             Handler handler = null;
115             if (hasLogFile) {
116               try {
117                 handler = new FileHandler(logFile, true);
118               } catch (IOException ioe) {
119                 throw new RuntimeException(ioe);
120               }
121             } else {
122               handler = new ConsoleHandler();
123             }
124             handler.setFormatter(new SimpleFormatter());
125             handler.setLevel(Level.ALL);
126             logger.addHandler(handler);
127 
128             logger.setUseParentHandlers(false);
129           }
130           
131           if (hasLogLevel) {
132             Level level = Level.parse(logLevel);
133             
134             logger.setLevel(level);
135 
136           }
137         }
138       }
139       
140       if (hasGrouperClientLogLevel) {
141         Level level = Level.parse(grouperClientLogLevel);
142         Log grouperClientLog = LogFactory.getLog("edu.internet2.middleware.grouperClient");
143         if (grouperClientLog instanceof Jdk14Logger) {
144           Jdk14Logger jdkLogger = (Jdk14Logger) grouperClientLog;
145           Logger logger = jdkLogger.getLogger();
146           logger.setLevel(level);
147         }
148       }
149       
150       configuredLogs = true;
151     }
152     
153     return new GrouperClientLog(theLog);
154     
155   }

244     /**
245      * retrieve class if class
246      * @param name
247      * @return class
248      */
249     private static Object retrieveClass(String name) {
250       if (isBlank(name)) {
251         return null;
252       }
253       
254       //see if fully qualified class
255       
256       Boolean knowsIfClass = jexlKnowsIfClass.get(name);
257       
258       //see if knows answer
259       if (knowsIfClass != null) {
260         //return class or null
261         return jexlClass.get(name);
262       }
263       
264       //see if valid class
265       if (jexlClassPattern.matcher(name).matches()) {
266         
267         jexlKnowsIfClass.put(name, true);
268         //try to load
269         try {
270           Class<?> theClass = Class.forName(name);
271           jexlClass.put(name, theClass);
272           return theClass;
273         } catch (Exception e) {
274           LOG.info("Cant load what looks like class: " + name, e);
275           //this is ok I guess, dont rethrow, not sure it is a class
276         }
277       }
278       return null;
279       
280     }

283      * @see edu.internet2.middleware.grouperClientExt.org.apache.commons.jexl2.MapContext#get(java.lang.String)

287       
288       //see if registered      
289       Object object = super.get(name);
290       
291       if (object != null) {

294       return retrieveClass(name);

326   /**
327    * substitute an EL for objects
328    * @param stringToParse
329    * @param variableMap
330    * @param allowStaticClasses if true allow static classes not registered with context
331    * @param silent if silent mode, swallow exceptions (warn), and dont warn when variable not found
332    * @param lenient false if undefined variables should throw an exception.  if lenient is true (default)
333    * then undefined variables are null
334    * @param logOnNull if null output of substitution should be logged
335    * @return the string
336    */
337   public static String substituteExpressionLanguage(String stringToParse, 
338       Map<String, Object> variableMap, boolean allowStaticClasses, boolean silent, boolean lenient, boolean logOnNull) {
339     if (isBlank(stringToParse)) {
340       return stringToParse;
341     }
342     String overallResult = null;
343     Exception exception = null;
344     try {
345       JexlContext jc = allowStaticClasses ? new ElMapContext() : new MapContext();
346 
347       int index = 0;
348       
349       variableMap = nonNull(variableMap);
350       
351       for (String key: variableMap.keySet()) {
352         jc.set(key, variableMap.get(key));
353       }
354       
355       //allow utility methods
356       jc.set("elUtils", new GcElUtilsSafe());
357       //if you add another one here, add it in the logs below
358       
359       // matching ${ exp }   (non-greedy)
360       Pattern pattern = Pattern.compile("\\$\\{(.*?)\\}");
361       Matcher matcher = pattern.matcher(stringToParse);
362       
363       StringBuilder result = new StringBuilder();
364   
365       //loop through and find each script
366       while(matcher.find()) {
367         result.append(stringToParse.substring(index, matcher.start()));
368         
369         //here is the script inside the curlies
370         String script = matcher.group(1);
371         
372         index = matcher.end();
373   
374         if (script.contains("{")) {
375           //we need to match up some curlies here...
376           int scriptStart = matcher.start(1);
377           int openCurlyCount = 0;
378           for (int i=scriptStart; i<stringToParse.length();i++) {
379             char curChar = stringToParse.charAt(i);
380             if (curChar == '{') {
381               openCurlyCount++;
382             }
383             if (curChar == '}') {
384               openCurlyCount--;
385               //negative 1 since we need to get to the close of the parent one...
386               if (openCurlyCount <= -1) {
387                 script = stringToParse.substring(scriptStart, i);
388                 index = i+1;
389                 break;
390               }
391             }
392           }
393         }
394         
395         JexlEngine jexlEngine = new JexlEngine();
396         jexlEngine.setSilent(silent);
397         jexlEngine.setLenient(lenient);
398   
399         Expression e = jexlEngine.createExpression(script);
400   
401         //this is the result of the evaluation
402         Object o = null;
403         
404         try {
405           o = e.evaluate(jc);
406         } catch (JexlException je) {
407           //exception-scrape to see if missing variable
408           if (!lenient && trimToEmpty(je.getMessage()).contains("undefined variable")) {
409             //clean up the message a little bit
410             // e.g. edu.internet2.middleware.grouper.util.GrouperUtil.substituteExpressionLanguage@8846![0,6]: 'amount < 50000 && amount2 < 23;' undefined variable amount
411             String message = je.getMessage();
412             //Pattern exceptionPattern = Pattern.compile("^" + GrouperUtil.class.getName() + "\\.substituteExpressionLanguage.*?]: '(.*)");
413             Pattern exceptionPattern = Pattern.compile("^.*undefined variable (.*)");
414             Matcher exceptionMatcher = exceptionPattern.matcher(message);
415             if (exceptionMatcher.matches()) {
416               //message = "'" + exceptionMatcher.group(1);
417               message = "variable '" + exceptionMatcher.group(1) + "' is not defined in script: '" + script + "'";
418             }
419             throw new GcExpressionLanguageMissingVariableException(message, je);
420           }
421           throw je;
422         }
423           
424         if (o == null) {
425           if (logOnNull) {
426             LOG.warn("expression returned null: " + script + ", in pattern: '" + stringToParse + "', available variables are: "
427                 + toStringForLog(variableMap.keySet()));
428           } else {
429             if (LOG.isDebugEnabled()) {
430               LOG.debug("expression returned null: " + script + ", in pattern: '" + stringToParse + "', available variables are: "
431                   + toStringForLog(variableMap.keySet()));
432             }            
433           }
434           o = "";
435         }
436         
437         if (o instanceof RuntimeException) {
438           throw (RuntimeException)o;
439         }
440         
441         result.append(o);
442         
443       }
444       
445       result.append(stringToParse.substring(index, stringToParse.length()));
446       overallResult = result.toString();
447       return overallResult;
448       
449     } catch (Exception e) {
450       exception = e;
451       if (e instanceof GcExpressionLanguageMissingVariableException) {
452         throw (GcExpressionLanguageMissingVariableException)e;
453       }
454       throw new RuntimeException("Error substituting string: '" + stringToParse + "'", e);
455     } finally {
456       if (LOG.isDebugEnabled()) {
457         Set<String> keysSet = new LinkedHashSet<String>(nonNull(variableMap).keySet());
458         keysSet.add("elUtils");
459         StringBuilder logMessage = new StringBuilder();
460         logMessage.append("Subsituting EL: '").append(stringToParse).append("', and with env vars: ");
461         String[] keys = keysSet.toArray(new String[]{});
462         for (int i=0;i<keys.length;i++) {
463           logMessage.append(keys[i]);
464           if (i != keys.length-1) {
465             logMessage.append(", ");
466           }
467         }
468         logMessage.append(" with result: '" + overallResult + "'");
469         if (exception != null) {
470           logMessage.append(", and exception: " + exception + ", " + getFullStackTrace(exception));
471         }
472         LOG.debug(logMessage.toString());
473       }
474     }
475   }

479    * @param wsSubject subject
480    * @param attributeNames list of attribute names in the subject
481    * @param attributeName to query
482    * @return the value or null

484   public static String subjectAttributeValue(WsSubject wsSubject, String[] attributeNames, String attributeName) {
485     
486     if (GrouperClientUtils.equals("subject__id", attributeName)) {
487       return wsSubject.getId();
488     }
489

491       return wsSubject.getName();
492     }
493

495       
496       if (GrouperClientUtils.equalsIgnoreCase(attributeName, attributeNames[i])
497           && GrouperClientUtils.length(wsSubject.getAttributeValues()) > i) {
498         //got it
499         return wsSubject.getAttributeValue(i);
500       }
501     }
502     return null;
503   }

560   /**
561    * name of the cache directory without trailing slash
562    * @return the name of the cache directory
563    */
564   public static String cacheDirectoryName() {
565     
566     if (cacheDirectoryName == null) {
567       String directoryName = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.cacheDirectory");
568       if (GrouperClientCommonUtils.isBlank(directoryName)) {
569         throw new RuntimeException("grouperClient.cacheDirectory is required in grouper.client.properties");
570       }
571   
572       directoryName = GrouperClientCommonUtils.stripEnd(directoryName, "/");
573       directoryName = GrouperClientCommonUtils.stripEnd(directoryName, "\\");
574       
575       File discoveryDir = new File(directoryName);
576       directoryName = discoveryDir.getAbsolutePath();
577       if (directoryName.endsWith("/.")) {
578         directoryName = directoryName.substring(0, directoryName.length()-2);
579       }
580       if (directoryName.endsWith("\\.")) {
581         directoryName = directoryName.substring(0, directoryName.length()-2);
582       }
583       cacheDirectoryName = directoryName;
584     }    
585     return cacheDirectoryName;
586   }

608   /**
609    * call HTTP with a url, optional request body, get a response body.  this assumes json by default
610    * @param urlSuffix is after the configured URL
611    * @param serviceAuthn is the config string that identifies the user/pass/url
612    * @param httpCallMethod HTTP
613    * @param body to send if applicable
614    * @return the response
615    */
616   public static HttpCallResponse httpCall(String urlSuffix, String serviceAuthn, HttpCallMethod httpCallMethod, String body) {
617     
618     //make sure right content type is in request (e.g. application/xhtml+xml
619     HttpClient httpClient = new HttpClient();
620 
621     DefaultHttpParams.getDefaultParams().setParameter(
622         HttpMethodParams.RETRY_HANDLER, new DefaultHttpMethodRetryHandler(0, false));
623 
624     httpClient.getParams().setAuthenticationPreemptive(true);
625     
626     int soTimeoutMillis = GrouperClientConfig.retrieveConfig().propertyValueIntRequired(
627         "grouperClient.webService.httpSocketTimeoutMillis");
628 
629     httpClient.getParams().setSoTimeout(soTimeoutMillis);
630     httpClient.getParams().setParameter(HttpMethodParams.HEAD_BODY_CHECK_TIMEOUT, soTimeoutMillis);
631 
632     int connectionManagerMillis = GrouperClientConfig.retrieveConfig().propertyValueIntRequired(
633         "grouperClient.webService.httpConnectionManagerTimeoutMillis");
634 
635     httpClient.getParams().setConnectionManagerTimeout(connectionManagerMillis);
636 
637     String user = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.webService." + serviceAuthn + ".user");
638 
639     boolean disableExternalFileLookup = GrouperClientConfig.retrieveConfig().propertyValueBooleanRequired(
640         "encrypt.disableExternalFileLookup");
641 
642     //lets lookup if file
643     String wsPass = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.webService." + serviceAuthn + ".pass");
644     String wsPassFromFile = GrouperClientUtils.readFromFileIfFile(wsPass, disableExternalFileLookup);
645 
646     String passPrefix = null;
647 
648     if (!GrouperClientUtils.equals(wsPass, wsPassFromFile)) {
649 
650       passPrefix = "WebService pass: reading encrypted value from file: " + wsPass;
651 
652       String encryptKey = GrouperClientUtils.encryptKey();
653       wsPass = new Crypto(encryptKey).decrypt(wsPassFromFile);
654       
655     } else {
656       passPrefix = "WebService pass: reading scalar value from grouper.client.properties";
657     }
658     
659     if (GrouperClientConfig.retrieveConfig().propertyValueBoolean("grouperClient.logging.logMaskedPassword", false)) {
660       LOG.debug(passPrefix + ": " + GrouperClientUtils.repeat("*", wsPass.length()));
661     }
662 
663     Credentials defaultcreds = new UsernamePasswordCredentials(user, wsPass);
664 
665     //set auth scope to null and negative so it applies to all hosts and ports
666     httpClient.getState().setCredentials(new AuthScope(null, -1), defaultcreds);
667 
668     String urlBase = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.webService." + serviceAuthn + ".url");
669     if (urlBase == null) {
670       throw new RuntimeException("urlBase is null");
671     }
672     
673     String url = urlBase + urlSuffix;
674 
675     String contentType = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.webService." + serviceAuthn + ".defaultRequestContentType");
676     contentType = GrouperClientUtils.defaultIfEmpty(contentType, "Application/json");
677 
678     HttpMethodBase httpMethodBase = null;
679     
680     if (httpCallMethod == null) {
681       throw new RuntimeException("You need to pass in an httpCallMethod");
682     }
683     
684     HttpCallResponse httpCallResponse = new HttpCallResponse();
685     httpCallResponse.setHttpCallMethod(httpCallMethod);
686     httpCallResponse.setContentType(contentType);
687     httpCallResponse.setUrl(url);
688     
689     //URL e.g. http://localhost:8093/grouper-ws/servicesRest/v1_3_000/...
690     //NOTE: aStem:aGroup urlencoded substitutes %3A for a colon
691     switch (httpCallMethod) {
692       case GET:
693         
694         httpMethodBase = new GetMethod(url);
695         if (!GrouperClientUtils.isBlank(body)) {
696           throw new RuntimeException("GET cannot have a body");
697         }
698         break;
699       case POST:
700         
701         httpMethodBase = new PostMethod(url);
702         if (!GrouperClientUtils.isBlank(body)) {
703           try {
704             ((PostMethod)httpMethodBase).setRequestEntity(new StringRequestEntity(body, "application/json", "UTF-8"));
705           } catch (UnsupportedEncodingException uee) {
706             throw new RuntimeException(uee);
707           }
708         }
709 
710         break;
711         
712       case PUT:
713         
714         httpMethodBase = new PutMethod(url);
715         if (!GrouperClientUtils.isBlank(body)) {
716           try {
717             ((PutMethod)httpMethodBase).setRequestEntity(new StringRequestEntity(body, "application/json", "UTF-8"));
718           } catch (UnsupportedEncodingException uee) {
719             throw new RuntimeException(uee);
720           }
721         }
722 
723         break;
724         
725       case DELETE:
726         
727         httpMethodBase = new DeleteMethod(url);
728         if (!GrouperClientUtils.isBlank(body)) {
729           throw new RuntimeException("DELETE cannot have a body");
730         }
731 
732         break;
733         
734       default: 
735         throw new RuntimeException("Not expecting method: " + httpCallMethod);
736     }
737     
738     httpMethodBase.addRequestHeader("Content-Type", contentType);
739     
740     //no keep alive so response is easier to indent for tests
741     httpMethodBase.setRequestHeader("Connection", "close");
742     
743     try {
744 
745       int responseCodeInt = httpClient.executeMethod(httpMethodBase);
746 
747       httpCallResponse.setHttpResponseCode(responseCodeInt);
748 
749     } catch (IOException ioe) {
750       throw new RuntimeException("Error in url: " + url + ", method: " + httpCallMethod + ", body: " + GrouperClientUtils.abbreviate(body, 5000), ioe);
751     }
752     
753     String theResponse = GrouperClientUtils.responseBodyAsString(httpMethodBase);
754     
755     httpCallResponse.setResponseBody(theResponse);
756     
757     return httpCallResponse;
758   }
