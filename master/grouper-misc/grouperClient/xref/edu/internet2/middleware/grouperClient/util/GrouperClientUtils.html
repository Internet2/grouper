54    /**
55     * @param theClass
56     * @return the log
57     */
58    public static Log retrieveLog(Class<?> theClass) {
59  
60      Log theLog = LogFactory.getLog(theClass);
61      
62      //if this isnt here, dont configure yet
63      if (isBlank(GrouperClientConfig.retrieveConfig().propertyValueString("encrypt.disableExternalFileLookup"))
64          || theClass.equals(GrouperClientCommonUtils.class)) {
65        return new GrouperClientLog(theLog);
66      }
67      
68      if (!configuredLogs) {
69        String logLevel = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.logging.logLevel");
70        String logFile = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.logging.logFile");
71        String grouperClientLogLevel = GrouperClientConfig.retrieveConfig().propertyValueString(
72            "grouperClient.logging.grouperClientOnly.logLevel");
73        
74        boolean hasLogLevel = !isBlank(logLevel);
75        boolean hasLogFile = !isBlank(logFile);
76        boolean hasGrouperClientLogLevel = !isBlank(grouperClientLogLevel);
77        
78        if (hasLogLevel || hasLogFile) {
79          if (theLog instanceof Jdk14Logger) {
80            Jdk14Logger jdkLogger = (Jdk14Logger) theLog;
81            Logger logger = jdkLogger.getLogger();
82            long timeToLive = 60;
83            while (logger.getParent() != null && timeToLive-- > 0) {
84              //this should be root logger
85              logger = logger.getParent();
86            }
87    
88            if (length(logger.getHandlers()) == 1) {
89    
90              //remove console appender if only one
91              if (logger.getHandlers()[0].getClass() == ConsoleHandler.class) {
92                logger.removeHandler(logger.getHandlers()[0]);
93              }
94            }
95    
96            if (length(logger.getHandlers()) == 0) {
97              Handler handler = null;
98              if (hasLogFile) {
99                try {
100                 handler = new FileHandler(logFile, true);
101               } catch (IOException ioe) {
102                 throw new RuntimeException(ioe);
103               }
104             } else {
105               handler = new ConsoleHandler();
106             }
107             handler.setFormatter(new SimpleFormatter());
108             handler.setLevel(Level.ALL);
109             logger.addHandler(handler);
110 
111             logger.setUseParentHandlers(false);
112           }
113           
114           if (hasLogLevel) {
115             Level level = Level.parse(logLevel);
116             
117             logger.setLevel(level);
118 
119           }
120         }
121       }
122       
123       if (hasGrouperClientLogLevel) {
124         Level level = Level.parse(grouperClientLogLevel);
125         Log grouperClientLog = LogFactory.getLog("edu.internet2.middleware.grouperClient");
126         if (grouperClientLog instanceof Jdk14Logger) {
127           Jdk14Logger jdkLogger = (Jdk14Logger) grouperClientLog;
128           Logger logger = jdkLogger.getLogger();
129           logger.setLevel(level);
130         }
131       }
132       
133       configuredLogs = true;
134     }
135     
136     return new GrouperClientLog(theLog);
137     
138   }

216   /** pattern to see if class or not */
217   private static Pattern jexlClassPattern = Pattern.compile("^[a-zA-Z0-9_.]*\\.[A-Z][a-zA-Z0-9_]*$");

227     /**
228      * retrieve class if class
229      * @param name
230      * @return class
231      */
232     private static Object retrieveClass(String name) {
233       if (isBlank(name)) {
234         return null;
235       }
236       
237       //see if fully qualified class
238       
239       Boolean knowsIfClass = jexlKnowsIfClass.get(name);
240       
241       //see if knows answer
242       if (knowsIfClass != null) {
243         //return class or null
244         return jexlClass.get(name);
245       }
246       
247       //see if valid class
248       if (jexlClassPattern.matcher(name).matches()) {
249         
250         jexlKnowsIfClass.put(name, true);
251         //try to load
252         try {
253           Class<?> theClass = Class.forName(name);
254           jexlClass.put(name, theClass);
255           return theClass;
256         } catch (Exception e) {
257           LOG.info("Cant load what looks like class: " + name, e);
258           //this is ok I guess, dont rethrow, not sure it is a class
259         }
260       }
261       return null;
262       
263     }
264     
265     /**
266      * @see edu.internet2.middleware.grouperClientExt.org.apache.commons.jexl2.MapContext#get(java.lang.String)
267      */
268     @Override
269     public Object get(String name) {
270       
271       //see if registered      
272       Object object = super.get(name);
273       
274       if (object != null) {
275         return object;
276       }
277       return retrieveClass(name);
278     }
279   
280     /**
281      * @see edu.internet2.middleware.grouperClientExt.org.apache.commons.jexl2.MapContext#has(java.lang.String)
282      */
283     @Override
284     public boolean has(String name) {
285       boolean superHas = super.has(name);
286       if (superHas) {
287         return true;
288       }
289       
290       return retrieveClass(name) != null;
291       
292     }

309   /**
310    * substitute an EL for objects
311    * @param stringToParse
312    * @param variableMap
313    * @param allowStaticClasses if true allow static classes not registered with context
314    * @param silent if silent mode, swallow exceptions (warn), and dont warn when variable not found
315    * @param lenient false if undefined variables should throw an exception.  if lenient is true (default)
316    * then undefined variables are null
317    * @param logOnNull if null output of substitution should be logged
318    * @return the string
319    */
320   public static String substituteExpressionLanguage(String stringToParse, 
321       Map<String, Object> variableMap, boolean allowStaticClasses, boolean silent, boolean lenient, boolean logOnNull) {
322     if (isBlank(stringToParse)) {
323       return stringToParse;
324     }
325     String overallResult = null;
326     Exception exception = null;
327     try {
328       JexlContext jc = allowStaticClasses ? new ElMapContext() : new MapContext();
329 
330       int index = 0;
331       
332       variableMap = nonNull(variableMap);
333       
334       for (String key: variableMap.keySet()) {
335         jc.set(key, variableMap.get(key));
336       }
337       
338       //allow utility methods
339       jc.set("elUtils", new GcElUtilsSafe());
340       //if you add another one here, add it in the logs below
341       
342       // matching ${ exp }   (non-greedy)
343       Pattern pattern = Pattern.compile("\\$\\{(.*?)\\}");
344       Matcher matcher = pattern.matcher(stringToParse);
345       
346       StringBuilder result = new StringBuilder();
347   
348       //loop through and find each script
349       while(matcher.find()) {
350         result.append(stringToParse.substring(index, matcher.start()));
351         
352         //here is the script inside the curlies
353         String script = matcher.group(1);
354         
355         index = matcher.end();
356   
357         if (script.contains("{")) {
358           //we need to match up some curlies here...
359           int scriptStart = matcher.start(1);
360           int openCurlyCount = 0;
361           for (int i=scriptStart; i<stringToParse.length();i++) {
362             char curChar = stringToParse.charAt(i);
363             if (curChar == '{') {
364               openCurlyCount++;
365             }
366             if (curChar == '}') {
367               openCurlyCount--;
368               //negative 1 since we need to get to the close of the parent one...
369               if (openCurlyCount <= -1) {
370                 script = stringToParse.substring(scriptStart, i);
371                 index = i+1;
372                 break;
373               }
374             }
375           }
376         }
377         
378         JexlEngine jexlEngine = new JexlEngine();
379         jexlEngine.setSilent(silent);
380         jexlEngine.setLenient(lenient);
381   
382         Expression e = jexlEngine.createExpression(script);
383   
384         //this is the result of the evaluation
385         Object o = null;
386         
387         try {
388           o = e.evaluate(jc);
389         } catch (JexlException je) {
390           //exception-scrape to see if missing variable
391           if (!lenient && trimToEmpty(je.getMessage()).contains("undefined variable")) {
392             //clean up the message a little bit
393             // e.g. edu.internet2.middleware.grouper.util.GrouperUtil.substituteExpressionLanguage@8846![0,6]: 'amount < 50000 && amount2 < 23;' undefined variable amount
394             String message = je.getMessage();
395             //Pattern exceptionPattern = Pattern.compile("^" + GrouperUtil.class.getName() + "\\.substituteExpressionLanguage.*?]: '(.*)");
396             Pattern exceptionPattern = Pattern.compile("^.*undefined variable (.*)");
397             Matcher exceptionMatcher = exceptionPattern.matcher(message);
398             if (exceptionMatcher.matches()) {
399               //message = "'" + exceptionMatcher.group(1);
400               message = "variable '" + exceptionMatcher.group(1) + "' is not defined in script: '" + script + "'";
401             }
402             throw new GcExpressionLanguageMissingVariableException(message, je);
403           }
404           throw je;
405         }
406           
407         if (o == null) {
408           if (logOnNull) {
409             LOG.warn("expression returned null: " + script + ", in pattern: '" + stringToParse + "', available variables are: "
410                 + toStringForLog(variableMap.keySet()));
411           } else {
412             if (LOG.isDebugEnabled()) {
413               LOG.debug("expression returned null: " + script + ", in pattern: '" + stringToParse + "', available variables are: "
414                   + toStringForLog(variableMap.keySet()));
415             }            
416           }
417           o = "";
418         }
419         
420         if (o instanceof RuntimeException) {
421           throw (RuntimeException)o;
422         }
423         
424         result.append(o);
425         
426       }
427       
428       result.append(stringToParse.substring(index, stringToParse.length()));
429       overallResult = result.toString();
430       return overallResult;
431       
432     } catch (Exception e) {
433       exception = e;
434       if (e instanceof GcExpressionLanguageMissingVariableException) {
435         throw (GcExpressionLanguageMissingVariableException)e;
436       }
437       throw new RuntimeException("Error substituting string: '" + stringToParse + "'", e);
438     } finally {
439       if (LOG.isDebugEnabled()) {
440         Set<String> keysSet = new LinkedHashSet<String>(nonNull(variableMap).keySet());
441         keysSet.add("elUtils");
442         StringBuilder logMessage = new StringBuilder();
443         logMessage.append("Subsituting EL: '").append(stringToParse).append("', and with env vars: ");
444         String[] keys = keysSet.toArray(new String[]{});
445         for (int i=0;i<keys.length;i++) {
446           logMessage.append(keys[i]);
447           if (i != keys.length-1) {
448             logMessage.append(", ");
449           }
450         }
451         logMessage.append(" with result: '" + overallResult + "'");
452         if (exception != null) {
453           logMessage.append(", and exception: " + exception + ", " + getFullStackTrace(exception));
454         }
455         LOG.debug(logMessage.toString());
456       }
457     }
458   }

460   /**
461    * get the attribute value of an attribute name of a subject
462    * @param wsSubject subject
463    * @param attributeNames list of attribute names in the subject
464    * @param attributeName to query
465    * @return the value or null
466    */
467   public static String subjectAttributeValue(WsSubject wsSubject, String[] attributeNames, String attributeName) {
468     
469     if (GrouperClientUtils.equals("subject__id", attributeName)) {
470       return wsSubject.getId();
471     }
472     
473     if (GrouperClientUtils.equals("subject__name", attributeName)) {
474       return wsSubject.getName();
475     }
476     
477     for (int i=0;i<GrouperClientUtils.length(attributeNames);i++) {
478       
479       if (GrouperClientUtils.equalsIgnoreCase(attributeName, attributeNames[i])
480           && GrouperClientUtils.length(wsSubject.getAttributeValues()) > i) {
481         //got it
482         return wsSubject.getAttributeValue(i);
483       }
484     }
485     return null;
486   }

492   public static String encryptKey() {
493     String encryptKey = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("encrypt.key");
494     
495     boolean disableExternalFileLookup = GrouperClientConfig.retrieveConfig().propertyValueBooleanRequired(
496         "encrypt.disableExternalFileLookup");
497     
498     //lets lookup if file
499     encryptKey = GrouperClientUtils.readFromFileIfFile(encryptKey, disableExternalFileLookup);
500     
501     //the server does this, so if the key is blank, it will still have something there, so be consistent
502     if (GrouperClientConfig.retrieveConfig().propertyValueBoolean("encrypt.encryptLikeServer", false)) {
503       
504       encryptKey += "w";
505     }
506     
507     return encryptKey;
508   }

510   /**
511    * name of the cache directory without trailing slash
512    * @return the name of the cache directory
513    */
514   public static String cacheDirectoryName() {
515     
516     if (cacheDirectoryName == null) {
517       String directoryName = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.cacheDirectory");
518       if (GrouperClientCommonUtils.isBlank(directoryName)) {
519         throw new RuntimeException("grouperClient.cacheDirectory is required in grouper.client.properties");
520       }
521   
522       directoryName = GrouperClientCommonUtils.stripEnd(directoryName, "/");
523       directoryName = GrouperClientCommonUtils.stripEnd(directoryName, "\\");
524       
525       File discoveryDir = new File(directoryName);
526       directoryName = discoveryDir.getAbsolutePath();
527       if (directoryName.endsWith("/.")) {
528         directoryName = directoryName.substring(0, directoryName.length()-2);
529       }
530       if (directoryName.endsWith("\\.")) {
531         directoryName = directoryName.substring(0, directoryName.length()-2);
532       }
533       cacheDirectoryName = directoryName;
534     }    
535     return cacheDirectoryName;
536   }
