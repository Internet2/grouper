40  /**
41   * All first level resources on a get request.
42   */
43  public enum VootWsRest {
44  
45    /** group get requests */
46    groups {
47  
48      /**
49       * Handle the incoming request based on url.
50       * 
51       * @param urlStrings not including the app name or servlet. for
52       *        http://localhost/grouper-ws/voot/groups/a:b the urlStrings
53       *        would be size two: {"groups", "subjectid"}
54       * @return the result object.
55       */
56      @Override
57      public Object service(List<String> urlStrings, VootRestHttpMethod vootRestHttpMethod,
58          Map<String, String[]> requestParameters) {
59        
60        if (vootRestHttpMethod != VootRestHttpMethod.GET) {
61          return new VootErrorResponse("Wrong method", "Not expecting method: " + vootRestHttpMethod.name());
62        }
63  
64        String userName = GrouperServiceUtils.popUrlString(urlStrings);
65        String sortBy = null;
66        if (requestParameters.get("sortBy") != null) {
67          sortBy = requestParameters.get("sortBy")[0];
68        }
69        int start = this.getStart(requestParameters);
70        int count = this.getCount(requestParameters);
71  
72        if (StringUtils.isBlank(userName)) {
73          String[] searchTerm = requestParameters.get("search");
74          return VootLogic.getGroups(StringUtils.join(searchTerm, "%"), sortBy, start, count);
75        }
76  
77        if (GrouperUtil.length(urlStrings) > 0) {
78          return new VootErrorResponse("Too many params", "Not expecting more URL strings: "
79              + GrouperUtil.toStringForLog(urlStrings));
80        }
81  
82        if (StringUtils.equals("@me", userName)) {
83          return VootLogic.getGroups(GrouperSession.staticGrouperSession().getSubject(), sortBy, start, count);
84        } else {
85          try {
86            return VootLogic.getGroups(SubjectFinder.findById(userName, true), sortBy, start, count);
87          } catch (SubjectNotFoundException e) {
88            return new VootErrorResponse("Subject error",
89                "Subject not found: " + userName + ", " + GrouperUtil.toStringForLog(urlStrings));
90          } catch (SubjectNotUniqueException e) {
91            return new VootErrorResponse("Subject error",
92                "Subject not unique: " + userName + ", " + GrouperUtil.toStringForLog(urlStrings));
93          }
94        }
95      }
96    },
97  
98    /** people get requests */
99    people {
100 
101     /**
102      * Handle the incoming request based on url.
103      * 
104      * @param urlStrings not including the app name or servlet. for
105      *        http://localhost/grouper-ws/voot/people/@me the urlStrings
106      *        would be size three: {"people", "subjectid", "a:b"}
107      * @return the result object.
108      */
109     @Override
110     public Object service(List<String> urlStrings, VootRestHttpMethod vootRestHttpMethod,
111         Map<String, String[]> requestParameters) {
112 
113       if (vootRestHttpMethod != VootRestHttpMethod.GET) {
114         throw new RuntimeException("Not expecting method: " + vootRestHttpMethod.name());
115       }
116 
117       String personId = GrouperServiceUtils.popUrlString(urlStrings);
118       String groupName = GrouperServiceUtils.popUrlString(urlStrings);
119       String sortBy = null;
120       if (requestParameters.get("sortBy") != null) {
121         sortBy = requestParameters.get("sortBy")[0];
122       }
123       int start = this.getStart(requestParameters);
124       int count = this.getCount(requestParameters);
125 
126       if (StringUtils.isBlank(personId)) {
127         return new VootErrorResponse("No username",
128             "Error: no userName passed, " + vootRestHttpMethod + ", " + GrouperUtil.toStringForLog(urlStrings));
129       }
130 
131       if (StringUtils.isBlank(groupName)) {
132         return new VootErrorResponse("No group name",
133             "Error: no group name passed, " + vootRestHttpMethod + ", " + GrouperUtil.toStringForLog(urlStrings));
134       }
135 
136       if (GrouperUtil.length(urlStrings) > 0) {
137         return new VootErrorResponse("Too many params",
138             "Not expecting more URL strings: " + GrouperUtil.toStringForLog(urlStrings));
139       }
140 
141       // get the members of that group
142       VootGroup vootGroup = new VootGroup();
143       vootGroup.setId(groupName);
144 
145       try {
146         if (StringUtils.equals("@me", personId)) {
147           return VootLogic.getMembers(GrouperSession.staticGrouperSession().getSubject(), vootGroup, sortBy, start, count);
148         } else {
149           return VootLogic.getMembers(SubjectFinder.findById(personId, true), vootGroup, sortBy, start, count);
150         }
151       } catch (SubjectNotFoundException e) {
152         return new VootErrorResponse("Subject error",
153             "Subject not found: " + personId + ", " + GrouperUtil.toStringForLog(urlStrings));
154       } catch (SubjectNotUniqueException e) {
155         return new VootErrorResponse("Subject error",
156             "Subject not unique: " + personId + ", " + GrouperUtil.toStringForLog(urlStrings));
157       } catch (GroupNotFoundException e) {
158         return new VootErrorResponse("Group not found", e.getMessage());
159       }
160     }
161   },
162 
163   /** blank or other request */
164   VOOT_BLANK {
165 
166     /**
167      * Returns an error for wrong method called.
168      * @return the error object. 
169      */
170     @Override
171     public Object service(List<String> urlStrings, VootRestHttpMethod vootRestHttpMethod,
172         Map<String, String[]> requestParameters) {
173       
174       if (vootRestHttpMethod != VootRestHttpMethod.GET) {
175         return new VootErrorResponse("Wrong params",  "Not expecting method: " + vootRestHttpMethod.name());
176       }
177       
178       return new VootErrorResponse("Wrong params", "Pass in a url string: groups or people");
179     }
180 
181   };
182 
183   /**
184    * Utility function to retrieve from the query string the startIndex parameter
185    * (if present).
186    * @return the integer value of the parameter passed. 
187    */
188   protected int getStart(Map<String, String[]> requestParameters) {
189     if (requestParameters.containsKey("startIndex")) {
190       return Integer.parseInt(requestParameters.get("startIndex")[0]);
191     }
192     return 0;
193   }
194 
195   /**
196    * Utility function to retrieve from the query string the count parameter
197    * (if present).
198    * @return the integer value of the parameter passed. 
199    */
200   protected int getCount(Map<String, String[]> requestParameters) {
201     if (requestParameters.containsKey("count")) {
202       return Integer.parseInt(requestParameters.get("count")[0]);
203     }
204     return -1;
205   }
206 
207   /**
208    * Handle the incoming request based on HTTP method.
209    * 
210    * @param urlStrings not including the app name or servlet. for
211    *        http://localhost/grouper-ws/servicesRest/groups/a:b the
212    *        urlStrings would be size two: {"group", "a:b"}
213    * @param vootRestHttpMethod is the method of the call.
214    * @param requestParameters the parameters passed to the request.
215    * @return the result object.
216    */
217   public abstract Object service(List<String> urlStrings, VootRestHttpMethod vootRestHttpMethod,
218       Map<String, String[]> requestParameters);
219 
220   /**
221    * Do a case-insensitive matching.
222    * 
223    * @param string the first commant to VOOT from query string.
224    * @param exceptionOnNotFound true if exception should be thrown on not found
225    * @return the enum or null or exception if not found.
226    * @throws RuntimeException if empty string passed.
227    * @throws GrouperRestInvalidRequest if there is a problem.
228    */
229   public static VootWsRest valueOfIgnoreCase(String string, boolean exceptionOnNotFound)
230       throws GrouperRestInvalidRequest {
231 
232     try { 
233       if (StringUtils.equals(VOOT_BLANK.name(), string)) {
234         throw new RuntimeException(VootWsRest.class.toString() + " cannot be " + VOOT_BLANK.name());
235       }
236   
237       // if blank, use the blank
238       if (StringUtils.isBlank(string)) {
239         string = VOOT_BLANK.name();
240       }
241   
242       return GrouperUtil.enumValueOfIgnoreCase(VootWsRest.class, string, exceptionOnNotFound);
243     } catch (RuntimeException e) {
244       if (exceptionOnNotFound) throw e;
245       return GrouperUtil.enumValueOfIgnoreCase(VootWsRest.class, VOOT_BLANK.name(), exceptionOnNotFound);
246     }
247   }
