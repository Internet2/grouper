66    /**
67     * Helper for find by approximate name queries.
68     * This isn't in 2.0.0.
69     * 
70     * @param name the name of the group to be searched
71     * @param scope the scope of the group to be searched
72     * @param currentNames search only within current names of the groups
73     * @param alternateNames search also in alternate names of the groups
74     * @param queryOptions options for the hibernate query
75     * @param typeOfGroups types of group to be included in the research
76     * @return Set of Group object representing found groups satisfying search parameters.
77     * @throws GrouperDAOException in case of a data access error
78     */
79    private static Set<Group> findAllByApproximateNameSecureHelper(final String name, final String scope,
80        final boolean currentNames, final boolean alternateNames, final QueryOptions queryOptions,
81        final Set<TypeOfGroup> typeOfGroups) throws GrouperDAOException {
82  
83      @SuppressWarnings("unchecked")
84      Set<Group> resultGroups = (Set<Group>) HibernateSession.callbackHibernateSession(
85          GrouperTransactionType.READONLY_OR_USE_EXISTING, AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {
86            public Object callback(HibernateHandlerBean hibernateHandlerBean) throws GrouperDAOException {
87              StringBuilder hql = new StringBuilder("select distinct theGroup from Group theGroup ");
88              ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
89              GrouperSession grouperSession = GrouperSession.staticGrouperSession();
90  
91              // see if we are adding more to the query
92              boolean changedQuery = grouperSession.getAccessResolver().hqlFilterGroupsWhereClause(
93                  grouperSession.getSubject(), byHqlStatic, hql, "theGroup.uuid",
94                  AccessPrivilege.VIEW_PRIVILEGES);
95  
96              hql.append((!changedQuery) ? " where " : " and ");
97              hql.append(" ( ");
98              
99              String lowerName = StringUtils.defaultString(name).toLowerCase();
100             if (currentNames) {
101               hql.append(" lower(theGroup.nameDb) like :theName or lower(theGroup.displayNameDb) like :theDisplayName ");
102               byHqlStatic.setString("theName", "%" + lowerName + "%");
103               byHqlStatic.setString("theDisplayName", "%" + lowerName + "%");
104             }
105 
106             if (alternateNames) {
107               if (currentNames) hql.append(" or ");
108               hql.append(" theGroup.alternateNameDb like :theAlternateName ");
109               byHqlStatic.setString("theAlternateName", "%" + lowerName + "%");
110             }
111 
112             hql.append(" ) ");
113 
114             if (scope != null) {
115               hql.append(" and theGroup.nameDb like :theStemScope ");
116               byHqlStatic.setString("theStemScope", scope + "%");
117             }
118 
119             // add in the typeOfGroups part
120             appendHqlQuery("theGroup", typeOfGroups, hql, byHqlStatic);
121             byHqlStatic.setCacheable(false);
122 
123             // reset sorting
124             if (queryOptions != null) {
125               byHqlStatic.options(queryOptions);
126             }
127 
128             byHqlStatic.createQuery(hql.toString());
129             Set<Group> groups = byHqlStatic.listSet(Group.class);
130 
131             return groups;
132           }
133         });
134     return resultGroups;
135   }

137   /**
138    * Append the type of groups part to the HQL query for searching groups.
139    * Note: this isn't in 2.0.0
140    * 
141    * @param groupAlias is the alias in the group hql query e.g. theGroup
142    * @param typeOfGroups the set of TypeOfGroup or null for all
143    * @param hql query so far
144    * @param hqlQuery object to append the stored params to
145    */
146   private static void appendHqlQuery(String groupAlias, Set<TypeOfGroup> typeOfGroups, StringBuilder hql, HqlQuery hqlQuery) {
147     if (GrouperUtil.length(typeOfGroups) <= 0) return;
148     
149     hql.append((hql.indexOf(" where ") > 0) ? " and ": " where ");
150     hql.append(groupAlias).append(".typeOfGroupDb in ( ");
151     
152     Set<String> typeOfGroupStrings = new LinkedHashSet<String>();
153     for (TypeOfGroup typeOfGroup : typeOfGroups) {
154       typeOfGroupStrings.add(typeOfGroup.name());
155     }
156     
157     hql.append(HibUtils.convertToInClause(typeOfGroupStrings, hqlQuery));
158     hql.append(" ) ");
159   }

161   /**
162    * Get the members for a group based on the VOOT group.
163    * 
164    * @param subject the subject querying the VOOT interface.
165    * @param vootGroup the group to be looked to find members.
166    * @param sortBy the field name to be used for sorting or null of no sorting.
167    * @param start the first element in the result set (0 means start from beginning).
168    * @param count the number of elements in the result set (-1 or 0 means find all).
169    * @return the response to be sent back to user in JSON format.
170    */
171   public static VootGetMembersResponse getMembers(Subject subject, VootGroup vootGroup, String sortBy, int start, int count) {
172     GrouperSession session = GrouperSession.staticGrouperSession();
173     // Non using session created by subject because findByName will only
174     // retrieve
175     // groups where the subject is administrator
176     // if
177     // (!GrouperSession.staticGrouperSession().getSubject().equals(subject))
178     // {
179     // session = GrouperSession.start(subject, false);
180     // }
181 
182     // note the name is the id
183     String groupName = vootGroup.getId();
184     // throws exception if the group is not found
185     Group group = GroupFinder.findByName(session, groupName, true);
186     Set<Subject> memberSubjects = new HashSet<Subject>();
187 
188     Set<Member> members = group.getMembers();
189     for (Member member : members) {
190       memberSubjects.add(member.getSubject());
191     }
192 
193     Set<Subject> admins = group.getAdmins();
194     Set<Subject> updaters = group.getUpdaters();
195 
196     // lets keep track of the subjects
197     // since subjects have a composite key, then keep track with multikey
198     Map<MultiKey, Subject> multiKeyToSubject = new HashMap<MultiKey, Subject>();
199 
200     // member, admin, manager (descriebd in VootGroup.GroupRoles enum)
201     Map<MultiKey, String> memberToRole = new HashMap<MultiKey, String>();
202 
203     boolean subjectInGroup = false;
204     for (Subject curSubject : memberSubjects) {
205       if (curSubject.getSourceId().equals(subject.getSourceId()) && curSubject.getId().equals(subject.getId()))
206         subjectInGroup = true;
207       MultiKey subjectMultiKey = new MultiKey(curSubject.getSourceId(), curSubject.getId());
208       multiKeyToSubject.put(subjectMultiKey, curSubject);
209       memberToRole.put(subjectMultiKey, VootGroup.GroupRoles.ADMIN.toString());
210     }
211     for (Subject curSubject : updaters) {
212       if (curSubject.getSourceId().equals(subject.getSourceId()) && curSubject.getId().equals(subject.getId()))
213         subjectInGroup = true;
214       MultiKey subjectMultiKey = new MultiKey(subject.getSourceId(), curSubject.getId());
215       multiKeyToSubject.put(subjectMultiKey, curSubject);
216       memberToRole.put(subjectMultiKey, VootGroup.GroupRoles.MANAGER.toString());
217     }
218     for (Subject curSubject : admins) {
219       if (curSubject.getSourceId().equals(subject.getSourceId()) && curSubject.getId().equals(subject.getId()))
220         subjectInGroup = true;
221       MultiKey subjectMultiKey = new MultiKey(curSubject.getSourceId(), curSubject.getId());
222       multiKeyToSubject.put(subjectMultiKey, curSubject);
223       memberToRole.put(subjectMultiKey, VootGroup.GroupRoles.MEMBER.toString());
224     }
225 
226     if (!GrouperSession.staticGrouperSession().getSubject().equals(subject) && !subjectInGroup) {
227       throw new GroupNotFoundException(E.GROUP_NOTFOUND + " by name: " + groupName);
228     }
229 
230     VootGetMembersResponse vootGetMembersResponse = new VootGetMembersResponse();
231     VootPerson[] result = new VootPerson[memberToRole.size()];
232 
233     // lets put them all back and make the person subjects
234     int index = 0;
235     for (MultiKey multiKey : memberToRole.keySet()) {
236       Subject curSubject = multiKeyToSubject.get(multiKey);
237       String role = memberToRole.get(multiKey);
238       VootPerson vootPerson = new VootPerson(curSubject);
239       vootPerson.setVoot_membership_role(role);
240       result[index] = vootPerson;
241 
242       index++;
243     }
244 
245     result = VootGetMembersResponse.sort(result, sortBy);
246     vootGetMembersResponse.paginate(result, start, count);
247     vootGetMembersResponse.setEntry(result, start, count);
248     return vootGetMembersResponse;
249   }

251   /**
252    * Get the groups that a person is in.
253    * 
254    * @param subject the subject representing the person used for search.
255    * @param sortBy the field name to be used for sorting or null of no sorting.
256    * @param start the first element in the result set (0 means start from beginning).
257    * @param count the number of elements in the result set (-1 or 0 means find all).
258    * @return the groups the subject passed as a parameter is part of.
259    */
260   public static VootGetGroupsResponse getGroups(Subject subject, String sortBy, int start, int count) {
261     GrouperSession session = GrouperSession.staticGrouperSession();
262     if (!GrouperSession.staticGrouperSession().getSubject().equals(subject)) {
263       session = GrouperSession.start(subject, false);
264     }
265 
266     Member member = MemberFinder.findBySubject(session, subject, false);
267 
268     VootGetGroupsResponse vootGetGroupsResponse = new VootGetGroupsResponse();
269 
270     if (member == null) {
271       vootGetGroupsResponse.paginate(null, start, count);
272       return vootGetGroupsResponse;
273     }
274 
275     // member, admin, manager
276     Set<Group> groups = member.getGroups();
277     Set<Group> admins = member.getGroups(FieldFinder.find(GROUPER_ADMIN_ROLE, true));
278     Set<Group> updaters = member.getGroups(FieldFinder.find(GROUPER_UPDATER_ROLE, true));
279 
280     Map<Group, String> groupToRole = new TreeMap<Group, String>();
281 
282     // if you are a member, and not an admin or updater, then you are a member
283     for (Group group : GrouperUtil.nonNull(groups)) {
284       groupToRole.put(group, VootGroup.GroupRoles.MEMBER.toString());
285     }
286 
287     // if you are an updater and not an admin, then you are a manager
288     for (Group group : GrouperUtil.nonNull(updaters)) {
289       groupToRole.put(group, VootGroup.GroupRoles.MANAGER.toString());
290     }
291 
292     // if you are an admin, then you are an admin
293     for (Group group : GrouperUtil.nonNull(admins)) {
294       groupToRole.put(group, VootGroup.GroupRoles.ADMIN.toString());
295     }
296 
297     if (groupToRole.size() == 0) {
298       vootGetGroupsResponse.paginate(null, start, count);
299       return vootGetGroupsResponse;
300     }
301 
302     VootGroup[] result = new VootGroup[groupToRole.size()];
303     int index = 0;
304     for (Group group : groupToRole.keySet()) {
305       VootGroup vootGroup = new VootGroup(group);
306       vootGroup.setVoot_membership_role(groupToRole.get(group));
307 
308       result[index] = vootGroup;
309 
310       index++;
311     }
312 
313     result = VootGetGroupsResponse.sort(result, sortBy);
314     vootGetGroupsResponse.paginate(result, start, count);
315     vootGetGroupsResponse.setEntry(result, start, count);
316     return vootGetGroupsResponse;
317   }

319   /**
320    * Get the groups that a person is in, searching by their name.
321    * 
322    * @param search the search term to be searched in group name.
323    * @param sortBy the field name to be used for sorting or null of no sorting.
324    * @param start the first element in the result set (0 means start from beginning).
325    * @param count the number of elements in the result set (-1 or 0 means find all).
326    * @return the groups found satisfying search criteria.
327    */
328   public static VootGetGroupsResponse getGroups(String search, String sortBy, int start, int count) {
329     VootGetGroupsResponse vootGetGroupsResponse = new VootGetGroupsResponse();
330 
331     // this isnt in 2.0.0
332     String searchString = "%";
333     if (search != null) {
334       searchString = "%" + search + "%";
335     }
336     Set<Group> groups = findAllByApproximateNameSecureHelper(searchString, null, true, true, null, null);
337 
338     if (GrouperUtil.length(groups) == 0) {
339       vootGetGroupsResponse.paginate(null, start, count);
340       return vootGetGroupsResponse;
341     }
342 
343     VootGroup[] result = new VootGroup[groups.size()];
344 
345     int index = 0;
346     for (Group group : groups) {
347       VootGroup vootGroup = new VootGroup(group);
348       result[index] = vootGroup;
349       index++;
350     }
351 
352     result = VootGetGroupsResponse.sort(result, sortBy);
353     vootGetGroupsResponse.paginate(result, start, count);
354     vootGetGroupsResponse.setEntry(result, start, count);
355     return vootGetGroupsResponse;
356   }
