41    public void testHasPermissionHelper() {
42      
43      Map<String, Set<GrouperActivemqPermission>> permissions = new HashMap<String, Set<GrouperActivemqPermission>>();
44      
45      permissions.put("test1", GrouperClientUtils.toSet(new GrouperActivemqPermission(GrouperActivemqPermissionAction.receiveMessage, "testA")));
46      permissions.put("test2", GrouperClientUtils.toSet(new GrouperActivemqPermission(GrouperActivemqPermissionAction.sendMessage, "testB")));
47      permissions.put("test3", GrouperClientUtils.toSet(new GrouperActivemqPermission(GrouperActivemqPermissionAction.receiveMessageInherit, "testC")));
48      permissions.put("test4", GrouperClientUtils.toSet(new GrouperActivemqPermission(GrouperActivemqPermissionAction.sendMessageInherit, "testD")));
49      permissions.put("test5", GrouperClientUtils.toSet(new GrouperActivemqPermission(GrouperActivemqPermissionAction.createDestination, "testE")));
50      permissions.put("test6", GrouperClientUtils.toSet(new GrouperActivemqPermission(GrouperActivemqPermissionAction.deleteDestination, "testF")));
51      permissions.put("test7", GrouperClientUtils.toSet(new GrouperActivemqPermission(GrouperActivemqPermissionAction.receiveMessage, "testG"),
52          new GrouperActivemqPermission(GrouperActivemqPermissionAction.receiveMessageInherit, "testH:testI")));
53  
54      // test1
55      assertTrue(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "test1", GrouperActivemqPermissionAction.receiveMessage, "testA"));
56      assertFalse(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "testQ", GrouperActivemqPermissionAction.receiveMessage, "testA"));
57      assertFalse(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "test1", GrouperActivemqPermissionAction.receiveMessage, "testA1"));
58      assertFalse(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "test1", GrouperActivemqPermissionAction.receiveMessage, "testA:2"));
59      assertFalse(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "test1", GrouperActivemqPermissionAction.createDestination, "testA"));
60      assertFalse(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "test1", GrouperActivemqPermissionAction.deleteDestination, "testA"));
61      assertFalse(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "test1", GrouperActivemqPermissionAction.receiveMessageInherit, "testA"));
62      assertFalse(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "test1", GrouperActivemqPermissionAction.sendMessage, "testA"));
63      assertFalse(GrouperActivemqPermissionsEngine.hasPermissionHelper(permissions, "test1", GrouperActivemqPermissionAction.sendMessageInherit, "testA"));
