18  /* JAAS imports */
19  import java.io.IOException;
20  
21  import javax.security.auth.callback.Callback;
22  import javax.security.auth.callback.CallbackHandler;
23  import javax.security.auth.callback.ConfirmationCallback;
24  import javax.security.auth.callback.NameCallback;
25  import javax.security.auth.callback.PasswordCallback;
26  import javax.security.auth.callback.TextOutputCallback;
27  import javax.security.auth.callback.UnsupportedCallbackException;
28  
29  /**
30   * this is a kerberos handler downloaded from the internet
31   * @see javax.security.auth.callback
32   */
33  
34  public class GrouperActivemqKerberosHandler implements CallbackHandler {
35  
36    /** principal */
37    private String principal;
38    
39    /** password */
40    private String password;
41    
42    /**
43     * <p>Creates a callback handler that prompts and reads from the
44     * command line for answers to authentication questions.
45     * This can be used by JAAS applications to instantiate a
46     * CallbackHandler.
47     * @param thePrincipal principal
48     * @param thePassword password
49  
50     */
51    public GrouperActivemqKerberosHandler(String thePrincipal, String thePassword) {
52      this.principal = thePrincipal;
53      this.password = thePassword;
54    }
55  
56    /**
57     * Handles the specified set of callbacks.
58     *
59     * @param callbacks the callbacks to handle
60     * @throws IOException if an input or output error occurs.
61     * @throws UnsupportedCallbackException if the callback is not an
62     * instance of NameCallback or PasswordCallback
63     */
64    public void handle(Callback[] callbacks) throws IOException,
65        UnsupportedCallbackException {
66      ConfirmationCallback confirmation = null;
67  
68      for (int i = 0; i < callbacks.length; i++) {
69        if (callbacks[i] instanceof TextOutputCallback) {
70          TextOutputCallback tc = (TextOutputCallback) callbacks[i];
71  
72          String text;
73          switch (tc.getMessageType()) {
74            case TextOutputCallback.INFORMATION:
75              text = "";
76              break;
77            case TextOutputCallback.WARNING:
78              text = "Warning: ";
79              break;
80            case TextOutputCallback.ERROR:
81              text = "Error: ";
82              break;
83            default:
84              throw new UnsupportedCallbackException(callbacks[i],
85                  "Unrecognized message type");
86          }
87  
88          String message = tc.getMessage();
89          if (message != null) {
90            text += message;
91          }
92  
93        } else if (callbacks[i] instanceof NameCallback) {
94          NameCallback nc = (NameCallback) callbacks[i];
95  
96          String result = this.principal;
97          if (result.equals("")) {
98            result = nc.getDefaultName();
99          }
100 
101         nc.setName(result);
102 
103       } else if (callbacks[i] instanceof PasswordCallback) {
104         PasswordCallback pc = (PasswordCallback) callbacks[i];
105 
106         pc.setPassword(this.password.toCharArray());
107 
108       } else if (callbacks[i] instanceof ConfirmationCallback) {
109         confirmation = (ConfirmationCallback) callbacks[i];
110 
111       } else {
112         throw new UnsupportedCallbackException(callbacks[i], "Unrecognized Callback");
113       }
114     }
115 
116     /* Do the confirmation callback last. */
117     if (confirmation != null) {
118       doConfirmation(confirmation);
119     }
120   }
121 
122   /**
123    * 
124    * @param confirmation
125    * @throws UnsupportedCallbackException
126    */
127   private void doConfirmation(ConfirmationCallback confirmation) 
128       throws UnsupportedCallbackException {
129     String prefix;
130     int messageType = confirmation.getMessageType();
131     switch (messageType) {
132       case ConfirmationCallback.WARNING:
133         prefix = "Warning: ";
134         break;
135       case ConfirmationCallback.ERROR:
136         prefix = "Error: ";
137         break;
138       case ConfirmationCallback.INFORMATION:
139         prefix = "";
140         break;
141       default:
142         throw new UnsupportedCallbackException(confirmation,
143             "Unrecognized message type: " + messageType);
144     }
145 
146     class OptionInfo {
147 
148       /**
149        * name
150        */
151       String name;
152 
153       int value;
154 
155       /**
156        * 
157        * @param name1
158        * @param value1
159        */
160       OptionInfo(String name1, int value1) {
161         this.name = name1;
162         //noop since this.name not read and got warning
163         this.value = Integer.parseInt(this.name);
164         this.value = value1;
165       }
166     }
167 
168     OptionInfo[] options;
169     int optionType = confirmation.getOptionType();
170     switch (optionType) {
171       case ConfirmationCallback.YES_NO_OPTION:
172         options = new OptionInfo[] { new OptionInfo("Yes", ConfirmationCallback.YES),
173             new OptionInfo("No", ConfirmationCallback.NO) };
174         break;
175       case ConfirmationCallback.YES_NO_CANCEL_OPTION:
176         options = new OptionInfo[] { new OptionInfo("Yes", ConfirmationCallback.YES),
177             new OptionInfo("No", ConfirmationCallback.NO),
178             new OptionInfo("Cancel", ConfirmationCallback.CANCEL) };
179         break;
180       case ConfirmationCallback.OK_CANCEL_OPTION:
181         options = new OptionInfo[] { new OptionInfo("OK", ConfirmationCallback.OK),
182             new OptionInfo("Cancel", ConfirmationCallback.CANCEL) };
183         break;
184       case ConfirmationCallback.UNSPECIFIED_OPTION:
185         String[] optionStrings = confirmation.getOptions();
186         options = new OptionInfo[optionStrings.length];
187         for (int i = 0; i < options.length; i++) {
188           options[i].value = i;
189         }
190         break;
191       default:
192         throw new UnsupportedCallbackException(confirmation, "Unrecognized option type: "
193             + optionType);
194     }
195 
196     int defaultOption = confirmation.getDefaultOption();
197 
198     String prompt = confirmation.getPrompt();
199     if (prompt == null) {
200       prompt = "";
201     }
202     prompt = prefix + prompt;
203     if (!prompt.equals("")) {
204       //System.err.println(prompt);
205     }
206 
207     int result;
208     try {
209       result = 0;//Integer.parseInt(readLine());
210       if (result < 0 || result > (options.length - 1)) {
211         result = defaultOption;
212       }
213       result = options[result].value;
214     } catch (NumberFormatException e) {
215       result = defaultOption;
216     }
217 
218     confirmation.setSelectedIndex(result);
219   }
220 }
