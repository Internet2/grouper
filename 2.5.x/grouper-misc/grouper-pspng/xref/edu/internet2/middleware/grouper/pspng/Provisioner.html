52  /**
53   * Top-Level provisioner class of PSPNG and is the superclass of Target-System-Specific 
54   * subclasses. This class is generic so that User & Group objects that the subclasses
55   * need are cached and referenced directly in overridden methods. For example, an
56   * LDAP-provisioning subclass will receive both Subject and LdapUser information.
57   * 
58   * This class is responsible for the following functionalities:
59   *   -Mapping the grouper world of Subjects & Groups into User and Group objects specific to the target system
60   *   -Caching both Grouper & Target-System information
61   *   -Driving Incremental and FullSync operations
62   *   -Keep track of what groups need to be full-synced (ie, there is a full-sync queue)
63   * 
64   * While there are several abstract methods:
65   *     fetchTargetSystemUsers & fetchTargetSystemGroups
66   *     addMembership & deleteMembership
67   *     createGroup & deleteGroup
68   *     doFullSync & doFullSync_cleanupExtraGroups
69   *   
70   * And that some subclasses can override the following method(s):
71   *     createUser
72   *   
73   *   
74   * The Provisioner lifecycle is as follows:
75   * -Constructed(Name and ProvisionerConfiguration subclass)
76   *   (setup empty caches)
77   * 
78   * Invoked with a batch of changes
79   *   -startProvisioningBatch(A collection of change events)
80   *      (refresh caches as necessary, do any bulk fetching that will help provisioning go faster)
81   *   -provisionItem or addMember/deleMembership/createGroup/deleteGroup (for each event in the batch)
82   *      (Determine and/or make the necessary changes)
83   *      
84   *   -finishProvisioningBatch(A collection of change events)
85   *      (Perform any batch updates that were determined, but not performed, by provisionItem step)
86   *      (Clear any caches specific to batch)
87   *      
88   *  FullSync
89   *    -Go through all Grouper groups and see which match the provisioner's selection filter
90   *    -Run the subclass's doFullSync method for each group that matches
91   *    -Run doFullSync_cleanupExtraGruops with the list of correct groups
92   *      
93   * @author Bert Bee-Lindgren
94   *
95   */

175   Provisioner(String provisionerConfigName, ConfigurationClass config, boolean fullSyncMode) {
176     this.provisionerConfigName = provisionerConfigName;
177 
178     if (fullSyncMode) {
179       this.provisionerDisplayName = provisionerConfigName + "-full";
180     } else {
181       this.provisionerDisplayName = provisionerConfigName;
182     }
183     this.fullSyncMode = fullSyncMode;
184     LOG = LoggerFactory.getLogger(String.format("%s.%s", getClass().getName(), provisionerDisplayName));
185 
186     this.config = config;
187 
188     checkAttributeDefinitions();
189 
190     // NOTE: PSPNG has different size=0 semantics... In PSPNG, size=0 means zero caching, but in
191     // GrouperCache/EhCache size=0 means unlimited caching. Therefore, we map size=0 to size=1.
192     grouperGroupInfoCache
193             = new GrouperCache<String, GrouperGroupInfo>(String.format("PSP-%s-GrouperGroupInfoCache", getConfigName()),
194             config.getGrouperGroupCacheSize() > 1 ? config.getGrouperGroupCacheSize() : 1,
195             false,
196             config.getDataCacheTime_secs(),
197             config.getDataCacheTime_secs(),
198             false);
199 
200     grouperSubjectCache
201               = new GrouperCache<String, Subject>(String.format("PSP-%s-GrouperSubjectCache", getConfigName()),
202               config.getGrouperSubjectCacheSize() > 1 ? config.getGrouperSubjectCacheSize() : 1,
203               false,
204               config.getDataCacheTime_secs(),
205               config.getDataCacheTime_secs(),
206               false);
207 
208     targetSystemUserCache
209               = new PspDatedCache<Subject, TSUserClass>(String.format("PSP-%s-TargetSystemUserCache", getConfigName()),
210               config.getTargetSystemUserCacheSize() > 1 ? config.getTargetSystemUserCacheSize() : 1,
211               false,
212               config.getDataCacheTime_secs(),
213               config.getDataCacheTime_secs(),
214               false);
215 
216     // This cache is set up with DisplayName of provisioner
217     // to keep the caches of the FullSync and Incremental instances separate
218     targetSystemGroupCache
219       = new GrouperCache<GrouperGroupInfo, TSGroupClass>(String.format("PSP-%s-TargetSystemGroupCache", getDisplayName()),
220           config.getTargetSystemGroupCacheSize() > 1 ? config.getTargetSystemGroupCacheSize() : 1,
221           false,
222           config.getDataCacheTime_secs(),
223           config.getDataCacheTime_secs(),
224           false);
225 
226     if ( config.needsTargetSystemUsers() ) {
227       tsUserFetchingService = Executors.newFixedThreadPool(
228               config.getNumberOfDataFetchingWorkers(),
229               new ThreadFactory() {
230                 AtomicInteger counter = new AtomicInteger(1);
231                 @Override
232                 public Thread newThread(Runnable r) {
233                   return new Thread(r, String.format("TSUserFetcher-%s-%d", getDisplayName(), counter.getAndIncrement()));
234                 }
235               });
236     } else {
237       tsUserFetchingService=null;
238     }
239   }

243    * This creates any attributes missing within the etc:pspng: folder.
244    */
245   private void checkAttributeDefinitions() {
246     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
247     if ( grouperSession == null )
248       grouperSession = GrouperSession.startRootSession();
249     
250     //GRP-1356: pspng should use the default configuration folder
251     //String pspngManagementStemName = "etc:pspng";
252     String pspngManagementStemName = GrouperConfig.retrieveConfig().propertyValueString(
253         "grouper.rootStemForBuiltinObjects", "etc") + ":pspng";
254     
255     Stem pspngManagementStem = StemFinder.findByName(grouperSession, pspngManagementStemName, false);
256     if (pspngManagementStem == null) {
257       pspngManagementStem = new StemSave(grouperSession).assignCreateParentStemsIfNotExist(true)
258         .assignDescription("Location for pspng-management objects.")
259         .assignName(pspngManagementStemName).save();
260     }
261 
262     //see if provision_to_def attributeDef is there
263     String provisionToDefName = pspngManagementStemName + ":" + PROVISION_TO_ATTRIBUTE + "_def";
264     AttributeDef provisionToDef = GrouperDAOFactory.getFactory().getAttributeDef().findByNameSecure(
265         provisionToDefName, false, new QueryOptions().secondLevelCache(false));
266     if (provisionToDef == null) {
267       provisionToDef = pspngManagementStem.addChildAttributeDef(PROVISION_TO_ATTRIBUTE + "_def", AttributeDefType.type);
268       provisionToDef.setAssignToGroup(true);
269       provisionToDef.setAssignToStem(true);
270       provisionToDef.setMultiAssignable(true);
271       provisionToDef.setValueType(AttributeDefValueType.string);
272       provisionToDef.store();
273     }
274     
275     //see if do_not_provision_to_def attributeDef is there
276     String doNotProvisionToDefName = pspngManagementStemName + ":" + DO_NOT_PROVISION_TO_ATTRIBUTE + "_def";
277     AttributeDef doNotProvisionToDef = GrouperDAOFactory.getFactory().getAttributeDef().findByNameSecure(
278         doNotProvisionToDefName, false, new QueryOptions().secondLevelCache(false));
279     if (doNotProvisionToDef == null) {
280       doNotProvisionToDef = pspngManagementStem.addChildAttributeDef(DO_NOT_PROVISION_TO_ATTRIBUTE+"_def", AttributeDefType.type);
281       doNotProvisionToDef.setAssignToGroup(true);
282       doNotProvisionToDef.setAssignToStem(true);
283       doNotProvisionToDef.setMultiAssignable(true);
284       doNotProvisionToDef.setValueType(AttributeDefValueType.string);
285       doNotProvisionToDef.store();
286     }
287     
288     GrouperCheckConfig.checkAttribute(pspngManagementStem, provisionToDef, PROVISION_TO_ATTRIBUTE, PROVISION_TO_ATTRIBUTE, "Defines what provisioners should process a group or groups within a folder", true);
289     GrouperCheckConfig.checkAttribute(pspngManagementStem, doNotProvisionToDef, DO_NOT_PROVISION_TO_ATTRIBUTE, DO_NOT_PROVISION_TO_ATTRIBUTE, "Defines what provisioners should not process a group or groups within a folder. Since the default is already for provisioners to not provision any groups, this attribute is to override a provision_to attribute set on an ancestor folder. ", true);
290   }

435   /**
436    * This method returns the work items that are supposed to be provisioned
437    * by calling shouldGroupBeProvisioned on each group mentioned
438    * by a workItem. If a workItem's group is within the scope of this provisioner
439    * or if the workItem is not related to a group, then it is included in the
440    * returned list of work items that should be processed further. Otherwise, it
441    * is marked as completed and not returned.
442    * 
443    * If the workItem is not to be processed, 
444    * @param workItems WorkItems read from the triggering source (Changelog or messaging).
445    * This will include both events that affect groups and those that do not. Generally, we
446    * pass on non-group changes in case a provisioner wants to process them. 
447    * @return The list of workItems that are to be provisioned
448    * @throws PspException
449    */
450   public List<ProvisioningWorkItem> 
451   filterWorkItems(List<ProvisioningWorkItem> workItems) throws PspException {
452     List<ProvisioningWorkItem> result = new ArrayList<ProvisioningWorkItem>();
453     
454     LOG.debug("Filtering provisioning batch of {} items", workItems.size());
455 
456     for ( ProvisioningWorkItem workItem : workItems ) {
457       GrouperGroupInfo group = workItem.getGroupInfo(this);
458 
459       // Groups that haven't been deleted: Skip them if they're not supposed to be provisioned
460       if ( group != null && !group.hasGroupBeenDeleted() && !shouldGroupBeProvisioned(group)) {
461           workItem.markAsSkipped("Ignoring work item because (existing) group should not be provisioned");
462           continue;
463       }
464 
465       if ( shouldWorkItemBeProcessed(workItem) ) {
466         result.add(workItem);
467       } else {
468         // Not going to process this item, so mark it as a success and don't add it to result
469           workItem.markAsSkipped("Ignoring work item");
470       }
471     }
472     
473     return result;
474   }

476   /**
477    * Used to filter workItems. This can be overridden by a subclass that was unhappy
478    * with the default filter behaviors. If overriding this, eg, to accept additional
479    * changes into a provisioner, then it would probably be useful to look for those
480    * additional types of changes and then call the super version of this.
481    *
482    * Note: In addition to returning a boolean about skipping the work item, it is helpful
483    * if this method actually calls markAsSkipped(reason) so that a more detailed
484    * reason can be noted for skipping it. In other words, this method has the
485    * (optional) side effect of actually marking the item as skipped. Finally, if this
486    * method does _not_ mark the item as skipped, it will be marked as skipped
487    * with a generic message.
488    *
489    * @param workItem
490    * @return
491    */
492   protected boolean shouldWorkItemBeProcessed(ProvisioningWorkItem workItem) {
493     if ( workItem.isSubjectUnresolvable(this) ) {
494       workItem.markAsSkippedAndWarn("Subject is unresolvable");
495       return false;
496     }
497 
498     // Check if we're configured to ignore changes to internal (g:gsa) subjects
499     // (default is that we do ignore such changes)
500     if ( getConfig().areChangesToInternalGrouperSubjectsIgnored() ) {
501       Subject subject = workItem.getSubject(this);
502       if ( subject != null && subject.getSourceId().equalsIgnoreCase("g:gsa") ) {
503         workItem.markAsSkipped("Ignoring event about a g:gsa subject");
504         return false;
505       }
506     }
507 
508     // Only group-deletions are processed for groups that have been deleted
509     // This skips all the membership-removal changelog entries that precede the
510     // DElETE changelog entry
511     if ( workItem.getGroupInfo(this) != null &&
512          workItem.getGroupInfo(this).hasGroupBeenDeleted() ) {
513       if (workItem.matchesChangelogType(ChangeLogTypeBuiltin.GROUP_DELETE) ) {
514         LOG.debug("{}: Group has been deleted within grouper, so we're processing group-deletion event: {}", getDisplayName(), workItem);
515       } else {
516         workItem.markAsSkipped("Group has been deleted within Grouper. Skipping event because it is not the actual GROUP_DELETE event");
517         return false;
518       }
519     }
520 
521     if ( !workItem.matchesChangelogType(ChangelogHandlingConfig.allRelevantChangelogTypes) ) {
522       workItem.markAsSkipped("Changelog type is not relevant to PSPNG provisioning (see ChangelogHandlingConfig.allRelevantChangelogTypes)");
523       return false;
524     }
525 
526     return true;
527   }

683   /**
684    * Make a JexlMap that contains subject and group information and evaluate the given
685    * expression.
686    *
687    * @param expressionName
688    * @param expression
689    * @param subject
690    * @param tsUser
691    * @param grouperGroupInfo
692    * @param tsGroup
693    * @param keysAndValues Key/Value pairs that will also be available within the Jexl's variable map
694    * @return
695    */
696   protected final String evaluateJexlExpression(String expressionName, String expression,
697       Subject subject, TSUserClass tsUser,
698       GrouperGroupInfo grouperGroupInfo, TSGroupClass tsGroup,
699       Object... keysAndValues) throws PspException {
700     
701     LOG.trace("Evaluating {} Jexl expression: {}", expressionName, expression);
702     
703     Map<String, Object> variableMap = new HashMap<>();
704 
705     variableMap.put("utils", new PspJexlUtils());
706 
707     // Copy any provided keys and values into the map
708     GrouperUtil.assertion(keysAndValues.length % 2 == 0, "KeysAndValues must be paired evenly");
709     for (int i=0; i<keysAndValues.length; i+=2)
710       variableMap.put(keysAndValues[i].toString(), keysAndValues[i+1]);
711 
712     // Give provisioner subclasses to add information
713     populateJexlMap(variableMap, 
714         subject, 
715         tsUser,
716         grouperGroupInfo, 
717         tsGroup);
718     
719     // Give our config a chance to add information
720     config.populateElMap(variableMap);
721     
722     try {
723       // In order to support nested expressions, we're going to repeatedly look for atomic (non-nested) expressions
724       // and replace each atomic ${...} with it's evaluation result until there are no more ${ in the string
725       // This kind of regular expression problem is discussed here:
726       // https://stackoverflow.com/questions/717644/regular-expression-that-doesnt-contain-certain-string
727 
728       Pattern atomicExpressionPattern = Pattern.compile("\\$\\{([^$]|\\$[^{])*?\\}" );
729       String result=expression;
730       Matcher atomicExpressionMatcher = atomicExpressionPattern.matcher(result);
731 
732       while ( atomicExpressionMatcher.find()) {
733         String atomicExpression = atomicExpressionMatcher.group();
734         String atomicExpressionResult;
735 
736         // Check to see if expression has a backup expression
737         //   xyz:-pdq ==> evaluate pdq if xyz cannot be evaluated
738         if ( ! atomicExpression.contains(":-") ) {
739           atomicExpressionResult = GrouperUtil.substituteExpressionLanguage(atomicExpression, variableMap, true, false, false);
740         }
741         else {
742           // Split atomicExpression on :- and add } & { to the first and second pieces
743 
744           String expressionOne = StringUtils.substringBefore(atomicExpression, ":-") + "}";
745           String expressionTwo = "${" + StringUtils.substringAfter(atomicExpression, ":-");
746 
747           try {
748             atomicExpressionResult = GrouperUtil.substituteExpressionLanguage(expressionOne, variableMap, true, false, false);
749           } catch (RuntimeException e) {
750             LOG.warn("{}: Problem evaluating '{}'. Will try :- expression '{}': {}",
751                     new Object[]{expressionName, expressionOne, expressionTwo, e.getMessage()});
752 
753 
754             atomicExpressionResult = GrouperUtil.substituteExpressionLanguage(expressionTwo, variableMap, true, false, false);
755           }
756         }
757         LOG.debug("Evaluated {} Jexl expression: '{}'", expressionName, atomicExpressionResult);
758         LOG.trace("Evaluated {} Jexl expression: '{}' FROM {} WITH variables {}",
759                 new Object[]{expressionName, atomicExpressionResult, atomicExpression, variableMap});
760 
761         // replaceFirst unescapes the string it is given (because it isn't a string literal, but can also
762         // refer to regex groupings: $1 refers to the first matching group in the Pattern. In order to get a $ in the
763         // replacement, you need to use \$. We're not using groups in this stuff, but we need to keep \, exactly
764         // as it is; therefore, We need to double-escape atomicExpressionResult... Yuck
765 
766         // This is even more confusing because replaceAll takes a regex and a string literal:
767         //   the first \\\\ is a regex expression for a single \
768         //     (java strings need \\ to make a \ and the regex needs \\ to make a single \)
769         //   and the second \\\\\\\\ (8 whacks) is a substitution string resulting in  two \\
770         result = atomicExpressionMatcher.replaceFirst(atomicExpressionResult.replaceAll("\\\\", "\\\\\\\\"));
771         atomicExpressionMatcher = atomicExpressionPattern.matcher(result);
772       }
773 
774       if ( expression.equals(result) ) {
775         LOG.trace("{} Jexl expression did not include any substitutions: {}", expressionName, expression);
776       } else {
777         LOG.debug("Evaluated entire {} Jexl expression: '{}'", expressionName, result);
778       }
779 
780       return result;
781     }
782     catch (RuntimeException e) {
783       LOG.error("Jexl Expression {} '{}' could not be evaluated for subject '{}/{}' and group '{}/{}' which used variableMap '{}'",
784           new Object[] {expressionName, expression,
785               subject, tsUser,
786               grouperGroupInfo, tsGroup,
787               variableMap, e});
788       throw new PspException("Jexl evaluation failed: %s", e.getMessage());
789     }
790   }

1208    * Dispatches an event to the right method, with generally
1209    * useful parameters. 
1210    * 
1211    * There is no need to override this method when implementing basic provisioning 
1212    * (add/delete groups and members), but a Provisioner subclasses might override this 
1213    * to capture other CategoryAndActions combinations. 
1214    * 
1215    * @param workItem
1216    */
1217   protected void provisionItem(ProvisioningWorkItem workItem) throws PspException {
1218     LOG.debug("Starting provisioning of item: {}", workItem);
1219     
1220     currentWorkItem.set(workItem);
1221     ChangeLogEntry entry = workItem.getChangelogEntry();
1222     String workItem_identifier = null;
1223     if ( workItem.getChangelogEntry()!=null ) {
1224       workItem_identifier = Long.toString(workItem.getChangelogEntry().getSequenceNumber());
1225     }
1226 
1227     try {
1228       if ( workItem.matchesChangelogType(ChangelogHandlingConfig.changelogTypesThatAreHandledIncrementally) ) {
1229         processIncrementalSyncEvent(workItem);
1230       }
1231       else if ( workItem.getGroupInfo(this) != null ) {
1232         // This is a changelog entry that modifies the group. Do a FullSync to see if any
1233         // provisioned information changed. Unfortunately, this will do a membership sync which
1234         // might slow down the processing of this changelog entry. However, non-membership
1235         // changes are expected to be infrequent, so we aren't creating an optimized code path
1236         // that doesn't sync memberships.
1237 
1238         FullSyncQueueItem fullSyncStatus = getFullSyncer()
1239                 .scheduleGroupForSync(
1240                         FullSyncProvisioner.QUEUE_TYPE.ASAP,
1241                         workItem.groupName,
1242                         workItem_identifier,
1243                         "Changelog: %s", workItem);
1244 
1245         workItem.markAsSuccess("Handled with scheduled FullSync (qid=%d)", fullSyncStatus.id);
1246       }
1247       else if (  workItemShouldBeHandledByFullSyncOfEverything(workItem) ) {
1248         LOG.info("{}: Performing sync of all groups to process work item: {}", getDisplayName(), workItem);
1249         getFullSyncer().queueAllGroupsForFullSync(FullSyncProvisioner.QUEUE_TYPE.CHANGELOG,workItem_identifier,"Work item invokes full sync for everything: %s", workItem);
1250         if ( getConfig().isGrouperAuthoritative() ) {
1251           getFullSyncer().scheduleGroupCleanup(FullSyncProvisioner.QUEUE_TYPE.CHANGELOG, workItem_identifier, "Changelog-initiated");
1252         }
1253         workItem.markAsSuccess("Scheduled a full-sync of all groups");
1254       }
1255       else
1256       {
1257         workItem.markAsSkipped("Nothing to do (not a supported change)");
1258       }
1259     } catch (PspException e) {
1260       LOG.error("Problem provisioning item {}", workItem, e);
1261       workItem.markAsFailure("Provisioning failure: %s", e.getMessage());
1262     } finally {
1263       currentWorkItem.set(null);

1268   private void processIncrementalSyncEvent(ProvisioningWorkItem workItem) throws PspException {
1269     ChangeLogEntry entry = workItem.getChangelogEntry();
1270 
1271     if ( entry.equalsCategoryAndAction(ChangeLogTypeBuiltin.GROUP_ADD ))
1272     {
1273       GrouperGroupInfo grouperGroupInfo = workItem.getGroupInfo(this);
1274 
1275       if ( grouperGroupInfo == null || grouperGroupInfo.hasGroupBeenDeleted() ) {
1276         workItem.markAsSkippedAndWarn("Ignored group-add: group does not exist any more");
1277         return;
1278       }
1279 
1280       if ( !shouldGroupBeProvisioned(grouperGroupInfo) ) {
1281         workItem.markAsSkipped("Group %s is not selected to be provisioned", grouperGroupInfo);
1282         return;
1283       }
1284 
1285       if ( tsGroupCache_shortTerm.containsKey(grouperGroupInfo) ) {
1286         workItem.markAsSuccess("Group %s already exists", grouperGroupInfo);
1287         return;
1288       }
1289       else
1290         createGroup(grouperGroupInfo, Collections.EMPTY_LIST);
1291     }
1292     else if ( entry.equalsCategoryAndAction(ChangeLogTypeBuiltin.GROUP_DELETE ))
1293     {
1294       // Can't tell right reliably if group was supposed to be provisioned when it existed, so
1295       // we just delete it if it currently exists in target
1296       GrouperGroupInfo grouperGroupInfo = workItem.getGroupInfo(this);
1297 
1298       if ( grouperGroupInfo == null ) {
1299         workItem.markAsSkippedAndWarn("Ignoring group-deletion event because group information was not found in grouper");
1300         return;
1301       }
1302 
1303       TSGroupClass tsGroup = tsGroupCache_shortTerm.get(grouperGroupInfo);
1304 
1305       // If system uses targetSystemGroup and one doesn't exist ==> Nothing to do
1306       if ( config.needsTargetSystemGroups() ) {
1307           if ( tsGroup == null ) {
1308               workItem.markAsSuccess("Nothing to do: Group does not exist in target system");
1309               return;
1310           } else {
1311               LOG.info("Deleting provisioned group: {}", grouperGroupInfo);
1312               deleteGroup(grouperGroupInfo, tsGroup);
1313           }
1314       }
1315       else {
1316         // System doesn't use targetSystemGroup, so we can't tell if this group
1317         // was provisioned or not. Therefore call deleteGroup just in case
1318 
1319         LOG.info("{}: Group has been deleted from grouper. Checking to see if it was provisioned to target system", getDisplayName());
1320         deleteGroup(grouperGroupInfo, tsGroup);
1321       }
1322     }
1323     else if ( entry.equalsCategoryAndAction(ChangeLogTypeBuiltin.MEMBERSHIP_ADD))
1324     {
1325       GrouperGroupInfo grouperGroupInfo = workItem.getGroupInfo(this);
1326 
1327       if ( grouperGroupInfo == null || grouperGroupInfo.hasGroupBeenDeleted() ) {
1328         workItem.markAsSkipped("Ignoring membership-add event for group that was deleted");
1329         return;
1330       }
1331 
1332       if ( !shouldGroupBeProvisioned(grouperGroupInfo) ) {
1333         workItem.markAsSkipped("Group %s is not selected to be provisioned", grouperGroupInfo);
1334         return;
1335       }
1336 
1337       TSGroupClass tsGroup = tsGroupCache_shortTerm.get(grouperGroupInfo);
1338       Subject subject = workItem.getSubject(this);
1339 
1340       if ( subject == null ) {
1341         workItem.markAsSkippedAndWarn("Ignoring membership-add event because subject is no longer in grouper");
1342         return;
1343       }
1344 
1345       if ( subject.getTypeName().equalsIgnoreCase("group") ) {
1346         workItem.markAsSkipped("Nested-group membership skipped");
1347         return;
1348       }
1349 
1350       TSUserClass tsUser = tsUserCache_shortTerm.get(subject);
1351 
1352       if ( config.needsTargetSystemUsers() && tsUser==null ) {
1353         workItem.markAsSkippedAndWarn("Skipped: subject doesn't exist in target system");
1354         return;
1355       }
1356 
1357       addMembership(grouperGroupInfo, tsGroup, subject, tsUser);
1358     }
1359     else if ( entry.equalsCategoryAndAction(ChangeLogTypeBuiltin.MEMBERSHIP_DELETE))
1360     {
1361       GrouperGroupInfo grouperGroupInfo = workItem.getGroupInfo(this);
1362 
1363       if ( grouperGroupInfo==null || grouperGroupInfo.hasGroupBeenDeleted() ) {
1364         workItem.markAsSkipped("Ignoring membership-delete event for group that was deleted");
1365         return;
1366       }
1367 
1368       if ( !shouldGroupBeProvisioned(grouperGroupInfo) ) {
1369         workItem.markAsSkipped("Group %s is not selected to be provisioned", grouperGroupInfo);
1370         return;
1371       }
1372 
1373       TSGroupClass tsGroup = tsGroupCache_shortTerm.get(grouperGroupInfo);
1374       Subject subject = workItem.getSubject(this);
1375 
1376       if ( subject == null ) {
1377         workItem.markAsSkippedAndWarn("Ignoring membership-delete event because subject is no longer in grouper");
1378         LOG.warn("Work item ignored: {}", workItem);
1379         return;
1380       }
1381 
1382       TSUserClass tsUser = tsUserCache_shortTerm.get(subject);
1383 
1384       if ( config.needsTargetSystemUsers() && tsUser==null ) {
1385         workItem.markAsSkippedAndWarn("Skipped: subject doesn't exist in target system");
1386         return;
1387       }
1388       deleteMembership(grouperGroupInfo, tsGroup, subject, tsUser);
1389     }

1743   public void provisionBatchOfItems(List<ProvisioningWorkItem> allWorkItems) {
1744     activeProvisioner.set(this);
1745     List<ProvisioningWorkItem> filteredWorkItems=null;
1746 
1747     // Mark the items as successful if we are not enabled.
1748     // Note: They are being marked as successful so that there is an easy mechanism
1749     // to get a provisioner up to date with the changelog (or other event system).
1750     // Additionally, if you just don't want to process events, then you can remove
1751     // this provisioner from the configuration.
1752     if (!config.isEnabled()) {
1753       for (ProvisioningWorkItem workItem : allWorkItems)
1754         workItem.markAsSkippedAndWarn("Provisioner %s is not enabled", getDisplayName());
1755       return;
1756     }
1757 
1758     // Let's see if there is a reason to flush our group cache. In particular,
1759     // we flush group information when groups and their attributes are edited
1760     // because we don't know what attributes of groups could be used in various
1761     // JEXL expressions
1762     MDC.put("step", "cache_eval");
1763     try {
1764       flushCachesIfNecessary(allWorkItems);
1765     } catch (PspException e) {
1766       LOG.error("Unable to evaluate our caches", e);
1767       MDC.remove("step");
1768       throw new RuntimeException("No entries provisioned. Cache evaluation failed: " + e.getMessage(), e);
1769     }
1770 
1771     // Let the provisioner filter out any unnecessary work items.
1772     // This particularly filters out groups that we're not supposed to provision
1773     MDC.put("step", "filter/");
1774     try {
1775       filteredWorkItems = filterWorkItems(allWorkItems);
1776       LOG.info("{}: {} work items need to be processed further", getDisplayName(), filteredWorkItems.size());
1777     } catch (PspException e) {
1778       LOG.error("Unable to filter the provisioning batch", e);
1779       MDC.remove("step");
1780       throw new RuntimeException("No entries provisioned. Batch-filtering failed: " + e.getMessage(), e);
1781     }
1782 
1783     // Tell the provisioner about this batch of workItems
1784     MDC.put("step", "start/");
1785     try {
1786       try {
1787         startCoordination(filteredWorkItems);
1788         startProvisioningBatch(filteredWorkItems);
1789       } catch (PspException e) {
1790         LOG.error("Unable to begin the provisioning batch", e);
1791         MDC.remove("step");
1792         throw new RuntimeException("No entries provisioned. Batch-Start failed: " + e.getMessage(), e);
1793       }
1794 
1795       // Go through the workItems that were not marked as processed by the startProvisioningBatch
1796       // and provision them
1797       for (ProvisioningWorkItem workItem : allWorkItems) {
1798         MDC.put("step", String.format("prov/%s/", workItem.getMdcLabel()));
1799         if (!workItem.hasBeenProcessed()) {
1800           try {
1801             provisionItem(workItem);
1802           } catch (PspException e) {
1803             LOG.error("Problem provisioning {}", workItem, e);
1804             workItem.markAsFailure(e.getMessage());
1805           }
1806         }
1807       }
1808 
1809       // Do 'finish' task for workItems that are not marked as processed before now
1810 
1811       MDC.put("step", "fin/");
1812       List<ProvisioningWorkItem> workItemsToFinish = new ArrayList<ProvisioningWorkItem>();
1813       try {
1814         for (ProvisioningWorkItem workItem : allWorkItems) {
1815           if (!workItem.hasBeenProcessed())
1816             workItemsToFinish.add(workItem);
1817         }
1818         finishProvisioningBatch(workItemsToFinish);
1819         finishCoordination(filteredWorkItems, true);
1820       } catch (PspException e) {
1821         LOG.error("Problem completing provisioning batch", e);
1822         for (ProvisioningWorkItem workItem : workItemsToFinish) {
1823           if (!workItem.hasBeenProcessed())
1824             workItem.markAsFailure("Unable to finish provisioning (%s)", e.getMessage());
1825         }
1826       }
1827       MDC.remove("step");
1828     }
1829     finally{
1830       finishCoordination(filteredWorkItems, false);
1831       activeProvisioner.remove();
1832     }
1833   }

1836   /**
1837    * Look at the batch of workItems and flush caches necessary to process the entries
1838    * properly. For instance, if we're being notified that an ATTRIBUTE-ASSIGNMENT has
1839    * occurred, we can flush our cache to get the best information possible.
1840    * 
1841    * @param allWorkItems
1842    * @throws PspException
1843    */
1844   protected void flushCachesIfNecessary(List<ProvisioningWorkItem> allWorkItems)  throws PspException{
1845     for (ProvisioningWorkItem workItem : allWorkItems ) {
1846 
1847       // Skip irrelevant changelog entries
1848       if (!workItem.matchesChangelogType(ChangelogHandlingConfig.allRelevantChangelogTypes)) {
1849         continue;
1850       }
1851 
1852       // Skip changelog entries that don't need cache flushing
1853       if (!workItem.matchesChangelogType(ChangelogHandlingConfig.relevantChangesThatNeedGroupCacheFlushing)) {
1854         continue;
1855       }
1856 
1857       // We know we need to flush something from the cache. If the entry is group-specific,
1858       // we'll only flush that group
1859       GrouperGroupInfo groupInfo = workItem.getGroupInfo(this);
1860       if (groupInfo != null) {
1861         uncacheGroup(groupInfo, null);
1862       } else {
1863         // Flush everything and return
1864         uncacheAllGroups();
1865         return;
1866       }
1867     }
1868   }

1882   /**
1883    * Some changes (eg, labeling a folder for syncing) can have a large effect and are best handled with
1884    * a complete sync of all groups.
1885    * @return true if this work item should initiate a full sync of all groups
1886    */
1887   public boolean workItemShouldBeHandledByFullSyncOfEverything(ProvisioningWorkItem workItem) {
1888     // Skip if this ChangelogHandlingConfig says this doesn't affect group selection
1889     if ( !workItem.matchesChangelogType(ChangelogHandlingConfig.changelogTypesThatAreHandledViaFullSync) ) {
1890       return false;
1891     }
1892 
1893     String attributeName = workItem.getAttributeName();
1894 
1895     // If we can't figure out what the attributeName being modified is, we need to assume
1896     // GroupSelection might be changed
1897     if (StringUtils.isEmpty(attributeName)) {
1898       LOG.info("{}: Change might change group selection: {}", getDisplayName(), workItem);
1899       return true;
1900     }
1901 
1902     if ( getConfig().attributesUsedInGroupSelectionExpression.contains(attributeName) ) {
1903       LOG.info("{}: Change changes {} which might change group selection: {}",
1904               new Object[]{getDisplayName(), attributeName, workItem});
1905       return true;
1906     }
1907 
1908     return false;
1909   }
