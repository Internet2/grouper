71    /**
72     * get the new deprovisioning settings
73     * @return the new configuration 
74     */
75    public GrouperDeprovisioningAttributeValue getGrouperDeprovisioningAttributeValueNew() {
76      
77      if (StringUtils.isBlank(this.getAffiliation())) {
78        return null;
79      }
80      this.grouperDeprovisioningOverallConfiguration = this.getGrouperDeprovisioningOverallConfiguration();
81      GrouperDeprovisioningConfiguration grouperDeprovisioningConfiguration = this.grouperDeprovisioningOverallConfiguration.getAffiliationToConfiguration().get(this.affiliation);
82      if (grouperDeprovisioningConfiguration == null) {
83        grouperDeprovisioningConfiguration = new GrouperDeprovisioningConfiguration();
84        grouperDeprovisioningConfiguration.setGrouperDeprovisioningOverallConfiguration(this.grouperDeprovisioningOverallConfiguration);
85        this.grouperDeprovisioningOverallConfiguration.getAffiliationToConfiguration().put(this.affiliation, grouperDeprovisioningConfiguration);
86      }
87      GrouperDeprovisioningAttributeValue grouperDeprovisioningAttributeValue = grouperDeprovisioningConfiguration.getNewConfig();
88      
89      // if theres no configuration, or if the configuration is inherited, then clear it out
90      if (grouperDeprovisioningAttributeValue == null) {
91        grouperDeprovisioningAttributeValue = new GrouperDeprovisioningAttributeValue();
92        grouperDeprovisioningAttributeValue.setAffiliationString(this.affiliation);
93        grouperDeprovisioningAttributeValue.setGrouperDeprovisioningConfiguration(grouperDeprovisioningConfiguration);
94        grouperDeprovisioningConfiguration.setNewConfig(grouperDeprovisioningAttributeValue);
95      }
96      
97      return grouperDeprovisioningAttributeValue;
98    }

132

139

160   public boolean isHasDeprovisioningOnThisObjectOrParent() {
161     this.attributeAssignableHelper();
162     if (this.grouperDeprovisioningOverallConfiguration == null || GrouperUtil.length(this.grouperDeprovisioningOverallConfiguration) == 0) {
163       return false;
164     }
165     for (GrouperDeprovisioningConfiguration grouperDeprovisioningConfiguration : this.grouperDeprovisioningOverallConfiguration.getAffiliationToConfiguration().values()) {
166       if (grouperDeprovisioningConfiguration.getOriginalConfig() != null && !StringUtils.isBlank(grouperDeprovisioningConfiguration.getOriginalConfig().getAffiliationString())) {
167         return true;
168       }
169     }
170     return false;
171   }

294   /**
295    * 
296    * @return true if can read
297    */
298   public boolean isCanReadDeprovisioning() {
299     
300     GuiGroup guiGroup = GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupContainer().getGuiGroup();
301     
302     if (guiGroup != null) {
303       if (GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupContainer().isCanRead()) {
304         return true;
305       }
306     }
307     
308     GuiStem guiStem = GrouperRequestContainer.retrieveFromRequestOrCreate().getStemContainer().getGuiStem();
309     
310     if (guiStem != null) {
311       if (GrouperRequestContainer.retrieveFromRequestOrCreate().getStemContainer().isCanAdminPrivileges()) {
312         return true;
313       }
314     }
315     
316     GuiAttributeDef guiAttributeDef = GrouperRequestContainer.retrieveFromRequestOrCreate().getAttributeDefContainer().getGuiAttributeDef();
317     
318     if (guiAttributeDef != null) {
319       if (GrouperRequestContainer.retrieveFromRequestOrCreate().getAttributeDefContainer().isCanRead()) {
320         return true;
321       }
322     }
323     return false;
324   }

361   /**
362    * need to setup stuff about attestation
363    */
364   private void attributeAssignableHelper() {
365 
366     if (this.setupAttributes) {
367       return;
368     }
369     boolean hasError = false;
370     try {
371     
372       if (!this.isCanReadDeprovisioning() )  {
373         return;
374       }
375       
376       GrouperSession.internal_callbackRootGrouperSession(new GrouperSessionHandler() {
377         
378         public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
379           
380           DeprovisioningContainer.this.attributeAssignableHelperAsGrouperSystem();
381           
382           return null;
383         }
384       });
385   
386     } catch (RuntimeException re) {
387       hasError = true;
388       throw re;
389     } finally {
390       if (!hasError) {
391         this.setupAttributes = true;
392       }
393     }
394   }

396   /**
397    * run the helper logic as grouper system
398    */
399   private void attributeAssignableHelperAsGrouperSystem() {
400     
401     GuiGroup guiGroup = GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupContainer().getGuiGroup();
402     
403     if (guiGroup != null) {
404       Group group = guiGroup.getGroup();
405 
406       this.grouperDeprovisioningOverallConfiguration = GrouperDeprovisioningOverallConfiguration.retrieveConfiguration(group);
407 
408     }
409 
410     GuiAttributeDef guiAttributeDef = GrouperRequestContainer.retrieveFromRequestOrCreate().getAttributeDefContainer().getGuiAttributeDef();
411     
412     if (guiAttributeDef != null) {
413       AttributeDef attributeDef = guiAttributeDef.getAttributeDef();
414 
415       this.grouperDeprovisioningOverallConfiguration = GrouperDeprovisioningOverallConfiguration.retrieveConfiguration(attributeDef);
416 
417     }
418 
419     GuiStem guiStem = GrouperRequestContainer.retrieveFromRequestOrCreate().getStemContainer().getGuiStem();
420 
421     if (guiStem != null) {
422       
423       Stem stem = guiStem.getStem();
424       
425       this.grouperDeprovisioningOverallConfiguration = GrouperDeprovisioningOverallConfiguration.retrieveConfiguration(stem);
426       
427     }
428 
429   }
